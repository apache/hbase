19  package org.apache.hadoop.hbase.util;
20  
21  import org.apache.hadoop.classification.InterfaceAudience;
22  import org.apache.hadoop.classification.InterfaceStability;
23  
24  /**
25   * Manages a singleton instance of the environment edge. This class shall
26   * implement static versions of the interface {@link EnvironmentEdge}, then
27   * defer to the delegate on invocation.
28   */
29  @InterfaceAudience.Public
30  @InterfaceStability.Evolving
31  public class EnvironmentEdgeManager {
32    private static volatile EnvironmentEdge delegate = new DefaultEnvironmentEdge();
33  
34    private EnvironmentEdgeManager() {
35  
36    }
37  
38    /**
39     * Retrieves the singleton instance of the {@link EnvironmentEdge} that is
40     * being managed.
41     *
42     * @return the edge.
43     */
44    public static EnvironmentEdge getDelegate() {
45      return delegate;
46    }
47  
48    /**
49     * Resets the managed instance to the default instance: {@link
50     * DefaultEnvironmentEdge}.
51     */
52    public static void reset() {
53      injectEdge(new DefaultEnvironmentEdge());
54    }
55  
56    /**
57     * Injects the given edge such that it becomes the managed entity. If null is
58     * passed to this method, the default type is assigned to the delegate.
59     *
60     * @param edge the new edge.
61     */
62    public static void injectEdge(EnvironmentEdge edge) {
63      if (edge == null) {
64        reset();
65      } else {
66        delegate = edge;
67      }
68    }
69  
70    /**
71     * Defers to the delegate and calls the
72     * {@link EnvironmentEdge#currentTimeMillis()} method.
73     *
74     * @return current time in millis according to the delegate.
75     */
76    public static long currentTimeMillis() {
77      return getDelegate().currentTimeMillis();
78    }
79  }
