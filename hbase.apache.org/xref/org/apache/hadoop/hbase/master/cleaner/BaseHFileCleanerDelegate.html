18  package org.apache.hadoop.hbase.master.cleaner;
19  
20  import org.apache.hadoop.classification.InterfaceAudience;
21  import org.apache.hadoop.hbase.BaseConfigurable;
22  
23  /**
24   * Base class for the hfile cleaning function inside the master. By default, only the
25   * {@link TimeToLiveHFileCleaner} is called.
26   * <p>
27   * If other effects are needed, implement your own LogCleanerDelegate and add it to the
28   * configuration "hbase.master.hfilecleaner.plugins", which is a comma-separated list of fully
29   * qualified class names. The <code>HFileCleaner<code> will build the cleaner chain in 
30   * order the order specified by the configuration.
31   * <p>
32   * For subclasses, setConf will be called exactly <i>once</i> before using the cleaner.
33   * <p>
34   * Since {@link BaseHFileCleanerDelegate HFileCleanerDelegates} are created in
35   * HFileCleaner by reflection, classes that implements this interface <b>must</b>
36   * provide a default constructor.
37   */
38  @InterfaceAudience.Private
39  public abstract class BaseHFileCleanerDelegate extends BaseConfigurable implements
40      FileCleanerDelegate {
41  
42    private boolean stopped = false;
43  
44    @Override
45    public void stop(String why) {
46      this.stopped = true;
47    }
48  
49    @Override
50    public boolean isStopped() {
51      return this.stopped;
52    }
53  }
