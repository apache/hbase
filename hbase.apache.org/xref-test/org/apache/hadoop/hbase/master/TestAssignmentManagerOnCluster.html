45   * This tests AssignmentManager with a testing cluster.

47  @Category(MediumTests.class)
48  public class TestAssignmentManagerOnCluster {
49    private final static byte[] FAMILY = Bytes.toBytes("FAMILY");
50    private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
51    private final static Configuration conf = TEST_UTIL.getConfiguration();
52    private static HBaseAdmin admin;

54    @BeforeClass
55    public static void setUpBeforeClass() throws Exception {
56      TEST_UTIL.startMiniCluster(3);
57      admin = TEST_UTIL.getHBaseAdmin();
58    }
59  
60    @AfterClass
61    public static void tearDownAfterClass() throws Exception {
62      TEST_UTIL.shutdownMiniCluster();
63    }

65    /**
66     * This tests region assignment
67     */
68    @Test
69    public void testAssignRegion() throws Exception {
70      String table = "testAssignRegion";
71      try {
72        HTableDescriptor desc = new HTableDescriptor(table);
73        desc.addFamily(new HColumnDescriptor(FAMILY));
74        admin.createTable(desc);
75  
76        HTable meta = new HTable(conf, HConstants.META_TABLE_NAME);
77        HRegionInfo hri = new HRegionInfo(
78          desc.getName(), Bytes.toBytes("A"), Bytes.toBytes("Z"));
79        MetaEditor.addRegionToMeta(meta, hri);
80  
81        HMaster master = TEST_UTIL.getHBaseCluster().getMaster();
82        master.assignRegion(hri);
83        master.getAssignmentManager().waitForAssignment(hri);
84  
85        ServerName serverName = master.getAssignmentManager().
86          getRegionStates().getRegionServerOfRegion(hri);
87        TEST_UTIL.assertRegionOnServer(hri, serverName, 200);
88      } finally {
89        TEST_UTIL.deleteTable(Bytes.toBytes(table));
90      }
91    }

93    /**
94     * This tests offlining a region
95     */
96    @Test
97    public void testOfflineRegion() throws Exception {
98      String table = "testOfflineRegion";
99      try {
100       HRegionInfo hri = createTableAndGetOneRegion(table);
101 
102       RegionStates regionStates = TEST_UTIL.getHBaseCluster().
103         getMaster().getAssignmentManager().getRegionStates();
104       ServerName serverName = regionStates.getRegionServerOfRegion(hri);
105       TEST_UTIL.assertRegionOnServer(hri, serverName, 200);
106       admin.offline(hri.getRegionName());
107 
108       long timeoutTime = System.currentTimeMillis() + 800;
109       while (true) {
110         List<HRegionInfo> regions =
111           regionStates.getRegionsOfTable(Bytes.toBytes(table));
112         if (!regions.contains(hri)) break;
113         long now = System.currentTimeMillis();
114         if (now > timeoutTime) {
115           fail("Failed to offline the region in time");
116           break;
117         }
118         Thread.sleep(10);
119       }
120       RegionState regionState = regionStates.getRegionState(hri);
121       assertTrue(regionState.isOffline());
122     } finally {
123       TEST_UTIL.deleteTable(Bytes.toBytes(table));
124     }
125   }

127   /**
128    * This tests moving a region
129    */
130   @Test
131   public void testMoveRegion() throws Exception {
132     String table = "testMoveRegion";
133     try {
134       HRegionInfo hri = createTableAndGetOneRegion(table);
135 
136       RegionStates regionStates = TEST_UTIL.getHBaseCluster().
137         getMaster().getAssignmentManager().getRegionStates();
138       ServerName serverName = regionStates.getRegionServerOfRegion(hri);
139       ServerName destServerName = null;
140       for (int i = 0; i < 3; i++) {
141         HRegionServer destServer = TEST_UTIL.getHBaseCluster().getRegionServer(i);
142         if (!destServer.getServerName().equals(serverName)) {
143           destServerName = destServer.getServerName();
144           break;
145         }
146       }
147       assertTrue(destServerName != null
148         && !destServerName.equals(serverName));
149       TEST_UTIL.getHBaseAdmin().move(hri.getEncodedNameAsBytes(),
150         Bytes.toBytes(destServerName.getServerName()));
151 
152       long timeoutTime = System.currentTimeMillis() + 800;
153       while (true) {
154         ServerName sn = regionStates.getRegionServerOfRegion(hri);
155         if (sn != null && sn.equals(destServerName)) {
156           TEST_UTIL.assertRegionOnServer(hri, sn, 200);
157           break;
158         }
159         long now = System.currentTimeMillis();
160         if (now > timeoutTime) {
161           fail("Failed to move the region in time");
162         }
163         regionStates.waitForUpdate(50);
164       }
165 
166     } finally {
167       TEST_UTIL.deleteTable(Bytes.toBytes(table));
168     }
169   }

171   HRegionInfo createTableAndGetOneRegion(
172       final String tableName) throws IOException, InterruptedException {
173     HTableDescriptor desc = new HTableDescriptor(tableName);
174     desc.addFamily(new HColumnDescriptor(FAMILY));
175     admin.createTable(desc, Bytes.toBytes("A"), Bytes.toBytes("Z"), 5);
176 
177     // wait till the table is assigned
178     HMaster master = TEST_UTIL.getHBaseCluster().getMaster();
179     long timeoutTime = System.currentTimeMillis() + 100;
180     while (true) {
181       List<HRegionInfo> regions = master.getAssignmentManager().
182         getRegionStates().getRegionsOfTable(Bytes.toBytes(tableName));
183       if (regions.size() > 3) {
184         return regions.get(2);
185       }
186       long now = System.currentTimeMillis();
187       if (now > timeoutTime) {
188         fail("Could not find an online region");
189       }
190       Thread.sleep(10);
191     }
192   }
