150   /**
151    * Called prior to writing the {@link StoreFile}s selected for compaction into
152    * a new {@code StoreFile}.  To override or modify the compaction process,
153    * implementing classes have two options:
154    * <ul>
155    *   <li>Wrap the provided {@link InternalScanner} with a custom
156    *   implementation that is returned from this method.  The custom scanner
157    *   can then inspect {@link KeyValue}s from the wrapped scanner, applying
158    *   its own policy to what gets written.</li>
159    *   <li>Call {@link org.apache.hadoop.hbase.coprocessor.ObserverContext#bypass()}
160    *   and provide a custom implementation for writing of new
161    *   {@link StoreFile}s.  <strong>Note: any implementations bypassing
162    *   core compaction using this approach must write out new store files
163    *   themselves or the existing data will no longer be available after
164    *   compaction.</strong></li>
165    * </ul>
166    * @param c the environment provided by the region server
167    * @param store the store being compacted
168    * @param scanner the scanner over existing data used in the store file
169    * rewriting
170    * @return the scanner to use during compaction.  Should not be {@code null}
171    * unless the implementation is writing new store files on its own.
172    * @throws IOException if an error occurred on the coprocessor
173    */
174   InternalScanner preCompact(final ObserverContext<RegionCoprocessorEnvironment> c,
175       final HStore store, final InternalScanner scanner) throws IOException;

177   /**
178    * Called prior to writing the {@link StoreFile}s selected for compaction into
179    * a new {@code StoreFile} and prior to creating the scanner used to read the
180    * input files.  To override or modify the compaction process,
181    * implementing classes can return a new scanner to provide the KeyValues to be
182    * stored into the new {@code StoreFile} or null to perform the default processing.
183    * Calling {@link org.apache.hadoop.hbase.coprocessor.ObserverContext#bypass()} has no
184    * effect in this hook.
185    * @param c the environment provided by the region server
186    * @param store the store being compacted
187    * @param scanners the list {@link StoreFileScanner}s to be read from
188    * @param scanType the {@link ScanType} indicating whether this is a major or minor compaction
189    * @param earliestPutTs timestamp of the earliest put that was found in any of the involved
190    * store files
191    * @param s the base scanner, if not {@code null}, from previous RegionObserver in the chain
192    * @return the scanner to use during compaction.  {@code null} if the default implementation
193    * is to be used.
194    * @throws IOException if an error occurred on the coprocessor
195    */
196   InternalScanner preCompactScannerOpen(final ObserverContext<RegionCoprocessorEnvironment> c,
197       final HStore store, List<? extends KeyValueScanner> scanners, final ScanType scanType,
