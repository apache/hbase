62    private static final NavigableMap<Long, Boolean> TIMESTAMP =
63      new TreeMap<Long, Boolean>();
64    static {
65      TIMESTAMP.put((long)1245620000, false);
66      TIMESTAMP.put((long)1245620005, true); // include
67      TIMESTAMP.put((long)1245620010, true); // include
68      TIMESTAMP.put((long)1245620055, true); // include
69      TIMESTAMP.put((long)1245620100, true); // include
70      TIMESTAMP.put((long)1245620150, false);
71      TIMESTAMP.put((long)1245620250, false);
72    }
73    static final long MINSTAMP = 1245620005;
74    static final long MAXSTAMP = 1245620100 + 1; // maxStamp itself is excluded. so increment it.

100   private static class ProcessTimeRangeMapper
101   extends TableMapper<ImmutableBytesWritable, MapWritable>
102   implements Configurable {
103 
104     private Configuration conf = null;
105     private HTable table = null;
106 
107     @Override
108     public void map(ImmutableBytesWritable key, Result result,
109         Context context)
110     throws IOException {
111       List<Long> tsList = new ArrayList<Long>();
112       for (KeyValue kv : result.list()) {
113         tsList.add(kv.getTimestamp());
114       }
115 
116       for (Long ts : tsList) {
117         Put put = new Put(key.get());
118         put.setWriteToWAL(false);
119         put.add(FAMILY_NAME, COLUMN_NAME, ts, Bytes.toBytes(true));
120         table.put(put);
121       }
122       table.flushCommits();
123     }
124 
125     @Override
126     public Configuration getConf() {
127       return conf;
128     }
129 
130     @Override
131     public void setConf(Configuration configuration) {
132       this.conf = configuration;
133       try {
134         table = new HTable(HBaseConfiguration.create(conf), TABLE_NAME);
135       } catch (IOException e) {
136         e.printStackTrace();
137       }
138     }
139   }

141   @Test
142   public void testTimeRangeMapRed()
143   throws IOException, InterruptedException, ClassNotFoundException {
144     final HTableDescriptor desc = new HTableDescriptor(TABLE_NAME);
145     final HColumnDescriptor col = new HColumnDescriptor(FAMILY_NAME);
146     col.setMaxVersions(Integer.MAX_VALUE);
147     desc.addFamily(col);
148     admin.createTable(desc);
149     HTable table = new HTable(UTIL.getConfiguration(), desc.getName());
150     prepareTest(table);
151     runTestOnTable();
152     verify(table);
153   }
154 
155   private void prepareTest(final HTable table) throws IOException {
156     for (Map.Entry<Long, Boolean> entry : TIMESTAMP.entrySet()) {
157       Put put = new Put(KEY);
158       put.setWriteToWAL(false);
159       put.add(FAMILY_NAME, COLUMN_NAME, entry.getKey(), Bytes.toBytes(false));
160       table.put(put);
161     }
162     table.flushCommits();
163   }
164 
165   private void runTestOnTable()
166   throws IOException, InterruptedException, ClassNotFoundException {
167     UTIL.startMiniMapReduceCluster();
168     Job job = null;
169     try {
170       job = new Job(UTIL.getConfiguration(), "test123");
171       job.setOutputFormatClass(NullOutputFormat.class);
172       job.setNumReduceTasks(0);
173       Scan scan = new Scan();
174       scan.addColumn(FAMILY_NAME, COLUMN_NAME);
175       scan.setTimeRange(MINSTAMP, MAXSTAMP);
176       scan.setMaxVersions();
177       TableMapReduceUtil.initTableMapperJob(Bytes.toString(TABLE_NAME),
178         scan, ProcessTimeRangeMapper.class, Text.class, Text.class, job);
179       job.waitForCompletion(true);
180     } catch (IOException e) {
181       // TODO Auto-generated catch block
182       e.printStackTrace();
183     } finally {
184       UTIL.shutdownMiniMapReduceCluster();
185       if (job != null) {
186         FileUtil.fullyDelete(
187           new File(job.getConfiguration().get("hadoop.tmp.dir")));
188       }
189     }
190   }
191 
192   private void verify(final HTable table) throws IOException {
193     Scan scan = new Scan();
194     scan.addColumn(FAMILY_NAME, COLUMN_NAME);
195     scan.setMaxVersions(1);
196     ResultScanner scanner = table.getScanner(scan);
197     for (Result r: scanner) {
198       for (KeyValue kv : r.list()) {
199         log.debug(Bytes.toString(r.getRow()) + "\t" + Bytes.toString(kv.getFamily())
200             + "\t" + Bytes.toString(kv.getQualifier())
201             + "\t" + kv.getTimestamp() + "\t" + Bytes.toBoolean(kv.getValue()));
202         org.junit.Assert.assertEquals(TIMESTAMP.get(kv.getTimestamp()),
203           (Boolean)Bytes.toBoolean(kv.getValue()));
204       }
205     }
206     scanner.close();
207   }
