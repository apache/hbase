70    /**
71     * Complex test of master failover that tests as many permutations of the
72     * different possible states that regions in transition could be in within ZK.
73     * <p>
74     * This tests the proper handling of these states by the failed-over master
75     * and includes a thorough testing of the timeout code as well.
76     * <p>
77     * Starts with a single master and three regionservers.
78     * <p>
79     * Creates two tables, enabledTable and disabledTable, each containing 5
80     * regions.  The disabledTable is then disabled.
81     * <p>
82     * After reaching steady-state, the master is killed.  We then mock several
83     * states in ZK.
84     * <p>
85     * After mocking them, we will startup a new master which should become the
86     * active master and also detect that it is a failover.  The primary test
87     * passing condition will be that all regions of the enabled table are
88     * assigned and all the regions of the disabled table are not assigned.
89     * <p>
90     * The different scenarios to be tested are below:
91     * <p>
92     * <b>ZK State:  OFFLINE</b>
93     * <p>A node can get into OFFLINE state if</p>
94     * <ul>
95     * <li>An RS fails to open a region, so it reverts the state back to OFFLINE
96     * <li>The Master is assigning the region to a RS before it sends RPC
97     * </ul>
98     * <p>We will mock the scenarios</p>
99     * <ul>
100    * <li>Master has assigned an enabled region but RS failed so a region is
101    *     not assigned anywhere and is sitting in ZK as OFFLINE</li>
102    * <li>This seems to cover both cases?</li>
103    * </ul>
104    * <p>
105    * <b>ZK State:  CLOSING</b>
106    * <p>A node can get into CLOSING state if</p>
107    * <ul>
108    * <li>An RS has begun to close a region
109    * </ul>
110    * <p>We will mock the scenarios</p>
111    * <ul>
112    * <li>Region of enabled table was being closed but did not complete
113    * <li>Region of disabled table was being closed but did not complete
114    * </ul>
115    * <p>
116    * <b>ZK State:  CLOSED</b>
117    * <p>A node can get into CLOSED state if</p>
118    * <ul>
119    * <li>An RS has completed closing a region but not acknowledged by master yet
120    * </ul>
121    * <p>We will mock the scenarios</p>
122    * <ul>
123    * <li>Region of a table that should be enabled was closed on an RS
124    * <li>Region of a table that should be disabled was closed on an RS
125    * </ul>
126    * <p>
127    * <b>ZK State:  OPENING</b>
128    * <p>A node can get into OPENING state if</p>
129    * <ul>
130    * <li>An RS has begun to open a region
131    * </ul>
132    * <p>We will mock the scenarios</p>
133    * <ul>
134    * <li>RS was opening a region of enabled table but never finishes
135    * </ul>
136    * <p>
137    * <b>ZK State:  OPENED</b>
138    * <p>A node can get into OPENED state if</p>
139    * <ul>
140    * <li>An RS has finished opening a region but not acknowledged by master yet
141    * </ul>
142    * <p>We will mock the scenarios</p>
143    * <ul>
144    * <li>Region of a table that should be enabled was opened on an RS
145    * <li>Region of a table that should be disabled was opened on an RS
146    * </ul>
147    * @throws Exception
148    */
149   @Test (timeout=180000)
150   public void testMasterFailoverWithMockedRIT() throws Exception {
151 
152     final int NUM_MASTERS = 1;
153     final int NUM_RS = 3;
154 
155     // Create config to use for this cluster
156     Configuration conf = HBaseConfiguration.create();
157     // Need to drop the timeout much lower
158     conf.setInt("hbase.master.assignment.timeoutmonitor.period", 2000);
159     conf.setInt("hbase.master.assignment.timeoutmonitor.timeout", 4000);
160     conf.setInt(ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART, 3);
161     conf.setInt(ServerManager.WAIT_ON_REGIONSERVERS_MAXTOSTART, 3);
162 
163     // Start the cluster
164     HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility(conf);
165     TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
166     MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
167     log("Cluster started");
168 
169     // Create a ZKW to use in the test
170     ZooKeeperWatcher zkw = HBaseTestingUtility.getZooKeeperWatcher(TEST_UTIL);
171 
172     // get all the master threads
173     List<MasterThread> masterThreads = cluster.getMasterThreads();
174     assertEquals(1, masterThreads.size());
175 
176     // only one master thread, let's wait for it to be initialized
177     assertTrue(cluster.waitForActiveAndReadyMaster());
178     HMaster master = masterThreads.get(0).getMaster();
179     assertTrue(master.isActiveMaster());
180     assertTrue(master.isInitialized());
181 
182     // disable load balancing on this master
183     master.balanceSwitch(false);
184 
185     // create two tables in META, each with 10 regions
186     byte [] FAMILY = Bytes.toBytes("family");
187     byte [][] SPLIT_KEYS = new byte [][] {
188         new byte[0], Bytes.toBytes("aaa"), Bytes.toBytes("bbb"),
189         Bytes.toBytes("ccc"), Bytes.toBytes("ddd"), Bytes.toBytes("eee"),
190         Bytes.toBytes("fff"), Bytes.toBytes("ggg"), Bytes.toBytes("hhh"),
191         Bytes.toBytes("iii"), Bytes.toBytes("jjj")
192     };
193 
194     byte [] enabledTable = Bytes.toBytes("enabledTable");
195     HTableDescriptor htdEnabled = new HTableDescriptor(enabledTable);
196     htdEnabled.addFamily(new HColumnDescriptor(FAMILY));
197 
198     FileSystem filesystem = FileSystem.get(conf);
199     Path rootdir = filesystem.makeQualified(
200         new Path(conf.get(HConstants.HBASE_DIR)));
201     // Write the .tableinfo
202     FSTableDescriptors.createTableDescriptor(filesystem, rootdir, htdEnabled);
203 
204     HRegionInfo hriEnabled = new HRegionInfo(htdEnabled.getName(), null, null);
205     createRegion(hriEnabled, rootdir, conf, htdEnabled);
206 
207     List<HRegionInfo> enabledRegions = TEST_UTIL.createMultiRegionsInMeta(
208         TEST_UTIL.getConfiguration(), htdEnabled, SPLIT_KEYS);
209 
210     byte [] disabledTable = Bytes.toBytes("disabledTable");
211     HTableDescriptor htdDisabled = new HTableDescriptor(disabledTable);
212     htdDisabled.addFamily(new HColumnDescriptor(FAMILY));
213     // Write the .tableinfo
214     FSTableDescriptors.createTableDescriptor(filesystem, rootdir, htdDisabled);
215     HRegionInfo hriDisabled = new HRegionInfo(htdDisabled.getName(), null, null);
216     createRegion(hriDisabled, rootdir, conf, htdDisabled);
217     List<HRegionInfo> disabledRegions = TEST_UTIL.createMultiRegionsInMeta(
218         TEST_UTIL.getConfiguration(), htdDisabled, SPLIT_KEYS);
219 
220     log("Regions in META have been created");
221 
222     // at this point we only expect 2 regions to be assigned out (catalogs)
223     assertEquals(2, cluster.countServedRegions());
224 
225     // Let's just assign everything to first RS
226     HRegionServer hrs = cluster.getRegionServer(0);
227     ServerName serverName = hrs.getServerName();
228     HRegionInfo closingRegion = enabledRegions.remove(0);
229     // we'll need some regions to already be assigned out properly on live RS
230     List<HRegionInfo> enabledAndAssignedRegions = new ArrayList<HRegionInfo>();
231     enabledAndAssignedRegions.add(enabledRegions.remove(0));
232     enabledAndAssignedRegions.add(enabledRegions.remove(0));
233     enabledAndAssignedRegions.add(closingRegion);
234 
235     List<HRegionInfo> disabledAndAssignedRegions = new ArrayList<HRegionInfo>();
236     disabledAndAssignedRegions.add(disabledRegions.remove(0));
237     disabledAndAssignedRegions.add(disabledRegions.remove(0));
238 
239     // now actually assign them
240     for (HRegionInfo hri : enabledAndAssignedRegions) {
241       master.assignmentManager.regionPlans.put(hri.getEncodedName(),
242           new RegionPlan(hri, null, serverName));
243       master.assignRegion(hri);
244     }
245     for (HRegionInfo hri : disabledAndAssignedRegions) {
246       master.assignmentManager.regionPlans.put(hri.getEncodedName(),
247           new RegionPlan(hri, null, serverName));
248       master.assignRegion(hri);
249     }
250 
251     // wait for no more RIT
252     log("Waiting for assignment to finish");
253     ZKAssign.blockUntilNoRIT(zkw);
254     log("Assignment completed");
255 
256     // Stop the master
257     log("Aborting master");
258     cluster.abortMaster(0);
259     cluster.waitOnMaster(0);
260     log("Master has aborted");
261 
262     /*
263      * Now, let's start mocking up some weird states as described in the method
264      * javadoc.
265      */
266 
267     List<HRegionInfo> regionsThatShouldBeOnline = new ArrayList<HRegionInfo>();
268     List<HRegionInfo> regionsThatShouldBeOffline = new ArrayList<HRegionInfo>();
269 
270     log("Beginning to mock scenarios");
271 
272     // Disable the disabledTable in ZK
273     ZKTable zktable = new ZKTable(zkw);
274     zktable.setDisabledTable(Bytes.toString(disabledTable));
275 
276     /*
277      *  ZK = OFFLINE
278      */
279 
280     // Region that should be assigned but is not and is in ZK as OFFLINE
281     HRegionInfo region = enabledRegions.remove(0);
282     regionsThatShouldBeOnline.add(region);
283     ZKAssign.createNodeOffline(zkw, region, serverName);
284 
285     /*
286      * ZK = CLOSING
287      */
288     regionsThatShouldBeOnline.add(closingRegion);
289     ZKAssign.createNodeClosing(zkw, closingRegion, serverName);
290 
291     /*
292      * ZK = CLOSED
293      */
294 
295     // Region of enabled table closed but not ack
296     region = enabledRegions.remove(0);
297     regionsThatShouldBeOnline.add(region);
298     int version = ZKAssign.createNodeClosing(zkw, region, serverName);
299     ZKAssign.transitionNodeClosed(zkw, region, serverName, version);
300 
301     // Region of disabled table closed but not ack
302     region = disabledRegions.remove(0);
303     regionsThatShouldBeOffline.add(region);
304     version = ZKAssign.createNodeClosing(zkw, region, serverName);
305     ZKAssign.transitionNodeClosed(zkw, region, serverName, version);
306 
307     /*
308      * ZK = OPENING
309      */
310 
311     // RS was opening a region of enabled table but never finishes
312     region = enabledRegions.remove(0);
313     regionsThatShouldBeOnline.add(region);
314     ZKAssign.createNodeOffline(zkw, region, serverName);
315     ZKAssign.transitionNodeOpening(zkw, region, serverName);
316 
317     /*
318      * ZK = OPENED
319      */
320 
321     // Region of enabled table was opened on RS
322     region = enabledRegions.remove(0);
323     regionsThatShouldBeOnline.add(region);
324     ZKAssign.createNodeOffline(zkw, region, serverName);
325     ProtobufUtil.openRegion(hrs, region);
326     while (true) {
327       byte [] bytes = ZKAssign.getData(zkw, region.getEncodedName());
328       RegionTransition rt = RegionTransition.parseFrom(bytes);
329       if (rt != null && rt.getEventType().equals(EventType.RS_ZK_REGION_OPENED)) {
330         break;
331       }
332       Thread.sleep(100);
333     }
334 
335     // Region of disable table was opened on RS
336     region = disabledRegions.remove(0);
337     regionsThatShouldBeOffline.add(region);
338     ZKAssign.createNodeOffline(zkw, region, serverName);
339     ProtobufUtil.openRegion(hrs, region);
340     while (true) {
341       byte [] bytes = ZKAssign.getData(zkw, region.getEncodedName());
342       RegionTransition rt = RegionTransition.parseFrom(bytes);
343       if (rt != null && rt.getEventType().equals(EventType.RS_ZK_REGION_OPENED)) {
344         break;
345       }
346       Thread.sleep(100);
347     }
348 
349     /*
350      * ZK = NONE
351      */
352 
353     /*
354      * DONE MOCKING
355      */
356 
357     log("Done mocking data up in ZK");
358 
359     // Start up a new master
360     log("Starting up a new master");
361     master = cluster.startMaster().getMaster();
362     log("Waiting for master to be ready");
363     cluster.waitForActiveAndReadyMaster();
364     log("Master is ready");
365 
366     // Failover should be completed, now wait for no RIT
367     log("Waiting for no more RIT");
368     ZKAssign.blockUntilNoRIT(zkw);
369     log("No more RIT in ZK, now doing final test verification");
370 
371     // Grab all the regions that are online across RSs
372     Set<HRegionInfo> onlineRegions = new TreeSet<HRegionInfo>();
373     for (JVMClusterUtil.RegionServerThread rst :
374       cluster.getRegionServerThreads()) {
375       onlineRegions.addAll(ProtobufUtil.getOnlineRegions(rst.getRegionServer()));
376     }
377 
378     // Now, everything that should be online should be online
379     for (HRegionInfo hri : regionsThatShouldBeOnline) {
380       assertTrue(onlineRegions.contains(hri));
381     }
382 
383     // Everything that should be offline should not be online
384     for (HRegionInfo hri : regionsThatShouldBeOffline) {
385       assertFalse(onlineRegions.contains(hri));
386     }
387 
388     log("Done with verification, all passed, shutting down cluster");
389 
390     // Done, shutdown the cluster
391     TEST_UTIL.shutdownMiniCluster();
392   }

397    * different possible states that regions in transition could be in within ZK

401    * and includes a thorough testing of the timeout code as well.
402    * <p>
403    * Starts with a single master and two regionservers.
404    * <p>

406    * regions.  The disabledTable is then disabled.
407    * <p>

409    * states in ZK.  And one of the RS will be killed.

411    * After mocking them and killing an RS, we will startup a new master which
412    * should become the active master and also detect that it is a failover.  The
413    * primary test passing condition will be that all regions of the enabled
414    * table are assigned and all the regions of the disabled table are not
415    * assigned.
416    * <p>
417    * The different scenarios to be tested are below:
418    * <p>
419    * <b>ZK State:  CLOSING</b>

440    * <p>A region could not have a transition node if</p>
441    * <ul>
442    * <li>The server hosting the region died and no master processed it
443    * </ul>
444    * <p>We will mock the scenarios</p>
445    * <ul>

911    * Simple test of master failover.
912    * <p>
913    * Starts with three masters.  Kills a backup master.  Then kills the active
914    * master.  Ensures the final master becomes active and we can still contact
915    * the cluster.
916    * @throws Exception

921     final int NUM_MASTERS = 3;
922     final int NUM_RS = 3;
923 
924     // Start the cluster
925     HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
926

928     MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
929 
930     // get all the master threads
931     List<MasterThread> masterThreads = cluster.getMasterThreads();
932 
933     // wait for each to come online
934     for (MasterThread mt : masterThreads) {
935       assertTrue(mt.isAlive());
936     }
937 
938     // verify only one is the active master and we have right number
939     int numActive = 0;
940     int activeIndex = -1;
941     ServerName activeName = null;
942     HMaster active = null;
943     for (int i = 0; i < masterThreads.size(); i++) {
944       if (masterThreads.get(i).getMaster().isActiveMaster()) {
945         numActive++;
946         activeIndex = i;
947         active = masterThreads.get(activeIndex).getMaster();
948         activeName = active.getServerName();
949       }

958     assertTrue(status.getMaster().equals(activeName));
959     assertEquals(2, status.getBackupMastersSize());
960     assertEquals(2, status.getBackupMasters().size());
961 
962     // attempt to stop one of the inactive masters
963     int backupIndex = (activeIndex == 0 ? 1 : activeIndex - 1);
964     HMaster master = cluster.getMaster(backupIndex);
965     LOG.debug("\n\nStopping a backup master: " + master.getServerName() + "\n");
966     cluster.stopMaster(backupIndex, false);

968

970     for (int i = 0; i < masterThreads.size(); i++) {
971       if (masterThreads.get(i).getMaster().isActiveMaster()) {
972         assertTrue(activeName.equals(masterThreads.get(i).getMaster().getServerName()));
973         activeIndex = i;
974         active = masterThreads.get(activeIndex).getMaster();
975       }
976     }
977     assertEquals(1, numActive);
978     assertEquals(2, masterThreads.size());
979     int rsCount = masterThreads.get(activeIndex).getMaster().getClusterStatus().getServersSize();
980     LOG.info("Active master " + active.getServerName() + " managing " + rsCount +  " regions servers");
981     assertEquals(3, rsCount);
982 
983     // Check that ClusterStatus reports the correct active and backup masters
984     assertNotNull(active);
985     status = active.getClusterStatus();
986     assertTrue(status.getMaster().equals(activeName));
987     assertEquals(1, status.getBackupMastersSize());
988     assertEquals(1, status.getBackupMasters().size());
989
