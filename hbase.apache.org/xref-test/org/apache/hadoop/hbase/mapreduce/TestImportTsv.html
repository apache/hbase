87      assertBytesEquals(Bytes.toBytes("col1"), parser.getFamily(1));

90      assertBytesEquals(Bytes.toBytes("scol2"), parser.getQualifier(3));
91      assertEquals(0, parser.getRowKeyColumnIndex());
92      assertTrue(parser.hasTimestamp());
93      assertEquals(2, parser.getTimestampKeyColumnIndex());

96    @Test
97    public void testTsvParser() throws BadTsvLineException {
98      TsvParser parser = new TsvParser("col_a,col_b:qual,HBASE_ROW_KEY,col_d", "\t");
99      assertBytesEquals(Bytes.toBytes("col_a"), parser.getFamily(0));
100     assertBytesEquals(HConstants.EMPTY_BYTE_ARRAY, parser.getQualifier(0));
101     assertBytesEquals(Bytes.toBytes("col_b"), parser.getFamily(1));
102     assertBytesEquals(Bytes.toBytes("qual"), parser.getQualifier(1));
103     assertNull(parser.getFamily(2));
104     assertNull(parser.getQualifier(2));
105     assertEquals(2, parser.getRowKeyColumnIndex());
106     
107     assertEquals(TsvParser.DEFAULT_TIMESTAMP_COLUMN_INDEX, parser
108         .getTimestampKeyColumnIndex());
109     
110     byte[] line = Bytes.toBytes("val_a\tval_b\tval_c\tval_d");
111     ParsedLine parsed = parser.parse(line, line.length);
112     checkParsing(parsed, Splitter.on("\t").split(Bytes.toString(line)));
113   }

134   private void checkParsing(ParsedLine parsed, Iterable<String> expected) {
135     ArrayList<String> parsedCols = new ArrayList<String>();
136     for (int i = 0; i < parsed.getColumnCount(); i++) {
137       parsedCols.add(Bytes.toString(
138           parsed.getLineBytes(),
139           parsed.getColumnOffset(i),
140           parsed.getColumnLength(i)));
141     }
142     if (!Iterables.elementsEqual(parsedCols, expected)) {
143       fail("Expected: " + Joiner.on(",").join(expected) + "\n" +
144           "Got:" + Joiner.on(",").join(parsedCols));
145     }
146   }

254   private void doMROnTableTest(String inputFile, String family,
255       String tableName, String data, String[] args, int valueMultiplier)
256       throws Exception {
257 
258     // Cluster
259     HBaseTestingUtility htu1 = new HBaseTestingUtility();
260 
261     htu1.startMiniCluster();
262     htu1.startMiniMapReduceCluster();
263 
264     GenericOptionsParser opts = new GenericOptionsParser(htu1.getConfiguration(), args);
265     Configuration conf = opts.getConfiguration();
266     args = opts.getRemainingArgs();
267 
268     try {
269 
270       FileSystem fs = FileSystem.get(conf);
271       FSDataOutputStream op = fs.create(new Path(inputFile), true);
272       if (data == null) {
273         data = "KEY\u001bVALUE1\u001bVALUE2\n";
274       }
275       op.write(Bytes.toBytes(data));
276       op.close();
277 
278       final byte[] FAM = Bytes.toBytes(family);
279       final byte[] TAB = Bytes.toBytes(tableName);
280       if (conf.get(ImportTsv.BULK_OUTPUT_CONF_KEY) == null) {
281         HTableDescriptor desc = new HTableDescriptor(TAB);
282         desc.addFamily(new HColumnDescriptor(FAM));
283         new HBaseAdmin(conf).createTable(desc);
284       } else { // set the hbaseAdmin as we are not going through main()
285         ImportTsv.createHbaseAdmin(conf);
286       }
287       Job job = ImportTsv.createSubmittableJob(conf, args);
288       job.waitForCompletion(false);
289       assertTrue(job.isSuccessful());
290       
291       HTable table = new HTable(new Configuration(conf), TAB);
292       boolean verified = false;
293       long pause = conf.getLong("hbase.client.pause", 5 * 1000);
294       int numRetries = conf.getInt("hbase.client.retries.number", 5);
295       for (int i = 0; i < numRetries; i++) {
296         try {
297           Scan scan = new Scan();
298           // Scan entire family.
299           scan.addFamily(FAM);
300           ResultScanner resScanner = table.getScanner(scan);
301           for (Result res : resScanner) {
302             assertTrue(res.size() == 2);
303             List<KeyValue> kvs = res.list();
304             assertEquals(toU8Str(kvs.get(0).getRow()),
305                 toU8Str(Bytes.toBytes("KEY")));
306             assertEquals(toU8Str(kvs.get(1).getRow()),
307                 toU8Str(Bytes.toBytes("KEY")));
308             assertEquals(toU8Str(kvs.get(0).getValue()),
309                 toU8Str(Bytes.toBytes("VALUE" + valueMultiplier)));
310             assertEquals(toU8Str(kvs.get(1).getValue()),
311                 toU8Str(Bytes.toBytes("VALUE" + 2*valueMultiplier)));
312             // Only one result set is expected, so let it loop.
313           }
314           verified = true;
315           break;
316         } catch (NullPointerException e) {
317           // If here, a cell was empty.  Presume its because updates came in
318           // after the scanner had been opened.  Wait a while and retry.
319         }
320         try {
321           Thread.sleep(pause);
322         } catch (InterruptedException e) {
323           // continue
324         }
325       }
326       assertTrue(verified);
327     } finally {
328       htu1.shutdownMiniMapReduceCluster();
329       htu1.shutdownMiniCluster();
330     }
331   }
