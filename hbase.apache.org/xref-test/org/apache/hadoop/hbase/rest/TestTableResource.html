58    private static String COLUMN_FAMILY = "test";
59    private static String COLUMN = COLUMN_FAMILY + ":qualifier";
60    private static Map<HRegionInfo, ServerName> regionMap;

68    @BeforeClass
69    public static void setUpBeforeClass() throws Exception {
70      TEST_UTIL.startMiniCluster(3);
71      REST_TEST_UTIL.startServletContainer(TEST_UTIL.getConfiguration());
72      client = new Client(new Cluster().add("localhost", 
73        REST_TEST_UTIL.getServletPort()));
74      context = JAXBContext.newInstance(
75          TableModel.class,
76          TableInfoModel.class,
77          TableListModel.class,
78          TableRegionModel.class);
79      HBaseAdmin admin = TEST_UTIL.getHBaseAdmin();
80      if (admin.tableExists(TABLE)) {
81        return;
82      }
83      HTableDescriptor htd = new HTableDescriptor(TABLE);
84      htd.addFamily(new HColumnDescriptor(COLUMN_FAMILY));
85      admin.createTable(htd);
86      HTable table = new HTable(TEST_UTIL.getConfiguration(), TABLE);
87      byte[] k = new byte[3];
88      byte [][] famAndQf = KeyValue.parseColumn(Bytes.toBytes(COLUMN));
89      for (byte b1 = 'a'; b1 < 'z'; b1++) {
90        for (byte b2 = 'a'; b2 < 'z'; b2++) {
91          for (byte b3 = 'a'; b3 < 'z'; b3++) {
92            k[0] = b1;
93            k[1] = b2;
94            k[2] = b3;
95            Put put = new Put(k);
96            put.setWriteToWAL(false);
97            put.add(famAndQf[0], famAndQf[1], k);
98            table.put(put);
99          }
100       }
101     }
102     table.flushCommits();
103     // get the initial layout (should just be one region)
104     Map<HRegionInfo, ServerName> m = table.getRegionLocations();
105     assertEquals(m.size(), 1);
106     // tell the master to split the table
107     admin.split(TABLE);
108     // give some time for the split to happen
109 
110     long timeout = System.currentTimeMillis() + (15 * 1000);
111     while (System.currentTimeMillis() < timeout && m.size()!=2){
112       try {
113         Thread.sleep(250);
114       } catch (InterruptedException e) {
115         LOG.warn(StringUtils.stringifyException(e));
116       }
117       // check again
118       m = table.getRegionLocations();
119     }
120 
121     // should have two regions now
122     assertEquals(m.size(), 2);
123     regionMap = m;
124     LOG.info("regions: " + regionMap);
125     table.close();
126   }

134   private static void checkTableList(TableListModel model) {
135     boolean found = false;
136     Iterator<TableModel> tables = model.getTables().iterator();
137     assertTrue(tables.hasNext());
138     while (tables.hasNext()) {
139       TableModel table = tables.next();
140       if (table.getName().equals(TABLE)) {
141         found = true;
142         break;
143       }
144     }
145     assertTrue(found);
146   }
147 
148   void checkTableInfo(TableInfoModel model) {
149     assertEquals(model.getName(), TABLE);
150     Iterator<TableRegionModel> regions = model.getRegions().iterator();
151     assertTrue(regions.hasNext());
152     while (regions.hasNext()) {
153       TableRegionModel region = regions.next();
154       boolean found = false;
155       for (Map.Entry<HRegionInfo, ServerName> e: regionMap.entrySet()) {
156         HRegionInfo hri = e.getKey();
157         String hriRegionName = hri.getRegionNameAsString();
158         String regionName = region.getName();
159         if (hriRegionName.equals(regionName)) {
160           found = true;
161           byte[] startKey = hri.getStartKey();
162           byte[] endKey = hri.getEndKey();
163           InetSocketAddress sa = new InetSocketAddress(e.getValue().getHostname(), e.getValue().getPort());
164           String location = sa.getHostName() + ":" +
165             Integer.valueOf(sa.getPort());
166           assertEquals(hri.getRegionId(), region.getId());
167           assertTrue(Bytes.equals(startKey, region.getStartKey()));
168           assertTrue(Bytes.equals(endKey, region.getEndKey()));
169           assertEquals(location, region.getLocation());
170           break;
171         }
172       }
173       assertTrue(found);
174     }
175   }
