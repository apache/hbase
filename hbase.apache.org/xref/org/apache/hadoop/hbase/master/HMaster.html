201 /**
202  * HMaster is the "master server" for HBase. An HBase cluster has one active
203  * master.  If many masters are started, all compete.  Whichever wins goes on to
204  * run the cluster.  All others park themselves in their constructor until
205  * master or cluster shutdown or until the active master loses its lease in
206  * zookeeper.  Thereafter, all running master jostle to take over master role.
207  *
208  * <p>The Master can be asked shutdown the cluster. See {@link #shutdown()}.  In
209  * this case it will tell all regionservers to go down and then wait on them
210  * all reporting in that they are down.  This master will then shut itself down.
211  *
212  * <p>You can also shutdown just this master.  Call {@link #stopMaster()}.
213  *
214  * @see MasterMonitorProtocol
215  * @see MasterAdminProtocol
216  * @see RegionServerStatusProtocol
217  * @see Watcher
218  */

323   /**
324    * Initializes the HMaster. The steps are as follows:
325    * <p>
326    * <ol>
327    * <li>Initialize HMaster RPC and address
328    * <li>Connect to ZooKeeper.
329    * </ol>
330    * <p>
331    * Remaining steps of initialization occur in {@link #run()} so that they
332    * run in their own thread rather than within the context of the constructor.
333    * @throws InterruptedException
334    */
335   public HMaster(final Configuration conf)
336   throws IOException, KeeperException, InterruptedException {
337     this.conf = new Configuration(conf);
338     // Disable the block cache on the master
339     this.conf.setFloat(HConstants.HFILE_BLOCK_CACHE_SIZE_KEY, 0.0f);
340     // Set how many times to retry talking to another server over HConnection.
341     HConnectionManager.setServerSideHConnectionRetries(this.conf, LOG);
342     // Server to handle client requests.
343     String hostname = Strings.domainNamePointerToHostName(DNS.getDefaultHost(
344       conf.get("hbase.master.dns.interface", "default"),
345       conf.get("hbase.master.dns.nameserver", "default")));
346     int port = conf.getInt(HConstants.MASTER_PORT, HConstants.DEFAULT_MASTER_PORT);
347     // Creation of a ISA will force a resolve.
348     InetSocketAddress initialIsa = new InetSocketAddress(hostname, port);
349     if (initialIsa.getAddress() == null) {
350       throw new IllegalArgumentException("Failed resolve of " + initialIsa);
351     }
352     int numHandlers = conf.getInt("hbase.master.handler.count",
353       conf.getInt("hbase.regionserver.handler.count", 25));
354     this.rpcServer = HBaseRPC.getServer(MasterMonitorProtocol.class, this,
355       new Class<?>[]{MasterMonitorProtocol.class,
356         MasterAdminProtocol.class, RegionServerStatusProtocol.class},
357         initialIsa.getHostName(), // BindAddress is IP we got for this server.
358         initialIsa.getPort(),
359         numHandlers,
360         0, // we dont use high priority handlers in master
361         conf.getBoolean("hbase.rpc.verbose", false), conf,
362         0); // this is a DNC w/o high priority handlers
363     // Set our address.
364     this.isa = this.rpcServer.getListenerAddress();
365     this.serverName = new ServerName(this.isa.getHostName(),
366       this.isa.getPort(), System.currentTimeMillis());
367     this.rsFatals = new MemoryBoundedLogMessageBuffer(
368         conf.getLong("hbase.master.buffer.for.rs.fatals", 1*1024*1024));
369 
370     // login the zookeeper client principal (if using security)
371     ZKUtil.loginClient(this.conf, "hbase.zookeeper.client.keytab.file",
372       "hbase.zookeeper.client.kerberos.principal", this.isa.getHostName());
373 
374     // initialize server principal (if using secure Hadoop)
375     User.login(conf, "hbase.master.keytab.file",
376       "hbase.master.kerberos.principal", this.isa.getHostName());
377 
378     LOG.info("hbase.rootdir=" + FSUtils.getRootDir(this.conf) +
379         ", hbase.cluster.distributed=" + this.conf.getBoolean("hbase.cluster.distributed", false));
380 
381     // set the thread name now we have an address
382     setName(MASTER + "-" + this.serverName.toString());
383 
384     Replication.decorateMasterConfiguration(this.conf);
385 
386     // Hack! Maps DFSClient => Master for logs.  HDFS made this
387     // config param for task trackers, but we can piggyback off of it.
388     if (this.conf.get("mapred.task.id") == null) {
389       this.conf.set("mapred.task.id", "hb_m_" + this.serverName.toString());
390     }
391 
392     this.zooKeeper = new ZooKeeperWatcher(conf, MASTER + ":" + isa.getPort(), this, true);
393     this.rpcServer.startThreads();
394 
395     // metrics interval: using the same property as region server.
396     this.msgInterval = conf.getInt("hbase.regionserver.msginterval", 3 * 1000);
397 
398     //should we check the compression codec type at master side, default true, HBASE-6370
399     this.masterCheckCompression = conf.getBoolean("hbase.master.check.compression", true);
400 
401     this.metricsMaster = new MetricsMaster( new MetricsMasterWrapperImpl(this));
402   }

404   /**
405    * Stall startup if we are designated a backup master; i.e. we want someone
406    * else to become the master before proceeding.
407    * @param c
408    * @param amm
409    * @throws InterruptedException
410    */
411   private static void stallIfBackupMaster(final Configuration c,
412       final ActiveMasterManager amm)
413   throws InterruptedException {
414     // If we're a backup master, stall until a primary to writes his address
415     if (!c.getBoolean(HConstants.MASTER_TYPE_BACKUP,
416       HConstants.DEFAULT_MASTER_TYPE_BACKUP)) {
417       return;
418     }
419     LOG.debug("HMaster started in backup mode.  " +
420       "Stalling until master znode is written.");
421     // This will only be a minute or so while the cluster starts up,
422     // so don't worry about setting watches on the parent znode
423     while (!amm.isActiveMaster()) {
424       LOG.debug("Waiting for master address ZNode to be written " +
425         "(Also watching cluster state node)");
426       Thread.sleep(c.getInt("zookeeper.session.timeout", 180 * 1000));
427     }

435   /**
436    * Main processing loop for the HMaster.
437    * <ol>
438    * <li>Block until becoming active master
439    * <li>Finish initialization via finishInitialization(MonitoredTask)
440    * <li>Enter loop until we are stopped
441    * <li>Stop services and perform cleanup once stopped
442    * </ol>
443    */
444   @Override
445   public void run() {
446     MonitoredTask startupStatus =
447       TaskMonitor.get().createStatus("Master startup");
448     startupStatus.setDescription("Master startup");
449     masterStartTime = System.currentTimeMillis();
450     try {
451       /*
452        * Block on becoming the active master.
453        *
454        * We race with other masters to write our address into ZooKeeper.  If we
455        * succeed, we are the primary/active master and finish initialization.
456        *
457        * If we do not succeed, there is another active master and we should
458        * now wait until it dies to try and become the next active master.  If we
459        * do not succeed on our first attempt, this is no longer a cluster startup.
460        */
461       becomeActiveMaster(startupStatus);
462 
463       // We are either the active master or we were asked to shutdown
464       if (!this.stopped) {
465         finishInitialization(startupStatus, false);
466         loop();
467       }
468     } catch (Throwable t) {
469       // HBASE-5680: Likely hadoop23 vs hadoop 20.x/1.x incompatibility
470       if (t instanceof NoClassDefFoundError &&
471           t.getMessage().contains("org/apache/hadoop/hdfs/protocol/FSConstants$SafeModeAction")) {
472           // improved error message for this special case
473           abort("HBase is having a problem with its Hadoop jars.  You may need to "
474               + "recompile HBase against Hadoop version "
475               +  org.apache.hadoop.util.VersionInfo.getVersion()
476               + " or change your hadoop jars to start properly", t);
477       } else {
478         abort("Unhandled exception. Starting shutdown.", t);
479       }
480     } finally {
481       startupStatus.cleanup();
482 
483       stopChores();
484       // Wait for all the remaining region servers to report in IFF we were
485       // running a cluster shutdown AND we were NOT aborting.
486       if (!this.abort && this.serverManager != null &&
487           this.serverManager.isClusterShutdown()) {
488         this.serverManager.letRegionServersShutdown();
489       }
490       stopServiceThreads();
491       // Stop services started for both backup and active masters
492       if (this.activeMasterManager != null) this.activeMasterManager.stop();
493       if (this.catalogTracker != null) this.catalogTracker.stop();
494       if (this.serverManager != null) this.serverManager.stop();
495       if (this.assignmentManager != null) this.assignmentManager.stop();
496       if (this.fileSystemManager != null) this.fileSystemManager.stop();
497       this.zooKeeper.close();
498     }
499     LOG.info("HMaster main thread exiting");
500   }

502   /**
503    * Try becoming active master.
504    * @param startupStatus
505    * @return True if we could successfully become the active master.
506    * @throws InterruptedException
507    */
508   private boolean becomeActiveMaster(MonitoredTask startupStatus)
509   throws InterruptedException {
510     // TODO: This is wrong!!!! Should have new servername if we restart ourselves,
511     // if we come back to life.
512     this.activeMasterManager = new ActiveMasterManager(zooKeeper, this.serverName,
513         this);
514     this.zooKeeper.registerListener(activeMasterManager);
515     stallIfBackupMaster(this.conf, this.activeMasterManager);
516 
517     // The ClusterStatusTracker is setup before the other
518     // ZKBasedSystemTrackers because it's needed by the activeMasterManager

610   /**
611    * Finish initialization of HMaster after becoming the primary master.
612    *
613    * <ol>
614    * <li>Initialize master components - file system manager, server manager,
615    *     assignment manager, region server tracker, catalog tracker, etc</li>
616    * <li>Start necessary service threads - rpc server, info server,
617    *     executor services, etc</li>
618    * <li>Set cluster as UP in ZooKeeper</li>
619    * <li>Wait for RegionServers to check-in</li>
620    * <li>Split logs and perform data recovery, if necessary</li>
621    * <li>Ensure assignment of root and meta regions<li>
622    * <li>Handle either fresh cluster start or master failover</li>
623    * </ol>
624    *
625    * @param masterRecovery
626    *
627    * @throws IOException
628    * @throws InterruptedException
629    * @throws KeeperException
630    */
631   private void finishInitialization(MonitoredTask status, boolean masterRecovery)
632   throws IOException, InterruptedException, KeeperException {
633 
634     isActiveMaster = true;
635 
636     /*
637      * We are active master now... go initialize components we need to run.
638      * Note, there may be dross in zk from previous runs; it'll get addressed
639      * below after we determine if cluster startup or failover.
640      */
641 
642     status.setStatus("Initializing Master file system");
643     this.masterActiveTime = System.currentTimeMillis();
644     // TODO: Do this using Dependency Injection, using PicoContainer, Guice or Spring.
645     this.fileSystemManager = new MasterFileSystem(this, this, metricsMaster, masterRecovery);
646 
647     this.tableDescriptors =
648       new FSTableDescriptors(this.fileSystemManager.getFileSystem(),
649       this.fileSystemManager.getRootDir());
650 
651     // publish cluster ID
652     status.setStatus("Publishing Cluster ID in ZooKeeper");
653     ZKClusterId.setClusterId(this.zooKeeper, fileSystemManager.getClusterId());
654 
655     if (!masterRecovery) {
656       this.executorService = new ExecutorService(getServerName().toString());
657       this.serverManager = createServerManager(this, this);
658     }
659 
660     status.setStatus("Initializing ZK system trackers");
661     initializeZKBasedSystemTrackers();
662 
663     if (!masterRecovery) {
664       // initialize master side coprocessors before we start handling requests
665       status.setStatus("Initializing master coprocessors");
666       this.cpHost = new MasterCoprocessorHost(this, this.conf);
667 
668       spanReceiverHost = new SpanReceiverHost(getConfiguration());
669       spanReceiverHost.loadSpanReceivers();
670 
671       // start up all service threads.
672       status.setStatus("Initializing master service threads");
673       startServiceThreads();
674     }
675 
676     // Wait for region servers to report in.
677     this.serverManager.waitForRegionServers(status);
678     // Check zk for region servers that are up but didn't register
679     for (ServerName sn: this.regionServerTracker.getOnlineServers()) {
680       if (!this.serverManager.isServerOnline(sn)) {
681         // Not registered; add it.
682         LOG.info("Registering server found up in zk but who has not yet " +
683           "reported in: " + sn);
684         this.serverManager.recordNewServer(sn, ServerLoad.EMPTY_SERVERLOAD);
685       }
686     }
687 
688     if (!masterRecovery) {
689       this.assignmentManager.startTimeOutMonitor();
690     }
691 
692     // TODO: Should do this in background rather than block master startup
693     status.setStatus("Splitting logs after master startup");
694     splitLogAfterStartup(this.fileSystemManager);
695 
696     // Make sure root and meta assigned before proceeding.
697     if (!assignRootAndMeta(status)) return;
698     enableServerShutdownHandler();
699 
700     // Update meta with new PB serialization if required. i.e migrate all HRI
701     // to PB serialization in meta and update the status in ROOT. This must happen
702     // before we assign all user regions or else the assignment will fail.
703     // TODO: Remove this after 0.96, when we do 0.98.
704     org.apache.hadoop.hbase.catalog.MetaMigrationConvertingToPB
705       .updateRootAndMetaIfNecessary(this);
706 
707     this.balancer.setMasterServices(this);
708     // Fix up assignment manager status
709     status.setStatus("Starting assignment manager");
710     this.assignmentManager.joinCluster();
711 
712     this.balancer.setClusterStatus(getClusterStatus());
713 
714     // Fixing up missing daughters if any
715     status.setStatus("Fixing up missing daughters");
716     fixupDaughters(status);
717 
718     if (!masterRecovery) {
719       // Start balancer and meta catalog janitor after meta and regions have
720       // been assigned.
721       status.setStatus("Starting balancer and catalog janitor");
722       this.clusterStatusChore = getAndStartClusterStatusChore(this);
723       this.balancerChore = getAndStartBalancerChore(this);
724       this.catalogJanitorChore = new CatalogJanitor(this, this);
725       startCatalogJanitorChore();
726     }
727 
728     status.markComplete("Initialization successful");
729     LOG.info("Master has completed initialization");
730     initialized = true;
731     // clear the dead servers with same host name and port of online server because we are not
732     // removing dead server with same hostname and port of rs which is trying to check in before
733     // master initialization. See HBASE-5916.
734     this.serverManager.clearDeadServersWithSameHostNameAndPortOfOnlineServer();
735 
736     if (!masterRecovery) {
737       if (this.cpHost != null) {
738         // don't let cp initialization errors kill the master
739         try {
740           this.cpHost.postStartMaster();
741         } catch (IOException ioe) {
742           LOG.error("Coprocessor postStartMaster() hook failed", ioe);
743         }
744       }
745     }
746   }

791   /**
792    * Check <code>-ROOT-</code> and <code>.META.</code> are assigned.  If not,
793    * assign them.
794    * @throws InterruptedException
795    * @throws IOException
796    * @throws KeeperException
797    * @return True if root and meta are healthy, assigned
798    */
799   boolean assignRootAndMeta(MonitoredTask status)
800   throws InterruptedException, IOException, KeeperException {
801     int assigned = 0;
802     long timeout = this.conf.getLong("hbase.catalog.verification.timeout", 1000);
803 
804     // Work on ROOT region.  Is it in zk in transition?
805     status.setStatus("Assigning ROOT region");
806     assignmentManager.getRegionStates().createRegionState(
807       HRegionInfo.ROOT_REGIONINFO);
808     boolean rit = this.assignmentManager.
809       processRegionInTransitionAndBlockUntilAssigned(HRegionInfo.ROOT_REGIONINFO);
810     ServerName currentRootServer = null;
811     boolean rootRegionLocation = catalogTracker.verifyRootRegionLocation(timeout);
812     if (!rit && !rootRegionLocation) {
813       currentRootServer = this.catalogTracker.getRootLocation();
814       splitLogAndExpireIfOnline(currentRootServer);
815       this.assignmentManager.assignRoot();
816       // Make sure a -ROOT- location is set.
817       if (!isRootLocation()) return false;
818       // This guarantees that the transition assigning -ROOT- has completed
819       this.assignmentManager.waitForAssignment(HRegionInfo.ROOT_REGIONINFO);
820       assigned++;
821     } else if (rit && !rootRegionLocation) {
822       // Make sure a -ROOT- location is set.
823       if (!isRootLocation()) return false;
824       // This guarantees that the transition assigning -ROOT- has completed
825       this.assignmentManager.waitForAssignment(HRegionInfo.ROOT_REGIONINFO);
826       assigned++;
827     } else if (rootRegionLocation) {
828       // Region already assigned.  We didn't assign it.  Add to in-memory state.
829       this.assignmentManager.regionOnline(HRegionInfo.ROOT_REGIONINFO,
830         this.catalogTracker.getRootLocation());
831     }
832     // Enable the ROOT table if on process fail over the RS containing ROOT
833     // was active.
834     enableCatalogTables(Bytes.toString(HConstants.ROOT_TABLE_NAME));
835     // Check for stopped, just in case
836     if (this.stopped) return false;
837     LOG.info("-ROOT- assigned=" + assigned + ", rit=" + rit +
838       ", location=" + catalogTracker.getRootLocation());
839 
840     // Work on meta region
841     status.setStatus("Assigning META region");
842     assignmentManager.getRegionStates().createRegionState(
843       HRegionInfo.FIRST_META_REGIONINFO);
844     rit = this.assignmentManager.
845       processRegionInTransitionAndBlockUntilAssigned(HRegionInfo.FIRST_META_REGIONINFO);
846     boolean metaRegionLocation = this.catalogTracker.verifyMetaRegionLocation(timeout);
847     if (!rit && !metaRegionLocation) {
848       ServerName currentMetaServer =
849         this.catalogTracker.getMetaLocationOrReadLocationFromRoot();
850       if (currentMetaServer != null
851           && !currentMetaServer.equals(currentRootServer)) {
852         splitLogAndExpireIfOnline(currentMetaServer);
853       }
854       assignmentManager.assignMeta();
855       enableSSHandWaitForMeta();
856       assigned++;
857     } else if (rit && !metaRegionLocation) {
858       // Wait until META region added to region server onlineRegions. See HBASE-5875.
859       enableSSHandWaitForMeta();
860       assigned++;
861     } else {
862       // Region already assigned.  We didn't assign it.  Add to in-memory state.
863       this.assignmentManager.regionOnline(HRegionInfo.FIRST_META_REGIONINFO,
864         this.catalogTracker.getMetaLocation());
865     }
866     enableCatalogTables(Bytes.toString(HConstants.META_TABLE_NAME));
867     LOG.info(".META. assigned=" + assigned + ", rit=" + rit +
868       ", location=" + catalogTracker.getMetaLocation());
869     status.setStatus("META and ROOT assigned.");
870     return true;
871   }

905   void fixupDaughters(final MonitoredTask status) throws IOException {
906     final Map<HRegionInfo, Result> offlineSplitParents =
907       new HashMap<HRegionInfo, Result>();
908     // This visitor collects offline split parents in the .META. table
909     MetaReader.Visitor visitor = new MetaReader.Visitor() {
910       @Override
911       public boolean visit(Result r) throws IOException {
912         if (r == null || r.isEmpty()) return true;
913         HRegionInfo info =
914           HRegionInfo.getHRegionInfo(r);
915         if (info == null) return true; // Keep scanning
916         if (info.isOffline() && info.isSplit()) {
917           offlineSplitParents.put(info, r);
918         }
919         // Returning true means "keep scanning"
920         return true;
921       }
922     };
923     // Run full scan of .META. catalog table passing in our custom visitor
924     MetaReader.fullScan(this.catalogTracker, visitor);
925     // Now work on our list of found parents. See if any we can clean up.
926     int fixups = 0;
927     for (Map.Entry<HRegionInfo, Result> e : offlineSplitParents.entrySet()) {
928       ServerName sn = HRegionInfo.getServerName(e.getValue());
929       if (!serverManager.isServerDead(sn)) { // Otherwise, let SSH take care of it
930         fixups += ServerShutdownHandler.fixupDaughters(
931           e.getValue(), assignmentManager, catalogTracker);
932       }
933     }
934     if (fixups != 0) {
935       LOG.info("Scanned the catalog and fixed up " + fixups +
936         " missing daughter region(s)");
937     }
938   }

1032       conf.getInt("hbase.master.executor.openregion.threads", 5));

1038       conf.getInt("hbase.master.executor.serverops.threads", 5));

1041    // at a time.  To do concurrency, would need fencing of enable/disable of
1042    // tables.
1043    this.executorService.startExecutorService(ExecutorType.MASTER_TABLE_OPERATIONS, 1);

1045    // Start log cleaner thread
1046    String n = Thread.currentThread().getName();
1047    int cleanerInterval = conf.getInt("hbase.master.cleaner.interval", 60 * 1000);

1061    int port = this.conf.getInt(HConstants.MASTER_INFO_PORT, 60010);

1069     }
1070 
1071     // Start allowing requests to happen.
1072     this.rpcServer.openServer();
1073     this.rpcServerOpen = true;
1074     if (LOG.isDebugEnabled()) {
1075       LOG.debug("Started service threads");
1076     }
1077   }

1274   public boolean balance() throws IOException {
1275     // if master not initialized, don't run balancer.
1276     if (!this.initialized) {
1277       LOG.debug("Master has not been initialized, don't run balancer.");
1278       return false;
1279     }
1280     // If balance not true, don't run balancer.
1281     if (!this.loadBalancerTracker.isBalancerOn()) return false;
1282     // Do this call outside of synchronized block.
1283     int maximumBalanceTime = getBalancerCutoffTime();
1284     long cutoffTime = System.currentTimeMillis() + maximumBalanceTime;
1285     boolean balancerRan;
1286     synchronized (this.balancer) {
1287       // Only allow one balance run at at time.
1288       if (this.assignmentManager.getRegionStates().isRegionsInTransition()) {
1289         Map<String, RegionState> regionsInTransition =
1290           this.assignmentManager.getRegionStates().getRegionsInTransition();
1291         LOG.debug("Not running balancer because " + regionsInTransition.size() +
1292           " region(s) in transition: " + org.apache.commons.lang.StringUtils.
1293             abbreviate(regionsInTransition.toString(), 256));
1294         return false;
1295       }
1296       if (this.serverManager.areDeadServersInProgress()) {
1297         LOG.debug("Not running balancer because processing dead regionserver(s): " +
1298           this.serverManager.getDeadServers());
1299         return false;
1300       }
1301 
1302       if (this.cpHost != null) {
1303         if (this.cpHost.preBalance()) {
1304           LOG.debug("Coprocessor bypassing balancer request");
1305           return false;
1306         }
1307       }
1308 
1309       Map<String, Map<ServerName, List<HRegionInfo>>> assignmentsByTable =
1310         this.assignmentManager.getRegionStates().getAssignmentsByTable();
1311 
1312       List<RegionPlan> plans = new ArrayList<RegionPlan>();
1313       //Give the balancer the current cluster state.
1314       this.balancer.setClusterStatus(getClusterStatus());
1315       for (Map<ServerName, List<HRegionInfo>> assignments : assignmentsByTable.values()) {
1316         List<RegionPlan> partialPlans = this.balancer.balanceCluster(assignments);
1317         if (partialPlans != null) plans.addAll(partialPlans);
1318       }
1319       int rpCount = 0;  // number of RegionPlans balanced so far
1320       long totalRegPlanExecTime = 0;
1321       balancerRan = plans != null;
1322       if (plans != null && !plans.isEmpty()) {
1323         for (RegionPlan plan: plans) {
1324           LOG.info("balance " + plan);
1325           long balStartTime = System.currentTimeMillis();
1326           this.assignmentManager.balance(plan);
1327           totalRegPlanExecTime += System.currentTimeMillis()-balStartTime;
1328           rpCount++;
1329           if (rpCount < plans.size() &&
1330               // if performing next balance exceeds cutoff time, exit the loop
1331               (System.currentTimeMillis() + (totalRegPlanExecTime / rpCount)) > cutoffTime) {
1332             LOG.debug("No more balancing till next balance run; maximumBalanceTime=" +
1333               maximumBalanceTime);
1334             break;
1335           }
1336         }
1337       }
1338       if (this.cpHost != null) {
1339         this.cpHost.postBalance(rpCount < plans.size() ? plans.subList(0, rpCount) : plans);
1340       }
1341     }
1342     return balancerRan;
1343   }

1939   /**
1940    * We do the following in a different thread.  If it is not completed
1941    * in time, we will time it out and assume it is not easy to recover.
1942    *
1943    * 1. Create a new ZK session. (since our current one is expired)
1944    * 2. Try to become a primary master again
1945    * 3. Initialize all ZK based system trackers.
1946    * 4. Assign root and meta. (they are already assigned, but we need to update our
1947    * internal memory state to reflect it)
1948    * 5. Process any RIT if any during the process of our recovery.
1949    *
1950    * @return True if we could successfully recover from ZK session expiry.
1951    * @throws InterruptedException
1952    * @throws IOException
1953    * @throws KeeperException
1954    * @throws ExecutionException
1955    */
1956   private boolean tryRecoveringExpiredZKSession() throws InterruptedException,
1957       IOException, KeeperException, ExecutionException {
1958 
1959     this.zooKeeper.reconnectAfterExpiration();
1960 
1961     Callable<Boolean> callable = new Callable<Boolean> () {
1962       public Boolean call() throws InterruptedException,
1963           IOException, KeeperException {
1964         MonitoredTask status =
1965           TaskMonitor.get().createStatus("Recovering expired ZK session");
1966         try {
1967           if (!becomeActiveMaster(status)) {
1968             return Boolean.FALSE;
1969           }
1970           serverShutdownHandlerEnabled = false;
1971           initialized = false;
1972           finishInitialization(status, true);
1973           return Boolean.TRUE;
1974         } finally {
1975           status.cleanup();
1976         }
1977       }
1978     };
1979 
1980     long timeout =
1981       conf.getLong("hbase.master.zksession.recover.timeout", 300000);
1982     java.util.concurrent.ExecutorService executor =
1983       Executors.newSingleThreadExecutor();
1984     Future<Boolean> result = executor.submit(callable);
1985     executor.shutdown();
1986     if (executor.awaitTermination(timeout, TimeUnit.MILLISECONDS)
1987         && result.isDone()) {
1988       Boolean recovered = result.get();
1989       if (recovered != null) {
1990         return recovered.booleanValue();
1991       }
1992     }
1993     executor.shutdownNow();
1994     return false;
1995   }
