67  /**
68   * The ServerManager class manages info about region servers.
69   * <p>
70   * Maintains lists of online and dead servers.  Processes the startups,
71   * shutdowns, and deaths of region servers.
72   * <p>
73   * Servers are distinguished in two different ways.  A given server has a
74   * location, specified by hostname and port, and of which there can only be one
75   * online at any given time.  A server instance is specified by the location
76   * (hostname and port) as well as the startcode (timestamp from when the server
77   * was started).  This is used to differentiate a restarted instance of a given
78   * server from the original instance.
79   * <p>
80   * If a sever is known not to be running any more, it is called dead. The dead
81   * server needs to be handled by a ServerShutdownHandler.  If the handler is not
82   * enabled yet, the server can't be handled right away so it is queued up.
83   * After the handler is enabled, the server will be submitted to a handler to handle.
84   * However, the handler may be just partially enabled.  If so,
85   * the server cannot be fully processed, and be queued up for further processing.
86   * A server is fully processed only after the handler is fully enabled
87   * and has completed the handling.
88   */

139    * Set of region servers which are dead but not processed immediately. If one
140    * server died before master enables ServerShutdownHandler, the server will be
141    * added to this set and will be processed through calling
142    * {@link ServerManager#processQueuedDeadServers()} by master.
143    * <p>
144    * A dead server is a server instance known to be dead, not listed in the /hbase/rs
145    * znode any more. It may have not been submitted to ServerShutdownHandler yet
146    * because the handler is not enabled.
147    * <p>
148    * A dead server, which has been submitted to ServerShutdownHandler while the
149    * handler is not enabled, is queued up.

152    * ServerShutdownHander for processing yet.

195   /**
196    * Let the server manager know a new regionserver has come online
197    * @param ia The remote address
198    * @param port The remote port
199    * @param serverStartcode
200    * @param serverCurrentTime The current time of the region server in ms
201    * @return The ServerName we know this server as.
202    * @throws IOException
203    */
204   ServerName regionServerStartup(final InetAddress ia, final int port,
205     final long serverStartcode, long serverCurrentTime)
206   throws IOException {
207     // Test for case where we get a region startup message from a regionserver
208     // that has been quickly restarted but whose znode expiration handler has
209     // not yet run, or from a server whose fail we are currently processing.
210     // Test its host+port combo is present in serverAddresstoServerInfo.  If it
211     // is, reject the server and trigger its expiration. The next time it comes
212     // in, it should have been removed from serverAddressToServerInfo and queued
213     // for processing by ProcessServerShutdown.
214     ServerName sn = new ServerName(ia.getHostName(), port, serverStartcode);
215     checkClockSkew(sn, serverCurrentTime);
216     checkIsDead(sn, "STARTUP");
217     checkAlreadySameHostPort(sn);
218     recordNewServer(sn, ServerLoad.EMPTY_SERVERLOAD);
219     return sn;
220   }

222   /**
223    * Updates last flushed sequence Ids for the regions on server sn
224    * @param sn
225    * @param hsl
226    */
227   private void updateLastFlushedSequenceIds(ServerName sn, ServerLoad hsl) {
228     Map<byte[], RegionLoad> regionsLoad = hsl.getRegionsLoad();
229     for (Entry<byte[], RegionLoad> entry : regionsLoad.entrySet()) {
230       Long existingValue = flushedSequenceIdByRegion.get(entry.getKey());
231       long l = entry.getValue().getCompleteSequenceId();
232       if (existingValue != null) {
233         if (l != -1 && l < existingValue) {
234           if (LOG.isDebugEnabled()) {
235             LOG.debug("RegionServer " + sn +
236                 " indicates a last flushed sequence id (" + entry.getValue() +
237                 ") that is less than the previous last flushed sequence id (" +
238                 existingValue + ") for region " +
239                 Bytes.toString(entry.getKey()) + " Ignoring.");
240           }
241           continue; // Don't let smaller sequence ids override greater
242           // sequence ids.
243         }
244       }
245       flushedSequenceIdByRegion.put(entry.getKey(), l);
246     }
247   }

249   void regionServerReport(ServerName sn, ServerLoad sl)
250   throws YouAreDeadException, PleaseHoldException {
251     checkIsDead(sn, "REPORT");
252     if (!this.onlineServers.containsKey(sn)) {
253       // Already have this host+port combo and its just different start code?
254       checkAlreadySameHostPort(sn);
255       // Just let the server in. Presume master joining a running cluster.
256       // recordNewServer is what happens at the end of reportServerStartup.
257       // The only thing we are skipping is passing back to the regionserver
258       // the ServerName to use. Here we presume a master has already done
259       // that so we'll press on with whatever it gave us for ServerName.
260       recordNewServer(sn, sl);
261     } else {
262       this.onlineServers.put(sn, sl);
263     }
264     updateLastFlushedSequenceIds(sn, sl);
265   }

267   /**
268    * Test to see if we have a server of same host and port already.
269    * @param serverName
270    * @throws PleaseHoldException
271    */
272   void checkAlreadySameHostPort(final ServerName serverName)
273   throws PleaseHoldException {
274     ServerName existingServer =
275       ServerName.findServerWithSameHostnamePort(getOnlineServersList(), serverName);
276     if (existingServer != null) {
277       String message = "Server serverName=" + serverName +
278         " rejected; we already have " + existingServer.toString() +
279         " registered with same hostname and port";
280       LOG.info(message);
281       if (existingServer.getStartcode() < serverName.getStartcode()) {
282         LOG.info("Triggering server recovery; existingServer " +
283           existingServer + " looks stale, new server:" + serverName);
284         expireServer(existingServer);
285       }
286       if (services.isServerShutdownHandlerEnabled()) {
287         // master has completed the initialization
288         throw new PleaseHoldException(message);
289       }
290     }
291   }

293   /**
294    * Checks if the clock skew between the server and the master. If the clock skew exceeds the
295    * configured max, it will throw an exception; if it exceeds the configured warning threshold,
296    * it will log a warning but start normally.
297    * @param serverName Incoming servers's name
298    * @param serverCurrentTime
299    * @throws ClockOutOfSyncException if the skew exceeds the configured max value
300    */
301   private void checkClockSkew(final ServerName serverName, final long serverCurrentTime)
302   throws ClockOutOfSyncException {
303     long skew = System.currentTimeMillis() - serverCurrentTime;
304     if (skew > maxSkew) {
305       String message = "Server " + serverName + " has been " +
306         "rejected; Reported time is too far out of sync with master.  " +
307         "Time difference of " + skew + "ms > max allowed of " + maxSkew + "ms";
308       LOG.warn(message);
309       throw new ClockOutOfSyncException(message);
310     } else if (skew > warningSkew){
311       String message = "Reported time for server " + serverName + " is out of sync with master " +
312         "by " + skew + "ms. (Warning threshold is " + warningSkew + "ms; " +
313         "error threshold is " + maxSkew + "ms)";
314       LOG.warn(message);
315     }
316   }

318   /**
319    * If this server is on the dead list, reject it with a YouAreDeadException.
320    * If it was dead but came back with a new start code, remove the old entry
321    * from the dead list.
322    * @param serverName
323    * @param what START or REPORT
324    * @throws YouAreDeadException
325    */
326   private void checkIsDead(final ServerName serverName, final String what)
327       throws YouAreDeadException {
328     if (this.deadservers.isDeadServer(serverName)) {
329       // host name, port and start code all match with existing one of the
330       // dead servers. So, this server must be dead.
331       String message = "Server " + what + " rejected; currently processing " +
332           serverName + " as dead server";
333       LOG.debug(message);
334       throw new YouAreDeadException(message);
335     }
336     // remove dead server with same hostname and port of newly checking in rs after master
337     // initialization.See HBASE-5916 for more information.
338     if ((this.services == null || ((HMaster) this.services).isInitialized())
339         && this.deadservers.cleanPreviousInstance(serverName)) {
340       // This server has now become alive after we marked it as dead.
341       // We removed it's previous entry from the dead list to reflect it.
342       LOG.debug(what + ":" + " Server " + serverName + " came back up," +
343           " removed it from the dead servers list");
344     }
345   }

374   /**
375    * Compute the average load across all region servers.
376    * Currently, this uses a very naive computation - just uses the number of
377    * regions being served, ignoring stats about number of requests.
378    * @return the average load
379    */
380   public double getAverageLoad() {
381     int totalLoad = 0;
382     int numServers = 0;
383     double averageLoad = 0.0;
384     for (ServerLoad sl: this.onlineServers.values()) {
385         numServers++;
386         totalLoad += sl.getNumberOfRegions();
387     }
388     averageLoad = (double)totalLoad / (double)numServers;
389     return averageLoad;
390   }

420   void letRegionServersShutdown() {
421     long previousLogTime = 0;
422     while (!onlineServers.isEmpty()) {
423 
424       if (System.currentTimeMillis() > (previousLogTime + 1000)) {
425         StringBuilder sb = new StringBuilder();
426         for (ServerName key : this.onlineServers.keySet()) {
427           if (sb.length() > 0) {
428             sb.append(", ");
429           }
430           sb.append(key);
431         }
432         LOG.info("Waiting on regionserver(s) to go down " + sb.toString());
433         previousLogTime = System.currentTimeMillis();
434       }
435 
436       synchronized (onlineServers) {
437         try {
438           onlineServers.wait(100);
439         } catch (InterruptedException ignored) {
440           // continue
441         }
442       }
443     }
444   }

446   /*
447    * Expire the passed server.  Add it to list of dead servers and queue a
448    * shutdown processing.
449    */
450   public synchronized void expireServer(final ServerName serverName) {
451     if (!services.isServerShutdownHandlerEnabled()) {
452       LOG.info("Master doesn't enable ServerShutdownHandler during initialization, "
453           + "delay expiring server " + serverName);
454       this.queuedDeadServers.add(serverName);
455       return;
456     }
457     if (!this.onlineServers.containsKey(serverName)) {
458       LOG.warn("Received expiration of " + serverName +
459         " but server is not currently online");
460     }
461     if (this.deadservers.contains(serverName)) {
462       // TODO: Can this happen?  It shouldn't be online in this case?
463       LOG.warn("Received expiration of " + serverName +
464           " but server shutdown is already in progress");
465       return;
466     }
467     // Remove the server from the known servers lists and update load info BUT
468     // add to deadservers first; do this so it'll show in dead servers list if
469     // not in online servers list.
470     this.deadservers.add(serverName);
471     this.onlineServers.remove(serverName);
472     synchronized (onlineServers) {
473       onlineServers.notifyAll();
474     }
475     this.serverConnections.remove(serverName);
476     // If cluster is going down, yes, servers are going to be expiring; don't
477     // process as a dead server
478     if (this.clusterShutdown) {
479       LOG.info("Cluster shutdown set; " + serverName +
480         " expired; onlineServers=" + this.onlineServers.size());
481       if (this.onlineServers.isEmpty()) {
482         master.stop("Cluster shutdown set; onlineServer=0");
483       }
484       return;
485     }
486 
487     boolean carryingRoot = services.getAssignmentManager().isCarryingRoot(serverName);
488     boolean carryingMeta = services.getAssignmentManager().isCarryingMeta(serverName);
489     if (carryingRoot || carryingMeta) {
490       this.services.getExecutorService().submit(new MetaServerShutdownHandler(this.master,
491         this.services, this.deadservers, serverName, carryingRoot, carryingMeta));
492     } else {
493       this.services.getExecutorService().submit(new ServerShutdownHandler(this.master,
494         this.services, this.deadservers, serverName, true));
495     }
496     LOG.debug("Added=" + serverName +
497       " to dead servers, submitted shutdown handler to be executed, root=" +
498         carryingRoot + ", meta=" + carryingMeta);
499   }

501   public synchronized void processDeadServer(final ServerName serverName) {
502     // When assignment manager is cleaning up the zookeeper nodes and rebuilding the
503     // in-memory region states, region servers could be down. Root/meta table can and
504     // should be re-assigned, log splitting can be done too. However, it is better to
505     // wait till the cleanup is done before re-assigning user regions.
506     //
507     // We should not wait in the server shutdown handler thread since it can clog
508     // the handler threads and root/meta table could not be re-assigned in case
509     // the corresponding server is down. So we queue them up here instead.
510     if (!services.getAssignmentManager().isFailoverCleanupDone()) {
511       requeuedDeadServers.add(serverName);
512       return;
513     }
514 
515     this.deadservers.add(serverName);
516     this.services.getExecutorService().submit(new ServerShutdownHandler(
517       this.master, this.services, this.deadservers, serverName, false));
518   }
519 
520   /**
521    * Process the servers which died during master's initialization. It will be
522    * called after HMaster#assignRootAndMeta and AssignmentManager#joinCluster.
523    * */
524   synchronized void processQueuedDeadServers() {
525     if (!services.isServerShutdownHandlerEnabled()) {
526       LOG.info("Master hasn't enabled ServerShutdownHandler");
527     }
528     Iterator<ServerName> serverIterator = queuedDeadServers.iterator();
529     while (serverIterator.hasNext()) {
530       expireServer(serverIterator.next());
531       serverIterator.remove();
532     }
533 
534     if (!services.getAssignmentManager().isFailoverCleanupDone()) {
535       LOG.info("AssignmentManager hasn't finished failover cleanup");
536     }
537     serverIterator = requeuedDeadServers.iterator();
538     while (serverIterator.hasNext()) {
539       processDeadServer(serverIterator.next());
540       serverIterator.remove();
541     }
542   }

544   /*
545    * Remove the server from the drain list.
546    */
547   public boolean removeServerFromDrainList(final ServerName sn) {
548     // Warn if the server (sn) is not online.  ServerName is of the form:
549     // <hostname> , <port> , <startcode>
550 
551     if (!this.isServerOnline(sn)) {
552       LOG.warn("Server " + sn + " is not currently online. " +
553                "Removing from draining list anyway, as requested.");
554     }
555     // Remove the server from the draining servers lists.
556     return this.drainingServers.remove(sn);
557   }
558 
559   /*
560    * Add the server to the drain list.
561    */
562   public boolean addServerToDrainList(final ServerName sn) {
563     // Warn if the server (sn) is not online.  ServerName is of the form:
564     // <hostname> , <port> , <startcode>
565 
566     if (!this.isServerOnline(sn)) {
567       LOG.warn("Server " + sn + " is not currently online. " +
568                "Ignoring request to add it to draining list.");
569       return false;
570     }
571     // Add the server to the draining servers lists, if it's not already in
572     // it.
573     if (this.drainingServers.contains(sn)) {
574       LOG.warn("Server " + sn + " is already in the draining server list." +
575                "Ignoring request to add it again.");
576       return false;
577     }
578     return this.drainingServers.add(sn);
579   }

583   /**
584    * Sends an OPEN RPC to the specified server to open the specified region.
585    * <p>
586    * Open should not fail but can if server just crashed.
587    * <p>
588    * @param server server to open a region
589    * @param region region to open
590    * @param versionOfOfflineNode that needs to be present in the offline node
591    * when RS tries to change the state from OFFLINE to other states.
592    */
593   public RegionOpeningState sendRegionOpen(final ServerName server,
594       HRegionInfo region, int versionOfOfflineNode)
595   throws IOException {
596     AdminProtocol admin = getServerConnection(server);
597     if (admin == null) {
598       LOG.warn("Attempting to send OPEN RPC to server " + server.toString() +
599         " failed because no RPC connection found to this server");
600       return RegionOpeningState.FAILED_OPENING;
601     }
602     OpenRegionRequest request =
603       RequestConverter.buildOpenRegionRequest(region, versionOfOfflineNode);
604     try {
605       OpenRegionResponse response = admin.openRegion(null, request);
606       return ResponseConverter.getRegionOpeningState(response);
607     } catch (ServiceException se) {
608       throw ProtobufUtil.getRemoteException(se);
609     }
610   }

612   /**
613    * Sends an OPEN RPC to the specified server to open the specified region.
614    * <p>
615    * Open should not fail but can if server just crashed.
616    * <p>
617    * @param server server to open a region
618    * @param regionOpenInfos info of a list of regions to open
619    * @return a list of region opening states
620    */
621   public List<RegionOpeningState> sendRegionOpen(ServerName server,
622       List<Pair<HRegionInfo, Integer>> regionOpenInfos)
623   throws IOException {
624     AdminProtocol admin = getServerConnection(server);
625     if (admin == null) {

692   /**
693    * Wait for the region servers to report in.
694    * We will wait until one of this condition is met:
695    *  - the master is stopped
696    *  - the 'hbase.master.wait.on.regionservers.maxtostart' number of
697    *    region servers is reached
698    *  - the 'hbase.master.wait.on.regionservers.mintostart' is reached AND
699    *   there have been no new region server in for
700    *      'hbase.master.wait.on.regionservers.interval' time AND
701    *   the 'hbase.master.wait.on.regionservers.timeout' is reached
702    *
703    * @throws InterruptedException
704    */
705   public void waitForRegionServers(MonitoredTask status)
706   throws InterruptedException {
707     final long interval = this.master.getConfiguration().
708       getLong(WAIT_ON_REGIONSERVERS_INTERVAL, 1500);
709     final long timeout = this.master.getConfiguration().
710       getLong(WAIT_ON_REGIONSERVERS_TIMEOUT, 4500);
711     int minToStart = this.master.getConfiguration().
712       getInt(WAIT_ON_REGIONSERVERS_MINTOSTART, 1);
713     if (minToStart < 1) {
714       LOG.warn(String.format(
715         "The value of '%s' (%d) can not be less than 1, ignoring.",
716         WAIT_ON_REGIONSERVERS_MINTOSTART, minToStart));
717       minToStart = 1;
718     }
719     int maxToStart = this.master.getConfiguration().
720       getInt(WAIT_ON_REGIONSERVERS_MAXTOSTART, Integer.MAX_VALUE);
721     if (maxToStart < minToStart) {
722         LOG.warn(String.format(
723             "The value of '%s' (%d) is set less than '%s' (%d), ignoring.",
724             WAIT_ON_REGIONSERVERS_MAXTOSTART, maxToStart,
725             WAIT_ON_REGIONSERVERS_MINTOSTART, minToStart));
726         maxToStart = Integer.MAX_VALUE;
727     }
728 
729     long now =  System.currentTimeMillis();
730     final long startTime = now;
731     long slept = 0;
732     long lastLogTime = 0;
733     long lastCountChange = startTime;
734     int count = countOfRegionServers();
735     int oldCount = 0;
736     while (
737       !this.master.isStopped() &&
738         count < maxToStart &&
739         (lastCountChange+interval > now || timeout > slept || count < minToStart)
740       ){
741 
742       // Log some info at every interval time or if there is a change
743       if (oldCount != count || lastLogTime+interval < now){
744         lastLogTime = now;
745         String msg =
746           "Waiting for region servers count to settle; currently"+
747             " checked in " + count + ", slept for " + slept + " ms," +
748             " expecting minimum of " + minToStart + ", maximum of "+ maxToStart+
749             ", timeout of "+timeout+" ms, interval of "+interval+" ms.";
750         LOG.info(msg);
751         status.setStatus(msg);
752       }
753 
754       // We sleep for some time
755       final long sleepTime = 50;
756       Thread.sleep(sleepTime);
757       now =  System.currentTimeMillis();
758       slept = now - startTime;
759 
760       oldCount = count;
761       count = countOfRegionServers();
762       if (count != oldCount) {
763         lastCountChange = now;
764       }
765     }
766 
767     LOG.info("Finished waiting for region servers count to settle;" +
768       " checked in " + count + ", slept for " + slept + " ms," +
769       " expecting minimum of " + minToStart + ", maximum of "+ maxToStart+","+
770       " master is "+ (this.master.isStopped() ? "stopped.": "running.")
771     );
772   }
