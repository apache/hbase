35  /**
36   * This is a Filter wrapper class which is used in the server side. Some filter
37   * related hooks can be defined in this wrapper. The only way to create a
38   * FilterWrapper instance is passing a client side Filter instance through
39   * {@link org.apache.hadoop.hbase.client.Scan#getFilter()}.
40   * 
41   */
42  @InterfaceAudience.Private
43  @InterfaceStability.Evolving
44  public class FilterWrapper extends Filter {
45    Filter filter = null;
46  
47    public FilterWrapper( Filter filter ) {
48      if (null == filter) {
49        // ensure the filter instance is not null
50        throw new NullPointerException("Cannot create FilterWrapper with null Filter");
51      }
52      this.filter = filter;
53    }
54  
55    /**
56     * @return The filter serialized using pb
57     */
58    public byte [] toByteArray() {
59      FilterProtos.FilterWrapper.Builder builder =
60        FilterProtos.FilterWrapper.newBuilder();
61      builder.setFilter(ProtobufUtil.toFilter(this.filter));
62      return builder.build().toByteArray();
63    }
64  
65    /**
66     * @param pbBytes A pb serialized {@link FilterWrapper} instance
67     * @return An instance of {@link FilterWrapper} made from <code>bytes</code>
68     * @throws DeserializationException
69     * @see #toByteArray
70     */
71    public static FilterWrapper parseFrom(final byte [] pbBytes)
72    throws DeserializationException {
73      FilterProtos.FilterWrapper proto;
74      try {
75        proto = FilterProtos.FilterWrapper.parseFrom(pbBytes);
76      } catch (InvalidProtocolBufferException e) {
77        throw new DeserializationException(e);
78      }
79      try {
80        return new FilterWrapper(ProtobufUtil.toFilter(proto.getFilter()));
81      } catch (IOException ioe) {
82        throw new DeserializationException(ioe);
83      }
84    }
85  
86    @Override
87    public void reset() {
88      this.filter.reset();
89    }

92    public boolean filterAllRemaining() {
93      return this.filter.filterAllRemaining();
94    }

124   }

126   @Override
127   public void filterRow(List<KeyValue> kvs) {
128     //To fix HBASE-6429, 
129     //Filter with filterRow() returning true is incompatible with scan with limit
130     //1. hasFilterRow() returns true, if either filterRow() or filterRow(kvs) is implemented.
131     //2. filterRow() is merged with filterRow(kvs),
132     //so that to make all those row related filtering stuff in the same function.
133     this.filter.filterRow(kvs);
134     if (!kvs.isEmpty() && this.filter.filterRow()) {
135       kvs.clear();
136     }
137   }

139   /**
140    * @param other
141    * @return true if and only if the fields of the filter that are serialized
142    * are equal to the corresponding fields in other.  Used for testing.
143    */
144   boolean areSerializedFieldsEqual(Filter o) {
145     if (o == this) return true;
146     if (!(o instanceof FilterWrapper)) return false;
147 
148     FilterWrapper other = (FilterWrapper)o;
149     return this.filter.areSerializedFieldsEqual(other.filter);
150   }
