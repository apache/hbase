49    @SuppressWarnings("unchecked")
50    @Test
51    public void testDuplicateVersions() throws IOException {
52      String family = "Family";
53      byte[] familyBytes = Bytes.toBytes("Family");
54      String table = "TestDuplicateVersions";
55  
56      HColumnDescriptor hcd =
57          new HColumnDescriptor(familyBytes).setMaxVersions(1000);
58      HTableDescriptor htd = new HTableDescriptor(table);
59      htd.addFamily(hcd);
60      HRegionInfo info = new HRegionInfo(Bytes.toBytes(table), null, null, false);
61      HRegion region =
62          HRegion.createHRegion(info, TEST_UTIL.getDataTestDir(), TEST_UTIL
63              .getConfiguration(), htd);
64      try {
65        List<String> rows = generateRandomWords(10, "row");
66        List<String> allColumns = generateRandomWords(10, "column");
67        List<String> values = generateRandomWords(100, "value");
68  
69        long maxTimestamp = 2;
70        double selectPercent = 0.5;
71        int numberOfTests = 5;
72        double flushPercentage = 0.2;
73        double minorPercentage = 0.2;
74        double majorPercentage = 0.2;
75        double putPercentage = 0.2;
76  
77        HashMap<String, KeyValue> allKVMap = new HashMap<String, KeyValue>();
78  
79        HashMap<String, KeyValue>[] kvMaps = new HashMap[numberOfTests];
80        ArrayList<String>[] columnLists = new ArrayList[numberOfTests];
81  
82        for (int i = 0; i < numberOfTests; i++) {
83          kvMaps[i] = new HashMap<String, KeyValue>();
84          columnLists[i] = new ArrayList<String>();
85          for (String column : allColumns) {
86            if (Math.random() < selectPercent) {
87              columnLists[i].add(column);
88            }
89          }
90        }
91  
92        for (String value : values) {
93          for (String row : rows) {
94            Put p = new Put(Bytes.toBytes(row));
95            p.setWriteToWAL(false);
96            for (String column : allColumns) {
97              for (long timestamp = 1; timestamp <= maxTimestamp; timestamp++) {
98                KeyValue kv =
99                    KeyValueTestUtil.create(row, family, column, timestamp, value);
100               if (Math.random() < putPercentage) {
101                 p.add(kv);
102                 allKVMap.put(kv.getKeyString(), kv);
103                 for (int i = 0; i < numberOfTests; i++) {
104                   if (columnLists[i].contains(column)) {
105                     kvMaps[i].put(kv.getKeyString(), kv);
106                   }
107                 }
108               }
109             }
110           }
111           region.put(p);
112           if (Math.random() < flushPercentage) {
113             LOG.info("Flushing... ");
114             region.flushcache();
115           }
116 
117           if (Math.random() < minorPercentage) {
118             LOG.info("Minor compacting... ");
119             region.compactStores(false);
120           }
121 
122           if (Math.random() < majorPercentage) {
123             LOG.info("Major compacting... ");
124             region.compactStores(true);
125           }
126         }
127       }
128 
129       for (int i = 0; i < numberOfTests + 1; i++) {
130         Collection<KeyValue> kvSet;
131         Scan scan = new Scan();
132         scan.setMaxVersions();
133         if (i < numberOfTests) {
134           kvSet = kvMaps[i].values();
135           for (String column : columnLists[i]) {
136             scan.addColumn(familyBytes, Bytes.toBytes(column));
137           }
138           LOG.info("ExplicitColumns scanner");
139           LOG.info("Columns: " + columnLists[i].size() + "  Keys: "
140               + kvSet.size());
141         } else {
142           kvSet = allKVMap.values();
143           LOG.info("Wildcard scanner");
144           LOG.info("Columns: " + allColumns.size() + "  Keys: " + kvSet.size());
145 
146         }
147         InternalScanner scanner = region.getScanner(scan);
148         List<KeyValue> results = new ArrayList<KeyValue>();
149         while (scanner.next(results))
150           ;
151         assertEquals(kvSet.size(), results.size());
152         assertTrue(results.containsAll(kvSet));
153       }
154     } finally {
155       HRegion.closeHRegion(region);
156     }
157 
158     HRegion.closeHRegion(region);
159   }

269   List<String> generateRandomWords(int numberOfWords, String suffix) {
270     Set<String> wordSet = new HashSet<String>();
271     for (int i = 0; i < numberOfWords; i++) {
272       int lengthOfWords = (int) (Math.random() * 5) + 1;
273       char[] wordChar = new char[lengthOfWords];
274       for (int j = 0; j < wordChar.length; j++) {
275         wordChar[j] = (char) (Math.random() * 26 + 97);
276       }
277       String word;
278       if (suffix == null) {
279         word = new String(wordChar);
280       } else {
281         word = new String(wordChar) + suffix;
282       }
283       wordSet.add(word);
284     }
285     List<String> wordList = new ArrayList<String>(wordSet);
286     return wordList;
287   }
