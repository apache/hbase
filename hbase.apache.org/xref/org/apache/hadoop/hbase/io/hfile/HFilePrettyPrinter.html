85    public HFilePrettyPrinter() {
86      options.addOption("v", "verbose", false,
87          "Verbose output; emits file and meta data delimiters");
88      options.addOption("p", "printkv", false, "Print key/value pairs");
89      options.addOption("e", "printkey", false, "Print keys");
90      options.addOption("m", "printmeta", false, "Print meta data of file");
91      options.addOption("b", "printblocks", false, "Print block index meta data");
92      options.addOption("k", "checkrow", false,
93          "Enable row order check; looks for out-of-order keys");
94      options.addOption("a", "checkfamily", false, "Enable family check");
95      options.addOption("f", "file", true,
96          "File to scan. Pass full-path; e.g. hdfs://a:9000/hbase/.META./12/34");
97      options.addOption("w", "seekToRow", true,
98        "Seek to this row and print all the kvs for this row only");
99      options.addOption("r", "region", true,
100         "Region to scan. Pass region name; e.g. '.META.,,1'");
101     options.addOption("s", "stats", false, "Print statistics");
102   }

104   public boolean parseOptions(String args[]) throws ParseException,
105       IOException {
106     if (args.length == 0) {
107       HelpFormatter formatter = new HelpFormatter();
108       formatter.printHelp("HFile", options, true);
109       return false;
110     }
111     CommandLineParser parser = new PosixParser();
112     CommandLine cmd = parser.parse(options, args);
113 
114     verbose = cmd.hasOption("v");
115     printValue = cmd.hasOption("p");
116     printKey = cmd.hasOption("e") || printValue;
117     shouldPrintMeta = cmd.hasOption("m");
118     printBlocks = cmd.hasOption("b");
119     printStats = cmd.hasOption("s");
120     checkRow = cmd.hasOption("k");
121     checkFamily = cmd.hasOption("a");
122 
123     if (cmd.hasOption("f")) {
124       files.add(new Path(cmd.getOptionValue("f")));
125     }
126 
127     if (cmd.hasOption("w")) {
128       String key = cmd.getOptionValue("w");
129       if (key != null && key.length() != 0) {
130         row = key.getBytes();
131         isSeekToRow = true;
132       } else {
133         System.err.println("Invalid row is specified.");
134         System.exit(-1);
135       }
136     }
137 
138     if (cmd.hasOption("r")) {
139       String regionName = cmd.getOptionValue("r");
140       byte[] rn = Bytes.toBytes(regionName);
141       byte[][] hri = HRegionInfo.parseRegionName(rn);
142       Path rootDir = FSUtils.getRootDir(conf);
143       Path tableDir = new Path(rootDir, Bytes.toString(hri[0]));
144       String enc = HRegionInfo.encodeRegionName(rn);
145       Path regionDir = new Path(tableDir, enc);
146       if (verbose)
147         System.out.println("region dir -> " + regionDir);
148       List<Path> regionFiles = HFile.getStoreFiles(FileSystem.get(conf),
149           regionDir);
150       if (verbose)
151         System.out.println("Number of region files found -> "
152             + regionFiles.size());
153       if (verbose) {
154         int i = 1;
155         for (Path p : regionFiles) {
156           if (verbose)
157             System.out.println("Found file[" + i++ + "] -> " + p);
158         }
159       }
160       files.addAll(regionFiles);
161     }
162 
163     return true;
164   }

166   /**
167    * Runs the command-line pretty-printer, and returns the desired command
168    * exit code (zero for success, non-zero for failure).
169    */
170   public int run(String[] args) {
171     conf = HBaseConfiguration.create();
172     conf.set("fs.defaultFS",
173         conf.get(org.apache.hadoop.hbase.HConstants.HBASE_DIR));
174     conf.set("fs.default.name",
175         conf.get(org.apache.hadoop.hbase.HConstants.HBASE_DIR));
176     try {
177       if (!parseOptions(args))
178         return 1;
179     } catch (IOException ex) {
180       LOG.error("Error parsing command-line options", ex);
181       return 1;
182     } catch (ParseException ex) {
183       LOG.error("Error parsing command-line options", ex);
184       return 1;
185     }
186 
187     // iterate over all files found
188     for (Path fileName : files) {
189       try {
190         processFile(fileName);
191       } catch (IOException ex) {
192         LOG.error("Error reading " + fileName, ex);
193       }
194     }
195 
196     if (verbose || printKey) {
197       System.out.println("Scanned kv count -> " + count);
198     }
199 
200     return 0;
201   }

203   private void processFile(Path file) throws IOException {
204     if (verbose)
205       System.out.println("Scanning -> " + file);
206     FileSystem fs = file.getFileSystem(conf);
207     if (!fs.exists(file)) {
208       System.err.println("ERROR, file doesnt exist: " + file);
209     }
210 
211     HFile.Reader reader = HFile.createReader(fs, file, new CacheConfig(conf));
212 
213     Map<byte[], byte[]> fileInfo = reader.loadFileInfo();
214 
215     KeyValueStatsCollector fileStats = null;
216 
217     if (verbose || printKey || checkRow || checkFamily || printStats) {
218       // scan over file and read key/value's and check if requested
219       HFileScanner scanner = reader.getScanner(false, false, false);
220       fileStats = new KeyValueStatsCollector();
221       boolean shouldScanKeysValues = false;
222       if (this.isSeekToRow) {
223         // seek to the first kv on this row
224         shouldScanKeysValues = 
225           (scanner.seekTo(KeyValue.createFirstOnRow(this.row).getKey()) != -1);
226       } else {
227         shouldScanKeysValues = scanner.seekTo();
228       }
229       if (shouldScanKeysValues)
230         scanKeysValues(file, fileStats, scanner, row);
231     }
232 
233     // print meta data
234     if (shouldPrintMeta) {
235       printMeta(reader, fileInfo);
236     }
237 
238     if (printBlocks) {
239       System.out.println("Block Index:");
240       System.out.println(reader.getDataBlockIndexReader());
241     }
242 
243     if (printStats) {
244       fileStats.finish();
245       System.out.println("Stats:\n" + fileStats);
246     }
247 
248     reader.close();
249   }

251   private void scanKeysValues(Path file, KeyValueStatsCollector fileStats,
252       HFileScanner scanner,  byte[] row) throws IOException {
253     KeyValue pkv = null;
254     do {
255       KeyValue kv = scanner.getKeyValue();
256       if (row != null && row.length != 0) {
257         int result = Bytes.compareTo(kv.getRow(), row);
258         if (result > 0) {
259           break;
260         } else if (result < 0) {
261           continue;
262         }
263       }
264       // collect stats
265       if (printStats) {
266         fileStats.collect(kv);
267       }
268       // dump key value
269       if (printKey) {
270         System.out.print("K: " + kv);
271         if (printValue) {
272           System.out.print(" V: " + Bytes.toStringBinary(kv.getValue()));
273         }
274         System.out.println();
275       }
276       // check if rows are in order
277       if (checkRow && pkv != null) {
278         if (Bytes.compareTo(pkv.getRow(), kv.getRow()) > 0) {
279           System.err.println("WARNING, previous row is greater then"
280               + " current row\n\tfilename -> " + file + "\n\tprevious -> "
281               + Bytes.toStringBinary(pkv.getKey()) + "\n\tcurrent  -> "
282               + Bytes.toStringBinary(kv.getKey()));
283         }
284       }
285       // check if families are consistent
286       if (checkFamily) {
287         String fam = Bytes.toString(kv.getFamily());
288         if (!file.toString().contains(fam)) {
289           System.err.println("WARNING, filename does not match kv family,"
290               + "\n\tfilename -> " + file + "\n\tkeyvalue -> "
291               + Bytes.toStringBinary(kv.getKey()));
292         }
293         if (pkv != null
294             && !Bytes.equals(pkv.getFamily(), kv.getFamily())) {
295           System.err.println("WARNING, previous kv has different family"
296               + " compared to current key\n\tfilename -> " + file
297               + "\n\tprevious -> " + Bytes.toStringBinary(pkv.getKey())
298               + "\n\tcurrent  -> " + Bytes.toStringBinary(kv.getKey()));
299         }
300       }
301       pkv = kv;
302       ++count;
303     } while (scanner.next());
304   }

306   /**
307    * Format a string of the form "k1=v1, k2=v2, ..." into separate lines
308    * with a four-space indentation.
309    */
310   private static String asSeparateLines(String keyValueStr) {
311     return keyValueStr.replaceAll(", ([a-zA-Z]+=)",
312                                   ",\n" + FOUR_SPACES + "$1");
313   }

315   private void printMeta(HFile.Reader reader, Map<byte[], byte[]> fileInfo)
316       throws IOException {
317     System.out.println("Block index size as per heapsize: "
318         + reader.indexSize());
319     System.out.println(asSeparateLines(reader.toString()));
320     System.out.println("Trailer:\n    "
321         + asSeparateLines(reader.getTrailer().toString()));
322     System.out.println("Fileinfo:");
323     for (Map.Entry<byte[], byte[]> e : fileInfo.entrySet()) {
324       System.out.print(FOUR_SPACES + Bytes.toString(e.getKey()) + " = ");
325       if (Bytes.compareTo(e.getKey(), Bytes.toBytes("MAX_SEQ_ID_KEY")) == 0) {
326         long seqid = Bytes.toLong(e.getValue());
327         System.out.println(seqid);
328       } else if (Bytes.compareTo(e.getKey(), Bytes.toBytes("TIMERANGE")) == 0) {
329         TimeRangeTracker timeRangeTracker = new TimeRangeTracker();
330         Writables.copyWritable(e.getValue(), timeRangeTracker);
331         System.out.println(timeRangeTracker.getMinimumTimestamp() + "...."
332             + timeRangeTracker.getMaximumTimestamp());
333       } else if (Bytes.compareTo(e.getKey(), FileInfo.AVG_KEY_LEN) == 0
334           || Bytes.compareTo(e.getKey(), FileInfo.AVG_VALUE_LEN) == 0) {
335         System.out.println(Bytes.toInt(e.getValue()));
336       } else {
337         System.out.println(Bytes.toStringBinary(e.getValue()));
338       }
339     }
340 
341     System.out.println("Mid-key: " + Bytes.toStringBinary(reader.midkey()));
342 
343     // Printing general bloom information
344     DataInput bloomMeta = reader.getGeneralBloomFilterMetadata();
345     BloomFilter bloomFilter = null;
346     if (bloomMeta != null)
347       bloomFilter = BloomFilterFactory.createFromMeta(bloomMeta, reader);
348 
349     System.out.println("Bloom filter:");
350     if (bloomFilter != null) {
351       System.out.println(FOUR_SPACES + bloomFilter.toString().replaceAll(
352           ByteBloomFilter.STATS_RECORD_SEP, "\n" + FOUR_SPACES));
353     } else {
354       System.out.println(FOUR_SPACES + "Not present");
355     }
356 
357     // Printing delete bloom information
358     bloomMeta = reader.getDeleteBloomFilterMetadata();
359     bloomFilter = null;
360     if (bloomMeta != null)
361       bloomFilter = BloomFilterFactory.createFromMeta(bloomMeta, reader);
362 
363     System.out.println("Delete Family Bloom filter:");
364     if (bloomFilter != null) {
365       System.out.println(FOUR_SPACES
366           + bloomFilter.toString().replaceAll(ByteBloomFilter.STATS_RECORD_SEP,
367               "\n" + FOUR_SPACES));
368     } else {
369       System.out.println(FOUR_SPACES + "Not present");
370     }
371   }

373   private static class LongStats {
374     private long min = Long.MAX_VALUE;
375     private long max = Long.MIN_VALUE;
376     private long sum = 0;
377     private long count = 0;
378 
379     void collect(long d) {
380       if (d < min) min = d;
381       if (d > max) max = d;
382       sum += d;
383       count++;
384     }
385 
386     public String toString() {
387       return "count: " + count +
388         "\tmin: " + min +
389         "\tmax: " + max +
390         "\tmean: " + ((double)sum/count);
391     }
392   }

394   private static class KeyValueStatsCollector {
395     LongStats keyLen = new LongStats();
396     LongStats valLen = new LongStats();
397     LongStats rowSizeBytes = new LongStats();
398     LongStats rowSizeCols = new LongStats();
399 
400     long curRowBytes = 0;
401     long curRowCols = 0;
402 
403     byte[] biggestRow = null;
404 
405     private KeyValue prevKV = null;
406     private long maxRowBytes = 0;
407 
408     public void collect(KeyValue kv) {
409       keyLen.collect(kv.getKeyLength());
410       valLen.collect(kv.getValueLength());
411       if (prevKV != null &&
412           KeyValue.COMPARATOR.compareRows(prevKV, kv) != 0) {
413         // new row
414         collectRow();
415       }
416       curRowBytes += kv.getLength();
417       curRowCols++;
418       prevKV = kv;
419     }
420 
421     private void collectRow() {
422       rowSizeBytes.collect(curRowBytes);
423       rowSizeCols.collect(curRowCols);
424 
425       if (curRowBytes > maxRowBytes && prevKV != null) {
426         biggestRow = prevKV.getRow();
427       }
428 
429       curRowBytes = 0;
430       curRowCols = 0;
431     }
432 
433     public void finish() {
434       if (curRowCols > 0) {
435         collectRow();
436       }
437     }
438 
439     @Override
440     public String toString() {
441       if (prevKV == null)
442         return "no data available for statistics";
443 
444       return
445         "Key length: " + keyLen + "\n" +
446         "Val length: " + valLen + "\n" +
447         "Row size (bytes): " + rowSizeBytes + "\n" +
448         "Row size (columns): " + rowSizeCols + "\n" +
449         "Key of biggest row: " + Bytes.toStringBinary(biggestRow);
450     }
