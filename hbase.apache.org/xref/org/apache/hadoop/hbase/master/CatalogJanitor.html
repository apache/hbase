53  /**
54   * A janitor for the catalog tables.  Scans the <code>.META.</code> catalog
55   * table on a period looking for unused regions to garbage collect.
56   */
57  @InterfaceAudience.Private
58  class CatalogJanitor extends Chore {
59    private static final Log LOG = LogFactory.getLog(CatalogJanitor.class.getName());
60    private final Server server;
61    private final MasterServices services;
62    private AtomicBoolean enabled = new AtomicBoolean(true);
63    private AtomicBoolean alreadyRunning = new AtomicBoolean(false);

65    CatalogJanitor(final Server server, final MasterServices services) {
66      super(server.getServerName() + "-CatalogJanitor",
67        server.getConfiguration().getInt("hbase.catalogjanitor.interval", 300000),
68        server);
69      this.server = server;
70      this.services = services;

74    protected boolean initialChore() {

108   /**
109    * Scans META and returns a number of scanned rows, and
110    * an ordered map of split parents.
111    */
112   Pair<Integer, Map<HRegionInfo, Result>> getSplitParents() throws IOException {
113     // TODO: Only works with single .META. region currently.  Fix.
114     final AtomicInteger count = new AtomicInteger(0);
115     // Keep Map of found split parents.  There are candidates for cleanup.
116     // Use a comparator that has split parents come before its daughters.
117     final Map<HRegionInfo, Result> splitParents =
118       new TreeMap<HRegionInfo, Result>(new SplitParentFirstComparator());
119     // This visitor collects split parents and counts rows in the .META. table
120     MetaReader.Visitor visitor = new MetaReader.Visitor() {
121       @Override
122       public boolean visit(Result r) throws IOException {
123         if (r == null || r.isEmpty()) return true;
124         count.incrementAndGet();
125         HRegionInfo info = HRegionInfo.getHRegionInfo(r);
126         if (info == null) return true; // Keep scanning
127         if (info.isSplitParent()) splitParents.put(info, r);
128         // Returning true means "keep scanning"
129         return true;
130       }
131     };
132     // Run full scan of .META. catalog table passing in our custom visitor
133     MetaReader.fullScan(this.server.getCatalogTracker(), visitor);
134 
135     return new Pair<Integer, Map<HRegionInfo, Result>>(count.get(), splitParents);
136   }

138   /**
139    * Run janitorial scan of catalog <code>.META.</code> table looking for
140    * garbage to collect.
141    * @throws IOException
142    */
143   int scan() throws IOException {
144     try {
145       if (!alreadyRunning.compareAndSet(false, true)) {
146         return 0;
147       }
148       Pair<Integer, Map<HRegionInfo, Result>> pair = getSplitParents();
149       int count = pair.getFirst();
150       Map<HRegionInfo, Result> splitParents = pair.getSecond();
151 
152       // Now work on our list of found parents. See if any we can clean up.
153       int cleaned = 0;
154     //regions whose parents are still around
155       HashSet<String> parentNotCleaned = new HashSet<String>();
156       for (Map.Entry<HRegionInfo, Result> e : splitParents.entrySet()) {
157         if (!parentNotCleaned.contains(e.getKey().getEncodedName()) &&
158             cleanParent(e.getKey(), e.getValue())) {
159           cleaned++;
160         } else {
161           // We could not clean the parent, so it's daughters should not be cleaned either (HBASE-6160)
162           PairOfSameType<HRegionInfo> daughters = HRegionInfo.getDaughterRegions(e.getValue());
163           parentNotCleaned.add(daughters.getFirst().getEncodedName());
164           parentNotCleaned.add(daughters.getSecond().getEncodedName());
165         }
166       }
167       if (cleaned != 0) {
168         LOG.info("Scanned " + count + " catalog row(s) and gc'd " + cleaned +
169             " unreferenced parent region(s)");
170       } else if (LOG.isDebugEnabled()) {
171         LOG.debug("Scanned " + count + " catalog row(s) and gc'd " + cleaned +
172             " unreferenced parent region(s)");
173       }
174       return cleaned;
175     } finally {
176       alreadyRunning.set(false);
177     }
178   }

180   /**
181    * Compare HRegionInfos in a way that has split parents sort BEFORE their
182    * daughters.
183    */
184   static class SplitParentFirstComparator implements Comparator<HRegionInfo> {
185     @Override
186     public int compare(HRegionInfo left, HRegionInfo right) {
187       // This comparator differs from the one HRegionInfo in that it sorts
188       // parent before daughters.
189       if (left == null) return -1;
190       if (right == null) return 1;
191       // Same table name.
192       int result = Bytes.compareTo(left.getTableName(),
193           right.getTableName());
194       if (result != 0) return result;
195       // Compare start keys.
196       result = Bytes.compareTo(left.getStartKey(), right.getStartKey());
197       if (result != 0) return result;
198       // Compare end keys.
199       result = Bytes.compareTo(left.getEndKey(), right.getEndKey());
200       if (result != 0) {
201         if (left.getStartKey().length != 0
202                 && left.getEndKey().length == 0) {
203             return -1;  // left is last region
204         }
205         if (right.getStartKey().length != 0
206                 && right.getEndKey().length == 0) {
207             return 1;  // right is the last region
208         }
209         return -result; // Flip the result so parent comes first.
210       }
211       return result;
212     }
213   }

215   /**
216    * If daughters no longer hold reference to the parents, delete the parent.
217    * @param parent HRegionInfo of split offlined parent
218    * @param rowContent Content of <code>parent</code> row in
219    * <code>metaRegionName</code>
220    * @return True if we removed <code>parent</code> from meta table and from
221    * the filesystem.
222    * @throws IOException
223    */
224   boolean cleanParent(final HRegionInfo parent, Result rowContent)
225   throws IOException {
226     boolean result = false;
227     // Run checks on each daughter split.
228     PairOfSameType<HRegionInfo> daughters = HRegionInfo.getDaughterRegions(rowContent);
229     Pair<Boolean, Boolean> a =
230       checkDaughterInFs(parent, daughters.getFirst());
231     Pair<Boolean, Boolean> b =
232       checkDaughterInFs(parent, daughters.getSecond());
233     if (hasNoReferences(a) && hasNoReferences(b)) {
234       LOG.debug("Deleting region " + parent.getRegionNameAsString() +
235         " because daughter splits no longer hold references");
236       // wipe out daughter references from parent region in meta
237       removeDaughtersFromParent(parent);
238 
239       // This latter regionOffline should not be necessary but is done for now
240       // until we let go of regionserver to master heartbeats.  See HBASE-3368.
241       if (this.services.getAssignmentManager() != null) {
242         // The mock used in testing catalogjanitor returns null for getAssignmnetManager.
243         // Allow for null result out of getAssignmentManager.
244         this.services.getAssignmentManager().regionOffline(parent);
245       }
246       FileSystem fs = this.services.getMasterFileSystem().getFileSystem();
247       LOG.debug("Archiving parent region:" + parent);
248       HFileArchiver.archiveRegion(fs, parent);
249       MetaEditor.deleteRegion(this.server.getCatalogTracker(), parent);
250       result = true;
251     }
252     return result;
253   }

255   /**
256    * @param p A pair where the first boolean says whether or not the daughter
257    * region directory exists in the filesystem and then the second boolean says
258    * whether the daughter has references to the parent.
259    * @return True the passed <code>p</code> signifies no references.
260    */
261   private boolean hasNoReferences(final Pair<Boolean, Boolean> p) {
262     return !p.getFirst() || !p.getSecond();
263   }

275   /**
276    * Checks if a daughter region -- either splitA or splitB -- still holds
277    * references to parent.
278    * @param parent Parent region
279    * @param daughter Daughter region
280    * @return A pair where the first boolean says whether or not the daughter
281    * region directory exists in the filesystem and then the second boolean says
282    * whether the daughter has references to the parent.
283    * @throws IOException
284    */
285   Pair<Boolean, Boolean> checkDaughterInFs(final HRegionInfo parent, final HRegionInfo daughter)
286   throws IOException {
287     boolean references = false;
288     boolean exists = false;
289     if (daughter == null)  {
290       return new Pair<Boolean, Boolean>(Boolean.FALSE, Boolean.FALSE);
291     }
292     FileSystem fs = this.services.getMasterFileSystem().getFileSystem();
293     Path rootdir = this.services.getMasterFileSystem().getRootDir();
294     Path tabledir = new Path(rootdir, daughter.getTableNameAsString());
295     Path regiondir = new Path(tabledir, daughter.getEncodedName());
296     exists = fs.exists(regiondir);
297     if (!exists) {
298       LOG.warn("Daughter regiondir does not exist: " + regiondir.toString());
299       return new Pair<Boolean, Boolean>(exists, Boolean.FALSE);
300     }
301     HTableDescriptor parentDescriptor = getTableDescriptor(parent.getTableName());
302 
303     for (HColumnDescriptor family: parentDescriptor.getFamilies()) {
304       Path p = HStore.getStoreHomedir(tabledir, daughter.getEncodedName(),
305         family.getName());
306       if (!fs.exists(p)) continue;
307       // Look for reference files.  Call listStatus with anonymous instance of PathFilter.
308       FileStatus [] ps = FSUtils.listStatus(fs, p,
309           new PathFilter () {
310             public boolean accept(Path path) {
311               return StoreFile.isReference(path);
312             }
313           }
314       );
315 
316       if (ps != null && ps.length > 0) {
317         references = true;
318         break;
319       }
320     }
321     return new Pair<Boolean, Boolean>(Boolean.valueOf(exists),
322       Boolean.valueOf(references));
323   }

325   private HTableDescriptor getTableDescriptor(byte[] tableName)
