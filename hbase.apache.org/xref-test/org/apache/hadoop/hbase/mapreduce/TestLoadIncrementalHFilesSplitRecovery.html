148   /**
149    * Split the known table in half.  (this is hard coded for this test suite)
150    */
151   private void forceSplit(String table) {
152     try {
153       // need to call regions server to by synchronous but isn't visible.
154       HRegionServer hrs = util.getRSForFirstRegionInTable(Bytes
155           .toBytes(table));
156 
157       for (HRegionInfo hri : ProtobufUtil.getOnlineRegions(hrs)) {
158         if (Bytes.equals(hri.getTableName(), Bytes.toBytes(table))) {
159           // splitRegion doesn't work if startkey/endkey are null
160           ProtobufUtil.split(hrs, hri, rowkey(ROWCOUNT / 2)); // hard code split
161         }
162       }
163 
164       // verify that split completed.
165       int regions;
166       do {
167         regions = 0;
168         for (HRegionInfo hri : ProtobufUtil.getOnlineRegions(hrs)) {
169           if (Bytes.equals(hri.getTableName(), Bytes.toBytes(table))) {
170             regions++;
171           }
172         }
173         if (regions != 2) {
174           LOG.info("Taking some time to complete split...");
175           Thread.sleep(250);
176         }
177       } while (regions != 2);
178     } catch (IOException e) {
179       e.printStackTrace();
180     } catch (InterruptedException e) {
181       e.printStackTrace();
182     }
183   }

198    * expected number of rows.

200   void assertExpectedTable(String table, int count, int value) {
201     try {
202       assertEquals(util.getHBaseAdmin().listTables(table).length, 1);
203 
204       HTable t = new HTable(util.getConfiguration(), table);
205       Scan s = new Scan();
206       ResultScanner sr = t.getScanner(s);
207       int i = 0;
208       for (Result r : sr) {
209         i++;
210         for (NavigableMap<byte[], byte[]> nm : r.getNoVersionMap().values()) {
211           for (byte[] val : nm.values()) {
212             assertTrue(Bytes.equals(val, value(value)));
213           }
214         }
215       }
216       assertEquals(count, i);
217     } catch (IOException e) {
218       fail("Failed due to exception");
219     }
220   }

223    * Test that shows that exception thrown from the RS side will result in an
224    * exception on the LIHFile client.
225    */
226   @Test(expected=IOException.class)
227   public void testBulkLoadPhaseFailure() throws Exception {
228     String table = "bulkLoadPhaseFailure";
229     setupTable(table, 10);
230 
231     final AtomicInteger attmptedCalls = new AtomicInteger();
232     final AtomicInteger failedCalls = new AtomicInteger();
233     LoadIncrementalHFiles lih = new LoadIncrementalHFiles(
234         util.getConfiguration()) {
235 
236       protected List<LoadQueueItem> tryAtomicRegionLoad(final HConnection conn,
237           byte[] tableName, final byte[] first, Collection<LoadQueueItem> lqis)
238       throws IOException {
239         int i = attmptedCalls.incrementAndGet();
240         if (i == 1) {
241           HConnection errConn = null;
242           try {
243             errConn = getMockedConnection(util.getConfiguration());
244           } catch (Exception e) {
245             LOG.fatal("mocking cruft, should never happen", e);
246             throw new RuntimeException("mocking cruft, should never happen");
247           }
248           failedCalls.incrementAndGet();
249           return super.tryAtomicRegionLoad(errConn, tableName, first, lqis);
250         }
251 
252         return super.tryAtomicRegionLoad(conn, tableName, first, lqis);
253       }
254     };
255 
256     // create HFiles for different column families
257     Path dir = buildBulkFiles(table, 1);
258     HTable t = new HTable(util.getConfiguration(), Bytes.toBytes(table));
259     lih.doBulkLoad(dir, t);
260 
261     fail("doBulkLoad should have thrown an exception");
262   }

285   /**
286    * This test exercises the path where there is a split after initial
287    * validation but before the atomic bulk load call. We cannot use presplitting
288    * to test this path, so we actually inject a split just before the atomic
289    * region load.
290    */
291   @Test
292   public void testSplitWhileBulkLoadPhase() throws Exception {
293     final String table = "splitWhileBulkloadPhase";
294     setupTable(table, 10);
295     populateTable(table,1);
296     assertExpectedTable(table, ROWCOUNT, 1);
297 
298     // Now let's cause trouble.  This will occur after checks and cause bulk
299     // files to fail when attempt to atomically import.  This is recoverable.
300     final AtomicInteger attemptedCalls = new AtomicInteger();
301     LoadIncrementalHFiles lih2 = new LoadIncrementalHFiles(
302         util.getConfiguration()) {
303 
304       protected void bulkLoadPhase(final HTable htable, final HConnection conn,
305           ExecutorService pool, Deque<LoadQueueItem> queue,
306           final Multimap<ByteBuffer, LoadQueueItem> regionGroups) throws IOException {
307         int i = attemptedCalls.incrementAndGet();
308         if (i == 1) {
309           // On first attempt force a split.
310           forceSplit(table);
311         }
312 
313         super.bulkLoadPhase(htable, conn, pool, queue, regionGroups);
314       }
315     };
316 
317     // create HFiles for different column families
318     HTable t = new HTable(util.getConfiguration(), Bytes.toBytes(table));
319     Path bulk = buildBulkFiles(table, 2);
320     lih2.doBulkLoad(bulk, t);
321 
322     // check that data was loaded
323     // The three expected attempts are 1) failure because need to split, 2)
324     // load of split top 3) load of split bottom
325     assertEquals(attemptedCalls.get(), 3);
326     assertExpectedTable(table, ROWCOUNT, 2);
327   }

330    * This test splits a table and attempts to bulk load.  The bulk import files
331    * should be split before atomically importing.
332    */
333   @Test
334   public void testGroupOrSplitPresplit() throws Exception {
335     final String table = "groupOrSplitPresplit";
336     setupTable(table, 10);
337     populateTable(table, 1);
338     assertExpectedTable(table, ROWCOUNT, 1);
339     forceSplit(table);
340 
341     final AtomicInteger countedLqis= new AtomicInteger();
342     LoadIncrementalHFiles lih = new LoadIncrementalHFiles(
343         util.getConfiguration()) {
344       protected List<LoadQueueItem> groupOrSplit(
345           Multimap<ByteBuffer, LoadQueueItem> regionGroups,
346           final LoadQueueItem item, final HTable htable,
347           final Pair<byte[][], byte[][]> startEndKeys) throws IOException {
348         List<LoadQueueItem> lqis = super.groupOrSplit(regionGroups, item, htable, startEndKeys);
349         if (lqis != null) {
350           countedLqis.addAndGet(lqis.size());
351         }
352         return lqis;
353       }
354     };
355 
356     // create HFiles for different column families
357     Path bulk = buildBulkFiles(table, 2);
358     HTable ht = new HTable(util.getConfiguration(), Bytes.toBytes(table));
359     lih.doBulkLoad(bulk, ht);
360 
361     assertExpectedTable(table, ROWCOUNT, 2);
362     assertEquals(20, countedLqis.get());
363   }
