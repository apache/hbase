185    * Test for HBASE-4288.  Throw an IOE when trying to verify meta region and
186    * prove it doesn't cause master shutdown.
187    * @see <a href="https://issues.apache.org/jira/browse/HBASE-4288">HBASE-4288</a>
188    * @throws IOException
189    * @throws InterruptedException
190    * @throws KeeperException
191    */
192   @Test
193   public void testServerNotRunningIOException()
194   throws IOException, InterruptedException, KeeperException, ServiceException {
195     // Mock an Admin and a Client.
196     final AdminProtocol admin = Mockito.mock(AdminProtocol.class);
197     final ClientProtocol client = Mockito.mock(ClientProtocol.class);
198     HConnection connection = mockConnection(admin, client);
199     try {
200       // If a 'getRegionInfo' is called on mocked AdminProtocol, throw IOE
201       // the first time.  'Succeed' the second time we are called.
202       GetRegionInfoResponse.Builder builder = GetRegionInfoResponse.newBuilder();
203       builder.setRegionInfo(HRegionInfo.convert(new HRegionInfo(Bytes.toBytes("test"))));
204       Mockito.when(admin.getRegionInfo((RpcController)Mockito.any(),
205         (GetRegionInfoRequest)Mockito.any())).thenThrow(
206           new ServiceException(new IOException("Server not running, aborting"))).
207         thenReturn(builder.build());
208 
209       // After we encounter the above 'Server not running', we should catch the
210       // IOE and go into retrying for the meta mode.  We'll do gets on -ROOT- to
211       // get new meta location.  Return something so this 'get' succeeds
212       // (here we mock up getRegionServerWithRetries, the wrapper around
213       // the actual get).
214 
215       // TODO: Refactor.  This method has been moved out of HConnection.
216       // It works for now but has been deprecated.
217       Mockito.when(connection.getRegionServerWithRetries((ServerCallable<Result>)Mockito.any())).
218         thenReturn(getMetaTableRowResult());
219 
220       Mockito.when(client.get((RpcController)Mockito.any(), (GetRequest)Mockito.any())).
221         thenReturn(GetResponse.newBuilder().build());
222       // Now start up the catalogtracker with our doctored Connection.
223       final CatalogTracker ct = constructAndStartCatalogTracker(connection);
224       try {
225         // Set a location for root and meta.
226         RootRegionTracker.setRootLocation(this.watcher, SN);
227         ct.setMetaLocation(SN);
228         // Call the method that HBASE-4288 calls.  It will try and verify the
229         // meta location and will fail on first attempt then go into a long wait.
230         // So, do this in a thread and then reset meta location to break it out
231         // of its wait after a bit of time.
232         final AtomicBoolean metaSet = new AtomicBoolean(false);
233         Thread t = new Thread() {
234           @Override
235           public void run() {
236             try {
237               metaSet.set(ct.waitForMetaServerConnectionDefault() !=  null);
238             } catch (NotAllMetaRegionsOnlineException e) {
239               throw new RuntimeException(e);
240             } catch (IOException e) {
241               throw new RuntimeException(e);
242             }
243           }
244         };
245         t.start();
246         while(!t.isAlive()) Threads.sleep(1);
247         Threads.sleep(1);
248         // Now reset the meta as though it were redeployed.
249         ct.setMetaLocation(SN);
250         t.join();
251         Assert.assertTrue(metaSet.get());
252       } finally {
253         // Clean out root and meta locations or later tests will be confused...
254         // they presume start fresh in zk.
255         ct.resetMetaLocation();
256         RootRegionTracker.deleteRootLocation(this.watcher);
257       }
258     } finally {
259       // Clear out our doctored connection or could mess up subsequent tests.
260       HConnectionManager.deleteConnection(UTIL.getConfiguration(), true);
261     }
262   }

539   /**
540    * Wait on META.
541    * Default is wait on -ROOT-.
542    */
543   class WaitOnMetaThread extends Thread {
544     final CatalogTracker ct;
545 
546     WaitOnMetaThread(final CatalogTracker ct) {
547       super("WaitOnMeta");
548       this.ct = ct;
549     }
550 
551     @Override
552     public void run() {
553       try {
554         doWaiting();
555       } catch (InterruptedException e) {
556         throw new RuntimeException("Failed wait", e);
557       }
558       LOG.info("Exiting " + getName());
559     }
560 
561     void doWaiting() throws InterruptedException {
562       try {
563         while (this.ct.waitForRoot(100) == null);
564       } catch (NotAllMetaRegionsOnlineException e) {
565         // Ignore.
566       }
567     }
568   }
