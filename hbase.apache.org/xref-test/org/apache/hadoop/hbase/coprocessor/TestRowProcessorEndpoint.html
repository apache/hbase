208   private void concurrentExec(
209       final Runnable task, final int numThreads) throws Throwable {
210     startSignal = new CountDownLatch(numThreads);
211     doneSignal = new CountDownLatch(numThreads);
212     for (int i = 0; i < numThreads; ++i) {
213       new Thread(new Runnable() {
214         @Override
215         public void run() {
216           try {
217             startSignal.countDown();
218             startSignal.await();
219             task.run();
220           } catch (Throwable e) {
221             failures.incrementAndGet();
222             e.printStackTrace();
223           }
224           doneSignal.countDown();
225         }
226       }).start();
227     }
228     doneSignal.await();
229   }

327       @Override
328       public void process(long now, HRegion region,
329           List<KeyValue> mutations, WALEdit walEdit) throws IOException {
330         // Scan current counter
331         List<KeyValue> kvs = new ArrayList<KeyValue>();
332         Scan scan = new Scan(row, row);
333         scan.addColumn(FAM, COUNTER);
334         doScan(region, scan, kvs);
335         counter = kvs.size() == 0 ? 0 :
336           Bytes.toInt(kvs.iterator().next().getValue());

338         // Assert counter value
339         assertEquals(expectedCounter, counter);

341         // Increment counter and send it to both memstore and wal edit
342         counter += 1;
343         expectedCounter += 1;
344 
345 
346         KeyValue kv =
347             new KeyValue(row, FAM, COUNTER, now, Bytes.toBytes(counter));
348         mutations.add(kv);
349         walEdit.add(kv);
350 
351         // We can also inject some meta data to the walEdit
352         KeyValue metaKv = new KeyValue(
353             row, HLog.METAFAMILY,
354             Bytes.toBytes("I just increment counter"),
355             Bytes.toBytes(counter));
356         walEdit.add(metaKv);
357       }

375         BaseRowProcessor<FriendsOfFriendsProcessorRequest, FriendsOfFriendsProcessorResponse> {
376       byte[] row = null;
377       byte[] person = null;

383       FriendsOfFriendsProcessor() {

393         return Collections.singleton(row);

396       @Override
397       public FriendsOfFriendsProcessorResponse getResult() {
398         FriendsOfFriendsProcessorResponse.Builder builder = 
399             FriendsOfFriendsProcessorResponse.newBuilder();
400         builder.addAllResult(result);
401         return builder.build();
402       }

406         return true;

409       @Override
410       public void process(long now, HRegion region,
411           List<KeyValue> mutations, WALEdit walEdit) throws IOException {
412         List<KeyValue> kvs = new ArrayList<KeyValue>();
413         { // First scan to get friends of the person
414           Scan scan = new Scan(row, row);
415           scan.addColumn(FAM, person);
416           doScan(region, scan, kvs);
417         }
418 
419         // Second scan to get friends of friends
420         Scan scan = new Scan(row, row);
421         for (KeyValue kv : kvs) {
422           byte[] friends = kv.getValue();
423           for (byte f : friends) {
424             scan.addColumn(FAM, new byte[]{f});
425           }
426         }
427         doScan(region, scan, kvs);
428 
429         // Collect result
430         result.clear();
431         for (KeyValue kv : kvs) {
432           for (byte b : kv.getValue()) {
433             result.add((char)b + "");
434           }
435         }
436       }

439       public FriendsOfFriendsProcessorRequest getRequestData() throws IOException {
440         FriendsOfFriendsProcessorRequest.Builder builder =
441             FriendsOfFriendsProcessorRequest.newBuilder();
442         builder.setPerson(ByteString.copyFrom(person));
443         builder.setRow(ByteString.copyFrom(row));
444         builder.addAllResult(result);
445         FriendsOfFriendsProcessorRequest f = builder.build();
446         return f;
447       }

459     public static class RowSwapProcessor extends
460         BaseRowProcessor<RowSwapProcessorRequest, RowSwapProcessorResponse> {
461       byte[] row1 = new byte[0];
462       byte[] row2 = new byte[0];
463 
464       /**
465        * Empty constructor for Writable
466        */
467       RowSwapProcessor() {
468       }
469 
470       RowSwapProcessor(byte[] row1, byte[] row2) {
471         this.row1 = row1;
472         this.row2 = row2;
473       }
474 
475       @Override
476       public Collection<byte[]> getRowsToLock() {
477         List<byte[]> rows = new ArrayList<byte[]>();
478         rows.add(row1);
479         rows.add(row2);
480         return rows;
481       }
482 
483       @Override
484       public boolean readOnly() {
485         return false;
486       }
487 
488       @Override
489       public RowSwapProcessorResponse getResult() {
490         return RowSwapProcessorResponse.getDefaultInstance();
491       }
492 
493       @Override
494       public void process(long now, HRegion region,
495           List<KeyValue> mutations, WALEdit walEdit) throws IOException {
496 
497         // Override the time to avoid race-condition in the unit test caused by
498         // inacurate timer on some machines
499         now = myTimer.getAndIncrement();
500 
501         // Scan both rows
502         List<KeyValue> kvs1 = new ArrayList<KeyValue>();
503         List<KeyValue> kvs2 = new ArrayList<KeyValue>();
504         doScan(region, new Scan(row1, row1), kvs1);
505         doScan(region, new Scan(row2, row2), kvs2);
506 
507         // Assert swapped
508         if (swapped) {
509           assertEquals(rowSize, kvs2.size());
510           assertEquals(row2Size, kvs1.size());
511         } else {
512           assertEquals(rowSize, kvs1.size());
513           assertEquals(row2Size, kvs2.size());
514         }
515         swapped = !swapped;
516 
517         // Add and delete keyvalues
518         List<List<KeyValue>> kvs = new ArrayList<List<KeyValue>>();
519         kvs.add(kvs1);
520         kvs.add(kvs2);
521         byte[][] rows = new byte[][]{row1, row2};
522         for (int i = 0; i < kvs.size(); ++i) {
523           for (KeyValue kv : kvs.get(i)) {
524             // Delete from the current row and add to the other row
525             KeyValue kvDelete =
526                 new KeyValue(rows[i], kv.getFamily(), kv.getQualifier(),
527                     kv.getTimestamp(), KeyValue.Type.Delete);
528             KeyValue kvAdd =
529                 new KeyValue(rows[1 - i], kv.getFamily(), kv.getQualifier(),
530                     now, kv.getValue());
531             mutations.add(kvDelete);
532             walEdit.add(kvDelete);
533             mutations.add(kvAdd);
534             walEdit.add(kvAdd);
535           }
536         }
537       }

617     public static void doScan(
618         HRegion region, Scan scan, List<KeyValue> result) throws IOException {
619       InternalScanner scanner = null;
620       try {
621         scan.setIsolationLevel(IsolationLevel.READ_UNCOMMITTED);
622         scanner = region.getScanner(scan);
623         result.clear();
624         scanner.next(result);
625       } finally {
626         if (scanner != null) scanner.close();
627       }
628     }

631   static String stringifyKvs(Collection<KeyValue> kvs) {
632     StringBuilder out = new StringBuilder();
633     out.append("[");
634     if (kvs != null) {
635       for (KeyValue kv : kvs) {
636         byte[] col = kv.getQualifier();
637         byte[] val = kv.getValue();
638         if (Bytes.equals(col, COUNTER)) {
639           out.append(Bytes.toStringBinary(col) + ":" +
640                      Bytes.toInt(val) + " ");
641         } else {
642           out.append(Bytes.toStringBinary(col) + ":" +
643                      Bytes.toStringBinary(val) + " ");
644         }
645       }
646     }
647     out.append("]");
648     return out.toString();
649   }
