19  package org.apache.hadoop.hbase;

21  import junit.framework.TestCase;

23  import org.apache.hadoop.hbase.util.Bytes;
24  import org.junit.experimental.categories.Category;
25  
26  /**
27   * Test comparing HBase objects.
28   */
29  @Category(SmallTests.class)
30  public class TestCompare extends TestCase {
31  
32    /**
33     * Sort of HRegionInfo.
34     */
35    public void testHRegionInfo() {
36      HRegionInfo a = new HRegionInfo(Bytes.toBytes("a"), null, null);
37      HRegionInfo b = new HRegionInfo(Bytes.toBytes("b"), null, null);
38      assertTrue(a.compareTo(b) != 0);
39      HTableDescriptor t = new HTableDescriptor("t");
40      byte [] midway = Bytes.toBytes("midway");
41      a = new HRegionInfo(t.getName(), null, midway);
42      b = new HRegionInfo(t.getName(), midway, null);
43      assertTrue(a.compareTo(b) < 0);
44      assertTrue(b.compareTo(a) > 0);
45      assertEquals(a, a);
46      assertTrue(a.compareTo(a) == 0);
47      a = new HRegionInfo(t.getName(), Bytes.toBytes("a"), Bytes.toBytes("d"));
48      b = new HRegionInfo(t.getName(), Bytes.toBytes("e"), Bytes.toBytes("g"));
49      assertTrue(a.compareTo(b) < 0);
50      a = new HRegionInfo(t.getName(), Bytes.toBytes("aaaa"), Bytes.toBytes("dddd"));
51      b = new HRegionInfo(t.getName(), Bytes.toBytes("e"), Bytes.toBytes("g"));
52      assertTrue(a.compareTo(b) < 0);
53      a = new HRegionInfo(t.getName(), Bytes.toBytes("aaaa"), Bytes.toBytes("dddd"));
54      b = new HRegionInfo(t.getName(), Bytes.toBytes("aaaa"), Bytes.toBytes("eeee"));
55      assertTrue(a.compareTo(b) < 0);
56    }
