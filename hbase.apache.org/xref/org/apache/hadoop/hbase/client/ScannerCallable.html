47  /**
48   * Retries scanner operations such as create, next, etc.
49   * Used by {@link ResultScanner}s made by {@link HTable}.
50   */
51  @InterfaceAudience.Public
52  @InterfaceStability.Stable
53  public class ScannerCallable extends ServerCallable<Result[]> {
54    public static final String LOG_SCANNER_LATENCY_CUTOFF
55      = "hbase.client.log.scanner.latency.cutoff";
56    public static final String LOG_SCANNER_ACTIVITY = "hbase.client.log.scanner.activity";
57  
58    private static final Log LOG = LogFactory.getLog(ScannerCallable.class);
59    private long scannerId = -1L;
60    private boolean instantiated = false;
61    private boolean closed = false;
62    private Scan scan;
63    private int caching = 1;
64    private ScanMetrics scanMetrics;
65    private boolean logScannerActivity = false;
66    private int logCutOffLatency = 1000;
67  
68    // indicate if it is a remote server call
69    private boolean isRegionServerRemote = true;
70    private long nextCallSeq = 0;
71    
72    /**
73     * @param connection which connection
74     * @param tableName table callable is on
75     * @param scan the scan to execute
76     * @param scanMetrics the ScanMetrics to used, if it is null, ScannerCallable
77     * won't collect metrics
78     */
79    public ScannerCallable (HConnection connection, byte [] tableName, Scan scan,
80      ScanMetrics scanMetrics) {
81      super(connection, tableName, scan.getStartRow());
82      this.scan = scan;
83      this.scanMetrics = scanMetrics;
84      Configuration conf = connection.getConfiguration();
85      logScannerActivity = conf.getBoolean(LOG_SCANNER_ACTIVITY, false);
86      logCutOffLatency = conf.getInt(LOG_SCANNER_LATENCY_CUTOFF, 1000);
87    }

89    /**
90     * @param reload force reload of server location
91     * @throws IOException
92     */
93    @Override
94    public void connect(boolean reload) throws IOException {
95      if (!instantiated || reload) {
96        super.connect(reload);
97        checkIfRegionServerIsRemote();
98        instantiated = true;
99      }
100 
101     // check how often we retry.
102     // HConnectionManager will call instantiateServer with reload==true
103     // if and only if for retries.
104     if (reload && this.scanMetrics != null) {
105       this.scanMetrics.countOfRPCRetries.incrementAndGet();
106       if (isRegionServerRemote) {
107         this.scanMetrics.countOfRemoteRPCRetries.incrementAndGet();
108       }
109     }
110   }

112   /**
113    * compare the local machine hostname with region server's hostname
114    * to decide if hbase client connects to a remote region server
115    * @throws UnknownHostException.
116    */
117   private void checkIfRegionServerIsRemote() throws UnknownHostException {
118     String myAddress = DNS.getDefaultHost("default", "default");
119     if (this.location.getHostname().equalsIgnoreCase(myAddress)) {
120       isRegionServerRemote = false;
121     } else {
122       isRegionServerRemote = true;
123     }
124   }

126   /**
127    * @see java.util.concurrent.Callable#call()
128    */
129   public Result [] call() throws IOException {
130     if (closed) {
131       if (scannerId != -1) {
132         close();
133       }
134     } else {
135       if (scannerId == -1L) {
136         this.scannerId = openScanner();
137       } else {
138         Result [] rrs = null;
139         try {
140           incRPCcallsMetrics();
141           ScanRequest request =
142             RequestConverter.buildScanRequest(scannerId, caching, false, nextCallSeq);
143           try {
144             ScanResponse response = server.scan(null, request);
145             // Client and RS maintain a nextCallSeq number during the scan. Every next() call
146             // from client to server will increment this number in both sides. Client passes this
147             // number along with the request and at RS side both the incoming nextCallSeq and its
148             // nextCallSeq will be matched. In case of a timeout this increment at the client side
149             // should not happen. If at the server side fetching of next batch of data was over,
150             // there will be mismatch in the nextCallSeq number. Server will throw
151             // OutOfOrderScannerNextException and then client will reopen the scanner with startrow
152             // as the last successfully retrieved row.
153             // See HBASE-5974
154             nextCallSeq++;
155             long timestamp = System.currentTimeMillis();
156             rrs = ResponseConverter.getResults(response);
157             if (logScannerActivity) {
158               long now = System.currentTimeMillis();
159               if (now - timestamp > logCutOffLatency) {
160                 int rows = rrs == null ? 0 : rrs.length;
161                 LOG.info("Took " + (now-timestamp) + "ms to fetch "
162                   + rows + " rows from scanner=" + scannerId);
163               }
164             }
165             if (response.hasMoreResults()
166                 && !response.getMoreResults()) {
167               scannerId = -1L;
168               closed = true;
169               return null;
170             }
171           } catch (ServiceException se) {
172             throw ProtobufUtil.getRemoteException(se);
173           }
174           updateResultsMetrics(rrs);
175         } catch (IOException e) {
176           if (logScannerActivity) {
177             LOG.info("Got exception in fetching from scanner="
178               + scannerId, e);
179           }
180           IOException ioe = e;
181           if (e instanceof RemoteException) {
182             ioe = RemoteExceptionHandler.decodeRemoteException((RemoteException)e);
183           }
184           if (logScannerActivity && (ioe instanceof UnknownScannerException)) {
185             try {
186               HRegionLocation location =
187                 connection.relocateRegion(tableName, scan.getStartRow());
188               LOG.info("Scanner=" + scannerId
189                 + " expired, current region location is " + location.toString()
190                 + " ip:" + location.getHostnamePort());
191             } catch (Throwable t) {
192               LOG.info("Failed to relocate region", t);
193             }
194           }
195           if (ioe instanceof NotServingRegionException) {
196             // Throw a DNRE so that we break out of cycle of calling NSRE
197             // when what we need is to open scanner against new location.
198             // Attach NSRE to signal client that it needs to resetup scanner.
199             if (this.scanMetrics != null) {
200               this.scanMetrics.countOfNSRE.incrementAndGet();
201             }
202             throw new DoNotRetryIOException("Reset scanner", ioe);
203           } else if (ioe instanceof RegionServerStoppedException) {
204             // Throw a DNRE so that we break out of cycle of calling RSSE
205             // when what we need is to open scanner against new location.
206             // Attach RSSE to signal client that it needs to resetup scanner.
207             throw new DoNotRetryIOException("Reset scanner", ioe);
208           } else {
209             // The outer layers will retry
210             throw ioe;
211           }
212         }
213         return rrs;
214       }
215     }
216     return null;
217   }

219   private void incRPCcallsMetrics() {
220     if (this.scanMetrics == null) {
221       return;
222     }
223     this.scanMetrics.countOfRPCcalls.incrementAndGet();
224     if (isRegionServerRemote) {
225       this.scanMetrics.countOfRemoteRPCcalls.incrementAndGet();
226     }
227   }

229   private void updateResultsMetrics(Result[] rrs) {
230     if (this.scanMetrics == null || rrs == null) {
231       return;
232     }
233     /*

235     for (Result rr : rrs) {

238         if (isRegionServerRemote) {

247   private void close() {
248     if (this.scannerId == -1L) {
249       return;
250     }
251     try {
252       incRPCcallsMetrics();
253       ScanRequest request =
254         RequestConverter.buildScanRequest(this.scannerId, 0, true);
255       try {
256         server.scan(null, request);
257       } catch (ServiceException se) {
258         throw ProtobufUtil.getRemoteException(se);
259       }
260     } catch (IOException e) {
261       LOG.warn("Ignore, probably already closed", e);
262     }
263     this.scannerId = -1L;
264   }
265 
266   protected long openScanner() throws IOException {
267     incRPCcallsMetrics();
268     ScanRequest request =
269       RequestConverter.buildScanRequest(
270         this.location.getRegionInfo().getRegionName(),
271         this.scan, 0, false);
272     try {
273       ScanResponse response = server.scan(null, request);
274       long id = response.getScannerId();
275       if (logScannerActivity) {
276         LOG.info("Open scanner=" + id + " for scan=" + scan.toString()
277           + " on region " + this.location.toString() + " ip:"
278           + this.location.getHostnamePort());
279       }
280       return id;
281     } catch (ServiceException se) {
282       throw ProtobufUtil.getRemoteException(se);
283     }
284   }
