102 /**
103  * A non-instantiable class that manages {@link HConnection}s.
104  * This class has a static Map of {@link HConnection} instances keyed by
105  * {@link Configuration}; all invocations of {@link #getConnection(Configuration)}
106  * that pass the same {@link Configuration} instance will be returned the same
107  * {@link  HConnection} instance (Adding properties to a Configuration
108  * instance does not change its object identity).  Sharing {@link HConnection}
109  * instances is usually what you want; all clients of the {@link HConnection}
110  * instances share the HConnections' cache of Region locations rather than each
111  * having to discover for itself the location of meta, root, etc.  It makes
112  * sense for the likes of the pool of HTables class {@link HTablePool}, for
113  * instance (If concerned that a single {@link HConnection} is insufficient
114  * for sharing amongst clients in say an heavily-multithreaded environment,
115  * in practise its not proven to be an issue.  Besides, {@link HConnection} is
116  * implemented atop Hadoop RPC and as of this writing, Hadoop RPC does a
117  * connection per cluster-member, exclusively).
118  *
119  * <p>But sharing connections
120  * makes clean up of {@link HConnection} instances a little awkward.  Currently,
121  * clients cleanup by calling
122  * {@link #deleteConnection(Configuration, boolean)}.  This will shutdown the
123  * zookeeper connection the HConnection was using and clean up all
124  * HConnection resources as well as stopping proxies to servers out on the
125  * cluster. Not running the cleanup will not end the world; it'll
126  * just stall the closeup some and spew some zookeeper connection failed
127  * messages into the log.  Running the cleanup on a {@link HConnection} that is
128  * subsequently used by another will cause breakage so be careful running
129  * cleanup.
130  * <p>To create a {@link HConnection} that is not shared by others, you can
131  * create a new {@link Configuration} instance, pass this new instance to
132  * {@link #getConnection(Configuration)}, and then when done, close it up by
133  * doing something like the following:
134  * <pre>
135  * {@code
136  * Configuration newConfig = new Configuration(originalConf);
137  * HConnection connection = HConnectionManager.getConnection(newConfig);
138  * // Use the connection to your hearts' delight and then when done...
139  * HConnectionManager.deleteConnection(newConfig, true);
140  * }
141  * </pre>
142  * <p>Cleanup used to be done inside in a shutdown hook.  On startup we'd
143  * register a shutdown hook that called {@link #deleteAllConnections(boolean)}
144  * on its way out but the order in which shutdown hooks run is not defined so
145  * were problematic for clients of HConnection that wanted to register their
146  * own shutdown hooks so we removed ours though this shifts the onus for
147  * cleanup to the client.
148  */

208   public static HConnection getConnection(Configuration conf)
209   throws ZooKeeperConnectionException {
210     HConnectionKey connectionKey = new HConnectionKey(conf);
211     synchronized (HBASE_INSTANCES) {
212       HConnectionImplementation connection = HBASE_INSTANCES.get(connectionKey);
213       if (connection == null) {
214         connection = new HConnectionImplementation(conf, true);
215         HBASE_INSTANCES.put(connectionKey, connection);
216       }
217       connection.incCount();
218       return connection;
219     }

282   private static void deleteConnection(HConnection connection, boolean stopProxy,
283       boolean staleConnection) {
284     synchronized (HBASE_INSTANCES) {
285       for (Entry<HConnectionKey, HConnectionImplementation> connectionEntry : HBASE_INSTANCES
286           .entrySet()) {
287         if (connectionEntry.getValue() == connection) {
288           deleteConnection(connectionEntry.getKey(), stopProxy, staleConnection);
289           break;
290         }
291       }
292     }
293   }

295   private static void deleteConnection(HConnectionKey connectionKey,
296       boolean stopProxy, boolean staleConnection) {
297     synchronized (HBASE_INSTANCES) {
298       HConnectionImplementation connection = HBASE_INSTANCES
299           .get(connectionKey);
300       if (connection != null) {
301         connection.decCount();
302         if (connection.isZeroReference() || staleConnection) {
303           HBASE_INSTANCES.remove(connectionKey);
304           connection.close(stopProxy);
305         } else if (stopProxy) {
306           connection.stopProxyOnClose(stopProxy);
307         }
308       }else {
309         LOG.error("Connection not found in the list, can't delete it "+
310           "(connection key="+connectionKey+"). May be the key was modified?");
311       }
312     }

350    * This class makes it convenient for one to execute a command in the context
351    * of a {@link HConnection} instance based on the given {@link Configuration}.
352    *
353    * <p>
354    * If you find yourself wanting to use a {@link HConnection} for a relatively

363   public static abstract class HConnectable<T> {
364     public Configuration conf;
365

367       this.conf = conf;
368     }
369 
370     public abstract T connect(HConnection connection) throws IOException;
371   }

373   /**
374    * This convenience method invokes the given {@link HConnectable#connect}
375    * implementation using a {@link HConnection} instance that lasts just for the
376    * duration of that invocation.
377    *
378    * @param <T> the return type of the connect method
379    * @param connectable the {@link HConnectable} instance
380    * @return the value returned by the connect method
381    * @throws IOException
382    */
383   public static <T> T execute(HConnectable<T> connectable) throws IOException {
384     if (connectable == null || connectable.conf == null) {
385       return null;
386     }
387     Configuration conf = connectable.conf;
388     HConnection connection = HConnectionManager.getConnection(conf);
389     boolean connectSucceeded = false;
390     try {
391       T returnValue = connectable.connect(connection);
392       connectSucceeded = true;
393       return returnValue;
394     } finally {
395       try {
396         connection.close();
397       } catch (Exception e) {
398         if (connectSucceeded) {
399           throw new IOException("The connection to " + connection
400               + " could not be deleted.", e);
401         }
402       }
403     }
404   }

430     public HConnectionKey(Configuration conf) {
431       Map<String, String> m = new HashMap<String, String>();
432       if (conf != null) {
433         for (String property : CONNECTION_PROPERTIES) {
434           String value = conf.get(property);
435           if (value != null) {
436             m.put(property, value);
437           }
438         }
439       }
440       this.properties = Collections.unmodifiableMap(m);
441 
442       try {
443         User currentUser = User.getCurrent();
444         if (currentUser != null) {
445           username = currentUser.getName();
446         }
447       } catch (IOException ioe) {
448         LOG.warn("Error obtaining current user, skipping username in HConnectionKey",
449             ioe);
450       }
451     }

470     @Override

472       if (this == obj)

474       if (obj == null)

476       if (getClass() != obj.getClass())
477         return false;
478       HConnectionKey that = (HConnectionKey) obj;
479       if (this.username != null && !this.username.equals(that.username)) {

481       } else if (this.username == null && that.username != null) {

483       }
484       if (this.properties == null) {
485         if (that.properties != null) {
486           return false;
487         }
488       } else {
489         if (that.properties == null) {
490           return false;
491         }

493           String thisValue = this.properties.get(property);
494           String thatValue = that.properties.get(property);
495           if (thisValue == thatValue) {
496             continue;
497           }
498           if (thisValue == null || !thisValue.equals(thatValue)) {
499             return false;
500           }
501         }
502       }
503       return true;
504     }

689      */

691         MasterProtocolState masterProtocolState)
692         throws IOException, KeeperException, ServiceException {
693 
694       ZooKeeperKeepAliveConnection zkw;
695       try {
696         zkw = getKeepAliveZooKeeperWatcher();
697       } catch (IOException e) {
698         throw new ZooKeeperConnectionException("Can't connect to ZooKeeper", e);
699       }
700 
701       try {
702

708           LOG.info(msg);

734     /**
735      * Create a master, retries if necessary.
736      */
737     private MasterProtocol createMasterWithRetries(
738       MasterProtocolState masterProtocolState) throws MasterNotRunningException {
739 
740       // The lock must be at the beginning to prevent multiple master creation
741       //  (and leaks) in a multithread context
742       synchronized (this.masterAndZKLock) {
743         Exception exceptionCaught = null;
744         MasterProtocol master = null;
745         int tries = 0;
746         while (
747           !this.closed && master == null
748           ) {
749           tries++;
750           try {
751             master = createMasterInterface(masterProtocolState);
752           } catch (IOException e) {
753             exceptionCaught = e;
754           } catch (KeeperException e) {
755             exceptionCaught = e;
756           } catch (ServiceException e) {
757             exceptionCaught = e;
758           }
759 
760           if (exceptionCaught != null)
761             // It failed. If it's not the last try, we're going to wait a little
762           if (tries < numRetries) {
763             long pauseTime = ConnectionUtils.getPauseTime(this.pause, tries);
764             LOG.info("getMaster attempt " + tries + " of " + numRetries +
765               " failed; retrying after sleep of " +pauseTime, exceptionCaught);
766 
767             try {
768               Thread.sleep(pauseTime);
769             } catch (InterruptedException e) {
770               Thread.currentThread().interrupt();
771               throw new RuntimeException(
772                 "Thread was interrupted while trying to connect to master.", e);
773             }
774 
775           } else {
776             // Enough tries, we stop now
777             LOG.info("getMaster attempt " + tries + " of " + numRetries +
778               " failed; no more retrying.", exceptionCaught);
779             throw new MasterNotRunningException(exceptionCaught);
780           }
781         }
782 
783         if (master == null) {
784           // implies this.closed true
785           throw new MasterNotRunningException(
786             "Connection was closed while trying to get master");
787         }
788 
789         return master;
790       }

929       final byte [] row, boolean useCache, boolean retry)
930     throws IOException {

932       if (tableName == null || tableName.length == 0) {
933         throw new IllegalArgumentException(
934             "table name cannot be null or zero length");

938         ZooKeeperKeepAliveConnection zkw = getKeepAliveZooKeeperWatcher();

962           useCache, userRegionLock, retry);

967      * Search .META. for the HRegionLocation info that contains the table and
968      * row we're seeking. It will prefetch certain number of regions info and
969      * save them to the global region cache.
970      */
971     private void prefetchRegionCache(final byte[] tableName,
972         final byte[] row) {
973       // Implement a new visitor for MetaScanner, and use it to walk through
974       // the .META.
975       MetaScannerVisitor visitor = new MetaScannerVisitorBase() {
976         public boolean processRow(Result result) throws IOException {
977           try {
978             HRegionInfo regionInfo = MetaScanner.getHRegionInfo(result);
979             if (regionInfo == null) {
980               return true;
981             }
982 
983             // possible we got a region of a different table...
984             if (!Bytes.equals(regionInfo.getTableName(), tableName)) {
985               return false; // stop scanning
986             }
987             if (regionInfo.isOffline()) {
988               // don't cache offline regions
989               return true;
990             }
991 
992             ServerName serverName = HRegionInfo.getServerName(result);
993             if (serverName == null) {
994               return true; // don't cache it
995             }
996             // instantiate the location
997             HRegionLocation loc = new HRegionLocation(regionInfo, serverName.getHostname(),
998                 serverName.getPort());
999             // cache this meta entry
1000             cacheLocation(tableName, loc);
1001 
1002             return true;
1003           } catch (RuntimeException e) {
1004             throw new IOException(e);
1005           }
1006         }
1007       };
1008       try {
1009         // pre-fetch certain number of regions info at region cache.
1010         MetaScanner.metaScan(conf, visitor, tableName, row,
1011             this.prefetchRegionLimit);
1012       } catch (IOException e) {
1013         LOG.warn("Encountered problems when prefetch META table: ", e);
1014       }
1015     }

1017     /*
1018       * Search one of the meta tables (-ROOT- or .META.) for the HRegionLocation
1019       * info that contains the table and row we're seeking.
1020       */
1021     private HRegionLocation locateRegionInMeta(final byte [] parentTable,
1022       final byte [] tableName, final byte [] row, boolean useCache,
1023       Object regionLockObject, boolean retry)
1024     throws IOException {
1025       HRegionLocation location;
1026       // If we are supposed to be using the cache, look in the cache to see if
1027       // we already have the region.
1028       if (useCache) {
1029         location = getCachedLocation(tableName, row);
1030         if (location != null) {
1031           return location;
1032         }
1033       }
1034       int localNumRetries = retry ? numRetries : 1;
1035       // build the key of the meta region we should be looking for.
1036       // the extra 9's on the end are necessary to allow "exact" matches
1037       // without knowing the precise region names.
1038       byte [] metaKey = HRegionInfo.createRegionName(tableName, row,
1039         HConstants.NINES, false);
1040       for (int tries = 0; true; tries++) {
1041         if (tries >= localNumRetries) {
1042           throw new NoServerForRegionException("Unable to find region for "
1043             + Bytes.toStringBinary(row) + " after " + numRetries + " tries.");
1044         }
1045 
1046         HRegionLocation metaLocation = null;
1047         try {
1048           // locate the root or meta region
1049           metaLocation = locateRegion(parentTable, metaKey, true, false);
1050           // If null still, go around again.
1051           if (metaLocation == null) continue;
1052           ClientProtocol server =
1053             getClient(metaLocation.getHostname(), metaLocation.getPort());
1054 
1055           Result regionInfoRow = null;
1056           // This block guards against two threads trying to load the meta
1057           // region at the same time. The first will load the meta region and
1058           // the second will use the value that the first one found.
1059           synchronized (regionLockObject) {
1060             // If the parent table is META, we may want to pre-fetch some
1061             // region info into the global region cache for this table.
1062             if (Bytes.equals(parentTable, HConstants.META_TABLE_NAME) &&
1063                 (getRegionCachePrefetch(tableName)) )  {
1064               prefetchRegionCache(tableName, row);
1065             }
1066 
1067             // Check the cache again for a hit in case some other thread made the
1068             // same query while we were waiting on the lock. If not supposed to
1069             // be using the cache, delete any existing cached location so it won't
1070             // interfere.
1071             if (useCache) {
1072               location = getCachedLocation(tableName, row);
1073               if (location != null) {
1074                 return location;
1075               }
1076             } else {
1077               deleteCachedLocation(tableName, row);
1078             }
1079 
1080             // Query the root or meta region for the location of the meta region
1081             regionInfoRow = ProtobufUtil.getRowOrBefore(server,
1082               metaLocation.getRegionInfo().getRegionName(), metaKey,
1083               HConstants.CATALOG_FAMILY);
1084           }
1085           if (regionInfoRow == null) {
1086             throw new TableNotFoundException(Bytes.toString(tableName));
1087           }
1088 
1089           // convert the row result into the HRegionLocation we need!
1090           HRegionInfo regionInfo = MetaScanner.getHRegionInfo(regionInfoRow);
1091           if (regionInfo == null) {
1092             throw new IOException("HRegionInfo was null or empty in " +
1093               Bytes.toString(parentTable) + ", row=" + regionInfoRow);
1094           }
1095 
1096           // possible we got a region of a different table...
1097           if (!Bytes.equals(regionInfo.getTableName(), tableName)) {
1098             throw new TableNotFoundException(
1099                   "Table '" + Bytes.toString(tableName) + "' was not found, got: " +
1100                   Bytes.toString(regionInfo.getTableName()) + ".");
1101           }
1102           if (regionInfo.isSplit()) {
1103             throw new RegionOfflineException("the only available region for" +
1104               " the required row is a split parent," +
1105               " the daughters should be online soon: " +
1106               regionInfo.getRegionNameAsString());
1107           }
1108           if (regionInfo.isOffline()) {
1109             throw new RegionOfflineException("the region is offline, could" +
1110               " be caused by a disable table call: " +
1111               regionInfo.getRegionNameAsString());
1112           }
1113 
1114           ServerName serverName = HRegionInfo.getServerName(regionInfoRow);
1115           if (serverName == null) {
1116             throw new NoServerForRegionException("No server address listed " +
1117               "in " + Bytes.toString(parentTable) + " for region " +
1118               regionInfo.getRegionNameAsString() + " containing row " +
1119               Bytes.toStringBinary(row));
1120           }
1121 
1122           // Instantiate the location
1123           location =
1124               new HRegionLocation(regionInfo, serverName.getHostname(), serverName.getPort());
1125           cacheLocation(tableName, location);
1126           return location;
1127         } catch (TableNotFoundException e) {
1128           // if we got this error, probably means the table just plain doesn't
1129           // exist. rethrow the error immediately. this should always be coming
1130           // from the HTable constructor.
1131           throw e;
1132         } catch (IOException e) {
1133           if (e instanceof RemoteException) {
1134             e = RemoteExceptionHandler.decodeRemoteException((RemoteException) e);
1135           }
1136           if (tries < numRetries - 1) {
1137             if (LOG.isDebugEnabled()) {
1138               LOG.debug("locateRegionInMeta parentTable=" +
1139                 Bytes.toString(parentTable) + ", metaLocation=" +
1140                 ((metaLocation == null)? "null": "{" + metaLocation + "}") +
1141                 ", attempt=" + tries + " of " +
1142                 this.numRetries + " failed; retrying after sleep of " +
1143                 ConnectionUtils.getPauseTime(this.pause, tries) + " because: " + e.getMessage());
1144             }
1145           } else {
1146             throw e;
1147           }
1148           // Only relocate the parent region if necessary
1149           if(!(e instanceof RegionOfflineException ||
1150               e instanceof NoServerForRegionException)) {
1151             relocateRegion(parentTable, metaKey);
1152           }
1153         }
1154         try{
1155           Thread.sleep(ConnectionUtils.getPauseTime(this.pause, tries));
1156         } catch (InterruptedException e) {
1157           Thread.currentThread().interrupt();
1158           throw new IOException("Giving up trying to location region in " +
1159             "meta: thread is interrupted.");
1160         }
1161       }
1162     }

1164     /*
1165      * Search the cache for a location that fits our table and row key.
1166      * Return null if no suitable region is located. TODO: synchronization note
1167      *
1168      * <p>TODO: This method during writing consumes 15% of CPU doing lookup
1169      * into the Soft Reference SortedMap.  Improve.
1170      *
1171      * @param tableName
1172      * @param row
1173      * @return Null or region location found in cache.
1174      */
1175     HRegionLocation getCachedLocation(final byte [] tableName,
1176         final byte [] row) {
1177       SoftValueSortedMap<byte [], HRegionLocation> tableLocations =
1178         getTableLocations(tableName);
1179 
1180       // start to examine the cache. we can only do cache actions
1181       // if there's something in the cache for this table.
1182       if (tableLocations.isEmpty()) {
1183         return null;
1184       }
1185 
1186       HRegionLocation possibleRegion = tableLocations.get(row);
1187       if (possibleRegion != null) {
1188         return possibleRegion;
1189       }
1190 
1191       possibleRegion = tableLocations.lowerValueByKey(row);
1192       if (possibleRegion == null) {
1193         return null;
1194       }
1195 
1196       // make sure that the end key is greater than the row we're looking
1197       // for, otherwise the row actually belongs in the next region, not
1198       // this one. the exception case is when the endkey is
1199       // HConstants.EMPTY_END_ROW, signifying that the region we're
1200       // checking is actually the last region in the table.
1201       byte[] endKey = possibleRegion.getRegionInfo().getEndKey();
1202       if (Bytes.equals(endKey, HConstants.EMPTY_END_ROW) ||
1203           KeyValue.getRowComparator(tableName).compareRows(
1204               endKey, 0, endKey.length, row, 0, row.length) > 0) {
1205         return possibleRegion;
1206       }
1207 
1208       // Passed all the way through, so we got nothin - complete cache miss
1209       return null;
1210     }

1357     VersionedProtocol getProtocol(final String hostname,
1358         final int port, final Class <? extends VersionedProtocol> protocolClass,
1359         final long version) throws IOException {
1360       String rsName = Addressing.createHostAndPortStr(hostname, port);
1361       // See if we already have a connection (common case)
1362       Map<String, VersionedProtocol> protocols = this.servers.get(rsName);
1363       if (protocols == null) {
1364         protocols = new HashMap<String, VersionedProtocol>();
1365         Map<String, VersionedProtocol> existingProtocols =
1366           this.servers.putIfAbsent(rsName, protocols);
1367         if (existingProtocols != null) {
1368           protocols = existingProtocols;
1369         }
1370       }
1371       String protocol = protocolClass.getName();
1372       VersionedProtocol server = protocols.get(protocol);
1373       if (server == null) {
1374         // create a unique lock for this RS + protocol (if necessary)
1375         String lockKey = protocol + "@" + rsName;
1376         this.connectionLock.putIfAbsent(lockKey, lockKey);
1377         // get the RS lock
1378         synchronized (this.connectionLock.get(lockKey)) {
1379           // do one more lookup in case we were stalled above
1380           server = protocols.get(protocol);
1381           if (server == null) {
1382             try {
1383               // Only create isa when we need to.
1384               InetSocketAddress address = new InetSocketAddress(hostname, port);
1385               // definitely a cache miss. establish an RPC for this RS
1386               server = HBaseRPC.waitForProxy(
1387                   protocolClass, version, address, this.conf,
1388                   this.maxRPCAttempts, this.rpcTimeout, this.rpcTimeout);
1389               protocols.put(protocol, server);
1390             } catch (RemoteException e) {
1391               LOG.warn("RemoteException connecting to RS", e);
1392               // Throw what the RemoteException was carrying.
1393               throw e.unwrapRemoteException();
1394             }
1395           }
1396         }
1397       }

1880        *  added to the inProgress list.

1883        * @throws IOException - if we can't locate a region after multiple retries.
1884        */
1885       private void submit(List<Action<R>> actionsList, final long sleepTime) throws IOException {
1886         // group per location => regions server
1887         final Map<HRegionLocation, MultiAction<R>> actionsByServer =
1888           new HashMap<HRegionLocation, MultiAction<R>>();
1889         for (Action<R> aAction : actionsList) {
1890           final Row row = aAction.getAction();

1909         for (Entry<HRegionLocation, MultiAction<R>> e : actionsByServer.entrySet()) {

1911             createDelayedCallable(sleepTime, e.getKey(), e.getValue());

1913             new Triple<MultiAction<R>, HRegionLocation, Future<MultiResponse>>(
1914               e.getValue(), e.getKey(), this.pool.submit(callable));

1916         }

1937       /**
1938        * Parameterized batch processing, allowing varying return types for
1939        * different {@link Row} implementations.
1940        * Throws an exception on error. If there are no exceptions, it means that the 'results'
1941        *  array is clean.
1942        */
1943       private void processBatchCallback() throws IOException, InterruptedException {
1944         if (this.results.length != this.rows.size()) {
1945           throw new IllegalArgumentException(
1946             "argument results (size="+results.length+") must be the same size as " +
1947               "argument list (size="+this.rows.size()+")");
1948         }
1949         if (this.rows.isEmpty()) {
1950           return;
1951         }
1952 
1953         // We keep the number of retry per action.
1954         int[] nbRetries = new int[this.results.length];
1955

1957         //  indexes will remain constant, allowing a direct lookup.

1968         //  we had an exception 'DoNotRetry'
1969         //  we had more than numRetries for any action
1970         //  In this case, we will finish the current retries but we won't start new ones.
1971         boolean lastRetry = false;
1972         // despite its name numRetries means number of tries. So if numRetries == 1 it means we
1973         //  won't retry. And we compare vs. 2 in case someone set it to zero.

2000                 if (noRetry) {
2001                   addToErrorsLists(exception, row, currentTask);
2002                 } else {
2003                   lastRetry = addToReplay(nbRetries, action);
2004                 }
2005               }
2006             }
2007           } else { // Success or partial success
2008             // Analyze detailed results. We can still have individual failures to be redo.
2009             // two specific exceptions are managed:
2010             //  - DoNotRetryIOException: we continue to retry for other actions
2011             //  - RegionMovedException: we update the cache with the new region location
2012             for (Entry<byte[], List<Pair<Integer, Object>>> resultsForRS :
2013                 responses.getResults().entrySet()) {
2014               for (Pair<Integer, Object> regionResult : resultsForRS.getValue()) {

2018

2020                 if (result == null || result instanceof Throwable) {

2022                   hci.updateCachedLocations(this.tableName, row, result);
2023                   if (result instanceof DoNotRetryIOException || noRetry) {
2024                     addToErrorsLists((Exception)result, row, currentTask);
2025                   } else {
2026                     lastRetry = addToReplay(nbRetries, correspondingAction);
2027                   }

2030                     this.callback.update(resultsForRS.getKey(),
2031                       this.rows.get(regionResult.getFirst()).getRow(),

2033                 }

2035             }
2036           }
2037 
2038           // Retry all actions in toReplay then clear it.
2039           if (!noRetry && !toReplay.isEmpty()) {
2040             doRetry();

2042               noRetry = true;
2043             }
2044           }
2045         }
2046 
2047         if (!exceptions.isEmpty()) {
