77    /**
78     * Remove an entire region from the table directory via archiving the region's hfiles.
79     * @param fs {@link FileSystem} from which to remove the region
80     * @param rootdir {@link Path} to the root directory where hbase files are stored (for building
81     *          the archive path)
82     * @param tableDir {@link Path} to where the table is being stored (for building the archive path)
83     * @param regionDir {@link Path} to where a region is being stored (for building the archive path)
84     * @return <tt>true</tt> if the region was sucessfully deleted. <tt>false</tt> if the filesystem
85     *         operations could not complete.
86     * @throws IOException if the request cannot be completed
87     */
88    public static boolean archiveRegion(FileSystem fs, Path rootdir, Path tableDir, Path regionDir)
89        throws IOException {
90      if (LOG.isDebugEnabled()) {
91        LOG.debug("ARCHIVING region " + regionDir.toString());
92      }
93  
94      // otherwise, we archive the files
95      // make sure we can archive
96      if (tableDir == null || regionDir == null) {
97        LOG.error("No archive directory could be found because tabledir (" + tableDir
98            + ") or regiondir (" + regionDir + "was null. Deleting files instead.");
99        deleteRegionWithoutArchiving(fs, regionDir);
100       // we should have archived, but failed to. Doesn't matter if we deleted
101       // the archived files correctly or not.
102       return false;
103     }
104 
105     // make sure the regiondir lives under the tabledir
106     Preconditions.checkArgument(regionDir.toString().startsWith(tableDir.toString()));
107     Path regionArchiveDir = HFileArchiveUtil.getRegionArchiveDir(fs.getConf(), tableDir, regionDir);
108 
109     LOG.debug("Have an archive directory, preparing to move files");
110     FileStatusConverter getAsFile = new FileStatusConverter(fs);
111     // otherwise, we attempt to archive the store files
112 
113     // build collection of just the store directories to archive
114     Collection<File> toArchive = new ArrayList<File>();
115     final PathFilter dirFilter = new FSUtils.DirFilter(fs);
116     PathFilter nonHidden = new PathFilter() {
117       @Override
118       public boolean accept(Path file) {
119         return dirFilter.accept(file) && !file.getName().toString().startsWith(".");
120       }
121     };
122     FileStatus[] storeDirs = FSUtils.listStatus(fs, regionDir, nonHidden);
123     // if there no files, we can just delete the directory and return;
124     if (storeDirs == null) {
125       LOG.debug("Region directory (" + regionDir + ") was empty, just deleting and returning!");
126       return deleteRegionWithoutArchiving(fs, regionDir);
127     }
128 
129     // convert the files in the region to a File
130     toArchive.addAll(Lists.transform(Arrays.asList(storeDirs), getAsFile));
131     LOG.debug("Archiving:" + toArchive);
132     boolean success = false;
133     try {
134       success = resolveAndArchive(fs, regionArchiveDir, toArchive);
135     } catch (IOException e) {
136       success = false;
137     }
138 
139     // if that was successful, then we delete the region
140     if (success) {
141       LOG.debug("Successfully resolved and archived, now can just delete region.");
142       return deleteRegionWithoutArchiving(fs, regionDir);
143     }
144 
145     throw new IOException("Received error when attempting to archive files (" + toArchive
146         + "), cannot delete region directory.");
147   }

149   /**
150    * Remove from the specified region the store files of the specified column family,
151    * either by archiving them or outright deletion
152    * @param fs the filesystem where the store files live
153    * @param conf {@link Configuration} to examine to determine the archive directory
154    * @param parent Parent region hosting the store files
155    * @param tableDir {@link Path} to where the table is being stored (for building the archive path)
156    * @param family the family hosting the store files
157    * @throws IOException if the files could not be correctly disposed.
158    */
159   public static void archiveFamily(FileSystem fs, Configuration conf,
160       HRegionInfo parent, Path tableDir, byte[] family) throws IOException {
161     Path familyDir = new Path(tableDir, new Path(parent.getEncodedName(), Bytes.toString(family)));
162     FileStatus[] storeFiles = FSUtils.listStatus(fs, familyDir);
163     if (storeFiles == null) {
164       LOG.debug("No store files to dispose for region=" + parent.getRegionNameAsString() +
165           ", family=" + Bytes.toString(family));
166       return;
167     }
168 
169     FileStatusConverter getAsFile = new FileStatusConverter(fs);
170     Collection<File> toArchive = Lists.transform(Arrays.asList(storeFiles), getAsFile);
171     Path storeArchiveDir = HFileArchiveUtil.getStoreArchivePath(conf, parent, tableDir, family);
172 
173     // do the actual archive
174     if (!resolveAndArchive(fs, storeArchiveDir, toArchive)) {
175       throw new IOException("Failed to archive/delete all the files for region:"
176           + Bytes.toString(parent.getRegionName()) + ", family:" + Bytes.toString(family)
177           + " into " + storeArchiveDir + ". Something is probably awry on the filesystem.");
178     }
179   }

181   /**
182    * Remove the store files, either by archiving them or outright deletion
183    * @param fs the filesystem where the store files live
184    * @param parent Parent region hosting the store files
185    * @param conf {@link Configuration} to examine to determine the archive directory
186    * @param family the family hosting the store files
187    * @param compactedFiles files to be disposed of. No further reading of these files should be
188    *          attempted; otherwise likely to cause an {@link IOException}
189    * @throws IOException if the files could not be correctly disposed.
190    */
191   public static void archiveStoreFiles(FileSystem fs, HRegion parent,
192       Configuration conf, byte[] family, Collection<StoreFile> compactedFiles) throws IOException {
193 
194     // sometimes in testing, we don't have rss, so we need to check for that
195     if (fs == null) {
196       LOG.warn("Passed filesystem is null, so just deleting the files without archiving for region:"
197           + Bytes.toString(parent.getRegionName()) + ", family:" + Bytes.toString(family));
198       deleteStoreFilesWithoutArchiving(compactedFiles);
199       return;
200     }
201 
202     // short circuit if we don't have any files to delete
203     if (compactedFiles.size() == 0) {
204       LOG.debug("No store files to dispose, done!");
205       return;
206     }
207 
208     // build the archive path
209     if (parent == null || family == null) throw new IOException(
210         "Need to have a parent region and a family to archive from.");
211 
212     Path storeArchiveDir = HFileArchiveUtil.getStoreArchivePath(conf, parent, family);
213 
214     // make sure we don't archive if we can't and that the archive dir exists
215     if (!fs.mkdirs(storeArchiveDir)) {
216       throw new IOException("Could not make archive directory (" + storeArchiveDir + ") for store:"
217           + Bytes.toString(family) + ", deleting compacted files instead.");
218     }
219 
220     // otherwise we attempt to archive the store files
221     LOG.debug("Archiving compacted store files.");
222 
223     // wrap the storefile into a File
224     StoreToFile getStorePath = new StoreToFile(fs);
225     Collection<File> storeFiles = Collections2.transform(compactedFiles, getStorePath);
226 
227     // do the actual archive
228     if (!resolveAndArchive(fs, storeArchiveDir, storeFiles)) {
229       throw new IOException("Failed to archive/delete all the files for region:"
230           + Bytes.toString(parent.getRegionName()) + ", family:" + Bytes.toString(family)
231           + " into " + storeArchiveDir + ". Something is probably awry on the filesystem.");
232     }
233   }

235   /**
236    * Archive the given files and resolve any conflicts with existing files via appending the time
237    * archiving started (so all conflicts in the same group have the same timestamp appended).
238    * <p>
239    * If any of the passed files to archive are directories, archives all the files under that
240    * directory. Archive directory structure for children is the base archive directory name + the
241    * parent directory and is built recursively is passed files are directories themselves.
242    * @param fs {@link FileSystem} on which to archive the files
243    * @param baseArchiveDir base archive directory to archive the given files
244    * @param toArchive files to be archived
245    * @return <tt>true</tt> on success, <tt>false</tt> otherwise
246    * @throws IOException on unexpected failure
247    */
248   private static boolean resolveAndArchive(FileSystem fs, Path baseArchiveDir,
249       Collection<File> toArchive) throws IOException {
250     LOG.debug("Starting to archive files:" + toArchive);
251     long start = EnvironmentEdgeManager.currentTimeMillis();
252     List<File> failures = resolveAndArchive(fs, baseArchiveDir, toArchive, start);
253 
254     // clean out the failures by just deleting them
255     if (failures.size() > 0) {
256       try {
257         LOG.error("Failed to complete archive, deleting extra store files.");
258         deleteFilesWithoutArchiving(failures);
259       } catch (IOException e) {
260         LOG.warn("Failed to delete store file(s) when archiving failed", e);
261       }
262       return false;
263     }
264     return true;
265   }

267   /**
268    * Resolve any conflict with an existing archive file via timestamp-append
269    * renaming of the existing file and then archive the passed in files.
270    * @param fs {@link FileSystem} on which to archive the files
271    * @param baseArchiveDir base archive directory to store the files. If any of
272    *          the files to archive are directories, will append the name of the
273    *          directory to the base archive directory name, creating a parallel
274    *          structure.
275    * @param toArchive files/directories that need to be archvied
276    * @param start time the archiving started - used for resolving archive
277    *          conflicts.
278    * @return the list of failed to archive files.
279    * @throws IOException if an unexpected file operation exception occured
280    */
281   private static List<File> resolveAndArchive(FileSystem fs, Path baseArchiveDir,
282       Collection<File> toArchive, long start) throws IOException {
283     // short circuit if no files to move
284     if (toArchive.size() == 0) return Collections.emptyList();
285 
286     LOG.debug("moving files to the archive directory: " + baseArchiveDir);
287 
288     // make sure the archive directory exists
289     if (!fs.exists(baseArchiveDir)) {
290       if (!fs.mkdirs(baseArchiveDir)) {
291         throw new IOException("Failed to create the archive directory:" + baseArchiveDir
292             + ", quitting archive attempt.");
293       }
294       LOG.debug("Created archive directory:" + baseArchiveDir);
295     }
296 
297     List<File> failures = new ArrayList<File>();
298     String startTime = Long.toString(start);
299     for (File file : toArchive) {
300       // if its a file archive it
301       try {
302         LOG.debug("Archiving:" + file);
303         if (file.isFile()) {
304           // attempt to archive the file
305           if (!resolveAndArchiveFile(baseArchiveDir, file, startTime)) {
306             LOG.warn("Couldn't archive " + file + " into backup directory: " + baseArchiveDir);
307             failures.add(file);
308           }
309         } else {
310           // otherwise its a directory and we need to archive all files
311           LOG.debug(file + " is a directory, archiving children files");
312           // so we add the directory name to the one base archive
313           Path parentArchiveDir = new Path(baseArchiveDir, file.getName());
314           // and then get all the files from that directory and attempt to
315           // archive those too
316           Collection<File> children = file.getChildren();
317           failures.addAll(resolveAndArchive(fs, parentArchiveDir, children, start));
318         }
319       } catch (IOException e) {
320         LOG.warn("Failed to archive file: " + file, e);
321         failures.add(file);
322       }
323     }
324     return failures;
325   }

327   /**
328    * Attempt to archive the passed in file to the archive directory.
329    * <p>
330    * If the same file already exists in the archive, it is moved to a timestamped directory under
331    * the archive directory and the new file is put in its place.
332    * @param archiveDir {@link Path} to the directory that stores the archives of the hfiles
333    * @param currentFile {@link Path} to the original HFile that will be archived
334    * @param archiveStartTime time the archiving started, to resolve naming conflicts
335    * @return <tt>true</tt> if the file is successfully archived. <tt>false</tt> if there was a
336    *         problem, but the operation still completed.
337    * @throws IOException on failure to complete {@link FileSystem} operations.
338    */
339   private static boolean resolveAndArchiveFile(Path archiveDir, File currentFile,
340       String archiveStartTime) throws IOException {
341     // build path as it should be in the archive
342     String filename = currentFile.getName();
343     Path archiveFile = new Path(archiveDir, filename);
344     FileSystem fs = currentFile.getFileSystem();
345 
346     // if the file already exists in the archive, move that one to a timestamped backup. This is a
347     // really, really unlikely situtation, where we get the same name for the existing file, but
348     // is included just for that 1 in trillion chance.
349     if (fs.exists(archiveFile)) {
350       if (LOG.isDebugEnabled()) {
351         LOG.debug("File:" + archiveFile + " already exists in archive, moving to "
352             + "timestamped backup and overwriting current.");
353       }
354 
355       // move the archive file to the stamped backup
356       Path backedupArchiveFile = new Path(archiveDir, filename + SEPARATOR + archiveStartTime);
357       if (!fs.rename(archiveFile, backedupArchiveFile)) {
358         LOG.error("Could not rename archive file to backup: " + backedupArchiveFile
359             + ", deleting existing file in favor of newer.");
360         // try to delete the exisiting file, if we can't rename it
361         if (!fs.delete(archiveFile, false)) {
362           throw new IOException("Couldn't delete existing archive file (" + archiveFile
363               + ") or rename it to the backup file (" + backedupArchiveFile
364               + ")to make room for similarly named file.");
365         }
366       }
367       LOG.debug("Backed up archive file from: " + archiveFile);
368     }
369 
370     LOG.debug("No existing file in archive for:" + archiveFile + ", free to archive original file.");
371 
372     // at this point, we should have a free spot for the archive file
373     if (currentFile.moveAndClose(archiveFile)) {
374       LOG.error("Failed to archive file:" + currentFile);
375       return false;
376     } else if (LOG.isDebugEnabled()) {
377       LOG.debug("Finished archiving file from: " + currentFile + ", to: " + archiveFile);
378     }
379     return true;
380   }

382   /**
383    * Simple delete of regular files from the {@link FileSystem}.
384    * <p>
385    * This method is a more generic implementation that the other deleteXXX
386    * methods in this class, allowing more code reuse at the cost of a couple
387    * more, short-lived objects (which should have minimum impact on the jvm).
388    * @param fs {@link FileSystem} where the files live
389    * @param files {@link Collection} of files to be deleted
390    * @throws IOException if a file cannot be deleted. All files will be
391    *           attempted to deleted before throwing the exception, rather than
392    *           failing at the first file.
393    */
394   private static void deleteFilesWithoutArchiving(Collection<File> files) throws IOException {
395     List<IOException> errors = new ArrayList<IOException>(0);
396     for (File file : files) {
397       try {
398         LOG.debug("Deleting region file:" + file);
399         file.delete();
400       } catch (IOException e) {
401         LOG.error("Failed to delete file:" + file);
402         errors.add(e);
403       }
404     }
405     if (errors.size() > 0) {
406       throw MultipleIOException.createIOException(errors);
407     }
408   }

411    * Without regard for backup, delete a region. Should be used with caution.
412    * @param regionDir {@link Path} to the region to be deleted.
413    * @param fs FileSystem from which to delete the region
414    * @return <tt>true</tt> on successful deletion, <tt>false</tt> otherwise
415    * @throws IOException on filesystem operation failure
416    */
417   private static boolean deleteRegionWithoutArchiving(FileSystem fs, Path regionDir)
418       throws IOException {
419     if (fs.delete(regionDir, true)) {
420       LOG.debug("Deleted all region files in: " + regionDir);
421       return true;
422     }
423     LOG.debug("Failed to delete region directory:" + regionDir);
424     return false;

427   /**
428    * Just do a simple delete of the given store files
429    * <p>
430    * A best effort is made to delete each of the files, rather than bailing on the first failure.
431    * <p>
432    * This method is preferable to {@link #deleteFilesWithoutArchiving(Collection)} since it consumes
433    * less resources, but is limited in terms of usefulness
434    * @param compactedFiles store files to delete from the file system.
435    * @throws IOException if a file cannot be deleted. All files will be attempted to deleted before
436    *           throwing the exception, rather than failing at the first file.
437    */
438   private static void deleteStoreFilesWithoutArchiving(Collection<StoreFile> compactedFiles)
439       throws IOException {
440     LOG.debug("Deleting store files without archiving.");
441     List<IOException> errors = new ArrayList<IOException>(0);
442     for (StoreFile hsf : compactedFiles) {
443       try {
444         hsf.deleteReader();
445       } catch (IOException e) {
446         LOG.error("Failed to delete store file:" + hsf.getPath());
447         errors.add(e);
448       }
449     }
450     if (errors.size() > 0) {
451       throw MultipleIOException.createIOException(errors);
452     }
453   }

569   /**
570    * A {@link File} that wraps a simple {@link Path} on a {@link FileSystem}.
571    */
572   private static class FileablePath extends File {
573     private final Path file;
574     private final FileStatusConverter getAsFile;
575 
576     public FileablePath(FileSystem fs, Path file) {
577       super(fs);
578       this.file = file;
579       this.getAsFile = new FileStatusConverter(fs);
580     }
581 
582     @Override
583     public void delete() throws IOException {
584       if (!fs.delete(file, true)) throw new IOException("Failed to delete:" + this.file);
585     }
586 
587     @Override
588     public String getName() {
589       return file.getName();
590     }
591 
592     @Override
593     public Collection<File> getChildren() throws IOException {
594       if (fs.isFile(file)) return Collections.emptyList();
595       return Collections2.transform(Arrays.asList(fs.listStatus(file)), getAsFile);
596     }
597 
598     @Override
599     public boolean isFile() throws IOException {
600       return fs.isFile(file);
601     }
602 
603     @Override
604     public void close() throws IOException {
605       // NOOP - files are implicitly closed on removal
606     }
607 
608     @Override
609     Path getPath() {
610       return file;
611     }
612   }
