74    private static int insertData(String tableName, String column, double prob)
75        throws IOException {
76      Random rng = new Random();
77      int count = 0;
78      HTable table = new HTable(TEST_UTIL.getConfiguration(), tableName);
79      byte[] k = new byte[3];
80      byte [][] famAndQf = KeyValue.parseColumn(Bytes.toBytes(column));
81      for (byte b1 = 'a'; b1 < 'z'; b1++) {
82        for (byte b2 = 'a'; b2 < 'z'; b2++) {
83          for (byte b3 = 'a'; b3 < 'z'; b3++) {
84            if (rng.nextDouble() < prob) {
85              k[0] = b1;
86              k[1] = b2;
87              k[2] = b3;
88              Put put = new Put(k);
89              put.setWriteToWAL(false);
90              put.add(famAndQf[0], famAndQf[1], k);
91              table.put(put);
92              count++;
93            }
94          }
95        }
96      }
97      table.flushCommits();
98      return count;
99    }

101   private static int countCellSet(CellSetModel model) {
102     int count = 0;
103     Iterator<RowModel> rows = model.getRows().iterator();
104     while (rows.hasNext()) {
105       RowModel row = rows.next();
106       Iterator<CellModel> cells = row.getCells().iterator();
107       while (cells.hasNext()) {
108         cells.next();
109         count++;
110       }
111     }
112     return count;
113   }

115   private static int fullTableScan(ScannerModel model) throws IOException {
116     model.setBatch(100);
117     Response response = client.put("/" + TABLE + "/scanner",
118       Constants.MIMETYPE_PROTOBUF, model.createProtobufOutput());
119     assertEquals(response.getCode(), 201);
120     String scannerURI = response.getLocation();
121     assertNotNull(scannerURI);
122     int count = 0;
123     while (true) {
124       response = client.get(scannerURI, Constants.MIMETYPE_PROTOBUF);
125       assertTrue(response.getCode() == 200 || response.getCode() == 204);
126       if (response.getCode() == 200) {
127         CellSetModel cellSet = new CellSetModel();
128         cellSet.getObjectFromMessage(response.getBody());
129         Iterator<RowModel> rows = cellSet.getRows().iterator();
130         while (rows.hasNext()) {
131           RowModel row = rows.next();
132           Iterator<CellModel> cells = row.getCells().iterator();
133           while (cells.hasNext()) {
134             cells.next();
135             count++;
136           }
137         }
138       } else {
139         break;
140       }
141     }
142     // delete the scanner
143     response = client.delete(scannerURI);
144     assertEquals(response.getCode(), 200);
145     return count;
146   }

180   @Test
181   public void testSimpleScannerXML() throws IOException, JAXBException {
182     final int BATCH_SIZE = 5;
183     // new scanner
184     ScannerModel model = new ScannerModel();
185     model.setBatch(BATCH_SIZE);
186     model.addColumn(Bytes.toBytes(COLUMN_1));
187     StringWriter writer = new StringWriter();
188     marshaller.marshal(model, writer);
189     byte[] body = Bytes.toBytes(writer.toString());
190 
191     // test put operation is forbidden in read-only mode
192     conf.set("hbase.rest.readonly", "true");
193     Response response = client.put("/" + TABLE + "/scanner",
194       Constants.MIMETYPE_XML, body);
195     assertEquals(response.getCode(), 403);
196     String scannerURI = response.getLocation();
197     assertNull(scannerURI);
198 
199     // recall previous put operation with read-only off
200     conf.set("hbase.rest.readonly", "false");
201     response = client.put("/" + TABLE + "/scanner", Constants.MIMETYPE_XML,
202       body);
203     assertEquals(response.getCode(), 201);
204     scannerURI = response.getLocation();
205     assertNotNull(scannerURI);
206 
207     // get a cell set
208     response = client.get(scannerURI, Constants.MIMETYPE_XML);
209     assertEquals(response.getCode(), 200);
210     CellSetModel cellSet = (CellSetModel)
211       unmarshaller.unmarshal(new ByteArrayInputStream(response.getBody()));
212     // confirm batch size conformance
213     assertEquals(countCellSet(cellSet), BATCH_SIZE);
214 
215     // test delete scanner operation is forbidden in read-only mode
216     conf.set("hbase.rest.readonly", "true");
217     response = client.delete(scannerURI);
218     assertEquals(response.getCode(), 403);
219 
220     // recall previous delete scanner operation with read-only off
221     conf.set("hbase.rest.readonly", "false");
222     response = client.delete(scannerURI);
223     assertEquals(response.getCode(), 200);
224   }

285     conf.set("hbase.rest.readonly", "false");
286     response = client.put("/" + TABLE + "/scanner",
287       Constants.MIMETYPE_PROTOBUF, model.createProtobufOutput());
288     assertEquals(response.getCode(), 201);
289     scannerURI = response.getLocation();
290     assertNotNull(scannerURI);
291 
292     // get a cell
293     response = client.get(scannerURI, Constants.MIMETYPE_BINARY);
294     assertEquals(response.getCode(), 200);
295     // verify that data was returned
296     assertTrue(response.getBody().length > 0);
297     // verify that the expected X-headers are present
298     boolean foundRowHeader = false, foundColumnHeader = false,
299       foundTimestampHeader = false;
300     for (Header header: response.getHeaders()) {
301       if (header.getName().equals("X-Row")) {
302         foundRowHeader = true;
303       } else if (header.getName().equals("X-Column")) {
304         foundColumnHeader = true;
305       } else if (header.getName().equals("X-Timestamp")) {
306         foundTimestampHeader = true;
307       }
308     }
309     assertTrue(foundRowHeader);
310     assertTrue(foundColumnHeader);
311     assertTrue(foundTimestampHeader);
312 
313     // test delete scanner operation is forbidden in read-only mode
314     conf.set("hbase.rest.readonly", "true");
315     response = client.delete(scannerURI);
316     assertEquals(response.getCode(), 403);
317 
318     // recall previous delete scanner operation with read-only off
319     conf.set("hbase.rest.readonly", "false");
320     response = client.delete(scannerURI);
321     assertEquals(response.getCode(), 200);
