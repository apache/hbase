136   public ZooKeeperWatcher(Configuration conf, String descriptor,
137       Abortable abortable, boolean canCreateBaseZNode)
138   throws IOException, ZooKeeperConnectionException {
139     this.conf = conf;
140     // Capture a stack trace now.  Will print it out later if problem so we can
141     // distingush amongst the myriad ZKWs.
142     try {
143       throw new Exception("ZKW CONSTRUCTOR STACK TRACE FOR DEBUGGING");
144     } catch (Exception e) {
145       this.constructorCaller = e;
146     }
147     this.quorum = ZKConfig.getZKQuorumServersString(conf);
148     // Identifier will get the sessionid appended later below down when we
149     // handle the syncconnect event.
150     this.identifier = descriptor;
151     this.abortable = abortable;
152     setNodeNames(conf);
153     this.recoverableZooKeeper = ZKUtil.connect(conf, quorum, this, descriptor);
154     if (canCreateBaseZNode) {
155       createBaseZNodes();
156     }
157   }

171           prefix("Unexpected KeeperException creating base node"), e);

257   /**
258    * Method called from ZooKeeper for events and connection status.
259    * <p>
260    * Valid events are passed along to listeners.  Connection status changes
261    * are dealt with locally.
262    */
263   @Override
264   public void process(WatchedEvent event) {
265     LOG.debug(prefix("Received ZooKeeper Event, " +
266         "type=" + event.getType() + ", " +
267         "state=" + event.getState() + ", " +
268         "path=" + event.getPath()));
269 
270     switch(event.getType()) {
271 
272       // If event type is NONE, this is a connection status change
273       case None: {
274         connectionEvent(event);
275         break;
276       }
277 
278       // Otherwise pass along to the listeners
279 
280       case NodeCreated: {
281         for(ZooKeeperListener listener : listeners) {
282           listener.nodeCreated(event.getPath());
283         }
284         break;
285       }
286 
287       case NodeDeleted: {
288         for(ZooKeeperListener listener : listeners) {
289           listener.nodeDeleted(event.getPath());
290         }
291         break;
292       }
293 
294       case NodeDataChanged: {
295         for(ZooKeeperListener listener : listeners) {
296           listener.nodeDataChanged(event.getPath());
297         }
298         break;
299       }
300 
301       case NodeChildrenChanged: {
302         for(ZooKeeperListener listener : listeners) {
303           listener.nodeChildrenChanged(event.getPath());
304         }
305         break;
306       }
307     }
308   }

312   /**
313    * Called when there is a connection-related event via the Watcher callback.
314    * <p>
315    * If Disconnected or Expired, this should shutdown the cluster. But, since
316    * we send a KeeperException.SessionExpiredException along with the abort
317    * call, it's possible for the Abortable to catch it and try to create a new
318    * session with ZooKeeper. This is what the client does in HCM.
319    * <p>
320    * @param event
321    */
322   private void connectionEvent(WatchedEvent event) {
323     switch(event.getState()) {
324       case SyncConnected:
325         // Now, this callback can be invoked before the this.zookeeper is set.
326         // Wait a little while.
327         long finished = System.currentTimeMillis() +
328           this.conf.getLong("hbase.zookeeper.watcher.sync.connected.wait", 2000);
329         while (System.currentTimeMillis() < finished) {
330           Threads.sleep(1);
331           if (this.recoverableZooKeeper != null) break;
332         }
333         if (this.recoverableZooKeeper == null) {
334           LOG.error("ZK is null on connection event -- see stack trace " +
335             "for the stack trace when constructor was called on this zkw",
336             this.constructorCaller);
337           throw new NullPointerException("ZK is null");
338         }
339         this.identifier = this.identifier + "-0x" +
340           Long.toHexString(this.recoverableZooKeeper.getSessionId());
341         // Update our identifier.  Otherwise ignore.
342         LOG.debug(this.identifier + " connected");
343         break;
344 
345       case SaslAuthenticated:
346         if (ZKUtil.isSecureZooKeeper(this.conf)) {
347           // We are authenticated, clients can proceed.
348           saslLatch.countDown();
349         }
350         break;
351 
352       case AuthFailed:
353         if (ZKUtil.isSecureZooKeeper(this.conf)) {
354           // We could not be authenticated, but clients should proceed anyway.
355           // Only access to znodes that require SASL authentication will be
356           // denied. The client may never need to access them.
357           saslLatch.countDown();
358         }
359         break;
360 
361       // Abort the server if Disconnected or Expired
362       case Disconnected:
363         LOG.debug(prefix("Received Disconnected from ZooKeeper, ignoring"));
364         break;
365 
366       case Expired:
367         if (ZKUtil.isSecureZooKeeper(this.conf)) {
368           // We consider Expired equivalent to AuthFailed for this
369           // connection. Authentication is never going to complete. The
370           // client should proceed to do cleanup.
371           saslLatch.countDown();
372         }
373         String msg = prefix(this.identifier + " received expired from " +
374           "ZooKeeper, aborting");
375         // TODO: One thought is to add call to ZooKeeperListener so say,
376         // ZooKeeperNodeTracker can zero out its data values.
377         if (this.abortable != null) this.abortable.abort(msg,
378             new KeeperException.SessionExpiredException());
379         break;
380         
381       case ConnectedReadOnly:
382         break; 
383         
384       default:
385         throw new IllegalStateException("Received event is not valid.");
386     }
387   }

389   /**
390    * Forces a synchronization of this ZooKeeper client connection.
391    * <p>
392    * Executing this method before running other methods will ensure that the
393    * subsequent operations are up-to-date and consistent as of the time that
394    * the sync is complete.
395    * <p>
396    * This is used for compareAndSwap type operations where we need to read the
397    * data of an existing node and delete or transition that node, utilizing the
398    * previously read version and data.  We want to ensure that the version read
399    * is up-to-date from when we begin the operation.
400    */
401   public void sync(String path) {
402     this.recoverableZooKeeper.sync(path, null, null);
403   }
