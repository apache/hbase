139   static class FailedServers {
140     private final LinkedList<Pair<Long, String>> failedServers = new
141         LinkedList<Pair<Long, java.lang.String>>();
142     private final int recheckServersTimeout;
143 
144     FailedServers(Configuration conf) {
145       this.recheckServersTimeout = conf.getInt(
146           FAILED_SERVER_EXPIRY_KEY, FAILED_SERVER_EXPIRY_DEFAULT);
147     }
148 
149     /**
150      * Add an address to the list of the failed servers list.
151      */
152     public synchronized void addToFailedServers(InetSocketAddress address) {
153       final long expiry = EnvironmentEdgeManager.currentTimeMillis() + recheckServersTimeout;
154       failedServers.addFirst(new Pair<Long, String>(expiry, address.toString()));
155     }
156 
157     /**
158      * Check if the server should be considered as bad. Clean the old entries of the list.
159      *
160      * @return true if the server is in the failed servers list
161      */
162     public synchronized boolean isFailedServer(final InetSocketAddress address) {
163       if (failedServers.isEmpty()) {
164         return false;
165       }
166 
167       final String lookup = address.toString();
168       final long now = EnvironmentEdgeManager.currentTimeMillis();
169 
170       // iterate, looking for the search entry and cleaning expired entries
171       Iterator<Pair<Long, String>> it = failedServers.iterator();
172       while (it.hasNext()) {
173         Pair<Long, String> cur = it.next();
174         if (cur.getFirst() < now) {
175           it.remove();
176         } else {
177           if (lookup.equals(cur.getSecond())) {
178             return true;
179           }
180         }
181       }
182 
183       return false;
184     }
185

320   /** Thread that reads responses and notifies callers.  Each connection owns a
321    * socket connected to a remote address.  Calls are multiplexed through this
322    * socket: responses may be delivered out of order. */

325     protected ConnectionId remoteId;
326     protected Socket socket = null;                 // connected socket

328     protected DataOutputStream out;

330     private String serverPrincipal;  // server's krb5 principal name

339     protected final AtomicLong lastActivity = new AtomicLong();// last I/O activity time

343     Connection(ConnectionId remoteId) throws IOException {
344       if (remoteId.getAddress().isUnresolved()) {
345         throw new UnknownHostException("unknown host: " +
346                                        remoteId.getAddress().getHostName());
347       }
348       this.server = remoteId.getAddress();
349 
350       UserGroupInformation ticket = remoteId.getTicket().getUGI();
351       Class<?> protocol = remoteId.getProtocol();
352       this.useSasl = User.isHBaseSecurityEnabled(conf);
353       if (useSasl && protocol != null) {
354         TokenInfo tokenInfo = protocol.getAnnotation(TokenInfo.class);
355         if (tokenInfo != null) {
356           TokenSelector<? extends TokenIdentifier> tokenSelector =
357               tokenHandlers.get(tokenInfo.value());
358           if (tokenSelector != null) {
359             token = tokenSelector.selectToken(new Text(clusterId),
360                 ticket.getTokens());
361           } else if (LOG.isDebugEnabled()) {
362             LOG.debug("No token selector found for type "+tokenInfo.value());
363           }
364         }
365         KerberosInfo krbInfo = protocol.getAnnotation(KerberosInfo.class);
366         if (krbInfo != null) {
367           String serverKey = krbInfo.serverPrincipal();
368           if (serverKey == null) {
369             throw new IOException(
370                 "Can't obtain server Kerberos config key from KerberosInfo");
371           }
372           serverPrincipal = SecurityUtil.getServerPrincipal(
373               conf.get(serverKey), server.getAddress().getCanonicalHostName().toLowerCase());
374           if (LOG.isDebugEnabled()) {
375             LOG.debug("RPC Server Kerberos principal name for protocol="
376                 + protocol.getCanonicalName() + " is " + serverPrincipal);
377           }
378         }
379       }
380 
381       if (!useSasl) {
382         authMethod = AuthMethod.SIMPLE;
383       } else if (token != null) {
384         authMethod = AuthMethod.DIGEST;
385       } else {
386         authMethod = AuthMethod.KERBEROS;
387       }
388 
389       if (LOG.isDebugEnabled())
390         LOG.debug("Use " + authMethod + " authentication for protocol "
391             + protocol.getSimpleName());
392 
393       reloginMaxBackoff = conf.getInt("hbase.security.relogin.maxbackoff", 5000);
394       this.remoteId = remoteId;
395 
396       ConnectionHeader.Builder builder = ConnectionHeader.newBuilder();
397       builder.setProtocol(protocol == null ? "" : protocol.getName());
398       UserInformation userInfoPB;
399       if ((userInfoPB = getUserInfoPB(ticket)) != null) {
400         builder.setUserInfo(userInfoPB);
401       }
402       this.header = builder.build();
403 
404       this.setName("IPC Client (" + socketFactory.hashCode() +") connection to " +
405         remoteId.getAddress().toString() +
406         ((ticket==null)?" from an unknown user": (" from " 
407         + ticket.getUserName())));
408       this.setDaemon(true);
409     }
410 
411     private UserInformation getUserInfoPB(UserGroupInformation ugi) {
412       if (ugi == null || authMethod == AuthMethod.DIGEST) {
413         // Don't send user for token auth
414         return null;
415       }
416       UserInformation.Builder userInfoPB = UserInformation.newBuilder();
417       if (ugi != null) {
418         if (authMethod == AuthMethod.KERBEROS) {
419           // Send effective user for Kerberos auth
420           userInfoPB.setEffectiveUser(ugi.getUserName());
421         } else if (authMethod == AuthMethod.SIMPLE) {
422           //Send both effective user and real user for simple auth
423           userInfoPB.setEffectiveUser(ugi.getUserName());
424           if (ugi.getRealUser() != null) {
425             userInfoPB.setRealUser(ugi.getRealUser().getUserName());
426           }
427         }
428       }
429       return userInfoPB.build();
430     }

437     /**
438      * Add a call to this connection's call queue and notify
439      * a listener; synchronized. If the connection is dead, the call is not added, and the
440      * caller is notified.
441      * This function can return a connection that is already marked as 'shouldCloseConnection'
442      *  It is up to the user code to check this status.
443      * @param call to add
444      */
445     protected synchronized void addCall(Call call) {
446       // If the connection is about to close, we manage this as if the call was already added
447       //  to the connection calls list. If not, the connection creations are serialized, as
448       //  mentioned in HBASE-6364
449       if (this.shouldCloseConnection.get()) {
450         if (this.closeException == null) {
451           call.setException(new IOException(
452               "Call " + call.id + " not added as the connection " + remoteId + " is closing"));
453         } else {
454           call.setException(this.closeException);
455         }
456         synchronized (call) {
457           call.notifyAll();
458         }
459       } else {
460         calls.put(call.id, call);
461         notify();
462       }
463     }

475       /* Process timeout exception
476        * if the connection is not going to be closed, send a ping.
477        * otherwise, throw the timeout exception.
478        */
479       private void handleTimeout(SocketTimeoutException e) throws IOException {
480         if (shouldCloseConnection.get() || !running.get() ||
481             remoteId.rpcTimeout > 0) {
482           throw e;
483         }
484         sendPing();
485       }
486 
487       /** Read a byte from the stream.
488        * Send a ping if timeout on read. Retries if no failure is detected
489        * until a byte is read.
490        * @throws IOException for any IO problem other than socket timeout
491        */
492       @Override
493       public int read() throws IOException {
494         do {
495           try {
496             return super.read();
497           } catch (SocketTimeoutException e) {
498             handleTimeout(e);
499           }
500         } while (true);
501       }
502 
503       /** Read bytes into a buffer starting from offset <code>off</code>
504        * Send a ping if timeout on read. Retries if no failure is detected
505        * until a byte is read.
506        *
507        * @return the total number of bytes read; -1 if the connection is closed.
508        */
509       @Override
510       public int read(byte[] buf, int off, int len) throws IOException {
511         do {
512           try {
513             return super.read(buf, off, len);
514           } catch (SocketTimeoutException e) {
515             handleTimeout(e);
516           }
517         } while (true);
518       }
519     }

521     protected synchronized void setupConnection() throws IOException {
522       short ioFailures = 0;
523       short timeoutFailures = 0;
524       while (true) {
525         try {
526           this.socket = socketFactory.createSocket();
527           this.socket.setTcpNoDelay(tcpNoDelay);
528           this.socket.setKeepAlive(tcpKeepAlive);
529           // connection time out is 20s
530           NetUtils.connect(this.socket, remoteId.getAddress(),
531               getSocketTimeout(conf));
532           if (remoteId.rpcTimeout > 0) {
533             pingInterval = remoteId.rpcTimeout; // overwrite pingInterval
534           }
535           this.socket.setSoTimeout(pingInterval);
536           return;
537         } catch (SocketTimeoutException toe) {
538           /* The max number of retries is 45,
539            * which amounts to 20s*45 = 15 minutes retries.
540            */
541           handleConnectionFailure(timeoutFailures++, maxRetries, toe);
542         } catch (IOException ie) {
543           handleConnectionFailure(ioFailures++, maxRetries, ie);
544         }
545       }
546     }
547 
548     protected void closeConnection() {
549       // close the current connection
550       if (socket != null) {
551         try {
552           socket.close();
553         } catch (IOException e) {
554           LOG.warn("Not able to close a socket", e);
555         }
556       }
557       // set socket to null so that the next call to setupIOstreams
558       // can start the process of connect all over again.
559       socket = null;
560     }

562     /**
563      *  Handle connection failures
564      *
565      * If the current number of retries is equal to the max number of retries,
566      * stop retrying and throw the exception; Otherwise backoff N seconds and
567      * try connecting again.
568      *
569      * This Method is only called from inside setupIOstreams(), which is
570      * synchronized. Hence the sleep is synchronized; the locks will be retained.
571      *
572      * @param curRetries current number of retries
573      * @param maxRetries max number of retries allowed
574      * @param ioe failure reason
575      * @throws IOException if max number of retries is reached
576      */
577     private void handleConnectionFailure(
578         int curRetries, int maxRetries, IOException ioe) throws IOException {
579 
580       closeConnection();
581 
582       // throw the exception if the maximum number of retries is reached
583       if (curRetries >= maxRetries) {
584         throw ioe;
585       }
586 
587       // otherwise back off and retry
588       try {
589         Thread.sleep(failureSleep);
590       } catch (InterruptedException ignored) {}
591 
592       LOG.info("Retrying connect to server: " + remoteId.getAddress() +
593         " after sleeping " + failureSleep + "ms. Already tried " + curRetries +
594         " time(s).");
595     }
596 
597     /* wait till someone signals us to start reading RPC response or
598      * it is idle too long, it is marked as to be closed,
599      * or the client is marked as not running.
600      *
601      * Return true if it is time to read a response; false otherwise.
602      */
603     protected synchronized boolean waitForWork() {
604       if (calls.isEmpty() && !shouldCloseConnection.get()  && running.get())  {
605         long timeout = maxIdleTime-
606               (System.currentTimeMillis()-lastActivity.get());
607         if (timeout>0) {
608           try {
609             wait(timeout);
610           } catch (InterruptedException ignored) {}
611         }
612       }
613 
614       if (!calls.isEmpty() && !shouldCloseConnection.get() && running.get()) {
615         return true;
616       } else if (shouldCloseConnection.get()) {
617         return false;
618       } else if (calls.isEmpty()) { // idle connection closed or stopped
619         markClosed(null);
620         return false;
621       } else { // get stopped but there are still pending requests
622         markClosed((IOException)new IOException().initCause(
623             new InterruptedException()));
624         return false;
625       }
626     }

632     /* Send a ping to the server if the time elapsed
633      * since last I/O activity is equal to or greater than the ping interval
634      */
635     protected synchronized void sendPing() throws IOException {
636       long curTime = System.currentTimeMillis();
637       if ( curTime - lastActivity.get() >= pingInterval) {
638         lastActivity.set(curTime);
639         //noinspection SynchronizeOnNonFinalField
640         synchronized (this.out) {
641           out.writeInt(PING_CALL_ID);
642           out.flush();
643         }
644       }
645     }

647     @Override
648     public void run() {
649       if (LOG.isDebugEnabled())
650         LOG.debug(getName() + ": starting, having connections "
651             + connections.size());
652 
653       try {
654         while (waitForWork()) {//wait here for work - read or close connection
655           receiveResponse();
656         }
657       } catch (Throwable t) {
658         LOG.warn("Unexpected exception receiving call responses", t);
659         markClosed(new IOException("Unexpected exception receiving call responses", t));
660       }
661 
662       close();
663 
664       if (LOG.isDebugEnabled())
665         LOG.debug(getName() + ": stopped, remaining connections "
666             + connections.size());
667     }

680     private synchronized boolean shouldAuthenticateOverKrb() throws IOException {
681       UserGroupInformation loginUser = UserGroupInformation.getLoginUser();
682       UserGroupInformation currentUser =
683         UserGroupInformation.getCurrentUser();
684       UserGroupInformation realUser = currentUser.getRealUser();
685       return authMethod == AuthMethod.KERBEROS &&
686           loginUser != null &&
687           //Make sure user logged in using Kerberos either keytab or TGT
688           loginUser.hasKerberosCredentials() &&
689           // relogin only in case it is the login user (e.g. JT)
690           // or superuser (like oozie).
691           (loginUser.equals(currentUser) || loginUser.equals(realUser));
692     }

700     /**
701      * If multiple clients with the same principal try to connect
702      * to the same server at the same time, the server assumes a
703      * replay attack is in progress. This is a feature of kerberos.
704      * In order to work around this, what is done is that the client
705      * backs off randomly and tries to initiate the connection
706      * again.
707      * The other problem is to do with ticket expiry. To handle that,
708      * a relogin is attempted.
709      * <p>
710      * The retry logic is governed by the {@link #shouldAuthenticateOverKrb}
711      * method. In case when the user doesn't have valid credentials, we don't
712      * need to retry (from cache or ticket). In such cases, it is prudent to
713      * throw a runtime exception when we receive a SaslException from the
714      * underlying authentication implementation, so there is no retry from 
715      * other high level (for eg, HCM or HBaseAdmin).
716      * </p>
717      */
718     private synchronized void handleSaslConnectionFailure(
719         final int currRetries,
720         final int maxRetries, final Exception ex, final Random rand,
721         final UserGroupInformation user)
722     throws IOException, InterruptedException{
723       user.doAs(new PrivilegedExceptionAction<Object>() {
724         public Object run() throws IOException, InterruptedException {
725           closeConnection();
726           if (shouldAuthenticateOverKrb()) {
727             if (currRetries < maxRetries) {
728               LOG.debug("Exception encountered while connecting to " +
729                   "the server : " + ex);
730               //try re-login
731               if (UserGroupInformation.isLoginKeytabBased()) {
732                 UserGroupInformation.getLoginUser().reloginFromKeytab();
733               } else {
734                 UserGroupInformation.getLoginUser().reloginFromTicketCache();
735               }
736               disposeSasl();
737               //have granularity of milliseconds
738               //we are sleeping with the Connection lock held but since this
739               //connection instance is being used for connecting to the server
740               //in question, it is okay
741               Thread.sleep((rand.nextInt(reloginMaxBackoff) + 1));
742               return null;
743             } else {
744               String msg = "Couldn't setup connection for " +
745               UserGroupInformation.getLoginUser().getUserName() +
746               " to " + serverPrincipal;
747               LOG.warn(msg);
748               throw (IOException) new IOException(msg).initCause(ex);
749             }
750           } else {
751             LOG.warn("Exception encountered while connecting to " +
752                 "the server : " + ex);
753           }
754           if (ex instanceof RemoteException) {
755             throw (RemoteException)ex;
756           }
757           if (ex instanceof SaslException) {
758             String msg = "SASL authentication failed." +
759               " The most likely cause is missing or invalid credentials." +
760               " Consider 'kinit'.";
761             LOG.fatal(msg, ex);
762             throw new RuntimeException(msg, ex);
763           }
764           throw new IOException(ex);
765         }
766       });
767     }

769     protected synchronized void setupIOstreams()
770         throws IOException, InterruptedException {
771       if (socket != null || shouldCloseConnection.get()) {
772         return;
773       }
774 
775       if (failedServers.isFailedServer(remoteId.getAddress())) {
776         if (LOG.isDebugEnabled()) {
777           LOG.debug("Not trying to connect to " + server +
778               " this server is in the failed servers list");
779         }
780         IOException e = new FailedServerException(
781             "This server is in the failed servers list: " + server);
782         markClosed(e);
783         close();
784         throw e;
785       }
786 
787       try {
788         if (LOG.isDebugEnabled()) {
789           LOG.debug("Connecting to "+server);
790         }
791         short numRetries = 0;
792         final short MAX_RETRIES = 5;
793         Random rand = null;
794         while (true) {
795           setupConnection();
796           InputStream inStream = NetUtils.getInputStream(socket);
797           OutputStream outStream = NetUtils.getOutputStream(socket);
798           writeRpcHeader(outStream);
799           if (useSasl) {
800             final InputStream in2 = inStream;
801             final OutputStream out2 = outStream;
802             UserGroupInformation ticket = remoteId.getTicket().getUGI();
803             if (authMethod == AuthMethod.KERBEROS) {
804               if (ticket != null && ticket.getRealUser() != null) {
805                 ticket = ticket.getRealUser();
806               }
807             }
808             boolean continueSasl = false;
809             try {
810               continueSasl =
811                 ticket.doAs(new PrivilegedExceptionAction<Boolean>() {
812                   @Override
813                   public Boolean run() throws IOException {
814                     return setupSaslConnection(in2, out2);
815                   }
816                 });
817             } catch (Exception ex) {
818               if (rand == null) {
819                 rand = new Random();
820               }
821               handleSaslConnectionFailure(numRetries++, MAX_RETRIES, ex, rand,
822                    ticket);
823               continue;
824             }
825             if (continueSasl) {
826               // Sasl connect is successful. Let's set up Sasl i/o streams.
827               inStream = saslRpcClient.getInputStream(inStream);
828               outStream = saslRpcClient.getOutputStream(outStream);
829             } else {
830               // fall back to simple auth because server told us so.
831               authMethod = AuthMethod.SIMPLE;
832               useSasl = false;
833             }
834           }
835           this.in = new DataInputStream(new BufferedInputStream
836               (new PingInputStream(inStream)));
837           this.out = new DataOutputStream
838           (new BufferedOutputStream(outStream));
839           writeHeader();
840 
841           // update last activity time
842           touch();
843 
844           // start the receiver thread after the socket connection has been set up
845           start();
846           return;
847         }
848       } catch (IOException e) {
849         failedServers.addToFailedServers(remoteId.address);
850         markClosed(e);
851         close();
852 
853         throw e;
854       }
855     }

877     /** Close the connection. */
878     protected synchronized void close() {
879       if (!shouldCloseConnection.get()) {
880         LOG.error("The connection is not in the closed state");
881         return;
882       }
883 
884       // release the resources
885       // first thing to do;take the connection out of the connection list
886       synchronized (connections) {
887         if (connections.get(remoteId) == this) {
888           connections.remove(remoteId);
889         }
890       }
891 
892       // close the streams and therefore the socket
893       IOUtils.closeStream(out);
894       IOUtils.closeStream(in);
895       disposeSasl();
896 
897       // clean up all calls
898       if (closeException == null) {
899         if (!calls.isEmpty()) {
900           LOG.warn(
901               "A connection is closed for no cause and calls are not empty. " +
902               "#Calls: " + calls.size());
903 
904           // clean up calls anyway
905           closeException = new IOException("Unexpected closed connection");
906           cleanupCalls();
907         }
908       } else {
909         // log the info
910         if (LOG.isDebugEnabled()) {
911           LOG.debug("closing ipc connection to " + server + ": " +
912               closeException.getMessage(),closeException);
913         }
914 
915         // cleanup calls
916         cleanupCalls();
917       }
918       if (LOG.isDebugEnabled())
919         LOG.debug(getName() + ": closed");
920     }

922     /* Initiates a call by sending the parameter to the remote server.
923      * Note: this is not called from the Connection thread, but by other
924      * threads.
925      */
926     protected void sendParam(Call call) {
927       if (shouldCloseConnection.get()) {
928         return;
929       }
930       try {
931         if (LOG.isDebugEnabled())
932           LOG.debug(getName() + " sending #" + call.id);
933 
934         RpcRequestHeader.Builder headerBuilder = RPCProtos.RpcRequestHeader.newBuilder();
935         headerBuilder.setCallId(call.id);
936 
937         if (Trace.isTracing()) {
938           Span s = Trace.currentTrace();
939           headerBuilder.setTinfo(RPCTInfo.newBuilder()
940               .setParentId(s.getSpanId())
941               .setTraceId(s.getTraceId()));
942         }
943 
944         //noinspection SynchronizeOnNonFinalField
945         synchronized (this.out) { // FindBugs IS2_INCONSISTENT_SYNC
946           RpcRequestHeader header = headerBuilder.build();
947           int serializedHeaderSize = header.getSerializedSize();
948           int requestSerializedSize = call.param.getSerializedSize();
949           this.out.writeInt(serializedHeaderSize +
950               CodedOutputStream.computeRawVarint32Size(serializedHeaderSize) +
951               requestSerializedSize +
952               CodedOutputStream.computeRawVarint32Size(requestSerializedSize));
953           header.writeDelimitedTo(this.out);
954           call.param.writeDelimitedTo(this.out);
955           this.out.flush();
956         }
957       } catch(IOException e) {
958         markClosed(e);
959       }
960     }

962     /* Receive a response.
963      * Because only one receiver, so no synchronization on in.
964      */
965     protected void receiveResponse() {
966       if (shouldCloseConnection.get()) {
967         return;
968       }
969       touch();
970 
971       try {
972         // See HBaseServer.Call.setResponse for where we write out the response.
973         // It writes the call.id (int), a boolean signifying any error (and if 
974         // so the exception name/trace), and the response bytes
975 
976         // Read the call id.
977         RpcResponseHeader response = RpcResponseHeader.parseDelimitedFrom(in);
978         if (response == null) {
979           // When the stream is closed, protobuf doesn't raise an EOFException,
980           // instead, it returns a null message object. 
981           throw new EOFException();
982         }
983         int id = response.getCallId();
984 
985         if (LOG.isDebugEnabled())
986           LOG.debug(getName() + " got value #" + id);
987         Call call = calls.get(id);
988 
989         Status status = response.getStatus();
990         if (status == Status.SUCCESS) {
991           Message rpcResponseType;
992           try {
993             rpcResponseType = ProtobufRpcEngine.Invoker.getReturnProtoType(
994                 ProtobufRpcEngine.Server.getMethod(remoteId.getProtocol(),
995                     call.param.getMethodName()));
996           } catch (Exception e) {
997             throw new RuntimeException(e); //local exception
998           }
999           Builder builder = rpcResponseType.newBuilderForType();
1000           builder.mergeDelimitedFrom(in);
1001           Message value = builder.build();
1002           // it's possible that this call may have been cleaned up due to a RPC
1003           // timeout, so check if it still exists before setting the value.
1004           if (call != null) {
1005             call.setValue(value);
1006           }
1007           calls.remove(id);
1008         } else if (status == Status.ERROR) {
1009           RpcException exceptionResponse = RpcException.parseDelimitedFrom(in);
1010           if (call != null) {
1011             //noinspection ThrowableInstanceNeverThrown
1012             call.setException(new RemoteException(
1013                 exceptionResponse.getExceptionName(),
1014                 exceptionResponse.getStackTrace()));
1015             calls.remove(id);
1016           }
1017         } else if (status == Status.FATAL) {
1018           RpcException exceptionResponse = RpcException.parseDelimitedFrom(in);
1019           // Close the connection
1020           markClosed(new RemoteException(
1021               exceptionResponse.getExceptionName(),
1022               exceptionResponse.getStackTrace()));
1023         }
1024       } catch (IOException e) {
1025         if (e instanceof SocketTimeoutException && remoteId.rpcTimeout > 0) {
1026           // Clean up open calls but don't treat this as a fatal condition,
1027           // since we expect certain responses to not make it by the specified
1028           // {@link ConnectionId#rpcTimeout}.
1029           closeException = e;
1030         } else {
1031           // Since the server did not respond within the default ping interval
1032           // time, treat this as a fatal condition and close this connection
1033           markClosed(e);
1034         }
1035       } finally {
1036         if (remoteId.rpcTimeout > 0) {
1037           cleanupCalls(remoteId.rpcTimeout);
1038         }
1039       }
1040     }
1041 
1042     protected synchronized void markClosed(IOException e) {
1043       if (shouldCloseConnection.compareAndSet(false, true)) {
1044         closeException = e;
1045         notifyAll();
1046       }
1047     }

1049     /* Cleanup all calls and mark them as done */
1050     protected void cleanupCalls() {
1051       cleanupCalls(0);
1052     }

1054     protected void cleanupCalls(long rpcTimeout) {
1055       Iterator<Entry<Integer, Call>> itor = calls.entrySet().iterator();
1056       while (itor.hasNext()) {
1057         Call c = itor.next().getValue();
1058         long waitTime = System.currentTimeMillis() - c.getStartTime();
1059         if (waitTime >= rpcTimeout) {
1060           if (this.closeException == null) {
1061             // There may be no exception in the case that there are many calls
1062             // being multiplexed over this connection and these are succeeding
1063             // fine while this Call object is taking a long time to finish
1064             // over on the server; e.g. I just asked the regionserver to bulk
1065             // open 3k regions or its a big fat multiput into a heavily-loaded
1066             // server (Perhaps this only happens at the extremes?)
1067             this.closeException = new CallTimeoutException("Call id=" + c.id +
1068               ", waitTime=" + waitTime + ", rpcTimetout=" + rpcTimeout);
1069           }
1070           c.setException(this.closeException);
1071           synchronized (c) {
1072             c.notifyAll();
1073           }
1074           itor.remove();
1075         } else {
1076           break;
1077         }
1078       }
1079       try {
1080         if (!calls.isEmpty()) {
1081           Call firstCall = calls.get(calls.firstKey());
1082           long maxWaitTime = System.currentTimeMillis() - firstCall.getStartTime();
1083           if (maxWaitTime < rpcTimeout) {
1084             rpcTimeout -= maxWaitTime;
1085           }
1086         }
1087         if (!shouldCloseConnection.get()) {
1088           closeException = null;
1089           if (socket != null) {
1090             socket.setSoTimeout((int) rpcTimeout);
1091           }
1092         }
1093       } catch (SocketException e) {
1094         LOG.debug("Couldn't lower timeout, which may result in longer than expected calls");
1095       }
1096     }
1097   }

1367     // TODO this synchronization block doesnt make any sense, we should possibly fix it

1369     synchronized (results) {

1394    * pool.  Connections to a given host/port are reused. */
1395   protected Connection getConnection(InetSocketAddress addr,
1396                                    Class<? extends VersionedProtocol> protocol,
1397                                    User ticket,
1398                                    int rpcTimeout,
1399                                    Call call)
1400                                    throws IOException, InterruptedException {
1401     if (!running.get()) {
1402       // the client is stopped
1403       throw new IOException("The client is stopped");
1404     }
1405     Connection connection;
1406     /* we could avoid this allocation for each RPC by having a
1407      * connectionsId object and with set() method. We need to manage the
1408      * refs for keys in HashMap properly. For now its ok.
1409      */
1410     ConnectionId remoteId = new ConnectionId(addr, protocol, ticket, rpcTimeout);
1411     synchronized (connections) {
1412       connection = connections.get(remoteId);
1413       if (connection == null) {
1414         connection = createConnection(remoteId);
1415         connections.put(remoteId, connection);
1416       }
1417     }
1418     connection.addCall(call);
1419 
1420     //we don't invoke the method below inside "synchronized (connections)"
1421     //block above. The reason for that is if the server happens to be slow,

1425     // waiting here; as setupIOstreams is synchronized. If the connection fails with a
1426     // timeout, they will all fail simultaneously. This is checked in setupIOstreams.
