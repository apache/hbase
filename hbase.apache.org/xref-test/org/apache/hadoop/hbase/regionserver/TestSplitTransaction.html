106   @Test public void testFailAfterPONR() throws IOException, KeeperException {
107     final int rowcount = TEST_UTIL.loadRegion(this.parent, CF);
108     assertTrue(rowcount > 0);
109     int parentRowCount = countRows(this.parent);
110     assertEquals(rowcount, parentRowCount);
111 
112     // Start transaction.
113     SplitTransaction st = prepareGOOD_SPLIT_ROW();
114     SplitTransaction spiedUponSt = spy(st);
115     Mockito
116         .doThrow(new MockedFailedDaughterOpen())
117         .when(spiedUponSt)
118         .openDaughterRegion((Server) Mockito.anyObject(),
119             (HRegion) Mockito.anyObject());
120 
121     // Run the execute.  Look at what it returns.
122     boolean expectedException = false;
123     Server mockServer = Mockito.mock(Server.class);
124     when(mockServer.getConfiguration()).thenReturn(TEST_UTIL.getConfiguration());
125     try {
126       spiedUponSt.execute(mockServer, null);
127     } catch (IOException e) {
128       if (e.getCause() != null &&
129           e.getCause() instanceof MockedFailedDaughterOpen) {
130         expectedException = true;
131       }
132     }
133     assertTrue(expectedException);
134     // Run rollback returns that we should restart.
135     assertFalse(spiedUponSt.rollback(null, null));
136     // Make sure that region a and region b are still in the filesystem, that
137     // they have not been removed; this is supposed to be the case if we go
138     // past point of no return.
139     Path tableDir =  this.parent.getRegionDir().getParent();
140     Path daughterADir =
141       new Path(tableDir, spiedUponSt.getFirstDaughter().getEncodedName());
142     Path daughterBDir =

220   @Test public void testWholesomeSplit() throws IOException {
221     final int rowcount = TEST_UTIL.loadRegion(this.parent, CF, true);
222     assertTrue(rowcount > 0);
223     int parentRowCount = countRows(this.parent);
224     assertEquals(rowcount, parentRowCount);
225 
226     // Pretend region's blocks are not in the cache, used for
227     // testWholesomeSplitWithHFileV1
228     CacheConfig cacheConf = new CacheConfig(TEST_UTIL.getConfiguration());
229     ((LruBlockCache) cacheConf.getBlockCache()).clearCache();
230 
231     // Start transaction.
232     SplitTransaction st = prepareGOOD_SPLIT_ROW();
233 
234     // Run the execute.  Look at what it returns.
235     Server mockServer = Mockito.mock(Server.class);
236     when(mockServer.getConfiguration()).thenReturn(TEST_UTIL.getConfiguration());
237     PairOfSameType<HRegion> daughters = st.execute(mockServer, null);
238     // Do some assertions about execution.
239     assertTrue(this.fs.exists(st.getSplitDir()));
240     // Assert the parent region is closed.
241     assertTrue(this.parent.isClosed());
242 
243     // Assert splitdir is empty -- because its content will have been moved out
244     // to be under the daughter region dirs.
245     assertEquals(0, this.fs.listStatus(st.getSplitDir()).length);
246     // Check daughters have correct key span.
247     assertTrue(Bytes.equals(this.parent.getStartKey(), daughters.getFirst().getStartKey()));
248     assertTrue(Bytes.equals(GOOD_SPLIT_ROW, daughters.getFirst().getEndKey()));
249     assertTrue(Bytes.equals(daughters.getSecond().getStartKey(), GOOD_SPLIT_ROW));
250     assertTrue(Bytes.equals(this.parent.getEndKey(), daughters.getSecond().getEndKey()));
251     // Count rows. daughters are already open
252     int daughtersRowCount = 0;
253     for (HRegion openRegion: daughters) {
254       try {
255         int count = countRows(openRegion);
256         assertTrue(count > 0 && count != rowcount);
257         daughtersRowCount += count;
258       } finally {
259         HRegion.closeHRegion(openRegion);
260       }
261     }
262     assertEquals(rowcount, daughtersRowCount);
263     // Assert the write lock is no longer held on parent
264     assertTrue(!this.parent.lock.writeLock().isHeldByCurrentThread());
265   }

267   @Test public void testRollback() throws IOException {
268     final int rowcount = TEST_UTIL.loadRegion(this.parent, CF);
269     assertTrue(rowcount > 0);
270     int parentRowCount = countRows(this.parent);
271     assertEquals(rowcount, parentRowCount);
272 
273     // Start transaction.
274     SplitTransaction st = prepareGOOD_SPLIT_ROW();
275     SplitTransaction spiedUponSt = spy(st);
276     when(spiedUponSt.createDaughterRegion(spiedUponSt.getSecondDaughter(), null)).
277       thenThrow(new MockedFailedDaughterCreation());
278     // Run the execute.  Look at what it returns.
279     boolean expectedException = false;
280     Server mockServer = Mockito.mock(Server.class);
281     when(mockServer.getConfiguration()).thenReturn(TEST_UTIL.getConfiguration());
282     try {
283       spiedUponSt.execute(mockServer, null);
284     } catch (MockedFailedDaughterCreation e) {
285       expectedException = true;
286     }
287     assertTrue(expectedException);
288     // Run rollback
289     assertTrue(spiedUponSt.rollback(null, null));
290 
291     // Assert I can scan parent.
292     int parentRowCount2 = countRows(this.parent);
293     assertEquals(parentRowCount, parentRowCount2);
294 
295     // Assert rollback cleaned up stuff in fs
296     assertTrue(!this.fs.exists(HRegion.getRegionDir(this.testdir, st.getFirstDaughter())));
297     assertTrue(!this.fs.exists(HRegion.getRegionDir(this.testdir, st.getSecondDaughter())));
298     assertTrue(!this.parent.lock.writeLock().isHeldByCurrentThread());
299 
300     // Now retry the split but do not throw an exception this time.
301     assertTrue(st.prepare());
302     PairOfSameType<HRegion> daughters = st.execute(mockServer, null);
303     // Count rows. daughters are already open
304     int daughtersRowCount = 0;
305     for (HRegion openRegion: daughters) {
306       try {
307         int count = countRows(openRegion);
308         assertTrue(count > 0 && count != rowcount);
309         daughtersRowCount += count;
310       } finally {
311         HRegion.closeHRegion(openRegion);
312       }
313     }
314     assertEquals(rowcount, daughtersRowCount);
315     // Assert the write lock is no longer held on parent
316     assertTrue(!this.parent.lock.writeLock().isHeldByCurrentThread());
317     assertTrue("Rollback hooks should be called.", wasRollBackHookCalled());
318   }

320   private boolean wasRollBackHookCalled(){
321     return (preRollBackCalled && postRollBackCalled);
322   }

331   private int countRows(final HRegion r) throws IOException {
332     int rowcount = 0;
333     InternalScanner scanner = r.getScanner(new Scan());
334     try {
335       List<KeyValue> kvs = new ArrayList<KeyValue>();
336       boolean hasNext = true;
337       while (hasNext) {
338         hasNext = scanner.next(kvs);
339         if (!kvs.isEmpty()) rowcount++;
340       }
341     } finally {
342       scanner.close();
343     }
344     return rowcount;
345   }

347   HRegion createRegion(final Path testdir, final HLog wal)
348   throws IOException {
349     // Make a region with start and end keys. Use 'aaa', to 'AAA'.  The load
350     // region utility will add rows between 'aaa' and 'zzz'.
351     HTableDescriptor htd = new HTableDescriptor("table");
352     HColumnDescriptor hcd = new HColumnDescriptor(CF);
353     htd.addFamily(hcd);
354     HRegionInfo hri = new HRegionInfo(htd.getName(), STARTROW, ENDROW);
355     HRegion r = HRegion.createHRegion(hri, testdir, TEST_UTIL.getConfiguration(), htd);
356     HRegion.closeHRegion(r);
357     return HRegion.openHRegion(testdir, hri, htd, wal,
358       TEST_UTIL.getConfiguration());
359   }
