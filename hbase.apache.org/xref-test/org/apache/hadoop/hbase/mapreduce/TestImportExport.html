86    /**
87     * When running on Hadoop 2, we need to copy (or add) configuration values for keys
88     * that start with "yarn." (from the map reduce minicluster) to the
89     * configuration that will be used during the test (from the HBase minicluster). 
90     * YARN configuration values are set properly in the map reduce minicluster, 
91     * but not necessarily in the HBase mini cluster.
92     * @param srcConf the configuration to copy from (the map reduce minicluster version)
93     * @param destConf the configuration to copy to (the HBase minicluster version)
94     */
95    private void copyConfigurationValues(Configuration srcConf, Configuration destConf) {
96      Iterator<Map.Entry<String,String>> it = srcConf.iterator();
97      while (it.hasNext()) {
98        Map.Entry<String, String> entry = it.next();
99        String key = entry.getKey();
100       String value = entry.getValue();
101       if (key.startsWith("yarn.") && !value.isEmpty()) {
102         destConf.set(key, value);
103       }
104     }
105   }

125

128         OUTPUT_DIR,

131     };

134     Configuration conf = opts.getConfiguration();

199   @Test
200   public void testWithDeletes() throws Exception {
201     String EXPORT_TABLE = "exportWithDeletes";
202     HTableDescriptor desc = new HTableDescriptor(EXPORT_TABLE);
203     desc.addFamily(new HColumnDescriptor(FAMILYA)
204         .setMaxVersions(5)
205         .setKeepDeletedCells(true)
206     );
207     UTIL.getHBaseAdmin().createTable(desc);
208     HTable t = new HTable(UTIL.getConfiguration(), EXPORT_TABLE);
209 
210     Put p = new Put(ROW1);
211     p.add(FAMILYA, QUAL, now, QUAL);
212     p.add(FAMILYA, QUAL, now+1, QUAL);
213     p.add(FAMILYA, QUAL, now+2, QUAL);
214     p.add(FAMILYA, QUAL, now+3, QUAL);
215     p.add(FAMILYA, QUAL, now+4, QUAL);
216     t.put(p);
217 
218     Delete d = new Delete(ROW1, now+3, null);
219     t.delete(d);
220     d = new Delete(ROW1);
221     d.deleteColumns(FAMILYA, QUAL, now+2);
222     t.delete(d);
223     
224     String[] args = new String[] {
225         "-D" + Export.RAW_SCAN + "=true",
226         EXPORT_TABLE,
227         OUTPUT_DIR,
228 	EXPORT_BATCHING,
229         "1000"
230     };
231 
232     GenericOptionsParser opts = new GenericOptionsParser(new Configuration(cluster.getConfiguration()), args);
233     Configuration conf = opts.getConfiguration();
234 
235     // copy or add the necessary configuration values from the map reduce config to the hbase config
236     copyConfigurationValues(UTIL.getConfiguration(), conf);
237     args = opts.getRemainingArgs();
238 
239     Job job = Export.createSubmittableJob(conf, args);
240     job.getConfiguration().set("mapreduce.framework.name", "yarn");
241     job.waitForCompletion(false);
242     assertTrue(job.isSuccessful());
243 
244 
245     String IMPORT_TABLE = "importWithDeletes";
246     desc = new HTableDescriptor(IMPORT_TABLE);
247     desc.addFamily(new HColumnDescriptor(FAMILYA)
248         .setMaxVersions(5)
249         .setKeepDeletedCells(true)
250     );
251     UTIL.getHBaseAdmin().createTable(desc);
252     t.close();
253     t = new HTable(UTIL.getConfiguration(), IMPORT_TABLE);
254     args = new String[] {
255         IMPORT_TABLE,
256         OUTPUT_DIR
257     };
258 
259     opts = new GenericOptionsParser(new Configuration(cluster.getConfiguration()), args);
260     conf = opts.getConfiguration();
261 
262     // copy or add the necessary configuration values from the map reduce config to the hbase config
263     copyConfigurationValues(UTIL.getConfiguration(), conf);
264     args = opts.getRemainingArgs();
265 
266     job = Import.createSubmittableJob(conf, args);
267     job.getConfiguration().set("mapreduce.framework.name", "yarn");
268     job.waitForCompletion(false);
269     assertTrue(job.isSuccessful());
270 
271     Scan s = new Scan();
272     s.setMaxVersions();
273     s.setRaw(true);
274     ResultScanner scanner = t.getScanner(s);
275     Result r = scanner.next();
276     KeyValue[] res = r.raw();
277     assertTrue(res[0].isDeleteFamily());
278     assertEquals(now+4, res[1].getTimestamp());
279     assertEquals(now+3, res[2].getTimestamp());
280     assertTrue(res[3].isDelete());
281     assertEquals(now+2, res[4].getTimestamp());
282     assertEquals(now+1, res[5].getTimestamp());
283     assertEquals(now, res[6].getTimestamp());
