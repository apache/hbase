42  /**
43   * The {@link HFile} has a fixed trailer which contains offsets to other
44   * variable parts of the file. Also includes basic metadata on this file. The
45   * trailer size is fixed within a given {@link HFile} format version only, but
46   * we always store the version number as the last four-byte integer of the file.
47   * The version number itself is split into two portions, a major 
48   * version and a minor version. 
49   * The last three bytes of a file is the major
50   * version and a single preceding byte is the minor number. The major version
51   * determines which readers/writers to use to read/write a hfile while a minor
52   * version determines smaller changes in hfile format that do not need a new
53   * reader/writer type.
54   */
55  @InterfaceAudience.Private
56  public class FixedFileTrailer {
57

130   private static int[] computeTrailerSizeByVersion() {
131     int versionToSize[] = new int[HFile.MAX_FORMAT_VERSION + 1];
132     for (int version = MIN_FORMAT_VERSION;
133          version <= MAX_FORMAT_VERSION;
134          ++version) {
135       FixedFileTrailer fft = new FixedFileTrailer(version, 
136                                    HFileBlock.MINOR_VERSION_NO_CHECKSUM);
137       DataOutputStream dos = new DataOutputStream(new NullOutputStream());
138       try {
139         fft.serialize(dos);
140       } catch (IOException ex) {
141         // The above has no reason to fail.
142         throw new RuntimeException(ex);
143       }
144       versionToSize[version] = dos.size();
145     }
146     return versionToSize;
147   }
148 
149   private static int getMaxTrailerSize() {
150     int maxSize = 0;
151     for (int version = MIN_FORMAT_VERSION;
152          version <= MAX_FORMAT_VERSION;
153          ++version)
154       maxSize = Math.max(getTrailerSize(version), maxSize);
155     return maxSize;
156   }

169   /**
170    * Write the trailer to a data stream. We support writing version 1 for
171    * testing and for determining version 1 trailer size. It is also easy to see
172    * what fields changed in version 2.
173    *
174    * @param outputStream
175    * @throws IOException
176    */
177   void serialize(DataOutputStream outputStream) throws IOException {
178     HFile.checkFormatVersion(majorVersion);
179 
180     ByteArrayOutputStream baos = new ByteArrayOutputStream();
181     DataOutput baosDos = new DataOutputStream(baos);
182 
183     BlockType.TRAILER.write(baosDos);
184     baosDos.writeLong(fileInfoOffset);
185     baosDos.writeLong(loadOnOpenDataOffset);
186     baosDos.writeInt(dataIndexCount);
187 
188     if (majorVersion == 1) {
189       // This used to be metaIndexOffset, but it was not used in version 1.
190       baosDos.writeLong(0);
191     } else {
192       baosDos.writeLong(uncompressedDataIndexSize);
193     }
194 
195     baosDos.writeInt(metaIndexCount);
196     baosDos.writeLong(totalUncompressedBytes);
197     if (majorVersion == 1) {
198       baosDos.writeInt((int) Math.min(Integer.MAX_VALUE, entryCount));
199     } else {
200       // This field is long from version 2 onwards.
201       baosDos.writeLong(entryCount);
202     }
203     baosDos.writeInt(compressionCodec.ordinal());
204 
205     if (majorVersion > 1) {
206       baosDos.writeInt(numDataIndexLevels);
207       baosDos.writeLong(firstDataBlockOffset);
208       baosDos.writeLong(lastDataBlockOffset);
209       Bytes.writeStringFixedSize(baosDos, comparatorClassName,
210           MAX_COMPARATOR_NAME_LENGTH);
211     }
212 
213     // serialize the major and minor versions
214     baosDos.writeInt(materializeVersion(majorVersion, minorVersion));
215 
216     outputStream.write(baos.toByteArray());
217   }

219   /**
220    * Deserialize the fixed file trailer from the given stream. The version needs
221    * to already be specified. Make sure this is consistent with
222    * {@link #serialize(DataOutputStream)}.
223    *
224    * @param inputStream
225    * @param version
226    * @throws IOException
227    */
228   void deserialize(DataInputStream inputStream) throws IOException {
229     HFile.checkFormatVersion(majorVersion);
230 
231     BlockType.TRAILER.readAndCheck(inputStream);
232 
233     fileInfoOffset = inputStream.readLong();
234     loadOnOpenDataOffset = inputStream.readLong();
235     dataIndexCount = inputStream.readInt();
236 
237     if (majorVersion == 1) {
238       inputStream.readLong(); // Read and skip metaIndexOffset.
239     } else {
240       uncompressedDataIndexSize = inputStream.readLong();
241     }
242     metaIndexCount = inputStream.readInt();
243 
244     totalUncompressedBytes = inputStream.readLong();
245     entryCount = majorVersion == 1 ? inputStream.readInt() : inputStream.readLong();
246     compressionCodec = Compression.Algorithm.values()[inputStream.readInt()];
247     if (majorVersion > 1) {
248       numDataIndexLevels = inputStream.readInt();
249       firstDataBlockOffset = inputStream.readLong();
250       lastDataBlockOffset = inputStream.readLong();
251       comparatorClassName =
252           Bytes.readStringFixedSize(inputStream, MAX_COMPARATOR_NAME_LENGTH);
253     }
254 
255     int version = inputStream.readInt();
256     expectMajorVersion(extractMajorVersion(version));
257     expectMinorVersion(extractMinorVersion(version));
258   }

289   /**
290    * Reads a file trailer from the given file.
291    *
292    * @param istream the input stream with the ability to seek. Does not have to
293    *          be buffered, as only one read operation is made.
294    * @param fileSize the file size. Can be obtained using
295    *          {@link org.apache.hadoop.fs.FileSystem#getFileStatus(
296    *          org.apache.hadoop.fs.Path)}.
297    * @return the fixed file trailer read
298    * @throws IOException if failed to read from the underlying stream, or the
299    *           trailer is corrupted, or the version of the trailer is
300    *           unsupported
301    */
302   public static FixedFileTrailer readFromStream(FSDataInputStream istream,
303       long fileSize) throws IOException {
304     int bufferSize = MAX_TRAILER_SIZE;
305     long seekPoint = fileSize - bufferSize;
306     if (seekPoint < 0) {
307       // It is hard to imagine such a small HFile.
308       seekPoint = 0;
309       bufferSize = (int) fileSize;
310     }
311 
312     istream.seek(seekPoint);
313     ByteBuffer buf = ByteBuffer.allocate(bufferSize);
314     istream.readFully(buf.array(), buf.arrayOffset(),
315         buf.arrayOffset() + buf.limit());
316 
317     // Read the version from the last int of the file.
318     buf.position(buf.limit() - Bytes.SIZEOF_INT);
319     int version = buf.getInt();
320 
321     // Extract the major and minor versions.
322     int majorVersion = extractMajorVersion(version);
323     int minorVersion = extractMinorVersion(version);
324 
325     HFile.checkFormatVersion(majorVersion); // throws IAE if invalid
326 
327     int trailerSize = getTrailerSize(majorVersion);
328 
329     FixedFileTrailer fft = new FixedFileTrailer(majorVersion, minorVersion);
330     fft.deserialize(new DataInputStream(new ByteArrayInputStream(buf.array(),
331         buf.arrayOffset() + bufferSize - trailerSize, trailerSize)));
332     return fft;
333   }

512   /**
513    * Extracts the major version for a 4-byte serialized version data.
514    * The major version is the 3 least significant bytes
515    */
516   private static int extractMajorVersion(int serializedVersion) {
517     return (serializedVersion & 0x00ffffff);
518   }
519 
520   /**
521    * Extracts the minor version for a 4-byte serialized version data.
522    * The major version are the 3 the most significant bytes
523    */
524   private static int extractMinorVersion(int serializedVersion) {
525     return (serializedVersion >>> 24);
526   }
527 
528   /**
529    * Create a 4 byte serialized version number by combining the
530    * minor and major version numbers.
531    */
532   private static int materializeVersion(int majorVersion, int minorVersion) {
533     return ((majorVersion & 0x00ffffff) | (minorVersion << 24));
534   }
