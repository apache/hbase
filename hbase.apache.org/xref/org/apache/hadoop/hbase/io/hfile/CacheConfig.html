112   /**
113    * Create a cache configuration using the specified configuration object and
114    * family descriptor.
115    * @param conf hbase configuration
116    * @param family column family configuration
117    */
118   public CacheConfig(Configuration conf, HColumnDescriptor family) {
119     this(CacheConfig.instantiateBlockCache(conf),
120         family.isBlockCacheEnabled(),
121         family.isInMemory(),
122         // For the following flags we enable them regardless of per-schema settings
123         // if they are enabled in the global configuration.
124         conf.getBoolean(CACHE_BLOCKS_ON_WRITE_KEY,
125             DEFAULT_CACHE_DATA_ON_WRITE) || family.shouldCacheDataOnWrite(),
126         conf.getBoolean(CACHE_INDEX_BLOCKS_ON_WRITE_KEY,
127             DEFAULT_CACHE_INDEXES_ON_WRITE) || family.shouldCacheIndexesOnWrite(),
128         conf.getBoolean(CACHE_BLOOM_BLOCKS_ON_WRITE_KEY,
129             DEFAULT_CACHE_BLOOMS_ON_WRITE) || family.shouldCacheBloomsOnWrite(),
130         conf.getBoolean(EVICT_BLOCKS_ON_CLOSE_KEY,
131             DEFAULT_EVICT_ON_CLOSE) || family.shouldEvictBlocksOnClose(),
132         conf.getBoolean(CACHE_DATA_BLOCKS_COMPRESSED_KEY, DEFAULT_COMPRESSED_CACHE)
133      );
134   }

311   /**
312    * Static reference to the block cache, or null if no caching should be used
313    * at all.
314    */
315   private static BlockCache globalBlockCache;
316 
317   /** Boolean whether we have disabled the block cache entirely. */
318   private static boolean blockCacheDisabled = false;
319 
320   /**
321    * Returns the block cache or <code>null</code> in case none should be used.
322    *
323    * @param conf  The current configuration.
324    * @return The block cache or <code>null</code>.
325    */
326   private static synchronized BlockCache instantiateBlockCache(
327       Configuration conf) {
328     if (globalBlockCache != null) return globalBlockCache;
329     if (blockCacheDisabled) return null;
330 
331     float cachePercentage = conf.getFloat(HConstants.HFILE_BLOCK_CACHE_SIZE_KEY,
332       HConstants.HFILE_BLOCK_CACHE_SIZE_DEFAULT);
333     if (cachePercentage == 0L) {
334       blockCacheDisabled = true;
335       return null;
336     }
337     if (cachePercentage > 1.0) {
338       throw new IllegalArgumentException(HConstants.HFILE_BLOCK_CACHE_SIZE_KEY +
339         " must be between 0.0 and 1.0, and not > 1.0");
340     }
341 
342     // Calculate the amount of heap to give the heap.
343     MemoryUsage mu = ManagementFactory.getMemoryMXBean().getHeapMemoryUsage();
344     long cacheSize = (long)(mu.getMax() * cachePercentage);
345     int blockSize = conf.getInt("hbase.offheapcache.minblocksize",
346         HFile.DEFAULT_BLOCKSIZE);
347     long offHeapCacheSize =
348       (long) (conf.getFloat("hbase.offheapcache.percentage", (float) 0) *
349           DirectMemoryUtils.getDirectMemorySize());
350     LOG.info("Allocating LruBlockCache with maximum size " +
351       StringUtils.humanReadableInt(cacheSize));
352     if (offHeapCacheSize <= 0) {
353       globalBlockCache = new LruBlockCache(cacheSize,
354           StoreFile.DEFAULT_BLOCKSIZE_SMALL, conf);
355     } else {
356       globalBlockCache = new DoubleBlockCache(cacheSize, offHeapCacheSize,
357           StoreFile.DEFAULT_BLOCKSIZE_SMALL, blockSize, conf);
358     }
359     return globalBlockCache;
360   }
