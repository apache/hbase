46  /**
47   * HLogPrettyPrinter prints the contents of a given HLog with a variety of
48   * options affecting formatting and extent of content.
49   * 
50   * It targets two usage cases: pretty printing for ease of debugging directly by
51   * humans, and JSON output for consumption by monitoring and/or maintenance
52   * scripts.
53   * 
54   * It can filter by row, region, or sequence id.
55   * 
56   * It can also toggle output of values.
57   * 
58   */
59  @InterfaceAudience.Public
60  @InterfaceStability.Evolving
61  public class HLogPrettyPrinter {
62    private boolean outputValues;
63    private boolean outputJSON;
64    // The following enable filtering by sequence, region, and row, respectively
65    private long sequence;
66    private String region;
67    private String row;
68    // enable in order to output a single list of transactions from several files
69    private boolean persistentOutput;
70    private boolean firstTxn;
71    // useful for programatic capture of JSON output
72    private PrintStream out;
73    // for JSON encoding
74    private ObjectMapper mapper;

91    /**
92     * Fully specified constructor.
93     * 
94     * @param outputValues
95     *          when true, enables output of values along with other log

97     * @param outputJSON
98     *          when true, enables output in JSON format rather than a
99     *          "pretty string"
100    * @param sequence
101    *          when nonnegative, serves as a filter; only log entries with this
102    *          sequence id will be printed
103    * @param region
104    *          when not null, serves as a filter; only log entries from this

106    * @param row
107    *          when not null, serves as a filter; only log entries from this row

111    *          endPersistentOutput() method must be used!
112    * @param out
113    *          Specifies an alternative to stdout for the destination of this 
114    *          PrettyPrinter's output.

116   public HLogPrettyPrinter(boolean outputValues, boolean outputJSON,
117       long sequence, String region, String row, boolean persistentOutput,
118       PrintStream out) {
119     this.outputValues = outputValues;
120     this.outputJSON = outputJSON;
121     this.sequence = sequence;
122     this.region = region;

218   /**
219    * reads a log file and outputs its contents, one transaction at a time, as
220    * specified by the currently configured options
221    * 
222    * @param conf
223    *          the HBase configuration relevant to this log file
224    * @param p
225    *          the path of the log file to be read
226    * @throws IOException
227    *           may be unable to access the configured filesystem or requested
228    *           file.
229    */
230   public void processFile(final Configuration conf, final Path p)
231       throws IOException {
232     FileSystem fs = FileSystem.get(conf);
233     if (!fs.exists(p)) {
234       throw new FileNotFoundException(p.toString());
235     }
236     if (!fs.isFile(p)) {
237       throw new IOException(p + " is not a file");
238     }
239     if (outputJSON && !persistentOutput) {
240       out.print("[");
241       firstTxn = true;
242     }
243     Reader log = HLogFactory.createReader(fs, p, conf);
244     try {
245       FSHLog.Entry entry;
246       while ((entry = log.next()) != null) {
247         HLogKey key = entry.getKey();
248         WALEdit edit = entry.getEdit();
249         // begin building a transaction structure
250         Map<String, Object> txn = key.toStringMap();
251         long writeTime = key.getWriteTime();
252         // check output filters
253         if (sequence >= 0 && ((Long) txn.get("sequence")) != sequence)
254           continue;
255         if (region != null && !((String) txn.get("region")).equals(region))
256           continue;
257         // initialize list into which we will store atomic actions
258         List<Map> actions = new ArrayList<Map>();
259         for (KeyValue kv : edit.getKeyValues()) {
260           // add atomic operation to txn
261           Map<String, Object> op = 
262             new HashMap<String, Object>(kv.toStringMap());
263           if (outputValues)
264             op.put("value", Bytes.toStringBinary(kv.getValue()));
265           // check row output filter
266           if (row == null || ((String) op.get("row")).equals(row))
267             actions.add(op);
268         }
269         if (actions.size() == 0)
270           continue;
271         txn.put("actions", actions);
272         if (outputJSON) {
273           // JSON output is a straightforward "toString" on the txn object
274           if (firstTxn)
275             firstTxn = false;
276           else
277             out.print(",");
278           // encode and print JSON
279           out.print(mapper.writeValueAsString(txn));
280         } else {
281           // Pretty output, complete with indentation by atomic action
282           out.println("Sequence " + txn.get("sequence") + " "
283               + "from region " + txn.get("region") + " " + "in table "
284               + txn.get("table") + " at write timestamp: " + new Date(writeTime));
285           for (int i = 0; i < actions.size(); i++) {
286             Map op = actions.get(i);
287             out.println("  Action:");
288             out.println("    row: " + op.get("row"));
289             out.println("    column: " + op.get("family") + ":"
290                 + op.get("qualifier"));
291             out.println("    timestamp: "
292                 + (new Date((Long) op.get("timestamp"))));
293             if (outputValues)
294               out.println("    value: " + op.get("value"));
295           }
296         }
297       }
298     } finally {
299       log.close();
300     }
301     if (outputJSON && !persistentOutput) {
302       out.print("]");
303     }
304   }

310   /**
311    * Pass one or more log file names and formatting options and it will dump out
312    * a text version of the contents on <code>stdout</code>.
313    * 
314    * @param args
315    *          Command line arguments
316    * @throws IOException
317    *           Thrown upon file system errors etc.
318    * @throws ParseException
319    *           Thrown if command-line parsing fails.
320    */
321   public static void run(String[] args) throws IOException {
322     // create options
323     Options options = new Options();
324     options.addOption("h", "help", false, "Output help message");
325     options.addOption("j", "json", false, "Output JSON");
326     options.addOption("p", "printvals", false, "Print values");
327     options.addOption("r", "region", true,
328         "Region to filter by. Pass region name; e.g. '.META.,,1'");
329     options.addOption("s", "sequence", true,
330         "Sequence to filter by. Pass sequence number.");
331     options.addOption("w", "row", true, "Row to filter by. Pass row name.");
332 
333     HLogPrettyPrinter printer = new HLogPrettyPrinter();
334     CommandLineParser parser = new PosixParser();
335     List files = null;
336     try {
337       CommandLine cmd = parser.parse(options, args);
338       files = cmd.getArgList();
339       if (files.size() == 0 || cmd.hasOption("h")) {
340         HelpFormatter formatter = new HelpFormatter();
341         formatter.printHelp("HLog <filename...>", options, true);
342         System.exit(-1);
343       }
344       // configure the pretty printer using command line options
345       if (cmd.hasOption("p"))
346         printer.enableValues();
347       if (cmd.hasOption("j"))
348         printer.enableJSON();
349       if (cmd.hasOption("r"))
350         printer.setRegionFilter(cmd.getOptionValue("r"));
351       if (cmd.hasOption("s"))
352         printer.setSequenceFilter(Long.parseLong(cmd.getOptionValue("s")));
353       if (cmd.hasOption("w"))
354         printer.setRowFilter(cmd.getOptionValue("w"));
355     } catch (ParseException e) {
356       e.printStackTrace();
357       HelpFormatter formatter = new HelpFormatter();
358       formatter.printHelp("HFile filename(s) ", options, true);
359       System.exit(-1);
360     }
361     // get configuration, file system, and process the given files
362     Configuration conf = HBaseConfiguration.create();
363     conf.set("fs.defaultFS",
364         conf.get(org.apache.hadoop.hbase.HConstants.HBASE_DIR));
365     conf.set("fs.default.name",
366         conf.get(org.apache.hadoop.hbase.HConstants.HBASE_DIR));
367     // begin output
368     printer.beginPersistentOutput();
369     for (Object f : files) {
370       Path file = new Path((String) f);
371       FileSystem fs = file.getFileSystem(conf);
372       if (!fs.exists(file)) {
373         System.err.println("ERROR, file doesnt exist: " + file);
374         return;
375       }
376       printer.processFile(conf, file);
377     }
378     printer.endPersistentOutput();
379   }
