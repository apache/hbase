39  /**
40   * Run bulk assign.  Does one RCP per regionserver passing a
41   * batch of regions using {@link GeneralBulkAssigner.SingleServerBulkAssigner}.
42   */
43  @InterfaceAudience.Private
44  public class GeneralBulkAssigner extends BulkAssigner {
45    private static final Log LOG = LogFactory.getLog(GeneralBulkAssigner.class);
46  
47    private Map<ServerName, List<HRegionInfo>> failedPlans
48      = new ConcurrentHashMap<ServerName, List<HRegionInfo>>();
49    private ExecutorService pool;
50  
51    final Map<ServerName, List<HRegionInfo>> bulkPlan;
52    final AssignmentManager assignmentManager;
53  
54    GeneralBulkAssigner(final Server server,
55        final Map<ServerName, List<HRegionInfo>> bulkPlan,
56        final AssignmentManager am) {
57      super(server);
58      this.bulkPlan = bulkPlan;
59      this.assignmentManager = am;
60    }
61  
62    @Override
63    protected String getThreadNamePrefix() {
64      return this.server.getServerName() + "-GeneralBulkAssigner";
65    }
66  
67    @Override
68    protected void populatePool(ExecutorService pool) {
69      this.pool = pool; // shut it down later in case some assigner hangs
70      for (Map.Entry<ServerName, List<HRegionInfo>> e: this.bulkPlan.entrySet()) {
71        pool.execute(new SingleServerBulkAssigner(e.getKey(), e.getValue(),
72          this.assignmentManager, this.failedPlans));
73      }
74    }
75  
76    /**
77     *
78     * @param timeout How long to wait.
79     * @return true if done.
80     */
81    @Override
82    protected boolean waitUntilDone(final long timeout)
83    throws InterruptedException {
84      Set<HRegionInfo> regionSet = new HashSet<HRegionInfo>();
85      for (List<HRegionInfo> regionList : bulkPlan.values()) {
86        regionSet.addAll(regionList);
87      }
88  
89      pool.shutdown(); // no more task allowed
90      int serverCount = bulkPlan.size();
91      int regionCount = regionSet.size();
92      long startTime = System.currentTimeMillis();
93      long rpcWaitTime = startTime + timeout;
94      while (!server.isStopped() && !pool.isTerminated()
95          && rpcWaitTime > System.currentTimeMillis()) {
96        if (failedPlans.isEmpty()) {
97          pool.awaitTermination(100, TimeUnit.MILLISECONDS);
98        } else {
99          reassignFailedPlans();
100       }
101     }
102     if (!pool.isTerminated()) {
103       LOG.warn("bulk assigner is still running after "
104         + (System.currentTimeMillis() - startTime) + "ms, shut it down now");
105       // some assigner hangs, can't wait any more, shutdown the pool now
106       List<Runnable> notStarted = pool.shutdownNow();
107       if (notStarted != null && !notStarted.isEmpty()) {
108         server.abort("some single server assigner hasn't started yet"
109           + " when the bulk assigner timed out", null);
110         return false;
111       }
112     }
113 
114     int reassigningRegions = 0;
115     if (!failedPlans.isEmpty() && !server.isStopped()) {
116       reassigningRegions = reassignFailedPlans();
117     }
118 
119     Configuration conf = server.getConfiguration();
120     long perRegionOpenTimeGuesstimate =
121       conf.getLong("hbase.bulk.assignment.perregion.open.time", 1000);
122     long endTime = Math.max(System.currentTimeMillis(), rpcWaitTime)
123       + perRegionOpenTimeGuesstimate * (reassigningRegions + 1);
124     RegionStates regionStates = assignmentManager.getRegionStates();
125     // We're not synchronizing on regionsInTransition now because we don't use any iterator.
126     while (!regionSet.isEmpty() && !server.isStopped() && endTime > System.currentTimeMillis()) {
127       Iterator<HRegionInfo> regionInfoIterator = regionSet.iterator();
128       while (regionInfoIterator.hasNext()) {
129         HRegionInfo hri = regionInfoIterator.next();
130         RegionState state = regionStates.getRegionState(hri);
131         if ((!regionStates.isRegionInTransition(hri) && regionStates.isRegionAssigned(hri))
132             || state.isSplit() || state.isSplitting()) {
133           regionInfoIterator.remove();
134         }
135       }
136       if (!regionSet.isEmpty()) {
137         regionStates.waitForUpdate(100);
138       }
139     }
140 
141     if (LOG.isDebugEnabled()) {
142       long elapsedTime = System.currentTimeMillis() - startTime;
143       String status = "successfully";
144       if (!regionSet.isEmpty()) {
145         status = "with " + regionSet.size() + " regions still not assigned yet";
146       }
147       LOG.debug("bulk assigning total " + regionCount + " regions to "
148         + serverCount + " servers, took " + elapsedTime + "ms, " + status);
149     }
150     return regionSet.isEmpty();
151   }
152 
153   @Override
154   protected long getTimeoutOnRIT() {
155     // Guess timeout.  Multiply the max number of regions on a server
156     // by how long we think one region takes opening.
157     Configuration conf = server.getConfiguration();
158     long perRegionOpenTimeGuesstimate =
159       conf.getLong("hbase.bulk.assignment.perregion.open.time", 1000);
160     int maxRegionsPerServer = 1;
161     for (List<HRegionInfo> regionList : bulkPlan.values()) {
162       int size = regionList.size();
163       if (size > maxRegionsPerServer) {
164         maxRegionsPerServer = size;
165       }
166     }
167     long timeout = perRegionOpenTimeGuesstimate * maxRegionsPerServer
168       + conf.getLong("hbase.regionserver.rpc.startup.waittime", 60000)
169       + conf.getLong("hbase.bulk.assignment.perregionserver.rpc.waittime",
170         30000) * bulkPlan.size();
171     LOG.debug("Timeout-on-RIT=" + timeout);
172     return timeout;
173   }
174 
175   @Override
176   protected UncaughtExceptionHandler getUncaughtExceptionHandler() {
177     return new UncaughtExceptionHandler() {
178       @Override
179       public void uncaughtException(Thread t, Throwable e) {
180         LOG.warn("Assigning regions in " + t.getName(), e);
181       }
182     };
183   }
184 
185   private int reassignFailedPlans() {
186     List<HRegionInfo> reassigningRegions = new ArrayList<HRegionInfo>();
187     for (Map.Entry<ServerName, List<HRegionInfo>> e : failedPlans.entrySet()) {
188       LOG.info("Failed assigning " + e.getValue().size()
189           + " regions to server " + e.getKey() + ", reassigning them");
190       reassigningRegions.addAll(failedPlans.remove(e.getKey()));
191     }
192     for (HRegionInfo region : reassigningRegions) {
193       assignmentManager.invokeAssign(region);
194     }
195     return reassigningRegions.size();
196   }

198   /**
199    * Manage bulk assigning to a server.
200    */
201   static class SingleServerBulkAssigner implements Runnable {
202     private final ServerName regionserver;
203     private final List<HRegionInfo> regions;
204     private final AssignmentManager assignmentManager;
205     private final Map<ServerName, List<HRegionInfo>> failedPlans;
206 
207     SingleServerBulkAssigner(final ServerName regionserver,
208         final List<HRegionInfo> regions, final AssignmentManager am,
209         final Map<ServerName, List<HRegionInfo>> failedPlans) {
210       this.regionserver = regionserver;
211       this.regions = regions;
212       this.assignmentManager = am;
213       this.failedPlans = failedPlans;
214     }
215 
216     @Override
217     public void run() {
218       try {
219        if (!assignmentManager.assign(regionserver, regions)) {
220          failedPlans.put(regionserver, regions);
221        }
222       } catch (Throwable t) {
223         LOG.warn("Failed bulking assigning " + regions.size()
224             + " region(s) to " + regionserver.getServerName()
225             + ", and continue to bulk assign others", t);
226         failedPlans.put(regionserver, regions);
227       }
228     }
229   }
