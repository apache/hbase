54    @Test
55    public void testReadAndWriteHRegionInfoFile() throws IOException, InterruptedException {
56      HBaseTestingUtility htu = new HBaseTestingUtility();
57      HRegionInfo hri = HRegionInfo.FIRST_META_REGIONINFO;
58      Path basedir = htu.getDataTestDir();
59      // Create a region.  That'll write the .regioninfo file.
60      HRegion r = HRegion.createHRegion(hri, basedir, htu.getConfiguration(),
61        HTableDescriptor.META_TABLEDESC);
62      // Get modtime on the file.
63      long modtime = getModTime(r);
64      HRegion.closeHRegion(r);
65      Thread.sleep(1001);
66      r = HRegion.createHRegion(hri, basedir, htu.getConfiguration(), HTableDescriptor.META_TABLEDESC);
67      // Ensure the file is not written for a second time.
68      long modtime2 = getModTime(r);
69      assertEquals(modtime, modtime2);
70      // Now load the file.
71      HRegionInfo deserializedHri =
72        HRegion.loadDotRegionInfoFileContent(FileSystem.get(htu.getConfiguration()), r.getRegionDir());
73      assertTrue(hri.equals(deserializedHri));
74    }
75  
76    long getModTime(final HRegion r) throws IOException {
77      FileStatus [] statuses =
78        r.getFilesystem().listStatus(new Path(r.getRegionDir(), HRegion.REGIONINFO_FILE));
79      assertTrue(statuses != null && statuses.length == 1);
80      return statuses[0].getModificationTime();
81    }

132   @Test
133   public void testContainsRange() {
134     HTableDescriptor tableDesc = new HTableDescriptor("testtable");
135     HRegionInfo hri = new HRegionInfo(
136         tableDesc.getName(), Bytes.toBytes("a"), Bytes.toBytes("g"));
137     // Single row range at start of region
138     assertTrue(hri.containsRange(Bytes.toBytes("a"), Bytes.toBytes("a")));
139     // Fully contained range
140     assertTrue(hri.containsRange(Bytes.toBytes("b"), Bytes.toBytes("c")));
141     // Range overlapping start of region
142     assertTrue(hri.containsRange(Bytes.toBytes("a"), Bytes.toBytes("c")));
143     // Fully contained single-row range
144     assertTrue(hri.containsRange(Bytes.toBytes("c"), Bytes.toBytes("c")));
145     // Range that overlaps end key and hence doesn't fit
146     assertFalse(hri.containsRange(Bytes.toBytes("a"), Bytes.toBytes("g")));
147     // Single row range on end key
148     assertFalse(hri.containsRange(Bytes.toBytes("g"), Bytes.toBytes("g")));
149     // Single row range entirely outside
150     assertFalse(hri.containsRange(Bytes.toBytes("z"), Bytes.toBytes("z")));
151     
152     // Degenerate range
153     try {
154       hri.containsRange(Bytes.toBytes("z"), Bytes.toBytes("a"));
155       fail("Invalid range did not throw IAE");
156     } catch (IllegalArgumentException iae) {
157     }
158   }
