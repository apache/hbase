97    /**
98     * Return an array of scanners corresponding to the given set of store files,
99     * And set the ScanQueryMatcher for each store file scanner for further
100    * optimization
101    */
102   public static List<StoreFileScanner> getScannersForStoreFiles(
103       Collection<StoreFile> files, boolean cacheBlocks, boolean usePread,
104       boolean isCompaction, ScanQueryMatcher matcher) throws IOException {
105     List<StoreFileScanner> scanners = new ArrayList<StoreFileScanner>(
106         files.size());
107     for (StoreFile file : files) {
108       StoreFile.Reader r = file.createReader();
109       StoreFileScanner scanner = r.getStoreFileScanner(cacheBlocks, usePread,
110           isCompaction);
111       scanner.setScanQueryMatcher(matcher);
112       scanners.add(scanner);
113     }
114     return scanners;
115   }

125   public KeyValue next() throws IOException {
126     KeyValue retKey = cur;
127 
128     try {
129       // only seek if we aren't at the end. cur == null implies 'end'.
130       if (cur != null) {
131         hfs.next();
132         cur = hfs.getKeyValue();
133         skipKVsNewerThanReadpoint();
134       }
135     } catch(IOException e) {
136       throw new IOException("Could not iterate " + this, e);
137     }
138     return retKey;
139   }

141   public boolean seek(KeyValue key) throws IOException {
142     seekCount.incrementAndGet();
143 
144     try {
145       try {
146         if(!seekAtOrAfter(hfs, key)) {
147           close();
148           return false;
149         }
150 
151         this.isReseekable = true;
152         cur = hfs.getKeyValue();
153 
154         return skipKVsNewerThanReadpoint();
155       } finally {
156         realSeekDone = true;
157       }
158     } catch (IOException ioe) {
159       throw new IOException("Could not seek " + this + " to key " + key, ioe);
160     }
161   }
162 
163   public boolean reseek(KeyValue key) throws IOException {
164     seekCount.incrementAndGet();
165 
166     try {
167       try {
168         if (!reseekAtOrAfter(hfs, key)) {
169           close();
170           return false;
171         }
172         cur = hfs.getKeyValue();
173 
174         return skipKVsNewerThanReadpoint();
175       } finally {
176         realSeekDone = true;
177       }
178     } catch (IOException ioe) {
179       throw new IOException("Could not reseek " + this + " to key " + key,
180           ioe);
181     }
182   }

184   protected boolean skipKVsNewerThanReadpoint() throws IOException {
185     long readPoint = MultiVersionConsistencyControl.getThreadReadPoint();
186 
187     // We want to ignore all key-values that are newer than our current
188     // readPoint
189     while(enforceMVCC
190         && cur != null
191         && (cur.getMemstoreTS() > readPoint)) {
192       hfs.next();
193       cur = hfs.getKeyValue();
194     }
195 
196     if (cur == null) {
197       close();
198       return false;
199     }
200 
201     // For the optimisation in HBASE-4346, we set the KV's memstoreTS to
202     // 0, if it is older than all the scanners' read points. It is possible
203     // that a newer KV's memstoreTS was reset to 0. But, there is an
204     // older KV which was not reset to 0 (because it was
205     // not old enough during flush). Make sure that we set it correctly now,
206     // so that the comparision order does not change.
207     if (cur.getMemstoreTS() <= readPoint) {
208       cur.setMemstoreTS(0);
209     }
210     return true;
211   }

218   /**
219    *
220    * @param s
221    * @param k
222    * @return false if not found or if k is after the end.
223    * @throws IOException
224    */
225   public static boolean seekAtOrAfter(HFileScanner s, KeyValue k)
226   throws IOException {
227     int result = s.seekTo(k.getBuffer(), k.getKeyOffset(), k.getKeyLength());
228     if(result < 0) {
229       // Passed KV is smaller than first KV in file, work from start of file
230       return s.seekTo();
231     } else if(result > 0) {
232       // Passed KV is larger than current KV in file, if there is a next
233       // it is the "after", if not then this scanner is done.
234       return s.next();
235     }
236     // Seeked to the exact key
237     return true;
238   }
239 
240   static boolean reseekAtOrAfter(HFileScanner s, KeyValue k)
241   throws IOException {
242     //This function is similar to seekAtOrAfter function
243     int result = s.reseekTo(k.getBuffer(), k.getKeyOffset(), k.getKeyLength());
244     if (result <= 0) {
245       return true;
246     } else {
247       // passed KV is larger than current KV in file, if there is a next
248       // it is after, if not then this scanner is done.
249       return s.next();
250     }
251   }

258   /**
259    * Pretend we have done a seek but don't do it yet, if possible. The hope is
260    * that we find requested columns in more recent files and won't have to seek
261    * in older files. Creates a fake key/value with the given row/column and the
262    * highest (most recent) possible timestamp we might get from this file. When
263    * users of such "lazy scanner" need to know the next KV precisely (e.g. when
264    * this scanner is at the top of the heap), they run {@link #enforceSeek()}.
265    * <p>
266    * Note that this function does guarantee that the current KV of this scanner
267    * will be advanced to at least the given KV. Because of this, it does have
268    * to do a real seek in cases when the seek timestamp is older than the
269    * highest timestamp of the file, e.g. when we are trying to seek to the next
270    * row/column and use OLDEST_TIMESTAMP in the seek key.
271    */
272   @Override
273   public boolean requestSeek(KeyValue kv, boolean forward, boolean useBloom)
274       throws IOException {
275     if (kv.getFamilyLength() == 0) {
276       useBloom = false;
277     }
278 
279     boolean haveToSeek = true;
280     if (useBloom) {
281       // check ROWCOL Bloom filter first.
282       if (reader.getBloomFilterType() == StoreFile.BloomType.ROWCOL) {
283         haveToSeek = reader.passesGeneralBloomFilter(kv.getBuffer(),
284             kv.getRowOffset(), kv.getRowLength(), kv.getBuffer(),
285             kv.getQualifierOffset(), kv.getQualifierLength());
286       } else if (this.matcher != null && !matcher.hasNullColumnInQuery() &&
287           kv.isDeleteFamily()) {
288         // if there is no such delete family kv in the store file,
289         // then no need to seek.
290         haveToSeek = reader.passesDeleteFamilyBloomFilter(kv.getBuffer(),
291             kv.getRowOffset(), kv.getRowLength());
292       }
293     }
294 
295     delayedReseek = forward;
296     delayedSeekKV = kv;
297 
298     if (haveToSeek) {
299       // This row/column might be in this store file (or we did not use the
300       // Bloom filter), so we still need to seek.
301       realSeekDone = false;
302       long maxTimestampInFile = reader.getMaxTimestamp();
303       long seekTimestamp = kv.getTimestamp();
304       if (seekTimestamp > maxTimestampInFile) {
305         // Create a fake key that is not greater than the real next key.
306         // (Lower timestamps correspond to higher KVs.)
307         // To understand this better, consider that we are asked to seek to
308         // a higher timestamp than the max timestamp in this file. We know that
309         // the next point when we have to consider this file again is when we
310         // pass the max timestamp of this file (with the same row/column).
311         cur = kv.createFirstOnRowColTS(maxTimestampInFile);
312       } else {
313         // This will be the case e.g. when we need to seek to the next
314         // row/column, and we don't know exactly what they are, so we set the
315         // seek key's timestamp to OLDEST_TIMESTAMP to skip the rest of this
316         // row/column.
317         enforceSeek();
318       }
319       return cur != null;
320     }
321 
322     // Multi-column Bloom filter optimization.
323     // Create a fake key/value, so that this scanner only bubbles up to the top
324     // of the KeyValueHeap in StoreScanner after we scanned this row/column in
325     // all other store files. The query matcher will then just skip this fake
326     // key/value and the store scanner will progress to the next column. This
327     // is obviously not a "real real" seek, but unlike the fake KV earlier in
328     // this method, we want this to be propagated to ScanQueryMatcher.
329     cur = kv.createLastOnRowCol();
330 
331     realSeekDone = true;
332     return true;
333   }

344   @Override
345   public void enforceSeek() throws IOException {
346     if (realSeekDone)
347       return;
348 
349     if (delayedReseek && this.isReseekable) {
350       reseek(delayedSeekKV);
351     } else {
352       seek(delayedSeekKV);
353     }
354   }
