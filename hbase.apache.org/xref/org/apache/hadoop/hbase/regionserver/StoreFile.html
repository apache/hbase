79  /**
80   * A Store data file.  Stores usually have one or more of these files.  They
81   * are produced by flushing the memstore to disk.  To
82   * create, instantiate a writer using {@link StoreFile.WriterBuilder}
83   * and append data. Be sure to add any metadata before calling close on the
84   * Writer (Use the appendMetadata convenience methods). On close, a StoreFile
85   * is sitting in the Filesystem.  To refer to it, create a StoreFile instance
86   * passing filesystem and path.  To read, call {@link #createReader()}.
87   * <p>StoreFiles may also reference store files in another Store.
88   *
89   * The reason for this weird pattern where you use a different instance for the
90   * writer and a reader is that we write once but read a lot more.
91   */

223   /**
224    * Constructor, loads a reader and it's indices, etc. May allocate a
225    * substantial amount of ram depending on the underlying files (10-20MB?).
226    *
227    * @param fs  The current file system to use.
228    * @param p  The path of the file.
229    * @param conf  The current configuration.
230    * @param cacheConf  The cache configuration and block cache reference.
231    * @param cfBloomType The bloom type to use for this store file as specified
232    *          by column family configuration. This may or may not be the same
233    *          as the Bloom filter type actually present in the HFile, because
234    *          column family configuration might change. If this is
235    *          {@link BloomType#NONE}, the existing Bloom filter is ignored.
236    * @param dataBlockEncoder data block encoding algorithm.
237    * @throws IOException When opening the reader fails.
238    */
239   public StoreFile(final FileSystem fs,
240             final Path p,
241             final Configuration conf,
242             final CacheConfig cacheConf,
243             final BloomType cfBloomType,
244             final HFileDataBlockEncoder dataBlockEncoder)
245       throws IOException {
246     this.fs = fs;
247     this.path = p;
248     this.cacheConf = cacheConf;
249     this.dataBlockEncoder =
250         dataBlockEncoder == null ? NoOpDataBlockEncoder.INSTANCE
251             : dataBlockEncoder;
252 
253     if (HFileLink.isHFileLink(p)) {
254       this.link = new HFileLink(conf, p);
255       LOG.debug("Store file " + p + " is a link");
256     } else if (isReference(p)) {
257       this.reference = Reference.read(fs, p);
258       this.referencePath = getReferredToFile(this.path);
259       LOG.debug("Store file " + p + " is a reference");
260     }
261 
262     if (BloomFilterFactory.isGeneralBloomEnabled(conf)) {
263       this.cfBloomType = cfBloomType;
264     } else {
265       LOG.info("Ignoring bloom filter check for file " + path + ": " +
266           "cfBloomType=" + cfBloomType + " (disabled in config)");
267       this.cfBloomType = BloomType.NONE;
268     }
269 
270     // cache the modification time stamp of this store file
271     FileStatus[] stats = FSUtils.listStatus(fs, p, null);
272     if (stats != null && stats.length == 1) {
273       this.modificationTimeStamp = stats[0].getModificationTime();
274     } else {
275       this.modificationTimeStamp = 0;
276     }
277

318   /**
319    * @param p Path to check.
320    * @param m Matcher to use.
321    * @return True if the path has format of a HStoreFile reference.
322    */
323   public static boolean isReference(final Path p, final Matcher m) {
324     if (m == null || !m.matches()) {
325       LOG.warn("Failed match of store file name " + p.toString());
326       throw new RuntimeException("Failed match of store file name " +
327           p.toString());
328     }
329     return m.groupCount() > 1 && m.group(2) != null;
330   }

332   /*
333    * Return path to the file referred to by a Reference.  Presumes a directory
334    * hierarchy of <code>${hbase.rootdir}/tablename/regionname/familyname</code>.
335    * @param p Path to a Reference file.
336    * @return Calculated path to parent region file.
337    * @throws IOException
338    */
339   static Path getReferredToFile(final Path p) {
340     Matcher m = REF_NAME_PARSER.matcher(p.getName());
341     if (m == null || !m.matches()) {
342       LOG.warn("Failed match of store file name " + p.toString());
343       throw new RuntimeException("Failed match of store file name " +
344           p.toString());
345     }
346     // Other region name is suffix on the passed Reference file name
347     String otherRegion = m.group(2);
348     // Tabledir is up two directories from where Reference was written.
349     Path tableDir = p.getParent().getParent().getParent();
350     String nameStrippedOfSuffix = m.group(1);
351     // Build up new path with the referenced region in place of our current
352     // region in the reference path.  Also strip regionname suffix from name.
353     return new Path(new Path(new Path(tableDir, otherRegion),
354       p.getParent().getName()), nameStrippedOfSuffix);
355   }

447   /**
448    * helper function to compute HDFS blocks distribution of a given reference
449    * file.For reference file, we don't compute the exact value. We use some
450    * estimate instead given it might be good enough. we assume bottom part
451    * takes the first half of reference file, top part takes the second half
452    * of the reference file. This is just estimate, given
453    * midkey ofregion != midkey of HFile, also the number and size of keys vary.
454    * If this estimate isn't good enough, we can improve it later.
455    * @param fs  The FileSystem
456    * @param reference  The reference
457    * @param reference  The referencePath
458    * @return HDFS blocks distribution
459    */
460   static private HDFSBlocksDistribution computeRefFileHDFSBlockDistribution(
461     FileSystem fs, Reference reference, Path referencePath) throws IOException {
462     if ( referencePath == null) {
463       return null;
464     }
465 
466     FileStatus status = fs.getFileStatus(referencePath);
467     long start = 0;
468     long length = 0;
469 
470     if (Reference.isTopFileRegion(reference.getFileRegion())) {
471       start = status.getLen()/2;
472       length = status.getLen() - status.getLen()/2;
473     } else {
474       start = 0;
475       length = status.getLen()/2;
476     }
477     return FSUtils.computeHDFSBlocksDistribution(fs, status, start, length);
478   }

480   /**
481    * helper function to compute HDFS blocks distribution of a given file.
482    * For reference file, it is an estimate
483    * @param fs  The FileSystem
484    * @param p  The path of the file
485    * @return HDFS blocks distribution
486    */
487   static public HDFSBlocksDistribution computeHDFSBlockDistribution(
488     FileSystem fs, Path p) throws IOException {
489     if (isReference(p)) {
490       Reference reference = Reference.read(fs, p);
491       Path referencePath = getReferredToFile(p);
492       return computeRefFileHDFSBlockDistribution(fs, reference, referencePath);
493     } else {
494       if (HFileLink.isHFileLink(p)) p = HFileLink.getReferencedPath(fs, p);
495       FileStatus status = fs.getFileStatus(p);
496       long length = status.getLen();
497       return FSUtils.computeHDFSBlocksDistribution(fs, status, 0, length);
498     }
499   }

502   /**
503    * compute HDFS block distribution, for reference file, it is an estimate
504    */
505   private void computeHDFSBlockDistribution() throws IOException {
506     if (isReference()) {
507       this.hdfsBlocksDistribution = computeRefFileHDFSBlockDistribution(
508         this.fs, this.reference, this.referencePath);
509     } else {
510       FileStatus status;
511       if (isLink()) {
512         status = link.getFileStatus(fs);
513       } else {
514         status = this.fs.getFileStatus(path);
515       }
516       long length = status.getLen();
517       this.hdfsBlocksDistribution = FSUtils.computeHDFSBlocksDistribution(
518         this.fs, status, 0, length);
519     }
520   }

522   /**
523    * Opens reader on this store file.  Called by Constructor.
524    * @return Reader for the store file.
525    * @throws IOException
526    * @see #closeReader(boolean)
527    */
528   private Reader open() throws IOException {
529     if (this.reader != null) {
530       throw new IllegalAccessError("Already open");
531     }
532     if (isReference()) {
533       this.reader = new HalfStoreFileReader(this.fs, this.referencePath,
534           this.cacheConf, this.reference,
535           dataBlockEncoder.getEncodingInCache());
536     } else if (isLink()) {
537       long size = link.getFileStatus(fs).getLen();
538       this.reader = new Reader(this.fs, this.path, link, size, this.cacheConf,
539                                dataBlockEncoder.getEncodingInCache(), true);
540     } else {
541       this.reader = new Reader(this.fs, this.path, this.cacheConf,
542           dataBlockEncoder.getEncodingInCache());
543     }
544 
545     computeHDFSBlockDistribution();
546 
547     // Load up indices and fileinfo. This also loads Bloom filter type.
548     metadataMap = Collections.unmodifiableMap(this.reader.loadFileInfo());
549 
550     // Read in our metadata.
551     byte [] b = metadataMap.get(MAX_SEQ_ID_KEY);
552     if (b != null) {
553       // By convention, if halfhfile, top half has a sequence number > bottom
554       // half. Thats why we add one in below. Its done for case the two halves
555       // are ever merged back together --rare.  Without it, on open of store,
556       // since store files are distinguished by sequence id, the one half would
557       // subsume the other.
558       this.sequenceid = Bytes.toLong(b);
559       if (isReference()) {
560         if (Reference.isTopFileRegion(this.reference.getFileRegion())) {
561           this.sequenceid += 1;
562         }
563       }
564     }
565 
566     if (isBulkLoadResult()){
567       // generate the sequenceId from the fileName
568       // fileName is of the form <randomName>_SeqId_<id-when-loaded>_
569       String fileName = this.path.getName();
570       int startPos = fileName.indexOf("SeqId_");
571       if (startPos != -1) {
572         this.sequenceid = Long.parseLong(fileName.substring(startPos + 6,
573             fileName.indexOf('_', startPos + 6)));
574         // Handle reference files as done above.
575         if (isReference()) {
576           if (Reference.isTopFileRegion(this.reference.getFileRegion())) {
577             this.sequenceid += 1;
578           }
579         }
580       }
581     }
582     this.reader.setSequenceID(this.sequenceid);
583 
584     b = metadataMap.get(HFileWriterV2.MAX_MEMSTORE_TS_KEY);
585     if (b != null) {
586       this.maxMemstoreTS = Bytes.toLong(b);
587     }
588 
589     b = metadataMap.get(MAJOR_COMPACTION_KEY);
590     if (b != null) {
591       boolean mc = Bytes.toBoolean(b);
592       if (this.majorCompaction == null) {
593         this.majorCompaction = new AtomicBoolean(mc);
594       } else {
595         this.majorCompaction.set(mc);
596       }
597     } else {
598       // Presume it is not major compacted if it doesn't explicity say so
599       // HFileOutputFormat explicitly sets the major compacted key.
600       this.majorCompaction = new AtomicBoolean(false);
601     }
602 
603     b = metadataMap.get(EXCLUDE_FROM_MINOR_COMPACTION_KEY);
604     this.excludeFromMinorCompaction = (b != null && Bytes.toBoolean(b));
605 
606     BloomType hfileBloomType = reader.getBloomFilterType();
607     if (cfBloomType != BloomType.NONE) {
608       reader.loadBloomfilter(BlockType.GENERAL_BLOOM_META);
609       if (hfileBloomType != cfBloomType) {
610         LOG.info("HFile Bloom filter type for "
611             + reader.getHFileReader().getName() + ": " + hfileBloomType
612             + ", but " + cfBloomType + " specified in column family "
613             + "configuration");
614       }
615     } else if (hfileBloomType != BloomType.NONE) {
616       LOG.info("Bloom filter turned off by CF config for "
617           + reader.getHFileReader().getName());
618     }
619 
620     // load delete family bloom filter
621     reader.loadBloomfilter(BlockType.DELETE_FAMILY_BLOOM_META);
622 
623     try {
624       byte [] timerangeBytes = metadataMap.get(TIMERANGE_KEY);
625       if (timerangeBytes != null) {
626         this.reader.timeRangeTracker = new TimeRangeTracker();
627         Writables.copyWritable(timerangeBytes, this.reader.timeRangeTracker);
628       }
629     } catch (IllegalArgumentException e) {
630       LOG.error("Error reading timestamp range data from meta -- " +
631           "proceeding without", e);
632       this.reader.timeRangeTracker = null;
633     }
634     return this.reader;
635   }

996      * @param bloomType bloom filter setting
997      * @param maxKeys the expected maximum number of keys to be added. Was used
998      *        for Bloom filter size in {@link HFile} format version 1.

1001      * @throws IOException problem writing to FS

1007         final KVComparator comparator, BloomType bloomType, long maxKeys,

1015           .withCompression(compress)

1102     private void appendGeneralBloomfilter(final KeyValue kv) throws IOException {
1103       if (this.generalBloomFilterWriter != null) {
1104         // only add to the bloom filter on a new, unique key
1105         boolean newKey = true;
1106         if (this.lastKv != null) {
1107           switch(bloomType) {
1108           case ROW:
1109             newKey = ! kvComparator.matchingRows(kv, lastKv);
1110             break;
1111           case ROWCOL:
1112             newKey = ! kvComparator.matchingRowColumn(kv, lastKv);
1113             break;
1114           case NONE:
1115             newKey = false;
1116             break;
1117           default:
1118             throw new IOException("Invalid Bloom filter type: " + bloomType +
1119                 " (ROW or ROWCOL expected)");
1120           }
1121         }
1122         if (newKey) {
1123           /*
1124            * http://2.bp.blogspot.com/_Cib_A77V54U/StZMrzaKufI/AAAAAAAAADo/ZhK7bGoJdMQ/s400/KeyValue.png
1125            * Key = RowLen + Row + FamilyLen + Column [Family + Qualifier] + TimeStamp
1126            *
1127            * 2 Types of Filtering:
1128            *  1. Row = Row
1129            *  2. RowCol = Row + Qualifier
1130            */
1131           byte[] bloomKey;
1132           int bloomKeyOffset, bloomKeyLen;
1133 
1134           switch (bloomType) {
1135           case ROW:
1136             bloomKey = kv.getBuffer();
1137             bloomKeyOffset = kv.getRowOffset();
1138             bloomKeyLen = kv.getRowLength();
1139             break;
1140           case ROWCOL:
1141             // merge(row, qualifier)
1142             // TODO: could save one buffer copy in case of compound Bloom
1143             // filters when this involves creating a KeyValue
1144             bloomKey = generalBloomFilterWriter.createBloomKey(kv.getBuffer(),
1145                 kv.getRowOffset(), kv.getRowLength(), kv.getBuffer(),
1146                 kv.getQualifierOffset(), kv.getQualifierLength());
1147             bloomKeyOffset = 0;
1148             bloomKeyLen = bloomKey.length;
1149             break;
1150           default:
1151             throw new IOException("Invalid Bloom filter type: " + bloomType +
1152                 " (ROW or ROWCOL expected)");
1153           }
1154           generalBloomFilterWriter.add(bloomKey, bloomKeyOffset, bloomKeyLen);
1155           if (lastBloomKey != null
1156               && generalBloomFilterWriter.getComparator().compare(bloomKey,
1157                   bloomKeyOffset, bloomKeyLen, lastBloomKey,
1158                   lastBloomKeyOffset, lastBloomKeyLen) <= 0) {
1159             throw new IOException("Non-increasing Bloom keys: "
1160                 + Bytes.toStringBinary(bloomKey, bloomKeyOffset, bloomKeyLen)
1161                 + " after "
1162                 + Bytes.toStringBinary(lastBloomKey, lastBloomKeyOffset,
1163                     lastBloomKeyLen));
1164           }
1165           lastBloomKey = bloomKey;
1166           lastBloomKeyOffset = bloomKeyOffset;
1167           lastBloomKeyLen = bloomKeyLen;
1168           this.lastKv = kv;
1169         }
1170       }
1171     }

1173     private void appendDeleteFamilyBloomFilter(final KeyValue kv)
1174         throws IOException {
1175       if (!kv.isDeleteFamily()) {
1176         return;
1177       }
1178 
1179       // increase the number of delete family in the store file
1180       deleteFamilyCnt++;
1181       if (null != this.deleteFamilyBloomFilterWriter) {
1182         boolean newKey = true;
1183         if (lastDeleteFamilyKV != null) {
1184           newKey = !kvComparator.matchingRows(kv, lastDeleteFamilyKV);
1185         }
1186         if (newKey) {
1187           this.deleteFamilyBloomFilterWriter.add(kv.getBuffer(),
1188               kv.getRowOffset(), kv.getRowLength());
1189           this.lastDeleteFamilyKV = kv;
1190         }
1191       }
1192     }

1503         int rowLen, byte[] col, int colOffset, int colLen) {
1504       // Cache Bloom filter as a local variable in case it is set to null by
1505       // another thread on an IO error.
1506       BloomFilter bloomFilter = this.generalBloomFilter;
1507       if (bloomFilter == null) {
1508         return true;
1509       }
1510 
1511       byte[] key;
1512       switch (bloomFilterType) {
1513         case ROW:
1514           if (col != null) {
1515             throw new RuntimeException("Row-only Bloom filter called with " +
1516                 "column specified");
1517           }

1519               throw new AssertionError("For row-only Bloom filters the row "
1520                   + "must occupy the whole array");
1521           }
1522           key = row;
1523           break;
1524 
1525         case ROWCOL:
1526           key = bloomFilter.createBloomKey(row, rowOffset, rowLen, col,
1527               colOffset, colLen);
1528           break;
1529 
1530         default:
1531           return true;
1532       }
1533 
1534       // Empty file
1535       if (reader.getTrailer().getEntryCount() == 0)
1536         return false;

1540         ByteBuffer bloom;
1541         if (bloomFilter.supportsAutoLoading()) {
1542           bloom = null;
1543           shouldCheckBloom = true;
1544         } else {

1546               true);
1547           shouldCheckBloom = bloom != null;
1548         }
1549 
1550         if (shouldCheckBloom) {
1551           boolean exists;
1552 
1553           // Whether the primary Bloom key is greater than the last Bloom key
1554           // from the file info. For row-column Bloom filters this is not yet
1555           // a sufficient condition to return false.
1556           boolean keyIsAfterLast = lastBloomKey != null
1557               && bloomFilter.getComparator().compare(key, lastBloomKey) > 0;
1558 
1559           if (bloomFilterType == BloomType.ROWCOL) {
1560             // Since a Row Delete is essentially a DeleteFamily applied to all
1561             // columns, a file might be skipped if using row+col Bloom filter.
1562             // In order to ensure this file is included an additional check is
1563             // required looking only for a row bloom.
1564             byte[] rowBloomKey = bloomFilter.createBloomKey(row, 0, row.length,
1565                 null, 0, 0);
1566 
1567             if (keyIsAfterLast
1568                 && bloomFilter.getComparator().compare(rowBloomKey,
1569                     lastBloomKey) > 0) {
1570               exists = false;
1571             } else {
1572               exists =
1573                   bloomFilter.contains(key, 0, key.length, bloom) ||
1574                   bloomFilter.contains(rowBloomKey, 0, rowBloomKey.length,
1575                       bloom);
1576             }
1577           } else {
1578             exists = !keyIsAfterLast
1579                 && bloomFilter.contains(key, 0, key.length, bloom);
1580           }
1581 
1582           return exists;
1583         }
1584       } catch (IOException e) {

1586             e);

1588       } catch (IllegalArgumentException e) {

1590         setGeneralBloomFilterFaulty();
1591       }
