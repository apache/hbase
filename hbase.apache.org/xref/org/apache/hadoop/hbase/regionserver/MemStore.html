47  /**
48   * The MemStore holds in-memory modifications to the Store.  Modifications
49   * are {@link KeyValue}s.  When asked to flush, current memstore is moved
50   * to snapshot and is cleared.  We continue to serve edits out of new memstore
51   * and backing snapshot until flusher reports in that the flush succeeded. At
52   * this point we let the snapshot go.
53   * TODO: Adjust size of the memstore when we remove items because they have
54   * been deleted.
55   * TODO: With new KVSLS, need to make sure we update HeapSize with difference
56   * in KV size.
57   */
58  @InterfaceAudience.Private
59  public class MemStore implements HeapSize {
60    private static final Log LOG = LogFactory.getLog(MemStore.class);

63      "hbase.hregion.memstore.mslab.enabled";
64    private static final boolean USEMSLAB_DEFAULT = false;

137   /**
138    * Creates a snapshot of the current memstore.
139    * Snapshot must be cleared by call to {@link #clearSnapshot(SortedSet<KeyValue>)}
140    * To get the snapshot made by this method, use {@link #getSnapshot()}
141    */
142   void snapshot() {
143     this.lock.writeLock().lock();
144     try {
145       // If snapshot currently has entries, then flusher failed or didn't call
146       // cleanup.  Log a warning.
147       if (!this.snapshot.isEmpty()) {
148         LOG.warn("Snapshot called again without clearing previous. " +
149           "Doing nothing. Another ongoing flush or did we fail last attempt?");
150       } else {
151         if (!this.kvset.isEmpty()) {
152           this.snapshot = this.kvset;
153           this.kvset = new KeyValueSkipListSet(this.comparator);
154           this.snapshotTimeRangeTracker = this.timeRangeTracker;
155           this.timeRangeTracker = new TimeRangeTracker();
156           // Reset heap to not include any keys
157           this.size.set(DEEP_OVERHEAD);
158           // Reset allocator so we get a fresh buffer for the new memstore
159           if (allocator != null) {
160             this.allocator = new MemStoreLAB(conf);
161           }
162         }
163       }
164     } finally {
165       this.lock.writeLock().unlock();
166     }
167   }

181   /**
182    * The passed snapshot was successfully persisted; it can be let go.
183    * @param ss The snapshot to clean out.
184    * @throws UnexpectedException
185    * @see {@link #snapshot()}
186    */
187   void clearSnapshot(final SortedSet<KeyValue> ss)
188   throws UnexpectedException {
189     this.lock.writeLock().lock();
190     try {
191       if (this.snapshot != ss) {
192         throw new UnexpectedException("Current snapshot is " +
193           this.snapshot + ", was passed " + ss);
194       }
195       // OK. Passed in snapshot is same as current snapshot.  If not-empty,
196       // create a new snapshot and let the old one go.
197       if (!ss.isEmpty()) {
198         this.snapshot = new KeyValueSkipListSet(this.comparator);
199         this.snapshotTimeRangeTracker = new TimeRangeTracker();
200       }
201     } finally {
202       this.lock.writeLock().unlock();
203     }
204   }

206   /**
207    * Write an update
208    * @param kv
209    * @return approximate size of the passed key and value.
210    */
211   long add(final KeyValue kv) {
212     this.lock.readLock().lock();
213     try {
214       KeyValue toAdd = maybeCloneWithAllocator(kv);
215       return internalAdd(toAdd);
216     } finally {
217       this.lock.readLock().unlock();
218     }
219   }

221   /**
222    * Internal version of add() that doesn't clone KVs with the
223    * allocator, and doesn't take the lock.
224    *
225    * Callers should ensure they already have the read lock taken
226    */
227   private long internalAdd(final KeyValue toAdd) {
228     long s = heapSizeChange(toAdd, this.kvset.add(toAdd));
229     timeRangeTracker.includeTimestamp(toAdd);
230     this.size.addAndGet(s);
231     return s;
232   }
233 
234   private KeyValue maybeCloneWithAllocator(KeyValue kv) {
235     if (allocator == null) {
236       return kv;
237     }
238 
239     int len = kv.getLength();
240     Allocation alloc = allocator.allocateBytes(len);
241     if (alloc == null) {
242       // The allocation was too large, allocator decided
243       // not to do anything with it.
244       return kv;
245     }
246     assert alloc != null && alloc.getData() != null;
247     System.arraycopy(kv.getBuffer(), kv.getOffset(), alloc.getData(), alloc.getOffset(), len);
248     KeyValue newKv = new KeyValue(alloc.getData(), alloc.getOffset(), len);
249     newKv.setMemstoreTS(kv.getMemstoreTS());
250     return newKv;
251   }

253   /**
254    * Remove n key from the memstore. Only kvs that have the same key and the
255    * same memstoreTS are removed.  It is ok to not update timeRangeTracker
256    * in this call. It is possible that we can optimize this method by using
257    * tailMap/iterator, but since this method is called rarely (only for
258    * error recovery), we can leave those optimization for the future.
259    * @param kv
260    */
261   void rollback(final KeyValue kv) {
262     this.lock.readLock().lock();
263     try {
264       // If the key is in the snapshot, delete it. We should not update
265       // this.size, because that tracks the size of only the memstore and
266       // not the snapshot. The flush of this snapshot to disk has not
267       // yet started because Store.flush() waits for all rwcc transactions to
268       // commit before starting the flush to disk.
269       KeyValue found = this.snapshot.get(kv);
270       if (found != null && found.getMemstoreTS() == kv.getMemstoreTS()) {
271         this.snapshot.remove(kv);
272       }
273       // If the key is in the memstore, delete it. Update this.size.
274       found = this.kvset.get(kv);
275       if (found != null && found.getMemstoreTS() == kv.getMemstoreTS()) {
276         this.kvset.remove(kv);
277         long s = heapSizeChange(kv, true);
278         this.size.addAndGet(-s);
279       }
280     } finally {
281       this.lock.readLock().unlock();
282     }
283   }

285   /**
286    * Write a delete
287    * @param delete
288    * @return approximate size of the passed key and value.
289    */
290   long delete(final KeyValue delete) {
291     long s = 0;
292     this.lock.readLock().lock();
293     try {
294       KeyValue toAdd = maybeCloneWithAllocator(delete);
295       s += heapSizeChange(toAdd, this.kvset.add(toAdd));
296       timeRangeTracker.includeTimestamp(toAdd);
297     } finally {
298       this.lock.readLock().unlock();
299     }
300     this.size.addAndGet(s);
301     return s;
302   }

306    * first.
307    * @return Next row or null if none found.

309   KeyValue getNextRow(final KeyValue kv) {
310     this.lock.readLock().lock();
311     try {
312       return getLowest(getNextRow(kv, this.kvset), getNextRow(kv, this.snapshot));
313     } finally {
314       this.lock.readLock().unlock();

318   /*
319    * @param a
320    * @param b
321    * @return Return lowest of a or b or null if both a and b are null
322    */
323   private KeyValue getLowest(final KeyValue a, final KeyValue b) {
324     if (a == null) {
325       return b;
326     }
327     if (b == null) {
328       return a;
329     }
330     return comparator.compareRows(a, b) <= 0? a: b;
331   }

333   /*
334    * @param key Find row that follows this one.  If null, return first.
335    * @param map Set to look in for a row beyond <code>row</code>.
336    * @return Next row or null if none found.  If one found, will be a new
337    * KeyValue -- can be destroyed by subsequent calls to this method.
338    */
339   private KeyValue getNextRow(final KeyValue key,
340       final NavigableSet<KeyValue> set) {
341     KeyValue result = null;
342     SortedSet<KeyValue> tail = key == null? set: set.tailSet(key);
343     // Iterate until we fall into the next row; i.e. move off current row
344     for (KeyValue kv: tail) {
345       if (comparator.compareRows(kv, key) <= 0)
346         continue;
347       // Note: Not suppressing deletes or expired cells.  Needs to be handled
348       // by higher up functions.
349       result = kv;
350       break;
351     }
352     return result;
353   }

368   /*
369    * @param set
370    * @param state Accumulates deletes and candidates.
371    */
372   private void getRowKeyAtOrBefore(final NavigableSet<KeyValue> set,
373       final GetClosestRowBeforeTracker state) {
374     if (set.isEmpty()) {
375       return;
376     }
377     if (!walkForwardInSingleRow(set, state.getTargetKey(), state)) {
378       // Found nothing in row.  Try backing up.
379       getRowKeyBefore(set, state);
380     }
381   }

383   /*
384    * Walk forward in a row from <code>firstOnRow</code>.  Presumption is that
385    * we have been passed the first possible key on a row.  As we walk forward
386    * we accumulate deletes until we hit a candidate on the row at which point
387    * we return.
388    * @param set
389    * @param firstOnRow First possible key on this row.
390    * @param state
391    * @return True if we found a candidate walking this row.
392    */
393   private boolean walkForwardInSingleRow(final SortedSet<KeyValue> set,
394       final KeyValue firstOnRow, final GetClosestRowBeforeTracker state) {
395     boolean foundCandidate = false;
396     SortedSet<KeyValue> tail = set.tailSet(firstOnRow);
397     if (tail.isEmpty()) return foundCandidate;
398     for (Iterator<KeyValue> i = tail.iterator(); i.hasNext();) {
399       KeyValue kv = i.next();
400       // Did we go beyond the target row? If so break.
401       if (state.isTooFar(kv, firstOnRow)) break;
402       if (state.isExpired(kv)) {
403         i.remove();
404         continue;
405       }
406       // If we added something, this row is a contender. break.
407       if (state.handle(kv)) {
408         foundCandidate = true;
409         break;
410       }
411     }
412     return foundCandidate;
413   }

415   /*
416    * Walk backwards through the passed set a row at a time until we run out of
417    * set or until we get a candidate.
418    * @param set
419    * @param state
420    */
421   private void getRowKeyBefore(NavigableSet<KeyValue> set,
422       final GetClosestRowBeforeTracker state) {
423     KeyValue firstOnRow = state.getTargetKey();
424     for (Member p = memberOfPreviousRow(set, state, firstOnRow);
425         p != null; p = memberOfPreviousRow(p.set, state, firstOnRow)) {
426       // Make sure we don't fall out of our table.
427       if (!state.isTargetTable(p.kv)) break;
428       // Stop looking if we've exited the better candidate range.
429       if (!state.isBetterCandidate(p.kv)) break;
430       // Make into firstOnRow
431       firstOnRow = new KeyValue(p.kv.getRow(), HConstants.LATEST_TIMESTAMP);
432       // If we find something, break;
433       if (walkForwardInSingleRow(p.set, firstOnRow, state)) break;
434     }
435   }

437   /**
438    * Only used by tests. TODO: Remove
439    *
440    * Given the specs of a column, update it, first by inserting a new record,
441    * then removing the old one.  Since there is only 1 KeyValue involved, the memstoreTS
442    * will be set to 0, thus ensuring that they instantly appear to anyone. The underlying
443    * store will ensure that the insert/delete each are atomic. A scanner/reader will either
444    * get the new value, or the old value and all readers will eventually only see the new
445    * value after the old was removed.
446    *
447    * @param row
448    * @param family
449    * @param qualifier
450    * @param newValue
451    * @param now
452    * @return  Timestamp
453    */
454   long updateColumnValue(byte[] row,
455                                 byte[] family,
456                                 byte[] qualifier,
457                                 long newValue,
458                                 long now) {
459    this.lock.readLock().lock();
460     try {
461       KeyValue firstKv = KeyValue.createFirstOnRow(
462           row, family, qualifier);
463       // Is there a KeyValue in 'snapshot' with the same TS? If so, upgrade the timestamp a bit.
464       SortedSet<KeyValue> snSs = snapshot.tailSet(firstKv);
465       if (!snSs.isEmpty()) {
466         KeyValue snKv = snSs.first();
467         // is there a matching KV in the snapshot?
468         if (snKv.matchingRow(firstKv) && snKv.matchingQualifier(firstKv)) {
469           if (snKv.getTimestamp() == now) {
470             // poop,
471             now += 1;
472           }
473         }
474       }
475 
476       // logic here: the new ts MUST be at least 'now'. But it could be larger if necessary.
477       // But the timestamp should also be max(now, mostRecentTsInMemstore)
478 
479       // so we cant add the new KV w/o knowing what's there already, but we also
480       // want to take this chance to delete some kvs. So two loops (sad)
481 
482       SortedSet<KeyValue> ss = kvset.tailSet(firstKv);
483       Iterator<KeyValue> it = ss.iterator();
484       while ( it.hasNext() ) {
485         KeyValue kv = it.next();
486 
487         // if this isnt the row we are interested in, then bail:
488         if (!kv.matchingColumn(family,qualifier) || !kv.matchingRow(firstKv) ) {
489           break; // rows dont match, bail.
490         }
491 
492         // if the qualifier matches and it's a put, just RM it out of the kvset.
493         if (kv.getType() == KeyValue.Type.Put.getCode() &&
494             kv.getTimestamp() > now && firstKv.matchingQualifier(kv)) {
495           now = kv.getTimestamp();
496         }
497       }
498 
499       // create or update (upsert) a new KeyValue with
500       // 'now' and a 0 memstoreTS == immediately visible
501       return upsert(Arrays.asList(
502           new KeyValue(row, family, qualifier, now, Bytes.toBytes(newValue))), 1L
503       );
504     } finally {
505       this.lock.readLock().unlock();
506     }
507   }

509   /**
510    * Update or insert the specified KeyValues.
511    * <p>
512    * For each KeyValue, insert into MemStore.  This will atomically upsert the
513    * value for that row/family/qualifier.  If a KeyValue did already exist,
514    * it will then be removed.
515    * <p>
516    * Currently the memstoreTS is kept at 0 so as each insert happens, it will
517    * be immediately visible.  May want to change this so it is atomic across
518    * all KeyValues.
519    * <p>
520    * This is called under row lock, so Get operations will still see updates
521    * atomically.  Scans will only see each KeyValue update as atomic.
522    *
523    * @param kvs
524    * @param readpoint readpoint below which we can safely remove duplicate KVs 
525    * @return change in memstore size
526    */
527   public long upsert(Iterable<KeyValue> kvs, long readpoint) {
528    this.lock.readLock().lock();
529     try {
530       long size = 0;
531       for (KeyValue kv : kvs) {
532         size += upsert(kv, readpoint);
533       }
534       return size;
535     } finally {
536       this.lock.readLock().unlock();
537     }
538   }

540   /**
541    * Inserts the specified KeyValue into MemStore and deletes any existing
542    * versions of the same row/family/qualifier as the specified KeyValue.
543    * <p>
544    * First, the specified KeyValue is inserted into the Memstore.
545    * <p>
546    * If there are any existing KeyValues in this MemStore with the same row,
547    * family, and qualifier, they are removed.
548    * <p>
549    * Callers must hold the read lock.
550    *
551    * @param kv
552    * @return change in size of MemStore
553    */
554   private long upsert(KeyValue kv, long readpoint) {
555     // Add the KeyValue to the MemStore
556     // Use the internalAdd method here since we (a) already have a lock
557     // and (b) cannot safely use the MSLAB here without potentially
558     // hitting OOME - see TestMemStore.testUpsertMSLAB for a
559     // test that triggers the pathological case if we don't avoid MSLAB
560     // here.
561     long addedSize = internalAdd(kv);
562 
563     // Get the KeyValues for the row/family/qualifier regardless of timestamp.
564     // For this case we want to clean up any other puts
565     KeyValue firstKv = KeyValue.createFirstOnRow(
566         kv.getBuffer(), kv.getRowOffset(), kv.getRowLength(),
567         kv.getBuffer(), kv.getFamilyOffset(), kv.getFamilyLength(),
568         kv.getBuffer(), kv.getQualifierOffset(), kv.getQualifierLength());
569     SortedSet<KeyValue> ss = kvset.tailSet(firstKv);
570     Iterator<KeyValue> it = ss.iterator();
571     // versions visible to oldest scanner
572     int versionsVisible = 0;
573     while ( it.hasNext() ) {
574       KeyValue cur = it.next();
575 
576       if (kv == cur) {
577         // ignore the one just put in
578         continue;
579       }
580       // check that this is the row and column we are interested in, otherwise bail
581       if (kv.matchingRow(cur) && kv.matchingQualifier(cur)) {
582         // only remove Puts that concurrent scanners cannot possibly see
583         if (cur.getType() == KeyValue.Type.Put.getCode() && cur.getMemstoreTS() <= readpoint) {
584           if (versionsVisible > 1) {
585             // if we get here we have seen at least one version visible to the oldest scanner,
586             // which means we can prove that no scanner will see this version
587 
588             // false means there was a change, so give us the size.
589             addedSize -= heapSizeChange(cur, true);
590             it.remove();
591           } else {
592             versionsVisible++;
593           }
594         }
595       } else {
596         // past the row or column, done
597         break;
598       }
599     }
600     return addedSize;
601   }

616   /*
617    * @param set Set to walk back in.  Pass a first in row or we'll return
618    * same row (loop).
619    * @param state Utility and context.
620    * @param firstOnRow First item on the row after the one we want to find a
621    * member in.
622    * @return Null or member of row previous to <code>firstOnRow</code>
623    */
624   private Member memberOfPreviousRow(NavigableSet<KeyValue> set,
625       final GetClosestRowBeforeTracker state, final KeyValue firstOnRow) {
626     NavigableSet<KeyValue> head = set.headSet(firstOnRow, false);
627     if (head.isEmpty()) return null;
628     for (Iterator<KeyValue> i = head.descendingIterator(); i.hasNext();) {
629       KeyValue found = i.next();
630       if (state.isExpired(found)) {
631         i.remove();
632         continue;
633       }
634       return new Member(head, found);
635     }
636     return null;
637   }

657   public boolean shouldSeek(Scan scan, long oldestUnexpiredTS) {
658     return (timeRangeTracker.includesTimeRange(scan.getTimeRange()) ||
659         snapshotTimeRangeTracker.includesTimeRange(scan.getTimeRange()))
660         && (Math.max(timeRangeTracker.getMaximumTimestamp(),
661                      snapshotTimeRangeTracker.getMaximumTimestamp()) >=

669   /*
670    * MemStoreScanner implements the KeyValueScanner.
671    * It lets the caller scan the contents of a memstore -- both current
672    * map and snapshot.
673    * This behaves as if it were a real scanner but does not maintain position.
674    */
675   protected class MemStoreScanner extends NonLazyKeyValueScanner {
676     // Next row information for either kvset or snapshot
677     private KeyValue kvsetNextRow = null;
678     private KeyValue snapshotNextRow = null;
679 
680     // last iterated KVs for kvset and snapshot (to restore iterator state after reseek)
681     private KeyValue kvsetItRow = null;
682     private KeyValue snapshotItRow = null;
683     
684     // iterator based scanning.
685     private Iterator<KeyValue> kvsetIt;
686     private Iterator<KeyValue> snapshotIt;
687 
688     // The kvset and snapshot at the time of creating this scanner
689     private KeyValueSkipListSet kvsetAtCreation;
690     private KeyValueSkipListSet snapshotAtCreation;
691 
692     // the pre-calculated KeyValue to be returned by peek() or next()
693     private KeyValue theNext;
694 
695     /*
696     Some notes...
697 
698      So memstorescanner is fixed at creation time. this includes pointers/iterators into
699     existing kvset/snapshot.  during a snapshot creation, the kvset is null, and the
700     snapshot is moved.  since kvset is null there is no point on reseeking on both,
701       we can save us the trouble. During the snapshot->hfile transition, the memstore
702       scanner is re-created by StoreScanner#updateReaders().  StoreScanner should
703       potentially do something smarter by adjusting the existing memstore scanner.
704 
705       But there is a greater problem here, that being once a scanner has progressed
706       during a snapshot scenario, we currently iterate past the kvset then 'finish' up.
707       if a scan lasts a little while, there is a chance for new entries in kvset to
708       become available but we will never see them.  This needs to be handled at the
709       StoreScanner level with coordination with MemStoreScanner.
710 
711       Currently, this problem is only partly managed: during the small amount of time
712       when the StoreScanner has not yet created a new MemStoreScanner, we will miss
713       the adds to kvset in the MemStoreScanner.
714     */

716     MemStoreScanner() {
717       super();
718 
719       kvsetAtCreation = kvset;
720       snapshotAtCreation = snapshot;
721     }

723     private KeyValue getNext(Iterator<KeyValue> it) {
724       long readPoint = MultiVersionConsistencyControl.getThreadReadPoint();
725 
726       KeyValue v = null;
727       try {
728         while (it.hasNext()) {
729           v = it.next();
730           if (v.getMemstoreTS() <= readPoint) {
731             return v;
732           }
733         }
734 
735         return null;
736       } finally {
737         if (v != null) {
738           // in all cases, remember the last KV iterated to
739           if (it == snapshotIt) {
740             snapshotItRow = v;
741           } else {
742             kvsetItRow = v;
743           }
744         }
745       }
746     }

748     /**
749      *  Set the scanner at the seek key.
750      *  Must be called only once: there is no thread safety between the scanner
751      *   and the memStore.
752      * @param key seek value
753      * @return false if the key is null or if there is no data
754      */
755     @Override
756     public synchronized boolean seek(KeyValue key) {
757       if (key == null) {
758         close();
759         return false;
760       }
761 
762       // kvset and snapshot will never be null.
763       // if tailSet can't find anything, SortedSet is empty (not null).
764       kvsetIt = kvsetAtCreation.tailSet(key).iterator();
765       snapshotIt = snapshotAtCreation.tailSet(key).iterator();
766       kvsetItRow = null;
767       snapshotItRow = null;
768 
769       return seekInSubLists(key);
770     }
771 
772 
773     /**
774      * (Re)initialize the iterators after a seek or a reseek.
775      */
776     private synchronized boolean seekInSubLists(KeyValue key){
777       kvsetNextRow = getNext(kvsetIt);
778       snapshotNextRow = getNext(snapshotIt);
779 
780       // Calculate the next value
781       theNext = getLowest(kvsetNextRow, snapshotNextRow);
782 
783       // has data
784       return (theNext != null);
785     }

788     /**
789      * Move forward on the sub-lists set previously by seek.
790      * @param key seek value (should be non-null)
791      * @return true if there is at least one KV to read, false otherwise
792      */
793     @Override
794     public synchronized boolean reseek(KeyValue key) {
795       /*
796       See HBASE-4195 & HBASE-3855 & HBASE-6591 for the background on this implementation.
797       This code is executed concurrently with flush and puts, without locks.
798       Two points must be known when working on this code:
799       1) It's not possible to use the 'kvTail' and 'snapshot'
800        variables, as they are modified during a flush.
801       2) The ideal implementation for performance would use the sub skip list
802        implicitly pointed by the iterators 'kvsetIt' and
803        'snapshotIt'. Unfortunately the Java API does not offer a method to
804        get it. So we remember the last keys we iterated to and restore
805        the reseeked set to at least that point.
806        */
807 
808       kvsetIt = kvsetAtCreation.tailSet(getHighest(key, kvsetItRow)).iterator();
809       snapshotIt = snapshotAtCreation.tailSet(getHighest(key, snapshotItRow)).iterator();
810 
811       return seekInSubLists(key);
812     }

815     @Override
816     public synchronized KeyValue peek() {
817       //DebugPrint.println(" MS@" + hashCode() + " peek = " + getLowest());
818       return theNext;
819     }

821     @Override
822     public synchronized KeyValue next() {
823       if (theNext == null) {
824           return null;
825       }
826 
827       final KeyValue ret = theNext;
828 
829       // Advance one of the iterators
830       if (theNext == kvsetNextRow) {
831         kvsetNextRow = getNext(kvsetIt);
832       } else {
833         snapshotNextRow = getNext(snapshotIt);
834       }
835 
836       // Calculate the next value
837       theNext = getLowest(kvsetNextRow, snapshotNextRow);
838 
839       //long readpoint = ReadWriteConsistencyControl.getThreadReadPoint();
840       //DebugPrint.println(" MS@" + hashCode() + " next: " + theNext + " next_next: " +
841       //    getLowest() + " threadpoint=" + readpoint);
842       return ret;
843     }

845     /*
846      * Returns the lower of the two key values, or null if they are both null.
847      * This uses comparator.compare() to compare the KeyValue using the memstore
848      * comparator.
849      */
850     private KeyValue getLowest(KeyValue first, KeyValue second) {
851       if (first == null && second == null) {
852         return null;
853       }
854       if (first != null && second != null) {
855         int compare = comparator.compare(first, second);
856         return (compare <= 0 ? first : second);
857       }
858       return (first != null ? first : second);
859     }

861     /*
862      * Returns the higher of the two key values, or null if they are both null.
863      * This uses comparator.compare() to compare the KeyValue using the memstore
864      * comparator.
865      */
866     private KeyValue getHighest(KeyValue first, KeyValue second) {
867       if (first == null && second == null) {
868         return null;
869       }
870       if (first != null && second != null) {
871         int compare = comparator.compare(first, second);
872         return (compare > 0 ? first : second);
873       }
874       return (first != null ? first : second);
875     }
