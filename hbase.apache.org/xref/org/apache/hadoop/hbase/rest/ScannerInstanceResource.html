46  @InterfaceAudience.Private
47  public class ScannerInstanceResource extends ResourceBase {
48    private static final Log LOG =
49      LogFactory.getLog(ScannerInstanceResource.class);
50  
51    static CacheControl cacheControl;
52    static {
53      cacheControl = new CacheControl();
54      cacheControl.setNoCache(true);
55      cacheControl.setNoTransform(false);
56    }

58    ResultGenerator generator;
59    String id;
60    int batch = 1;
61  
62    public ScannerInstanceResource(String table, String id, 
63        ResultGenerator generator, int batch) throws IOException {
64      this.id = id;
65      this.generator = generator;
66      this.batch = batch;
67    }

69    @GET
70    @Produces({MIMETYPE_XML, MIMETYPE_JSON, MIMETYPE_PROTOBUF})
71    public Response get(final @Context UriInfo uriInfo, 
72        @QueryParam("n") int maxRows, final @QueryParam("c") int maxValues) {
73      if (LOG.isDebugEnabled()) {
74        LOG.debug("GET " + uriInfo.getAbsolutePath());
75      }
76      servlet.getMetrics().incrementRequests(1);
77      CellSetModel model = new CellSetModel();
78      RowModel rowModel = null;
79      byte[] rowKey = null;
80      int limit = batch;
81      if (maxValues > 0) {
82        limit = maxValues;
83      }
84      int count = limit;
85      do {
86        KeyValue value = null;
87        try {
88          value = generator.next();
89        } catch (IllegalStateException e) {
90          if (ScannerResource.delete(id)) {
91            servlet.getMetrics().incrementSucessfulDeleteRequests(1);
92          } else {
93            servlet.getMetrics().incrementFailedDeleteRequests(1);
94          }
95          servlet.getMetrics().incrementFailedGetRequests(1);
96          throw new WebApplicationException(Response.Status.GONE);
97        }
98        if (value == null) {
99          LOG.info("generator exhausted");
100         // respond with 204 (No Content) if an empty cell set would be
101         // returned
102         if (count == limit) {
103           return Response.noContent().build();
104         }
105         break;
106       }
107       if (rowKey == null) {
108         rowKey = value.getRow();
109         rowModel = new RowModel(rowKey);
110       }
111       if (!Bytes.equals(value.getRow(), rowKey)) {
112         // if maxRows was given as a query param, stop if we would exceed the
113         // specified number of rows
114         if (maxRows > 0) { 
115           if (--maxRows == 0) {
116             generator.putBack(value);
117             break;
118           }
119         }
120         model.addRow(rowModel);
121         rowKey = value.getRow();
122         rowModel = new RowModel(rowKey);
123       }
124       rowModel.addCell(
125         new CellModel(value.getFamily(), value.getQualifier(), 
126           value.getTimestamp(), value.getValue()));
127     } while (--count > 0);
128     model.addRow(rowModel);
129     ResponseBuilder response = Response.ok(model);
130     response.cacheControl(cacheControl);
131     servlet.getMetrics().incrementSucessfulGetRequests(1);
132     return response.build();
133   }

135   @GET
136   @Produces(MIMETYPE_BINARY)
137   public Response getBinary(final @Context UriInfo uriInfo) {
138     if (LOG.isDebugEnabled()) {
139       LOG.debug("GET " + uriInfo.getAbsolutePath() + " as " +
140         MIMETYPE_BINARY);
141     }
142     servlet.getMetrics().incrementRequests(1);
143     try {
144       KeyValue value = generator.next();
145       if (value == null) {
146         LOG.info("generator exhausted");
147         return Response.noContent().build();
148       }
149       ResponseBuilder response = Response.ok(value.getValue());
150       response.cacheControl(cacheControl);
151       response.header("X-Row", Base64.encodeBytes(value.getRow()));      
152       response.header("X-Column", 
153         Base64.encodeBytes(
154           KeyValue.makeColumn(value.getFamily(), value.getQualifier())));
155       response.header("X-Timestamp", value.getTimestamp());
156       servlet.getMetrics().incrementSucessfulGetRequests(1);
157       return response.build();
158     } catch (IllegalStateException e) {
159       if (ScannerResource.delete(id)) {
160         servlet.getMetrics().incrementSucessfulDeleteRequests(1);
161       } else {
162         servlet.getMetrics().incrementFailedDeleteRequests(1);
163       }
164       servlet.getMetrics().incrementFailedGetRequests(1);
165       throw new WebApplicationException(Response.Status.GONE);
166     }
167   }

169   @DELETE

171     if (LOG.isDebugEnabled()) {
172       LOG.debug("DELETE " + uriInfo.getAbsolutePath());
173     }
174     servlet.getMetrics().incrementRequests(1);
175     if (servlet.isReadOnly()) {
176       throw new WebApplicationException(Response.Status.FORBIDDEN);
177     }
178     if (ScannerResource.delete(id)) {
179       servlet.getMetrics().incrementSucessfulDeleteRequests(1);
180     } else {
181       servlet.getMetrics().incrementFailedDeleteRequests(1);
182     }
183     return Response.ok().build();
184   }
