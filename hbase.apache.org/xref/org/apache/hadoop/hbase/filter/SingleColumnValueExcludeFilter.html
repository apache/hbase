35  /**
36   * A {@link Filter} that checks a single column value, but does not emit the
37   * tested column. This will enable a performance boost over
38   * {@link SingleColumnValueFilter}, if the tested column value is not actually
39   * needed as input (besides for the filtering itself).
40   */
41  @InterfaceAudience.Public
42  @InterfaceStability.Stable
43  public class SingleColumnValueExcludeFilter extends SingleColumnValueFilter {
44  
45    /**
46     * Constructor for binary compare of the value of a single column. If the
47     * column is found and the condition passes, all columns of the row will be
48     * emitted; except for the tested column value. If the column is not found or
49     * the condition fails, the row will not be emitted.
50     *
51     * @param family name of column family
52     * @param qualifier name of column qualifier
53     * @param compareOp operator
54     * @param value value to compare column values against
55     */
56    public SingleColumnValueExcludeFilter(byte[] family, byte[] qualifier,
57        CompareOp compareOp, byte[] value) {
58      super(family, qualifier, compareOp, value);
59    }
60  
61    /**
62     * Constructor for binary compare of the value of a single column. If the
63     * column is found and the condition passes, all columns of the row will be
64     * emitted; except for the tested column value. If the condition fails, the
65     * row will not be emitted.
66     * <p>
67     * Use the filterIfColumnMissing flag to set whether the rest of the columns
68     * in a row will be emitted if the specified column to check is not found in
69     * the row.
70     *
71     * @param family name of column family
72     * @param qualifier name of column qualifier
73     * @param compareOp operator
74     * @param comparator Comparator to use.
75     */
76    public SingleColumnValueExcludeFilter(byte[] family, byte[] qualifier,
77        CompareOp compareOp, ByteArrayComparable comparator) {
78      super(family, qualifier, compareOp, comparator);
79    }

81    /**
82     * Constructor for protobuf deserialization only.
83     * @param family
84     * @param qualifier
85     * @param compareOp
86     * @param comparator
87     * @param foundColumn
88     * @param matchedColumn
89     * @param filterIfMissing
90     * @param latestVersionOnly
91     */
92    protected SingleColumnValueExcludeFilter(final byte[] family, final byte [] qualifier,
93      final CompareOp compareOp, ByteArrayComparable comparator, final boolean foundColumn,
94      final boolean matchedColumn, final boolean filterIfMissing, final boolean latestVersionOnly) {
95      super(family,qualifier,compareOp,comparator,foundColumn,
96        matchedColumn,filterIfMissing,latestVersionOnly);
97    }
98  
99    public ReturnCode filterKeyValue(KeyValue keyValue) {
100     ReturnCode superRetCode = super.filterKeyValue(keyValue);
101     if (superRetCode == ReturnCode.INCLUDE) {
102       // If the current column is actually the tested column,
103       // we will skip it instead.
104       if (keyValue.matchingColumn(this.columnFamily, this.columnQualifier)) {
105         return ReturnCode.SKIP;
106       }
107     }
108     return superRetCode;
109   }

111   public static Filter createFilterFromArguments(ArrayList<byte []> filterArguments) {
112     SingleColumnValueFilter tempFilter = (SingleColumnValueFilter)
113       SingleColumnValueFilter.createFilterFromArguments(filterArguments);
114     SingleColumnValueExcludeFilter filter = new SingleColumnValueExcludeFilter (
115       tempFilter.getFamily(), tempFilter.getQualifier(),
116       tempFilter.getOperator(), tempFilter.getComparator());
117 
118     if (filterArguments.size() == 6) {
119       filter.setFilterIfMissing(tempFilter.getFilterIfMissing());
120       filter.setLatestVersionOnly(tempFilter.getLatestVersionOnly());
121     }
122     return filter;
123   }
