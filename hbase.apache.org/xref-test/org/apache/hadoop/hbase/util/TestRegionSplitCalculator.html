44    /**
45     * This is range uses a user specified start and end keys. It also has an
46     * extra time based tiebreaker so that different ranges with the same
47     * start/end key pair count as different regions.
48     */
49    static class SimpleRange implements KeyRange {
50      byte[] start, end;
51      long tiebreaker;
52  
53      SimpleRange(byte[] start, byte[] end) {
54        this.start = start;
55        this.end = end;
56        this.tiebreaker = System.nanoTime();
57      }
58  
59      @Override
60      public byte[] getStartKey() {
61        return start;
62      }
63  
64      @Override
65      public byte[] getEndKey() {
66        return end;
67      }
68  
69      public String toString() {
70        return "[" + Bytes.toString(start) + ", " + Bytes.toString(end) + "]";
71      }
72    }
73  
74    Comparator<SimpleRange> cmp = new Comparator<SimpleRange>() {
75      @Override
76      public int compare(SimpleRange sr1, SimpleRange sr2) {
77        ComparisonChain cc = ComparisonChain.start();
78        cc = cc.compare(sr1.getStartKey(), sr2.getStartKey(),
79            Bytes.BYTES_COMPARATOR);
80        cc = cc.compare(sr1.getEndKey(), sr2.getEndKey(),
81            RegionSplitCalculator.BYTES_COMPARATOR);
82        cc = cc.compare(sr1.tiebreaker, sr2.tiebreaker);
83        return cc.result();
84      }
85    };

87    /**
88     * Check the "depth" (number of regions included at a split) of a generated
89     * split calculation
90     */
91    void checkDepths(SortedSet<byte[]> splits,
92        Multimap<byte[], SimpleRange> regions, Integer... depths) {
93      assertEquals(splits.size(), depths.length);
94      int i = 0;
95      for (byte[] k : splits) {
96        Collection<SimpleRange> rs = regions.get(k);
97        int sz = rs == null ? 0 : rs.size();
98        assertEquals((int) depths[i], sz);
99        i++;
100     }
101   }

103   /**
104    * This dumps data in a visually reasonable way for visual debugging. It has
105    * the basic iteration structure.
106    */
107   String dump(SortedSet<byte[]> splits, Multimap<byte[], SimpleRange> regions) {
108     // we display this way because the last end key should be displayed as well.
109     StringBuilder sb = new StringBuilder();
110     for (byte[] k : splits) {
111       sb.append(Bytes.toString(k) + ":\t");
112       for (SimpleRange r : regions.get(k)) {
113         sb.append(r.toString() + "\t");
114       }
115       sb.append("\n");
116     }
117     String s = sb.toString();
118     LOG.info("\n" + s);
119     return s;
120   }

357   @Test
358   public void testBigRanges() {
359     SimpleRange ai = new SimpleRange(Bytes.toBytes("A"), Bytes.toBytes("I"));
360     SimpleRange ae = new SimpleRange(Bytes.toBytes("A"), Bytes.toBytes("E"));
361     SimpleRange ac = new SimpleRange(Bytes.toBytes("A"), Bytes.toBytes("C"));
362 
363     Collection<SimpleRange> bigOverlap = new ArrayList<SimpleRange>();
364     bigOverlap.add(new SimpleRange(Bytes.toBytes("A"), Bytes.toBytes("E")));
365     bigOverlap.add(new SimpleRange(Bytes.toBytes("A"), Bytes.toBytes("C")));
366     bigOverlap.add(new SimpleRange(Bytes.toBytes("A"), Bytes.toBytes("B")));
367     bigOverlap.add(new SimpleRange(Bytes.toBytes("B"), Bytes.toBytes("C")));
368     bigOverlap.add(new SimpleRange(Bytes.toBytes("E"), Bytes.toBytes("H")));
369     bigOverlap.add(ai);
370     bigOverlap.add(ae);
371     bigOverlap.add(ac);
372 
373     // Expect 1 range to be returned: ai
374     List<SimpleRange> bigRanges = RegionSplitCalculator.findBigRanges(bigOverlap, 1);
375     assertEquals(1, bigRanges.size());
376     assertEquals(ai, bigRanges.get(0));
377 
378     // Expect 3 ranges to be returned: ai, ae and ac
379     bigRanges = RegionSplitCalculator.findBigRanges(bigOverlap, 3);
380     assertEquals(3, bigRanges.size());
381     assertEquals(ai, bigRanges.get(0));
382 
383     SimpleRange r1 = bigRanges.get(1);
384     SimpleRange r2 = bigRanges.get(2);
385     assertEquals(Bytes.toString(r1.start), "A");
386     assertEquals(Bytes.toString(r2.start), "A");
387     String r1e = Bytes.toString(r1.end);
388     String r2e = Bytes.toString(r2.end);
389     assertTrue((r1e.equals("C") && r2e.equals("E"))
390       || (r1e.equals("E") && r2e.equals("C")));
391   }
