39  /**
40   * This class is a wrapper for the implementation of
41   * com.sun.management.UnixOperatingSystemMXBean
42   * It will decide to use the sun api or its own implementation
43   * depending on the runtime (vendor) used.
44   */
45  
46  @InterfaceAudience.Public
47  @InterfaceStability.Evolving
48  public class JVM 
49  {
50    static final Logger LOG = LoggerFactory.getLogger(JVM.class);
51  
52    private OperatingSystemMXBean osMbean;
53  
54    private static final boolean ibmvendor =
55      System.getProperty("java.vendor").contains("IBM");
56    private static final boolean windows = 
57      System.getProperty("os.name").startsWith("Windows");
58    private static final boolean linux =
59      System.getProperty("os.name").startsWith("Linux");
60  
61    /**
62     * Constructor. Get the running Operating System instance
63     */
64    public JVM () {
65      this.osMbean = ManagementFactory.getOperatingSystemMXBean();
66    }
67   
68    /**
69     * Check if the OS is unix. 
70     * 
71     * @return whether this is unix or not.
72     */
73    public boolean isUnix() {
74      if (windows) {
75        return false;
76      }
77      return (ibmvendor ? linux : true);
78    }
79  
80    /**
81     * Load the implementation of UnixOperatingSystemMXBean for Oracle jvm
82     * and runs the desired method. 
83     * @param mBeanMethodName : method to run from the interface UnixOperatingSystemMXBean
84     * @return the method result
85     */
86    private Long runUnixMXBeanMethod (String mBeanMethodName) {
87    
88      Object unixos;
89      Class<?> classRef;
90      Method mBeanMethod;
91  
92      try {
93        classRef = Class.forName("com.sun.management.UnixOperatingSystemMXBean");
94        if (classRef.isInstance(osMbean)) {
95          mBeanMethod = classRef.getDeclaredMethod(mBeanMethodName,
96            new Class[0]);
97          unixos = classRef.cast(osMbean);
98          return (Long)mBeanMethod.invoke(unixos);
99        }
100     }
101     catch(Exception e) {
102       LOG.warn("Not able to load class or method for com.sun.managment.UnixOperatingSystemMXBean.", e);
103     }
104     return null;
105   }
106 
107   /**
108    * Get the number of opened filed descriptor for the runtime jvm.
109    * If Oracle java, it will use the com.sun.management interfaces.
110    * Otherwise, this methods implements it (linux only).  
111    * @return number of open file descriptors for the jvm
112    */
113   public long getOpenFileDescriptorCount() {
114 
115     Long ofdc;
116     
117     if (!ibmvendor) {
118       ofdc = runUnixMXBeanMethod("getOpenFileDescriptorCount");
119       return (ofdc != null ? ofdc.longValue () : -1);
120     }
121     try {
122       //need to get the PID number of the process first
123       RuntimeMXBean rtmbean = ManagementFactory.getRuntimeMXBean();
124       String rtname = rtmbean.getName();
125       String[] pidhost = rtname.split("@");
126 
127       //using linux bash commands to retrieve info
128       Process p = Runtime.getRuntime().exec(
129       new String[] { "bash", "-c",
130           "ls /proc/" + pidhost[0] + "/fdinfo | wc -l" });
131       InputStream in = p.getInputStream();
132       BufferedReader output = new BufferedReader(
133         		new InputStreamReader(in));
134 
135       String openFileDesCount;
136       if ((openFileDesCount = output.readLine()) != null)      
137              return Long.parseLong(openFileDesCount);
138      } catch (IOException ie) {
139      	     LOG.warn("Not able to get the number of open file descriptors", ie);
140     }
141     return -1;
142   }
143 
144   /**
145    * Get the number of the maximum file descriptors the system can use.
146    * If Oracle java, it will use the com.sun.management interfaces.
147    * Otherwise, this methods implements it (linux only).  
148    * @return max number of file descriptors the operating system can use.
149    */
150   public long getMaxFileDescriptorCount() {
151 
152     Long mfdc;
153 
154     if (!ibmvendor) {
155       mfdc = runUnixMXBeanMethod("getMaxFileDescriptorCount");
156       return (mfdc != null ? mfdc.longValue () : -1);
157     }
158     try {
159       
160       //using linux bash commands to retrieve info
161       Process p = Runtime.getRuntime().exec(
162         	  new String[] { "bash", "-c",
163         	  "ulimit -n" });
164       InputStream in = p.getInputStream();
165       BufferedReader output = new BufferedReader(
166         new InputStreamReader(in));
167 
168       String maxFileDesCount;
169       if ((maxFileDesCount = output.readLine()) != null)      
170         	return Long.parseLong(maxFileDesCount);
171     }   catch (IOException ie) {
172       		LOG.warn("Not able to get the max number of file descriptors", ie);
173     }
174     return -1;
175  }
