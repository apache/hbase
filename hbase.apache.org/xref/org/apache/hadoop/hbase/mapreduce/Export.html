44  /**
45  * Export an HBase table.
46  * Writes content to sequence files up in HDFS.  Use {@link Import} to read it
47  * back in again.
48  */
49  @InterfaceAudience.Public
50  @InterfaceStability.Stable
51  public class Export {
52    private static final Log LOG = LogFactory.getLog(Export.class);
53    final static String NAME = "export";
54    final static String RAW_SCAN = "hbase.mapreduce.include.deleted.rows";
55    final static String EXPORT_BATCHING = "hbase.export.scanner.batch";
56  
57    /**
58     * Mapper.
59     */
60    static class Exporter
61    extends TableMapper<ImmutableBytesWritable, Result> {
62      /**
63       * @param row  The current table row key.
64       * @param value  The columns.
65       * @param context  The current context.
66       * @throws IOException When something is broken with the data.
67       * @see org.apache.hadoop.mapreduce.Mapper#map(KEYIN, VALUEIN,
68       *   org.apache.hadoop.mapreduce.Mapper.Context)
69       */
70      @Override
71      public void map(ImmutableBytesWritable row, Result value,
72        Context context)
73      throws IOException {
74        try {
75          context.write(row, value);
76        } catch (InterruptedException e) {
77          e.printStackTrace();
78        }
79      }
80    }
81  
82    /**
83     * Sets up the actual job.
84     *
85     * @param conf  The current configuration.
86     * @param args  The command line parameters.
87     * @return The newly created job.
88     * @throws IOException When setting up the job fails.
89     */
90    public static Job createSubmittableJob(Configuration conf, String[] args)
91    throws IOException {
92      String tableName = args[0];
93      Path outputDir = new Path(args[1]);
94      Job job = new Job(conf, NAME + "_" + tableName);
95      job.setJobName(NAME + "_" + tableName);
96      job.setJarByClass(Exporter.class);
97      // Set optional scan parameters
98      Scan s = getConfiguredScanForJob(conf, args);
99      TableMapReduceUtil.initTableMapperJob(tableName, s, Exporter.class, null,
100       null, job);
101     // No reducers.  Just write straight to output files.
102     job.setNumReduceTasks(0);
103     job.setOutputFormatClass(SequenceFileOutputFormat.class);
104     job.setOutputKeyClass(ImmutableBytesWritable.class);
105     job.setOutputValueClass(Result.class);
106     FileOutputFormat.setOutputPath(job, outputDir);
107     return job;
108   }

110   private static Scan getConfiguredScanForJob(Configuration conf, String[] args) throws IOException {
111     Scan s = new Scan();
112     // Optional arguments.
113     // Set Scan Versions
114     int versions = args.length > 2? Integer.parseInt(args[2]): 1;
115     s.setMaxVersions(versions);
116     // Set Scan Range
117     long startTime = args.length > 3? Long.parseLong(args[3]): 0L;
118     long endTime = args.length > 4? Long.parseLong(args[4]): Long.MAX_VALUE;
119     s.setTimeRange(startTime, endTime);
120     // Set cache blocks
121     s.setCacheBlocks(false);
122     // Set Scan Column Family
123     boolean raw = Boolean.parseBoolean(conf.get(RAW_SCAN));
124     if (raw) {
125       s.setRaw(raw);
126     }
127     
128     if (conf.get(TableInputFormat.SCAN_COLUMN_FAMILY) != null) {
129       s.addFamily(Bytes.toBytes(conf.get(TableInputFormat.SCAN_COLUMN_FAMILY)));
130     }
131     // Set RowFilter or Prefix Filter if applicable.
132     Filter exportFilter = getExportFilter(args);
133     if (exportFilter!= null) {
134         LOG.info("Setting Scan Filter for Export.");
135       s.setFilter(exportFilter);
136     }
137 
138     int batching = conf.getInt(EXPORT_BATCHING, -1);
139     if (batching !=  -1){
140       try{
141         s.setBatch(batching);
142 	} catch (RuntimeException e) {
143 	    LOG.error("Batching could not be set", e);
144       }
145     }
146     LOG.info("versions=" + versions + ", starttime=" + startTime +
147       ", endtime=" + endTime + ", keepDeletedCells=" + raw);
148     return s;
149   }
150 
151   private static Filter getExportFilter(String[] args) {
152     Filter exportFilter = null;
153     String filterCriteria = (args.length > 5) ? args[5]: null;
154     if (filterCriteria == null) return null;
155     if (filterCriteria.startsWith("^")) {
156       String regexPattern = filterCriteria.substring(1, filterCriteria.length());
157       exportFilter = new RowFilter(CompareOp.EQUAL, new RegexStringComparator(regexPattern));
158     } else {
159       exportFilter = new PrefixFilter(Bytes.toBytes(filterCriteria));
160     }
161     return exportFilter;
162   }

164   /*
165    * @param errorMsg Error message.  Can be null.
166    */
167   private static void usage(final String errorMsg) {
168     if (errorMsg != null && errorMsg.length() > 0) {
169       System.err.println("ERROR: " + errorMsg);
170     }
171     System.err.println("Usage: Export [-D <property=value>]* <tablename> <outputdir> [<versions> " +
172       "[<starttime> [<endtime>]] [^[regex pattern] or [Prefix] to filter]]\n");
173     System.err.println("  Note: -D properties will be applied to the conf used. ");
174     System.err.println("  For example: ");
175     System.err.println("   -D mapred.output.compress=true");
176     System.err.println("   -D mapred.output.compression.codec=org.apache.hadoop.io.compress.GzipCodec");
177     System.err.println("   -D mapred.output.compression.type=BLOCK");
178     System.err.println("  Additionally, the following SCAN properties can be specified");
179     System.err.println("  to control/limit what is exported..");
180     System.err.println("   -D " + TableInputFormat.SCAN_COLUMN_FAMILY + "=<familyName>");
181     System.err.println("   -D " + RAW_SCAN + "=true");
182     System.err.println("For performance consider the following properties:\n"
183         + "   -Dhbase.client.scanner.caching=100\n"
184         + "   -Dmapred.map.tasks.speculative.execution=false\n"
185         + "   -Dmapred.reduce.tasks.speculative.execution=false");
186     System.err.println("For tables with very wide rows consider setting the batch size as below:\n"
187         + "   -D" + EXPORT_BATCHING + "=10");
188   }

190   /**
191    * Main entry point.
192    *
193    * @param args  The command line parameters.
194    * @throws Exception When running the job fails.
195    */
196   public static void main(String[] args) throws Exception {
197     Configuration conf = HBaseConfiguration.create();
198     String[] otherArgs = new GenericOptionsParser(conf, args).getRemainingArgs();
199     if (otherArgs.length < 2) {
200       usage("Wrong number of arguments: " + otherArgs.length);
201       System.exit(-1);
202     }
203     Job job = createSubmittableJob(conf, otherArgs);
204     System.exit(job.waitForCompletion(true)? 0 : 1);
205   }
