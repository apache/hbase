55  /**
56   * Reading {@link HFile} version 1 and 2 blocks, and writing version 2 blocks.
57   * <ul>
58   * <li>In version 1 all blocks are always compressed or uncompressed, as
59   * specified by the {@link HFile}'s compression algorithm, with a type-specific
60   * magic record stored in the beginning of the compressed data (i.e. one needs
61   * to uncompress the compressed block to determine the block type). There is
62   * only a single compression algorithm setting for all blocks. Offset and size
63   * information from the block index are required to read a block.
64   * <li>In version 2 a block is structured as follows:
65   * <ul>
66   * <li>Magic record identifying the block type (8 bytes)
67   * <li>Compressed block size, header not included (4 bytes)
68   * <li>Uncompressed block size, header not included (4 bytes)
69   * <li>The offset of the previous block of the same type (8 bytes). This is
70   * used to be able to navigate to the previous block without going to the block
71   * <li>For minorVersions >=1, there is an additional 4 byte field 
72   * bytesPerChecksum that records the number of bytes in a checksum chunk.
73   * <li>For minorVersions >=1, there is a 4 byte value to store the size of
74   * data on disk (excluding the checksums)
75   * <li>For minorVersions >=1, a series of 4 byte checksums, one each for
76   * the number of bytes specified by bytesPerChecksum.
77   * index.
78   * <li>Compressed data (or uncompressed data if compression is disabled). The
79   * compression algorithm is the same for all the blocks in the {@link HFile},
80   * similarly to what was done in version 1.
81   * </ul>
82   * </ul>
83   * The version 2 block representation in the block cache is the same as above,
84   * except that the data section is always uncompressed in the cache.
85   */

245   /**
246    * Creates a block from an existing buffer starting with a header. Rewinds
247    * and takes ownership of the buffer. By definition of rewind, ignores the
248    * buffer position, but if you slice the buffer beforehand, it will rewind
249    * to that point. The reason this has a minorNumber and not a majorNumber is
250    * because majorNumbers indicate the format of a HFile whereas minorNumbers 
251    * indicate the format inside a HFileBlock.
252    */
253   HFileBlock(ByteBuffer b, int minorVersion) throws IOException {
254     b.rewind();
255     blockType = BlockType.read(b);
256     onDiskSizeWithoutHeader = b.getInt();
257     uncompressedSizeWithoutHeader = b.getInt();
258     prevBlockOffset = b.getLong();
259     this.minorVersion = minorVersion;
260     if (minorVersion >= MINOR_VERSION_WITH_CHECKSUM) {
261       this.checksumType = b.get();
262       this.bytesPerChecksum = b.getInt();
263       this.onDiskDataSizeWithHeader = b.getInt();
264     } else {
265       this.checksumType = ChecksumType.NULL.getCode();
266       this.bytesPerChecksum = 0;
267       this.onDiskDataSizeWithHeader = onDiskSizeWithoutHeader +
268                                        HEADER_SIZE_NO_CHECKSUM;
269     }
270     buf = b;
271     buf.rewind();
272   }

381   /**
382    * Checks if the block is internally consistent, i.e. the first
383    * {@link #HEADER_SIZE} bytes of the buffer contain a valid header consistent
384    * with the fields. This function is primary for testing and debugging, and
385    * is not thread-safe, because it alters the internal buffer pointer.
386    */
387   void sanityCheck() throws IOException {
388     buf.rewind();
389 
390     {
391       BlockType blockTypeFromBuf = BlockType.read(buf);
392       if (blockTypeFromBuf != blockType) {
393         throw new IOException("Block type stored in the buffer: " +
394             blockTypeFromBuf + ", block type field: " + blockType);
395       }
396     }
397 
398     sanityCheckAssertion(buf.getInt(), onDiskSizeWithoutHeader,
399         "onDiskSizeWithoutHeader");
400 
401     sanityCheckAssertion(buf.getInt(), uncompressedSizeWithoutHeader,
402         "uncompressedSizeWithoutHeader");
403 
404     sanityCheckAssertion(buf.getLong(), prevBlockOffset, "prevBlocKOffset");
405     if (minorVersion >= MINOR_VERSION_WITH_CHECKSUM) {
406       sanityCheckAssertion(buf.get(), checksumType, "checksumType");
407       sanityCheckAssertion(buf.getInt(), bytesPerChecksum, "bytesPerChecksum");
408       sanityCheckAssertion(buf.getInt(), onDiskDataSizeWithHeader, 
409                            "onDiskDataSizeWithHeader");
410     }
411 
412     int cksumBytes = totalChecksumBytes();
413     int hdrSize = headerSize();
414     int expectedBufLimit = uncompressedSizeWithoutHeader + headerSize() +
415                            cksumBytes;
416     if (buf.limit() != expectedBufLimit) {
417       throw new AssertionError("Expected buffer limit " + expectedBufLimit
418           + ", got " + buf.limit());
419     }
420 
421     // We might optionally allocate HEADER_SIZE more bytes to read the next
422     // block's, header, so there are two sensible values for buffer capacity.
423     int size = uncompressedSizeWithoutHeader + hdrSize + cksumBytes;
424     if (buf.capacity() != size &&
425         buf.capacity() != size + hdrSize) {
426       throw new AssertionError("Invalid buffer capacity: " + buf.capacity() +
427           ", expected " + size + " or " + (size + hdrSize));
428     }
429   }

461   /**
462    * Always allocates a new buffer of the correct size. Copies header bytes
463    * from the existing buffer. Does not change header fields. 
464    * Reserve room to keep checksum bytes too.
465    *
466    * @param extraBytes whether to reserve room in the buffer to read the next
467    *          block's header
468    */
469   private void allocateBuffer(boolean extraBytes) {
470     int cksumBytes = totalChecksumBytes();
471     int capacityNeeded = headerSize() + uncompressedSizeWithoutHeader +
472         cksumBytes +
473         (extraBytes ? headerSize() : 0);
474 
475     ByteBuffer newBuf = ByteBuffer.allocate(capacityNeeded);
476 
477     // Copy header bytes.
478     System.arraycopy(buf.array(), buf.arrayOffset(), newBuf.array(),
479         newBuf.arrayOffset(), headerSize());
480 
481     buf = newBuf;
482     buf.limit(headerSize() + uncompressedSizeWithoutHeader + cksumBytes);
483   }

524   @Override
525   public long heapSize() {
526     long size = ClassSize.align(
527         ClassSize.OBJECT +
528         // Block type and byte buffer references
529         2 * ClassSize.REFERENCE +
530         // On-disk size, uncompressed size, and next block's on-disk size
531         // bytePerChecksum,  onDiskDataSize and minorVersion
532         6 * Bytes.SIZEOF_INT +
533         // Checksum type
534         1 * Bytes.SIZEOF_BYTE +
535         // This and previous block offset
536         2 * Bytes.SIZEOF_LONG +
537         // "Include memstore timestamp" flag
538         Bytes.SIZEOF_BOOLEAN
539     );
540 
541     if (buf != null) {
542       // Deep overhead of the byte buffer. Needs to be aligned separately.
543       size += ClassSize.align(buf.capacity() + BYTE_BUFFER_HEAP_SIZE);
544     }
545 
546     return ClassSize.align(size);
547   }

549   /**
550    * Read from an input stream. Analogous to
551    * {@link IOUtils#readFully(InputStream, byte[], int, int)}, but specifies a
552    * number of "extra" bytes that would be desirable but not absolutely
553    * necessary to read.
554    *
555    * @param in the input stream to read from
556    * @param buf the buffer to read into
557    * @param bufOffset the destination offset in the buffer
558    * @param necessaryLen the number of bytes that are absolutely necessary to
559    *          read
560    * @param extraLen the number of extra bytes that would be nice to read
561    * @return true if succeeded reading the extra bytes
562    * @throws IOException if failed to read the necessary bytes
563    */
564   public static boolean readWithExtra(InputStream in, byte buf[],
565       int bufOffset, int necessaryLen, int extraLen) throws IOException {
566     int bytesRemaining = necessaryLen + extraLen;
567     while (bytesRemaining > 0) {
568       int ret = in.read(buf, bufOffset, bytesRemaining);
569       if (ret == -1 && bytesRemaining <= extraLen) {
570         // We could not read the "extra data", but that is OK.
571         break;
572       }
573 
574       if (ret < 0) {
575         throw new IOException("Premature EOF from inputStream (read "
576             + "returned " + ret + ", was trying to read " + necessaryLen
577             + " necessary bytes and " + extraLen + " extra bytes, "
578             + "successfully read "
579             + (necessaryLen + extraLen - bytesRemaining));
580       }
581       bufOffset += ret;
582       bytesRemaining -= ret;
583     }
584     return bytesRemaining <= 0;
585   }

597    * Unified version 2 {@link HFile} block writer. The intended usage pattern
598    * is as follows:
599    * <ol>
600    * <li>Construct an {@link HFileBlock.Writer}, providing a compression algorithm.
601    * <li>Call {@link Writer#startWriting} and get a data stream to write to.
602    * <li>Write your data into the stream.
603    * <li>Call {@link Writer#writeHeaderAndData(FSDataOutputStream)} as many times as you need to.
604    * store the serialized block into an external stream.
605    * <li>Repeat to write more blocks.
606    * </ol>
607    * <p>
608    */

610

612       INIT,

636     /**

704           ? dataBlockEncoder : NoOpDataBlockEncoder.INSTANCE;

728     /**

730      *
731      * @return the stream the user can write their data into
732      * @throws IOException
733      */

735         throws IOException {
736       if (state == State.BLOCK_READY && startOffset != -1) {
737         // We had a previous block that was written to a stream at a specific
738         // offset. Save that offset as the last offset of a block of that type.
739         prevOffsetByType[blockType.getId()] = startOffset;
740       }
741 
742       startOffset = -1;
743       blockType = newBlockType;

745       baosInMemory.reset();
746       baosInMemory.write(DUMMY_HEADER);
747 
748       state = State.WRITING;
749 
750       // We will compress it later in finishBlock()
751       userDataStream = new DataOutputStream(baosInMemory);
752       return userDataStream;
753     }

767     /**
768      * Transitions the block writer from the "writing" state to the "block
769      * ready" state.  Does nothing if a block is already finished.
770      */
771     private void ensureBlockReady() throws IOException {
772       Preconditions.checkState(state != State.INIT,
773           "Unexpected state: " + state);
774 
775       if (state == State.BLOCK_READY)
776         return;
777 
778       // This will set state to BLOCK_READY.
779       finishBlock();
780     }

782     /**
783      * An internal method that flushes the compressing stream (if using
784      * compression), serializes the header, and takes care of the separate
785      * uncompressed stream for caching on write, if applicable. Sets block
786      * write state to "block ready".
787      */
788     private void finishBlock() throws IOException {
789       userDataStream.flush();
790       // This does an array copy, so it is safe to cache this byte array.
791       uncompressedBytesWithHeader = baosInMemory.toByteArray();
792       prevOffset = prevOffsetByType[blockType.getId()];
793 
794       // We need to set state before we can package the block up for
795       // cache-on-write. In a way, the block is ready, but not yet encoded or
796       // compressed.
797       state = State.BLOCK_READY;
798       if (blockType == BlockType.DATA) {
799         encodeDataBlockForDisk();
800       } else {
801         defaultBlockEncodingCtx.compressAfterEncoding(
802             uncompressedBytesWithHeader, blockType);
803         onDiskBytesWithHeader =
804           defaultBlockEncodingCtx.getOnDiskBytesWithHeader();
805       }
806 
807       int numBytes = (int) ChecksumUtil.numBytes(
808           onDiskBytesWithHeader.length,
809           bytesPerChecksum);
810 
811       // put the header for on disk bytes
812       putHeader(onDiskBytesWithHeader, 0,
813           onDiskBytesWithHeader.length + numBytes,
814           uncompressedBytesWithHeader.length, onDiskBytesWithHeader.length);
815       // set the header for the uncompressed bytes (for cache-on-write)
816       putHeader(uncompressedBytesWithHeader, 0,
817           onDiskBytesWithHeader.length + numBytes,
818           uncompressedBytesWithHeader.length, onDiskBytesWithHeader.length);
819 
820       onDiskChecksum = new byte[numBytes];
821       ChecksumUtil.generateChecksums(
822           onDiskBytesWithHeader, 0, onDiskBytesWithHeader.length,
823           onDiskChecksum, 0, checksumType, bytesPerChecksum);
824     }

826     /**
827      * Encodes this block if it is a data block and encoding is turned on in
828      * {@link #dataBlockEncoder}.
829      */
830     private void encodeDataBlockForDisk() throws IOException {
831       // do data block encoding, if data block encoder is set
832       ByteBuffer rawKeyValues =
833           ByteBuffer.wrap(uncompressedBytesWithHeader, HEADER_SIZE,
834               uncompressedBytesWithHeader.length - HEADER_SIZE).slice();
835 
836       //do the encoding
837       dataBlockEncoder.beforeWriteToDisk(rawKeyValues,
838               includesMemstoreTS, dataBlockEncodingCtx, blockType);
839 
840       uncompressedBytesWithHeader =
841           dataBlockEncodingCtx.getUncompressedBytesWithHeader();
842       onDiskBytesWithHeader =
843           dataBlockEncodingCtx.getOnDiskBytesWithHeader();
844       blockType = dataBlockEncodingCtx.getBlockType();
845     }

847     /**
848      * Put the header into the given byte array at the given offset.
849      * @param onDiskSize size of the block on disk header + data + checksum
850      * @param uncompressedSize size of the block after decompression (but
851      *          before optional data block decoding) including header
852      * @param onDiskDataSize size of the block on disk with header
853      *        and data but not including the checksums
854      */
855     private void putHeader(byte[] dest, int offset, int onDiskSize,
856         int uncompressedSize, int onDiskDataSize) {
857       offset = blockType.put(dest, offset);
858       offset = Bytes.putInt(dest, offset, onDiskSize - HEADER_SIZE);
859       offset = Bytes.putInt(dest, offset, uncompressedSize - HEADER_SIZE);
860       offset = Bytes.putLong(dest, offset, prevOffset);
861       offset = Bytes.putByte(dest, offset, checksumType.getCode());
862       offset = Bytes.putInt(dest, offset, bytesPerChecksum);
863       Bytes.putInt(dest, offset, onDiskDataSize);
864     }

1193     /**
1194      * Does a positional read or a seek and read into the given buffer. Returns
1195      * the on-disk size of the next block, or -1 if it could not be determined.
1196      *
1197      * @param dest destination buffer
1198      * @param destOffset offset in the destination buffer
1199      * @param size size of the block to be read
1200      * @param peekIntoNextBlock whether to read the next block's on-disk size
1201      * @param fileOffset position in the stream to read at
1202      * @param pread whether we should do a positional read
1203      * @param istream The input source of data
1204      * @return the on-disk size of the next block with header size included, or
1205      *         -1 if it could not be determined
1206      * @throws IOException
1207      */
1208     protected int readAtOffset(FSDataInputStream istream,
1209         byte[] dest, int destOffset, int size,
1210         boolean peekIntoNextBlock, long fileOffset, boolean pread)
1211         throws IOException {
1212       if (peekIntoNextBlock &&
1213           destOffset + size + hdrSize > dest.length) {
1214         // We are asked to read the next block's header as well, but there is
1215         // not enough room in the array.
1216         throw new IOException("Attempted to read " + size + " bytes and " +
1217             hdrSize + " bytes of next header into a " + dest.length +
1218             "-byte array at offset " + destOffset);
1219       }
1220 
1221       if (!pread && streamLock.tryLock()) {
1222         // Seek + read. Better for scanning.
1223         try {
1224           istream.seek(fileOffset);
1225 
1226           long realOffset = istream.getPos();
1227           if (realOffset != fileOffset) {
1228             throw new IOException("Tried to seek to " + fileOffset + " to "
1229                 + "read " + size + " bytes, but pos=" + realOffset
1230                 + " after seek");
1231           }
1232 
1233           if (!peekIntoNextBlock) {
1234             IOUtils.readFully(istream, dest, destOffset, size);
1235             return -1;
1236           }
1237 
1238           // Try to read the next block header.
1239           if (!readWithExtra(istream, dest, destOffset, size, hdrSize))
1240             return -1;
1241         } finally {
1242           streamLock.unlock();
1243         }
1244       } else {
1245         // Positional read. Better for random reads; or when the streamLock is already locked.
1246         int extraSize = peekIntoNextBlock ? hdrSize : 0;
1247 
1248         int ret = istream.read(fileOffset, dest, destOffset, size + extraSize);
1249         if (ret < size) {
1250           throw new IOException("Positional read of " + size + " bytes " +
1251               "failed at offset " + fileOffset + " (returned " + ret + ")");
1252         }
1253 
1254         if (ret == size || ret < size + extraSize) {
1255           // Could not read the next block's header, or did not try.
1256           return -1;
1257         }
1258       }
1259 
1260       assert peekIntoNextBlock;
1261       return Bytes.toInt(dest, destOffset + size + BlockType.MAGIC_LENGTH) +
1262           hdrSize;
1263     }

1307     /**
1308      * Read a version 1 block. There is no uncompressed header, and the block
1309      * type (the magic record) is part of the compressed data. This
1310      * implementation assumes that the bounded range file input stream is
1311      * needed to stop the decompressor reading into next block, because the
1312      * decompressor just grabs a bunch of data without regard to whether it is
1313      * coming to end of the compressed section.
1314      *
1315      * The block returned is still a version 2 block, and in particular, its
1316      * first {@link #HEADER_SIZE} bytes contain a valid version 2 header.
1317      *
1318      * @param offset the offset of the block to read in the file
1319      * @param onDiskSizeWithMagic the on-disk size of the version 1 block,
1320      *          including the magic record, which is the part of compressed
1321      *          data if using compression
1322      * @param uncompressedSizeWithMagic uncompressed size of the version 1
1323      *          block, including the magic record
1324      */
1325     @Override
1326     public HFileBlock readBlockData(long offset, long onDiskSizeWithMagic,
1327         int uncompressedSizeWithMagic, boolean pread) throws IOException {
1328       if (uncompressedSizeWithMagic <= 0) {
1329         throw new IOException("Invalid uncompressedSize="
1330             + uncompressedSizeWithMagic + " for a version 1 block");
1331       }
1332 
1333       if (onDiskSizeWithMagic <= 0 || onDiskSizeWithMagic >= Integer.MAX_VALUE)
1334       {
1335         throw new IOException("Invalid onDiskSize=" + onDiskSizeWithMagic
1336             + " (maximum allowed: " + Integer.MAX_VALUE + ")");
1337       }
1338 
1339       int onDiskSize = (int) onDiskSizeWithMagic;
1340 
1341       if (uncompressedSizeWithMagic < MAGIC_LENGTH) {
1342         throw new IOException("Uncompressed size for a version 1 block is "
1343             + uncompressedSizeWithMagic + " but must be at least "
1344             + MAGIC_LENGTH);
1345       }
1346 
1347       // The existing size already includes magic size, and we are inserting
1348       // a version 2 header.
1349       ByteBuffer buf = ByteBuffer.allocate(uncompressedSizeWithMagic
1350           + HEADER_DELTA);
1351 
1352       int onDiskSizeWithoutHeader;
1353       if (compressAlgo == Compression.Algorithm.NONE) {
1354         // A special case when there is no compression.
1355         if (onDiskSize != uncompressedSizeWithMagic) {
1356           throw new IOException("onDiskSize=" + onDiskSize
1357               + " and uncompressedSize=" + uncompressedSizeWithMagic
1358               + " must be equal for version 1 with no compression");
1359         }
1360 
1361         // The first MAGIC_LENGTH bytes of what this will read will be
1362         // overwritten.
1363         readAtOffset(istream, buf.array(), buf.arrayOffset() + HEADER_DELTA,
1364             onDiskSize, false, offset, pread);
1365 
1366         onDiskSizeWithoutHeader = uncompressedSizeWithMagic - MAGIC_LENGTH;
1367       } else {
1368         InputStream bufferedBoundedStream = createBufferedBoundedStream(
1369             offset, onDiskSize, pread);
1370         Compression.decompress(buf.array(), buf.arrayOffset()
1371             + HEADER_DELTA, bufferedBoundedStream, onDiskSize,
1372             uncompressedSizeWithMagic, this.compressAlgo);
1373 
1374         // We don't really have a good way to exclude the "magic record" size
1375         // from the compressed block's size, since it is compressed as well.
1376         onDiskSizeWithoutHeader = onDiskSize;
1377       }
1378 
1379       BlockType newBlockType = BlockType.parse(buf.array(), buf.arrayOffset()
1380           + HEADER_DELTA, MAGIC_LENGTH);
1381 
1382       // We set the uncompressed size of the new HFile block we are creating
1383       // to the size of the data portion of the block without the magic record,
1384       // since the magic record gets moved to the header.
1385       HFileBlock b = new HFileBlock(newBlockType, onDiskSizeWithoutHeader,
1386           uncompressedSizeWithMagic - MAGIC_LENGTH, -1L, buf, FILL_HEADER,
1387           offset, MemStore.NO_PERSISTENT_TS, 0, 0, ChecksumType.NULL.getCode(),
1388           onDiskSizeWithoutHeader + HEADER_SIZE_NO_CHECKSUM);
1389       return b;
1390     }
1391   }

1478     /**
1479      * Reads a version 2 block. Tries to do as little memory allocation as
1480      * possible, using the provided on-disk size.
1481      *
1482      * @param offset the offset in the stream to read at
1483      * @param onDiskSizeWithHeaderL the on-disk size of the block, including
1484      *          the header, or -1 if unknown
1485      * @param uncompressedSize the uncompressed size of the the block. Always
1486      *          expected to be -1. This parameter is only used in version 1.
1487      * @param pread whether to use a positional read
1488      */
1489     @Override
1490     public HFileBlock readBlockData(long offset, long onDiskSizeWithHeaderL,
1491         int uncompressedSize, boolean pread) throws IOException {
1492 
1493       // It is ok to get a reference to the stream here without any
1494       // locks because it is marked final.
1495       FSDataInputStream is = this.istreamNoFsChecksum;
1496 
1497       // get a copy of the current state of whether to validate
1498       // hbase checksums or not for this read call. This is not 
1499       // thread-safe but the one constaint is that if we decide 
1500       // to skip hbase checksum verification then we are 
1501       // guaranteed to use hdfs checksum verification.
1502       boolean doVerificationThruHBaseChecksum = this.useHBaseChecksum;
1503       if (!doVerificationThruHBaseChecksum) {
1504         is = this.istream;
1505       }
1506                      
1507       HFileBlock blk = readBlockDataInternal(is, offset, 
1508                          onDiskSizeWithHeaderL, 
1509                          uncompressedSize, pread,
1510                          doVerificationThruHBaseChecksum);
1511       if (blk == null) {
1512         HFile.LOG.warn("HBase checksum verification failed for file " +
1513                        path + " at offset " +
1514                        offset + " filesize " + fileSize +
1515                        ". Retrying read with HDFS checksums turned on...");
1516 
1517         if (!doVerificationThruHBaseChecksum) {
1518           String msg = "HBase checksum verification failed for file " +
1519                        path + " at offset " +
1520                        offset + " filesize " + fileSize + 
1521                        " but this cannot happen because doVerify is " +
1522                        doVerificationThruHBaseChecksum;
1523           HFile.LOG.warn(msg);
1524           throw new IOException(msg); // cannot happen case here
1525         }
1526         HFile.checksumFailures.incrementAndGet(); // update metrics
1527  
1528         // If we have a checksum failure, we fall back into a mode where
1529         // the next few reads use HDFS level checksums. We aim to make the
1530         // next CHECKSUM_VERIFICATION_NUM_IO_THRESHOLD reads avoid
1531         // hbase checksum verification, but since this value is set without
1532         // holding any locks, it can so happen that we might actually do
1533         // a few more than precisely this number.
1534         this.checksumOffCount = CHECKSUM_VERIFICATION_NUM_IO_THRESHOLD;
1535         this.useHBaseChecksum = false;
1536         doVerificationThruHBaseChecksum = false;
1537         is = this.istream;
1538         blk = readBlockDataInternal(is, offset, onDiskSizeWithHeaderL,
1539                                     uncompressedSize, pread,
1540                                     doVerificationThruHBaseChecksum);
1541         if (blk != null) {
1542           HFile.LOG.warn("HDFS checksum verification suceeded for file " +
1543                          path + " at offset " +
1544                          offset + " filesize " + fileSize);
1545         }
1546       } 
1547       if (blk == null && !doVerificationThruHBaseChecksum) {
1548         String msg = "readBlockData failed, possibly due to " +
1549                      "checksum verification failed for file " + path +
1550                      " at offset " + offset + " filesize " + fileSize;
1551         HFile.LOG.warn(msg);
1552         throw new IOException(msg);
1553       }
1554 
1555       // If there is a checksum mismatch earlier, then retry with 
1556       // HBase checksums switched off and use HDFS checksum verification.
1557       // This triggers HDFS to detect and fix corrupt replicas. The
1558       // next checksumOffCount read requests will use HDFS checksums.
1559       // The decrementing of this.checksumOffCount is not thread-safe,
1560       // but it is harmless because eventually checksumOffCount will be
1561       // a negative number.
1562       if (!this.useHBaseChecksum && this.useHBaseChecksumConfigured) {
1563         if (this.checksumOffCount-- < 0) {
1564           this.useHBaseChecksum = true; // auto re-enable hbase checksums
1565         }
1566       }
1567       return blk;
1568     }

1570     /**
1571      * Reads a version 2 block. 
1572      *
1573      * @param offset the offset in the stream to read at
1574      * @param onDiskSizeWithHeaderL the on-disk size of the block, including
1575      *          the header, or -1 if unknown
1576      * @param uncompressedSize the uncompressed size of the the block. Always
1577      *          expected to be -1. This parameter is only used in version 1.
1578      * @param pread whether to use a positional read
1579      * @param verifyChecksum Whether to use HBase checksums. 
1580      *        If HBase checksum is switched off, then use HDFS checksum.
1581      * @return the HFileBlock or null if there is a HBase checksum mismatch
1582      */
1583     private HFileBlock readBlockDataInternal(FSDataInputStream is, long offset, 
1584         long onDiskSizeWithHeaderL, int uncompressedSize, boolean pread,
1585         boolean verifyChecksum) throws IOException {
1586       if (offset < 0) {
1587         throw new IOException("Invalid offset=" + offset + " trying to read "
1588             + "block (onDiskSize=" + onDiskSizeWithHeaderL
1589             + ", uncompressedSize=" + uncompressedSize + ")");
1590       }
1591       if (uncompressedSize != -1) {
1592         throw new IOException("Version 2 block reader API does not need " +
1593             "the uncompressed size parameter");
1594       }
1595 
1596       if ((onDiskSizeWithHeaderL < hdrSize && onDiskSizeWithHeaderL != -1)
1597           || onDiskSizeWithHeaderL >= Integer.MAX_VALUE) {
1598         throw new IOException("Invalid onDisksize=" + onDiskSizeWithHeaderL
1599             + ": expected to be at least " + hdrSize
1600             + " and at most " + Integer.MAX_VALUE + ", or -1 (offset="
1601             + offset + ", uncompressedSize=" + uncompressedSize + ")");
1602       }
1603 
1604       int onDiskSizeWithHeader = (int) onDiskSizeWithHeaderL;
1605       // See if we can avoid reading the header. This is desirable, because
1606       // we will not incur a backward seek operation if we have already
1607       // read this block's header as part of the previous read's look-ahead.
1608       // And we also want to skip reading the header again if it has already
1609       // been read.
1610       PrefetchedHeader prefetchedHeader = prefetchedHeaderForThread.get();
1611       ByteBuffer headerBuf = prefetchedHeader.offset == offset ?
1612           prefetchedHeader.buf : null;
1613 
1614       int nextBlockOnDiskSize = 0;
1615       // Allocate enough space to fit the next block's header too.
1616       byte[] onDiskBlock = null;
1617 
1618       HFileBlock b = null;
1619       if (onDiskSizeWithHeader > 0) {
1620         // We know the total on-disk size but not the uncompressed size. Read
1621         // the entire block into memory, then parse the header and decompress
1622         // from memory if using compression. This code path is used when
1623         // doing a random read operation relying on the block index, as well as
1624         // when the client knows the on-disk size from peeking into the next
1625         // block's header (e.g. this block's header) when reading the previous
1626         // block. This is the faster and more preferable case.
1627 
1628         // Size that we have to skip in case we have already read the header.
1629         int preReadHeaderSize = headerBuf == null ? 0 : hdrSize;
1630         onDiskBlock = new byte[onDiskSizeWithHeader + hdrSize];
1631         nextBlockOnDiskSize = readAtOffset(is, onDiskBlock,
1632             preReadHeaderSize, onDiskSizeWithHeader - preReadHeaderSize,
1633             true, offset + preReadHeaderSize, pread);
1634         if (headerBuf != null) {
1635           // the header has been read when reading the previous block, copy
1636           // to this block's header
1637           System.arraycopy(headerBuf.array(),
1638               headerBuf.arrayOffset(), onDiskBlock, 0, hdrSize);
1639         } else {
1640           headerBuf = ByteBuffer.wrap(onDiskBlock, 0, hdrSize);
1641         }
1642         // We know the total on-disk size but not the uncompressed size. Read
1643         // the entire block into memory, then parse the header and decompress
1644         // from memory if using compression. Here we have already read the
1645         // block's header
1646         try {
1647           b = new HFileBlock(headerBuf, getMinorVersion());
1648         } catch (IOException ex) {
1649           // Seen in load testing. Provide comprehensive debug info.
1650           throw new IOException("Failed to read compressed block at "
1651               + offset
1652               + ", onDiskSizeWithoutHeader="
1653               + onDiskSizeWithHeader
1654               + ", preReadHeaderSize="
1655               + hdrSize
1656               + ", header.length="
1657               + prefetchedHeader.header.length
1658               + ", header bytes: "
1659               + Bytes.toStringBinary(prefetchedHeader.header, 0,
1660                   hdrSize), ex);
1661         }
1662         // if the caller specifies a onDiskSizeWithHeader, validate it.
1663         int onDiskSizeWithoutHeader = onDiskSizeWithHeader - hdrSize;
1664         assert onDiskSizeWithoutHeader >= 0;
1665         b.validateOnDiskSizeWithoutHeader(onDiskSizeWithoutHeader);
1666       } else {
1667         // Check headerBuf to see if we have read this block's header as part of
1668         // reading the previous block. This is an optimization of peeking into
1669         // the next block's header (e.g.this block's header) when reading the
1670         // previous block. This is the faster and more preferable case. If the
1671         // header is already there, don't read the header again.
1672 
1673         // Unfortunately, we still have to do a separate read operation to
1674         // read the header.
1675         if (headerBuf == null) {
1676           // From the header, determine the on-disk size of the given hfile
1677           // block, and read the remaining data, thereby incurring two read
1678           // operations. This might happen when we are doing the first read
1679           // in a series of reads or a random read, and we don't have access
1680           // to the block index. This is costly and should happen very rarely.
1681           headerBuf = ByteBuffer.allocate(hdrSize);
1682           readAtOffset(is, headerBuf.array(), headerBuf.arrayOffset(),
1683               hdrSize, false, offset, pread);
1684         }
1685 
1686         b = new HFileBlock(headerBuf, getMinorVersion());
1687         onDiskBlock = new byte[b.getOnDiskSizeWithHeader() + hdrSize];
1688         System.arraycopy(headerBuf.array(),
1689               headerBuf.arrayOffset(), onDiskBlock, 0, hdrSize);
1690         nextBlockOnDiskSize =
1691           readAtOffset(is, onDiskBlock, hdrSize, b.getOnDiskSizeWithHeader()
1692               - hdrSize, true, offset + hdrSize, pread);
1693         onDiskSizeWithHeader = b.onDiskSizeWithoutHeader + hdrSize;
1694       }
1695 
1696       boolean isCompressed =
1697         compressAlgo != null
1698             && compressAlgo != Compression.Algorithm.NONE;
1699       if (!isCompressed) {
1700         b.assumeUncompressed();
1701       }
1702 
1703       if (verifyChecksum &&
1704           !validateBlockChecksum(b, onDiskBlock, hdrSize)) {
1705         return null;             // checksum mismatch
1706       }
1707 
1708       if (isCompressed) {
1709         // This will allocate a new buffer but keep header bytes.
1710         b.allocateBuffer(nextBlockOnDiskSize > 0);
1711         if (b.blockType.equals(BlockType.ENCODED_DATA)) {
1712           encodedBlockDecodingCtx.prepareDecoding(b.getOnDiskSizeWithoutHeader(),
1713               b.getUncompressedSizeWithoutHeader(), b.getBufferWithoutHeader(), onDiskBlock,
1714               hdrSize);
1715         } else {
1716           defaultDecodingCtx.prepareDecoding(b.getOnDiskSizeWithoutHeader(),
1717               b.getUncompressedSizeWithoutHeader(), b.getBufferWithoutHeader(), onDiskBlock,
1718               hdrSize);
1719         }
1720         if (nextBlockOnDiskSize > 0) {
1721           // Copy next block's header bytes into the new block if we have them.
1722           System.arraycopy(onDiskBlock, onDiskSizeWithHeader, b.buf.array(),
1723               b.buf.arrayOffset() + hdrSize
1724               + b.uncompressedSizeWithoutHeader + b.totalChecksumBytes(),
1725               hdrSize);
1726         }
1727       } else {
1728         // The onDiskBlock will become the headerAndDataBuffer for this block.
1729         // If nextBlockOnDiskSizeWithHeader is not zero, the onDiskBlock already
1730         // contains the header of next block, so no need to set next
1731         // block's header in it.
1732         b = new HFileBlock(ByteBuffer.wrap(onDiskBlock, 0,
1733                 onDiskSizeWithHeader), getMinorVersion());
1734       }
1735 
1736       b.nextBlockOnDiskSizeWithHeader = nextBlockOnDiskSize;
1737 
1738       // Set prefetched header
1739       if (b.nextBlockOnDiskSizeWithHeader > 0) {
1740         prefetchedHeader.offset = offset + b.getOnDiskSizeWithHeader();
1741         System.arraycopy(onDiskBlock, onDiskSizeWithHeader,
1742             prefetchedHeader.header, 0, hdrSize);
1743       }
1744 
1745       b.includesMemstoreTS = includesMemstoreTS;
1746       b.offset = offset;
1747       return b;
1748     }
