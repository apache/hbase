63      byte[] val = "val".getBytes();
64      KeyValue kv = new KeyValue(row, fam, qf, ts, val);

75      assertEquals(kv.getLength(), deserializedKv.getLength());
76    }

551   /**
552    * Create a table of name <code>name</code> with {@link COLUMNS} for
553    * families.
554    * @param name Name to give table.
555    * @param versions How many versions to allow per column.
556    * @return Column descriptor.
557    */
558   protected HTableDescriptor createTableDescriptor(final String name,
559       final int versions) {
560     HTableDescriptor htd = new HTableDescriptor(name);
561     htd.addFamily(new HColumnDescriptor(fam1)
562         .setMaxVersions(versions)
563         .setBlockCacheEnabled(false)
564     );

566         .setMaxVersions(versions)
567         .setBlockCacheEnabled(false)
568     );
569     htd.addFamily(new HColumnDescriptor(fam3)
570         .setMaxVersions(versions)
571         .setBlockCacheEnabled(false)
572     );
