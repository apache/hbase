19  package org.apache.hadoop.hbase.master.handler;
20  
21  import java.io.IOException;
22  import java.util.List;
23  
24  import org.apache.hadoop.classification.InterfaceAudience;
25  import org.apache.hadoop.hbase.HRegionInfo;
26  import org.apache.hadoop.hbase.HTableDescriptor;
27  import org.apache.hadoop.hbase.Server;
28  import org.apache.hadoop.hbase.master.HMaster;
29  import org.apache.hadoop.hbase.master.MasterCoprocessorHost;
30  import org.apache.hadoop.hbase.master.MasterServices;
31  
32  @InterfaceAudience.Private
33  public class ModifyTableHandler extends TableEventHandler {
34    private final HTableDescriptor htd;
35  
36    public ModifyTableHandler(final byte [] tableName,
37        final HTableDescriptor htd, final Server server,
38        final MasterServices masterServices)
39    throws IOException {
40      super(EventType.C_M_MODIFY_TABLE, tableName, server, masterServices);
41      // Check table exists.
42      getTableDescriptor();
43      // This is the new schema we are going to write out as this modification.
44      this.htd = htd;
45    }
46  
47    @Override
48    protected void handleTableOperation(List<HRegionInfo> hris)
49    throws IOException {
50      MasterCoprocessorHost cpHost = ((HMaster) this.server)
51          .getCoprocessorHost();
52      if (cpHost != null) {
53        cpHost.preModifyTableHandler(this.tableName, this.htd);
54      }
55      // Update descriptor
56      this.masterServices.getTableDescriptors().add(this.htd);
57      if (cpHost != null) {
58        cpHost.postModifyTableHandler(this.tableName, this.htd);
59      }
60    }
61  
62    @Override
63    public String toString() {
64      String name = "UnknownServerName";
65      if(server != null && server.getServerName() != null) {
66        name = server.getServerName().toString();
67      }
68      return getClass().getSimpleName() + "-" + name + "-" + getSeqid() + "-" +
69        tableNameStr;
70    }
71  }
