239   /**
240    * Delete a region from assignments, meta, or completely from hdfs.
241    * @param unassign if true unassign region if assigned
242    * @param metaRow  if true remove region's row from META
243    * @param hdfs if true remove region's dir in HDFS
244    * @param regionInfoOnly if true remove a region dir's .regioninfo file
245    */
246   private void deleteRegion(Configuration conf, final HTableDescriptor htd,
247       byte[] startKey, byte[] endKey, boolean unassign, boolean metaRow,
248       boolean hdfs, boolean regionInfoOnly) throws IOException, InterruptedException {
249     LOG.info("** Before delete:");
250     dumpMeta(htd.getName());
251 
252     Map<HRegionInfo, ServerName> hris = tbl.getRegionLocations();
253     for (Entry<HRegionInfo, ServerName> e: hris.entrySet()) {
254       HRegionInfo hri = e.getKey();
255       ServerName hsa = e.getValue();
256       if (Bytes.compareTo(hri.getStartKey(), startKey) == 0
257           && Bytes.compareTo(hri.getEndKey(), endKey) == 0) {
258 
259         LOG.info("RegionName: " +hri.getRegionNameAsString());
260         byte[] deleteRow = hri.getRegionName();
261 
262         if (unassign) {
263           LOG.info("Undeploying region " + hri + " from server " + hsa);
264           undeployRegion(new HBaseAdmin(conf), hsa, hri);
265         }
266 
267         if (regionInfoOnly) {
268           LOG.info("deleting hdfs .regioninfo data: " + hri.toString() + hsa.toString());
269           Path rootDir = new Path(conf.get(HConstants.HBASE_DIR));
270           FileSystem fs = rootDir.getFileSystem(conf);
271           Path p = new Path(rootDir + "/" + htd.getNameAsString(), hri.getEncodedName());
272           Path hriPath = new Path(p, HRegion.REGIONINFO_FILE);
273           fs.delete(hriPath, true);
274         }
275 
276         if (hdfs) {
277           LOG.info("deleting hdfs data: " + hri.toString() + hsa.toString());
278           Path rootDir = new Path(conf.get(HConstants.HBASE_DIR));
279           FileSystem fs = rootDir.getFileSystem(conf);
280           Path p = new Path(rootDir + "/" + htd.getNameAsString(), hri.getEncodedName());
281           HBaseFsck.debugLsr(conf, p);
282           boolean success = fs.delete(p, true);
283           LOG.info("Deleted " + p + " sucessfully? " + success);
284           HBaseFsck.debugLsr(conf, p);
285         }
286 
287         if (metaRow) {
288           HTable meta = new HTable(conf, HConstants.META_TABLE_NAME);
289           Delete delete = new Delete(deleteRow);
290           meta.delete(delete);
291         }
292       }
293       LOG.info(hri.toString() + hsa.toString());
294     }
295 
296     TEST_UTIL.getMetaTableRows(htd.getName());
297     LOG.info("*** After delete:");
298     dumpMeta(htd.getName());
299   }

301   /**
302    * Setup a clean table before we start mucking with it.
303    *
304    * @throws IOException
305    * @throws InterruptedException
306    * @throws KeeperException
307    */
308   HTable setupTable(String tablename) throws Exception {
309     HTableDescriptor desc = new HTableDescriptor(tablename);
310     HColumnDescriptor hcd = new HColumnDescriptor(Bytes.toString(FAM));
311     desc.addFamily(hcd); // If a table has no CF's it doesn't get checked
312     TEST_UTIL.getHBaseAdmin().createTable(desc, SPLITS);
313     tbl = new HTable(TEST_UTIL.getConfiguration(), tablename);
314 
315     List<Put> puts = new ArrayList<Put>();
316     for (byte[] row : ROWKEYS) {
317       Put p = new Put(row);
318       p.add(FAM, Bytes.toBytes("val"), row);
319       puts.add(p);
320     }
321     tbl.put(puts);
322     tbl.flushCommits();
323     return tbl;
324   }

497   Map<ServerName, List<String>> getDeployedHRIs(
498       final HBaseAdmin admin) throws IOException {
499     ClusterStatus status = admin.getClusterStatus();
500     Collection<ServerName> regionServers = status.getServers();
501     Map<ServerName, List<String>> mm =
502         new HashMap<ServerName, List<String>>();
503     HConnection connection = admin.getConnection();
504     for (ServerName hsi : regionServers) {
505       AdminProtocol server =
506         connection.getAdmin(hsi.getHostname(), hsi.getPort());
507 
508       // list all online regions from this region server
509       List<HRegionInfo> regions = ProtobufUtil.getOnlineRegions(server);
510       List<String> regionNames = new ArrayList<String>();
511       for (HRegionInfo hri : regions) {
512         regionNames.add(hri.getRegionNameAsString());
513       }
514       mm.put(hsi, regionNames);
515     }
516     return mm;
517   }
