285    * A convenience method to bind to a given address and report
286    * better exceptions if the address is not a valid host.
287    * @param socket the socket to bind
288    * @param address the address to bind to
289    * @param backlog the number of connections allowed in the queue

294   public static void bind(ServerSocket socket, InetSocketAddress address,
295                           int backlog) throws IOException {
296     try {
297       socket.bind(address, backlog);
298     } catch (BindException e) {
299       BindException bindException =
300         new BindException("Problem binding to " + address + " : " +
301             e.getMessage());
302       bindException.initCause(e);
303       throw bindException;
304     } catch (SocketException e) {
305       // If they try to bind to a different host's address, give a better
306       // error message.
307       if ("Unresolved address".equals(e.getMessage())) {
308         throw new UnknownHostException("Invalid hostname for server: " +
309                                        address.getHostName());
310       }
311       throw e;
312     }

354     protected synchronized void setResponse(Object value, Status status,
355         String errorClass, String error) {
356       if (this.isError)
357         return;
358       if (errorClass != null) {
359         this.isError = true;
360       }
361  
362       ByteBufferOutputStream buf = null;
363       if (value != null) {
364         buf = new ByteBufferOutputStream(((Message)value).getSerializedSize());
365       } else {
366         buf = new ByteBufferOutputStream(BUFFER_INITIAL_SIZE);
367       }
368       DataOutputStream out = new DataOutputStream(buf);
369       try {
370         RpcResponseHeader.Builder builder = RpcResponseHeader.newBuilder();
371         // Call id.
372         builder.setCallId(this.id);
373         builder.setStatus(status);
374         builder.build().writeDelimitedTo(out);
375         if (error != null) {
376           RpcException.Builder b = RpcException.newBuilder();
377           b.setExceptionName(errorClass);
378           b.setStackTrace(error);
379           b.build().writeDelimitedTo(out);
380         } else {
381           if (value != null) {
382             ((Message)value).writeDelimitedTo(out);
383           }
384         }
385         if (connection.useWrap) {
386           wrapWithSasl(buf);
387         }
388       } catch (IOException e) {
389         LOG.warn("Exception while creating response " + e);
390       }
391       ByteBuffer bb = buf.getByteBuffer();
392       bb.position(0);
393       this.response = bb;
394     }

396     private void wrapWithSasl(ByteBufferOutputStream response)
397         throws IOException {
398       if (connection.useSasl) {
399         // getByteBuffer calls flip()
400         ByteBuffer buf = response.getByteBuffer();
401         byte[] token;
402         // synchronization may be needed since there can be multiple Handler
403         // threads using saslServer to wrap responses.
404         synchronized (connection.saslServer) {
405           token = connection.saslServer.wrap(buf.array(),
406               buf.arrayOffset(), buf.remaining());
407         }
408         if (LOG.isDebugEnabled())
409           LOG.debug("Adding saslServer wrapped token of size " + token.length
410               + " as call response.");
411         buf.clear();
412         DataOutputStream saslOut = new DataOutputStream(response);
413         saslOut.writeInt(token.length);
414         saslOut.write(token, 0, token.length);
415       }
416     }

537     private class Reader implements Runnable {
538       private volatile boolean adding = false;
539       private final Selector readSelector;
540 
541       Reader() throws IOException {
542         this.readSelector = Selector.open();
543       }
544       public void run() {
545         try {
546           doRunLoop();
547         } finally {
548           try {
549             readSelector.close();
550           } catch (IOException ioe) {
551             LOG.error("Error closing read selector in " + getName(), ioe);
552           }
553         }
554       }
555 
556       private synchronized void doRunLoop() {
557         while (running) {
558           SelectionKey key = null;
559           try {
560             readSelector.select();
561             while (adding) {
562               this.wait(1000);
563             }
564 
565             Iterator<SelectionKey> iter = readSelector.selectedKeys().iterator();
566             while (iter.hasNext()) {
567               key = iter.next();
568               iter.remove();
569               if (key.isValid()) {
570                 if (key.isReadable()) {
571                   doRead(key);
572                 }
573               }
574               key = null;
575             }
576           } catch (InterruptedException e) {
577             if (running) {                      // unexpected -- log it
578               LOG.info(getName() + " unexpectedly interrupted: " +
579                   StringUtils.stringifyException(e));
580             }
581           } catch (IOException ex) {
582             LOG.error("Error in Reader", ex);
583           }
584         }
585       }
586 
587       /**
588        * This gets reader into the state that waits for the new channel
589        * to be registered with readSelector. If it was waiting in select()
590        * the thread will be woken up, otherwise whenever select() is called
591        * it will return even if there is nothing to read and wait
592        * in while(adding) for finishAdd call
593        */
594       public void startAdd() {
595         adding = true;
596         readSelector.wakeup();
597       }
598 
599       public synchronized SelectionKey registerChannel(SocketChannel channel)
600         throws IOException {
601         return channel.register(readSelector, SelectionKey.OP_READ);
602       }
603 
604       public synchronized void finishAdd() {
605         adding = false;
606         this.notify();
607       }
608     }

610     /** cleanup connections from connectionList. Choose a random range
611      * to scan and also have a limit on the number of the connections
612      * that will be cleanedup per run. The criteria for cleanup is the time
613      * for which the connection was idle. If 'force' is true then all
614      * connections will be looked at for the cleanup.
615      * @param force all connections will be looked at for cleanup
616      */
617     private void cleanupConnections(boolean force) {
618       if (force || numConnections > thresholdIdleConnections) {
619         long currentTime = System.currentTimeMillis();
620         if (!force && (currentTime - lastCleanupRunTime) < cleanupInterval) {
621           return;
622         }
623         int start = 0;
624         int end = numConnections - 1;
625         if (!force) {
626           start = rand.nextInt() % numConnections;
627           end = rand.nextInt() % numConnections;
628           int temp;
629           if (end < start) {
630             temp = start;
631             start = end;
632             end = temp;
633           }
634         }
635         int i = start;
636         int numNuked = 0;
637         while (i <= end) {
638           Connection c;
639           synchronized (connectionList) {
640             try {
641               c = connectionList.get(i);
642             } catch (Exception e) {return;}
643           }
644           if (c.timedOut(currentTime)) {
645             if (LOG.isDebugEnabled())
646               LOG.debug(getName() + ": disconnecting client " + c.getHostAddress());
647             closeConnection(c);
648             numNuked++;
649             end--;
650             //noinspection UnusedAssignment
651             c = null;
652             if (!force && numNuked == maxConnectionsToNuke) break;
653           }
654           else i++;
655         }
656         lastCleanupRunTime = System.currentTimeMillis();
657       }
658     }

660     @Override
661     public void run() {
662       LOG.info(getName() + ": starting");
663       SERVER.set(HBaseServer.this);
664 
665       while (running) {
666         SelectionKey key = null;
667         try {
668           selector.select(); // FindBugs IS2_INCONSISTENT_SYNC
669           Iterator<SelectionKey> iter = selector.selectedKeys().iterator();
670           while (iter.hasNext()) {
671             key = iter.next();
672             iter.remove();
673             try {
674               if (key.isValid()) {
675                 if (key.isAcceptable())
676                   doAccept(key);
677               }
678             } catch (IOException ignored) {
679             }
680             key = null;
681           }
682         } catch (OutOfMemoryError e) {
683           if (errorHandler != null) {
684             if (errorHandler.checkOOME(e)) {
685               LOG.info(getName() + ": exiting on OOME");
686               closeCurrentConnection(key, e);
687               cleanupConnections(true);
688               return;
689             }
690           } else {

692             // log the event and sleep for a minute and give
693             // some thread(s) a chance to finish
694             LOG.warn("Out of Memory in server select", e);
695             closeCurrentConnection(key, e);
696             cleanupConnections(true);
697             try { Thread.sleep(60000); } catch (Exception ignored) {}
698           }
699         } catch (Exception e) {
700           closeCurrentConnection(key, e);
701         }
702         cleanupConnections(false);
703       }
704       LOG.info("Stopping " + this.getName());
705

707         try {
708           acceptChannel.close();
709           selector.close();
710         } catch (IOException ignored) { }
711 
712         selector= null;
713         acceptChannel= null;
714 
715         // clean up all connections
716         while (!connectionList.isEmpty()) {
717           closeConnection(connectionList.remove(0));
718         }
719       }
720     }

850     private void doRunLoop() {
851       long lastPurgeTime = 0;   // last check for old calls.
852 
853       while (running) {
854         try {
855           waitPending();     // If a channel is being registered, wait.
856           writeSelector.select(purgeTimeout);
857           Iterator<SelectionKey> iter = writeSelector.selectedKeys().iterator();
858           while (iter.hasNext()) {
859             SelectionKey key = iter.next();
860             iter.remove();
861             try {
862               if (key.isValid() && key.isWritable()) {
863                   doAsyncWrite(key);
864               }
865             } catch (IOException e) {
866               LOG.info(getName() + ": doAsyncWrite threw exception " + e);
867             }
868           }
869           long now = System.currentTimeMillis();
870           if (now < lastPurgeTime + purgeTimeout) {
871             continue;
872           }
873           lastPurgeTime = now;
874           //
875           // If there were some calls that have not been sent out for a
876           // long time, discard them.
877           //
878           LOG.debug("Checking for old call responses.");
879           ArrayList<Call> calls;
880 
881           // get the list of channels from list of keys.
882           synchronized (writeSelector.keys()) {
883             calls = new ArrayList<Call>(writeSelector.keys().size());
884             iter = writeSelector.keys().iterator();
885             while (iter.hasNext()) {
886               SelectionKey key = iter.next();
887               Call call = (Call)key.attachment();
888               if (call != null && key.channel() == call.connection.channel) {
889                 calls.add(call);
890               }
891             }
892           }
893 
894           for(Call call : calls) {
895             try {
896               doPurge(call, now);
897             } catch (IOException e) {
898               LOG.warn("Error in purging old calls " + e);
899             }
900           }
901         } catch (OutOfMemoryError e) {
902           if (errorHandler != null) {
903             if (errorHandler.checkOOME(e)) {
904               LOG.info(getName() + ": exiting on OOME");
905               return;
906             }
907           } else {
908             //
909             // we can run out of memory if we have too many threads
910             // log the event and sleep for a minute and give
911             // some thread(s) a chance to finish
912             //
913             LOG.warn("Out of Memory in server select", e);
914             try { Thread.sleep(60000); } catch (Exception ignored) {}
915           }
916         } catch (Exception e) {
917           LOG.warn("Exception in Responder " +
918                    StringUtils.stringifyException(e));
919         }
920       }
921       LOG.info("Stopping " + this.getName());
922     }

924     private void doAsyncWrite(SelectionKey key) throws IOException {
925       Call call = (Call)key.attachment();
926       if (call == null) {
927         return;
928       }
929       if (key.channel() != call.connection.channel) {
930         throw new IOException("doAsyncWrite: bad channel");
931       }
932 
933       synchronized(call.connection.responseQueue) {
934         if (processResponse(call.connection.responseQueue, false)) {
935           try {
936             key.interestOps(0);
937           } catch (CancelledKeyException e) {
938             /* The Listener/reader might have closed the socket.
939              * We don't explicitly cancel the key, so not sure if this will
940              * ever fire.
941              * This warning could be removed.
942              */
943             LOG.warn("Exception while changing ops : " + e);
944           }
945         }
946       }
947     }

949     //
950     // Remove calls that have been pending in the responseQueue
951     // for a long time.
952     //
953     private void doPurge(Call call, long now) throws IOException {
954       synchronized (call.connection.responseQueue) {
955         Iterator<Call> iter = call.connection.responseQueue.listIterator(0);
956         while (iter.hasNext()) {
957           Call nextCall = iter.next();
958           if (now > nextCall.timestamp + purgeTimeout) {
959             closeConnection(nextCall.connection);
960             break;
961           }
962         }
963       }

966     // Processes one response. Returns true if there are no more pending
967     // data for this channel.
968     //
969     private boolean processResponse(final LinkedList<Call> responseQueue,
970                                     boolean inHandler) throws IOException {
971       boolean error = true;
972       boolean done = false;       // there is more data for this channel.
973       int numElements;
974       Call call = null;
975       try {
976         //noinspection SynchronizationOnLocalVariableOrMethodParameter
977         synchronized (responseQueue) {
978           //
979           // If there are no items for this channel, then we are done
980           //
981           numElements = responseQueue.size();
982           if (numElements == 0) {
983             error = false;
984             return true;              // no more data for this channel.
985           }
986           //
987           // Extract the first call
988           //
989           call = responseQueue.removeFirst();
990           SocketChannel channel = call.connection.channel;
991           if (LOG.isDebugEnabled()) {
992             LOG.debug(getName() + ": responding to #" + call.id + " from " +
993                       call.connection);
994           }
995           //
996           // Send as much data as we can in the non-blocking fashion
997           //
998           int numBytes = channelWrite(channel, call.response);
999           if (numBytes < 0) {
1000             return true;
1001           }
1002           if (!call.response.hasRemaining()) {
1003             responseQueueLen--;
1004             call.connection.decRpcCount();
1005             //noinspection RedundantIfStatement
1006             if (numElements == 1) {    // last call fully processes.
1007               done = true;             // no more data for this channel.
1008             } else {
1009               done = false;            // more calls pending to be sent.
1010             }
1011             if (LOG.isDebugEnabled()) {
1012               LOG.debug(getName() + ": responding to #" + call.id + " from " +
1013                         call.connection + " Wrote " + numBytes + " bytes.");
1014             }
1015           } else {
1016             //
1017             // If we were unable to write the entire response out, then
1018             // insert in Selector queue.
1019             //
1020             call.connection.responseQueue.addFirst(call);
1021 
1022             if (inHandler) {
1023               // set the serve time when the response has to be sent later
1024               call.timestamp = System.currentTimeMillis();
1025               if (enqueueInSelector(call))
1026                 done = true;
1027             }
1028             if (LOG.isDebugEnabled()) {
1029               LOG.debug(getName() + ": responding to #" + call.id + " from " +
1030                         call.connection + " Wrote partial " + numBytes +
1031                         " bytes.");
1032             }
1033           }
1034           error = false;              // everything went off well
1035         }
1036       } finally {
1037         if (error && call != null) {
1038           LOG.warn(getName()+", call " + call + ": output error");
1039           done = true;               // error. no more data for this channel.
1040           closeConnection(call.connection);
1041         }
1042       }
1043       return done;
1044     }

1046     //
1047     // Enqueue for background thread to send responses out later.
1048     //
1049     private boolean enqueueInSelector(Call call) throws IOException {
1050       boolean done = false;
1051       incPending();
1052       try {
1053         // Wake up the thread blocked on select, only then can the call
1054         // to channel.register() complete.
1055         SocketChannel channel = call.connection.channel;
1056         writeSelector.wakeup();
1057         channel.register(writeSelector, SelectionKey.OP_WRITE, call);
1058       } catch (ClosedChannelException e) {
1059         //It's OK.  Channel might be closed else where.
1060         done = true;
1061       } finally {
1062         decPending();
1063       }
1064       return done;
1065     }

1067     //
1068     // Enqueue a response from the application.
1069     //
1070     void doRespond(Call call) throws IOException {
1071       // set the serve time when the response has to be sent later
1072       call.timestamp = System.currentTimeMillis();
1073       responseQueueLen++;
1074 
1075       boolean doRegister = false;
1076       synchronized (call.connection.responseQueue) {
1077         call.connection.responseQueue.addLast(call);
1078         if (call.connection.responseQueue.size() == 1) {
1079           doRegister = !processResponse(call.connection.responseQueue, false);
1080         }
1081       }
1082       if (doRegister) {
1083         enqueueInSelector(call);
1084       }
1085     }

1087     private synchronized void incPending() {   // call waiting to be enqueued.
1088       pending++;
1089     }
1090 
1091     private synchronized void decPending() { // call done enqueueing.
1092       pending--;
1093       notify();
1094     }

1096     private synchronized void waitPending() throws InterruptedException {
1097       while (pending > 0) {
1098         wait();
1099       }

1231     private void saslReadAndProcess(byte[] saslToken) throws IOException,
1232         InterruptedException {

1245       } else {

1257                   HBaseSaslRpcServer.SASL_PROPS, new SaslDigestCallbackHandler(

1262               .getCurrentUser();

1268                 throw new AccessControlException(
1269                     "Kerberos principal name does NOT have the expected "
1270                         + "hostname part: " + fullName);
1271               }
1272               current.doAs(new PrivilegedExceptionAction<Object>() {
1273                 @Override
1274                 public Object run() throws SaslException {
1275                   saslServer = Sasl.createSaslServer(AuthMethod.KERBEROS
1276                       .getMechanismName(), names[0], names[1],
1277                       HBaseSaslRpcServer.SASL_PROPS, new SaslGssCallbackHandler());
1278                   return null;
1279                 }
1280               });
1281             }
1282             if (saslServer == null)
1283               throw new AccessControlException(
1284                   "Unable to find SASL server implementation for "
1285                       + authMethod.getMechanismName());

1300               break;

1317               null);
1318         }

1325               + user + ". Negotiated QoP is "
1326               + saslServer.getNegotiatedProperty(Sasl.QOP));
1327           }          
1328           metrics.authenticationSuccess();

1330           saslContextEstablished = true;
1331         }
1332       }

1365     public int readAndProcess() throws IOException, InterruptedException {
1366       while (true) {
1367         /* Read at most one RPC. If the header is not read completely yet
1368          * then iterate until we read first RPC or until there is no data left.
1369          */
1370         int count = -1;
1371         if (dataLengthBuffer.remaining() > 0) {
1372           count = channelRead(channel, dataLengthBuffer);
1373           if (count < 0 || dataLengthBuffer.remaining() > 0)
1374             return count;
1375         }
1376 
1377         if (!rpcHeaderRead) {
1378           //Every connection is expected to send the header.
1379           if (rpcHeaderBuffer == null) {
1380             rpcHeaderBuffer = ByteBuffer.allocate(2);
1381           }
1382           count = channelRead(channel, rpcHeaderBuffer);
1383           if (count < 0 || rpcHeaderBuffer.remaining() > 0) {
1384             return count;
1385           }
1386           int version = rpcHeaderBuffer.get(0);
1387           byte[] method = new byte[] {rpcHeaderBuffer.get(1)};
1388           authMethod = AuthMethod.read(new DataInputStream(
1389               new ByteArrayInputStream(method)));
1390           dataLengthBuffer.flip();
1391           if (!HEADER.equals(dataLengthBuffer) || version != CURRENT_VERSION) {
1392               LOG.warn("Incorrect header or version mismatch from " +
1393                   hostAddress + ":" + remotePort +
1394                   " got version " + version +
1395                   " expected version " + CURRENT_VERSION);
1396             setupBadVersionResponse(version);
1397             return -1;
1398           }
1399           dataLengthBuffer.clear();
1400           if (authMethod == null) {
1401             throw new IOException("Unable to read authentication method");
1402           }
1403           if (isSecurityEnabled && authMethod == AuthMethod.SIMPLE) {
1404             AccessControlException ae = new AccessControlException(
1405                 "Authentication is required");
1406             setupResponse(authFailedResponse, authFailedCall, Status.FATAL,
1407                 ae.getClass().getName(), ae.getMessage());
1408             responder.doRespond(authFailedCall);
1409             throw ae;
1410           }
1411           if (!isSecurityEnabled && authMethod != AuthMethod.SIMPLE) {
1412             doRawSaslReply(SaslStatus.SUCCESS, new IntWritable(
1413                 HBaseSaslRpcServer.SWITCH_TO_SIMPLE_AUTH), null, null);
1414             authMethod = AuthMethod.SIMPLE;
1415             // client has already sent the initial Sasl message and we
1416             // should ignore it. Both client and server should fall back
1417             // to simple auth from now on.
1418             skipInitialSaslHandshake = true;
1419           }
1420           if (authMethod != AuthMethod.SIMPLE) {
1421             useSasl = true;
1422           }
1423 
1424           rpcHeaderBuffer = null;
1425           rpcHeaderRead = true;
1426           continue;
1427         }
1428 
1429         if (data == null) {
1430           dataLengthBuffer.flip();
1431           dataLength = dataLengthBuffer.getInt();
1432 
1433           if (dataLength == HBaseClient.PING_CALL_ID) {
1434             if(!useWrap) { //covers the !useSasl too
1435               dataLengthBuffer.clear();
1436               return 0;  //ping message
1437             }
1438           }
1439           if (dataLength < 0) {
1440             throw new IllegalArgumentException("Unexpected data length " 
1441                 + dataLength + "!! from " + getHostAddress());
1442           }
1443           data = ByteBuffer.allocate(dataLength);
1444           incRpcCount();  // Increment the rpc count
1445         }
1446 
1447         count = channelRead(channel, data);
1448 
1449         if (data.remaining() == 0) {
1450           dataLengthBuffer.clear();
1451           data.flip();
1452           if (skipInitialSaslHandshake) {
1453             data = null;
1454             skipInitialSaslHandshake = false;
1455             continue;
1456           }
1457           boolean isHeaderRead = headerRead;
1458           if (useSasl) {
1459             saslReadAndProcess(data.array());
1460           } else {
1461             processOneRpc(data.array());
1462           }
1463           data = null;
1464           if (!isHeaderRead) {
1465             continue;
1466           }
1467         }
1468         return count;
1469       }
1470     }

1501     /// Reads the connection header following version
1502     private void processHeader(byte[] buf) throws IOException {
1503       DataInputStream in =
1504         new DataInputStream(new ByteArrayInputStream(buf));
1505       header = ConnectionHeader.parseFrom(in);
1506       try {
1507         String protocolClassName = header.getProtocol();
1508         if (protocolClassName != null) {
1509           protocol = getProtocolClass(header.getProtocol(), conf);
1510         }
1511       } catch (ClassNotFoundException cnfe) {
1512         throw new IOException("Unknown protocol: " + header.getProtocol());
1513       }
1514 
1515       UserGroupInformation protocolUser = createUser(header);
1516       if (!useSasl) {
1517         user = protocolUser;
1518         if (user != null) {
1519           user.setAuthenticationMethod(AuthMethod.SIMPLE.authenticationMethod);
1520         }
1521       } else {
1522         // user is authenticated
1523         user.setAuthenticationMethod(authMethod.authenticationMethod);
1524         //Now we check if this is a proxy user case. If the protocol user is
1525         //different from the 'user', it is a proxy user scenario. However,
1526         //this is not allowed if user authenticated with DIGEST.
1527         if ((protocolUser != null)
1528             && (!protocolUser.getUserName().equals(user.getUserName()))) {
1529           if (authMethod == AuthMethod.DIGEST) {
1530             // Not allowed to doAs if token authentication is used
1531             throw new AccessControlException("Authenticated user (" + user
1532                 + ") doesn't match what the client claims to be ("
1533                 + protocolUser + ")");
1534           } else {
1535             // Effective user can be different from authenticated user
1536             // for simple auth or kerberos auth
1537             // The user is the real user. Now we create a proxy user
1538             UserGroupInformation realUser = user;
1539             user = UserGroupInformation.createProxyUser(protocolUser
1540                 .getUserName(), realUser);
1541             // Now the user is a proxy user, set Authentication method Proxy.
1542             user.setAuthenticationMethod(AuthenticationMethod.PROXY);
1543           }
1544         }
1545       }
1546     }

1548     private void processUnwrappedData(byte[] inBuf) throws IOException,
1549     InterruptedException {
1550       ReadableByteChannel ch = Channels.newChannel(new ByteArrayInputStream(
1551           inBuf));
1552       // Read all RPCs contained in the inBuf, even partial ones
1553       while (true) {
1554         int count = -1;
1555         if (unwrappedDataLengthBuffer.remaining() > 0) {
1556           count = channelRead(ch, unwrappedDataLengthBuffer);
1557           if (count <= 0 || unwrappedDataLengthBuffer.remaining() > 0)
1558             return;
1559         }
1560 
1561         if (unwrappedData == null) {
1562           unwrappedDataLengthBuffer.flip();
1563           int unwrappedDataLength = unwrappedDataLengthBuffer.getInt();
1564 
1565           if (unwrappedDataLength == HBaseClient.PING_CALL_ID) {
1566             if (LOG.isDebugEnabled())
1567               LOG.debug("Received ping message");
1568             unwrappedDataLengthBuffer.clear();
1569             continue; // ping message
1570           }
1571           unwrappedData = ByteBuffer.allocate(unwrappedDataLength);
1572         }
1573 
1574         count = channelRead(ch, unwrappedData);
1575         if (count <= 0 || unwrappedData.remaining() > 0)
1576           return;
1577 
1578         if (unwrappedData.remaining() == 0) {
1579           unwrappedDataLengthBuffer.clear();
1580           unwrappedData.flip();
1581           processOneRpc(unwrappedData.array());
1582           unwrappedData = null;
1583         }
1584       }
1585     }

1587     private void processOneRpc(byte[] buf) throws IOException,
1588     InterruptedException {
1589       if (headerRead) {
1590         processData(buf);
1591       } else {
1592         processHeader(buf);
1593         headerRead = true;
1594         if (!authorizeConnection()) {
1595           throw new AccessControlException("Connection from " + this
1596               + " for protocol " + header.getProtocol()
1597               + " is unauthorized for user " + user);
1598         }
1599       }

1602     protected void processData(byte[] buf) throws  IOException, InterruptedException {
1603       DataInputStream dis =
1604         new DataInputStream(new ByteArrayInputStream(buf));
1605       RpcRequestHeader request = RpcRequestHeader.parseDelimitedFrom(dis);
1606 
1607       int id = request.getCallId();
1608       long callSize = buf.length;
1609 
1610       if (LOG.isDebugEnabled()) {
1611         LOG.debug(" got call #" + id + ", " + callSize + " bytes");
1612       }
1613       // Enforcing the call queue size, this triggers a retry in the client
1614       if ((callSize + callQueueSize.get()) > maxQueueSize) {
1615         final Call callTooBig = new Call(id, null, this, responder, callSize,
1616             null);
1617         ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
1618         setupResponse(responseBuffer, callTooBig, Status.FATAL,
1619             IOException.class.getName(),
1620             "Call queue is full, is ipc.server.max.callqueue.size too small?");
1621         responder.doRespond(callTooBig);
1622         return;
1623       }
1624 
1625       RpcRequestBody rpcRequestBody;
1626       try {
1627         rpcRequestBody = RpcRequestBody.parseDelimitedFrom(dis);
1628       } catch (Throwable t) {
1629         LOG.warn("Unable to read call parameters for client " +
1630                  getHostAddress(), t);
1631         final Call readParamsFailedCall = new Call(id, null, this, responder,
1632             callSize, null);
1633         ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
1634 
1635         setupResponse(responseBuffer, readParamsFailedCall, Status.FATAL,
1636             t.getClass().getName(),
1637             "IPC server unable to read call parameters: " + t.getMessage());
1638         responder.doRespond(readParamsFailedCall);
1639         return;
1640       }
1641 
1642       Call call;
1643       if (request.hasTinfo()) {
1644         call = new Call(id, rpcRequestBody, this, responder, callSize,
1645             new TraceInfo(request.getTinfo().getTraceId(), request.getTinfo()
1646                 .getParentId()));
1647       } else {
1648         call = new Call(id, rpcRequestBody, this, responder, callSize, null);
1649       }
1650 
1651       callQueueSize.add(callSize);
1652 
1653       if (priorityCallQueue != null && getQosLevel(rpcRequestBody) > highPriorityLevel) {
1654         priorityCallQueue.put(call);
1655       } else if (replicationQueue != null
1656           && getQosLevel(rpcRequestBody) == HConstants.REPLICATION_QOS) {
1657         replicationQueue.put(call);
1658       } else {
1659         callQueue.put(call);              // queue the call; maybe blocked here
1660       }
1661     }

1663     private boolean authorizeConnection() throws IOException {
1664       try {
1665         // If auth method is DIGEST, the token was obtained by the
1666         // real user for the effective user, therefore not required to
1667         // authorize real user. doAs is allowed only for simple or kerberos
1668         // authentication
1669         if (user != null && user.getRealUser() != null
1670             && (authMethod != AuthMethod.DIGEST)) {
1671           ProxyUsers.authorize(user, this.getHostAddress(), conf);
1672         }
1673         authorize(user, header, getHostInetAddress());
1674         if (LOG.isDebugEnabled()) {
1675           LOG.debug("Successfully authorized " + header);
1676         }
1677         metrics.authorizationSuccess();
1678       } catch (AuthorizationException ae) {
1679         LOG.debug("Connection authorization failed: "+ae.getMessage(), ae);
1680         metrics.authorizationFailure();
1681         setupResponse(authFailedResponse, authFailedCall, Status.FATAL,
1682             ae.getClass().getName(), ae.getMessage());
1683         responder.doRespond(authFailedCall);
1684         return false;
1685       }
1686       return true;
1687     }

1702     private UserGroupInformation createUser(ConnectionHeader head) {
1703       UserGroupInformation ugi = null;
1704 
1705       if (!head.hasUserInfo()) {
1706         return null;
1707       }
1708       UserInformation userInfoProto = head.getUserInfo();
1709       String effectiveUser = null;
1710       if (userInfoProto.hasEffectiveUser()) {
1711         effectiveUser = userInfoProto.getEffectiveUser();
1712       }
1713       String realUser = null;
1714       if (userInfoProto.hasRealUser()) {
1715         realUser = userInfoProto.getRealUser();
1716       }
1717       if (effectiveUser != null) {
1718         if (realUser != null) {
1719           UserGroupInformation realUserUgi =
1720               UserGroupInformation.createRemoteUser(realUser);
1721           ugi = UserGroupInformation.createProxyUser(effectiveUser, realUserUgi);
1722         } else {
1723           ugi = UserGroupInformation.createRemoteUser(effectiveUser);
1724         }
1725       }
1726       return ugi;

1735     public Handler(final BlockingQueue<Call> cq, int instanceNumber) {
1736       this.myCallQueue = cq;
1737       this.setDaemon(true);
1738

1740       if (cq == priorityCallQueue) {
1741         // this is just an amazing hack, but it works.
1742         threadName = "PRI " + threadName;
1743       } else if (cq == replicationQueue) {

1750     @Override
1751     public void run() {
1752       LOG.info(getName() + ": starting");
1753       status.setStatus("starting");
1754       SERVER.set(HBaseServer.this);
1755       while (running) {
1756 
1757         try {
1758           status.pause("Waiting for a call");
1759           Call call = myCallQueue.take(); // pop the queue; maybe blocked here
1760           status.setStatus("Setting up call");
1761           status.setConnection(call.connection.getHostAddress(), 
1762               call.connection.getRemotePort());
1763 
1764           if (LOG.isDebugEnabled())
1765             LOG.debug(getName() + ": has #" + call.id + " from " +
1766                       call.connection);
1767 
1768           String errorClass = null;
1769           String error = null;
1770           Message value = null;
1771 
1772           CurCall.set(call);
1773           Span currentRequestSpan = NullSpan.getInstance();
1774           try {
1775             if (!started)
1776               throw new ServerNotRunningYetException("Server is not running yet");
1777 
1778             if (call.tinfo != null) {
1779               currentRequestSpan = Trace.startSpan(
1780                   "handling " + call.toString(), call.tinfo, Sampler.ALWAYS);
1781             }
1782 
1783             if (LOG.isDebugEnabled()) {
1784               UserGroupInformation remoteUser = call.connection.user;
1785               LOG.debug(getName() + ": call #" + call.id + " executing as "
1786                   + (remoteUser == null ? "NULL principal" :
1787                     remoteUser.getUserName()));
1788             }
1789 
1790             RequestContext.set(User.create(call.connection.user), getRemoteIp(),
1791                 call.connection.protocol);
1792 
1793             // make the call
1794             value = call(call.connection.protocol, call.rpcRequestBody, call.timestamp,
1795                 status);
1796           } catch (Throwable e) {
1797             LOG.debug(getName()+", call "+call+": error: " + e, e);
1798             errorClass = e.getClass().getName();
1799             error = StringUtils.stringifyException(e);
1800           } finally {
1801             currentRequestSpan.stop();
1802             // Must always clear the request context to avoid leaking
1803             // credentials between requests.
1804             RequestContext.clear();
1805           }
1806           CurCall.set(null);
1807           callQueueSize.add(call.getSize() * -1);
1808           // Set the response for undelayed calls and delayed calls with
1809           // undelayed responses.
1810           if (!call.isDelayed() || !call.isReturnValueDelayed()) {
1811             call.setResponse(value,
1812               errorClass == null? Status.SUCCESS: Status.ERROR,
1813                 errorClass, error);
1814           }
1815           call.sendResponseIfReady();
1816           status.markComplete("Sent response");
1817         } catch (InterruptedException e) {
1818           if (running) {                          // unexpected -- log it
1819             LOG.info(getName() + " caught: " +
1820                      StringUtils.stringifyException(e));
1821           }
1822         } catch (OutOfMemoryError e) {
1823           if (errorHandler != null) {
1824             if (errorHandler.checkOOME(e)) {
1825               LOG.info(getName() + ": exiting on OOME");
1826               return;
1827             }
1828           } else {
1829             // rethrow if no handler
1830             throw e;
1831           }
1832        } catch (ClosedChannelException cce) {
1833           LOG.warn(getName() + " caught a ClosedChannelException, " +
1834             "this means that the server was processing a " +
1835             "request but the client went away. The error message was: " +
1836             cce.getMessage());
1837         } catch (Exception e) {
1838           LOG.warn(getName() + " caught: " +
1839                    StringUtils.stringifyException(e));
1840         }
1841       }
1842       LOG.info(getName() + ": exiting");
1843     }
