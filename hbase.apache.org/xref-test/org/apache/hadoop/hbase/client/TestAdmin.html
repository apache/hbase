358   public void testOnlineChangeTableSchema() throws IOException, InterruptedException {
359     final byte [] tableName = Bytes.toBytes("changeTableSchemaOnline");
360     TEST_UTIL.getMiniHBaseCluster().getMaster().getConfiguration().setBoolean(
361         "hbase.online.schema.update.enable", true);
362     HTableDescriptor [] tables = admin.listTables();
363     int numTables = tables.length;
364     TEST_UTIL.createTable(tableName, HConstants.CATALOG_FAMILY).close();
365     tables = this.admin.listTables();
366     assertEquals(numTables + 1, tables.length);
367 
368     // FIRST, do htabledescriptor changes.
369     HTableDescriptor htd = this.admin.getTableDescriptor(tableName);
370     // Make a copy and assert copy is good.
371     HTableDescriptor copy = new HTableDescriptor(htd);
372     assertTrue(htd.equals(copy));
373     // Now amend the copy. Introduce differences.
374     long newFlushSize = htd.getMemStoreFlushSize() / 2;
375     copy.setMemStoreFlushSize(newFlushSize);
376     final String key = "anyoldkey";
377     assertTrue(htd.getValue(key) == null);
378     copy.setValue(key, key);
379     boolean expectedException = false;
380     try {
381       modifyTable(tableName, copy);
382     } catch (TableNotDisabledException re) {
383       expectedException = true;
384     }
385     assertFalse(expectedException);
386     HTableDescriptor modifiedHtd = this.admin.getTableDescriptor(tableName);
387     assertFalse(htd.equals(modifiedHtd));
388     assertTrue(copy.equals(modifiedHtd));
389     assertEquals(newFlushSize, modifiedHtd.getMemStoreFlushSize());
390     assertEquals(key, modifiedHtd.getValue(key));
391 
392     // Now work on column family changes.
393     htd = this.admin.getTableDescriptor(tableName);
394     int countOfFamilies = modifiedHtd.getFamilies().size();
395     assertTrue(countOfFamilies > 0);
396     HColumnDescriptor hcd = modifiedHtd.getFamilies().iterator().next();
397     int maxversions = hcd.getMaxVersions();
398     final int newMaxVersions = maxversions + 1;
399     hcd.setMaxVersions(newMaxVersions);
400     final byte [] hcdName = hcd.getName();
401     expectedException = false;
402     try {
403       this.admin.modifyColumn(tableName, hcd);
404     } catch (TableNotDisabledException re) {
405       expectedException = true;
406     }
407     assertFalse(expectedException);
408     modifiedHtd = this.admin.getTableDescriptor(tableName);
409     HColumnDescriptor modifiedHcd = modifiedHtd.getFamily(hcdName);
410     assertEquals(newMaxVersions, modifiedHcd.getMaxVersions());
411 
412     // Try adding a column
413     assertFalse(this.admin.isTableDisabled(tableName));
414     final String xtracolName = "xtracol";
415     htd = this.admin.getTableDescriptor(tableName);
416     HColumnDescriptor xtracol = new HColumnDescriptor(xtracolName);
417     xtracol.setValue(xtracolName, xtracolName);
418     expectedException = false;
419     try {
420       this.admin.addColumn(tableName, xtracol);
421     } catch (TableNotDisabledException re) {
422       expectedException = true;
423     }
424     // Add column should work even if the table is enabled
425     assertFalse(expectedException);
426     modifiedHtd = this.admin.getTableDescriptor(tableName);
427     hcd = modifiedHtd.getFamily(xtracol.getName());
428     assertTrue(hcd != null);
429     assertTrue(hcd.getValue(xtracolName).equals(xtracolName));
430 
431     // Delete the just-added column.
432     this.admin.deleteColumn(tableName, xtracol.getName());
433     modifiedHtd = this.admin.getTableDescriptor(tableName);
434     hcd = modifiedHtd.getFamily(xtracol.getName());
435     assertTrue(hcd == null);
436 
437     // Delete the table
438     this.admin.disableTable(tableName);
439     this.admin.deleteTable(tableName);
440     this.admin.listTables();
441     assertFalse(this.admin.tableExists(tableName));
442   }

476   /**
477    * Modify table is async so wait on completion of the table operation in master.
478    * @param tableName
479    * @param htd
480    * @throws IOException
481    */
482   private void modifyTable(final byte [] tableName, final HTableDescriptor htd)
483   throws IOException {
484     MasterServices services = TEST_UTIL.getMiniHBaseCluster().getMaster();
485     ExecutorService executor = services.getExecutorService();
486     AtomicBoolean done = new AtomicBoolean(false);
487     executor.registerListener(EventType.C_M_MODIFY_TABLE, new DoneListener(done));
488     this.admin.modifyTable(tableName, htd);
489     while (!done.get()) {
490       synchronized (done) {
491         try {
492           done.wait(100);
493         } catch (InterruptedException e) {
494           e.printStackTrace();
495         }
496       }
497     }
498     executor.unregisterListener(EventType.C_M_MODIFY_TABLE);
499   }

501   /**
502    * Listens for when an event is done in Master.
503    */
504   static class DoneListener implements EventHandler.EventHandlerListener {
505     private final AtomicBoolean done;
506 
507     DoneListener(final AtomicBoolean done) {
508       super();
509       this.done = done;
510     }
511 
512     @Override
513     public void afterProcess(EventHandler event) {
514       this.done.set(true);
515       synchronized (this.done) {
516         // Wake anyone waiting on this value to change.
517         this.done.notifyAll();
518       }
519     }
520 
521     @Override
522     public void beforeProcess(EventHandler event) {
523       // continue
524     }
525   }

527   protected void verifyRoundRobinDistribution(HTable ht, int expectedRegions) throws IOException {
528     int numRS = ht.getConnection().getCurrentNrHRS();
529     Map<HRegionInfo, ServerName> regions = ht.getRegionLocations();
530     Map<ServerName, List<HRegionInfo>> server2Regions = new HashMap<ServerName, List<HRegionInfo>>();
531     for (Map.Entry<HRegionInfo, ServerName> entry : regions.entrySet()) {
532       ServerName server = entry.getValue();
533       List<HRegionInfo> regs = server2Regions.get(server);
534       if (regs == null) {
535         regs = new ArrayList<HRegionInfo>();
536         server2Regions.put(server, regs);
537       }
538       regs.add(entry.getKey());
539     }
540     float average = (float) expectedRegions/numRS;
541     int min = (int)Math.floor(average);
542     int max = (int)Math.ceil(average);
543     for (List<HRegionInfo> regionList : server2Regions.values()) {
544       assertTrue(regionList.size() == min || regionList.size() == max);
545     }
546   }

869   void splitTest(byte[] splitPoint, byte[][] familyNames, int[] rowCounts,
870     int numVersions, int blockSize) throws Exception {
871     byte [] tableName = Bytes.toBytes("testForceSplit");
872     assertFalse(admin.tableExists(tableName));
873     final HTable table = TEST_UTIL.createTable(tableName, familyNames,
874       numVersions, blockSize);
875     try {
876       int rowCount = 0;
877       byte[] q = new byte[0];
878 
879       // insert rows into column families. The number of rows that have values
880       // in a specific column family is decided by rowCounts[familyIndex]
881       for (int index = 0; index < familyNames.length; index++) {
882         ArrayList<Put> puts = new ArrayList<Put>(rowCounts[index]);
883         for (int i = 0; i < rowCounts[index]; i++) {
884           byte[] k = Bytes.toBytes(i);
885           Put put = new Put(k);
886           put.add(familyNames[index], q, k);
887           puts.add(put);
888         }
889         table.put(puts);
890 
891         if ( rowCount < rowCounts[index] ) {
892           rowCount = rowCounts[index];
893         }
894       }
895 
896       // get the initial layout (should just be one region)
897       Map<HRegionInfo, ServerName> m = table.getRegionLocations();
898       System.out.println("Initial regions (" + m.size() + "): " + m);
899       assertTrue(m.size() == 1);
900 
901       // Verify row count
902       Scan scan = new Scan();
903       ResultScanner scanner = table.getScanner(scan);
904       int rows = 0;
905       for(@SuppressWarnings("unused") Result result : scanner) {
906         rows++;
907       }
908       scanner.close();
909       assertEquals(rowCount, rows);
910 
911       // Have an outstanding scan going on to make sure we can scan over splits.
912       scan = new Scan();
913       scanner = table.getScanner(scan);
914       // Scan first row so we are into first region before split happens.
915       scanner.next();
916 
917       final AtomicInteger count = new AtomicInteger(0);
918       Thread t = new Thread("CheckForSplit") {
919         public void run() {
920           for (int i = 0; i < 20; i++) {
921             try {
922               sleep(1000);
923             } catch (InterruptedException e) {
924               continue;
925             }
926             // check again    table = new HTable(conf, tableName);
927             Map<HRegionInfo, ServerName> regions = null;
928             try {
929               regions = table.getRegionLocations();
930             } catch (IOException e) {
931               e.printStackTrace();
932             }
933             if (regions == null) continue;
934             count.set(regions.size());
935             if (count.get() >= 2) break;
936             LOG.debug("Cycle waiting on split");
937           }
938         }
939       };
940       t.start();
941       // Split the table
942       this.admin.split(tableName, splitPoint);
943       t.join();
944 
945       // Verify row count
946       rows = 1; // We counted one row above.
947       for (@SuppressWarnings("unused") Result result : scanner) {
948         rows++;
949         if (rows > rowCount) {
950           scanner.close();
951           assertTrue("Scanned more than expected (" + rowCount + ")", false);
952         }
953       }
954       scanner.close();
955       assertEquals(rowCount, rows);
956 
957       Map<HRegionInfo, ServerName> regions = null;
958       try {
959         regions = table.getRegionLocations();
960       } catch (IOException e) {
961         e.printStackTrace();
962       }
963       assertEquals(2, regions.size());
964       HRegionInfo[] r = regions.keySet().toArray(new HRegionInfo[0]);
965       if (splitPoint != null) {
966         // make sure the split point matches our explicit configuration
967         assertEquals(Bytes.toString(splitPoint),
968             Bytes.toString(r[0].getEndKey()));
969         assertEquals(Bytes.toString(splitPoint),
970             Bytes.toString(r[1].getStartKey()));
971         LOG.debug("Properly split on " + Bytes.toString(splitPoint));
972       } else {
973         if (familyNames.length > 1) {
974           int splitKey = Bytes.toInt(r[0].getEndKey());
975           // check if splitKey is based on the largest column family
976           // in terms of it store size
977           int deltaForLargestFamily = Math.abs(rowCount/2 - splitKey);
978           for (int index = 0; index < familyNames.length; index++) {
979             int delta = Math.abs(rowCounts[index]/2 - splitKey);
980             assertTrue(delta >= deltaForLargestFamily);
981           }
982         }
983       }
984     } finally {
985       TEST_UTIL.deleteTable(tableName);
986       table.close();
987     }
988   }
