84    @SuppressWarnings("rawtypes")
85    protected String buildRowSpec(final byte[] row, final Map familyMap, 
86        final long startTime, final long endTime, final int maxVersions) {
87      StringBuffer sb = new StringBuffer();
88      sb.append('/');
89      sb.append(Bytes.toStringBinary(name));
90      sb.append('/');
91      sb.append(Bytes.toStringBinary(row));
92      Set families = familyMap.entrySet();
93      if (families != null) {
94        Iterator i = familyMap.entrySet().iterator();
95        if (i.hasNext()) {
96          sb.append('/');
97        }
98        while (i.hasNext()) {
99          Map.Entry e = (Map.Entry)i.next();
100         Collection quals = (Collection)e.getValue();
101         if (quals != null && !quals.isEmpty()) {
102           Iterator ii = quals.iterator();
103           while (ii.hasNext()) {
104             sb.append(Bytes.toStringBinary((byte[])e.getKey()));
105             sb.append(':');
106             Object o = ii.next();
107             // Puts use byte[] but Deletes use KeyValue
108             if (o instanceof byte[]) {
109               sb.append(Bytes.toStringBinary((byte[])o));
110             } else if (o instanceof KeyValue) {
111               sb.append(Bytes.toStringBinary(((KeyValue)o).getQualifier()));
112             } else {
113               throw new RuntimeException("object type not handled");
114             }
115             if (ii.hasNext()) {
116               sb.append(',');
117             }
118           }
119         } else {
120           sb.append(Bytes.toStringBinary((byte[])e.getKey()));
121           sb.append(':');
122         }
123         if (i.hasNext()) {
124           sb.append(',');
125         }
126       }
127     }
128     if (startTime != 0 && endTime != Long.MAX_VALUE) {
129       sb.append('/');
130       sb.append(startTime);
131       if (startTime != endTime) {
132         sb.append(',');
133         sb.append(endTime);
134       }
135     } else if (endTime != Long.MAX_VALUE) {
136       sb.append('/');
137       sb.append(endTime);
138     }
139     if (maxVersions > 1) {
140       sb.append("?v=");
141       sb.append(maxVersions);
142     }
143     return sb.toString();
144   }

146   protected String buildMultiRowSpec(final byte[][] rows, int maxVersions) {
147     StringBuilder sb = new StringBuilder();
148     sb.append('/');
149     sb.append(Bytes.toStringBinary(name));
150     sb.append("/multiget/");
151     if (rows == null || rows.length == 0) {
152       return sb.toString();
153     }
154     sb.append("?");
155     for(int i=0; i<rows.length; i++) {
156       byte[] rk = rows[i];
157       if (i != 0) {
158         sb.append('&');
159       }
160       sb.append("row=");
161       sb.append(Bytes.toStringBinary(rk));
162     }
163     sb.append("&v=");
164     sb.append(maxVersions);
165 
166     return sb.toString();
167   }

169   protected Result[] buildResultFromModel(final CellSetModel model) {
170     List<Result> results = new ArrayList<Result>();
171     for (RowModel row: model.getRows()) {
172       List<KeyValue> kvs = new ArrayList<KeyValue>();
173       for (CellModel cell: row.getCells()) {
174         byte[][] split = KeyValue.parseColumn(cell.getColumn());
175         byte[] column = split[0];
176         byte[] qualifier = split.length > 1 ? split[1] : null;
177         kvs.add(new KeyValue(row.getKey(), column, qualifier, 
178           cell.getTimestamp(), cell.getValue()));
179       }
180       results.add(new Result(kvs));
181     }
182     return results.toArray(new Result[results.size()]);
183   }
184 
185   protected CellSetModel buildModelFromPut(Put put) {
186     RowModel row = new RowModel(put.getRow());
187     long ts = put.getTimeStamp();
188     for (List<KeyValue> kvs: put.getFamilyMap().values()) {
189       for (KeyValue kv: kvs) {
190         row.addCell(new CellModel(kv.getFamily(), kv.getQualifier(),
191           ts != HConstants.LATEST_TIMESTAMP ? ts : kv.getTimestamp(),
192           kv.getValue()));
193       }
194     }
195     CellSetModel model = new CellSetModel();
196     model.addRow(row);
197     return model;
198   }

241   public HTableDescriptor getTableDescriptor() throws IOException {
242     StringBuilder sb = new StringBuilder();
243     sb.append('/');
244     sb.append(Bytes.toStringBinary(name));
245     sb.append('/');
246     sb.append("schema");
247     for (int i = 0; i < maxRetries; i++) {
248       Response response = client.get(sb.toString(), Constants.MIMETYPE_PROTOBUF);
249       int code = response.getCode();
250       switch (code) {
251       case 200:
252         TableSchemaModel schema = new TableSchemaModel();
253         schema.getObjectFromMessage(response.getBody());
254         return schema.getTableDescriptor();
255       case 509: 
256         try {
257           Thread.sleep(sleepTime);
258         } catch (InterruptedException e) { }
259         break;
260       default:
261         throw new IOException("schema request returned " + code);
262       }
263     }
264     throw new IOException("schema request timed out");
265   }

271   public Result get(Get get) throws IOException {
272     TimeRange range = get.getTimeRange();
273     String spec = buildRowSpec(get.getRow(), get.getFamilyMap(),
274       range.getMin(), range.getMax(), get.getMaxVersions());
275     if (get.getFilter() != null) {
276       LOG.warn("filters not supported on gets");
277     }
278     Result[] results = getResults(spec);
279     if (results.length > 0) {
280       if (results.length > 1) {
281         LOG.warn("too many results for get (" + results.length + ")");
282       }
283       return results[0];
284     } else {
285       return new Result();
286     }
287   }

289   public Result[] get(List<Get> gets) throws IOException {
290     byte[][] rows = new byte[gets.size()][];
291     int maxVersions = 1;
292     int count = 0;
293 
294     for(Get g:gets) {
295 
296       if ( count == 0 ) {
297         maxVersions = g.getMaxVersions();
298       } else if (g.getMaxVersions() != maxVersions) {
299         LOG.warn("MaxVersions on Gets do not match, using the first in the list ("+maxVersions+")");
300       }
301 
302       if (g.getFilter() != null) {
303         LOG.warn("filters not supported on gets");
304       }
305 
306       rows[count] = g.getRow();
307       count ++;
308     }
309 
310     String spec = buildMultiRowSpec(rows, maxVersions);
311 
312     return getResults(spec);
313   }

315   private Result[] getResults(String spec) throws IOException {
316     for (int i = 0; i < maxRetries; i++) {
317       Response response = client.get(spec, Constants.MIMETYPE_PROTOBUF);
318       int code = response.getCode();
319       switch (code) {
320         case 200:
321           CellSetModel model = new CellSetModel();
322           model.getObjectFromMessage(response.getBody());
323           Result[] results = buildResultFromModel(model);
324           if ( results.length > 0) {
325             return results;
326           }
327           // fall through
328         case 404:
329           return new Result[0];
330 
331         case 509:
332           try {
333             Thread.sleep(sleepTime);
334           } catch (InterruptedException e) { }
335           break;
336         default:
337           throw new IOException("get request returned " + code);
338       }
339     }
340     throw new IOException("get request timed out");
341   }

375   public void put(List<Put> puts) throws IOException {
376     // this is a trick: The gateway accepts multiple rows in a cell set and
377     // ignores the row specification in the URI
378 
379     // separate puts by row
380     TreeMap<byte[],List<KeyValue>> map =
381       new TreeMap<byte[],List<KeyValue>>(Bytes.BYTES_COMPARATOR);
382     for (Put put: puts) {
383       byte[] row = put.getRow();
384       List<KeyValue> kvs = map.get(row);
385       if (kvs == null) {
386         kvs = new ArrayList<KeyValue>();
387         map.put(row, kvs);
388       }
389       for (List<KeyValue> l: put.getFamilyMap().values()) {
390         kvs.addAll(l);
391       }
392     }
393 
394     // build the cell set
395     CellSetModel model = new CellSetModel();
396     for (Map.Entry<byte[], List<KeyValue>> e: map.entrySet()) {
397       RowModel row = new RowModel(e.getKey());
398       for (KeyValue kv: e.getValue()) {
399         row.addCell(new CellModel(kv));
400       }
401       model.addRow(row);
402     }
403 
404     // build path for multiput
405     StringBuilder sb = new StringBuilder();
406     sb.append('/');
407     sb.append(Bytes.toStringBinary(name));
408     sb.append("/$multiput"); // can be any nonexistent row
409     for (int i = 0; i < maxRetries; i++) {
410       Response response = client.put(sb.toString(), Constants.MIMETYPE_PROTOBUF,
411         model.createProtobufOutput());
412       int code = response.getCode();
413       switch (code) {
414       case 200:
415         return;
416       case 509:
417         try {
418           Thread.sleep(sleepTime);
419         } catch (InterruptedException e) { }
420         break;
421       default:
422         throw new IOException("multiput request failed with " + code);
423       }
424     }
425     throw new IOException("multiput request timed out");
426   }

463     public Scanner(Scan scan) throws IOException {
464       ScannerModel model;
465       try {
466         model = ScannerModel.fromScan(scan);
467       } catch (Exception e) {
468         throw new IOException(e);
469       }
470       StringBuffer sb = new StringBuffer();
471       sb.append('/');
472       sb.append(Bytes.toStringBinary(name));
473       sb.append('/');
474       sb.append("scanner");
475       for (int i = 0; i < maxRetries; i++) {
476         Response response = client.post(sb.toString(),
477           Constants.MIMETYPE_PROTOBUF, model.createProtobufOutput());
478         int code = response.getCode();
479         switch (code) {
480         case 201:
481           uri = response.getLocation();
482           return;
483         case 509:
484           try {
485             Thread.sleep(sleepTime);
486           } catch (InterruptedException e) { }
487           break;
488         default:
489           throw new IOException("scan request failed with " + code);
490         }
491       }
492       throw new IOException("scan request timed out");
493     }

495     @Override
496     public Result[] next(int nbRows) throws IOException {
497       StringBuilder sb = new StringBuilder(uri);
498       sb.append("?n=");
499       sb.append(nbRows);
500       for (int i = 0; i < maxRetries; i++) {
501         Response response = client.get(sb.toString(),
502           Constants.MIMETYPE_PROTOBUF);
503         int code = response.getCode();
504         switch (code) {
505         case 200:
506           CellSetModel model = new CellSetModel();
507           model.getObjectFromMessage(response.getBody());
508           return buildResultFromModel(model);
509         case 204:
510         case 206:
511           return null;
512         case 509:
513           try {
514             Thread.sleep(sleepTime);
515           } catch (InterruptedException e) { }
516           break;
517         default:
518           throw new IOException("scanner.next request failed with " + code);
519         }
520       }
521       throw new IOException("scanner.next request timed out");
522     }
523 
524     @Override
525     public Result next() throws IOException {
526       Result[] results = next(1);
527       if (results == null || results.length < 1) {
528         return null;
529       }
530       return results[0];
531     }

618   public boolean checkAndPut(byte[] row, byte[] family, byte[] qualifier,
619       byte[] value, Put put) throws IOException {
620     // column to check-the-value
621     put.add(new KeyValue(row, family, qualifier, value));
622 
623     CellSetModel model = buildModelFromPut(put);
624     StringBuilder sb = new StringBuilder();
625     sb.append('/');
626     sb.append(Bytes.toStringBinary(name));
627     sb.append('/');
628     sb.append(Bytes.toStringBinary(put.getRow()));
629     sb.append("?check=put");
630 
631     for (int i = 0; i < maxRetries; i++) {
632       Response response = client.put(sb.toString(),
633         Constants.MIMETYPE_PROTOBUF, model.createProtobufOutput());
634       int code = response.getCode();
635       switch (code) {
636       case 200:
637         return true;
638       case 304: // NOT-MODIFIED
639         return false;
640       case 509:
641         try {
642           Thread.sleep(sleepTime);
643         } catch (final InterruptedException e) {
644         }
645         break;
646       default:
647         throw new IOException("checkAndPut request failed with " + code);
648       }
649     }
650     throw new IOException("checkAndPut request timed out");
651   }
