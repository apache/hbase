85    /**
86     * Test that if we fail a flush, abort gets set on close.
87     * @see <a href="https://issues.apache.org/jira/browse/HBASE-4270">HBASE-4270</a>
88     * @throws IOException
89     * @throws NodeExistsException
90     * @throws KeeperException
91     */
92    @Test public void testFailedFlushAborts()
93    throws IOException, NodeExistsException, KeeperException {
94      final Server server = new MockServer(HTU, false);
95      final RegionServerServices rss = new MockRegionServerServices();
96      HTableDescriptor htd = TEST_HTD;
97      final HRegionInfo hri =
98        new HRegionInfo(htd.getName(), HConstants.EMPTY_END_ROW,
99          HConstants.EMPTY_END_ROW);
100     HRegion region =
101       HRegion.createHRegion(hri, HTU.getDataTestDir(),
102         HTU.getConfiguration(), htd);
103     try {
104       assertNotNull(region);
105       // Spy on the region so can throw exception when close is called.
106       HRegion spy = Mockito.spy(region);
107       final boolean abort = false;
108       Mockito.when(spy.close(abort)).
109       thenThrow(new RuntimeException("Mocked failed close!"));
110       // The CloseRegionHandler will try to get an HRegion that corresponds
111       // to the passed hri -- so insert the region into the online region Set.
112       rss.addToOnlineRegions(spy);
113       // Assert the Server is NOT stopped before we call close region.
114       assertFalse(server.isStopped());
115       CloseRegionHandler handler =
116           new CloseRegionHandler(server, rss, hri, false, false, -1);
117       boolean throwable = false;
118       try {
119         handler.process();
120       } catch (Throwable t) {
121         throwable = true;
122       } finally {
123         assertTrue(throwable);
124         // Abort calls stop so stopped flag should be set.
125         assertTrue(server.isStopped());
126       }
127     } finally {
128       HRegion.closeHRegion(region);
129     }
130   }

132      /**
133       * Test if close region can handle ZK closing node version mismatch
134       * @throws IOException
135       * @throws NodeExistsException
136       * @throws KeeperException
137      * @throws DeserializationException 
138       */
139      @Test public void testZKClosingNodeVersionMismatch()
140      throws IOException, NodeExistsException, KeeperException, DeserializationException {
141        final Server server = new MockServer(HTU);
142        final MockRegionServerServices rss = new MockRegionServerServices();
143        rss.setFileSystem(HTU.getTestFileSystem());
144    
145        HTableDescriptor htd = TEST_HTD;
146        final HRegionInfo hri = TEST_HRI;
147    
148        // open a region first so that it can be closed later
149        OpenRegion(server, rss, htd, hri);
150    
151        // close the region
152        // Create it CLOSING, which is what Master set before sending CLOSE RPC
153        int versionOfClosingNode = ZKAssign.createNodeClosing(server.getZooKeeper(),
154          hri, server.getServerName());
155    
156        // The CloseRegionHandler will validate the expected version
157        // Given it is set to invalid versionOfClosingNode+1,
158        // CloseRegionHandler should be M_ZK_REGION_CLOSING
159        CloseRegionHandler handler =
160          new CloseRegionHandler(server, rss, hri, false, true,
161          versionOfClosingNode+1);
162        handler.process();
163    
164        // Handler should remain in M_ZK_REGION_CLOSING
165        RegionTransition rt =
166          RegionTransition.parseFrom(ZKAssign.getData(server.getZooKeeper(), hri.getEncodedName()));
167        assertTrue(rt.getEventType().equals(EventType.M_ZK_REGION_CLOSING ));
168      }

207      private void OpenRegion(Server server, RegionServerServices rss,
208          HTableDescriptor htd, HRegionInfo hri)
209      throws IOException, NodeExistsException, KeeperException, DeserializationException {
210        // Create it OFFLINE node, which is what Master set before sending OPEN RPC
211        ZKAssign.createNodeOffline(server.getZooKeeper(), hri, server.getServerName());
212        OpenRegionHandler openHandler = new OpenRegionHandler(server, rss, hri, htd);
213        rss.getRegionsInTransitionInRS().put(hri.getEncodedNameAsBytes(), Boolean.TRUE);
214        openHandler.process();
215        // This parse is not used?
216        RegionTransition.parseFrom(ZKAssign.getData(server.getZooKeeper(), hri.getEncodedName()));
217        // delete the node, which is what Master do after the region is opened
218        ZKAssign.deleteNode(server.getZooKeeper(), hri.getEncodedName(), EventType.RS_ZK_REGION_OPENED);
219      }
