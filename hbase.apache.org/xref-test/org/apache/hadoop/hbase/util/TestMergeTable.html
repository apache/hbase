40  /**
41   * Tests merging a normal table's regions
42   */
43  @Category(MediumTests.class)
44  public class TestMergeTable {
45    private static final Log LOG = LogFactory.getLog(TestMergeTable.class);
46    private final HBaseTestingUtility UTIL = new HBaseTestingUtility();
47    private static final byte [] COLUMN_NAME = Bytes.toBytes("contents");
48    private static final byte [] VALUE;
49    static {
50      // We will use the same value for the rows as that is not really important here
51      String partialValue = String.valueOf(System.currentTimeMillis());
52      StringBuilder val = new StringBuilder();
53      while (val.length() < 1024) {
54        val.append(partialValue);
55      }
56      VALUE = Bytes.toBytes(val.toString());
57    }
58  
59    /**
60     * Test merge.
61     * Hand-makes regions of a mergeable size and adds the hand-made regions to
62     * hand-made meta.  The hand-made regions are created offline.  We then start
63     * up mini cluster, disables the hand-made table and starts in on merging.
64     * @throws Exception
65     */
66    @Test (timeout=300000) public void testMergeTable() throws Exception {
67      // Table we are manually creating offline.
68      HTableDescriptor desc = new HTableDescriptor(Bytes.toBytes("test"));
69      desc.addFamily(new HColumnDescriptor(COLUMN_NAME));
70  
71      // Set maximum regionsize down.
72      UTIL.getConfiguration().setLong(HConstants.HREGION_MAX_FILESIZE, 64L * 1024L * 1024L);
73      // Make it so we don't split.
74      UTIL.getConfiguration().setInt("hbase.regionserver.regionSplitLimit", 0);
75      // Startup hdfs.  Its in here we'll be putting our manually made regions.
76      UTIL.startMiniDFSCluster(1);
77      // Create hdfs hbase rootdir.
78      Path rootdir = UTIL.createRootDir();
79      FileSystem fs = FileSystem.get(UTIL.getConfiguration());
80      if (fs.exists(rootdir)) {
81        if (fs.delete(rootdir, true)) {
82          LOG.info("Cleaned up existing " + rootdir);
83        }
84      }
85  
86      // Now create three data regions: The first is too large to merge since it
87      // will be > 64 MB in size. The second two will be smaller and will be
88      // selected for merging.
89  
90      // To ensure that the first region is larger than 64MB we need to write at
91      // least 65536 rows. We will make certain by writing 70000
92      byte [] row_70001 = Bytes.toBytes("row_70001");
93      byte [] row_80001 = Bytes.toBytes("row_80001");
94  
95      // Create regions and populate them at same time.  Create the tabledir
96      // for them first.
97      FSTableDescriptors.createTableDescriptor(fs, rootdir, desc);
98      HRegion [] regions = {
99        createRegion(desc, null, row_70001, 1, 70000, rootdir),
100       createRegion(desc, row_70001, row_80001, 70001, 10000, rootdir),
101       createRegion(desc, row_80001, null, 80001, 11000, rootdir)
102     };
103 
104     // Now create the root and meta regions and insert the data regions
105     // created above into .META.
106     setupROOTAndMeta(rootdir, regions);
107     try {
108       LOG.info("Starting mini zk cluster");
109       UTIL.startMiniZKCluster();
110       LOG.info("Starting mini hbase cluster");
111       UTIL.startMiniHBaseCluster(1, 1);
112       Configuration c = new Configuration(UTIL.getConfiguration());
113       CatalogTracker ct = new CatalogTracker(c);
114       ct.start();
115       List<HRegionInfo> originalTableRegions =
116         MetaReader.getTableRegions(ct, desc.getName());
117       LOG.info("originalTableRegions size=" + originalTableRegions.size() +
118         "; " + originalTableRegions);
119       HBaseAdmin admin = new HBaseAdmin(c);
120       admin.disableTable(desc.getName());
121       HMerge.merge(c, FileSystem.get(c), desc.getName());
122       List<HRegionInfo> postMergeTableRegions =
123         MetaReader.getTableRegions(ct, desc.getName());
124       LOG.info("postMergeTableRegions size=" + postMergeTableRegions.size() +
125         "; " + postMergeTableRegions);
126       assertTrue("originalTableRegions=" + originalTableRegions.size() +
127         ", postMergeTableRegions=" + postMergeTableRegions.size(),
128         postMergeTableRegions.size() < originalTableRegions.size());
129       LOG.info("Done with merge");
130     } finally {
131       UTIL.shutdownMiniCluster();
132       LOG.info("After cluster shutdown");
133     }
134   }
135 
136   private HRegion createRegion(final HTableDescriptor desc,
137       byte [] startKey, byte [] endKey, int firstRow, int nrows, Path rootdir)
138   throws IOException {
139     HRegionInfo hri = new HRegionInfo(desc.getName(), startKey, endKey);
140     HRegion region = HRegion.createHRegion(hri, rootdir, UTIL.getConfiguration(), desc);
141     LOG.info("Created region " + region.getRegionNameAsString());
142     for(int i = firstRow; i < firstRow + nrows; i++) {
143       Put put = new Put(Bytes.toBytes("row_" + String.format("%1$05d", i)));
144       put.setWriteToWAL(false);
145       put.add(COLUMN_NAME, null,  VALUE);
146       region.put(put);
147       if (i % 10000 == 0) {
148         LOG.info("Flushing write #" + i);
149         region.flushcache();
150       }
151     }
152     HRegion.closeHRegion(region);
153     return region;
154   }

156   protected void setupROOTAndMeta(Path rootdir, final HRegion [] regions)
157   throws IOException {
158     HRegion root =
159       HRegion.createHRegion(HRegionInfo.ROOT_REGIONINFO, rootdir,
160           UTIL.getConfiguration(), HTableDescriptor.ROOT_TABLEDESC);
161     HRegion meta =
162       HRegion.createHRegion(HRegionInfo.FIRST_META_REGIONINFO, rootdir,
163       UTIL.getConfiguration(), HTableDescriptor.META_TABLEDESC);
164     HRegion.addRegionToMETA(root, meta);
165     for (HRegion r: regions) {
166       HRegion.addRegionToMETA(meta, r);
167     }
168     HRegion.closeHRegion(meta);
169     HRegion.closeHRegion(root);
170   }
