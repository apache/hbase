132   @Test (timeout = 300000) public void testRSSplitEphemeralsDisappearButDaughtersAreOnlinedAfterShutdownHandling()
133   throws IOException, InterruptedException, NodeExistsException, KeeperException, DeserializationException, ServiceException {
134     final byte [] tableName =
135       Bytes.toBytes("ephemeral");
136 
137     // Create table then get the single region for our new table.
138     HTable t = TESTING_UTIL.createTable(tableName, HConstants.CATALOG_FAMILY);
139 
140     List<HRegion> regions = cluster.getRegions(tableName);
141     HRegionInfo hri = getAndCheckSingleTableRegion(regions);
142 
143     int tableRegionIndex = ensureTableRegionNotOnSameServerAsMeta(admin, hri);
144 
145     // Turn off balancer so it doesn't cut in and mess up our placements.
146     this.admin.setBalancerRunning(false, true);
147     // Turn off the meta scanner so it don't remove parent on us.
148     cluster.getMaster().setCatalogJanitorEnabled(false);
149     try {
150       // Add a bit of load up into the table so splittable.
151       TESTING_UTIL.loadTable(t, HConstants.CATALOG_FAMILY);
152       // Get region pre-split.
153       HRegionServer server = cluster.getRegionServer(tableRegionIndex);
154       printOutRegions(server, "Initial regions: ");
155       int regionCount = ProtobufUtil.getOnlineRegions(server).size();
156       // Now, before we split, set special flag in master, a flag that has
157       // it FAIL the processing of split.
158       SplitRegionHandler.TEST_SKIP = true;
159       // Now try splitting and it should work.
160       split(hri, server, regionCount);
161       // Get daughters
162       List<HRegion> daughters = checkAndGetDaughters(tableName);
163       // Assert the ephemeral node is up in zk.
164       String path = ZKAssign.getNodeName(TESTING_UTIL.getZooKeeperWatcher(),
165         hri.getEncodedName());
166       Stat stats =
167         TESTING_UTIL.getZooKeeperWatcher().getRecoverableZooKeeper().exists(path, false);
168       LOG.info("EPHEMERAL NODE BEFORE SERVER ABORT, path=" + path + ", stats=" + stats);
169       RegionTransition rt =
170         RegionTransition.parseFrom(ZKAssign.getData(TESTING_UTIL.getZooKeeperWatcher(),
171           hri.getEncodedName()));
172       // State could be SPLIT or SPLITTING.
173       assertTrue(rt.getEventType().equals(EventType.RS_ZK_REGION_SPLIT) ||
174         rt.getEventType().equals(EventType.RS_ZK_REGION_SPLITTING));
175       // Now crash the server
176       cluster.abortRegionServer(tableRegionIndex);
177       waitUntilRegionServerDead();
178 
179       // Wait till regions are back on line again.
180       while(cluster.getRegions(tableName).size() < daughters.size()) {
181         LOG.info("Waiting for repair to happen");
182         Thread.sleep(1000);
183       }
184       // Assert daughters are online.
185       regions = cluster.getRegions(tableName);
186       for (HRegion r: regions) {
187         assertTrue(daughters.contains(r));
188       }
189       // Finally assert that the ephemeral SPLIT znode was cleaned up.
190       for (int i=0; i<100; i++) {
191         // wait a bit (10s max) for the node to disappear
192         stats = TESTING_UTIL.getZooKeeperWatcher().getRecoverableZooKeeper().exists(path, false);
193         if (stats == null) break;
194         Thread.sleep(100);
195       }
196       LOG.info("EPHEMERAL NODE AFTER SERVER ABORT, path=" + path + ", stats=" + stats);
197       assertTrue(stats == null);
198     } finally {
199       // Set this flag back.
200       SplitRegionHandler.TEST_SKIP = false;
201       admin.setBalancerRunning(true, false);
202       cluster.getMaster().setCatalogJanitorEnabled(true);
203     }
204   }

858   /**
859    * Ensure single table region is not on same server as the single .META. table
860    * region.
861    * @param admin
862    * @param hri
863    * @return Index of the server hosting the single table region
864    * @throws UnknownRegionException
865    * @throws MasterNotRunningException
866    * @throws ZooKeeperConnectionException
867    * @throws InterruptedException
868    */
869   private int ensureTableRegionNotOnSameServerAsMeta(final HBaseAdmin admin,
870       final HRegionInfo hri)
871   throws UnknownRegionException, MasterNotRunningException,
872   ZooKeeperConnectionException, InterruptedException {
873     MiniHBaseCluster cluster = TESTING_UTIL.getMiniHBaseCluster();
874     // Now make sure that the table region is not on same server as that hosting
875     // .META.  We don't want .META. replay polluting our test when we later crash
876     // the table region serving server.
877     int metaServerIndex = cluster.getServerWithMeta();
878     assertTrue(metaServerIndex != -1);
879     HRegionServer metaRegionServer = cluster.getRegionServer(metaServerIndex);
880     int tableRegionIndex = cluster.getServerWith(hri.getRegionName());
881     assertTrue(tableRegionIndex != -1);
882     HRegionServer tableRegionServer = cluster.getRegionServer(tableRegionIndex);
883     if (metaRegionServer.getServerName().equals(tableRegionServer.getServerName())) {
884       HRegionServer hrs = getOtherRegionServer(cluster, metaRegionServer);
885       assertNotNull(hrs);
886       assertNotNull(hri);
887       LOG.
888         info("Moving " + hri.getRegionNameAsString() + " to " +
889         hrs.getServerName() + "; metaServerIndex=" + metaServerIndex);
890       admin.move(hri.getEncodedNameAsBytes(),
891         Bytes.toBytes(hrs.getServerName().toString()));
892     }
893     // Wait till table region is up on the server that is NOT carrying .META..
894     while (true) {
895       tableRegionIndex = cluster.getServerWith(hri.getRegionName());
896       if (tableRegionIndex != -1 && tableRegionIndex != metaServerIndex) break;
897       LOG.debug("Waiting on region move off the .META. server; current index " +
898         tableRegionIndex + " and metaServerIndex=" + metaServerIndex);
899       Thread.sleep(100);
900     }
901     // Verify for sure table region is not on same server as .META.
902     tableRegionIndex = cluster.getServerWith(hri.getRegionName());
903     assertTrue(tableRegionIndex != -1);
904     assertNotSame(metaServerIndex, tableRegionIndex);
905     return tableRegionIndex;
906   }

910    * Can't rely on indexes into list of regionservers since crashed servers
911    * occupy an index.

920       HRegionServer hrs = rst.getRegionServer();
