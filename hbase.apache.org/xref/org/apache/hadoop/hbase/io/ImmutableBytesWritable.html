33  /**
34   * A byte sequence that is usable as a key or value.  Based on
35   * {@link org.apache.hadoop.io.BytesWritable} only this class is NOT resizable
36   * and DOES NOT distinguish between the size of the seqeunce and the current
37   * capacity as {@link org.apache.hadoop.io.BytesWritable} does. Hence its
38   * comparatively 'immutable'. When creating a new instance of this class,
39   * the underlying byte [] is not copied, just referenced.  The backing
40   * buffer is accessed when we go to serialize.
41   */
42  @InterfaceAudience.Public
43  @InterfaceStability.Stable
44  public class ImmutableBytesWritable
45  implements WritableComparable<ImmutableBytesWritable> {
46    private byte[] bytes;
47    private int offset;
48    private int length;
49  
50    /**
51     * Create a zero-size sequence.
52     */
53    public ImmutableBytesWritable() {
54      super();
55    }
56  
57    /**
58     * Create a ImmutableBytesWritable using the byte array as the initial value.
59     * @param bytes This array becomes the backing storage for the object.
60     */
61    public ImmutableBytesWritable(byte[] bytes) {
62      this(bytes, 0, bytes.length);
63    }
64  
65    /**
66     * Set the new ImmutableBytesWritable to the contents of the passed
67     * <code>ibw</code>.
68     * @param ibw the value to set this ImmutableBytesWritable to.
69     */
70    public ImmutableBytesWritable(final ImmutableBytesWritable ibw) {
71      this(ibw.get(), 0, ibw.getSize());
72    }

75     * Set the value to a given byte range
76     * @param bytes the new byte range to set to

148   public void readFields(final DataInput in) throws IOException {
149     this.length = in.readInt();
150     this.bytes = new byte[this.length];
151     in.readFully(this.bytes, 0, this.length);
152     this.offset = 0;
153   }
154 
155   public void write(final DataOutput out) throws IOException {
156     out.writeInt(this.length);
157     out.write(this.bytes, this.offset, this.length);
158   }

161   @Override
162   public int hashCode() {
163     int hash = 1;
164     for (int i = offset; i < offset + length; i++)
165       hash = (31 * hash) + (int)bytes[i];
166     return hash;

198     if (right_obj instanceof byte []) {
199       return compareTo((byte [])right_obj) == 0;
200     }

202       return compareTo((ImmutableBytesWritable)right_obj) == 0;

207   /**
208    * @see java.lang.Object#toString()
209    */
210   @Override
211   public String toString() {
212     StringBuilder sb = new StringBuilder(3*this.bytes.length);
213     for (int idx = offset; idx < offset + length; idx++) {
214       // if not the first, put a blank separator in
215       if (idx != offset) {
216         sb.append(' ');
217       }
218       String num = Integer.toHexString(bytes[idx]);
219       // if it is only one digit, add a leading 0.
220       if (num.length() < 2) {
221         sb.append('0');
222       }
223       sb.append(num);
224     }
225     return sb.toString();
226   }
