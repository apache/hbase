62    /**
63     * Gives the maximum for a given combination of column qualifier and column
64     * family, in the given row range as defined in the Scan object. In its
65     * current implementation, it takes one column family and one column qualifier
66     * (if provided). In case of null column qualifier, maximum value for the
67     * entire column family will be returned.
68     */
69    @Override
70    public void getMax(RpcController controller, AggregateArgument request,
71        RpcCallback<AggregateResponse> done) {
72      InternalScanner scanner = null;
73      AggregateResponse response = null;
74      T max = null;
75      try {
76        ColumnInterpreter<T, S> ci = constructColumnInterpreterFromRequest(request);
77        T temp;
78        Scan scan = ProtobufUtil.toScan(request.getScan());
79        scanner = env.getRegion().getScanner(scan);
80        List<KeyValue> results = new ArrayList<KeyValue>();
81        byte[] colFamily = scan.getFamilies()[0];
82        NavigableSet<byte[]> qualifiers = scan.getFamilyMap().get(colFamily);
83        byte[] qualifier = null;
84        if (qualifiers != null && !qualifiers.isEmpty()) {
85          qualifier = qualifiers.pollFirst();
86        }
87        // qualifier can be null.
88        boolean hasMoreRows = false;
89        do {
90          hasMoreRows = scanner.next(results);
91          for (KeyValue kv : results) {
92            temp = ci.getValue(colFamily, qualifier, kv);
93            max = (max == null || (temp != null && ci.compare(temp, max) > 0)) ? temp : max;
94          }
95          results.clear();
96        } while (hasMoreRows);
97        if (max != null) {
98          AggregateResponse.Builder builder = AggregateResponse.newBuilder();
99          builder.addFirstPart(ci.getProtoForCellType(max));
100         response = builder.build();
101       }
102     } catch (IOException e) {
103       ResponseConverter.setControllerException(controller, e);
104     } finally {
105       if (scanner != null) {
106         try {
107           scanner.close();
108         } catch (IOException ignored) {}
109       }
110     }
111     log.info("Maximum from this region is "
112         + env.getRegion().getRegionNameAsString() + ": " + max);
113     done.run(response);
114   }

230     AggregateResponse response = null;
231     long counter = 0l;
232     List<KeyValue> results = new ArrayList<KeyValue>();
233     InternalScanner scanner = null;
234     try {
235       Scan scan = ProtobufUtil.toScan(request.getScan());
236       byte[] colFamily = scan.getFamilies()[0];
237       NavigableSet<byte[]> qualifiers = scan.getFamilyMap().get(colFamily);
238       byte[] qualifier = null;
239       if (qualifiers != null && !qualifiers.isEmpty()) {
240         qualifier = qualifiers.pollFirst();
241       }
242       if (scan.getFilter() == null && qualifier == null)

244       scanner = env.getRegion().getScanner(scan);
245       boolean hasMoreRows = false;

252       } while (hasMoreRows);
253       ByteBuffer bb = ByteBuffer.allocate(8).putLong(counter);
254       bb.rewind();

338    * Scan object. In its current implementation, it takes one column family and

346     InternalScanner scanner = null;
347     AggregateResponse response = null;
348     try {
349       ColumnInterpreter<T, S> ci = constructColumnInterpreterFromRequest(request);
350       S sumVal = null, sumSqVal = null, tempVal = null;
351       long rowCountVal = 0l;
352       Scan scan = ProtobufUtil.toScan(request.getScan());
353       scanner = env.getRegion().getScanner(scan);
354       byte[] colFamily = scan.getFamilies()[0];

399   /**
400    * Gives a List containing sum of values and sum of weights.
401    * It is computed for the combination of column
402    * family and column qualifier(s) in the given row range as defined in the
403    * Scan object. In its current implementation, it takes one column family and
404    * two column qualifiers. The first qualifier is for values column and 
405    * the second qualifier (optional) is for weight column.
406    */
407   @Override
408   public void getMedian(RpcController controller, AggregateArgument request,
409       RpcCallback<AggregateResponse> done) {
410     AggregateResponse response = null;
411     InternalScanner scanner = null;
412     try {
413       ColumnInterpreter<T, S> ci = constructColumnInterpreterFromRequest(request);
414       S sumVal = null, sumWeights = null, tempVal = null, tempWeight = null;
415       Scan scan = ProtobufUtil.toScan(request.getScan());
416       scanner = env.getRegion().getScanner(scan);
417       byte[] colFamily = scan.getFamilies()[0];
418       NavigableSet<byte[]> qualifiers = scan.getFamilyMap().get(colFamily);
419       byte[] valQualifier = null, weightQualifier = null;
420       if (qualifiers != null && !qualifiers.isEmpty()) {
421         valQualifier = qualifiers.pollFirst();
422         // if weighted median is requested, get qualifier for the weight column
423         weightQualifier = qualifiers.pollLast();
424       }
425       List<KeyValue> results = new ArrayList<KeyValue>();
426 
427       boolean hasMoreRows = false;
428     
429       do {
430         tempVal = null;
431         tempWeight = null;
432         hasMoreRows = scanner.next(results);
433         for (KeyValue kv : results) {
434           tempVal = ci.add(tempVal, ci.castToReturnType(ci.getValue(colFamily,
435               valQualifier, kv)));
436           if (weightQualifier != null) {
437             tempWeight = ci.add(tempWeight,
438                 ci.castToReturnType(ci.getValue(colFamily, weightQualifier, kv)));
439           }
440         }
441         results.clear();
442         sumVal = ci.add(sumVal, tempVal);
443         sumWeights = ci.add(sumWeights, tempWeight);
444       } while (hasMoreRows);
445       ByteString first_sumVal = ci.getProtoForPromotedType(sumVal);
446       S s = sumWeights == null ? ci.castToReturnType(ci.getMinValue()) : sumWeights;
447       ByteString first_sumWeights = ci.getProtoForPromotedType(s);
448       AggregateResponse.Builder pair = AggregateResponse.newBuilder();
449       pair.addFirstPart(first_sumVal);
450       pair.addFirstPart(first_sumWeights); 
451       response = pair.build();
452     } catch (IOException e) {
453       ResponseConverter.setControllerException(controller, e);
454     } finally {
455       if (scanner != null) {
456         try {
457           scanner.close();
458         } catch (IOException ignored) {}
459       }
460     }
461     done.run(response);
462   }
