45  /**
46   * Status information on the HBase cluster.
47   * <p>
48   * <tt>ClusterStatus</tt> provides clients with information such as:
49   * <ul>
50   * <li>The count and names of region servers in the cluster.</li>
51   * <li>The count and names of dead region servers in the cluster.</li>
52   * <li>The name of the active master for the cluster.</li>
53   * <li>The name(s) of the backup master(s) for the cluster, if they exist.</li>
54   * <li>The average cluster load.</li>
55   * <li>The number of regions deployed on the cluster.</li>
56   * <li>The number of requests since last report.</li>
57   * <li>Detailed region server loading and resource usage information,
58   *  per server and per region.</li>
59   * <li>Regions in transition at master</li>
60   * <li>The unique cluster ID</li>
61   * </ul>
62   */

274    /**
275     * Convert a ClutserStatus to a protobuf ClusterStatus
276     *
277     * @return the protobuf ClusterStatus
278     */
279   public ClusterStatusProtos.ClusterStatus convert() {
280     ClusterStatusProtos.ClusterStatus.Builder builder = ClusterStatusProtos.ClusterStatus.newBuilder();
281     builder.setHbaseVersion(HBaseVersionFileContent.newBuilder().setVersion(getHBaseVersion()));
282 
283     for (Map.Entry<ServerName, ServerLoad> entry : liveServers.entrySet()) {
284       LiveServerInfo.Builder lsi =
285         LiveServerInfo.newBuilder().setServer(ProtobufUtil.toServerName(entry.getKey()));
286       lsi.setServerLoad(entry.getValue().obtainServerLoadPB());
287       builder.addLiveServers(lsi.build());
288     }
289     for (ServerName deadServer : getDeadServerNames()) {
290       builder.addDeadServers(ProtobufUtil.toServerName(deadServer));
291     }
292     for (Map.Entry<String, RegionState> rit : getRegionsInTransition().entrySet()) {
293       ClusterStatusProtos.RegionState rs = rit.getValue().convert();
294       RegionSpecifier.Builder spec =
295         RegionSpecifier.newBuilder().setType(RegionSpecifierType.REGION_NAME);
296       spec.setValue(ByteString.copyFrom(Bytes.toBytes(rit.getKey())));
297 
298       RegionInTransition pbRIT =
299         RegionInTransition.newBuilder().setSpec(spec.build()).setRegionState(rs).build();
300       builder.addRegionsInTransition(pbRIT);
301     }
302     builder.setClusterId(new ClusterId(getClusterId()).convert());
303     for (String coprocessor : getMasterCoprocessors()) {
304       builder.addMasterCoprocessors(HBaseProtos.Coprocessor.newBuilder().setName(coprocessor));
305     }
306     builder.setMaster(
307       ProtobufUtil.toServerName(getMaster()));
308     for (ServerName backup : getBackupMasters()) {
309       builder.addBackupMasters(ProtobufUtil.toServerName(backup));
310     }
311     builder.setBalancerOn(balancerOn);
312     return builder.build();
313   }

315   /**
316    * Convert a protobuf ClusterStatus to a ClusterStatus
317    *
318    * @param proto the protobuf ClusterStatus
319    * @return the converted ClusterStatus
320    */
321   public static ClusterStatus convert(ClusterStatusProtos.ClusterStatus proto) {
322     Map<ServerName, ServerLoad> servers = new HashMap<ServerName, ServerLoad>();
323     for (LiveServerInfo lsi : proto.getLiveServersList()) {
324       servers.put(ProtobufUtil.toServerName(lsi.getServer()), new ServerLoad(lsi.getServerLoad()));
325     }
326     Collection<ServerName> deadServers = new LinkedList<ServerName>();
327     for (HBaseProtos.ServerName sn : proto.getDeadServersList()) {
328       deadServers.add(ProtobufUtil.toServerName(sn));
329     }
330     Collection<ServerName> backupMasters = new LinkedList<ServerName>();
331     for (HBaseProtos.ServerName sn : proto.getBackupMastersList()) {
332       backupMasters.add(ProtobufUtil.toServerName(sn));
333     }
334     final Map<String, RegionState> rit = new HashMap<String, RegionState>();
335     for (RegionInTransition region : proto.getRegionsInTransitionList()) {
336       String key = new String(region.getSpec().getValue().toByteArray());
337       RegionState value = RegionState.convert(region.getRegionState());
338       rit.put(key,value);
339     }
340     final int numMasterCoprocessors = proto.getMasterCoprocessorsCount();
341     final String[] masterCoprocessors = new String[numMasterCoprocessors];
342     for (int i = 0; i < numMasterCoprocessors; i++) {
343       masterCoprocessors[i] = proto.getMasterCoprocessors(i).getName();
344     }
345     return new ClusterStatus(proto.getHbaseVersion().getVersion(),
346       ClusterId.convert(proto.getClusterId()).toString(),servers,deadServers,
347       ProtobufUtil.toServerName(proto.getMaster()),backupMasters,rit,masterCoprocessors,
348       proto.getBalancerOn());
349   }
