20  package org.apache.hadoop.hbase.ipc;
21  
22  public abstract class MetricsHBaseServerSourceFactory {
23    /**
24     * The name of the metrics
25     */
26    static final String METRICS_NAME = "IPC";
27  
28    /**
29     * Description
30     */
31    static final String METRICS_DESCRIPTION = "Metrics about HBase Server IPC";
32  
33    /**
34     * The Suffix of the JMX Context that a MetricsHBaseServerSource will register under.
35     *
36     * JMX_CONTEXT will be created by createContextName(serverClassName) + METRICS_JMX_CONTEXT_SUFFIX
37     */
38    static final String METRICS_JMX_CONTEXT_SUFFIX = ",sub=" + METRICS_NAME;
39  
40    abstract MetricsHBaseServerSource create(String serverName, MetricsHBaseServerWrapper wrapper);
41  
42    /**
43     * From the name of the class that's starting up create the
44     * context that an IPC source should register itself.
45     *
46     * @param serverName The name of the class that's starting up.
47     * @return The Camel Cased context name.
48     */
49    protected static String createContextName(String serverName) {
50      if (serverName.contains("HMaster")) {
51        return "Master";
52      } else if (serverName.contains("HRegion")) {
53        return "RegionServer";
54      }
55      return "IPC";
56    }
57  }
