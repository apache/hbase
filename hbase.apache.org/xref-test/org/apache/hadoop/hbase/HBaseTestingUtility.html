231   /**
232    * Home our data in a dir under {@link #DEFAULT_BASE_TEST_DIRECTORY}.
233    * Give it a random name so can have many concurrent tests running if
234    * we need to.  It needs to amend the {@link #TEST_DIRECTORY_KEY}
235    * System property, as it's what minidfscluster bases
236    * it data dir on.  Moding a System property is not the way to do concurrent
237    * instances -- another instance could grab the temporary
238    * value unintentionally -- but not anything can do about it at moment;
239    * single instance only is how the minidfscluster works.
240    *
241    * We also create the underlying directory for
242    *  hadoop.log.dir, mapred.local.dir and hadoop.tmp.dir, and set the values
243    *  in the conf, and as a system property for hadoop.tmp.dir
244    *
245    * @return The calculated data test build directory, if newly-created.
246    */
247   @Override
248   protected Path setupDataTestDir() {
249     Path testPath = super.setupDataTestDir();
250     if (null == testPath) {
251       return null;
252     }
253 
254     createSubDirAndSystemProperty(
255       "hadoop.log.dir",
256       testPath, "hadoop-log-dir");
257 
258     // This is defaulted in core-default.xml to /tmp/hadoop-${user.name}, but
259     //  we want our own value to ensure uniqueness on the same machine
260     createSubDirAndSystemProperty(
261       "hadoop.tmp.dir",
262       testPath, "hadoop-tmp-dir");
263 
264     // Read and modified in org.apache.hadoop.mapred.MiniMRCluster
265     createSubDir(
266       "mapred.local.dir",
267       testPath, "mapred-local-dir");
268 
269     return testPath;
270   }

279   private void createSubDirAndSystemProperty(
280     String propertyName, Path parent, String subDirName){
281 
282     String sysValue = System.getProperty(propertyName);
283 
284     if (sysValue != null) {
285       // There is already a value set. So we do nothing but hope
286       //  that there will be no conflicts
287       LOG.info("System.getProperty(\""+propertyName+"\") already set to: "+
288         sysValue + " so I do NOT create it in " + parent);
289       String confValue = conf.get(propertyName);
290       if (confValue != null && !confValue.endsWith(sysValue)){
291        LOG.warn(
292          propertyName + " property value differs in configuration and system: "+
293          "Configuration="+confValue+" while System="+sysValue+
294          " Erasing configuration value by system value."
295        );
296       }
297       conf.set(propertyName, sysValue);
298     } else {
299       // Ok, it's not set, so we create it as a subdirectory
300       createSubDir(propertyName, parent, subDirName);
301       System.setProperty(propertyName, conf.get(propertyName));
302     }
303   }

445 
446 
447     this.dfsCluster = new MiniDFSCluster(0, this.conf, servers, true, true,

453     // Do old style too just to be safe.
454     this.conf.set("fs.default.name", fs.getUri().toString());

456     // Wait for the cluster to be totally up
457     this.dfsCluster.waitClusterUp();
458 
459     //reset the test directory for test file system

461

530   /** Enable the short circuit read, unless configured differently.
531    * Set both HBase and HDFS settings, including skipping the hdfs checksum checks.
532    */
533   private void enableShortCircuit() {
534     if (isReadShortCircuitOn()) {
535       String curUser = System.getProperty("user.name");
536       LOG.info("read short circuit is ON for user " + curUser);
537       // read short circuit, for hdfs
538       conf.set("dfs.block.local-path-access.user", curUser);
539       // read short circuit, for hbase
540       conf.setBoolean("dfs.client.read.shortcircuit", true);
541       // Skip checking checksum, for the hdfs client and the datanode
542       conf.setBoolean("dfs.client.read.shortcircuit.skip.checksum", true);
543     } else {
544       LOG.info("read short circuit is OFF");
545     }
546   }

1298   public int createMultiRegions(final Configuration c, final HTable table,
1299       final byte[] columnFamily, byte [][] startKeys)
1300   throws IOException {
1301     Arrays.sort(startKeys, Bytes.BYTES_COMPARATOR);
1302     HTable meta = new HTable(c, HConstants.META_TABLE_NAME);
1303     HTableDescriptor htd = table.getTableDescriptor();
1304     if(!htd.hasFamily(columnFamily)) {
1305       HColumnDescriptor hcd = new HColumnDescriptor(columnFamily);
1306       htd.addFamily(hcd);
1307     }
1308     // remove empty region - this is tricky as the mini cluster during the test

1311     // including the new start region from empty to "bbb". lg

1313     String regionToDeleteInFS = table

1319     for (int i = 0; i < startKeys.length; i++) {
1320       int j = (i + 1) % startKeys.length;
1321       HRegionInfo hri = new HRegionInfo(table.getTableName(),
1322         startKeys[i], startKeys[j]);
1323       MetaEditor.addRegionToMeta(meta, hri);
1324       newRegions.add(hri);
1325       count++;
1326     }
1327     // see comment above, remove "old" (or previous) single region
1328     for (byte[] row : rows) {
1329       LOG.info("createMultiRegions: deleting meta row -> " +
1330         Bytes.toStringBinary(row));
1331       meta.delete(new Delete(row));
1332     }
1333     // remove the "old" region from FS
1334     Path tableDir = new Path(getDefaultRootDirPath().toString()
1335         + System.getProperty("file.separator") + htd.getNameAsString()
1336         + System.getProperty("file.separator") + regionToDeleteInFS);

1339     HConnection conn = table.getConnection();
1340     conn.clearRegionCache();
1341     // assign all the new regions IF table is enabled.
1342     HBaseAdmin admin = getHBaseAdmin();
1343     if (admin.isTableEnabled(table.getTableName())) {

1345         admin.assign(hri.getRegionName());
1346       }

1615   /**
1616    * Expire a ZooKeeper session as recommended in ZooKeeper documentation
1617    * http://wiki.apache.org/hadoop/ZooKeeper/FAQ#A4
1618    * There are issues when doing this:
1619    * [1] http://www.mail-archive.com/dev@zookeeper.apache.org/msg01942.html
1620    * [2] https://issues.apache.org/jira/browse/ZOOKEEPER-1105
1621    *
1622    * @param nodeZK - the ZK watcher to expire
1623    * @param checkStatus - true to check if we can create an HTable with the
1624    *                    current configuration.
1625    */
1626   public void expireSession(ZooKeeperWatcher nodeZK, boolean checkStatus)
1627     throws Exception {
1628     Configuration c = new Configuration(this.conf);
1629     String quorumServers = ZKConfig.getZKQuorumServersString(c);
1630     ZooKeeper zk = nodeZK.getRecoverableZooKeeper().getZooKeeper();
1631     byte[] password = zk.getSessionPasswd();
1632     long sessionID = zk.getSessionId();
1633 
1634     // Expiry seems to be asynchronous (see comment from P. Hunt in [1]),
1635     //  so we create a first watcher to be sure that the
1636     //  event was sent. We expect that if our watcher receives the event
1637     //  other watchers on the same machine will get is as well.
1638     // When we ask to close the connection, ZK does not close it before
1639     //  we receive all the events, so don't have to capture the event, just
1640     //  closing the connection should be enough.
1641     ZooKeeper monitor = new ZooKeeper(quorumServers,
1642       1000, new org.apache.zookeeper.Watcher(){
1643       @Override
1644       public void process(WatchedEvent watchedEvent) {
1645         LOG.info("Monitor ZKW received event="+watchedEvent);
1646       }
1647     } , sessionID, password);
1648 
1649     // Making it expire
1650     ZooKeeper newZK = new ZooKeeper(quorumServers,
1651         1000, EmptyWatcher.instance, sessionID, password);
1652 
1653     //ensure that we have connection to the server before closing down, otherwise
1654     //the close session event will be eaten out before we start CONNECTING state
1655     long start = System.currentTimeMillis();
1656     while (newZK.getState() != States.CONNECTED
1657          && System.currentTimeMillis() - start < 1000) {
1658        Thread.sleep(1);
1659     }
1660     newZK.close();
1661     LOG.info("ZK Closed Session 0x" + Long.toHexString(sessionID));
1662 
1663     // Now closing & waiting to be sure that the clients get it.
1664     monitor.close();
1665 
1666     if (checkStatus) {
1667       new HTable(new Configuration(conf), HConstants.META_TABLE_NAME).close();
1668     }
1669   }
