118   private void waitForServiceToStop(ServiceType service, ServerName serverName, long timeout)
119     throws IOException {
120     LOG.info("Waiting service:" + service + " to stop: " + serverName.getServerName());
121     long start = System.currentTimeMillis();
122 
123     while ((System.currentTimeMillis() - start) < timeout) {
124       if (!clusterManager.isRunning(service, serverName.getHostname())) {
125         return;
126       }
127       Threads.sleep(1000);
128     }
129     throw new IOException("did timeout waiting for service to stop:" + serverName);
130   }

167   @Override
168   public boolean waitForActiveAndReadyMaster(long timeout) throws IOException {
169     long start = System.currentTimeMillis();
170     while (System.currentTimeMillis() - start < timeout) {
171       try {
172         getMasterAdmin();
173         return true;
174       } catch (MasterNotRunningException m) {
175         LOG.warn("Master not started yet " + m);
176       } catch (ZooKeeperConnectionException e) {
177         LOG.warn("Failed to connect to ZK " + e);
178       }
179       Threads.sleep(1000);
180     }
181     return false;
182   }

214   @Override
215   public void restoreClusterStatus(ClusterStatus initial) throws IOException {
216     //TODO: caution: not tested throughly
217     ClusterStatus current = getClusterStatus();
218 
219     //restore masters
220 
221     //check whether current master has changed
222     if (!ServerName.isSameHostnameAndPort(initial.getMaster(), current.getMaster())) {
223       //master has changed, we would like to undo this.
224       //1. Kill the current backups
225       //2. Stop current master
226       //3. Start a master at the initial hostname (if not already running as backup)
227       //4. Start backup masters
228       boolean foundOldMaster = false;
229       for (ServerName currentBackup : current.getBackupMasters()) {
230         if (!ServerName.isSameHostnameAndPort(currentBackup, initial.getMaster())) {
231           stopMaster(currentBackup);
232         } else {
233           foundOldMaster = true;
234         }
235       }
236       stopMaster(current.getMaster());
237       if (foundOldMaster) { //if initial master is not running as a backup
238         startMaster(initial.getMaster().getHostname());
239       }
240       waitForActiveAndReadyMaster(); //wait so that active master takes over
241 
242       //start backup masters
243       for (ServerName backup : initial.getBackupMasters()) {
244         //these are not started in backup mode, but we should already have an active master
245         startMaster(backup.getHostname());
246       }
247     } else {
248       //current master has not changed, match up backup masters
249       HashMap<String, ServerName> initialBackups = new HashMap<String, ServerName>();
250       HashMap<String, ServerName> currentBackups = new HashMap<String, ServerName>();
251 
252       for (ServerName server : initial.getBackupMasters()) {
253         initialBackups.put(server.getHostname(), server);
254       }
255       for (ServerName server : current.getBackupMasters()) {
256         currentBackups.put(server.getHostname(), server);
257       }
258 
259       for (String hostname : Sets.difference(initialBackups.keySet(), currentBackups.keySet())) {
260         startMaster(hostname);
261       }
262 
263       for (String hostname : Sets.difference(currentBackups.keySet(), initialBackups.keySet())) {
264         stopMaster(currentBackups.get(hostname));
265       }
266     }
267 
268     //restore region servers
269     HashMap<String, ServerName> initialServers = new HashMap<String, ServerName>();
270     HashMap<String, ServerName> currentServers = new HashMap<String, ServerName>();
271 
272     for (ServerName server : initial.getServers()) {
273       initialServers.put(server.getHostname(), server);
274     }
275     for (ServerName server : current.getServers()) {
276       currentServers.put(server.getHostname(), server);
277     }
278 
279     for (String hostname : Sets.difference(initialServers.keySet(), currentServers.keySet())) {
280       startRegionServer(hostname);
281     }
282 
283     for (String hostname : Sets.difference(currentServers.keySet(), initialServers.keySet())) {
284       stopRegionServer(currentServers.get(hostname));
285     }
286   }
