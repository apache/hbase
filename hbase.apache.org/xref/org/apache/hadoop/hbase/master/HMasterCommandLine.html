50    private static final String USAGE =
51      "Usage: Master [opts] start|stop|clear\n" +
52      " start  Start Master. If local mode, start Master and RegionServer in same JVM\n" +
53      " stop   Start cluster shutdown; Master signals RegionServer shutdown\n" +
54      " clear  Delete the master znode in ZooKeeper after a master crashes\n "+
55      " where [opts] are:\n" +
56      "   --minServers=<servers>    Minimum RegionServers needed to host user tables.\n" +
57      "   --backup                  Master should start in backup mode";

62      this.masterClass = masterClass;
63    }

66      return USAGE;

70    public int run(String args[]) throws Exception {
71      Options opt = new Options();
72      opt.addOption("minServers", true, "Minimum RegionServers needed to host user tables");
73      opt.addOption("backup", false, "Do not try to become HMaster until the primary fails");
74  
75  
76      CommandLine cmd;
77      try {
78        cmd = new GnuParser().parse(opt, args);
79      } catch (ParseException e) {
80        LOG.error("Could not parse: ", e);
81        usage(null);
82        return -1;
83      }
84  
85  
86      if (cmd.hasOption("minServers")) {
87        String val = cmd.getOptionValue("minServers");
88        getConf().setInt("hbase.regions.server.count.min",
89                    Integer.valueOf(val));
90        LOG.debug("minServers set to " + val);
91      }
92  
93      // check if we are the backup master - override the conf if so
94      if (cmd.hasOption("backup")) {
95        getConf().setBoolean(HConstants.MASTER_TYPE_BACKUP, true);
96      }
97  
98      List<String> remainingArgs = cmd.getArgList();
99      if (remainingArgs.size() != 1) {
100       usage(null);
101       return -1;
102     }
103 
104     String command = remainingArgs.get(0);
105 
106     if ("start".equals(command)) {
107       return startMaster();
108     } else if ("stop".equals(command)) {
109       return stopMaster();
110     } else if ("clear".equals(command)) {
111       return (ZNodeClearer.clear(getConf()) ? 0 : -1);
112     } else {
113       usage("Invalid command: " + command);
114       return -1;
115     }
116   }

118   private int startMaster() {
119     Configuration conf = getConf();
120     try {
121       // If 'local', defer to LocalHBaseCluster instance.  Starts master
122       // and regionserver both in the one JVM.
123       if (LocalHBaseCluster.isLocal(conf)) {
124         final MiniZooKeeperCluster zooKeeperCluster =
125           new MiniZooKeeperCluster();
126         File zkDataPath = new File(conf.get(HConstants.ZOOKEEPER_DATA_DIR));
127         int zkClientPort = conf.getInt(HConstants.ZOOKEEPER_CLIENT_PORT, 0);
128         if (zkClientPort == 0) {
129           throw new IOException("No config value for "
130               + HConstants.ZOOKEEPER_CLIENT_PORT);
131         }
132         zooKeeperCluster.setDefaultClientPort(zkClientPort);
133 
134         // login the zookeeper server principal (if using security)
135         ZKUtil.loginServer(conf, "hbase.zookeeper.server.keytab.file",
136           "hbase.zookeeper.server.kerberos.principal", null);
137 
138         int clientPort = zooKeeperCluster.startup(zkDataPath);
139         if (clientPort != zkClientPort) {
140           String errorMsg = "Could not start ZK at requested port of " +
141             zkClientPort + ".  ZK was started at port: " + clientPort +
142             ".  Aborting as clients (e.g. shell) will not be able to find " +
143             "this ZK quorum.";
144           System.err.println(errorMsg);
145           throw new IOException(errorMsg);
146         }
147         conf.set(HConstants.ZOOKEEPER_CLIENT_PORT,
148                  Integer.toString(clientPort));
149         // Need to have the zk cluster shutdown when master is shutdown.
150         // Run a subclass that does the zk cluster shutdown on its way out.
151         LocalHBaseCluster cluster = new LocalHBaseCluster(conf, 1, 1,
152                                                           LocalHMaster.class, HRegionServer.class);
153         ((LocalHMaster)cluster.getMaster(0)).setZKCluster(zooKeeperCluster);
154         cluster.startup();
155         waitOnMasterThreads(cluster);
156       } else {
157         HMaster master = HMaster.constructMaster(masterClass, conf);
158         if (master.isStopped()) {
159           LOG.info("Won't bring the Master up as a shutdown is requested");
160           return -1;
161         }
162         master.start();
163         master.join();
164         if(master.isAborted())
165           throw new RuntimeException("HMaster Aborted");
166       }
167     } catch (Throwable t) {
168       LOG.error("Failed to start master", t);
169       return -1;
170     }
171     return 0;
172   }

174   private int stopMaster() {
175     HBaseAdmin adm = null;
176     try {
177       Configuration conf = getConf();
178       // Don't try more than once
179       conf.setInt("hbase.client.retries.number", 1);
180       adm = new HBaseAdmin(getConf());
181     } catch (MasterNotRunningException e) {
182       LOG.error("Master not running");
183       return -1;
184     } catch (ZooKeeperConnectionException e) {
185       LOG.error("ZooKeeper not available");
186       return -1;
187     }
188     try {
189       adm.shutdown();
190     } catch (Throwable t) {
191       LOG.error("Failed to stop master", t);
192       return -1;
193     }
194     return 0;
195   }

197   private void waitOnMasterThreads(LocalHBaseCluster cluster) throws InterruptedException{
198     List<JVMClusterUtil.MasterThread> masters = cluster.getMasters();
199     List<JVMClusterUtil.RegionServerThread> regionservers = cluster.getRegionServers();
200 	  
201     if (masters != null) { 
202       for (JVMClusterUtil.MasterThread t : masters) {
203         t.join();
204         if(t.getMaster().isAborted()) {
205           closeAllRegionServerThreads(regionservers);
206           throw new RuntimeException("HMaster Aborted");
207         }
208       }
209     }
210   }
211 
212   private static void closeAllRegionServerThreads(List<JVMClusterUtil.RegionServerThread> regionservers) {
213     for(JVMClusterUtil.RegionServerThread t : regionservers){
214       t.getRegionServer().stop("HMaster Aborted; Bringing down regions servers");
215     }
216   }

218   /*
219    * Version of master that will shutdown the passed zk cluster on its way out.
220    */
221   public static class LocalHMaster extends HMaster {
222     private MiniZooKeeperCluster zkcluster = null;
223 
224     public LocalHMaster(Configuration conf)
225     throws IOException, KeeperException, InterruptedException {
226       super(conf);
227     }
228 
229     @Override
230     public void run() {
231       super.run();
232       if (this.zkcluster != null) {
233         try {
234           this.zkcluster.shutdown();
235         } catch (IOException e) {
236           e.printStackTrace();
237         }
238       }
239     }
240 
241     void setZKCluster(final MiniZooKeeperCluster zkcluster) {
242       this.zkcluster = zkcluster;
243     }
