109  * Provides an interface to manage HBase database table metadata + general
110  * administrative functions.  Use HBaseAdmin to create, drop, list, enable and
111  * disable tables. Use it also to add and drop table column families.

113  * <p>See {@link HTable} to add, update, and delete data from an individual table.

304   private long getPauseTime(int tries) {
305     int triesCount = tries;
306     if (triesCount >= HConstants.RETRY_BACKOFF.length) {
307       triesCount = HConstants.RETRY_BACKOFF.length - 1;
308     }
309     return this.pause * HConstants.RETRY_BACKOFF[triesCount];

330   /**
331    * Creates a new table with the specified number of regions.  The start key
332    * specified will become the end key of the first region of the table, and
333    * the end key specified will become the start key of the last region of the
334    * table (the first region has a null start key and the last region has a
335    * null end key).
336    *
337    * BigInteger math will be used to divide the key range specified into
338    * enough segments to make the required number of total regions.
339    *
340    * Synchronous operation.
341    *
342    * @param desc table descriptor for table
343    * @param startKey beginning of key range
344    * @param endKey end of key range
345    * @param numRegions the total number of regions to create
346    *
347    * @throws IllegalArgumentException if the table name is reserved
348    * @throws MasterNotRunningException if master is not running
349    * @throws TableExistsException if table already exists (If concurrent
350    * threads, the table may have been created between test-for-existence
351    * and attempt-at-creation).
352    * @throws IOException
353    */
354   public void createTable(HTableDescriptor desc, byte [] startKey,
355       byte [] endKey, int numRegions)
356   throws IOException {
357     HTableDescriptor.isLegalTableName(desc.getName());
358     if(numRegions < 3) {
359       throw new IllegalArgumentException("Must create at least three regions");
360     } else if(Bytes.compareTo(startKey, endKey) >= 0) {
361       throw new IllegalArgumentException("Start key must be smaller than end key");
362     }
363     byte [][] splitKeys = Bytes.split(startKey, endKey, numRegions - 3);
364     if(splitKeys == null || splitKeys.length != numRegions - 1) {
365       throw new IllegalArgumentException("Unable to split key range into enough regions");
366     }
367     createTable(desc, splitKeys);
368   }

370   /**
371    * Creates a new table with an initial set of empty regions defined by the
372    * specified split keys.  The total number of regions created will be the
373    * number of split keys plus one. Synchronous operation.
374    * Note : Avoid passing empty split key.
375    *
376    * @param desc table descriptor for table
377    * @param splitKeys array of split keys for the initial regions of the table
378    *
379    * @throws IllegalArgumentException if the table name is reserved, if the split keys
380    * are repeated and if the split key has empty byte array.
381    * @throws MasterNotRunningException if master is not running
382    * @throws TableExistsException if table already exists (If concurrent
383    * threads, the table may have been created between test-for-existence
384    * and attempt-at-creation).
385    * @throws IOException
386    */
387   public void createTable(final HTableDescriptor desc, byte [][] splitKeys)
388   throws IOException {
389     HTableDescriptor.isLegalTableName(desc.getName());
390     try {
391       createTableAsync(desc, splitKeys);
392     } catch (SocketTimeoutException ste) {
393       LOG.warn("Creating " + desc.getNameAsString() + " took too long", ste);
394     }
395     int numRegs = splitKeys == null ? 1 : splitKeys.length + 1;
396     int prevRegCount = 0;
397     boolean doneWithMetaScan = false;
398     for (int tries = 0; tries < this.numRetries * this.retryLongerMultiplier;
399       ++tries) {
400       if (!doneWithMetaScan) {
401         // Wait for new table to come on-line
402         final AtomicInteger actualRegCount = new AtomicInteger(0);
403         MetaScannerVisitor visitor = new MetaScannerVisitorBase() {
404           @Override
405           public boolean processRow(Result rowResult) throws IOException {
406             HRegionInfo info = HRegionInfo.getHRegionInfo(rowResult);
407             if (info == null) {
408               LOG.warn("No serialized HRegionInfo in " + rowResult);
409               return true;
410             }
411             if (!(Bytes.equals(info.getTableName(), desc.getName()))) {
412               return false;
413             }
414             ServerName serverName = HRegionInfo.getServerName(rowResult);
415             // Make sure that regions are assigned to server
416             if (!(info.isOffline() || info.isSplit()) && serverName != null
417                 && serverName.getHostAndPort() != null) {
418               actualRegCount.incrementAndGet();
419             }
420             return true;
421           }
422         };
423         MetaScanner.metaScan(conf, visitor, desc.getName());
424         if (actualRegCount.get() != numRegs) {
425           if (tries == this.numRetries * this.retryLongerMultiplier - 1) {
426             throw new RegionOfflineException("Only " + actualRegCount.get() +
427               " of " + numRegs + " regions are online; retries exhausted.");
428           }
429           try { // Sleep
430             Thread.sleep(getPauseTime(tries));
431           } catch (InterruptedException e) {
432             throw new InterruptedIOException("Interrupted when opening" +
433               " regions; " + actualRegCount.get() + " of " + numRegs +
434               " regions processed so far");
435           }
436           if (actualRegCount.get() > prevRegCount) { // Making progress
437             prevRegCount = actualRegCount.get();
438             tries = -1;
439           }
440         } else {
441           doneWithMetaScan = true;
442           tries = -1;
443         }
444       } else if (isTableEnabled(desc.getName())) {
445         return;
446       } else {
447         try { // Sleep
448           Thread.sleep(getPauseTime(tries));
449         } catch (InterruptedException e) {
450           throw new InterruptedIOException("Interrupted when waiting" +
451             " for table to be enabled; meta scan was done");
452         }
453       }
454     }
455     throw new TableNotEnabledException(
456       "Retries exhausted while still waiting for table: "
457       + desc.getNameAsString() + " to be enabled");
458   }

460   /**
461    * Creates a new table but does not block and wait for it to come online.
462    * Asynchronous operation.  To check if the table exists, use
463    * {@link #isTableAvailable} -- it is not safe to create an HTable
464    * instance to this table before it is available.
465    * Note : Avoid passing empty split key.
466    * @param desc table descriptor for table
467    *
468    * @throws IllegalArgumentException Bad table name, if the split keys
469    * are repeated and if the split key has empty byte array.
470    * @throws MasterNotRunningException if master is not running
471    * @throws TableExistsException if table already exists (If concurrent
472    * threads, the table may have been created between test-for-existence
473    * and attempt-at-creation).
474    * @throws IOException
475    */
476   public void createTableAsync(
477     final HTableDescriptor desc, final byte [][] splitKeys)
478   throws IOException {
479     HTableDescriptor.isLegalTableName(desc.getName());
480     if(splitKeys != null && splitKeys.length > 0) {
481       Arrays.sort(splitKeys, Bytes.BYTES_COMPARATOR);
482       // Verify there are no duplicate split keys
483       byte [] lastKey = null;
484       for(byte [] splitKey : splitKeys) {
485         if (Bytes.compareTo(splitKey, HConstants.EMPTY_BYTE_ARRAY) == 0) {
486           throw new IllegalArgumentException(
487               "Empty split key must not be passed in the split keys.");
488         }
489         if(lastKey != null && Bytes.equals(splitKey, lastKey)) {
490           throw new IllegalArgumentException("All split keys must be unique, " +
491             "found duplicate: " + Bytes.toStringBinary(splitKey) +
492             ", " + Bytes.toStringBinary(lastKey));
493         }
494         lastKey = splitKey;
495       }
496     }
497 
498     execute(new MasterAdminCallable<Void>() {
499       @Override
500       public Void call() throws ServiceException {
501         CreateTableRequest request = RequestConverter.buildCreateTableRequest(desc, splitKeys);
502         masterAdmin.createTable(null, request);
503         return null;
504       }
505     });
506   }

519   /**
520    * Deletes a table.
521    * Synchronous operation.
522    *
523    * @param tableName name of table to delete
524    * @throws IOException if a remote or network exception occurs
525    */
526   public void deleteTable(final byte [] tableName) throws IOException {
527     HTableDescriptor.isLegalTableName(tableName);
528     HRegionLocation firstMetaServer = getFirstMetaServerForTable(tableName);
529     boolean tableExists = true;
530 
531     execute(new MasterAdminCallable<Void>() {
532       @Override
533       public Void call() throws ServiceException {
534         DeleteTableRequest req = RequestConverter.buildDeleteTableRequest(tableName);
535         masterAdmin.deleteTable(null,req);
536         return null;
537       }
538     });
539 
540     // Wait until all regions deleted
541     ClientProtocol server =
542       connection.getClient(firstMetaServer.getHostname(), firstMetaServer.getPort());
543     for (int tries = 0; tries < (this.numRetries * this.retryLongerMultiplier); tries++) {
544       try {
545 
546         Scan scan = MetaReader.getScanForTableName(tableName);
547         scan.addColumn(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER);
548         ScanRequest request = RequestConverter.buildScanRequest(
549           firstMetaServer.getRegionInfo().getRegionName(), scan, 1, true);
550         Result[] values = null;
551         // Get a batch at a time.
552         try {
553           ScanResponse response = server.scan(null, request);
554           values = ResponseConverter.getResults(response);
555         } catch (ServiceException se) {
556           throw ProtobufUtil.getRemoteException(se);
557         }
558 
559         // let us wait until .META. table is updated and
560         // HMaster removes the table from its HTableDescriptors
561         if (values == null || values.length == 0) {
562           tableExists = false;
563           GetTableDescriptorsResponse htds;
564           MasterMonitorKeepAliveConnection master = connection.getKeepAliveMasterMonitor();
565           try {
566             GetTableDescriptorsRequest req =
567               RequestConverter.buildGetTableDescriptorsRequest(null);
568             htds = master.getTableDescriptors(null, req);
569           } catch (ServiceException se) {
570             throw ProtobufUtil.getRemoteException(se);
571           } finally {
572             master.close();
573           }
574           for (TableSchema ts : htds.getTableSchemaList()) {
575             if (Bytes.equals(tableName, ts.getName().toByteArray())) {
576               tableExists = true;
577               break;
578             }
579           }
580           if (!tableExists) {
581             break;
582           }
583         }
584       } catch (IOException ex) {
585         if(tries == numRetries - 1) {           // no more tries left
586           if (ex instanceof RemoteException) {
587             throw ((RemoteException) ex).unwrapRemoteException();
588           }else {
589             throw ex;
590           }
591         }
592       }
593       try {
594         Thread.sleep(getPauseTime(tries));
595       } catch (InterruptedException e) {
596         // continue
597       }
598     }
599 
600     if (tableExists) {
601       throw new IOException("Retries exhausted, it took too long to wait"+
602         " for the table " + Bytes.toString(tableName) + " to be deleted.");
603     }
604     // Delete cached information to prevent clients from using old locations
605     this.connection.clearRegionCache(tableName);
606     LOG.info("Deleted " + Bytes.toString(tableName));
607   }

1732    * @param ct A {@link CatalogTracker} instance (caller of this method usually has one).
1733    * @return a pair of HRegionInfo and ServerName if <code>tableNameOrRegionName</code> is
1734    *  a verified region name (we call {@link  MetaReader#getRegion( CatalogTracker, byte[])}
1735    *  else null.
1736    * Throw an exception if <code>tableNameOrRegionName</code> is null.
1737    * @throws IOException
1738    */

1740       final CatalogTracker ct) throws IOException {
1741     if (tableNameOrRegionName == null) {
1742       throw new IllegalArgumentException("Pass a table name or region name");
1743     }

1745     if (pair == null) {
1746       final AtomicReference<Pair<HRegionInfo, ServerName>> result =
1747         new AtomicReference<Pair<HRegionInfo, ServerName>>(null);
1748       final String encodedName = Bytes.toString(tableNameOrRegionName);
1749       MetaScannerVisitor visitor = new MetaScannerVisitorBase() {
1750         @Override
1751         public boolean processRow(Result data) throws IOException {
1752           HRegionInfo info = HRegionInfo.getHRegionInfo(data);
1753           if (info == null) {
1754             LOG.warn("No serialized HRegionInfo in " + data);
1755             return true;
1756           }
1757           if (!encodedName.equals(info.getEncodedName())) return true;
1758           ServerName sn = HRegionInfo.getServerName(data);
1759           result.set(new Pair<HRegionInfo, ServerName>(info, sn));
1760           return false; // found the region, stop
1761         }
1762       };
1763 
1764       MetaScanner.metaScan(conf, visitor);
1765       pair = result.get();
1766     }
1767     return pair;
1768   }
