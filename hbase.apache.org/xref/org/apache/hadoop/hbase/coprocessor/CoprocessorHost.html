127   /**
128    * Load system coprocessors. Read the class names from configuration.
129    * Called by constructor.
130    */
131   protected void loadSystemCoprocessors(Configuration conf, String confKey) {
132     Class<?> implClass = null;
133 
134     // load default coprocessors from configure file    
135     String[] defaultCPClasses = conf.getStrings(confKey);
136     if (defaultCPClasses == null || defaultCPClasses.length == 0)
137       return;
138 
139     int priority = Coprocessor.PRIORITY_SYSTEM;
140     List<E> configured = new ArrayList<E>();
141     for (String className : defaultCPClasses) {
142       className = className.trim();
143       if (findCoprocessor(className) != null) {
144         continue;
145       }
146       ClassLoader cl = this.getClass().getClassLoader();
147       Thread.currentThread().setContextClassLoader(cl);
148       try {
149         implClass = cl.loadClass(className);
150         configured.add(loadInstance(implClass, Coprocessor.PRIORITY_SYSTEM, conf));
151         LOG.info("System coprocessor " + className + " was loaded " +
152             "successfully with priority (" + priority++ + ").");
153       } catch (ClassNotFoundException e) {
154         LOG.warn("Class " + className + " cannot be found. " +
155             e.getMessage());
156       } catch (IOException e) {
157         LOG.warn("Load coprocessor " + className + " failed. " +
158             e.getMessage());
159       }
160     }
161 
162     // add entire set to the collection for COW efficiency
163     coprocessors.addAll(configured);
164   }

166   /**
167    * Load a coprocessor implementation into the host
168    * @param path path to implementation jar
169    * @param className the main class name
170    * @param priority chaining priority
171    * @param conf configuration for coprocessor
172    * @throws java.io.IOException Exception
173    */
174   @SuppressWarnings("deprecation")
175   public E load(Path path, String className, int priority,
176       Configuration conf) throws IOException {
177     Class<?> implClass = null;
178     LOG.debug("Loading coprocessor class " + className + " with path " + 
179         path + " and priority " + priority);
180 
181     ClassLoader cl = null;
182     if (path == null) {
183       try {
184         implClass = getClass().getClassLoader().loadClass(className);
185       } catch (ClassNotFoundException e) {
186         throw new IOException("No jar path specified for " + className);
187       }
188     } else {
189       // Have we already loaded the class, perhaps from an earlier region open
190       // for the same table?
191       cl = classLoadersCache.get(path);
192       if (cl != null){
193         LOG.debug("Found classloader "+ cl + "for "+path.toString());
194         try {
195           implClass = cl.loadClass(className);
196         } catch (ClassNotFoundException e) {
197           LOG.info("Class " + className + " needs to be loaded from a file - " +
198               path + ".");
199           // go ahead to load from file system.
200         }
201       }
202     }
203 
204     // If not, load
205     if (implClass == null) {
206       if (path == null) {
207         throw new IOException("No jar path specified for " + className);
208       }
209       // copy the jar to the local filesystem
210       if (!path.toString().endsWith(".jar")) {
211         throw new IOException(path.toString() + ": not a jar file?");
212       }
213       FileSystem fs = path.getFileSystem(HBaseConfiguration.create());
214       Path dst = new Path(System.getProperty("java.io.tmpdir") +
215           java.io.File.separator +"." + pathPrefix +
216           "." + className + "." + System.currentTimeMillis() + ".jar");
217       fs.copyToLocalFile(path, dst);
218       File tmpLocal = new File(dst.toString());
219       tmpLocal.deleteOnExit();
220 
221       // TODO: code weaving goes here
222 
223       // TODO: wrap heap allocations and enforce maximum usage limits
224 
225       /* TODO: inject code into loop headers that monitors CPU use and
226          aborts runaway user code */
227 
228       // load the jar and get the implementation main class
229       // NOTE: Path.toURL is deprecated (toURI instead) but the URLClassLoader
230       // unsurprisingly wants URLs, not URIs; so we will use the deprecated
231       // method which returns URLs for as long as it is available
232       List<URL> paths = new ArrayList<URL>();
233       URL url = new File(dst.toString()).getCanonicalFile().toURL();
234       paths.add(url);
235 
236       JarFile jarFile = new JarFile(dst.toString());
237       Enumeration<JarEntry> entries = jarFile.entries();
238       while (entries.hasMoreElements()) {
239         JarEntry entry = entries.nextElement();
240         if (entry.getName().matches("/lib/[^/]+\\.jar")) {
241           File file = new File(System.getProperty("java.io.tmpdir") +
242               java.io.File.separator +"." + pathPrefix +
243               "." + className + "." + System.currentTimeMillis() + "." + entry.getName().substring(5));
244           IOUtils.copyBytes(jarFile.getInputStream(entry), new FileOutputStream(file), conf, true);
245           file.deleteOnExit();
246           paths.add(file.toURL());
247         }
248       }
249       jarFile.close();
250 
251       cl = new CoprocessorClassLoader(paths, this.getClass().getClassLoader());
252       // cache cp classloader as a weak value, will be GC'ed when no reference left
253       ClassLoader prev = classLoadersCache.putIfAbsent(path, cl);
254       if (prev != null) {
255         //lost update race, use already added classloader
256         cl = prev;
257       }
258 
259       try {
260         implClass = cl.loadClass(className);
261       } catch (ClassNotFoundException e) {
262         throw new IOException("Cannot load external coprocessor class " + className, e);
263       }
264     }
265 
266     //load custom code for coprocessor
267     Thread currentThread = Thread.currentThread();
268     ClassLoader hostClassLoader = currentThread.getContextClassLoader();
269     try{
270       // switch temporarily to the thread classloader for custom CP
271       currentThread.setContextClassLoader(cl);
272       E cpInstance = loadInstance(implClass, priority, conf);
273       return cpInstance;
274     } finally {
275       // restore the fresh (host) classloader
276       currentThread.setContextClassLoader(hostClassLoader);
277     }
278   }

292   /**
293    * @param implClass Implementation class
294    * @param priority priority
295    * @param conf configuration
296    * @throws java.io.IOException Exception
297    */
298   public E loadInstance(Class<?> implClass, int priority, Configuration conf)
299       throws IOException {
300     if (!Coprocessor.class.isAssignableFrom(implClass)) {
301       throw new IOException("Configured class " + implClass.getName() + " must implement "
302           + Coprocessor.class.getName() + " interface ");
303     }
304 
305     // create the instance
306     Coprocessor impl;
307     Object o = null;
308     try {
309       o = implClass.newInstance();
310       impl = (Coprocessor)o;
311     } catch (InstantiationException e) {
312       throw new IOException(e);
313     } catch (IllegalAccessException e) {
314       throw new IOException(e);
315     }
316     // create the environment
317     E env = createEnvironment(implClass, impl, priority, ++loadSequence, conf);
318     if (env instanceof Environment) {
319       ((Environment)env).startup();
320     }
321     // HBASE-4014: maintain list of loaded coprocessors for later crash analysis
322     // if server (master or regionserver) aborts.
323     coprocessorNames.add(implClass.getName());
324     return env;
325   }

344    * @param className the class name
345    * @return the coprocessor, or null if not found

347   public Coprocessor findCoprocessor(String className) {
348     // initialize the coprocessors
349     for (E env: coprocessors) {
350       if (env.getInstance().getClass().getName().equals(className) ||
351           env.getInstance().getClass().getSimpleName().equals(className)) {
352         return env.getInstance();
353       }

355     return null;

358   /**
359    * Retrieves the set of classloaders used to instantiate Coprocessor classes defined in external
360    * jar files.
361    * @return A set of ClassLoader instances
362    */
363   Set<ClassLoader> getExternalClassLoaders() {
364     Set<ClassLoader> externalClassLoaders = new HashSet<ClassLoader>();
365     final ClassLoader systemClassLoader = this.getClass().getClassLoader();
366     for (E env : coprocessors) {
367       ClassLoader cl = env.getInstance().getClass().getClassLoader();
368       if (cl != systemClassLoader ){
369         //do not include system classloader
370         externalClassLoaders.add(cl);
371       }
372     }
373     return externalClassLoaders;
374   }

392   /**
393    * Environment priority comparator.
394    * Coprocessors are chained in sorted order.
395    */
396   static class EnvironmentPriorityComparator
397       implements Comparator<CoprocessorEnvironment> {
398     public int compare(final CoprocessorEnvironment env1,
399         final CoprocessorEnvironment env2) {
400       if (env1.getPriority() < env2.getPriority()) {
401         return -1;
402       } else if (env1.getPriority() > env2.getPriority()) {
403         return 1;
404       }
405       if (env1.getLoadSequence() < env2.getLoadSequence()) {
406         return -1;
407       } else if (env1.getLoadSequence() > env2.getLoadSequence()) {
408         return 1;
409       }
410       return 0;
411     }
412   }

654     /** The coprocessor */
655     public Coprocessor impl;
656     /** Chaining priority */
657     protected int priority = Coprocessor.PRIORITY_USER;
658     /** Current coprocessor state */
659     Coprocessor.State state = Coprocessor.State.UNINSTALLED;
660     /** Accounting for tables opened by the coprocessor */
661     protected List<HTableInterface> openTables =
662       Collections.synchronizedList(new ArrayList<HTableInterface>());
663     private int seq;

666     /**
667      * Constructor
668      * @param impl the coprocessor instance
669      * @param priority chaining priority
670      */
671     public Environment(final Coprocessor impl, final int priority,
672         final int seq, final Configuration conf) {
673       this.impl = impl;
674       this.priority = priority;
675       this.state = Coprocessor.State.INSTALLED;
676       this.seq = seq;
677       this.conf = conf;
678     }
679 
680     /** Initialize the environment */
681     public void startup() {
682       if (state == Coprocessor.State.INSTALLED ||
683           state == Coprocessor.State.STOPPED) {
684         state = Coprocessor.State.STARTING;
685         try {
686           impl.start(this);
687           state = Coprocessor.State.ACTIVE;
688         } catch (IOException ioe) {
689           LOG.error("Error starting coprocessor "+impl.getClass().getName(), ioe);
690         }
691       } else {
692         LOG.warn("Not starting coprocessor "+impl.getClass().getName()+
693             " because not inactive (state="+state.toString()+")");
694       }
695     }
696 
697     /** Clean up the environment */
698     protected void shutdown() {
699       if (state == Coprocessor.State.ACTIVE) {
700         state = Coprocessor.State.STOPPING;
701         try {
702           impl.stop(this);
703           state = Coprocessor.State.STOPPED;
704         } catch (IOException ioe) {
705           LOG.error("Error stopping coprocessor "+impl.getClass().getName(), ioe);
706         }
707       } else {
708         LOG.warn("Not stopping coprocessor "+impl.getClass().getName()+
709             " because not active (state="+state.toString()+")");
710       }
711       // clean up any table references
712       for (HTableInterface table: openTables) {
713         try {
714           ((HTableWrapper)table).internalClose();
715         } catch (IOException e) {
716           // nothing can be done here
717           LOG.warn("Failed to close " +
718               Bytes.toStringBinary(table.getTableName()), e);
719         }
720       }
721     }

786   /**
787    * This is used by coprocessor hooks which are declared to throw IOException
788    * (or its subtypes). For such hooks, we should handle throwable objects
789    * depending on the Throwable's type. Those which are instances of
790    * IOException should be passed on to the client. This is in conformance with
791    * the HBase idiom regarding IOException: that it represents a circumstance
792    * that should be passed along to the client for its own handling. For
793    * example, a coprocessor that implements access controls would throw a
794    * subclass of IOException, such as AccessDeniedException, in its preGet()
795    * method to prevent an unauthorized client's performing a Get on a particular
796    * table.
797    * @param env Coprocessor Environment
798    * @param e Throwable object thrown by coprocessor.
799    * @exception IOException Exception
800    */
801   protected void handleCoprocessorThrowable(final CoprocessorEnvironment env,
802                                             final Throwable e)
803       throws IOException {
804     if (e instanceof IOException) {
805       throw (IOException)e;
806     }
807     // If we got here, e is not an IOException. A loaded coprocessor has a
808     // fatal bug, and the server (master or regionserver) should remove the
809     // faulty coprocessor from its set of active coprocessors. Setting
810     // 'hbase.coprocessor.abortonerror' to true will cause abortServer(),
811     // which may be useful in development and testing environments where
812     // 'failing fast' for error analysis is desired.
813     if (env.getConfiguration().getBoolean("hbase.coprocessor.abortonerror",false)) {
814       // server is configured to abort.
815       abortServer(env, e);
816     } else {
817       LOG.error("Removing coprocessor '" + env.toString() + "' from " +
818           "environment because it threw:  " + e,e);
819       coprocessors.remove(env);
820       throw new DoNotRetryIOException("Coprocessor: '" + env.toString() +
821           "' threw: '" + e + "' and has been removed" + "from the active " +
822           "coprocessor set.", e);
823     }
824   }
