115    * Test that on a major compaction, if all cells are expired or deleted, then
116    * we'll end up with no product.  Make sure scanner over region returns
117    * right answer in this case - and that it just basically works.
118    * @throws IOException

120   public void testMajorCompactingToNoOutput() throws IOException {
121     createStoreFile(r);
122     for (int i = 0; i < compactionThreshold; i++) {
123       createStoreFile(r);
124     }
125     // Now delete everything.
126     InternalScanner s = r.getScanner(new Scan());
127     do {
128       List<KeyValue> results = new ArrayList<KeyValue>();
129       boolean result = s.next(results);
130       r.delete(new Delete(results.get(0).getRow()), null, false);
131       if (!result) break;
132     } while(true);
133     s.close();
134     // Flush
135     r.flushcache();
136     // Major compact.
137     r.compactStores(true);
138     s = r.getScanner(new Scan());
139     int counter = 0;
140     do {
141       List<KeyValue> results = new ArrayList<KeyValue>();
142       boolean result = s.next(results);
143       if (!result) break;
144       counter++;
145     } while(true);
146     assertEquals(0, counter);
147   }

190   private void majorCompaction() throws Exception {
191     createStoreFile(r);
192     for (int i = 0; i < compactionThreshold; i++) {
193       createStoreFile(r);
194     }
195     // Add more content.
196     addContent(new HRegionIncommon(r), Bytes.toString(COLUMN_FAMILY));
197 
198     // Now there are about 5 versions of each column.
199     // Default is that there only 3 (MAXVERSIONS) versions allowed per column.
200     //
201     // Assert == 3 when we ask for versions.
202     Result result = r.get(new Get(STARTROW).addFamily(COLUMN_FAMILY_TEXT).setMaxVersions(100), null);
203     assertEquals(compactionThreshold, result.size());
204 
205     // see if CompactionProgress is in place but null
206     for (Store store : this.r.stores.values()) {
207       assertNull(store.getCompactionProgress());
208     }
209 
210     r.flushcache();
211     r.compactStores(true);
212 
213     // see if CompactionProgress has done its thing on at least one store
214     int storeCount = 0;
215     for (Store store : this.r.stores.values()) {
216       CompactionProgress progress = store.getCompactionProgress();
217       if( progress != null ) {
218         ++storeCount;
219         assertTrue(progress.currentCompactedKVs > 0);
220         assertTrue(progress.totalCompactingKVs > 0);
221       }
222       assertTrue(storeCount > 0);
223     }
224 
225     // look at the second row
226     // Increment the least significant character so we get to next row.
227     byte [] secondRowBytes = START_KEY_BYTES.clone();
228     secondRowBytes[START_KEY_BYTES.length - 1]++;
229 
230     // Always 3 versions if that is what max versions is.
231     result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).
232         setMaxVersions(100), null);
233     LOG.debug("Row " + Bytes.toStringBinary(secondRowBytes) + " after " +
234         "initial compaction: " + result);
235     assertEquals("Invalid number of versions of row "
236         + Bytes.toStringBinary(secondRowBytes) + ".", compactionThreshold,
237         result.size());
238 
239     // Now add deletes to memstore and then flush it.
240     // That will put us over
241     // the compaction threshold of 3 store files.  Compacting these store files
242     // should result in a compacted store file that has no references to the
243     // deleted row.
244     LOG.debug("Adding deletes to memstore and flushing");
245     Delete delete = new Delete(secondRowBytes, System.currentTimeMillis(), null);
246     byte [][] famAndQf = {COLUMN_FAMILY, null};
247     delete.deleteFamily(famAndQf[0]);
248     r.delete(delete, null, true);
249 
250     // Assert deleted.
251     result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).setMaxVersions(100), null );
252     assertTrue("Second row should have been deleted", result.isEmpty());
253 
254     r.flushcache();
255 
256     result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).setMaxVersions(100), null );
257     assertTrue("Second row should have been deleted", result.isEmpty());
258 
259     // Add a bit of data and flush.  Start adding at 'bbb'.
260     createSmallerStoreFile(this.r);
261     r.flushcache();
262     // Assert that the second row is still deleted.
263     result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).setMaxVersions(100), null );
264     assertTrue("Second row should still be deleted", result.isEmpty());
265 
266     // Force major compaction.
267     r.compactStores(true);
268     assertEquals(r.getStore(COLUMN_FAMILY_TEXT).getStorefiles().size(), 1);
269 
270     result = r.get(new Get(secondRowBytes).addFamily(COLUMN_FAMILY_TEXT).setMaxVersions(100), null );
271     assertTrue("Second row should still be deleted", result.isEmpty());
272 
273     // Make sure the store files do have some 'aaa' keys in them -- exactly 3.
274     // Also, that compacted store files do not have any secondRowBytes because
275     // they were deleted.
276     verifyCounts(3,0);
277 
278     // Multiple versions allowed for an entry, so the delete isn't enough
279     // Lower TTL and expire to ensure that all our entries have been wiped
280     final int ttl = 1000;
281     for (Store hstore : this.r.stores.values()) {
282       HStore store = ((HStore) hstore);
283       HStore.ScanInfo old = store.scanInfo;
284       HStore.ScanInfo si = new HStore.ScanInfo(old.getFamily(),
285           old.getMinVersions(), old.getMaxVersions(), ttl,
286           old.getKeepDeletedCells(), 0, old.getComparator());
287       store.scanInfo = si;
288     }
289     Thread.sleep(1000);
290 
291     r.compactStores(true);
292     int count = count();
293     assertEquals("Should not see anything after TTL has expired", 0, count);
294   }

296   public void testTimeBasedMajorCompaction() throws Exception {
297     // create 2 storefiles and force a major compaction to reset the time
298     int delay = 10 * 1000; // 10 sec
299     float jitterPct = 0.20f; // 20%
300     conf.setLong(HConstants.MAJOR_COMPACTION_PERIOD, delay);
301     conf.setFloat("hbase.hregion.majorcompaction.jitter", jitterPct);
302 
303     HStore s = ((HStore) r.getStore(COLUMN_FAMILY));
304     s.compactionPolicy.updateConfiguration(conf, s);
305     try {
306       createStoreFile(r);
307       createStoreFile(r);
308       r.compactStores(true);
309 
310       // add one more file & verify that a regular compaction won't work
311       createStoreFile(r);
312       r.compactStores(false);
313       assertEquals(2, s.getStorefilesCount());
314 
315       // ensure that major compaction time is deterministic
316       CompactionPolicy c = s.compactionPolicy;
317       List<StoreFile> storeFiles = s.getStorefiles();
318       long mcTime = c.getNextMajorCompactTime(storeFiles);
319       for (int i = 0; i < 10; ++i) {
320         assertEquals(mcTime, c.getNextMajorCompactTime(storeFiles));
321       }
322 
323       // ensure that the major compaction time is within the variance
324       long jitter = Math.round(delay * jitterPct);
325       assertTrue(delay - jitter <= mcTime && mcTime <= delay + jitter);
326 
327       // wait until the time-based compaction interval
328       Thread.sleep(mcTime);
329 
330       // trigger a compaction request and ensure that it's upgraded to major
331       r.compactStores(false);
332       assertEquals(1, s.getStorefilesCount());
333     } finally {
334       // reset the timed compaction settings
335       conf.setLong(HConstants.MAJOR_COMPACTION_PERIOD, 1000*60*60*24);
336       conf.setFloat("hbase.hregion.majorcompaction.jitter", 0.20F);
337       // run a major to reset the cache
338       createStoreFile(r);
339       r.compactStores(true);
340       assertEquals(1, s.getStorefilesCount());
341     }
342   }

399   private void testMinorCompactionWithDelete(Delete delete, int expectedResultsAfterDelete) throws Exception {
400     HRegionIncommon loader = new HRegionIncommon(r);
401     for (int i = 0; i < compactionThreshold + 1; i++) {
402       addContent(loader, Bytes.toString(fam1), Bytes.toString(col1), firstRowBytes, thirdRowBytes, i);
403       addContent(loader, Bytes.toString(fam1), Bytes.toString(col2), firstRowBytes, thirdRowBytes, i);
404       addContent(loader, Bytes.toString(fam2), Bytes.toString(col1), firstRowBytes, thirdRowBytes, i);
405       addContent(loader, Bytes.toString(fam2), Bytes.toString(col2), firstRowBytes, thirdRowBytes, i);
406       r.flushcache();
407     }
408 
409     Result result = r.get(new Get(firstRowBytes).addColumn(fam1, col1).setMaxVersions(100), null);
410     assertEquals(compactionThreshold, result.size());
411     result = r.get(new Get(secondRowBytes).addColumn(fam2, col2).setMaxVersions(100), null);
412     assertEquals(compactionThreshold, result.size());
413 
414     // Now add deletes to memstore and then flush it.  That will put us over
415     // the compaction threshold of 3 store files.  Compacting these store files
416     // should result in a compacted store file that has no references to the
417     // deleted row.
418     r.delete(delete, null, true);
419 
420     // Make sure that we have only deleted family2 from secondRowBytes
421     result = r.get(new Get(secondRowBytes).addColumn(fam2, col2).setMaxVersions(100), null);
422     assertEquals(expectedResultsAfterDelete, result.size());
423     // but we still have firstrow
424     result = r.get(new Get(firstRowBytes).addColumn(fam1, col1).setMaxVersions(100), null);
425     assertEquals(compactionThreshold, result.size());
426 
427     r.flushcache();
428     // should not change anything.
429     // Let us check again
430 
431     // Make sure that we have only deleted family2 from secondRowBytes
432     result = r.get(new Get(secondRowBytes).addColumn(fam2, col2).setMaxVersions(100), null);
433     assertEquals(expectedResultsAfterDelete, result.size());
434     // but we still have firstrow
435     result = r.get(new Get(firstRowBytes).addColumn(fam1, col1).setMaxVersions(100), null);
436     assertEquals(compactionThreshold, result.size());
437 
438     // do a compaction
439     Store store2 = this.r.stores.get(fam2);
440     int numFiles1 = store2.getStorefiles().size();
441     assertTrue("Was expecting to see 4 store files", numFiles1 > compactionThreshold); // > 3
442     store2.compactRecentForTesting(compactionThreshold);   // = 3
443     int numFiles2 = store2.getStorefiles().size();
444     // Check that we did compact
445     assertTrue("Number of store files should go down", numFiles1 > numFiles2);
446     // Check that it was a minor compaction.
447     assertTrue("Was not supposed to be a major compaction", numFiles2 > 1);
448 
449     // Make sure that we have only deleted family2 from secondRowBytes
450     result = r.get(new Get(secondRowBytes).addColumn(fam2, col2).setMaxVersions(100), null);
451     assertEquals(expectedResultsAfterDelete, result.size());
452     // but we still have firstrow
453     result = r.get(new Get(firstRowBytes).addColumn(fam1, col1).setMaxVersions(100), null);
454     assertEquals(compactionThreshold, result.size());
455   }

476   /**
477    * Verify that you can stop a long-running compaction
478    * (used during RS shutdown)
479    * @throws Exception
480    */
481   public void testInterruptCompaction() throws Exception {
482     assertEquals(0, count());
483 
484     // lower the polling interval for this test
485     int origWI = HStore.closeCheckInterval;
486     HStore.closeCheckInterval = 10*1000; // 10 KB
487 
488     try {
489       // Create a couple store files w/ 15KB (over 10KB interval)
490       int jmax = (int) Math.ceil(15.0/compactionThreshold);
491       byte [] pad = new byte[1000]; // 1 KB chunk
492       for (int i = 0; i < compactionThreshold; i++) {
493         HRegionIncommon loader = new HRegionIncommon(r);
494         Put p = new Put(Bytes.add(STARTROW, Bytes.toBytes(i)));
495         p.setWriteToWAL(false);
496         for (int j = 0; j < jmax; j++) {
497           p.add(COLUMN_FAMILY, Bytes.toBytes(j), pad);
498         }
499         addContent(loader, Bytes.toString(COLUMN_FAMILY));
500         loader.put(p);
501         loader.flushcache();
502       }
503 
504       HRegion spyR = spy(r);
505       doAnswer(new Answer() {
506         public Object answer(InvocationOnMock invocation) throws Throwable {
507           r.writestate.writesEnabled = false;
508           return invocation.callRealMethod();
509         }
510       }).when(spyR).doRegionCompactionPrep();
511 
512       // force a minor compaction, but not before requesting a stop
513       spyR.compactStores();
514 
515       // ensure that the compaction stopped, all old files are intact,
516       Store s = r.stores.get(COLUMN_FAMILY);
517       assertEquals(compactionThreshold, s.getStorefilesCount());
518       assertTrue(s.getStorefilesSize() > 15*1000);
519       // and no new store files persisted past compactStores()
520       FileStatus[] ls = FileSystem.get(conf).listStatus(r.getTmpDir());
521       assertEquals(0, ls.length);
522 
523     } finally {
524       // don't mess up future tests
525       r.writestate.writesEnabled = true;
526       HStore.closeCheckInterval = origWI;
527 
528       // Delete all Store information once done using
529       for (int i = 0; i < compactionThreshold; i++) {
530         Delete delete = new Delete(Bytes.add(STARTROW, Bytes.toBytes(i)));
531         byte [][] famAndQf = {COLUMN_FAMILY, null};
532         delete.deleteFamily(famAndQf[0]);
533         r.delete(delete, null, true);
534       }
535       r.flushcache();
536 
537       // Multiple versions allowed for an entry, so the delete isn't enough
538       // Lower TTL and expire to ensure that all our entries have been wiped
539       final int ttl = 1000;
540       for (Store hstore: this.r.stores.values()) {
541         HStore store = (HStore)hstore;
542         HStore.ScanInfo old = store.scanInfo;
543         HStore.ScanInfo si = new HStore.ScanInfo(old.getFamily(),
544             old.getMinVersions(), old.getMaxVersions(), ttl,
545             old.getKeepDeletedCells(), 0, old.getComparator());
546         store.scanInfo = si;
547       }
548       Thread.sleep(ttl);
549 
550       r.compactStores(true);
551       assertEquals(0, count());
552     }
553   }

583   public void testCompactionWithCorruptResult() throws Exception {
584     int nfiles = 10;
585     for (int i = 0; i < nfiles; i++) {
586       createStoreFile(r);
587     }
588     HStore store = (HStore) r.getStore(COLUMN_FAMILY);

590     List<StoreFile> storeFiles = store.getStorefiles();
591     long maxId = StoreFile.getMaxSequenceIdInList(storeFiles, true);
592     Compactor tool = new Compactor(this.conf);
593 
594     StoreFile.Writer compactedFile =
