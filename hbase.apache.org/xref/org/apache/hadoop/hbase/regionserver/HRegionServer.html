438   /**
439    * Starts a HRegionServer at the default location
440    *
441    * @param conf
442    * @throws IOException
443    * @throws InterruptedException
444    */
445   public HRegionServer(Configuration conf)
446   throws IOException, InterruptedException {
447     this.fsOk = true;
448     this.conf = conf;
449     // Set how many times to retry talking to another server over HConnection.
450     HConnectionManager.setServerSideHConnectionRetries(this.conf, LOG);
451     this.isOnline = false;
452     checkCodecs(this.conf);
453 
454     // do we use checksum verfication in the hbase? If hbase checksum verification
455     // is enabled, then we automatically switch off hdfs checksum verification.
456     this.useHBaseChecksum = conf.getBoolean(
457       HConstants.HBASE_CHECKSUM_VERIFICATION, false);
458 
459     // Config'ed params
460     this.numRetries = conf.getInt("hbase.client.retries.number", 10);
461     this.threadWakeFrequency = conf.getInt(HConstants.THREAD_WAKE_FREQUENCY,
462       10 * 1000);
463     this.msgInterval = conf.getInt("hbase.regionserver.msginterval", 3 * 1000);
464 
465     this.sleeper = new Sleeper(this.msgInterval, this);
466 
467     this.maxScannerResultSize = conf.getLong(
468       HConstants.HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE_KEY,
469       HConstants.DEFAULT_HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE);
470 
471     this.numRegionsToReport = conf.getInt(
472       "hbase.regionserver.numregionstoreport", 10);
473 
474     this.rpcTimeout = conf.getInt(
475       HConstants.HBASE_RPC_TIMEOUT_KEY,
476       HConstants.DEFAULT_HBASE_RPC_TIMEOUT);
477 
478     this.abortRequested = false;
479     this.stopped = false;
480 
481     this.rowLockLeaseTimeoutPeriod = conf.getInt(
482       HConstants.HBASE_REGIONSERVER_ROWLOCK_TIMEOUT_PERIOD,
483       HConstants.DEFAULT_HBASE_REGIONSERVER_ROWLOCK_TIMEOUT_PERIOD);
484 
485     this.scannerLeaseTimeoutPeriod = conf.getInt(HConstants.HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD,
486       HConstants.DEFAULT_HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD);
487 
488     // Server to handle client requests.
489     String hostname = Strings.domainNamePointerToHostName(DNS.getDefaultHost(
490       conf.get("hbase.regionserver.dns.interface", "default"),
491       conf.get("hbase.regionserver.dns.nameserver", "default")));
492     int port = conf.getInt(HConstants.REGIONSERVER_PORT,
493       HConstants.DEFAULT_REGIONSERVER_PORT);
494     // Creation of a HSA will force a resolve.
495     InetSocketAddress initialIsa = new InetSocketAddress(hostname, port);
496     if (initialIsa.getAddress() == null) {
497       throw new IllegalArgumentException("Failed resolve of " + initialIsa);
498     }
499 
500     this.rpcServer = HBaseRPC.getServer(AdminProtocol.class, this,
501       new Class<?>[]{ClientProtocol.class,
502         AdminProtocol.class, HBaseRPCErrorHandler.class,
503         OnlineRegions.class},
504         initialIsa.getHostName(), // BindAddress is IP we got for this server.
505         initialIsa.getPort(),
506         conf.getInt("hbase.regionserver.handler.count", 10),
507         conf.getInt("hbase.regionserver.metahandler.count", 10),
508         conf.getBoolean("hbase.rpc.verbose", false),
509         conf, HConstants.QOS_THRESHOLD);
510     // Set our address.
511     this.isa = this.rpcServer.getListenerAddress();
512 
513     this.rpcServer.setErrorHandler(this);
514     this.rpcServer.setQosFunction((qosFunction = new QosFunction()));
515     this.startcode = System.currentTimeMillis();
516 
517     // login the zookeeper client principal (if using security)
518     ZKUtil.loginClient(this.conf, "hbase.zookeeper.client.keytab.file",
519       "hbase.zookeeper.client.kerberos.principal", this.isa.getHostName());
520 
521     // login the server principal (if using secure Hadoop)
522     User.login(this.conf, "hbase.regionserver.keytab.file",
523       "hbase.regionserver.kerberos.principal", this.isa.getHostName());
524     regionServerAccounting = new RegionServerAccounting();
525     cacheConfig = new CacheConfig(conf);
526     this.rsHost = new RegionServerCoprocessorHost(this, this.conf);
527   }

529   /**
530    * Run test on configured codecs to make sure supporting libs are in place.
531    * @param c
532    * @throws IOException
533    */
534   private static void checkCodecs(final Configuration c) throws IOException {
535     // check to see if the codec list is available:
536     String [] codecs = c.getStrings("hbase.regionserver.codecs", (String[])null);
537     if (codecs == null) return;
538     for (String codec : codecs) {
539       if (!CompressionTest.testCompression(codec)) {
540         throw new IOException("Compression codec " + codec +
541           " not supported, aborting RS construction");
542       }
543     }
544   }

568   /**
569    * Utility used ensuring higher quality of service for priority rpcs; e.g.
570    * rpcs to .META. and -ROOT-, etc.
571    */
572   class QosFunction implements Function<RpcRequestBody,Integer> {
573     private final Map<String, Integer> annotatedQos;
574     //We need to mock the regionserver instance for some unit tests (set via
575     //setRegionServer method.
576     //The field value is initially set to the enclosing instance of HRegionServer.
577     private HRegionServer hRegionServer = HRegionServer.this;
578 
579     //The logic for figuring out high priority RPCs is as follows:
580     //1. if the method is annotated with a QosPriority of QOS_HIGH,
581     //   that is honored
582     //2. parse out the protobuf message and see if the request is for meta
583     //   region, and if so, treat it as a high priority RPC
584     //Some optimizations for (2) are done here -
585     //Clients send the argument classname as part of making the RPC. The server
586     //decides whether to deserialize the proto argument message based on the
587     //pre-established set of argument classes (knownArgumentClasses below).
588     //This prevents the server from having to deserialize all proto argument
589     //messages prematurely.
590     //All the argument classes declare a 'getRegion' method that returns a
591     //RegionSpecifier object. Methods can be invoked on the returned object
592     //to figure out whether it is a meta region or not.
593     @SuppressWarnings("unchecked")
594     private final Class<? extends Message>[] knownArgumentClasses = new Class[]{
595         GetRegionInfoRequest.class,
596         GetStoreFileRequest.class,
597         CloseRegionRequest.class,
598         FlushRegionRequest.class,
599         SplitRegionRequest.class,
600         CompactRegionRequest.class,
601         GetRequest.class,
602         MutateRequest.class,
603         ScanRequest.class,
604         LockRowRequest.class,
605         UnlockRowRequest.class,
606         MultiRequest.class
607     };
608 
609     //Some caches for helping performance
610     private final Map<String, Class<? extends Message>> argumentToClassMap =
611         new HashMap<String, Class<? extends Message>>();
612     private final Map<String, Map<Class<? extends Message>, Method>>
613       methodMap = new HashMap<String, Map<Class<? extends Message>, Method>>();
614 
615     public QosFunction() {
616       Map<String, Integer> qosMap = new HashMap<String, Integer>();
617       for (Method m : HRegionServer.class.getMethods()) {
618         QosPriority p = m.getAnnotation(QosPriority.class);
619         if (p != null) {
620           qosMap.put(m.getName(), p.priority());
621         }
622       }
623 
624       annotatedQos = qosMap;
625       if (methodMap.get("parseFrom") == null) {
626         methodMap.put("parseFrom",
627             new HashMap<Class<? extends Message>, Method>());
628       }
629       if (methodMap.get("getRegion") == null) {
630         methodMap.put("getRegion", 
631             new HashMap<Class<? extends Message>, Method>());
632       }
633       for (Class<? extends Message> cls : knownArgumentClasses) {
634         argumentToClassMap.put(cls.getCanonicalName(), cls);
635         try {
636           methodMap.get("parseFrom").put(cls,
637                           cls.getDeclaredMethod("parseFrom",ByteString.class));
638           methodMap.get("getRegion").put(cls, cls.getDeclaredMethod("getRegion"));
639         } catch (Exception e) {
640           throw new RuntimeException(e);
641         }
642       }
643     }

645     void setRegionServer(HRegionServer server) {
646       this.hRegionServer = server;
647     }

649     public boolean isMetaRegion(byte[] regionName) {
650       HRegion region;
651       try {
652         region = hRegionServer.getRegion(regionName);
653       } catch (NotServingRegionException ignored) {
654         return false;
655       }
656       return region.getRegionInfo().isMetaRegion();
657     }

659     @Override
660     public Integer apply(RpcRequestBody from) {
661       String methodName = from.getMethodName();
662       Class<? extends Message> rpcArgClass = null;
663       if (from.hasRequestClassName()) {
664         String cls = from.getRequestClassName();
665         rpcArgClass = argumentToClassMap.get(cls);
666       }
667 
668       Integer priorityByAnnotation = annotatedQos.get(methodName);
669       if (priorityByAnnotation != null) {
670         return priorityByAnnotation;
671       }
672 
673       if (rpcArgClass == null || from.getRequest().isEmpty()) {
674         return HConstants.NORMAL_QOS;
675       }
676       Object deserializedRequestObj = null;
677       //check whether the request has reference to Meta region
678       try {
679         Method parseFrom = methodMap.get("parseFrom").get(rpcArgClass);
680         deserializedRequestObj = parseFrom.invoke(null, from.getRequest());
681         Method getRegion = methodMap.get("getRegion").get(rpcArgClass);
682         RegionSpecifier regionSpecifier =
683             (RegionSpecifier)getRegion.invoke(deserializedRequestObj,
684                 (Object[])null);
685         HRegion region = hRegionServer.getRegion(regionSpecifier);
686         if (region.getRegionInfo().isMetaRegion()) {
687           if (LOG.isDebugEnabled()) {
688             LOG.debug("High priority: " + from.toString());
689           }
690           return HConstants.HIGH_QOS;
691         }
692       } catch (Exception ex) {
693         throw new RuntimeException(ex);
694       }
695 
696       if (methodName.equals("scan")) { // scanner methods...
697         ScanRequest request = (ScanRequest)deserializedRequestObj;
698         if (!request.hasScannerId()) {
699           return HConstants.NORMAL_QOS;
700         }
701         RegionScanner scanner = hRegionServer.getScanner(request.getScannerId());
702         if (scanner != null && scanner.getRegionInfo().isMetaRegion()) {
703           if (LOG.isDebugEnabled()) {
704             LOG.debug("High priority scanner request: " + request.getScannerId());
705           }
706           return HConstants.HIGH_QOS;
707         }
708       }
709       if (LOG.isDebugEnabled()) {
710         LOG.debug("Low priority: " + from.toString());
711       }
712       return HConstants.NORMAL_QOS;
713     }

738   /**
739    * Bring up connection to zk ensemble and then wait until a master for this
740    * cluster and then after that, wait until cluster 'up' flag has been set.
741    * This is the order in which master does things.
742    * Finally put up a catalog tracker.
743    * @throws IOException
744    * @throws InterruptedException
745    */
746   private void initializeZooKeeper() throws IOException, InterruptedException {
747     // Open connection to zookeeper and set primary watcher
748     this.zooKeeper = new ZooKeeperWatcher(conf, REGIONSERVER + ":" +
749       this.isa.getPort(), this);
750 
751     // Create the master address manager, register with zk, and start it.  Then
752     // block until a master is available.  No point in starting up if no master
753     // running.
754     this.masterAddressManager = new MasterAddressTracker(this.zooKeeper, this);
755     this.masterAddressManager.start();
756     blockAndCheckIfStopped(this.masterAddressManager);
757 
758     // Wait on cluster being up.  Master will set this flag up in zookeeper
759     // when ready.
760     this.clusterStatusTracker = new ClusterStatusTracker(this.zooKeeper, this);
761     this.clusterStatusTracker.start();
762     blockAndCheckIfStopped(this.clusterStatusTracker);
763 
764     // Create the catalog tracker and start it;
765     this.catalogTracker = new CatalogTracker(this.zooKeeper, this.conf,
766       this, this.conf.getInt("hbase.regionserver.catalog.timeout", 600000));
767     catalogTracker.start();
768 
769     // Retrieve clusterId
770     // Since cluster status is now up
771     // ID should have already been set by HMaster
772     try {
773       String clusterId = ZKClusterId.readClusterIdZNode(this.zooKeeper);
774       if (clusterId == null) {
775         this.abort("Cluster ID has not been set");
776       }
777       this.conf.set(HConstants.CLUSTER_ID, clusterId);
778       LOG.info("ClusterId : "+clusterId);
779     } catch (KeeperException e) {
780       this.abort("Failed to retrieve Cluster ID",e);
781     }
782   }

834   /**
835    * The HRegionServer sticks in this loop until closed.
836    */
837   public void run() {
838     try {
839       // Do pre-registration initializations; zookeeper, lease threads, etc.
840       preRegistrationInitialization();
841     } catch (Throwable e) {
842       abort("Fatal exception during initialization", e);
843     }
844 
845     try {
846       // Try and register with the Master; tell it we are here.  Break if
847       // server is stopped or the clusterup flag is down or hdfs went wacky.
848       while (keepLooping()) {
849         RegionServerStartupResponse w = reportForDuty();
850         if (w == null) {
851           LOG.warn("reportForDuty failed; sleeping and then retrying.");
852           this.sleeper.sleep();
853         } else {
854           handleReportForDutyResponse(w);
855           break;
856         }
857       }
858 
859       // We registered with the Master.  Go into run mode.
860       long lastMsg = 0;
861       long oldRequestCount = -1;
862       // The main run loop.
863       while (!this.stopped && isHealthy()) {
864         if (!isClusterUp()) {
865           if (isOnlineRegionsEmpty()) {
866             stop("Exiting; cluster shutdown set and not carrying any regions");
867           } else if (!this.stopping) {
868             this.stopping = true;
869             LOG.info("Closing user regions");
870             closeUserRegions(this.abortRequested);
871           } else if (this.stopping) {
872             boolean allUserRegionsOffline = areAllUserRegionsOffline();
873             if (allUserRegionsOffline) {
874               // Set stopped if no requests since last time we went around the loop.
875               // The remaining meta regions will be closed on our way out.
876               if (oldRequestCount == this.requestCount.get()) {
877                 stop("Stopped; only catalog regions remaining online");
878                 break;
879               }
880               oldRequestCount = this.requestCount.get();
881             } else {
882               // Make sure all regions have been closed -- some regions may
883               // have not got it because we were splitting at the time of
884               // the call to closeUserRegions.
885               closeUserRegions(this.abortRequested);
886             }
887             LOG.debug("Waiting on " + getOnlineRegionsAsPrintableString());
888           }
889         }
890         long now = System.currentTimeMillis();
891         if ((now - lastMsg) >= msgInterval) {
892           tryRegionServerReport(lastMsg, now);
893           lastMsg = System.currentTimeMillis();
894         }
895         if (!this.stopped) this.sleeper.sleep();
896       } // for
897     } catch (Throwable t) {
898       if (!checkOOME(t)) {
899         abort("Unhandled exception: " + t.getMessage(), t);
900       }
901     }
902     // Run shutdown.
903     if (mxBean != null) {
904       MBeanUtil.unregisterMBean(mxBean);
905       mxBean = null;
906     }
907     if (this.thriftServer != null) this.thriftServer.shutdown();
908     this.leases.closeAfterLeasesExpire();
909     this.rpcServer.stop();
910     if (this.splitLogWorker != null) {
911       splitLogWorker.stop();
912     }
913     if (this.infoServer != null) {
914       LOG.info("Stopping infoServer");
915       try {
916         this.infoServer.stop();
917       } catch (Exception e) {
918         e.printStackTrace();
919       }
920     }
921     // Send cache a shutdown.
922     if (cacheConfig.isBlockCacheEnabled()) {
923       cacheConfig.getBlockCache().shutdown();
924     }
925 
926     movedRegionsCleaner.stop("Region Server stopping");
927 
928     // Send interrupts to wake up threads if sleeping so they notice shutdown.
929     // TODO: Should we check they are alive? If OOME could have exited already
930     if (this.cacheFlusher != null) this.cacheFlusher.interruptIfNecessary();
931     if (this.compactSplitThread != null) this.compactSplitThread.interruptIfNecessary();
932     if (this.hlogRoller != null) this.hlogRoller.interruptIfNecessary();
933     if (this.compactionChecker != null)
934       this.compactionChecker.interrupt();
935 
936     if (this.killed) {
937       // Just skip out w/o closing regions.  Used when testing.
938     } else if (abortRequested) {
939       if (this.fsOk) {
940         closeUserRegions(abortRequested); // Don't leave any open file handles
941       }
942       LOG.info("aborting server " + this.serverNameFromMasterPOV);
943     } else {
944       closeUserRegions(abortRequested);
945       closeAllScanners();
946       LOG.info("stopping server " + this.serverNameFromMasterPOV);
947     }
948     // Interrupt catalog tracker here in case any regions being opened out in
949     // handlers are stuck waiting on meta or root.
950     if (this.catalogTracker != null) this.catalogTracker.stop();
951 
952     // Closing the compactSplit thread before closing meta regions
953     if (!this.killed && containsMetaTableRegions()) {
954       if (!abortRequested || this.fsOk) {
955         if (this.compactSplitThread != null) {
956           this.compactSplitThread.join();
957           this.compactSplitThread = null;
958         }
959         closeMetaTableRegions(abortRequested);
960       }
961     }
962 
963     if (!this.killed && this.fsOk) {
964       waitOnAllRegionsToClose(abortRequested);
965       LOG.info("stopping server " + this.serverNameFromMasterPOV +
966         "; all regions closed.");
967     }
968 
969     //fsOk flag may be changed when closing regions throws exception.
970     if (!this.killed && this.fsOk) {
971       closeWAL(abortRequested ? false : true);
972     }
973 
974     // Make sure the proxy is down.
975     if (this.hbaseMaster != null) {
976       HBaseRPC.stopProxy(this.hbaseMaster);
977       this.hbaseMaster = null;
978     }
979     this.leases.close();
980 
981     if (!killed) {
982       join();
983     }
984 
985     try {
986       deleteMyEphemeralNode();
987     } catch (KeeperException e) {
988       LOG.warn("Failed deleting my ephemeral node", e);
989     }
990     // We may have failed to delete the znode at the previous step, but
991     //  we delete the file anyway: a second attempt to delete the znode is likely to fail again.
992     ZNodeClearer.deleteMyEphemeralNodeOnDisk();
993     this.zooKeeper.close();
994     LOG.info("stopping server " + this.serverNameFromMasterPOV +
995       "; zookeeper connection closed.");
996 
997     LOG.info(Thread.currentThread().getName() + " exiting");
998   }

1017   void tryRegionServerReport(long reportStartTime, long reportEndTime)
1018   throws IOException {
1019     HBaseProtos.ServerLoad sl = buildServerLoad(reportStartTime, reportEndTime);
1020     try {
1021       RegionServerReportRequest.Builder request = RegionServerReportRequest.newBuilder();
1022       ServerName sn = ServerName.parseVersionedServerName(
1023         this.serverNameFromMasterPOV.getVersionedBytes());
1024       request.setServer(ProtobufUtil.toServerName(sn));
1025       request.setLoad(sl);
1026       this.hbaseMaster.regionServerReport(null, request.build());
1027     } catch (ServiceException se) {
1028       IOException ioe = ProtobufUtil.getRemoteException(se);
1029       if (ioe instanceof YouAreDeadException) {
1030         // This will be caught and handled as a fatal error in run()
1031         throw ioe;
1032       }
1033       // Couldn't connect to the master, get location from zk and reconnect
1034       // Method blocks until new master is found or we are stopped
1035       getMaster();
1036     }
1037   }

1039   HBaseProtos.ServerLoad buildServerLoad(long reportStartTime, long reportEndTime) {
1040     // We're getting the MetricsRegionServerWrapper here because the wrapper computes requests
1041     // per second, and other metrics  As long as metrics are part of ServerLoad it's best to use
1042     // the wrapper to compute those numbers in one place.
1043     // In the long term most of these should be moved off of ServerLoad and the heart beat.
1044     // Instead they should be stored in an HBase table so that external visibility into HBase is
1045     // improved; Additionally the load balancer will be able to take advantage of a more complete
1046     // history.
1047     MetricsRegionServerWrapper regionServerWrapper = this.metricsRegionServer.getRegionServerWrapper();
1048     Collection<HRegion> regions = getOnlineRegionsLocalContext();
1049     MemoryUsage memory =
1050       ManagementFactory.getMemoryMXBean().getHeapMemoryUsage();
1051 
1052     HBaseProtos.ServerLoad.Builder serverLoad = HBaseProtos.ServerLoad.newBuilder();
1053     serverLoad.setNumberOfRequests((int) regionServerWrapper.getRequestsPerSecond());
1054     serverLoad.setTotalNumberOfRequests((int) regionServerWrapper.getTotalRequestCount());
1055     serverLoad.setUsedHeapMB((int)(memory.getUsed() / 1024 / 1024));
1056     serverLoad.setMaxHeapMB((int) (memory.getMax() / 1024 / 1024));
1057     Set<String> coprocessors = this.hlog.getCoprocessorHost().getCoprocessors();
1058     for (String coprocessor : coprocessors) {
1059       serverLoad.addCoprocessors(
1060         Coprocessor.newBuilder().setName(coprocessor).build());
1061     }
1062     for (HRegion region : regions) {
1063       serverLoad.addRegionLoads(createRegionLoad(region));
1064     }
1065     serverLoad.setReportStartTime(reportStartTime);
1066     serverLoad.setReportEndTime(reportEndTime);
1067     if (this.infoServer != null) {
1068       serverLoad.setInfoServerPort(this.infoServer.getPort());
1069     } else {
1070       serverLoad.setInfoServerPort(-1);
1071     }
1072     return serverLoad.build();
1073   }

1084   /**
1085    * Wait on regions close.
1086    */
1087   private void waitOnAllRegionsToClose(final boolean abort) {
1088     // Wait till all regions are closed before going out.
1089     int lastCount = -1;
1090     long previousLogTime = 0;
1091     Set<String> closedRegions = new HashSet<String>();
1092     while (!isOnlineRegionsEmpty()) {
1093       int count = getNumberOfOnlineRegions();
1094       // Only print a message if the count of regions has changed.
1095       if (count != lastCount) {
1096         // Log every second at most
1097         if (System.currentTimeMillis() > (previousLogTime + 1000)) {
1098           previousLogTime = System.currentTimeMillis();
1099           lastCount = count;
1100           LOG.info("Waiting on " + count + " regions to close");
1101           // Only print out regions still closing if a small number else will
1102           // swamp the log.
1103           if (count < 10 && LOG.isDebugEnabled()) {
1104             LOG.debug(this.onlineRegions);
1105           }
1106         }
1107       }
1108       // Ensure all user regions have been sent a close. Use this to
1109       // protect against the case where an open comes in after we start the
1110       // iterator of onlineRegions to close all user regions.
1111       for (Map.Entry<String, HRegion> e : this.onlineRegions.entrySet()) {
1112         HRegionInfo hri = e.getValue().getRegionInfo();
1113         if (!this.regionsInTransitionInRS.containsKey(hri.getEncodedNameAsBytes())
1114             && !closedRegions.contains(hri.getEncodedName())) {
1115           closedRegions.add(hri.getEncodedName());
1116           // Don't update zk with this close transition; pass false.
1117           closeRegion(hri, abort, false);
1118         }
1119       }
1120       // No regions in RIT, we could stop waiting now.
1121       if (this.regionsInTransitionInRS.isEmpty()) {
1122         if (!isOnlineRegionsEmpty()) {
1123           LOG.info("We were exiting though online regions are not empty," +
1124               " because some regions failed closing");
1125         }
1126         break;
1127       }
1128       Threads.sleep(200);
1129     }
1130   }

1158   /*
1159    * Run init. Sets up hlog and starts up all server threads.
1160    *
1161    * @param c Extra configuration.
1162    */
1163   protected void handleReportForDutyResponse(final RegionServerStartupResponse c)
1164   throws IOException {
1165     try {
1166       for (NameStringPair e : c.getMapEntriesList()) {
1167         String key = e.getName();
1168         // The hostname the master sees us as.
1169         if (key.equals(HConstants.KEY_FOR_HOSTNAME_SEEN_BY_MASTER)) {
1170           String hostnameFromMasterPOV = e.getValue();
1171           this.serverNameFromMasterPOV = new ServerName(hostnameFromMasterPOV,
1172             this.isa.getPort(), this.startcode);
1173           LOG.info("Master passed us hostname to use. Was=" +
1174             this.isa.getHostName() + ", Now=" +
1175             this.serverNameFromMasterPOV.getHostname());
1176           continue;
1177         }
1178         String value = e.getValue().toString();
1179         if (LOG.isDebugEnabled()) {
1180           LOG.debug("Config from master: " + key + "=" + value);
1181         }
1182         this.conf.set(key, value);
1183       }

1450   /*
1451    * Start maintanence Threads, Server, Worker and lease checker threads.
1452    * Install an UncaughtExceptionHandler that calls abort of RegionServer if we
1453    * get an unhandled exception. We cannot set the handler on all threads.
1454    * Server's internal Listener thread is off limits. For Server, if an OOME, it
1455    * waits a while then retries. Meantime, a flush or a compaction that tries to
1456    * run should trigger same critical condition and the shutdown will run. On
1457    * its way out, this server will shut down Server. Leases are sort of
1458    * inbetween. It has an internal thread that while it inherits from Chore, it
1459    * keeps its own internal stop mechanism so needs to be stopped by this
1460    * hosting server. Worker logs the exception and exits.
1461    */
1462   private void startServiceThreads() throws IOException {
1463     String n = Thread.currentThread().getName();
1464     UncaughtExceptionHandler handler = new UncaughtExceptionHandler() {
1465       public void uncaughtException(Thread t, Throwable e) {
1466         abort("Uncaught exception in service thread " + t.getName(), e);
1467       }
1468     };
1469 
1470     // Start executor services
1471     this.service = new ExecutorService(getServerName().toString());
1472     this.service.startExecutorService(ExecutorType.RS_OPEN_REGION,
1473       conf.getInt("hbase.regionserver.executor.openregion.threads", 3));
1474     this.service.startExecutorService(ExecutorType.RS_OPEN_ROOT,
1475       conf.getInt("hbase.regionserver.executor.openroot.threads", 1));
1476     this.service.startExecutorService(ExecutorType.RS_OPEN_META,
1477       conf.getInt("hbase.regionserver.executor.openmeta.threads", 1));
1478     this.service.startExecutorService(ExecutorType.RS_CLOSE_REGION,
1479       conf.getInt("hbase.regionserver.executor.closeregion.threads", 3));
1480     this.service.startExecutorService(ExecutorType.RS_CLOSE_ROOT,
1481       conf.getInt("hbase.regionserver.executor.closeroot.threads", 1));
1482     this.service.startExecutorService(ExecutorType.RS_CLOSE_META,
1483       conf.getInt("hbase.regionserver.executor.closemeta.threads", 1));
1484 
1485     Threads.setDaemonThreadRunning(this.hlogRoller.getThread(), n + ".logRoller", handler);
1486     Threads.setDaemonThreadRunning(this.cacheFlusher.getThread(), n + ".cacheFlusher",
1487       handler);
1488     Threads.setDaemonThreadRunning(this.compactionChecker.getThread(), n +
1489       ".compactionChecker", handler);
1490 
1491     // Leases is not a Thread. Internally it runs a daemon thread. If it gets
1492     // an unhandled exception, it will just exit.
1493     this.leases.setName(n + ".leaseChecker");
1494     this.leases.start();
1495 
1496     // Put up the webui.  Webui may come up on port other than configured if
1497     // that port is occupied. Adjust serverInfo if this is the case.
1498     this.webuiport = putUpWebUI();
1499 
1500     if (this.replicationSourceHandler == this.replicationSinkHandler &&
1501         this.replicationSourceHandler != null) {
1502       this.replicationSourceHandler.startReplicationService();
1503     } else if (this.replicationSourceHandler != null) {
1504       this.replicationSourceHandler.startReplicationService();
1505     } else if (this.replicationSinkHandler != null) {
1506       this.replicationSinkHandler.startReplicationService();
1507     }
1508 
1509     // Start Server.  This service is like leases in that it internally runs
1510     // a thread.
1511     this.rpcServer.start();
1512 
1513     // Create the log splitting worker and start it
1514     this.splitLogWorker = new SplitLogWorker(this.zooKeeper,
1515         this.getConfiguration(), this.getServerName(), this);
1516     splitLogWorker.start();
1517   }

1747   /**
1748    * Get the current master from ZooKeeper and open the RPC connection to it.
1749    *
1750    * Method will block until a master is available. You can break from this
1751    * block by requesting the server stop.
1752    *
1753    * @return master + port, or null if server has been stopped
1754    */
1755   private ServerName getMaster() {
1756     ServerName masterServerName = null;
1757     long previousLogTime = 0;
1758     RegionServerStatusProtocol master = null;
1759     boolean refresh = false; // for the first time, use cached data
1760     while (keepLooping() && master == null) {
1761       masterServerName = this.masterAddressManager.getMasterAddress(refresh);
1762       if (masterServerName == null) {
1763         if (!keepLooping()) {
1764           // give up with no connection.
1765           LOG.debug("No master found and cluster is stopped; bailing out");
1766           return null;
1767         }
1768         LOG.debug("No master found; retry");
1769         previousLogTime = System.currentTimeMillis();
1770         refresh = true; // let's try pull it from ZK directly
1771 
1772         sleeper.sleep();
1773         continue;
1774       }
1775 
1776       InetSocketAddress isa =
1777         new InetSocketAddress(masterServerName.getHostname(), masterServerName.getPort());
1778 
1779       LOG.info("Attempting connect to Master server at " +
1780         this.masterAddressManager.getMasterAddress());

2731       MutateType type = mutate.getMutateType();
2732       switch (type) {
2733       case APPEND:
2734         r = append(region, mutate);

2756               qualifier, compareOp, comparator, put, lock, true);

2758               result = region.getCoprocessorHost().postCheckAndPut(row, family,
2759                 qualifier, compareOp, comparator, put, result);
2760             }
2761             processed = Boolean.valueOf(result);
2762           }
2763         } else {
2764           region.put(put, lock);
2765           processed = Boolean.TRUE;
2766         }
2767         break;
2768       case DELETE:
2769         Delete delete = ProtobufUtil.toDelete(mutate);
2770         lock = getLockFromId(delete.getLockId());
2771         if (request.hasCondition()) {
2772           Condition condition = request.getCondition();
2773           byte[] row = condition.getRow().toByteArray();
2774           byte[] family = condition.getFamily().toByteArray();
2775           byte[] qualifier = condition.getQualifier().toByteArray();

2778             ProtobufUtil.toComparator(condition.getComparator());

2802         builder.setProcessed(processed.booleanValue());

2817   /**
2818    * Scan data in a table.
2819    *
2820    * @param controller the RPC controller
2821    * @param request the scan request
2822    * @throws ServiceException
2823    */
2824   @Override
2825   public ScanResponse scan(final RpcController controller,
2826       final ScanRequest request) throws ServiceException {
2827     Leases.Lease lease = null;
2828     String scannerName = null;
2829     try {
2830       if (!request.hasScannerId() && !request.hasScan()) {
2831         throw new DoNotRetryIOException(
2832           "Missing required input: scannerId or scan");
2833       }
2834       long scannerId = -1;
2835       if (request.hasScannerId()) {
2836         scannerId = request.getScannerId();
2837         scannerName = String.valueOf(scannerId);
2838       }
2839       try {
2840         checkOpen();
2841       } catch (IOException e) {
2842         // If checkOpen failed, server not running or filesystem gone,
2843         // cancel this lease; filesystem is gone or we're closing or something.
2844         if (scannerName != null) {
2845           try {
2846             leases.cancelLease(scannerName);
2847           } catch (LeaseException le) {
2848             LOG.info("Server shutting down and client tried to access missing scanner " +
2849               scannerName);
2850           }
2851         }
2852         throw e;
2853       }
2854       requestCount.increment();
2855 
2856       try {
2857         int ttl = 0;
2858         HRegion region = null;
2859         RegionScanner scanner = null;
2860         RegionScannerHolder rsh = null;
2861         boolean moreResults = true;
2862         boolean closeScanner = false;
2863         ScanResponse.Builder builder = ScanResponse.newBuilder();
2864         if (request.hasCloseScanner()) {
2865           closeScanner = request.getCloseScanner();
2866         }
2867         int rows = 1;
2868         if (request.hasNumberOfRows()) {
2869           rows = request.getNumberOfRows();
2870         }
2871         if (request.hasScannerId()) {
2872           rsh = scanners.get(scannerName);
2873           if (rsh == null) {
2874             throw new UnknownScannerException(
2875               "Name: " + scannerName + ", already closed?");
2876           }
2877           scanner = rsh.s;
2878           region = getRegion(scanner.getRegionInfo().getRegionName());
2879         } else {
2880           region = getRegion(request.getRegion());
2881           ClientProtos.Scan protoScan = request.getScan();
2882           Scan scan = ProtobufUtil.toScan(protoScan);
2883           region.prepareScanner(scan);
2884           if (region.getCoprocessorHost() != null) {
2885             scanner = region.getCoprocessorHost().preScannerOpen(scan);
2886           }
2887           if (scanner == null) {
2888             scanner = region.getScanner(scan);
2889           }
2890           if (region.getCoprocessorHost() != null) {
2891             scanner = region.getCoprocessorHost().postScannerOpen(scan, scanner);
2892           }
2893           scannerId = addScanner(scanner);
2894           scannerName = String.valueOf(scannerId);
2895           ttl = this.scannerLeaseTimeoutPeriod;
2896         }
2897 
2898         if (rows > 0) {
2899           // if nextCallSeq does not match throw Exception straight away. This needs to be
2900           // performed even before checking of Lease.
2901           // See HBASE-5974
2902           if (request.hasNextCallSeq()) {
2903             if (rsh == null) {
2904               rsh = scanners.get(scannerName);
2905             }
2906             if (rsh != null) {
2907               if (request.getNextCallSeq() != rsh.nextCallSeq) {
2908                 throw new OutOfOrderScannerNextException("Expected nextCallSeq: " + rsh.nextCallSeq
2909                     + " But the nextCallSeq got from client: " + request.getNextCallSeq());
2910               }
2911               // Increment the nextCallSeq value which is the next expected from client.
2912               rsh.nextCallSeq++;
2913             }
2914           }
2915           try {
2916             // Remove lease while its being processed in server; protects against case
2917             // where processing of request takes > lease expiration time.
2918             lease = leases.removeLease(scannerName);
2919             List<Result> results = new ArrayList<Result>(rows);
2920             long currentScanResultSize = 0;
2921 
2922             boolean done = false;
2923             // Call coprocessor. Get region info from scanner.
2924             if (region != null && region.getCoprocessorHost() != null) {
2925               Boolean bypass = region.getCoprocessorHost().preScannerNext(
2926                 scanner, results, rows);
2927               if (!results.isEmpty()) {
2928                 for (Result r : results) {
2929                   for (KeyValue kv : r.raw()) {
2930                     currentScanResultSize += kv.heapSize();
2931                   }
2932                 }
2933               }
2934               if (bypass != null && bypass.booleanValue()) {
2935                 done = true;
2936               }
2937             }
2938 
2939             if (!done) {
2940               long maxResultSize = scanner.getMaxResultSize();
2941               if (maxResultSize <= 0) {
2942                 maxResultSize = maxScannerResultSize;
2943               }
2944               List<KeyValue> values = new ArrayList<KeyValue>();
2945               MultiVersionConsistencyControl.setThreadReadPoint(scanner.getMvccReadPoint());
2946               region.startRegionOperation();
2947               try {
2948                 int i = 0;
2949                 synchronized(scanner) {
2950                   for (; i < rows
2951                       && currentScanResultSize < maxResultSize; i++) {
2952                     // Collect values to be returned here
2953                     boolean moreRows = scanner.nextRaw(values);
2954                     if (!values.isEmpty()) {
2955                       for (KeyValue kv : values) {
2956                         currentScanResultSize += kv.heapSize();
2957                       }
2958                       results.add(new Result(values));
2959                     }
2960                     if (!moreRows) {
2961                       break;
2962                     }
2963                     values.clear();
2964                   }
2965                 }
2966                 region.readRequestsCount.add(i);
2967               } finally {
2968                 region.closeRegionOperation();
2969               }
2970 
2971               // coprocessor postNext hook
2972               if (region != null && region.getCoprocessorHost() != null) {
2973                 region.getCoprocessorHost().postScannerNext(scanner, results, rows, true);
2974               }
2975             }
2976 
2977             // If the scanner's filter - if any - is done with the scan
2978             // and wants to tell the client to stop the scan. This is done by passing
2979             // a null result, and setting moreResults to false.
2980             if (scanner.isFilterDone() && results.isEmpty()) {
2981               moreResults = false;
2982               results = null;
2983             } else {
2984               for (Result result: results) {
2985                 if (result != null) {
2986                   builder.addResult(ProtobufUtil.toResult(result));
2987                 }
2988               }
2989             }
2990           } finally {
2991             // We're done. On way out re-add the above removed lease.
2992             // Adding resets expiration time on lease.
2993             if (scanners.containsKey(scannerName)) {
2994               if (lease != null) leases.addLease(lease);
2995               ttl = this.scannerLeaseTimeoutPeriod;
2996             }
2997           }
2998         }
2999 
3000         if (!moreResults || closeScanner) {
3001           ttl = 0;
3002           moreResults = false;
3003           if (region != null && region.getCoprocessorHost() != null) {
3004             if (region.getCoprocessorHost().preScannerClose(scanner)) {
3005               return builder.build(); // bypass
3006             }
3007           }
3008           rsh = scanners.remove(scannerName);
3009           if (rsh != null) {
3010             scanner = rsh.s;
3011             scanner.close();
3012             leases.cancelLease(scannerName);
3013             if (region != null && region.getCoprocessorHost() != null) {
3014               region.getCoprocessorHost().postScannerClose(scanner);
3015             }
3016           }
3017         }
3018 
3019         if (ttl > 0) {
3020           builder.setTtl(ttl);
3021         }
3022         builder.setScannerId(scannerId);
3023         builder.setMoreResults(moreResults);
3024         return builder.build();
3025       } catch (Throwable t) {
3026         if (scannerName != null &&
3027             t instanceof NotServingRegionException) {
3028           scanners.remove(scannerName);
3029         }
3030         throw convertThrowableToIOE(cleanup(t));
3031       }
3032     } catch (IOException ie) {
3033       throw new ServiceException(ie);
3034     }
3035   }
