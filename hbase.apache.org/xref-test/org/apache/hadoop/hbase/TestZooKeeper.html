109   private ZooKeeperWatcher getZooKeeperWatcher(HConnection c) throws
110     NoSuchMethodException, InvocationTargetException, IllegalAccessException {
111 
112     Method getterZK = c.getClass().getMethod("getKeepAliveZooKeeperWatcher");
113     getterZK.setAccessible(true);
114 
115     return (ZooKeeperWatcher) getterZK.invoke(c);

126   public void testClientSessionExpired() throws Exception {
127     Configuration c = new Configuration(TEST_UTIL.getConfiguration());
128 
129     // We don't want to share the connection as we will check its state
130     c.set(HConstants.HBASE_CLIENT_INSTANCE_ID, "1111");
131 
132     HConnection connection = HConnectionManager.getConnection(c);
133 
134     ZooKeeperWatcher connectionZK = getZooKeeperWatcher(connection);
135     LOG.info("ZooKeeperWatcher= 0x"+ Integer.toHexString(
136       connectionZK.hashCode()));
137     LOG.info("getRecoverableZooKeeper= 0x"+ Integer.toHexString(
138       connectionZK.getRecoverableZooKeeper().hashCode()));
139     LOG.info("session="+Long.toHexString(
140       connectionZK.getRecoverableZooKeeper().getSessionId()));
141 
142     TEST_UTIL.expireSession(connectionZK);
143 
144     LOG.info("Before using zkw state=" +
145       connectionZK.getRecoverableZooKeeper().getState());
146     // provoke session expiration by doing something with ZK
147     try {
148       connectionZK.getRecoverableZooKeeper().getZooKeeper().exists(
149         "/1/1", false);
150     } catch (KeeperException ignored) {
151     }
152 
153     // Check that the old ZK connection is closed, means we did expire
154     States state = connectionZK.getRecoverableZooKeeper().getState();
155     LOG.info("After using zkw state=" + state);
156     LOG.info("session="+Long.toHexString(
157       connectionZK.getRecoverableZooKeeper().getSessionId()));
158 
159     // It's asynchronous, so we may have to wait a little...
160     final long limit1 = System.currentTimeMillis() + 3000;
161     while (System.currentTimeMillis() < limit1 && state != States.CLOSED){
162       state = connectionZK.getRecoverableZooKeeper().getState();
163     }
164     LOG.info("After using zkw loop=" + state);
165     LOG.info("ZooKeeper should have timed out");
166     LOG.info("session="+Long.toHexString(
167       connectionZK.getRecoverableZooKeeper().getSessionId()));
168 
169     // It's surprising but sometimes we can still be in connected state.
170     // As it's known (even if not understood) we don't make the the test fail
171     // for this reason.)
172     // Assert.assertTrue("state=" + state, state == States.CLOSED);
173 
174     // Check that the client recovered
175     ZooKeeperWatcher newConnectionZK = getZooKeeperWatcher(connection);
176 
177     States state2 = newConnectionZK.getRecoverableZooKeeper().getState();
178     LOG.info("After new get state=" +state2);
179 
180     // As it's an asynchronous event we may got the same ZKW, if it's not
181     //  yet invalidated. Hence this loop.
182     final long limit2 = System.currentTimeMillis() + 3000;
183     while (System.currentTimeMillis() < limit2 &&
184       state2 != States.CONNECTED && state2 != States.CONNECTING) {
185 
186       newConnectionZK = getZooKeeperWatcher(connection);
187       state2 = newConnectionZK.getRecoverableZooKeeper().getState();
188     }
189     LOG.info("After new get state loop=" + state2);
190 
191     Assert.assertTrue(
192       state2 == States.CONNECTED || state2 == States.CONNECTING);
193 
194     connection.close();
195   }

258       Configuration otherConf = new Configuration(TEST_UTIL.getConfiguration());

261 
262       // dummy, just to open the connection
263       localMeta.exists(new Get(HConstants.LAST_ROW));
264       ipMeta.exists(new Get(HConstants.LAST_ROW));
265 
266       // make sure they aren't the same
267       ZooKeeperWatcher z1 =
268         getZooKeeperWatcher(HConnectionManager.getConnection(localMeta.getConfiguration()));
269       ZooKeeperWatcher z2 =
270         getZooKeeperWatcher(HConnectionManager.getConnection(otherConf));
271       assertFalse(z1 == z2);
272       assertFalse(z1.getQuorum().equals(z2.getQuorum()));
273 
274       localMeta.close();
275       ipMeta.close();
276     } catch (Exception e) {
277       e.printStackTrace();
278       fail();
279     }

283    * Create a bunch of znodes in a hierarchy, try deleting one that has childs
284    * (it will fail), then delete it recursively, then delete the last znode
285    * @throws Exception
286    */
287   @Test
288   public void testZNodeDeletes() throws Exception {
289     ZooKeeperWatcher zkw = new ZooKeeperWatcher(
290       new Configuration(TEST_UTIL.getConfiguration()), 
291       TestZooKeeper.class.getName(), null);
292     ZKUtil.createWithParents(zkw, "/l1/l2/l3/l4");
293     try {
294       ZKUtil.deleteNode(zkw, "/l1/l2");
295       fail("We should not be able to delete if znode has childs");
296     } catch (KeeperException ex) {
297       assertNotNull(ZKUtil.getDataNoWatch(zkw, "/l1/l2/l3/l4", null));
298     }
299     ZKUtil.deleteNodeRecursively(zkw, "/l1/l2");
300     // make sure it really is deleted
301     assertNull(ZKUtil.getDataNoWatch(zkw, "/l1/l2/l3/l4", null));
302 
303     // do the same delete again and make sure it doesn't crash
304     ZKUtil.deleteNodeRecursively(zkw, "/l1/l2");
305 
306     ZKUtil.deleteNode(zkw, "/l1");
307     assertNull(ZKUtil.getDataNoWatch(zkw, "/l1/l2", null));
308   }

321   private void testKey(String ensemble, String port, String znode)
322       throws IOException {
323     Configuration conf = new Configuration();
324     String key = ensemble+":"+port+":"+znode;
325     String[] parts = ZKUtil.transformClusterKey(key);
326     assertEquals(ensemble, parts[0]);
327     assertEquals(port, parts[1]);
328     assertEquals(znode, parts[2]);
329     ZKUtil.applyClusterKeyToConf(conf, key);
330     assertEquals(parts[0], conf.get(HConstants.ZOOKEEPER_QUORUM));
331     assertEquals(parts[1], conf.get(HConstants.ZOOKEEPER_CLIENT_PORT));
332     assertEquals(parts[2], conf.get(HConstants.ZOOKEEPER_ZNODE_PARENT));
333     String reconstructedKey = ZKUtil.getZooKeeperClusterKey(conf);
334     assertEquals(key, reconstructedKey);
335   }

346       KeeperException, IOException {
347     Configuration c = TEST_UTIL.getConfiguration();
348 
349     String aclZnode = "/aclRoot";
350     String quorumServers = ZKConfig.getZKQuorumServersString(c);
351     int sessionTimeout = 5 * 1000; // 5 seconds
352     ZooKeeper zk = new ZooKeeper(quorumServers, sessionTimeout, EmptyWatcher.instance);
353     zk.addAuthInfo("digest", "hbase:rox".getBytes());
354 
355     // Assumes the  root of the ZooKeeper space is writable as it creates a node
356     // wherever the cluster home is defined.
357     ZooKeeperWatcher zk2 = new ZooKeeperWatcher(TEST_UTIL.getConfiguration(),
358       "testMasterAddressManagerFromZK", null);
359 
360     // I set this acl after the attempted creation of the cluster home node.
361     // Add retries in case of retryable zk exceptions.
362     while (true) {
363       try {
364         zk.setACL("/", ZooDefs.Ids.CREATOR_ALL_ACL, -1);
365         break;
366       } catch (KeeperException e) {
367         switch (e.code()) {
368           case CONNECTIONLOSS:
369           case SESSIONEXPIRED:
370           case OPERATIONTIMEOUT:
371             LOG.warn("Possibly transient ZooKeeper exception: " + e);
372             Threads.sleep(100);
373             break;
374          default:
375             throw e;
376         }
377       }
378     }
379 
380     while (true) {
381       try {
382         zk.create(aclZnode, null, ZooDefs.Ids.CREATOR_ALL_ACL, CreateMode.PERSISTENT);
383         break;
384       } catch (KeeperException e) {
385         switch (e.code()) {
386           case CONNECTIONLOSS:
387           case SESSIONEXPIRED:
388           case OPERATIONTIMEOUT:
389             LOG.warn("Possibly transient ZooKeeper exception: " + e);
390             Threads.sleep(100);
391             break;
392          default:
393             throw e;
394         }
395       }
396     }
397     zk.close();
398     ZKUtil.createAndFailSilent(zk2, aclZnode);
399  }

414   /**
415    * Tests that the master does not call retainAssignment after recovery from expired zookeeper
416    * session. Without the HBASE-6046 fix master always tries to assign all the user regions by
417    * calling retainAssignment.
418    */
419   @Test
420   public void testRegionAssignmentAfterMasterRecoveryDueToZKExpiry() throws Exception {
421     MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
422     cluster.startRegionServer();
423     HMaster m = cluster.getMaster();
424     // now the cluster is up. So assign some regions.
425     HBaseAdmin admin = new HBaseAdmin(TEST_UTIL.getConfiguration());
426     byte[][] SPLIT_KEYS = new byte[][] { Bytes.toBytes("a"), Bytes.toBytes("b"),
427         Bytes.toBytes("c"), Bytes.toBytes("d"), Bytes.toBytes("e"), Bytes.toBytes("f"),
428         Bytes.toBytes("g"), Bytes.toBytes("h"), Bytes.toBytes("i"), Bytes.toBytes("j") };
429 
430     String tableName = "testRegionAssignmentAfterMasterRecoveryDueToZKExpiry";
431     admin.createTable(new HTableDescriptor(tableName), SPLIT_KEYS);
432     ZooKeeperWatcher zooKeeperWatcher = HBaseTestingUtility.getZooKeeperWatcher(TEST_UTIL);
433     ZKAssign.blockUntilNoRIT(zooKeeperWatcher);
434     m.getZooKeeperWatcher().close();
435     MockLoadBalancer.retainAssignCalled = false;
436     m.abort("Test recovery from zk session expired", new KeeperException.SessionExpiredException());
437     assertFalse(m.isStopped());
438     // The recovered master should not call retainAssignment, as it is not a
439     // clean startup.
440     assertFalse("Retain assignment should not be called", MockLoadBalancer.retainAssignCalled);
441   }

443   /**
444    * Tests whether the logs are split when master recovers from a expired zookeeper session and an
445    * RS goes down.
446    */
447   @Test(timeout = 240000)
448   public void testLogSplittingAfterMasterRecoveryDueToZKExpiry() throws IOException,
449       KeeperException, InterruptedException {
450     MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
451     cluster.startRegionServer();
452     HMaster m = cluster.getMaster();
453     // now the cluster is up. So assign some regions.
454     HBaseAdmin admin = new HBaseAdmin(TEST_UTIL.getConfiguration());
455     byte[][] SPLIT_KEYS = new byte[][] { Bytes.toBytes("1"), Bytes.toBytes("2"),
456         Bytes.toBytes("3"), Bytes.toBytes("4"), Bytes.toBytes("5") };
457 
458     String tableName = "testLogSplittingAfterMasterRecoveryDueToZKExpiry";
459     HTableDescriptor htd = new HTableDescriptor(tableName);
460     HColumnDescriptor hcd = new HColumnDescriptor("col");
461     htd.addFamily(hcd);
462     admin.createTable(htd, SPLIT_KEYS);
463     ZooKeeperWatcher zooKeeperWatcher = HBaseTestingUtility.getZooKeeperWatcher(TEST_UTIL);
464     ZKAssign.blockUntilNoRIT(zooKeeperWatcher);
465     HTable table = new HTable(TEST_UTIL.getConfiguration(), tableName);
466 
467     Put p = null;
468     int numberOfPuts = 0;
469     for (numberOfPuts = 0; numberOfPuts < 6; numberOfPuts++) {
470       p = new Put(Bytes.toBytes(numberOfPuts));
471       p.add(Bytes.toBytes("col"), Bytes.toBytes("ql"), Bytes.toBytes("value" + numberOfPuts));
472       table.put(p);
473     }
474     m.getZooKeeperWatcher().close();
475     m.abort("Test recovery from zk session expired", new KeeperException.SessionExpiredException());
476     assertFalse(m.isStopped());
477     cluster.getRegionServer(0).abort("Aborting");
478     // Without patch for HBASE-6046 this test case will always timeout
479     // with patch the test case should pass.
480     Scan scan = new Scan();
481     int numberOfRows = 0;
482     ResultScanner scanner = table.getScanner(scan);
483     Result[] result = scanner.next(1);
484     while (result != null && result.length > 0) {
485       numberOfRows++;
486       result = scanner.next(1);
487     }
488     assertEquals("Number of rows should be equal to number of puts.", numberOfPuts, numberOfRows);
489   }
