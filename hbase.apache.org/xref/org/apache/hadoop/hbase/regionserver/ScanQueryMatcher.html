41    // Optimization so we can skip lots of compares when we decide to skip

53    /*
54     * The following three booleans define how we deal with deletes.
55     * There are three different aspects:
56     * 1. Whether to keep delete markers. This is used in compactions.
57     *    Minor compactions always keep delete markers.
58     * 2. Whether to keep deleted rows. This is also used in compactions,
59     *    if the store is set to keep deleted rows. This implies keeping
60     *    the delete markers as well.
61     *    In this case deleted rows are subject to the normal max version
62     *    and TTL/min version rules just like "normal" rows.
63     * 3. Whether a scan can do time travel queries even before deleted
64     *    marker to reach deleted rows.
65     */
66    /** whether to retain delete markers */
67    private final boolean retainDeletesInOutput;
68    /** whether to return deleted rows */
69    private final boolean keepDeletedCells;
70    /** whether time range queries can see rows "behind" a delete */
71    private final boolean seePastDeleteMarkers;

75    private final ColumnTracker columns;

78    private final KeyValue startKey;

100   /**
101    * This variable shows whether there is an null column in the query. There
102    * always exists a null column in the wildcard column query.
103    * There maybe exists a null column in the explicit column query based on the
104    * first column.

106   private boolean hasNullColumn = true;

108   // By default, when hbase.hstore.time.to.purge.deletes is 0ms, a delete
109   // marker is always removed during a major compaction. If set to non-zero
110   // value then major compaction will try to keep a delete marker around for
111   // the given number of milliseconds. We want to keep the delete markers
112   // around a bit longer because old puts might appear out-of-order. For
113   // example, during log replication between two clusters.
114   //
115   // If the delete marker has lived longer than its column-family's TTL then
116   // the delete marker will be removed even if time.to.purge.deletes has not
117   // passed. This is because all the Puts that this delete marker can influence
118   // would have also expired. (Removing of delete markers on col family TTL will
119   // not happen if min-versions is set to non-zero)
120   //
121   // But, if time.to.purge.deletes has not expired then a delete
122   // marker will not be removed just because there are no Puts that it is
123   // currently influencing. This is because Puts, that this delete can
124   // influence.  may appear out of order.
125   private final long timeToPurgeDeletes;
126   
127   private final boolean isUserScan;

129   /**
130    * Construct a QueryMatcher for a scan
131    * @param scan
132    * @param scanInfo The store's immutable scan info
133    * @param columns
134    * @param scanType Type of the scan
135    * @param earliestPutTs Earliest put seen in any of the store files.
136    * @param oldestUnexpiredTS the oldest timestamp we are interested in,
137    *  based on TTL
138    */
139   public ScanQueryMatcher(Scan scan, HStore.ScanInfo scanInfo,
140       NavigableSet<byte[]> columns, ScanType scanType,
141       long readPointToUse, long earliestPutTs, long oldestUnexpiredTS) {
142     this.tr = scan.getTimeRange();
143     this.rowComparator = scanInfo.getComparator().getRawComparator();
144     this.deletes =  new ScanDeleteTracker();
145     this.stopRow = scan.getStopRow();
146     this.startKey = KeyValue.createFirstDeleteFamilyOnRow(scan.getStartRow(),
147         scanInfo.getFamily());
148     this.filter = scan.getFilter();
149     this.earliestPutTs = earliestPutTs;
150     this.maxReadPointToTrackVersions = readPointToUse;
151     this.timeToPurgeDeletes = scanInfo.getTimeToPurgeDeletes();
152 
153     /* how to deal with deletes */
154     this.isUserScan = scanType == ScanType.USER_SCAN;
155     // keep deleted cells: if compaction or raw scan
156     this.keepDeletedCells = (scanInfo.getKeepDeletedCells() && !isUserScan) || scan.isRaw();
157     // retain deletes: if minor compaction or raw scan
158     this.retainDeletesInOutput = scanType == ScanType.MINOR_COMPACT || scan.isRaw();
159     // seePastDeleteMarker: user initiated scans
160     this.seePastDeleteMarkers = scanInfo.getKeepDeletedCells() && isUserScan;
161 
162     int maxVersions = Math.min(scan.getMaxVersions(), scanInfo.getMaxVersions());
163     // Single branch to deal with two types of reads (columns vs all in family)
164     if (columns == null || columns.size() == 0) {
165       // there is always a null column in the wildcard column query.
166       hasNullColumn = true;
167 
168       // use a specialized scan for wildcard column tracker.
169       this.columns = new ScanWildcardColumnTracker(
170           scanInfo.getMinVersions(), maxVersions, oldestUnexpiredTS);
171     } else {
172       // whether there is null column in the explicit column query
173       hasNullColumn = (columns.first().length == 0);
174 
175       // We can share the ExplicitColumnTracker, diff is we reset
176       // between rows, not between storefiles.
177       this.columns = new ExplicitColumnTracker(columns,
178           scanInfo.getMinVersions(), maxVersions, oldestUnexpiredTS);
179     }
180   }

200   /**
201    * Determines if the caller should do one of several things:
202    * - seek/skip to the next row (MatchCode.SEEK_NEXT_ROW)
203    * - seek/skip to the next column (MatchCode.SEEK_NEXT_COL)
204    * - include the current KeyValue (MatchCode.INCLUDE)
205    * - ignore the current KeyValue (MatchCode.SKIP)
206    * - got to the next row (MatchCode.DONE)
207    *
208    * @param kv KeyValue to check
209    * @return The match code instance.
210    * @throws IOException in case there is an internal consistency problem
211    *      caused by a data corruption.
212    */
213   public MatchCode match(KeyValue kv) throws IOException {
214     if (filter != null && filter.filterAllRemaining()) {
215       return MatchCode.DONE_SCAN;
216     }
217 
218     byte [] bytes = kv.getBuffer();
219     int offset = kv.getOffset();
220     int initialOffset = offset;
221 
222     int keyLength = Bytes.toInt(bytes, offset, Bytes.SIZEOF_INT);
223     offset += KeyValue.ROW_OFFSET;
224 
225     short rowLength = Bytes.toShort(bytes, offset, Bytes.SIZEOF_SHORT);
226     offset += Bytes.SIZEOF_SHORT;
227 
228     int ret = this.rowComparator.compareRows(row, this.rowOffset, this.rowLength,
229         bytes, offset, rowLength);
230     if (ret <= -1) {
231       return MatchCode.DONE;
232     } else if (ret >= 1) {
233       // could optimize this, if necessary?
234       // Could also be called SEEK_TO_CURRENT_ROW, but this
235       // should be rare/never happens.
236       return MatchCode.SEEK_NEXT_ROW;
237     }
238 
239     // optimize case.
240     if (this.stickyNextRow)
241         return MatchCode.SEEK_NEXT_ROW;
242 
243     if (this.columns.done()) {
244       stickyNextRow = true;
245       return MatchCode.SEEK_NEXT_ROW;
246     }
247 
248     //Passing rowLength
249     offset += rowLength;
250 
251     //Skipping family
252     byte familyLength = bytes [offset];
253     offset += familyLength + 1;
254 
255     int qualLength = keyLength + KeyValue.ROW_OFFSET -
256       (offset - initialOffset) - KeyValue.TIMESTAMP_TYPE_SIZE;
257 
258     long timestamp = kv.getTimestamp();
259     // check for early out based on timestamp alone
260     if (columns.isDone(timestamp)) {
261         return columns.getNextRowOrNextColumn(bytes, offset, qualLength);
262     }
263 
264     /*
265      * The delete logic is pretty complicated now.
266      * This is corroborated by the following:
267      * 1. The store might be instructed to keep deleted rows around.
268      * 2. A scan can optionally see past a delete marker now.
269      * 3. If deleted rows are kept, we have to find out when we can
270      *    remove the delete markers.
271      * 4. Family delete markers are always first (regardless of their TS)
272      * 5. Delete markers should not be counted as version
273      * 6. Delete markers affect puts of the *same* TS
274      * 7. Delete marker need to be version counted together with puts
275      *    they affect
276      */
277     byte type = kv.getType();
278     if (kv.isDelete()) {
279       if (!keepDeletedCells) {
280         // first ignore delete markers if the scanner can do so, and the
281         // range does not include the marker
282         //
283         // during flushes and compactions also ignore delete markers newer
284         // than the readpoint of any open scanner, this prevents deleted
285         // rows that could still be seen by a scanner from being collected
286         boolean includeDeleteMarker = seePastDeleteMarkers ?
287             tr.withinTimeRange(timestamp) :
288             tr.withinOrAfterTimeRange(timestamp);
289         if (includeDeleteMarker
290             && kv.getMemstoreTS() <= maxReadPointToTrackVersions) {
291           this.deletes.add(bytes, offset, qualLength, timestamp, type);
292         }
293         // Can't early out now, because DelFam come before any other keys
294       }
295       if (retainDeletesInOutput
296           || (!isUserScan && (EnvironmentEdgeManager.currentTimeMillis() - timestamp) <= timeToPurgeDeletes)
297           || kv.getMemstoreTS() > maxReadPointToTrackVersions) {
298         // always include or it is not time yet to check whether it is OK
299         // to purge deltes or not
300         return MatchCode.INCLUDE;
301       } else if (keepDeletedCells) {
302         if (timestamp < earliestPutTs) {
303           // keeping delete rows, but there are no puts older than
304           // this delete in the store files.
305           return columns.getNextRowOrNextColumn(bytes, offset, qualLength);
306         }
307         // else: fall through and do version counting on the
308         // delete markers
309       } else {
310         return MatchCode.SKIP;
311       }
312       // note the following next else if...
313       // delete marker are not subject to other delete markers
314     } else if (!this.deletes.isEmpty()) {
315       DeleteResult deleteResult = deletes.isDeleted(bytes, offset, qualLength,
316           timestamp);
317       switch (deleteResult) {
318         case FAMILY_DELETED:
319         case COLUMN_DELETED:
320           return columns.getNextRowOrNextColumn(bytes, offset, qualLength);
321         case VERSION_DELETED:
322           return MatchCode.SKIP;
323         case NOT_DELETED:
324           break;
325         default:
326           throw new RuntimeException("UNEXPECTED");
327         }
328     }
329 
330     int timestampComparison = tr.compare(timestamp);
331     if (timestampComparison >= 1) {
332       return MatchCode.SKIP;
333     } else if (timestampComparison <= -1) {
334       return columns.getNextRowOrNextColumn(bytes, offset, qualLength);
335     }
336 
337     /**
338      * Filters should be checked before checking column trackers. If we do
339      * otherwise, as was previously being done, ColumnTracker may increment its
340      * counter for even that KV which may be discarded later on by Filter. This
341      * would lead to incorrect results in certain cases.
342      */
343     ReturnCode filterResponse = ReturnCode.SKIP;
344     if (filter != null) {
345       filterResponse = filter.filterKeyValue(kv);
346       if (filterResponse == ReturnCode.SKIP) {
347         return MatchCode.SKIP;
348       } else if (filterResponse == ReturnCode.NEXT_COL) {
349         return columns.getNextRowOrNextColumn(bytes, offset, qualLength);
350       } else if (filterResponse == ReturnCode.NEXT_ROW) {
351         stickyNextRow = true;
352         return MatchCode.SEEK_NEXT_ROW;
353       } else if (filterResponse == ReturnCode.SEEK_NEXT_USING_HINT) {
354         return MatchCode.SEEK_NEXT_USING_HINT;
355       }
356     }
357 
358     MatchCode colChecker = columns.checkColumn(bytes, offset, qualLength,
359         timestamp, type, kv.getMemstoreTS() > maxReadPointToTrackVersions);
360     /*
361      * According to current implementation, colChecker can only be
362      * SEEK_NEXT_COL, SEEK_NEXT_ROW, SKIP or INCLUDE. Therefore, always return
363      * the MatchCode. If it is SEEK_NEXT_ROW, also set stickyNextRow.
364      */
365     if (colChecker == MatchCode.SEEK_NEXT_ROW) {
366       stickyNextRow = true;
367     } else if (filter != null && colChecker == MatchCode.INCLUDE &&
368                filterResponse == ReturnCode.INCLUDE_AND_NEXT_COL) {
369       return MatchCode.INCLUDE_AND_SEEK_NEXT_COL;
370     }
371     return colChecker;
372 
373   }

375   public boolean moreRowsMayExistAfter(KeyValue kv) {
376     if (!Bytes.equals(stopRow , HConstants.EMPTY_END_ROW) &&
377         rowComparator.compareRows(kv.getBuffer(),kv.getRowOffset(),
378             kv.getRowLength(), stopRow, 0, stopRow.length) >= 0) {
379       // KV >= STOPROW
380       // then NO there is nothing left.
381       return false;
382     } else {
383       return true;
384     }
385   }

429   public KeyValue getKeyForNextColumn(KeyValue kv) {
430     ColumnCount nextColumn = columns.getColumnHint();
431     if (nextColumn == null) {
432       return KeyValue.createLastOnRow(
433           kv.getBuffer(), kv.getRowOffset(), kv.getRowLength(),
434           kv.getBuffer(), kv.getFamilyOffset(), kv.getFamilyLength(),
435           kv.getBuffer(), kv.getQualifierOffset(), kv.getQualifierLength());
436     } else {
437       return KeyValue.createFirstOnRow(
438           kv.getBuffer(), kv.getRowOffset(), kv.getRowLength(),
439           kv.getBuffer(), kv.getFamilyOffset(), kv.getFamilyLength(),
440           nextColumn.getBuffer(), nextColumn.getOffset(), nextColumn.getLength());
441     }
442   }

451   /**
452    * {@link #match} return codes.  These instruct the scanner moving through
453    * memstores and StoreFiles what to do with the current KeyValue.
454    * <p>
455    * Additionally, this contains "early-out" language to tell the scanner to
456    * move on to the next File (memstore or Storefile), or to return immediately.
457    */
458   public static enum MatchCode {
459     /**
460      * Include KeyValue in the returned result
461      */
462     INCLUDE,
463 
464     /**
465      * Do not include KeyValue in the returned result
466      */
467     SKIP,
468 
469     /**
470      * Do not include, jump to next StoreFile or memstore (in time order)
471      */
472     NEXT,
473 
474     /**
475      * Do not include, return current result
476      */
477     DONE,
478 
479     /**
480      * These codes are used by the ScanQueryMatcher
481      */
482 
483     /**
484      * Done with the row, seek there.
485      */
486     SEEK_NEXT_ROW,
487     /**
488      * Done with column, seek to next.
489      */
490     SEEK_NEXT_COL,
491 
492     /**
493      * Done with scan, thanks to the row filter.
494      */
495     DONE_SCAN,
496 
497     /*
498      * Seek to next key which is given as hint.
499      */
500     SEEK_NEXT_USING_HINT,
501 
502     /**
503      * Include KeyValue and done with column, seek to next.
504      */
505     INCLUDE_AND_SEEK_NEXT_COL,
506 
507     /**
508      * Include KeyValue and done with row, seek to next.
509      */
510     INCLUDE_AND_SEEK_NEXT_ROW,
511   }
