145   @Test
146   public void testCompoundBloomFilter() throws IOException {
147     conf.setBoolean(BloomFilterFactory.IO_STOREFILE_BLOOM_ENABLED, true);
148     for (int t = 0; t < NUM_TESTS; ++t) {
149       conf.setFloat(BloomFilterFactory.IO_STOREFILE_BLOOM_ERROR_RATE,
150           (float) TARGET_ERROR_RATES[t]);
151 
152       testIdMsg = "in test #" + t + ":";
153       Random generationRand = new Random(GENERATION_SEED);
154       List<KeyValue> kvs = createSortedKeyValues(generationRand, NUM_KV[t]);
155       BloomType bt = BLOOM_TYPES[t];
156       Path sfPath = writeStoreFile(t, bt, kvs);
157       readStoreFile(t, bt, kvs, sfPath);
158     }
159   }

161   /**
162    * Validates the false positive ratio by computing its z-value and comparing
163    * it to the provided threshold.
164    *
165    * @param falsePosRate experimental positive rate
166    * @param nTrials the number of Bloom filter checks
167    * @param zValueBoundary z-value boundary, positive for an upper bound and
168    *          negative for a lower bound
169    * @param cbf the compound Bloom filter we are using
170    * @param additionalMsg additional message to include in log output and
171    *          assertion failures
172    */
173   private void validateFalsePosRate(double falsePosRate, int nTrials,
174       double zValueBoundary, CompoundBloomFilter cbf, String additionalMsg) {
175     double p = BloomFilterFactory.getErrorRate(conf);
176     double zValue = (falsePosRate - p) / Math.sqrt(p * (1 - p) / nTrials);
177 
178     String assortedStatsStr = " (targetErrorRate=" + p + ", falsePosRate="
179         + falsePosRate + ", nTrials=" + nTrials + ")";
180     LOG.info("z-value is " + zValue + assortedStatsStr);
181 
182     boolean isUpperBound = zValueBoundary > 0;
183 
184     if (isUpperBound && zValue > zValueBoundary ||
185         !isUpperBound && zValue < zValueBoundary) {
186       String errorMsg = "False positive rate z-value " + zValue + " is "
187           + (isUpperBound ? "higher" : "lower") + " than " + zValueBoundary
188           + assortedStatsStr + ". Per-chunk stats:\n"
189           + cbf.formatTestingStats();
190       fail(errorMsg + additionalMsg);
191     }
192   }

194   private void readStoreFile(int t, BloomType bt, List<KeyValue> kvs,
195       Path sfPath) throws IOException {
196     StoreFile sf = new StoreFile(fs, sfPath, conf, cacheConf, bt,
197         NoOpDataBlockEncoder.INSTANCE);
198     StoreFile.Reader r = sf.createReader();
199     final boolean pread = true; // does not really matter
200     StoreFileScanner scanner = r.getStoreFileScanner(true, pread);
201 
202     {
203       // Test for false negatives (not allowed).
204       int numChecked = 0;
205       for (KeyValue kv : kvs) {
206         byte[] row = kv.getRow();
207         boolean present = isInBloom(scanner, row, kv.getQualifier());
208         assertTrue(testIdMsg + " Bloom filter false negative on row "
209             + Bytes.toStringBinary(row) + " after " + numChecked
210             + " successful checks", present);
211         ++numChecked;
212       }
213     }
214 
215     // Test for false positives (some percentage allowed). We test in two modes:
216     // "fake lookup" which ignores the key distribution, and production mode.
217     for (boolean fakeLookupEnabled : new boolean[] { true, false }) {
218       ByteBloomFilter.setFakeLookupMode(fakeLookupEnabled);
219       try {
220         String fakeLookupModeStr = ", fake lookup is " + (fakeLookupEnabled ?
221             "enabled" : "disabled");
222         CompoundBloomFilter cbf = (CompoundBloomFilter) r.getGeneralBloomFilter();
223         cbf.enableTestingStats();
224         int numFalsePos = 0;
225         Random rand = new Random(EVALUATION_SEED);
226         int nTrials = NUM_KV[t] * 10;
227         for (int i = 0; i < nTrials; ++i) {
228           byte[] query = TestHFileWriterV2.randomRowOrQualifier(rand);
229           if (isInBloom(scanner, query, bt, rand)) {
230             numFalsePos += 1;
231           }
232         }
233         double falsePosRate = numFalsePos * 1.0 / nTrials;
234         LOG.debug(String.format(testIdMsg
235             + " False positives: %d out of %d (%f)",
236             numFalsePos, nTrials, falsePosRate) + fakeLookupModeStr);
237 
238         // Check for obvious Bloom filter crashes.
239         assertTrue("False positive is too high: " + falsePosRate + " (greater "
240             + "than " + TOO_HIGH_ERROR_RATE + ")" + fakeLookupModeStr,
241             falsePosRate < TOO_HIGH_ERROR_RATE);
242 
243         // Now a more precise check to see if the false positive rate is not
244         // too high. The reason we use a relaxed restriction for the real-world
245         // case as opposed to the "fake lookup" case is that our hash functions
246         // are not completely independent.
247 
248         double maxZValue = fakeLookupEnabled ? 1.96 : 2.5;
249         validateFalsePosRate(falsePosRate, nTrials, maxZValue, cbf,
250             fakeLookupModeStr);
251 
252         // For checking the lower bound we need to eliminate the last chunk,
253         // because it is frequently smaller and the false positive rate in it
254         // is too low. This does not help if there is only one under-sized
255         // chunk, though.
256         int nChunks = cbf.getNumChunks();
257         if (nChunks > 1) {
258           numFalsePos -= cbf.getNumPositivesForTesting(nChunks - 1);
259           nTrials -= cbf.getNumQueriesForTesting(nChunks - 1);
260           falsePosRate = numFalsePos * 1.0 / nTrials;
261           LOG.info(testIdMsg + " False positive rate without last chunk is " +
262               falsePosRate + fakeLookupModeStr);
263         }
264 
265         validateFalsePosRate(falsePosRate, nTrials, -2.58, cbf,
266             fakeLookupModeStr);
267       } finally {
268         ByteBloomFilter.setFakeLookupMode(false);
269       }
270     }
271 
272     r.close(true); // end of test so evictOnClose
273   }

281   private boolean isInBloom(StoreFileScanner scanner, byte[] row,
282       byte[] qualifier) {
283     Scan scan = new Scan(row, row);
284     TreeSet<byte[]> columns = new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR);
285     columns.add(qualifier);
286     return scanner.shouldUseScanner(scan, columns, Long.MIN_VALUE);
287   }

289   private Path writeStoreFile(int t, BloomType bt, List<KeyValue> kvs)
290       throws IOException {
291     conf.setInt(BloomFilterFactory.IO_STOREFILE_BLOOM_BLOCK_SIZE,
292         BLOOM_BLOCK_SIZES[t]);
293     conf.setBoolean(CacheConfig.CACHE_BLOCKS_ON_WRITE_KEY, true);
294     cacheConf = new CacheConfig(conf);
295 
296     StoreFile.Writer w = new StoreFile.WriterBuilder(conf, cacheConf, fs,
297         BLOCK_SIZES[t])
298             .withOutputDir(TEST_UTIL.getDataTestDir())
299             .withBloomType(bt)
300             .withChecksumType(HFile.DEFAULT_CHECKSUM_TYPE)
301             .withBytesPerChecksum(HFile.DEFAULT_BYTES_PER_CHECKSUM)
302             .build();
303 
304     assertTrue(w.hasGeneralBloom());
305     assertTrue(w.getGeneralBloomWriter() instanceof CompoundBloomFilterWriter);
306     CompoundBloomFilterWriter cbbf =
307         (CompoundBloomFilterWriter) w.getGeneralBloomWriter();
308 
309     int keyCount = 0;
310     KeyValue prev = null;
311     LOG.debug("Total keys/values to insert: " + kvs.size());
312     for (KeyValue kv : kvs) {
313       w.append(kv);
314 
315       // Validate the key count in the Bloom filter.
316       boolean newKey = true;
317       if (prev != null) {
318         newKey = !(bt == BloomType.ROW ? KeyValue.COMPARATOR.matchingRows(kv,
319             prev) : KeyValue.COMPARATOR.matchingRowColumn(kv, prev));
320       }
321       if (newKey)
322         ++keyCount;
323       assertEquals(keyCount, cbbf.getKeyCount());
324 
325       prev = kv;
326     }
327     w.close();
328 
329     return w.getPath();
330   }

332   @Test
333   public void testCompoundBloomSizing() {
334     int bloomBlockByteSize = 4096;
335     int bloomBlockBitSize = bloomBlockByteSize * 8;
336     double targetErrorRate = 0.01;
337     long maxKeysPerChunk = ByteBloomFilter.idealMaxKeys(bloomBlockBitSize,
338         targetErrorRate);
339 
340     long bloomSize1 = bloomBlockByteSize * 8;
341     long bloomSize2 = ByteBloomFilter.computeBitSize(maxKeysPerChunk,
342         targetErrorRate);
343 
344     double bloomSizeRatio = (bloomSize2 * 1.0 / bloomSize1);
345     assertTrue(Math.abs(bloomSizeRatio - 0.9999) < 0.0001);
346   }

348   @Test
349   public void testCreateKey() {
350     CompoundBloomFilterBase cbfb = new CompoundBloomFilterBase();
351     byte[] row = "myRow".getBytes();
352     byte[] qualifier = "myQualifier".getBytes();
353     byte[] rowKey = cbfb.createBloomKey(row, 0, row.length,
354         row, 0, 0);
355     byte[] rowColKey = cbfb.createBloomKey(row, 0, row.length,
356         qualifier, 0, qualifier.length);
357     KeyValue rowKV = KeyValue.createKeyValueFromKey(rowKey);
358     KeyValue rowColKV = KeyValue.createKeyValueFromKey(rowColKey);
359     assertEquals(rowKV.getTimestamp(), rowColKV.getTimestamp());
360     assertEquals(Bytes.toStringBinary(rowKV.getRow()),
361         Bytes.toStringBinary(rowColKV.getRow()));
362     assertEquals(0, rowKV.getQualifier().length);
363   }
