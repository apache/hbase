70      /**
71       * Represent an HLogSplit, i.e. a single HLog file.
72       * Start- and EndTime are managed by the split, so that HLog files can be
73       * filtered before WALEdits are passed to the mapper(s).
74       * @param logFileName
75       * @param fileSize

79      public HLogSplit(String logFileName, long fileSize, long startTime, long endTime) {

82        this.startTime = startTime;
83        this.endTime = endTime;
84      }

89      }

94        return new String[] {};
95      }

98        return logFileName;
99      }

109     @Override
110     public void readFields(DataInput in) throws IOException {
111       logFileName = in.readUTF();
112       fileSize = in.readLong();
113       startTime = in.readLong();
114       endTime = in.readLong();
115     }
116 
117     @Override
118     public void write(DataOutput out) throws IOException {
119       out.writeUTF(logFileName);

131   /**
132    * {@link RecordReader} for an {@link HLog} file.
133    */
134   static class HLogRecordReader extends RecordReader<HLogKey, WALEdit> {
135     private HLog.Reader reader = null;
136     private HLog.Entry currentEntry = new HLog.Entry();
137     private long startTime;
138     private long endTime;
139 
140     @Override
141     public void initialize(InputSplit split, TaskAttemptContext context)
142         throws IOException, InterruptedException {
143       HLogSplit hsplit = (HLogSplit)split;
144       Path logFile = new Path(hsplit.getLogFileName());
145       Configuration conf = context.getConfiguration();
146       LOG.info("Opening reader for "+split);
147       try {
148         this.reader = HLogFactory.createReader(logFile.getFileSystem(conf), 
149             logFile, conf);
150       } catch (EOFException x) {
151         LOG.info("Ignoring corrupted HLog file: " + logFile
152             + " (This is normal when a RegionServer crashed.)");
153       }
154       this.startTime = hsplit.getStartTime();
155       this.endTime = hsplit.getEndTime();
156     }
157 
158     @Override
159     public boolean nextKeyValue() throws IOException, InterruptedException {
160       if (reader == null) return false;
161 
162       HLog.Entry temp;
163       long i = -1;
164       do {
165         // skip older entries
166         try {
167           temp = reader.next(currentEntry);
168           i++;
169         } catch (EOFException x) {
170           LOG.info("Corrupted entry detected. Ignoring the rest of the file."
171               + " (This is normal when a RegionServer crashed.)");
172           return false;
173         }
174       }
175       while(temp != null && temp.getKey().getWriteTime() < startTime);
176 
177       if (temp == null) {
178         if (i > 0) LOG.info("Skipped " + i + " entries.");
179         LOG.info("Reached end of file.");
180         return false;
181       } else if (i > 0) {
182         LOG.info("Skipped " + i + " entries, until ts: " + temp.getKey().getWriteTime() + ".");
183       }
184       boolean res = temp.getKey().getWriteTime() <= endTime;
185       if (!res) {
186         LOG.info("Reached ts: " + temp.getKey().getWriteTime() + " ignoring the rest of the file.");
187       }
188       return res;
189     }
190 
191     @Override
192     public HLogKey getCurrentKey() throws IOException, InterruptedException {
193       return currentEntry.getKey();
194     }
195 
196     @Override
197     public WALEdit getCurrentValue() throws IOException, InterruptedException {
198       return currentEntry.getEdit();
199     }
200 
201     @Override
202     public float getProgress() throws IOException, InterruptedException {
203       // N/A depends on total number of entries, which is unknown
204       return 0;
205     }
206 
207     @Override
208     public void close() throws IOException {
209       LOG.info("Closing reader");
210       if (reader != null) this.reader.close();
211     }
212   }

214   @Override
215   public List<InputSplit> getSplits(JobContext context) throws IOException,
216       InterruptedException {
217     Configuration conf = context.getConfiguration();
218     Path inputDir = new Path(conf.get("mapred.input.dir"));
219 
220     long startTime = conf.getLong(START_TIME_KEY, Long.MIN_VALUE);
221     long endTime = conf.getLong(END_TIME_KEY, Long.MAX_VALUE);
222 
223     FileSystem fs = inputDir.getFileSystem(conf);
224     List<FileStatus> files = getFiles(fs, inputDir, startTime, endTime);
225 
226     List<InputSplit> splits = new ArrayList<InputSplit>(files.size());
227     for (FileStatus file : files) {
228       splits.add(new HLogSplit(file.getPath().toString(), file.getLen(), startTime, endTime));
229     }
230     return splits;
231   }
232 
233   private List<FileStatus> getFiles(FileSystem fs, Path dir, long startTime, long endTime)
234       throws IOException {
235     List<FileStatus> result = new ArrayList<FileStatus>();
236     LOG.debug("Scanning " + dir.toString() + " for HLog files");
237 
238     FileStatus[] files = fs.listStatus(dir);
239     if (files == null) return Collections.emptyList();
240     for (FileStatus file : files) {
241       if (file.isDir()) {
242         // recurse into sub directories
243         result.addAll(getFiles(fs, file.getPath(), startTime, endTime));
244       } else {
245         String name = file.getPath().toString();
246         int idx = name.lastIndexOf('.');
247         if (idx > 0) {
248           try {
249             long fileStartTime = Long.parseLong(name.substring(idx+1));
250             if (fileStartTime <= endTime) {
251               LOG.info("Found: " + name);
252               result.add(file);
253             }
254           } catch (NumberFormatException x) {
255             idx = 0;
256           }
257         }
258         if (idx == 0) {
259           LOG.warn("File " + name + " does not appear to be an HLog file. Skipping...");
260         }
261       }
262     }
263     return result;
264   }
