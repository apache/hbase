70  /**
71   * Writes HFiles. Passed KeyValues must arrive in order.
72   * Writes current time as the sequence id for the file. Sets the major compacted
73   * attribute on created hfiles. Calling write(null,null) will forceably roll
74   * all HFiles being written.
75   * <p>
76   * Using this class as part of a MapReduce job is best done 
77   * using {@link #configureIncrementalLoad(Job, HTable)}.
78   * @see KeyValueSortReducer
79   */
80  @InterfaceAudience.Public
81  @InterfaceStability.Stable
82  public class HFileOutputFormat extends FileOutputFormat<ImmutableBytesWritable, KeyValue> {
83    static Log LOG = LogFactory.getLog(HFileOutputFormat.class);
84    static final String COMPRESSION_CONF_KEY = "hbase.hfileoutputformat.families.compression";
85    TimeRangeTracker trt = new TimeRangeTracker();
86    private static final String DATABLOCK_ENCODING_CONF_KEY =
87       "hbase.mapreduce.hfileoutputformat.datablock.encoding";
88  
89    public RecordWriter<ImmutableBytesWritable, KeyValue> getRecordWriter(final TaskAttemptContext context)
90    throws IOException, InterruptedException {
91      // Get the path of the temporary output file
92      final Path outputPath = FileOutputFormat.getOutputPath(context);
93      final Path outputdir = new FileOutputCommitter(outputPath, context).getWorkPath();
94      final Configuration conf = context.getConfiguration();
95      final FileSystem fs = outputdir.getFileSystem(conf);
96      // These configs. are from hbase-*.xml
97      final long maxsize = conf.getLong(HConstants.HREGION_MAX_FILESIZE,
98          HConstants.DEFAULT_MAX_FILE_SIZE);
99      final int blocksize = conf.getInt("hbase.mapreduce.hfileoutputformat.blocksize",
100         HFile.DEFAULT_BLOCKSIZE);
101     // Invented config.  Add to hbase-*.xml if other than default compression.
102     final String defaultCompression = conf.get("hfile.compression",
103         Compression.Algorithm.NONE.getName());
104     final boolean compactionExclude = conf.getBoolean(
105         "hbase.mapreduce.hfileoutputformat.compaction.exclude", false);
106 
107     // create a map from column family to the compression algorithm
108     final Map<byte[], String> compressionMap = createFamilyCompressionMap(conf);
109 
110     String dataBlockEncodingStr = conf.get(DATABLOCK_ENCODING_CONF_KEY);
111     final HFileDataBlockEncoder encoder;
112     if (dataBlockEncodingStr == null) {
113       encoder = NoOpDataBlockEncoder.INSTANCE;
114     } else {
115       try {
116         encoder = new HFileDataBlockEncoderImpl(DataBlockEncoding
117             .valueOf(dataBlockEncodingStr));
118       } catch (IllegalArgumentException ex) {
119         throw new RuntimeException(
120             "Invalid data block encoding type configured for the param "
121                 + DATABLOCK_ENCODING_CONF_KEY + " : " + dataBlockEncodingStr);
122       }
123     }
124 
125     return new RecordWriter<ImmutableBytesWritable, KeyValue>() {
126       // Map of families to writers and how much has been output on the writer.
127       private final Map<byte [], WriterLength> writers =
128         new TreeMap<byte [], WriterLength>(Bytes.BYTES_COMPARATOR);
129       private byte [] previousRow = HConstants.EMPTY_BYTE_ARRAY;
130       private final byte [] now = Bytes.toBytes(System.currentTimeMillis());
131       private boolean rollRequested = false;
132 
133       public void write(ImmutableBytesWritable row, KeyValue kv)
134       throws IOException {
135         // null input == user explicitly wants to flush
136         if (row == null && kv == null) {
137           rollWriters();
138           return;
139         }
140 
141         byte [] rowKey = kv.getRow();
142         long length = kv.getLength();
143         byte [] family = kv.getFamily();
144         WriterLength wl = this.writers.get(family);
145 
146         // If this is a new column family, verify that the directory exists
147         if (wl == null) {
148           fs.mkdirs(new Path(outputdir, Bytes.toString(family)));
149         }
150 
151         // If any of the HFiles for the column families has reached
152         // maxsize, we need to roll all the writers
153         if (wl != null && wl.written + length >= maxsize) {
154           this.rollRequested = true;
155         }
156 
157         // This can only happen once a row is finished though
158         if (rollRequested && Bytes.compareTo(this.previousRow, rowKey) != 0) {
159           rollWriters();
160         }
161 
162         // create a new HLog writer, if necessary
163         if (wl == null || wl.writer == null) {
164           wl = getNewWriter(family, conf);
165         }
166 
167         // we now have the proper HLog writer. full steam ahead
168         kv.updateLatestStamp(this.now);
169         trt.includeTimestamp(kv);
170         wl.writer.append(kv);
171         wl.written += length;
172 
173         // Copy the row so we know when a row transition.
174         this.previousRow = rowKey;
175       }
176 
177       private void rollWriters() throws IOException {
178         for (WriterLength wl : this.writers.values()) {
179           if (wl.writer != null) {
180             LOG.info("Writer=" + wl.writer.getPath() +
181                 ((wl.written == 0)? "": ", wrote=" + wl.written));
182             close(wl.writer);
183           }
184           wl.writer = null;
185           wl.written = 0;
186         }
187         this.rollRequested = false;
188       }
189 
190       /* Create a new HFile.Writer.
191        * @param family
192        * @return A WriterLength, containing a new HFile.Writer.
193        * @throws IOException
194        */
195       private WriterLength getNewWriter(byte[] family, Configuration conf)
196           throws IOException {
197         WriterLength wl = new WriterLength();
198         Path familydir = new Path(outputdir, Bytes.toString(family));
199         String compression = compressionMap.get(family);
200         compression = compression == null ? defaultCompression : compression;
201         wl.writer = HFile.getWriterFactoryNoCache(conf)
202             .withPath(fs, StoreFile.getUniqueFile(fs, familydir))
203             .withBlockSize(blocksize)
204             .withCompression(compression)
205             .withComparator(KeyValue.KEY_COMPARATOR)
206             .withDataBlockEncoder(encoder)
207             .withChecksumType(HStore.getChecksumType(conf))
208             .withBytesPerChecksum(HStore.getBytesPerChecksum(conf))
209             .create();
210         this.writers.put(family, wl);
211         return wl;
212       }
213 
214       private void close(final HFile.Writer w) throws IOException {
215         if (w != null) {
216           w.appendFileInfo(StoreFile.BULKLOAD_TIME_KEY,
217               Bytes.toBytes(System.currentTimeMillis()));
218           w.appendFileInfo(StoreFile.BULKLOAD_TASK_KEY,
219               Bytes.toBytes(context.getTaskAttemptID().toString()));
220           w.appendFileInfo(StoreFile.MAJOR_COMPACTION_KEY,
221               Bytes.toBytes(true));
222           w.appendFileInfo(StoreFile.EXCLUDE_FROM_MINOR_COMPACTION_KEY,
223               Bytes.toBytes(compactionExclude));
224           w.appendFileInfo(StoreFile.TIMERANGE_KEY,
225               WritableUtils.toByteArray(trt));
226           w.close();
227         }
228       }
229 
230       public void close(TaskAttemptContext c)
231       throws IOException, InterruptedException {
232         for (WriterLength wl: this.writers.values()) {
233           close(wl.writer);
234         }
235       }
236     };
237   }

247   /**
248    * Return the start keys of all of the regions in this table,
249    * as a list of ImmutableBytesWritable.
250    */
251   private static List<ImmutableBytesWritable> getRegionStartKeys(HTable table)
252   throws IOException {
253     byte[][] byteKeys = table.getStartKeys();
254     ArrayList<ImmutableBytesWritable> ret =
255       new ArrayList<ImmutableBytesWritable>(byteKeys.length);
256     for (byte[] byteKey : byteKeys) {
257       ret.add(new ImmutableBytesWritable(byteKey));
258     }
259     return ret;
260   }

262   /**
263    * Write out a SequenceFile that can be read by TotalOrderPartitioner
264    * that contains the split points in startKeys.
265    * @param partitionsPath output path for SequenceFile
266    * @param startKeys the region start keys
267    */
268   private static void writePartitions(Configuration conf, Path partitionsPath,
269       List<ImmutableBytesWritable> startKeys) throws IOException {
270     if (startKeys.isEmpty()) {
271       throw new IllegalArgumentException("No regions passed");
272     }
273 
274     // We're generating a list of split points, and we don't ever
275     // have keys < the first region (which has an empty start key)
276     // so we need to remove it. Otherwise we would end up with an
277     // empty reducer with index 0
278     TreeSet<ImmutableBytesWritable> sorted =
279       new TreeSet<ImmutableBytesWritable>(startKeys);
280 
281     ImmutableBytesWritable first = sorted.first();
282     if (!first.equals(HConstants.EMPTY_BYTE_ARRAY)) {
283       throw new IllegalArgumentException(
284           "First region of table should have empty start key. Instead has: "
285           + Bytes.toStringBinary(first.get()));
286     }
287     sorted.remove(first);
288 
289     // Write the actual file
290     FileSystem fs = partitionsPath.getFileSystem(conf);
291     SequenceFile.Writer writer = SequenceFile.createWriter(fs,
292         conf, partitionsPath, ImmutableBytesWritable.class, NullWritable.class);
293 
294     try {
295       for (ImmutableBytesWritable startKey : sorted) {
296         writer.append(startKey, NullWritable.get());
297       }
298     } finally {
299       writer.close();
300     }
301   }

303   /**
304    * Configure a MapReduce Job to perform an incremental load into the given
305    * table. This
306    * <ul>
307    *   <li>Inspects the table to configure a total order partitioner</li>
308    *   <li>Uploads the partitions file to the cluster and adds it to the DistributedCache</li>
309    *   <li>Sets the number of reduce tasks to match the current number of regions</li>
310    *   <li>Sets the output key/value class to match HFileOutputFormat's requirements</li>
311    *   <li>Sets the reducer up to perform the appropriate sorting (either KeyValueSortReducer or
312    *     PutSortReducer)</li>
313    * </ul>
314    * The user should be sure to set the map output value class to either KeyValue or Put before
315    * running this function.
316    */
317   public static void configureIncrementalLoad(Job job, HTable table)
318   throws IOException {
319     Configuration conf = job.getConfiguration();
320 
321     job.setPartitionerClass(TotalOrderPartitioner.class);
322     job.setOutputKeyClass(ImmutableBytesWritable.class);
323     job.setOutputValueClass(KeyValue.class);
324     job.setOutputFormatClass(HFileOutputFormat.class);
325 
326     // Based on the configured map output class, set the correct reducer to properly
327     // sort the incoming values.
328     // TODO it would be nice to pick one or the other of these formats.
329     if (KeyValue.class.equals(job.getMapOutputValueClass())) {
330       job.setReducerClass(KeyValueSortReducer.class);
331     } else if (Put.class.equals(job.getMapOutputValueClass())) {
332       job.setReducerClass(PutSortReducer.class);
333     } else {
334       LOG.warn("Unknown map output value type:" + job.getMapOutputValueClass());
335     }
336 
337     LOG.info("Looking up current regions for table " + table);
338     List<ImmutableBytesWritable> startKeys = getRegionStartKeys(table);
339     LOG.info("Configuring " + startKeys.size() + " reduce partitions " +
340         "to match current region count");
341     job.setNumReduceTasks(startKeys.size());
342 
343     Path partitionsPath = new Path(job.getWorkingDirectory(),
344                                    "partitions_" + UUID.randomUUID());
345     LOG.info("Writing partition information to " + partitionsPath);
346 
347     FileSystem fs = partitionsPath.getFileSystem(conf);
348     writePartitions(conf, partitionsPath, startKeys);
349     partitionsPath.makeQualified(fs);
350 
351     URI cacheUri;
352     try {
353       cacheUri = new URI(partitionsPath.toString() + "#" + TotalOrderPartitioner.DEFAULT_PATH);
354     } catch (URISyntaxException e) {
355       throw new IOException(e);
356     }
357     DistributedCache.addCacheFile(cacheUri, conf);
358     DistributedCache.createSymlink(conf);
359 
360     // Set compression algorithms based on column families
361     configureCompression(table, conf);
362 
363     TableMapReduceUtil.addDependencyJars(job);
364     LOG.info("Incremental table output configured.");
365   }

367   /**
368    * Run inside the task to deserialize column family to compression algorithm
369    * map from the
370    * configuration.
371    *
372    * Package-private for unit tests only.
373    *
374    * @return a map from column family to the name of the configured compression
375    *         algorithm
376    */
377   static Map<byte[], String> createFamilyCompressionMap(Configuration conf) {
378     Map<byte[], String> compressionMap = new TreeMap<byte[], String>(Bytes.BYTES_COMPARATOR);
379     String compressionConf = conf.get(COMPRESSION_CONF_KEY, "");
380     for (String familyConf : compressionConf.split("&")) {
381       String[] familySplit = familyConf.split("=");
382       if (familySplit.length != 2) {
383         continue;
384       }
385 
386       try {
387         compressionMap.put(URLDecoder.decode(familySplit[0], "UTF-8").getBytes(),
388             URLDecoder.decode(familySplit[1], "UTF-8"));
389       } catch (UnsupportedEncodingException e) {
390         // will not happen with UTF-8 encoding
391         throw new AssertionError(e);
392       }
393     }
394     return compressionMap;
395   }

397   /**
398    * Serialize column family to compression algorithm map to configuration.
399    * Invoked while configuring the MR job for incremental load.
400    *
401    * Package-private for unit tests only.
402    *
403    * @throws IOException
404    *           on failure to read column family descriptors
405    */
406   static void configureCompression(HTable table, Configuration conf) throws IOException {
407     StringBuilder compressionConfigValue = new StringBuilder();
408     HTableDescriptor tableDescriptor = table.getTableDescriptor();
409     if(tableDescriptor == null){
410       // could happen with mock table instance
411       return;
412     }
413     Collection<HColumnDescriptor> families = tableDescriptor.getFamilies();
414     int i = 0;
415     for (HColumnDescriptor familyDescriptor : families) {
416       if (i++ > 0) {
417         compressionConfigValue.append('&');
418       }
419       compressionConfigValue.append(URLEncoder.encode(familyDescriptor.getNameAsString(), "UTF-8"));
420       compressionConfigValue.append('=');
421       compressionConfigValue.append(URLEncoder.encode(familyDescriptor.getCompression().getName(), "UTF-8"));
422     }
423     // Get rid of the last ampersand
424     conf.set(COMPRESSION_CONF_KEY, compressionConfigValue.toString());
425   }
