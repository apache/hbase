131   @Test
132   public void testReplayEditsAfterRegionMovedWithMultiCF() throws Exception {
133     final byte[] tableName = Bytes
134         .toBytes("testReplayEditsAfterRegionMovedWithMultiCF");
135     byte[] family1 = Bytes.toBytes("cf1");
136     byte[] family2 = Bytes.toBytes("cf2");
137     byte[] qualifier = Bytes.toBytes("q");
138     byte[] value = Bytes.toBytes("testV");
139     byte[][] familys = { family1, family2 };
140     TEST_UTIL.createTable(tableName, familys);
141     HTable htable = new HTable(TEST_UTIL.getConfiguration(), tableName);
142     Put put = new Put(Bytes.toBytes("r1"));
143     put.add(family1, qualifier, value);
144     htable.put(put);
145     ResultScanner resultScanner = htable.getScanner(new Scan());
146     int count = 0;
147     while (resultScanner.next() != null) {
148       count++;
149     }
150     resultScanner.close();
151     assertEquals(1, count);
152 
153     MiniHBaseCluster hbaseCluster = TEST_UTIL.getMiniHBaseCluster();
154     List<HRegion> regions = hbaseCluster.getRegions(tableName);
155     assertEquals(1, regions.size());
156 
157     // move region to another regionserver
158     HRegion destRegion = regions.get(0);
159     int originServerNum = hbaseCluster
160         .getServerWith(destRegion.getRegionName());
161     assertTrue("Please start more than 1 regionserver", hbaseCluster
162         .getRegionServerThreads().size() > 1);
163     int destServerNum = 0;
164     while (destServerNum == originServerNum) {
165       destServerNum++;
166     }
167     HRegionServer originServer = hbaseCluster.getRegionServer(originServerNum);
168     HRegionServer destServer = hbaseCluster.getRegionServer(destServerNum);
169     // move region to destination regionserver
170     moveRegionAndWait(destRegion, destServer);
171 
172     // delete the row
173     Delete del = new Delete(Bytes.toBytes("r1"));
174     htable.delete(del);
175     resultScanner = htable.getScanner(new Scan());
176     count = 0;
177     while (resultScanner.next() != null) {
178       count++;
179     }
180     resultScanner.close();
181     assertEquals(0, count);
182 
183     // flush region and make major compaction
184     destServer.getOnlineRegion(destRegion.getRegionName()).flushcache();
185     // wait to complete major compaction
186     for (Store store : destServer.getOnlineRegion(destRegion.getRegionName())
187         .getStores().values()) {
188       store.triggerMajorCompaction();
189     }
190     destServer.getOnlineRegion(destRegion.getRegionName()).compactStores();
191 
192     // move region to origin regionserver
193     moveRegionAndWait(destRegion, originServer);
194     // abort the origin regionserver
195     originServer.abort("testing");
196 
197     // see what we get
198     Result result = htable.get(new Get(Bytes.toBytes("r1")));
199     if (result != null) {
200       assertTrue("Row is deleted, but we get" + result.toString(),
201           (result == null) || result.isEmpty());
202     }
203     resultScanner.close();
204   }

206   private void moveRegionAndWait(HRegion destRegion, HRegionServer destServer)
207       throws InterruptedException, MasterNotRunningException,
208       ZooKeeperConnectionException, IOException {
209     HMaster master = TEST_UTIL.getMiniHBaseCluster().getMaster();
210     TEST_UTIL.getHBaseAdmin().move(
211         destRegion.getRegionInfo().getEncodedNameAsBytes(),
212         Bytes.toBytes(destServer.getServerName().getServerName()));
213     while (true) {
214       ServerName serverName = master.getAssignmentManager()
215         .getRegionStates().getRegionServerOfRegion(destRegion.getRegionInfo());
216       if (serverName != null && serverName.equals(destServer.getServerName())) {
217         TEST_UTIL.assertRegionOnServer(
218           destRegion.getRegionInfo(), serverName, 200);
219         break;
220       }
221       Thread.sleep(10);
222     }
223   }

341   /**
342    * Test writing edits into an HRegion, closing it, splitting logs, opening
343    * Region again.  Verify seqids.
344    * @throws IOException
345    * @throws IllegalAccessException
346    * @throws NoSuchFieldException
347    * @throws IllegalArgumentException
348    * @throws SecurityException
349    */
350   @Test
351   public void testReplayEditsWrittenViaHRegion()
352   throws IOException, SecurityException, IllegalArgumentException,
353       NoSuchFieldException, IllegalAccessException, InterruptedException {
354     final String tableNameStr = "testReplayEditsWrittenViaHRegion";
355     final HRegionInfo hri = createBasic3FamilyHRegionInfo(tableNameStr);
356     final Path basedir = new Path(this.hbaseRootDir, tableNameStr);
357     deleteDir(basedir);
358     final byte[] rowName = Bytes.toBytes(tableNameStr);
359     final int countPerFamily = 10;
360     final HTableDescriptor htd = createBasic3FamilyHTD(tableNameStr);
361     HRegion region3 = HRegion.createHRegion(hri,
362             hbaseRootDir, this.conf, htd);
363     HRegion.closeHRegion(region3);
364     // Write countPerFamily edits into the three families.  Do a flush on one
365     // of the families during the load of edits so its seqid is not same as
366     // others to test we do right thing when different seqids.
367     HLog wal = createWAL(this.conf);
368     HRegion region = new HRegion(basedir, wal, this.fs, this.conf, hri, htd, null);
369     long seqid = region.initialize();
370     // HRegionServer usually does this. It knows the largest seqid across all regions.
371     wal.setSequenceNumber(seqid);
372     boolean first = true;
373     for (HColumnDescriptor hcd: htd.getFamilies()) {
374       addRegionEdits(rowName, hcd.getName(), countPerFamily, this.ee, region, "x");
375       if (first ) {
376         // If first, so we have at least one family w/ different seqid to rest.
377         region.flushcache();
378         first = false;
379       }
380     }
381     // Now assert edits made it in.
382     final Get g = new Get(rowName);
383     Result result = region.get(g, null);
384     assertEquals(countPerFamily * htd.getFamilies().size(),
385       result.size());
386     // Now close the region (without flush), split the log, reopen the region and assert that
387     // replay of log has the correct effect, that our seqids are calculated correctly so
388     // all edits in logs are seen as 'stale'/old.
389     region.close(true);
390     wal.close();
391     runWALSplit(this.conf);
392     HLog wal2 = createWAL(this.conf);
393     HRegion region2 = new HRegion(basedir, wal2, this.fs, this.conf, hri, htd, null);
394     long seqid2 = region2.initialize();
395     // HRegionServer usually does this. It knows the largest seqid across all regions.
396     wal2.setSequenceNumber(seqid2);
397     assertTrue(seqid + result.size() < seqid2);
398     final Result result1b = region2.get(g, null);
399     assertEquals(result.size(), result1b.size());
400 
401     // Next test.  Add more edits, then 'crash' this region by stealing its wal
402     // out from under it and assert that replay of the log adds the edits back
403     // correctly when region is opened again.
404     for (HColumnDescriptor hcd: htd.getFamilies()) {
405       addRegionEdits(rowName, hcd.getName(), countPerFamily, this.ee, region2, "y");
406     }
407     // Get count of edits.
408     final Result result2 = region2.get(g, null);
409     assertEquals(2 * result.size(), result2.size());
410     wal2.sync();
411     // Set down maximum recovery so we dfsclient doesn't linger retrying something
412     // long gone.
413     HBaseTestingUtility.setMaxRecoveryErrorCount(((FSHLog) wal2).getOutputStream(), 1);
414     final Configuration newConf = HBaseConfiguration.create(this.conf);
415     User user = HBaseTestingUtility.getDifferentUser(newConf,
416       tableNameStr);
417     user.runAs(new PrivilegedExceptionAction() {
418       public Object run() throws Exception {
419         runWALSplit(newConf);
420         FileSystem newFS = FileSystem.get(newConf);
421         // Make a new wal for new region open.
422         HLog wal3 = createWAL(newConf);
423         final AtomicInteger countOfRestoredEdits = new AtomicInteger(0);
424         HRegion region3 = new HRegion(basedir, wal3, newFS, newConf, hri, htd, null) {
425           @Override
426           protected boolean restoreEdit(Store s, KeyValue kv) {
427             boolean b = super.restoreEdit(s, kv);
428             countOfRestoredEdits.incrementAndGet();
429             return b;
430           }
431         };
432         long seqid3 = region3.initialize();
433         // HRegionServer usually does this. It knows the largest seqid across all regions.
434         wal3.setSequenceNumber(seqid3);
435         Result result3 = region3.get(g, null);
436         // Assert that count of cells is same as before crash.
437         assertEquals(result2.size(), result3.size());
438         assertEquals(htd.getFamilies().size() * countPerFamily,
439           countOfRestoredEdits.get());
440 
441         // I can't close wal1.  Its been appropriated when we split.
442         region3.close();
443         wal3.closeAndDelete();
444         return null;
445       }
446     });
447   }

449   /**
450    * Test that we recover correctly when there is a failure in between the
451    * flushes. i.e. Some stores got flushed but others did not.
452    *
453    * Unfortunately, there is no easy hook to flush at a store level. The way
454    * we get around this is by flushing at the region level, and then deleting
455    * the recently flushed store file for one of the Stores. This would put us
456    * back in the situation where all but that store got flushed and the region
457    * died.
458    *
459    * We restart Region again, and verify that the edits were replayed.
460    *
461    * @throws IOException
462    * @throws IllegalAccessException
463    * @throws NoSuchFieldException
464    * @throws IllegalArgumentException
465    * @throws SecurityException
466    */
467   @Test
468   public void testReplayEditsAfterPartialFlush()
469   throws IOException, SecurityException, IllegalArgumentException,
470       NoSuchFieldException, IllegalAccessException, InterruptedException {
471     final String tableNameStr = "testReplayEditsWrittenViaHRegion";
472     final HRegionInfo hri = createBasic3FamilyHRegionInfo(tableNameStr);
473     final Path basedir = new Path(this.hbaseRootDir, tableNameStr);
474     deleteDir(basedir);
475     final byte[] rowName = Bytes.toBytes(tableNameStr);
476     final int countPerFamily = 10;
477     final HTableDescriptor htd = createBasic3FamilyHTD(tableNameStr);
478     HRegion region3 = HRegion.createHRegion(hri,
479             hbaseRootDir, this.conf, htd);
480     HRegion.closeHRegion(region3);
481     // Write countPerFamily edits into the three families.  Do a flush on one
482     // of the families during the load of edits so its seqid is not same as
483     // others to test we do right thing when different seqids.
484     HLog wal = createWAL(this.conf);
485     HRegion region = new HRegion(basedir, wal, this.fs, this.conf, hri, htd, null);
486     long seqid = region.initialize();
487     // HRegionServer usually does this. It knows the largest seqid across all regions.
488     wal.setSequenceNumber(seqid);
489     for (HColumnDescriptor hcd: htd.getFamilies()) {
490       addRegionEdits(rowName, hcd.getName(), countPerFamily, this.ee, region, "x");
491     }
492 
493     // Now assert edits made it in.
494     final Get g = new Get(rowName);
495     Result result = region.get(g, null);
496     assertEquals(countPerFamily * htd.getFamilies().size(),
497       result.size());
498 
499     // Let us flush the region
500     region.flushcache();
501     region.close(true);
502     wal.close();
503 
504     // delete the store files in the second column family to simulate a failure
505     // in between the flushcache();
506     // we have 3 families. killing the middle one ensures that taking the maximum
507     // will make us fail.
508     int cf_count = 0;
509     for (HColumnDescriptor hcd: htd.getFamilies()) {
510       cf_count++;
511       if (cf_count == 2) {
512         this.fs.delete(new Path(region.getRegionDir(), Bytes.toString(hcd.getName()))
513             , true);
514       }
515     }
516 
517 
518     // Let us try to split and recover
519     runWALSplit(this.conf);
520     HLog wal2 = createWAL(this.conf);
521     HRegion region2 = new HRegion(basedir, wal2, this.fs, this.conf, hri, htd, null);
522     long seqid2 = region2.initialize();
523     // HRegionServer usually does this. It knows the largest seqid across all regions.
524     wal2.setSequenceNumber(seqid2);
525     assertTrue(seqid + result.size() < seqid2);
526 
527     final Result result1b = region2.get(g, null);
528     assertEquals(result.size(), result1b.size());
529   }

531   /**
532    * Create an HRegion with the result of a HLog split and test we only see the
533    * good edits
534    * @throws Exception
535    */
536   @Test
537   public void testReplayEditsWrittenIntoWAL() throws Exception {
538     final String tableNameStr = "testReplayEditsWrittenIntoWAL";
539     final HRegionInfo hri = createBasic3FamilyHRegionInfo(tableNameStr);
540     final Path basedir = new Path(hbaseRootDir, tableNameStr);
541     deleteDir(basedir);
542     fs.mkdirs(new Path(basedir, hri.getEncodedName()));
543     final HTableDescriptor htd = createBasic3FamilyHTD(tableNameStr);
544     HRegion region2 = HRegion.createHRegion(hri,
545             hbaseRootDir, this.conf, htd);
546     HRegion.closeHRegion(region2);
547     final HLog wal = createWAL(this.conf);
548     final byte[] tableName = Bytes.toBytes(tableNameStr);
549     final byte[] rowName = tableName;
550     final byte[] regionName = hri.getEncodedNameAsBytes();
551 
552     // Add 1k to each family.
553     final int countPerFamily = 1000;
554     for (HColumnDescriptor hcd: htd.getFamilies()) {
555       addWALEdits(tableName, hri, rowName, hcd.getName(), countPerFamily,
556           ee, wal, htd);
557     }
558 
559     // Add a cache flush, shouldn't have any effect
560     long logSeqId = wal.startCacheFlush(regionName);
561     wal.completeCacheFlush(regionName, tableName, logSeqId, hri.isMetaRegion());
562 
563     // Add an edit to another family, should be skipped.
564     WALEdit edit = new WALEdit();
565     long now = ee.currentTimeMillis();
566     edit.add(new KeyValue(rowName, Bytes.toBytes("another family"), rowName,
567       now, rowName));
568     wal.append(hri, tableName, edit, now, htd);
569 
570     // Delete the c family to verify deletes make it over.
571     edit = new WALEdit();
572     now = ee.currentTimeMillis();
573     edit.add(new KeyValue(rowName, Bytes.toBytes("c"), null, now,
574       KeyValue.Type.DeleteFamily));
575     wal.append(hri, tableName, edit, now, htd);
576 
577     // Sync.
578     wal.sync();
579     // Set down maximum recovery so we dfsclient doesn't linger retrying something
580     // long gone.
581     HBaseTestingUtility.setMaxRecoveryErrorCount(((FSHLog) wal).getOutputStream(), 1);
582     // Make a new conf and a new fs for the splitter to run on so we can take

728   private void addWALEdits (final byte [] tableName, final HRegionInfo hri,
729       final byte [] rowName, final byte [] family,
730       final int count, EnvironmentEdge ee, final HLog wal, final HTableDescriptor htd)
731   throws IOException {
732     String familyStr = Bytes.toString(family);
733     for (int j = 0; j < count; j++) {
734       byte[] qualifierBytes = Bytes.toBytes(Integer.toString(j));
735       byte[] columnBytes = Bytes.toBytes(familyStr + ":" + Integer.toString(j));
736       WALEdit edit = new WALEdit();
737       edit.add(new KeyValue(rowName, family, qualifierBytes,
738         ee.currentTimeMillis(), columnBytes));
739       wal.append(hri, tableName, edit, ee.currentTimeMillis(), htd);
740     }
741   }

751       r.put(p);
752     }
