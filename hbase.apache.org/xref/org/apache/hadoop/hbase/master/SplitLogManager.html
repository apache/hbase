69  /**
70   * Distributes the task of log splitting to the available region servers.
71   * Coordination happens via zookeeper. For every log file that has to be split a
72   * znode is created under <code>/hbase/splitlog</code>. SplitLogWorkers race to grab a task.
73   *
74   * <p>SplitLogManager monitors the task znodes that it creates using the
75   * timeoutMonitor thread. If a task's progress is slow then
76   * {@link #resubmit(String, Task, ResubmitDirective)} will take away the task from the owner
77   * {@link SplitLogWorker} and the task will be up for grabs again. When the task is done then the
78   * task's znode is deleted by SplitLogManager.
79   *
80   * <p>Clients call {@link #splitLogDistributed(Path)} to split a region server's
81   * log files. The caller thread waits in this method until all the log files
82   * have been split.
83   *
84   * <p>All the zookeeper calls made by this class are asynchronous. This is mainly
85   * to help reduce response time seen by the callers.
86   *
87   * <p>There is race in this design between the SplitLogManager and the
88   * SplitLogWorker. SplitLogManager might re-queue a task that has in reality
89   * already been completed by a SplitLogWorker. We rely on the idempotency of
90   * the log splitting task for correctness.
91   *
92   * <p>It is also assumed that every log splitting task is unique and once
93   * completed (either with success or with error) it will be not be submitted
94   * again. If a task is resubmitted then there is a risk that old "delete task"
95   * can delete the re-submission.
96   */
97  @InterfaceAudience.Private

126   /**
127    * Wrapper around {@link #SplitLogManager(ZooKeeperWatcher zkw, Configuration conf,
128    *   Stoppable stopper, MasterServices master, ServerName serverName, TaskFinisher tf)}
129    * that provides a task finisher for copying recovered edits to their final destination.
130    * The task finisher has to be robust because it can be arbitrarily restarted or called
131    * multiple times.
132    * 
133    * @param zkw
134    * @param conf
135    * @param stopper
136    * @param serverName
137    */
138   public SplitLogManager(ZooKeeperWatcher zkw, final Configuration conf,
139        Stoppable stopper, MasterServices master, ServerName serverName) {
140     this(zkw, conf, stopper,  master, serverName, new TaskFinisher() {
141       @Override
142       public Status finish(ServerName workerName, String logfile) {
143         try {
144           HLogSplitter.finishSplitLogFile(logfile, conf);
145         } catch (IOException e) {
146           LOG.warn("Could not finish splitting of log file " + logfile, e);
147           return Status.ERR;
148         }
149         return Status.DONE;
150       }
151     });
152   }

197   private FileStatus[] getFileList(List<Path> logDirs) throws IOException {
198     List<FileStatus> fileStatus = new ArrayList<FileStatus>();
199     for (Path hLogDir : logDirs) {
200       this.fs = hLogDir.getFileSystem(conf);
201       if (!fs.exists(hLogDir)) {
202         LOG.warn(hLogDir + " doesn't exist. Nothing to do!");
203         continue;
204       }
205       // TODO filter filenames?
206       FileStatus[] logfiles = FSUtils.listStatus(fs, hLogDir, null);
207       if (logfiles == null || logfiles.length == 0) {
208         LOG.info(hLogDir + " is empty dir, no logs to split");
209       } else {
210         for (FileStatus status : logfiles)
211           fileStatus.add(status);
212       }
213     }
214     FileStatus[] a = new FileStatus[fileStatus.size()];
215     return fileStatus.toArray(a);
216   }

231   /**
232    * The caller will block until all the log files of the given region server
233    * have been processed - successfully split or an error is encountered - by an
234    * available worker region server. This method must only be called after the
235    * region servers have been brought online.
236    *
237    * @param logDirs List of log dirs to split
238    * @throws IOException If there was an error while splitting any log file
239    * @return cumulative size of the logfiles split
240    */
241   public long splitLogDistributed(final List<Path> logDirs) throws IOException {
242     MonitoredTask status = TaskMonitor.get().createStatus(
243           "Doing distributed log split in " + logDirs);
244     FileStatus[] logfiles = getFileList(logDirs);
245     status.setStatus("Checking directory contents...");
246     LOG.debug("Scheduling batch of logs to split");
247     SplitLogCounters.tot_mgr_log_split_batch_start.incrementAndGet();
248     LOG.info("started splitting logs in " + logDirs);
249     long t = EnvironmentEdgeManager.currentTimeMillis();
250     long totalSize = 0;
251     TaskBatch batch = new TaskBatch();
252     for (FileStatus lf : logfiles) {
253       // TODO If the log file is still being written to - which is most likely
254       // the case for the last log file - then its length will show up here
255       // as zero. The size of such a file can only be retrieved after
256       // recover-lease is done. totalSize will be under in most cases and the
257       // metrics that it drives will also be under-reported.
258       totalSize += lf.getLen();
259       if (enqueueSplitTask(lf.getPath().toString(), batch) == false) {
260         throw new IOException("duplicate log split scheduled for " + lf.getPath());
261       }
262     }
263     waitForSplittingCompletion(batch, status);
264     if (batch.done != batch.installed) {
265       batch.isDead = true;
266       SplitLogCounters.tot_mgr_log_split_batch_err.incrementAndGet();
267       LOG.warn("error while splitting logs in " + logDirs +
268       " installed = " + batch.installed + " but only " + batch.done + " done");
269       String msg = "error or interrupted while splitting logs in "
270         + logDirs + " Task = " + batch;
271       status.abort(msg);
272       throw new IOException(msg);
273     }
274     for(Path logDir: logDirs){
275       status.setStatus("Cleaning up log directory...");
276       try {
277         if (fs.exists(logDir) && !fs.delete(logDir, false)) {
278           LOG.warn("Unable to delete log src dir. Ignoring. " + logDir);
279         }
280       } catch (IOException ioe) {
281         FileStatus[] files = fs.listStatus(logDir);
282         if (files != null && files.length > 0) {
283           LOG.warn("returning success without actually splitting and " + 
284               "deleting all the log files in path " + logDir);
285         } else {
286           LOG.warn("Unable to delete log src dir. Ignoring. " + logDir, ioe);
287         }
288       }
289       SplitLogCounters.tot_mgr_log_split_batch_success.incrementAndGet();
290     }
291     String msg = "finished splitting (more than or equal to) " + totalSize +
292         " bytes in " + batch.installed + " log files in " + logDirs + " in " +
293         (EnvironmentEdgeManager.currentTimeMillis() - t) + "ms";
294     status.markComplete(msg);
295     LOG.info(msg);
296     return totalSize;
297   }

299   /**
300    * Add a task entry to splitlog znode if it is not already there.
301    * 
302    * @param taskname the path of the log to be split
303    * @param batch the batch this task belongs to
304    * @return true if a new entry is created, false if it is already there.
305    */
306   boolean enqueueSplitTask(String taskname, TaskBatch batch) {
307     SplitLogCounters.tot_mgr_log_split_start.incrementAndGet();
308     // This is a znode path under the splitlog dir with the rest of the path made up of an
309     // url encoding of the passed in log to split.
310     String path = ZKSplitLog.getEncodedNodeName(watcher, taskname);
311     Task oldtask = createTaskIfAbsent(path, batch);
312     if (oldtask == null) {
313       // publish the task in zk
314       createNode(path, zkretries);
315       return true;
316     }
317     return false;
318   }

320   private void waitForSplittingCompletion(TaskBatch batch, MonitoredTask status) {
321     synchronized (batch) {
322       while ((batch.done + batch.error) != batch.installed) {
323         try {
324           status.setStatus("Waiting for distributed tasks to finish. "
325               + " scheduled=" + batch.installed
326               + " done=" + batch.done
327               + " error=" + batch.error);
328           int remaining = batch.installed - (batch.done + batch.error);
329           int actual = activeTasks(batch);
330           if (remaining != actual) {
331             LOG.warn("Expected " + remaining
332               + " active tasks, but actually there are " + actual);
333           }
334           int remainingInZK = remainingTasksInZK();
335           if (remainingInZK >= 0 && actual > remainingInZK) {
336             LOG.warn("Expected at least" + actual
337               + " tasks in ZK, but actually there are " + remainingInZK);
338           }
339           if (remainingInZK == 0 || actual == 0) {
340             LOG.warn("No more task remaining (ZK or task map), splitting "
341               + "should have completed. Remaining tasks in ZK " + remainingInZK
342               + ", active tasks in map " + actual);
343             if (remainingInZK == 0 && actual == 0) {
344               return;
345             }
346           }
347           batch.wait(100);
348           if (stopper.isStopped()) {
349             LOG.warn("Stopped while waiting for log splits to be completed");
350             return;
351           }
352         } catch (InterruptedException e) {
353           LOG.warn("Interrupted while waiting for log splits to be completed");
354           Thread.currentThread().interrupt();
355           return;
356         }
357       }
358     }
359   }

361   private int activeTasks(final TaskBatch batch) {
362     int count = 0;
363     for (Task t: tasks.values()) {
364       if (t.batch == batch && t.status == TerminationStatus.IN_PROGRESS) {
365         count++;
366       }
367     }
368     return count;
369   }

371   private int remainingTasksInZK() {
372     int count = 0;
373     try {
374       List<String> tasks =
375         ZKUtil.listChildrenNoWatch(watcher, watcher.splitLogZNode);
376       if (tasks != null) {
377         for (String t: tasks) {
378           if (!ZKSplitLog.isRescanNode(watcher, t)) {
379             count++;
380           }
381         }
382       }
383     } catch (KeeperException ke) {
384       LOG.warn("Failed to check remaining tasks", ke);
385       count = -1;
386     }
387     return count;
388   }

390   private void setDone(String path, TerminationStatus status) {
391     Task task = tasks.get(path);
392     if (task == null) {
393       if (!ZKSplitLog.isRescanNode(watcher, path)) {
394         SplitLogCounters.tot_mgr_unacquired_orphan_done.incrementAndGet();
395         LOG.debug("unacquired orphan task is done " + path);
396       }
397     } else {
398       synchronized (task) {
399         if (task.status == IN_PROGRESS) {
400           if (status == SUCCESS) {
401             SplitLogCounters.tot_mgr_log_split_success.incrementAndGet();
402             LOG.info("Done splitting " + path);
403           } else {
404             SplitLogCounters.tot_mgr_log_split_err.incrementAndGet();
405             LOG.warn("Error splitting " + path);
406           }
407           task.status = status;
408           if (task.batch != null) {
409             synchronized (task.batch) {
410               if (status == SUCCESS) {
411                 task.batch.done++;
412               } else {
413                 task.batch.error++;
414               }
415               task.batch.notify();
416             }
417           }
418         }
419       }
420     }
421     // delete the task node in zk. Keep trying indefinitely - its an async
422     // call and no one is blocked waiting for this node to be deleted. All
423     // task names are unique (log.<timestamp>) there is no risk of deleting
424     // a future task.
425     deleteNode(path, Long.MAX_VALUE);
426     return;
427   }

464   private void getDataSetWatchSuccess(String path, byte[] data, int version)
465   throws DeserializationException {
466     if (data == null) {
467       if (version == Integer.MIN_VALUE) {
468         // assume all done. The task znode suddenly disappeared.
469         setDone(path, SUCCESS);
470         return;
471       }
472       SplitLogCounters.tot_mgr_null_data.incrementAndGet();
473       LOG.fatal("logic error - got null data " + path);
474       setDone(path, FAILURE);
475       return;
476     }
477     data = this.watcher.getRecoverableZooKeeper().removeMetaData(data);
478     SplitLogTask slt = SplitLogTask.parseFrom(data);
479     if (slt.isUnassigned()) {
480       LOG.debug("task not yet acquired " + path + " ver = " + version);
481       handleUnassignedTask(path);
482     } else if (slt.isOwned()) {
483       heartbeat(path, version, slt.getServerName());
484     } else if (slt.isResigned()) {
485       LOG.info("task " + path + " entered state: " + slt.toString());
486       resubmitOrFail(path, FORCE);
487     } else if (slt.isDone()) {
488       LOG.info("task " + path + " entered state: " + slt.toString());
489       if (taskFinisher != null && !ZKSplitLog.isRescanNode(watcher, path)) {
490         if (taskFinisher.finish(slt.getServerName(), ZKSplitLog.getFileName(path)) == Status.DONE) {
491           setDone(path, SUCCESS);
492         } else {
493           resubmitOrFail(path, CHECK);
494         }
495       } else {
496         setDone(path, SUCCESS);
497       }
498     } else if (slt.isErr()) {
499       LOG.info("task " + path + " entered state: " + slt.toString());
500       resubmitOrFail(path, CHECK);
501     } else {
502       LOG.fatal("logic error - unexpected zk state for path = " + path + " data = " + slt.toString());
503       setDone(path, FAILURE);
504     }
505   }

512   /**
513    * It is possible for a task to stay in UNASSIGNED state indefinitely - say
514    * SplitLogManager wants to resubmit a task. It forces the task to UNASSIGNED
515    * state but it dies before it could create the RESCAN task node to signal
516    * the SplitLogWorkers to pick up the task. To prevent this scenario the
517    * SplitLogManager resubmits all orphan and UNASSIGNED tasks at startup.
518    *
519    * @param path
520    */
521   private void handleUnassignedTask(String path) {
522     if (ZKSplitLog.isRescanNode(watcher, path)) {
523       return;
524     }
525     Task task = findOrCreateOrphanTask(path);
526     if (task.isOrphan() && (task.incarnation == 0)) {
527       LOG.info("resubmitting unassigned orphan task " + path);
528       // ignore failure to resubmit. The timeout-monitor will handle it later
529       // albeit in a more crude fashion
530       resubmit(path, task, FORCE);
531     }
532   }

534   private void heartbeat(String path, int new_version, ServerName workerName) {
535     Task task = findOrCreateOrphanTask(path);
536     if (new_version != task.last_version) {
537       if (task.isUnassigned()) {
538         LOG.info("task " + path + " acquired by " + workerName);
539       }
540       task.heartbeat(EnvironmentEdgeManager.currentTimeMillis(), new_version, workerName);
541       SplitLogCounters.tot_mgr_heartbeat.incrementAndGet();
542     } else {
543       // duplicate heartbeats - heartbeats w/o zk node version
544       // changing - are possible. The timeout thread does
545       // getDataSetWatch() just to check whether a node still
546       // exists or not
547     }
548     return;
549   }

551   private boolean resubmit(String path, Task task, ResubmitDirective directive) {
552     // its ok if this thread misses the update to task.deleted. It will fail later
553     if (task.status != IN_PROGRESS) {
554       return false;
555     }
556     int version;
557     if (directive != FORCE) {
558       // We're going to resubmit:
559       //  1) immediately if the worker server is now marked as dead
560       //  2) after a configurable timeout if the server is not marked as dead but has still not
561       //       finished the task. This allows to continue if the worker cannot actually handle it,
562       //       for any reason.
563       final long time = EnvironmentEdgeManager.currentTimeMillis() - task.last_update;
564       final boolean alive = master.getServerManager() != null ?
565           master.getServerManager().isServerOnline(task.cur_worker_name) : true;
566       if (alive && time < timeout) {
567         LOG.trace("Skipping the resubmit of " + task.toString() + "  because the server " +
568             task.cur_worker_name + " is not marked as dead, we waited for " + time +
569             " while the timeout is " + timeout);
570         return false;
571       }
572       if (task.unforcedResubmits >= resubmit_threshold) {
573         if (!task.resubmitThresholdReached) {
574           task.resubmitThresholdReached = true;
575           SplitLogCounters.tot_mgr_resubmit_threshold_reached.incrementAndGet();
576           LOG.info("Skipping resubmissions of task " + path +
577               " because threshold " + resubmit_threshold + " reached");
578         }
579         return false;
580       }
581       // race with heartbeat() that might be changing last_version
582       version = task.last_version;
583     } else {
584       version = -1;
585     }
586     LOG.info("resubmitting task " + path);
587     task.incarnation++;
588     try {
589       // blocking zk call but this is done from the timeout thread
590       SplitLogTask slt = new SplitLogTask.Unassigned(this.serverName);
591       if (ZKUtil.setData(this.watcher, path, slt.toByteArray(), version) == false) {
592         LOG.debug("failed to resubmit task " + path +
593             " version changed");
594         task.heartbeatNoDetails(EnvironmentEdgeManager.currentTimeMillis());
595         return false;
596       }
597     } catch (NoNodeException e) {
598       LOG.warn("failed to resubmit because znode doesn't exist " + path +
599           " task done (or forced done by removing the znode)");
600       try {
601         getDataSetWatchSuccess(path, null, Integer.MIN_VALUE);
602       } catch (DeserializationException e1) {
603         LOG.debug("Failed to re-resubmit task " + path + " because of deserialization issue", e1);
604         task.heartbeatNoDetails(EnvironmentEdgeManager.currentTimeMillis());
605         return false;
606       }
607       return false;
608     } catch (KeeperException.BadVersionException e) {
609       LOG.debug("failed to resubmit task " + path + " version changed");
610       task.heartbeatNoDetails(EnvironmentEdgeManager.currentTimeMillis());
611       return false;
612     } catch (KeeperException e) {
613       SplitLogCounters.tot_mgr_resubmit_failed.incrementAndGet();
614       LOG.warn("failed to resubmit " + path, e);
615       return false;
616     }
617     // don't count forced resubmits
618     if (directive != FORCE) {
619       task.unforcedResubmits++;
620     }
621     task.setUnassigned();
622     createRescanNode(Long.MAX_VALUE);
623     SplitLogCounters.tot_mgr_resubmit.incrementAndGet();
624     return true;
625   }

700   /**
701    * @param path
702    * @param batch
703    * @return null on success, existing task on error
704    */
705   private Task createTaskIfAbsent(String path, TaskBatch batch) {
706     Task oldtask;
707     // batch.installed is only changed via this function and
708     // a single thread touches batch.installed.
709     Task newtask = new Task();
710     newtask.batch = batch;
711     oldtask = tasks.putIfAbsent(path, newtask);
712     if (oldtask == null) {
713       batch.installed++;
714       return  null;
715     }
716     // new task was not used.
717     synchronized (oldtask) {
718       if (oldtask.isOrphan()) {
719         if (oldtask.status == SUCCESS) {
720           // The task is already done. Do not install the batch for this
721           // task because it might be too late for setDone() to update
722           // batch.done. There is no need for the batch creator to wait for
723           // this task to complete.
724           return (null);
725         }
726         if (oldtask.status == IN_PROGRESS) {
727           oldtask.batch = batch;
728           batch.installed++;
729           LOG.debug("Previously orphan task " + path + " is now being waited upon");
730           return null;
731         }
732         while (oldtask.status == FAILURE) {
733           LOG.debug("wait for status of task " + path + " to change to DELETED");
734           SplitLogCounters.tot_mgr_wait_for_zk_delete.incrementAndGet();
735           try {
736             oldtask.wait();
737           } catch (InterruptedException e) {
738             Thread.currentThread().interrupt();
739             LOG.warn("Interrupted when waiting for znode delete callback");
740             // fall through to return failure
741             break;
742           }
743         }
744         if (oldtask.status != DELETED) {
745           LOG.warn("Failure because previously failed task" +
746               " state still present. Waiting for znode delete callback" +
747               " path=" + path);
748           return oldtask;
749         }
750         // reinsert the newTask and it must succeed this time
751         Task t = tasks.putIfAbsent(path, newtask);
752         if (t == null) {
753           batch.installed++;
754           return  null;
755         }
756         LOG.fatal("Logic error. Deleted task still present in tasks map");
757         assert false : "Deleted task still present in tasks map";
758         return t;
759       }
760       LOG.warn("Failure because two threads can't wait for the same task; path=" + path);
761       return oldtask;
762     }

924   /**
925    * Periodically checks all active tasks and resubmits the ones that have timed
926    * out
927    */
928   private class TimeoutMonitor extends Chore {
929     public TimeoutMonitor(final int period, Stoppable stopper) {
930       super("SplitLogManager Timeout Monitor", period, stopper);
931     }
932 
933     @Override
934     protected void chore() {
935       int resubmitted = 0;
936       int unassigned = 0;
937       int tot = 0;
938       boolean found_assigned_task = false;
939       Set<ServerName> localDeadWorkers;
940 
941       synchronized (deadWorkersLock) {
942         localDeadWorkers = deadWorkers;
943         deadWorkers = null;
944       }
945 
946       for (Map.Entry<String, Task> e : tasks.entrySet()) {
947         String path = e.getKey();
948         Task task = e.getValue();
949         ServerName cur_worker = task.cur_worker_name;
950         tot++;
951         // don't easily resubmit a task which hasn't been picked up yet. It
952         // might be a long while before a SplitLogWorker is free to pick up a
953         // task. This is because a SplitLogWorker picks up a task one at a
954         // time. If we want progress when there are no region servers then we
955         // will have to run a SplitLogWorker thread in the Master.
956         if (task.isUnassigned()) {
957           unassigned++;
958           continue;
959         }
960         found_assigned_task = true;
961         if (localDeadWorkers != null && localDeadWorkers.contains(cur_worker)) {
962           SplitLogCounters.tot_mgr_resubmit_dead_server_task.incrementAndGet();
963           if (resubmit(path, task, FORCE)) {
964             resubmitted++;
965           } else {
966             handleDeadWorker(cur_worker);
967             LOG.warn("Failed to resubmit task " + path + " owned by dead " +
968                 cur_worker + ", will retry.");
969           }
970         } else if (resubmit(path, task, CHECK)) {
971           resubmitted++;
972         }
973       }
974       if (tot > 0) {
975         LOG.debug("total tasks = " + tot + " unassigned = " + unassigned);
976       }
977       if (resubmitted > 0) {
978         LOG.info("resubmitted " + resubmitted + " out of " + tot + " tasks");
979       }
980       // If there are pending tasks and all of them have been unassigned for
981       // some time then put up a RESCAN node to ping the workers.
982       // ZKSplitlog.DEFAULT_UNASSIGNED_TIMEOUT is of the order of minutes
983       // because a. it is very unlikely that every worker had a
984       // transient error when trying to grab the task b. if there are no
985       // workers then all tasks wills stay unassigned indefinitely and the
986       // manager will be indefinitely creating RESCAN nodes. TODO may be the
987       // master should spawn both a manager and a worker thread to guarantee
988       // that there is always one worker in the system
989       if (tot > 0 && !found_assigned_task &&
990           ((EnvironmentEdgeManager.currentTimeMillis() - lastNodeCreateTime) >
991           unassignedTimeout)) {
992         for (Map.Entry<String, Task> e : tasks.entrySet()) {
993           String path = e.getKey();
994           Task task = e.getValue();
995           // we have to do task.isUnassigned() check again because tasks might
996           // have been asynchronously assigned. There is no locking required
997           // for these checks ... it is OK even if tryGetDataSetWatch() is
998           // called unnecessarily for a task
999           if (task.isUnassigned() && (task.status != FAILURE)) {
1000             // We just touch the znode to make sure its still there
1001             tryGetDataSetWatch(path);
1002           }
1003         }
1004         createRescanNode(Long.MAX_VALUE);
1005         SplitLogCounters.tot_mgr_resubmit_unassigned.incrementAndGet();
1006         LOG.debug("resubmitting unassigned task(s) after timeout");
1007       }
1008     }
1009   }
