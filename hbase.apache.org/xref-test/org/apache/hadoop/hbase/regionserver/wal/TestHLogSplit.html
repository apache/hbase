179    * Simulates splitting a WAL out from under a regionserver that is still trying to write it.  Ensures we do not
180    * lose edits.
181    * @throws IOException

184   @Test
185   public void testLogCannotBeWrittenOnceParsed() throws IOException, InterruptedException {
186     final AtomicLong counter = new AtomicLong(0);
187     AtomicBoolean stop = new AtomicBoolean(false);
188     // Region we'll write edits too and then later examine to make sure they all made it in.
189     final String region = REGIONS.get(0);
190     Thread zombie = new ZombieLastLogWriterRegionServer(this.conf, counter, stop, region);
191     try {
192       long startCount = counter.get();
193       zombie.start();
194       // Wait till writer starts going.
195       while (startCount == counter.get()) Threads.sleep(1);
196       // Give it a second to write a few appends.
197       Threads.sleep(1000);
198       final Configuration conf2 = HBaseConfiguration.create(this.conf);
199       final User robber = User.createUserForTesting(conf2, ROBBER, GROUP);
200       int count = robber.runAs(new PrivilegedExceptionAction<Integer>() {
201         @Override
202         public Integer run() throws Exception {
203           FileSystem fs = FileSystem.get(conf2);
204           HLogSplitter logSplitter = HLogSplitter.createLogSplitter(conf2, HBASEDIR, HLOGDIR, OLDLOGDIR, fs);
205           logSplitter.splitLog();
206           Path logfile = getLogForRegion(HBASEDIR, TABLE_NAME, region);
207           return countHLog(logfile, fs, conf2);
208         }
209       });
210       LOG.info("zombie=" + counter.get() + ", robber=" + count);
211       assertTrue("The log file could have at most 1 extra log entry, but can't have less. Zombie could write " +
212         counter.get() + " and logfile had only " + count,
213         counter.get() == count || counter.get() + 1 == count);
214     } finally {
215       stop.set(true);
216       zombie.interrupt();
217       Threads.threadDumpingIsAlive(zombie);
218     }
219   }

221   /**
222    * This thread will keep writing to a 'wal' file even after the split process has started.
223    * It simulates a region server that was considered dead but woke up and wrote some more to he last log entry.
224    * Does its writing as an alternate user in another filesystem instance to simulate better it being a regionserver.
225    */
226   static class ZombieLastLogWriterRegionServer extends Thread {
227     final AtomicLong editsCount;
228     final AtomicBoolean stop;
229     // final User user;
230     /**
231      * Region to write edits for.
232      */
233     final String region;
234     final Configuration conf;
235     final User user;
236 
237     public ZombieLastLogWriterRegionServer(final Configuration conf, AtomicLong counter, AtomicBoolean stop,
238         final String region)
239     throws IOException, InterruptedException {
240       super("ZombieLastLogWriterRegionServer");
241       setDaemon(true);
242       this.stop = stop;
243       this.editsCount = counter;
244       this.region = region;
245       this.conf = HBaseConfiguration.create(conf);
246       this.user = User.createUserForTesting(this.conf, ZOMBIE, GROUP);
247     }
248 
249     @Override
250     public void run() {
251       try {
252         doWriting();
253       } catch (IOException e) {
254         flushToConsole(getName() + " Writer exiting " + e);
255       } catch (InterruptedException e) {
256         flushToConsole(getName() + " Writer exiting " + e);
257       }
258     }
259 
260     private void doWriting() throws IOException, InterruptedException {
261       this.user.runAs(new PrivilegedExceptionAction<Object>() {
262         @Override
263         public Object run() throws Exception {
264           // Index of the WAL we want to keep open.  generateHLogs will leave open the WAL whose index we supply here.
265           int walToKeepOpen = 2;
266           // How many files to write.
267           final int numOfWriters = walToKeepOpen + 1;
268           // The below method writes numOfWriters files each with ENTRIES entries for a total of numOfWriters * ENTRIES
269           // added per column family in the region.
270           HLog.Writer[] writers = null;
271           try {
272             DistributedFileSystem dfs = (DistributedFileSystem)FileSystem.get(conf);
273             writers = generateHLogs(dfs, numOfWriters, ENTRIES, walToKeepOpen);
274           } catch (IOException e1) {
275             throw new RuntimeException("Failed", e1);
276           }
277           // Update counter so has all edits written so far.
278           editsCount.addAndGet(numOfWriters * NUM_WRITERS);
279           // This WAL should be open still after our call to generateHLogs -- we asked it leave it open.
280           HLog.Writer writer = writers[walToKeepOpen];
281           loop(writer);
282           return null;
283         }
284       });
285     }
286 
287     private void loop(final HLog.Writer writer) {
288       byte [] regionBytes = Bytes.toBytes(this.region);
289       while (true) {
290         try {
291           long seq = appendEntry(writer, TABLE_NAME, regionBytes, ("r" + editsCount.get()).getBytes(),
292             regionBytes, QUALIFIER, VALUE, 0);
293           long count = editsCount.incrementAndGet();
294           flushToConsole(getName() + " sync count=" + count + ", seq=" + seq);
295           try {
296             Thread.sleep(1);
297           } catch (InterruptedException e) {
298             //
299           }
300         } catch (IOException ex) {
301           flushToConsole(getName() + " ex " + ex.toString());
302           if (ex instanceof RemoteException) {
303             flushToConsole("Juliet: got RemoteException " + ex.getMessage() +
304               " while writing " + (editsCount.get() + 1));
305           } else {
306             flushToConsole(getName() + " failed to write....at " + editsCount.get());
307             assertTrue("Failed to write " + editsCount.get(), false);
308           }
309           break;
310         } catch (Throwable t) {
311           flushToConsole(getName() + " HOW? " + t);
312           t.printStackTrace();
313           break;
314         }
315       }
316       flushToConsole(getName() + " Writer exiting");
317     }
318   }

888   /**
889    * Sets up a log splitter with a mock reader and writer. The mock reader generates
890    * a specified number of edits spread across 5 regions. The mock writer optionally
891    * sleeps for each edit it is fed.
892    * *
893    * After the split is complete, verifies that the statistics show the correct number
894    * of edits output into each region.
895    *
896    * @param numFakeEdits number of fake edits to push through pipeline
897    * @param bufferSize size of in-memory buffer
898    * @param writerSlowness writer threads will sleep this many ms per edit
899    */
900   private void doTestThreading(final int numFakeEdits,
901       final int bufferSize,
902       final int writerSlowness) throws Exception {
903 
904     Configuration localConf = new Configuration(conf);
905     localConf.setInt("hbase.regionserver.hlog.splitlog.buffersize", bufferSize);
906 
907     // Create a fake log file (we'll override the reader to produce a stream of edits)
908     FSDataOutputStream out = fs.create(new Path(HLOGDIR, HLOG_FILE_PREFIX + ".fake"));
909     out.close();
910 
911     // Make region dirs for our destination regions so the output doesn't get skipped
912     final List<String> regions = ImmutableList.of("r0", "r1", "r2", "r3", "r4");
913     makeRegionDirs(fs, regions);
914 
915     // Create a splitter that reads and writes the data without touching disk
916     HLogSplitter logSplitter = new HLogSplitter(
917         localConf, HBASEDIR, HLOGDIR, OLDLOGDIR, fs, null) {
918 
919       /* Produce a mock writer that doesn't write anywhere */
920       protected HLog.Writer createWriter(FileSystem fs, Path logfile, Configuration conf)
921       throws IOException {
922         HLog.Writer mockWriter = Mockito.mock(HLog.Writer.class);
923         Mockito.doAnswer(new Answer<Void>() {
924           int expectedIndex = 0;
925 
926           @Override
927           public Void answer(InvocationOnMock invocation) {
928             if (writerSlowness > 0) {
929               try {
930                 Thread.sleep(writerSlowness);
931               } catch (InterruptedException ie) {
932                 Thread.currentThread().interrupt();
933               }
934             }
935             HLog.Entry entry = (Entry) invocation.getArguments()[0];
936             WALEdit edit = entry.getEdit();
937             List<KeyValue> keyValues = edit.getKeyValues();
938             assertEquals(1, keyValues.size());
939             KeyValue kv = keyValues.get(0);
940 
941             // Check that the edits come in the right order.
942             assertEquals(expectedIndex, Bytes.toInt(kv.getRow()));
943             expectedIndex++;
944             return null;
945           }
946         }).when(mockWriter).append(Mockito.<HLog.Entry>any());
947         return mockWriter;
948       }
949 
950 
951       /* Produce a mock reader that generates fake entries */
952       protected Reader getReader(FileSystem fs, Path curLogFile, Configuration conf)
953       throws IOException {
954         Reader mockReader = Mockito.mock(Reader.class);
955         Mockito.doAnswer(new Answer<HLog.Entry>() {
956           int index = 0;
957 
958           @Override
959           public HLog.Entry answer(InvocationOnMock invocation) throws Throwable {
960             if (index >= numFakeEdits) return null;
961 
962             // Generate r0 through r4 in round robin fashion
963             int regionIdx = index % regions.size();
964             byte region[] = new byte[] {(byte)'r', (byte) (0x30 + regionIdx)};
965 
966             HLog.Entry ret = createTestEntry(TABLE_NAME, region,
967                 Bytes.toBytes((int)(index / regions.size())),
968                 FAMILY, QUALIFIER, VALUE, index);
969             index++;
970             return ret;
971           }
972         }).when(mockReader).next();
973         return mockReader;
974       }
975     };
976 
977     logSplitter.splitLog();
978 
979     // Verify number of written edits per region
980 
981     Map<byte[], Long> outputCounts = logSplitter.getOutputCounts();
982     for (Map.Entry<byte[], Long> entry : outputCounts.entrySet()) {
983       LOG.info("Got " + entry.getValue() + " output edits for region " +
984           Bytes.toString(entry.getKey()));
985 
986       assertEquals((long)entry.getValue(), numFakeEdits / regions.size());
987     }
988     assertEquals(regions.size(), outputCounts.size());
989   }

1323   private void corruptHLog(Path path, Corruptions corruption, boolean close,
1324                            FileSystem fs) throws IOException {
1325 
1326     FSDataOutputStream out;
1327     int fileSize = (int) fs.listStatus(path)[0].getLen();
1328 
1329     FSDataInputStream in = fs.open(path);
1330     byte[] corrupted_bytes = new byte[fileSize];
1331     in.readFully(0, corrupted_bytes, 0, fileSize);
1332     in.close();
1333 
1334     switch (corruption) {
1335       case APPEND_GARBAGE:
1336         fs.delete(path, false);
1337         out = fs.create(path);
1338         out.write(corrupted_bytes);
1339         out.write("-----".getBytes());
1340         closeOrFlush(close, out);
1341         break;
1342 
1343       case INSERT_GARBAGE_ON_FIRST_LINE:
1344         fs.delete(path, false);
1345         out = fs.create(path);
1346         out.write(0);
1347         out.write(corrupted_bytes);
1348         closeOrFlush(close, out);
1349         break;
1350 
1351       case INSERT_GARBAGE_IN_THE_MIDDLE:
1352         fs.delete(path, false);
1353         out = fs.create(path);
1354         int middle = (int) Math.floor(corrupted_bytes.length / 2);
1355         out.write(corrupted_bytes, 0, middle);
1356         out.write(0);
1357         out.write(corrupted_bytes, middle, corrupted_bytes.length - middle);
1358         closeOrFlush(close, out);
1359         break;
1360 
1361       case TRUNCATE:
1362         fs.delete(path, false);
1363         out = fs.create(path);
1364         out.write(corrupted_bytes, 0, fileSize-32);
1365         closeOrFlush(close, out);
1366 
1367         break;
1368     }

1371   private void closeOrFlush(boolean close, FSDataOutputStream out)
1372   throws IOException {
1373     if (close) {
1374       out.close();
1375     } else {
1376       Method syncMethod = null;
1377       try {
1378         syncMethod = out.getClass().getMethod("hflush", new Class<?> []{});
1379       } catch (NoSuchMethodException e) {
1380         try {
1381           syncMethod = out.getClass().getMethod("sync", new Class<?> []{});
1382         } catch (NoSuchMethodException ex) {
1383           throw new IOException("This version of Hadoop supports " +
1384               "neither Syncable.sync() nor Syncable.hflush().");
1385         }
1386       }
1387       try {
1388         syncMethod.invoke(out, new Object[]{});
1389       } catch (Exception e) {
1390         throw new IOException(e);
1391       }
1392       // Not in 0out.hflush();
1393     }
1394   }
