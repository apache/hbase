34  /**
35   * Compress key by storing size of common prefix with previous KeyValue
36   * and storing raw size of rest.
37   *
38   * Format:
39   * 1-5 bytes: compressed key length minus prefix (7-bit encoding)
40   * 1-5 bytes: compressed value length (7-bit encoding)
41   * 1-3 bytes: compressed length of common key prefix
42   * ... bytes: rest of key (including timestamp)
43   * ... bytes: value
44   *
45   * In a worst case compressed KeyValue will be three bytes longer than original.
46   *
47   */
48  @InterfaceAudience.Private
49  public class PrefixKeyDeltaEncoder extends BufferedDataBlockEncoder {
50  
51    private int addKV(int prevKeyOffset, DataOutputStream out,
52        ByteBuffer in, int prevKeyLength) throws IOException {
53      int keyLength = in.getInt();
54      int valueLength = in.getInt();
55  
56      if (prevKeyOffset == -1) {
57        // copy the key, there is no common prefix with none
58        ByteBufferUtils.putCompressedInt(out, keyLength);
59        ByteBufferUtils.putCompressedInt(out, valueLength);
60        ByteBufferUtils.putCompressedInt(out, 0);
61        ByteBufferUtils.moveBufferToStream(out, in, keyLength + valueLength);
62      } else {
63        // find a common prefix and skip it
64        int common = ByteBufferUtils.findCommonPrefix(
65            in, prevKeyOffset + KeyValue.ROW_OFFSET,
66            in.position(),
67            Math.min(prevKeyLength, keyLength));
68  
69        ByteBufferUtils.putCompressedInt(out, keyLength - common);
70        ByteBufferUtils.putCompressedInt(out, valueLength);
71        ByteBufferUtils.putCompressedInt(out, common);
72  
73        ByteBufferUtils.skip(in, common);
74        ByteBufferUtils.moveBufferToStream(out, in, keyLength - common
75            + valueLength);
76      }
77  
78      return keyLength;
79    }

81    @Override
82    public void internalEncodeKeyValues(DataOutputStream writeHere,
83        ByteBuffer in, boolean includesMemstoreTS) throws IOException {
84      in.rewind();
85      ByteBufferUtils.putInt(writeHere, in.limit());
86      int prevOffset = -1;
87      int offset = 0;
88      int keyLength = 0;
89      while (in.hasRemaining()) {
90        offset = in.position();
91        keyLength = addKV(prevOffset, writeHere, in, keyLength);
92        afterEncodingKeyValue(in, writeHere, includesMemstoreTS);
93        prevOffset = offset;
94      }
95    }

97    @Override
98    public ByteBuffer decodeKeyValues(DataInputStream source,
99        int allocHeaderLength, int skipLastBytes, boolean includesMemstoreTS)
100           throws IOException {
101     int decompressedSize = source.readInt();
102     ByteBuffer buffer = ByteBuffer.allocate(decompressedSize +
103         allocHeaderLength);
104     buffer.position(allocHeaderLength);
105     int prevKeyOffset = 0;
106 
107     while (source.available() > skipLastBytes) {
108       prevKeyOffset = decodeKeyValue(source, buffer, prevKeyOffset);
109       afterDecodingKeyValue(source, buffer, includesMemstoreTS);
110     }
111 
112     if (source.available() != skipLastBytes) {
113       throw new IllegalStateException("Read too many bytes.");
114     }
115 
116     buffer.limit(buffer.position());
117     return buffer;
118   }

120   private int decodeKeyValue(DataInputStream source, ByteBuffer buffer,
121       int prevKeyOffset)
122           throws IOException, EncoderBufferTooSmallException {
123     int keyLength = ByteBufferUtils.readCompressedInt(source);
124     int valueLength = ByteBufferUtils.readCompressedInt(source);
125     int commonLength = ByteBufferUtils.readCompressedInt(source);
126     int keyOffset;
127     keyLength += commonLength;
128 
129     ensureSpace(buffer, keyLength + valueLength + KeyValue.ROW_OFFSET);
130 
131     buffer.putInt(keyLength);
132     buffer.putInt(valueLength);
133 
134     // copy the prefix
135     if (commonLength > 0) {
136       keyOffset = buffer.position();
137       ByteBufferUtils.copyFromBufferToBuffer(buffer, buffer, prevKeyOffset,
138           commonLength);
139     } else {
140       keyOffset = buffer.position();
141     }
142 
143     // copy rest of the key and value
144     int len = keyLength - commonLength + valueLength;
145     ByteBufferUtils.copyFromStreamToBuffer(buffer, source, len);
146     return keyOffset;
147   }

149   @Override
150   public ByteBuffer getFirstKeyInBlock(ByteBuffer block) {
151     block.mark();
152     block.position(Bytes.SIZEOF_INT);
153     int keyLength = ByteBufferUtils.readCompressedInt(block);
154     ByteBufferUtils.readCompressedInt(block);
155     int commonLength = ByteBufferUtils.readCompressedInt(block);
156     if (commonLength != 0) {
157       throw new AssertionError("Nonzero common length in the first key in "
158           + "block: " + commonLength);
159     }
160     int pos = block.position();
161     block.reset();
162     return ByteBuffer.wrap(block.array(), pos, keyLength).slice();
163   }

170   @Override
171   public EncodedSeeker createSeeker(RawComparator<byte[]> comparator,
172       final boolean includesMemstoreTS) {
173     return new BufferedEncodedSeeker<SeekerState>(comparator) {
174       @Override
175       protected void decodeNext() {
176         current.keyLength = ByteBufferUtils.readCompressedInt(currentBuffer);
177         current.valueLength = ByteBufferUtils.readCompressedInt(currentBuffer);
178         current.lastCommonPrefix =
179             ByteBufferUtils.readCompressedInt(currentBuffer);
180         current.keyLength += current.lastCommonPrefix;
181         current.ensureSpaceForKey();
182         currentBuffer.get(current.keyBuffer, current.lastCommonPrefix,
183             current.keyLength - current.lastCommonPrefix);
184         current.valueOffset = currentBuffer.position();
185         ByteBufferUtils.skip(currentBuffer, current.valueLength);
186         if (includesMemstoreTS) {
187           current.memstoreTS = ByteBufferUtils.readVLong(currentBuffer);
188         } else {
189           current.memstoreTS = 0;
190         }
191         current.nextKvOffset = currentBuffer.position();
192       }
193 
194       @Override
195       protected void decodeFirst() {
196         ByteBufferUtils.skip(currentBuffer, Bytes.SIZEOF_INT);
197         decodeNext();
198       }
199     };
