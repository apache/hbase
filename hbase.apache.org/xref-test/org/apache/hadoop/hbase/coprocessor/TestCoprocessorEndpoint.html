85      desc.addFamily(new HColumnDescriptor(TEST_FAMILY));

105   public void testGeneric() throws Throwable {
106     HTable table = new HTable(util.getConfiguration(), TEST_TABLE);
107     GenericProtocol protocol = table.coprocessorProxy(GenericProtocol.class,
108                                                       Bytes.toBytes("testRow"));
109     String workResult1 = protocol.doWork("foo");
110     assertEquals("foo", workResult1);
111     byte[] workResult2 = protocol.doWork(new byte[]{1});
112     assertArrayEquals(new byte[]{1}, workResult2);
113     byte workResult3 = protocol.doWork((byte)1);
114     assertEquals((byte)1, workResult3);
115     char workResult4 = protocol.doWork('c');
116     assertEquals('c', workResult4);
117     boolean workResult5 = protocol.doWork(true);
118     assertEquals(true, workResult5);
119     short workResult6 = protocol.doWork((short)1);
120     assertEquals((short)1, workResult6);
121     int workResult7 = protocol.doWork(5);
122     assertEquals(5, workResult7);
123     long workResult8 = protocol.doWork(5l);
124     assertEquals(5l, workResult8);
125     double workResult9 = protocol.doWork(6d);
126     assertEquals(6d, workResult9, 0.01);
127     float workResult10 = protocol.doWork(6f);
128     assertEquals(6f, workResult10, 0.01);
129     Text workResult11 = protocol.doWork(new Text("foo"));
130     assertEquals(new Text("foo"), workResult11);
131     table.close();

185   @Test
186   public void testCoprocessorService() throws Throwable {
187     HTable table = new HTable(util.getConfiguration(), TEST_TABLE);
188     NavigableMap<HRegionInfo,ServerName> regions = table.getRegionLocations();
189 
190     final TestProtos.EchoRequestProto request =
191         TestProtos.EchoRequestProto.newBuilder().setMessage("hello").build();
192     final Map<byte[], String> results = Collections.synchronizedMap(
193         new TreeMap<byte[], String>(Bytes.BYTES_COMPARATOR));
194     try {
195       // scan: for all regions
196       final RpcController controller = new ServerRpcController();
197       table.coprocessorService(TestRpcServiceProtos.TestProtobufRpcProto.class,
198           ROWS[0], ROWS[ROWS.length - 1],
199           new Batch.Call<TestRpcServiceProtos.TestProtobufRpcProto, TestProtos.EchoResponseProto>() {
200             public TestProtos.EchoResponseProto call(TestRpcServiceProtos.TestProtobufRpcProto instance)
201                 throws IOException {
202               LOG.debug("Default response is " + TestProtos.EchoRequestProto.getDefaultInstance());
203               BlockingRpcCallback<TestProtos.EchoResponseProto> callback = new BlockingRpcCallback<TestProtos.EchoResponseProto>();
204               instance.echo(controller, request, callback);
205               TestProtos.EchoResponseProto response = callback.get();
206               LOG.debug("Batch.Call returning result " + response);
207               return response;
208             }
209           },
210           new Batch.Callback<TestProtos.EchoResponseProto>() {
211             public void update(byte[] region, byte[] row, TestProtos.EchoResponseProto result) {
212               assertNotNull(result);
213               assertEquals("hello", result.getMessage());
214               results.put(region, result.getMessage());
215             }
216           }
217       );
218       for (Map.Entry<byte[], String> e : results.entrySet()) {
219         LOG.info("Got value "+e.getValue()+" for region "+Bytes.toStringBinary(e.getKey()));
220       }
221       assertEquals(3, results.size());
222       for (HRegionInfo info : regions.navigableKeySet()) {
223         LOG.info("Region info is "+info.getRegionNameAsString());
224         assertTrue(results.containsKey(info.getRegionName()));
225       }
226       results.clear();
227 
228       // scan: for region 2 and region 3
229       table.coprocessorService(TestRpcServiceProtos.TestProtobufRpcProto.class,
230           ROWS[rowSeperator1], ROWS[ROWS.length - 1],
231           new Batch.Call<TestRpcServiceProtos.TestProtobufRpcProto, TestProtos.EchoResponseProto>() {
232             public TestProtos.EchoResponseProto call(TestRpcServiceProtos.TestProtobufRpcProto instance)
233                 throws IOException {
234               LOG.debug("Default response is " + TestProtos.EchoRequestProto.getDefaultInstance());
235               BlockingRpcCallback<TestProtos.EchoResponseProto> callback = new BlockingRpcCallback<TestProtos.EchoResponseProto>();
236               instance.echo(controller, request, callback);
237               TestProtos.EchoResponseProto response = callback.get();
238               LOG.debug("Batch.Call returning result " + response);
239               return response;
240             }
241           },
242           new Batch.Callback<TestProtos.EchoResponseProto>() {
243             public void update(byte[] region, byte[] row, TestProtos.EchoResponseProto result) {
244               assertNotNull(result);
245               assertEquals("hello", result.getMessage());
246               results.put(region, result.getMessage());
247             }
248           }
249       );
250       for (Map.Entry<byte[], String> e : results.entrySet()) {
251         LOG.info("Got value "+e.getValue()+" for region "+Bytes.toStringBinary(e.getKey()));
252       }
253       assertEquals(2, results.size());
254     } finally {
255       table.close();
256     }
257   }

270   private static byte[][] makeN(byte[] base, int n) {
271     byte[][] ret = new byte[n][];
272     for (int i = 0; i < n; i++) {
273       ret[i] = Bytes.add(base, Bytes.toBytes(String.format("%02d", i)));
274     }
275     return ret;
