33  /**
34   * This is a generic filter to be used to filter by comparison.  It takes an
35   * operator (equal, greater, not equal, etc) and a byte [] comparator.
36   * <p>
37   * To filter by row key, use {@link RowFilter}.
38   * <p>
39   * To filter by column qualifier, use {@link QualifierFilter}.
40   * <p>
41   * To filter by value, use {@link SingleColumnValueFilter}.
42   * <p>
43   * These filters can be wrapped with {@link SkipFilter} and {@link WhileMatchFilter}
44   * to add more control.
45   * <p>
46   * Multiple filters can be combined using {@link FilterList}.
47   */
48  @InterfaceAudience.Public
49  @InterfaceStability.Stable
50  public abstract class CompareFilter extends FilterBase {
51  
52    /** Comparison operators. */
53    public enum CompareOp {
54      /** less than */
55      LESS,
56      /** less than or equal to */
57      LESS_OR_EQUAL,
58      /** equals */
59      EQUAL,
60      /** not equal */
61      NOT_EQUAL,
62      /** greater than or equal to */
63      GREATER_OR_EQUAL,
64      /** greater than */
65      GREATER,
66      /** no operation */
67      NO_OP,
68    }
69  
70    protected CompareOp compareOp;
71    protected ByteArrayComparable comparator;
72  
73    /**
74     * Constructor.
75     * @param compareOp the compare op for row matching
76     * @param comparator the comparator for row matching
77     */
78    public CompareFilter(final CompareOp compareOp,
79        final ByteArrayComparable comparator) {
80      this.compareOp = compareOp;
81      this.comparator = comparator;
82    }
83  
84    /**
85     * @return operator
86     */
87    public CompareOp getOperator() {
88      return compareOp;
89    }
90  
91    /**
92     * @return the comparator
93     */
94    public ByteArrayComparable getComparator() {
95      return comparator;
96    }
97  
98    protected boolean doCompare(final CompareOp compareOp,
99        final ByteArrayComparable comparator, final byte [] data,
100       final int offset, final int length) {
101     if (compareOp == CompareOp.NO_OP) {
102       return true;
103     }
104     int compareResult = comparator.compareTo(data, offset, length);
105     switch (compareOp) {
106       case LESS:
107         return compareResult <= 0;
108       case LESS_OR_EQUAL:
109         return compareResult < 0;
110       case EQUAL:
111         return compareResult != 0;
112       case NOT_EQUAL:
113         return compareResult == 0;
114       case GREATER_OR_EQUAL:
115         return compareResult > 0;
116       case GREATER:
117         return compareResult >= 0;
118       default:
119         throw new RuntimeException("Unknown Compare op " +
120           compareOp.name());
121     }
122   }
123 
124   public static ArrayList extractArguments(ArrayList<byte []> filterArguments) {
125     Preconditions.checkArgument(filterArguments.size() == 2,
126                                 "Expected 2 but got: %s", filterArguments.size());
127     CompareOp compareOp = ParseFilter.createCompareOp(filterArguments.get(0));
128     ByteArrayComparable comparator = ParseFilter.createComparator(
129       ParseFilter.removeQuotesFromByteArray(filterArguments.get(1)));
130 
131     if (comparator instanceof RegexStringComparator ||
132         comparator instanceof SubstringComparator) {
133       if (compareOp != CompareOp.EQUAL &&
134           compareOp != CompareOp.NOT_EQUAL) {
135         throw new IllegalArgumentException ("A regexstring comparator and substring comparator" +
136                                             " can only be used with EQUAL and NOT_EQUAL");
137       }
138     }
139     ArrayList arguments = new ArrayList();
140     arguments.add(compareOp);
141     arguments.add(comparator);
142     return arguments;
143   }
144 
145   /**
146    * @return A pb instance to represent this instance.
147    */
148   FilterProtos.CompareFilter convert() {
149     FilterProtos.CompareFilter.Builder builder =
150       FilterProtos.CompareFilter.newBuilder();
151     HBaseProtos.CompareType compareOp = CompareType.valueOf(this.compareOp.name());
152     builder.setCompareOp(compareOp);
153     if (this.comparator != null) builder.setComparator(ProtobufUtil.toComparator(this.comparator));
154     return builder.build();
155   }
156 
157   /**
158    *
159    * @param other
160    * @return true if and only if the fields of the filter that are serialized
161    * are equal to the corresponding fields in other.  Used for testing.
162    */
163   boolean areSerializedFieldsEqual(Filter o) {
164     if (o == this) return true;
165     if (!(o instanceof CompareFilter)) return false;
166 
167     CompareFilter other = (CompareFilter)o;
168     return this.getOperator().equals(other.getOperator()) &&
169       (this.getComparator() == other.getComparator()
170         || this.getComparator().areSerializedFieldsEqual(other.getComparator()));
171   }
172 
173   @Override
174   public String toString() {
175     return String.format("%s (%s, %s)",
176         this.getClass().getSimpleName(),
177         this.compareOp.name(),
178         Bytes.toStringBinary(this.comparator.getValue()));
179   }
