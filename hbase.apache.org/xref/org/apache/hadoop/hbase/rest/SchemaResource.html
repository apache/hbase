78    private HTableDescriptor getTableSchema() throws IOException,
79        TableNotFoundException {
80      HTablePool pool = servlet.getTablePool();
81      HTableInterface table = pool.getTable(tableResource.getName());
82      try {
83        return table.getTableDescriptor();
84      } finally {
85        table.close();
86      }
87    }

89    @GET
90    @Produces({MIMETYPE_TEXT, MIMETYPE_XML, MIMETYPE_JSON, MIMETYPE_PROTOBUF})
91    public Response get(final @Context UriInfo uriInfo) {
92      if (LOG.isDebugEnabled()) {
93        LOG.debug("GET " + uriInfo.getAbsolutePath());
94      }
95      servlet.getMetrics().incrementRequests(1);
96      try {
97        ResponseBuilder response =
98          Response.ok(new TableSchemaModel(getTableSchema()));
99        response.cacheControl(cacheControl);
100       servlet.getMetrics().incrementSucessfulGetRequests(1);
101       return response.build();
102     } catch (TableNotFoundException e) {
103       servlet.getMetrics().incrementFailedGetRequests(1);
104       throw new WebApplicationException(Response.Status.NOT_FOUND);
105     } catch (IOException e) {
106       servlet.getMetrics().incrementFailedGetRequests(1);
107       throw new WebApplicationException(e,
108                   Response.Status.SERVICE_UNAVAILABLE);
109     }
110   }

112   private Response replace(final byte[] name, final TableSchemaModel model,
113       final UriInfo uriInfo, final HBaseAdmin admin) {
114     if (servlet.isReadOnly()) {
115       throw new WebApplicationException(Response.Status.FORBIDDEN);
116     }
117     try {
118       HTableDescriptor htd = new HTableDescriptor(name);
119       for (Map.Entry<QName,Object> e: model.getAny().entrySet()) {
120         htd.setValue(e.getKey().getLocalPart(), e.getValue().toString());
121       }
122       for (ColumnSchemaModel family: model.getColumns()) {
123         HColumnDescriptor hcd = new HColumnDescriptor(family.getName());
124         for (Map.Entry<QName,Object> e: family.getAny().entrySet()) {
125           hcd.setValue(e.getKey().getLocalPart(), e.getValue().toString());
126         }
127         htd.addFamily(hcd);
128       }
129       if (admin.tableExists(name)) {
130         admin.disableTable(name);
131         admin.modifyTable(name, htd);
132         admin.enableTable(name);
133         servlet.getMetrics().incrementSucessfulPutRequests(1);
134       } else try {
135         admin.createTable(htd);
136         servlet.getMetrics().incrementSucessfulPutRequests(1);
137       } catch (TableExistsException e) {
138         // race, someone else created a table with the same name
139         throw new WebApplicationException(e, Response.Status.NOT_MODIFIED);
140       }
141       return Response.created(uriInfo.getAbsolutePath()).build();
142     } catch (IOException e) {
143       throw new WebApplicationException(e,
144             Response.Status.SERVICE_UNAVAILABLE);
145     }
146   }

148   private Response update(final byte[] name, final TableSchemaModel model,
149       final UriInfo uriInfo, final HBaseAdmin admin) {
150     if (servlet.isReadOnly()) {
151       throw new WebApplicationException(Response.Status.FORBIDDEN);
152     }
153     try {
154       HTableDescriptor htd = admin.getTableDescriptor(name);
155       admin.disableTable(name);
156       try {
157         for (ColumnSchemaModel family: model.getColumns()) {
158           HColumnDescriptor hcd = new HColumnDescriptor(family.getName());
159           for (Map.Entry<QName,Object> e: family.getAny().entrySet()) {
160             hcd.setValue(e.getKey().getLocalPart(), e.getValue().toString());
161           }
162           if (htd.hasFamily(hcd.getName())) {
163             admin.modifyColumn(name, hcd);
164           } else {
165             admin.addColumn(name, hcd);
166           }
167         }
168       } catch (IOException e) {
169         throw new WebApplicationException(e,
170             Response.Status.INTERNAL_SERVER_ERROR);
171       } finally {
172         admin.enableTable(tableResource.getName());
173       }
174       servlet.getMetrics().incrementSucessfulPutRequests(1);
175       return Response.ok().build();
176     } catch (IOException e) {
177       throw new WebApplicationException(e,
178           Response.Status.SERVICE_UNAVAILABLE);
179     }
180   }
181 
182   private Response update(final TableSchemaModel model, final boolean replace,
183       final UriInfo uriInfo) {
184     try {
185       byte[] name = Bytes.toBytes(tableResource.getName());
186       HBaseAdmin admin = new HBaseAdmin(servlet.getConfiguration());
187       if (replace || !admin.tableExists(name)) {
188         return replace(name, model, uriInfo, admin);
189       } else {
190         return update(name, model, uriInfo, admin);
191       }
192     } catch (IOException e) {
193       servlet.getMetrics().incrementFailedPutRequests(1);
194       throw new WebApplicationException(e, 
195             Response.Status.SERVICE_UNAVAILABLE);
196     }
197   }

221   @DELETE
222   public Response delete(final @Context UriInfo uriInfo) {
223     if (LOG.isDebugEnabled()) {
224       LOG.debug("DELETE " + uriInfo.getAbsolutePath());
225     }
226     servlet.getMetrics().incrementRequests(1);
227     try {
228       HBaseAdmin admin = new HBaseAdmin(servlet.getConfiguration());
229       boolean success = false;
230       for (int i = 0; i < 10; i++) try {
231         admin.disableTable(tableResource.getName());
232         success = true;
233         break;
234       } catch (IOException e) {
235       }
236       if (!success) {
237         throw new IOException("could not disable table");
238       }
239       admin.deleteTable(tableResource.getName());
240       servlet.getMetrics().incrementSucessfulDeleteRequests(1);
241       return Response.ok().build();
242     } catch (TableNotFoundException e) {
243       servlet.getMetrics().incrementFailedDeleteRequests(1);
244       throw new WebApplicationException(Response.Status.NOT_FOUND);
245     } catch (IOException e) {
246       servlet.getMetrics().incrementFailedDeleteRequests(1);
247       throw new WebApplicationException(e, 
248             Response.Status.SERVICE_UNAVAILABLE);
249     }
