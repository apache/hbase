203   @Test
204   public void testPersistence() throws Exception {
205     Configuration conf = UTIL.getConfiguration();
206     AccessControlLists.addUserPermission(conf,
207         new UserPermission(Bytes.toBytes("albert"), TEST_TABLE, null,
208                            (byte[])null, TablePermission.Action.READ));
209     AccessControlLists.addUserPermission(conf,
210         new UserPermission(Bytes.toBytes("betty"), TEST_TABLE, null,
211                            (byte[])null, TablePermission.Action.READ,
212                            TablePermission.Action.WRITE));
213     AccessControlLists.addUserPermission(conf,
214         new UserPermission(Bytes.toBytes("clark"),
215                            TEST_TABLE, TEST_FAMILY,
216                            TablePermission.Action.READ));
217     AccessControlLists.addUserPermission(conf,
218         new UserPermission(Bytes.toBytes("dwight"),
219                            TEST_TABLE, TEST_FAMILY, TEST_QUALIFIER,
220                            TablePermission.Action.WRITE));
221 
222     // verify permissions survive changes in table metadata
223     ListMultimap<String,TablePermission> preperms =
224         AccessControlLists.getTablePermissions(conf, TEST_TABLE);
225 
226     HTable table = new HTable(conf, TEST_TABLE);
227     table.put(new Put(Bytes.toBytes("row1"))
228         .add(TEST_FAMILY, TEST_QUALIFIER, Bytes.toBytes("v1")));
229     table.put(new Put(Bytes.toBytes("row2"))
230         .add(TEST_FAMILY, TEST_QUALIFIER, Bytes.toBytes("v2")));
231     HBaseAdmin admin = UTIL.getHBaseAdmin();
232     admin.split(TEST_TABLE);
233 
234     // wait for split
235     Thread.sleep(10000);
236 
237     ListMultimap<String,TablePermission> postperms =
238         AccessControlLists.getTablePermissions(conf, TEST_TABLE);
239 
240     checkMultimapEqual(preperms, postperms);
241   }

264   public void checkMultimapEqual(ListMultimap<String,TablePermission> first,
265       ListMultimap<String,TablePermission> second) {
266     assertEquals(first.size(), second.size());
267     for (String key : first.keySet()) {
268       List<TablePermission> firstPerms = first.get(key);
269       List<TablePermission> secondPerms = second.get(key);
270       assertNotNull(secondPerms);
271       assertEquals(firstPerms.size(), secondPerms.size());
272       LOG.info("First permissions: "+firstPerms.toString());
273       LOG.info("Second permissions: "+secondPerms.toString());
274       for (TablePermission p : firstPerms) {
275         assertTrue("Permission "+p.toString()+" not found", secondPerms.contains(p));
276       }
277     }

282     TablePermission p1 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);
283     TablePermission p2 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);

289     assertTrue(p1.equals(p2));
290     assertTrue(p2.equals(p1));
291 
292     p1 = new TablePermission(TEST_TABLE, TEST_FAMILY, TablePermission.Action.READ, TablePermission.Action.WRITE);
293     p2 = new TablePermission(TEST_TABLE, TEST_FAMILY, TablePermission.Action.WRITE, TablePermission.Action.READ);
294     assertTrue(p1.equals(p2));
295     assertTrue(p2.equals(p1));
296 
297     p1 = new TablePermission(TEST_TABLE, TEST_FAMILY, TEST_QUALIFIER, TablePermission.Action.READ, TablePermission.Action.WRITE);
298     p2 = new TablePermission(TEST_TABLE, TEST_FAMILY, TEST_QUALIFIER, TablePermission.Action.WRITE, TablePermission.Action.READ);
299     assertTrue(p1.equals(p2));
300     assertTrue(p2.equals(p1));
301 
302     p1 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);
303     p2 = new TablePermission(TEST_TABLE, TEST_FAMILY, TablePermission.Action.READ);
304     assertFalse(p1.equals(p2));
305     assertFalse(p2.equals(p1));
306 
307     p1 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);
308     p2 = new TablePermission(TEST_TABLE, null, TablePermission.Action.WRITE);
309     assertFalse(p1.equals(p2));
310     assertFalse(p2.equals(p1));
311     p2 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ, TablePermission.Action.WRITE);
312     assertFalse(p1.equals(p2));
313     assertFalse(p2.equals(p1));
314 
315     p1 = new TablePermission(TEST_TABLE, null, TablePermission.Action.READ);
316     p2 = new TablePermission(TEST_TABLE2, null, TablePermission.Action.READ);
317     assertFalse(p1.equals(p2));
318     assertFalse(p2.equals(p1));
319 
320     p2 = new TablePermission(TEST_TABLE, null);
321     assertFalse(p1.equals(p2));
322     assertFalse(p2.equals(p1));

348     assertEquals("Should have 1 permission for user2", 1, user2Perms.size());
349     assertEquals("user2 should have CREATE permission",
350                  new Permission.Action[] { Permission.Action.CREATE },
351                  user2Perms.get(0).getActions());

353     List<TablePermission> user3Perms = perms.get("user3");
354     assertEquals("Should have 1 permission for user3", 1, user3Perms.size());

359                  user3Perms.get(0).getActions());

362   @Test
363   public void testAuthManager() throws Exception {
364     Configuration conf = UTIL.getConfiguration();
365     /* test a race condition causing TableAuthManager to sometimes fail global permissions checks
366      * when the global cache is being updated
367      */
368     TableAuthManager authManager = TableAuthManager.get(ZKW, conf);
369     // currently running user is the system user and should have global admin perms
370     User currentUser = User.getCurrent();
371     assertTrue(authManager.authorize(currentUser, Permission.Action.ADMIN));
372     for (int i=1; i<=50; i++) {
373       AccessControlLists.addUserPermission(conf, new UserPermission(Bytes.toBytes("testauth"+i),
374           Permission.Action.ADMIN, Permission.Action.READ, Permission.Action.WRITE));
375       // make sure the system user still shows as authorized
376       assertTrue("Failed current user auth check on iter "+i,
377           authManager.authorize(currentUser, Permission.Action.ADMIN));
378     }
