51  /**
52   * Base class for performing operations against tables.
53   * Checks on whether the process can go forward are done in constructor rather
54   * than later on in {@link #process()}.  The idea is to fail fast rather than
55   * later down in an async invocation of {@link #process()} (which currently has
56   * no means of reporting back issues once started).
57   */
58  @InterfaceAudience.Private
59  public abstract class TableEventHandler extends EventHandler {
60    private static final Log LOG = LogFactory.getLog(TableEventHandler.class);
61    protected final MasterServices masterServices;
62    protected final byte [] tableName;
63    protected final String tableNameStr;
64    protected boolean persistedToZk = false;
65  
66    public TableEventHandler(EventType eventType, byte [] tableName, Server server,
67        MasterServices masterServices)
68    throws IOException {
69      super(server, eventType);
70      this.masterServices = masterServices;
71      this.tableName = tableName;
72      try {
73        this.masterServices.checkTableModifiable(tableName);
74      } catch (TableNotDisabledException ex)  {
75        if (isOnlineSchemaChangeAllowed()
76            && eventType.isOnlineSchemaChangeSupported()) {
77          LOG.debug("Ignoring table not disabled exception " +
78              "for supporting online schema changes.");
79        }	else {
80          throw ex;
81        }
82      }
83      this.tableNameStr = Bytes.toString(this.tableName);
84    }
85  
86   private boolean isOnlineSchemaChangeAllowed() {
87     return this.server.getConfiguration().getBoolean(
88       "hbase.online.schema.update.enable", false);
89   }
90  
91    @Override
92    public void process() {
93      try {
94        LOG.info("Handling table operation " + eventType + " on table " +
95            Bytes.toString(tableName));
96        List<HRegionInfo> hris =
97          MetaReader.getTableRegions(this.server.getCatalogTracker(),
98            tableName);
99        handleTableOperation(hris);
100       if (eventType.isOnlineSchemaChangeSupported() && this.masterServices.
101           getAssignmentManager().getZKTable().
102           isEnabledTable(Bytes.toString(tableName))) {
103         if (reOpenAllRegions(hris)) {
104           LOG.info("Completed table operation " + eventType + " on table " +
105               Bytes.toString(tableName));
106         } else {
107           LOG.warn("Error on reopening the regions");
108         }
109       }
110     } catch (IOException e) {
111       LOG.error("Error manipulating table " + Bytes.toString(tableName), e);
112     } catch (KeeperException e) {
113       LOG.error("Error manipulating table " + Bytes.toString(tableName), e);
114     } finally {
115       // notify the waiting thread that we're done persisting the request
116       setPersist();
117     }
118   }

120   public boolean reOpenAllRegions(List<HRegionInfo> regions) throws IOException {
121     boolean done = false;
122     LOG.info("Bucketing regions by region server...");
123     HTable table = new HTable(masterServices.getConfiguration(), tableName);
124     TreeMap<ServerName, List<HRegionInfo>> serverToRegions = Maps
125         .newTreeMap();
126     NavigableMap<HRegionInfo, ServerName> hriHserverMapping = table.getRegionLocations();
127     List<HRegionInfo> reRegions = new ArrayList<HRegionInfo>();
128     for (HRegionInfo hri : regions) {
129       ServerName rsLocation = hriHserverMapping.get(hri);
130 
131       // Skip the offlined split parent region
132       // See HBASE-4578 for more information.
133       if (null == rsLocation) {
134         LOG.info("Skip " + hri);
135         continue;
136       }
137       if (!serverToRegions.containsKey(rsLocation)) {
138         LinkedList<HRegionInfo> hriList = Lists.newLinkedList();
139         serverToRegions.put(rsLocation, hriList);
140       }
141       reRegions.add(hri);
142       serverToRegions.get(rsLocation).add(hri);
143     }
144     
145     LOG.info("Reopening " + reRegions.size() + " regions on "
146         + serverToRegions.size() + " region servers.");
147     this.masterServices.getAssignmentManager().setRegionsToReopen(reRegions);
148     BulkReOpen bulkReopen = new BulkReOpen(this.server, serverToRegions,
149         this.masterServices.getAssignmentManager());
150     while (true) {
151       try {
152         if (bulkReopen.bulkReOpen()) {
153           done = true;
154           break;
155         } else {
156           LOG.warn("Timeout before reopening all regions");
157         }
158       } catch (InterruptedException e) {
159         LOG.warn("Reopen was interrupted");
160         // Preserve the interrupt.
161         Thread.currentThread().interrupt();
162         break;
163       }
164     }
165     return done;
166   }

168   /**
169    * Table modifications are processed asynchronously, but provide an API for
170    * you to query their status.
171    *
172    * @throws IOException
173    */
174   public synchronized void waitForPersist() throws IOException {
175     if (!persistedToZk) {
176       try {
177         wait();
178       } catch (InterruptedException ie) {
179         throw (IOException) new InterruptedIOException().initCause(ie);
180       }
181       assert persistedToZk;
182     }
183   }
184 
185   private synchronized void setPersist() {
186     if (!persistedToZk) {
187       persistedToZk = true;
188       notify();
189     }
190   }

192   /**
193    * @return Table descriptor for this table
194    * @throws TableExistsException
195    * @throws FileNotFoundException
196    * @throws IOException
197    */
198   HTableDescriptor getTableDescriptor()
199   throws FileNotFoundException, IOException {
200     final String name = Bytes.toString(tableName);
201     HTableDescriptor htd =
202       this.masterServices.getTableDescriptors().get(name);
203     if (htd == null) {
204       throw new IOException("HTableDescriptor missing for " + name);
205     }
206     return htd;
207   }

209   byte [] hasColumnFamily(final HTableDescriptor htd, final byte [] cf)
210   throws InvalidFamilyOperationException {
211     if (!htd.hasFamily(cf)) {
212       throw new InvalidFamilyOperationException("Column family '" +
213         Bytes.toString(cf) + "' does not exist");
214     }
215     return cf;
216   }
