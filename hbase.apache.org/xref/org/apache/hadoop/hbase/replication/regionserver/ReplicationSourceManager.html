51  /**
52   * This class is responsible to manage all the replication
53   * sources. There are two classes of sources:
54   * <li> Normal sources are persistent and one per peer cluster</li>
55   * <li> Old sources are recovered from a failed region server and our
56   * only goal is to finish replicating the HLog queue it had up in ZK</li>
57   *
58   * When a region server dies, this class uses a watcher to get notified and it
59   * tries to grab a lock in order to transfer all the queues in a local
60   * old source.
61   */
62  @InterfaceAudience.Private
63  public class ReplicationSourceManager {
64    private static final Log LOG =
65        LogFactory.getLog(ReplicationSourceManager.class);
66    // List of all the sources that read this RS's logs
67    private final List<ReplicationSourceInterface> sources;
68    // List of all the sources we got from died RSs
69    private final List<ReplicationSourceInterface> oldsources;
70    // Indicates if we are currently replicating
71    private final AtomicBoolean replicating;
72    // Helper for zookeeper
73    private final ReplicationZookeeper zkHelper;
74    // All about stopping
75    private final Stoppable stopper;
76    // All logs we are currently trackign
77    private final Map<String, SortedSet<String>> hlogsById;
78    private final Configuration conf;
79    private final FileSystem fs;
80    // The path to the latest log we saw, for new coming sources
81    private Path latestPath;
82    // List of all the other region servers in this cluster
83    private final List<String> otherRegionServers = new ArrayList<String>();
84    // Path to the hlogs directories
85    private final Path logDir;
86    // Path to the hlog archive
87    private final Path oldLogDir;
88    // The number of ms that we wait before moving znodes, HBASE-3596
89    private final long sleepBeforeFailover;
90    // Homemade executer service for replication
91    private final ThreadPoolExecutor executor;

93    /**
94     * Creates a replication manager and sets the watch on all the other
95     * registered region servers
96     * @param zkHelper the zk helper for replication
97     * @param conf the configuration to use
98     * @param stopper the stopper object for this region server
99     * @param fs the file system to use
100    * @param replicating the status of the replication on this cluster
101    * @param logDir the directory that contains all hlog directories of live RSs
102    * @param oldLogDir the directory where old logs are archived
103    */
104   public ReplicationSourceManager(final ReplicationZookeeper zkHelper,
105                                   final Configuration conf,
106                                   final Stoppable stopper,
107                                   final FileSystem fs,
108                                   final AtomicBoolean replicating,
109                                   final Path logDir,
110                                   final Path oldLogDir) {
111     this.sources = new ArrayList<ReplicationSourceInterface>();
112     this.replicating = replicating;
113     this.zkHelper = zkHelper;
114     this.stopper = stopper;
115     this.hlogsById = new HashMap<String, SortedSet<String>>();
116     this.oldsources = new ArrayList<ReplicationSourceInterface>();
117     this.conf = conf;
118     this.fs = fs;
119     this.logDir = logDir;
120     this.oldLogDir = oldLogDir;
121     this.sleepBeforeFailover = conf.getLong("replication.sleep.before.failover", 2000);
122     this.zkHelper.registerRegionServerListener(
123         new OtherRegionServerWatcher(this.zkHelper.getZookeeperWatcher()));
124     this.zkHelper.registerRegionServerListener(
125         new PeersWatcher(this.zkHelper.getZookeeperWatcher()));
126     this.zkHelper.listPeersIdsAndWatch();
127     // It's preferable to failover 1 RS at a time, but with good zk servers
128     // more could be processed at the same time.
129     int nbWorkers = conf.getInt("replication.executor.workers", 1);
130     // use a short 100ms sleep since this could be done inline with a RS startup
131     // even if we fail, other region servers can take care of it
132     this.executor = new ThreadPoolExecutor(nbWorkers, nbWorkers,
133         100, TimeUnit.MILLISECONDS,
134         new LinkedBlockingQueue<Runnable>());
135     ThreadFactoryBuilder tfb = new ThreadFactoryBuilder();
136     tfb.setNameFormat("ReplicationExecutor-%d");
137     this.executor.setThreadFactory(tfb.build());
138   }

140   /**
141    * Provide the id of the peer and a log key and this method will figure which
142    * hlog it belongs to and will log, for this region server, the current
143    * position. It will also clean old logs from the queue.
144    * @param log Path to the log currently being replicated from
145    * replication status in zookeeper. It will also delete older entries.
146    * @param id id of the peer cluster
147    * @param position current location in the log
148    * @param queueRecovered indicates if this queue comes from another region server
149    * @param holdLogInZK if true then the log is retained in ZK
150    */
151   public void logPositionAndCleanOldLogs(Path log, String id, long position, 
152       boolean queueRecovered, boolean holdLogInZK) {
153     String key = log.getName();
154     this.zkHelper.writeReplicationStatus(key, id, position);
155     if (holdLogInZK) {
156      return;
157     }
158     synchronized (this.hlogsById) {
159       SortedSet<String> hlogs = this.hlogsById.get(id);
160       if (!queueRecovered && hlogs.first() != key) {
161         SortedSet<String> hlogSet = hlogs.headSet(key);
162         for (String hlog : hlogSet) {
163           this.zkHelper.removeLogFromList(hlog, id);
164         }
165         hlogSet.clear();
166       }
167     }
168   }

170   /**
171    * Adds a normal source per registered peer cluster and tries to process all
172    * old region server hlog queues
173    */
174   public void init() throws IOException {
175     for (String id : this.zkHelper.getPeerClusters().keySet()) {
176       addSource(id);
177     }
178     List<String> currentReplicators = this.zkHelper.getListOfReplicators();
179     if (currentReplicators == null || currentReplicators.size() == 0) {
180       return;
181     }
182     synchronized (otherRegionServers) {
183       refreshOtherRegionServersList();
184       LOG.info("Current list of replicators: " + currentReplicators
185           + " other RSs: " + otherRegionServers);
186     }
187     // Look if there's anything to process after a restart
188     for (String rs : currentReplicators) {
189       synchronized (otherRegionServers) {
190         if (!this.otherRegionServers.contains(rs)) {
191           transferQueues(rs);
192         }
193       }
194     }
195   }

197   /**
198    * Add a new normal source to this region server
199    * @param id the id of the peer cluster
200    * @return the source that was created
201    * @throws IOException
202    */
203   public ReplicationSourceInterface addSource(String id) throws IOException {
204     ReplicationSourceInterface src =
205         getReplicationSource(this.conf, this.fs, this, stopper, replicating, id);
206     synchronized (this.hlogsById) {
207       this.sources.add(src);
208       this.hlogsById.put(id, new TreeSet<String>());
209       // Add the latest hlog to that source's queue
210       if (this.latestPath != null) {
211         String name = this.latestPath.getName();
212         this.hlogsById.get(id).add(name);
213         try {
214           this.zkHelper.addLogToList(name, src.getPeerClusterZnode());
215         } catch (KeeperException ke) {
216           String message = "Cannot add log to zk for" +
217             " replication when creating a new source";
218           stopper.stop(message);
219           throw new IOException(message, ke);
220         }
221         src.enqueueLog(this.latestPath);
222       }
223     }
224     src.startup();
225     return src;
226   }

228   /**
229    * Terminate the replication on this region server
230    */
231   public void join() {
232     this.executor.shutdown();
233     if (this.sources.size() == 0) {
234       this.zkHelper.deleteOwnRSZNode();
235     }
236     for (ReplicationSourceInterface source : this.sources) {
237       source.terminate("Region server is closing");
238     }
239   }

257   void preLogRoll(Path newLog) throws IOException {
258     if (!this.replicating.get()) {
259       LOG.warn("Replication stopped, won't add new log");
260       return;
261     }
262 
263     synchronized (this.hlogsById) {
264       String name = newLog.getName();
265       for (ReplicationSourceInterface source : this.sources) {
266         try {
267           this.zkHelper.addLogToList(name, source.getPeerClusterZnode());
268         } catch (KeeperException ke) {
269           throw new IOException("Cannot add log to zk for replication", ke);
270         }
271       }
272       for (SortedSet<String> hlogs : this.hlogsById.values()) {
273         if (this.sources.isEmpty()) {
274           // If there's no slaves, don't need to keep the old hlogs since
275           // we only consider the last one when a new slave comes in
276           hlogs.clear();
277         }
278         hlogs.add(name);
279       }
280     }
281 
282     this.latestPath = newLog;
283   }
284 
285   void postLogRoll(Path newLog) throws IOException {
286     if (!this.replicating.get()) {
287       LOG.warn("Replication stopped, won't add new log");
288       return;
289     }
290 
291     // This only updates the sources we own, not the recovered ones
292     for (ReplicationSourceInterface source : this.sources) {
293       source.enqueueLog(newLog);    
294     }
295   }

339   /**
340    * Transfer all the queues of the specified to this region server.
341    * First it tries to grab a lock and if it works it will move the
342    * znodes and finally will delete the old znodes.
343    *
344    * It creates one old source for any type of source of the old rs.
345    * @param rsZnode
346    */
347   public void transferQueues(String rsZnode) {
348     NodeFailoverWorker transfer = new NodeFailoverWorker(rsZnode);
349     try {
350       this.executor.execute(transfer);
351     } catch (RejectedExecutionException ex) {
352       LOG.info("Cancelling the transfer of " + rsZnode +
353           " because of " + ex.getMessage());
354     }
355   }

367   /**
368    * Thie method first deletes all the recovered sources for the specified
369    * id, then deletes the normal source (deleting all related data in ZK).
370    * @param id The id of the peer cluster
371    */
372   public void removePeer(String id) {
373     LOG.info("Closing the following queue " + id + ", currently have "
374         + sources.size() + " and another "
375         + oldsources.size() + " that were recovered");
376     String terminateMessage = "Replication stream was removed by a user";
377     ReplicationSourceInterface srcToRemove = null;
378     List<ReplicationSourceInterface> oldSourcesToDelete =
379         new ArrayList<ReplicationSourceInterface>();
380     // First close all the recovered sources for this peer
381     for (ReplicationSourceInterface src : oldsources) {
382       if (id.equals(src.getPeerClusterId())) {
383         oldSourcesToDelete.add(src);
384       }
385     }
386     for (ReplicationSourceInterface src : oldSourcesToDelete) {
387       src.terminate(terminateMessage);
388       closeRecoveredQueue((src));
389     }
390     LOG.info("Number of deleted recovered sources for " + id + ": "
391         + oldSourcesToDelete.size());
392     // Now look for the one on this cluster
393     for (ReplicationSourceInterface src : this.sources) {
394       if (id.equals(src.getPeerClusterId())) {
395         srcToRemove = src;
396         break;
397       }
398     }
399     if (srcToRemove == null) {
400       LOG.error("The queue we wanted to close is missing " + id);
401       return;
402     }
403     srcToRemove.terminate(terminateMessage);
404     this.sources.remove(srcToRemove);
405     this.zkHelper.deleteSource(id, true);
406   }

408   /**
409    * Reads the list of region servers from ZK and atomically clears our
410    * local view of it and replaces it with the updated list.
411    * 
412    * @return true if the local list of the other region servers was updated
413    * with the ZK data (even if it was empty),
414    * false if the data was missing in ZK
415    */
416   private boolean refreshOtherRegionServersList() {
417     List<String> newRsList = zkHelper.getRegisteredRegionServers();
418     if (newRsList == null) {
419       return false;
420     } else {
421       synchronized (otherRegionServers) {
422         otherRegionServers.clear();
423         otherRegionServers.addAll(newRsList);
424       }
425     }
426     return true;
427   }

430    * Watcher used to be notified of the other region server's death
431    * in the local cluster. It initiates the process to transfer the queues
432    * if it is able to grab the lock.
433    */
434   public class OtherRegionServerWatcher extends ZooKeeperListener {
435 
436     /**
437      * Construct a ZooKeeper event listener.
438      */
439     public OtherRegionServerWatcher(ZooKeeperWatcher watcher) {
440       super(watcher);
441     }
442 
443     /**
444      * Called when a new node has been created.
445      * @param path full path of the new node
446      */
447     public void nodeCreated(String path) {
448       refreshListIfRightPath(path);
449     }
450 
451     /**
452      * Called when a node has been deleted
453      * @param path full path of the deleted node
454      */
455     public void nodeDeleted(String path) {
456       if (stopper.isStopped()) {
457         return;
458       }
459       boolean cont = refreshListIfRightPath(path);
460       if (!cont) {
461         return;
462       }
463       LOG.info(path + " znode expired, trying to lock it");
464       transferQueues(ReplicationZookeeper.getZNodeName(path));
465     }
466 
467     /**
468      * Called when an existing node has a child node added or removed.
469      * @param path full path of the node whose children have changed
470      */
471     public void nodeChildrenChanged(String path) {
472       if (stopper.isStopped()) {
473         return;
474       }
475       refreshListIfRightPath(path);
476     }
477 
478     private boolean refreshListIfRightPath(String path) {
479       if (!path.startsWith(zkHelper.getZookeeperWatcher().rsZNode)) {
480         return false;
481       }
482       return refreshOtherRegionServersList();
483     }
484   }

486   /**
487    * Watcher used to follow the creation and deletion of peer clusters.
488    */
489   public class PeersWatcher extends ZooKeeperListener {
490 
491     /**
492      * Construct a ZooKeeper event listener.
493      */
494     public PeersWatcher(ZooKeeperWatcher watcher) {
495       super(watcher);
496     }
497 
498     /**
499      * Called when a node has been deleted
500      * @param path full path of the deleted node
501      */
502     public void nodeDeleted(String path) {
503       List<String> peers = refreshPeersList(path);
504       if (peers == null) {
505         return;
506       }
507       String id = ReplicationZookeeper.getZNodeName(path);
508       removePeer(id);
509     }
510 
511     /**
512      * Called when an existing node has a child node added or removed.
513      * @param path full path of the node whose children have changed
514      */
515     public void nodeChildrenChanged(String path) {
516       List<String> peers = refreshPeersList(path);
517       if (peers == null) {
518         return;
519       }
520       for (String id : peers) {
521         try {
522           boolean added = zkHelper.connectToPeer(id);
523           if (added) {
524             addSource(id);
525           }
526         } catch (IOException e) {
527           // TODO manage better than that ?
528           LOG.error("Error while adding a new peer", e);
529         } catch (KeeperException e) {
530           LOG.error("Error while adding a new peer", e);
531         }
532       }
533     }
534 
535     /**
536      * Verify if this event is meant for us, and if so then get the latest
537      * peers' list from ZK. Also reset the watches.
538      * @param path path to check against
539      * @return A list of peers' identifiers if the event concerns this watcher,
540      * else null.
541      */
542     private List<String> refreshPeersList(String path) {
543       if (!path.startsWith(zkHelper.getPeersZNode())) {
544         return null;
545       }
546       return zkHelper.listPeersIdsAndWatch();
547     }

550   /**
551    * Class responsible to setup new ReplicationSources to take care of the
552    * queues from dead region servers.
553    */
554   class NodeFailoverWorker extends Thread {
555 
556     private String rsZnode;
557 
558     /**
559      *
560      * @param rsZnode
561      */
562     public NodeFailoverWorker(String rsZnode) {
563       super("Failover-for-"+rsZnode);
564       this.rsZnode = rsZnode;
565     }
566 
567     @Override
568     public void run() {
569       // Wait a bit before transferring the queues, we may be shutting down.
570       // This sleep may not be enough in some cases.
571       try {
572         Thread.sleep(sleepBeforeFailover);
573       } catch (InterruptedException e) {
574         LOG.warn("Interrupted while waiting before transferring a queue.");
575         Thread.currentThread().interrupt();
576       }
577       // We try to lock that rs' queue directory
578       if (stopper.isStopped()) {
579         LOG.info("Not transferring queue since we are shutting down");
580         return;
581       }
582       if (!zkHelper.lockOtherRS(rsZnode)) {
583         return;
584       }
585       LOG.info("Moving " + rsZnode + "'s hlogs to my queue");
586       SortedMap<String, SortedSet<String>> newQueues =
587           zkHelper.copyQueuesFromRS(rsZnode);
588       zkHelper.deleteRsQueues(rsZnode);
589       if (newQueues == null || newQueues.size() == 0) {
590         return;
591       }
592 
593       for (Map.Entry<String, SortedSet<String>> entry : newQueues.entrySet()) {
594         String peerId = entry.getKey();
595         try {
596           ReplicationSourceInterface src = getReplicationSource(conf,
597               fs, ReplicationSourceManager.this, stopper, replicating, peerId);
598           if (!zkHelper.getPeerClusters().containsKey(src.getPeerClusterId())) {
599             src.terminate("Recovered queue doesn't belong to any current peer");
600             break;
601           }
602           oldsources.add(src);
603           for (String hlog : entry.getValue()) {
604             src.enqueueLog(new Path(oldLogDir, hlog));
605           }
606           src.startup();
607         } catch (IOException e) {
608           // TODO manage it
609           LOG.error("Failed creating a source", e);
610         }
611       }
612     }
613   }
