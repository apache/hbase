37  /**
38   * Used to perform Put operations for a single row.
39   * <p>
40   * To perform a Put, instantiate a Put object with the row to insert to and
41   * for each column to be inserted, execute {@link #add(byte[], byte[], byte[]) add} or
42   * {@link #add(byte[], byte[], long, byte[]) add} if setting the timestamp.
43   */
44  @InterfaceAudience.Public
45  @InterfaceStability.Stable
46  public class Put extends Mutation implements HeapSize, Comparable<Row> {
47    private static final long OVERHEAD = ClassSize.align(
48        ClassSize.OBJECT + 2 * ClassSize.REFERENCE +
49        2 * Bytes.SIZEOF_LONG + Bytes.SIZEOF_BOOLEAN +
50        ClassSize.REFERENCE + ClassSize.TREEMAP);

79    /**
80     * Create a Put operation for the specified row, using a given timestamp, and an existing row lock.
81     * @param row row key
82     * @param ts timestamp
83     * @param rowLock previously acquired row lock, or null
84     */
85    public Put(byte [] row, long ts, RowLock rowLock) {
86      if(row == null || row.length > HConstants.MAX_ROW_LENGTH) {
87        throw new IllegalArgumentException("Row key is invalid");
88      }
89      this.row = Arrays.copyOf(row, row.length);
90      this.ts = ts;
91      if(rowLock != null) {
92        this.lockId = rowLock.getLockId();
93      }
94    }

96    /**
97     * Copy constructor.  Creates a Put operation cloned from the specified Put.
98     * @param putToCopy put to copy
99     */
100   public Put(Put putToCopy) {
101     this(putToCopy.getRow(), putToCopy.ts, putToCopy.getRowLock());
102     this.familyMap =
103       new TreeMap<byte [], List<KeyValue>>(Bytes.BYTES_COMPARATOR);
104     for(Map.Entry<byte [], List<KeyValue>> entry :
105       putToCopy.getFamilyMap().entrySet()) {
106       this.familyMap.put(entry.getKey(), entry.getValue());
107     }
108     this.writeToWAL = putToCopy.writeToWAL;
109   }

122   /**
123    * Add the specified column and value, with the specified timestamp as
124    * its version to this Put operation.
125    * @param family family name
126    * @param qualifier column qualifier
127    * @param ts version timestamp
128    * @param value column value
129    * @return this
130    */
131   public Put add(byte [] family, byte [] qualifier, long ts, byte [] value) {
132     List<KeyValue> list = getKeyValueList(family);
133     KeyValue kv = createPutKeyValue(family, qualifier, ts, value);
134     list.add(kv);
135     familyMap.put(kv.getFamily(), list);
136     return this;
137   }

139   /**
140    * Add the specified KeyValue to this Put operation.  Operation assumes that
141    * the passed KeyValue is immutable and its backing array will not be modified
142    * for the duration of this Put.
143    * @param kv individual KeyValue
144    * @return this
145    * @throws java.io.IOException e
146    */
147   public Put add(KeyValue kv) throws IOException{
148     byte [] family = kv.getFamily();
149     List<KeyValue> list = getKeyValueList(family);
150     //Checking that the row of the kv is the same as the put
151     int res = Bytes.compareTo(this.row, 0, row.length,
152         kv.getBuffer(), kv.getRowOffset(), kv.getRowLength());
153     if(res != 0) {
154       throw new IOException("The row in the recently added KeyValue " +
155           Bytes.toStringBinary(kv.getBuffer(), kv.getRowOffset(),
156         kv.getRowLength()) + " doesn't match the original one " +
157         Bytes.toStringBinary(this.row));
158     }
159     list.add(kv);
160     familyMap.put(family, list);
161     return this;
162   }

235   /*
236    * Private method to determine if this object's familyMap contains
237    * the given value assigned to the given family, qualifier and timestamp
238    * respecting the 2 boolean arguments
239    *
240    * @param family
241    * @param qualifier
242    * @param ts
243    * @param value
244    * @param ignoreTS
245    * @param ignoreValue
246    * @return returns true if the given family, qualifier timestamp and value
247    * already has an existing KeyValue object in the family map.
248    */
249   private boolean has(byte[] family, byte[] qualifier, long ts, byte[] value,
250                       boolean ignoreTS, boolean ignoreValue) {
251     List<KeyValue> list = getKeyValueList(family);
252     if (list.size() == 0) {
253       return false;
254     }
255     // Boolean analysis of ignoreTS/ignoreValue.
256     // T T => 2
257     // T F => 3 (first is always true)
258     // F T => 2
259     // F F => 1
260     if (!ignoreTS && !ignoreValue) {
261       for (KeyValue kv : list) {
262         if (Arrays.equals(kv.getFamily(), family) &&
263             Arrays.equals(kv.getQualifier(), qualifier) &&
264             Arrays.equals(kv.getValue(), value) &&
265             kv.getTimestamp() == ts) {
266           return true;
267         }
268       }
269     } else if (ignoreValue && !ignoreTS) {
270       for (KeyValue kv : list) {
271         if (Arrays.equals(kv.getFamily(), family) && Arrays.equals(kv.getQualifier(), qualifier)
272             && kv.getTimestamp() == ts) {
273           return true;
274         }
275       }
276     } else if (!ignoreValue && ignoreTS) {
277       for (KeyValue kv : list) {
278         if (Arrays.equals(kv.getFamily(), family) && Arrays.equals(kv.getQualifier(), qualifier)
279             && Arrays.equals(kv.getValue(), value)) {
280           return true;
281         }
282       }
283     } else {
284       for (KeyValue kv : list) {
285         if (Arrays.equals(kv.getFamily(), family) &&
286             Arrays.equals(kv.getQualifier(), qualifier)) {
287           return true;
288         }
289       }
290     }
291     return false;
292   }

294   /**
295    * Returns a list of all KeyValue objects with matching column family and qualifier.
296    *
297    * @param family column family
298    * @param qualifier column qualifier
299    * @return a list of KeyValue objects with the matching family and qualifier,
300    * returns an empty list if one doesnt exist for the given family.
301    */
302   public List<KeyValue> get(byte[] family, byte[] qualifier) {
303     List<KeyValue> filteredList = new ArrayList<KeyValue>();
304     for (KeyValue kv: getKeyValueList(family)) {
305       if (Arrays.equals(kv.getQualifier(), qualifier)) {
306         filteredList.add(kv);
307       }
308     }
309     return filteredList;
310   }

312   /**
313    * Creates an empty list if one doesnt exist for the given column family
314    * or else it returns the associated list of KeyValue objects.
315    *
316    * @param family column family
317    * @return a list of KeyValue objects, returns an empty list if one doesnt exist.
318    */
319   private List<KeyValue> getKeyValueList(byte[] family) {
320     List<KeyValue> list = familyMap.get(family);
321     if(list == null) {
322       list = new ArrayList<KeyValue>(0);
323     }
324     return list;
325   }

327   //HeapSize
328   public long heapSize() {
329     long heapsize = OVERHEAD;
330     //Adding row
331     heapsize += ClassSize.align(ClassSize.ARRAY + this.row.length);
332 
333     //Adding map overhead
334     heapsize +=
335       ClassSize.align(this.familyMap.size() * ClassSize.MAP_ENTRY);
336     for(Map.Entry<byte [], List<KeyValue>> entry : this.familyMap.entrySet()) {
337       //Adding key overhead
338       heapsize +=
339         ClassSize.align(ClassSize.ARRAY + entry.getKey().length);
340 
341       //This part is kinds tricky since the JVM can reuse references if you
342       //store the same value, but have a good match with SizeOf at the moment
343       //Adding value overhead
344       heapsize += ClassSize.align(ClassSize.ARRAYLIST);
345       int size = entry.getValue().size();
346       heapsize += ClassSize.align(ClassSize.ARRAY +
347           size * ClassSize.REFERENCE);
348 
349       for(KeyValue kv : entry.getValue()) {
350         heapsize += kv.heapSize();
351       }
352     }
353     heapsize += getAttributeSize();
354 
355     return ClassSize.align((int)heapsize);
356   }
