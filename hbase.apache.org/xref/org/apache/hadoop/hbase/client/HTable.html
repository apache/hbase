78  /**
79   * <p>Used to communicate with a single HBase table.
80   *
81   * <p>This class is not thread safe for reads nor write.
82   *
83   * <p>In case of writes (Put, Delete), the underlying write buffer can
84   * be corrupted if multiple threads contend over a single HTable instance.
85   *
86   * <p>In case of reads, some fields used by a Scan are shared among all threads.
87   * The HTable implementation can either not contract to be safe in case of a Get
88   *
89   * <p>To access a table in a multi threaded environment, please consider
90   * using the {@link HTablePool} class to create your HTable instances.
91   *
92   * <p>Instances of HTable passed the same {@link Configuration} instance will
93   * share connections to servers out on the cluster and to the zookeeper ensemble
94   * as well as caches of region locations.  This is usually a *good* thing and it
95   * is recommended to reuse the same configuration object for all your tables.
96   * This happens because they will all share the same underlying
97   * {@link HConnection} instance. See {@link HConnectionManager} for more on
98   * how this mechanism works.
99   *
100  * <p>{@link HConnection} will read most of the
101  * configuration it needs from the passed {@link Configuration} on initial
102  * construction.  Thereafter, for settings such as
103  * <code>hbase.client.pause</code>, <code>hbase.client.retries.number</code>,
104  * and <code>hbase.client.rpc.maxattempts</code> updating their values in the
105  * passed {@link Configuration} subsequent to {@link HConnection} construction
106  * will go unnoticed.  To run with changed values, make a new
107  * {@link HTable} passing a new {@link Configuration} instance that has the
108  * new configuration.
109  *
110  * <p>Note that this class implements the {@link Closeable} interface. When a
111  * HTable instance is no longer required, it *should* be closed in order to ensure
112  * that the underlying resources are promptly released. Please note that the close
113  * method can throw java.io.IOException that must be handled.
114  *
115  * @see HBaseAdmin for create, drop, list, enable and disable of tables.
116  * @see HConnection
117  * @see HConnectionManager
118  */
119 @InterfaceAudience.Public
120 @InterfaceStability.Stable
121 public class HTable implements HTableInterface {

156   /**
157    * Creates an object to access a HBase table.
158    * Shares zookeeper connection and other resources with other HTable instances
159    * created with the same <code>conf</code> instance.  Uses already-populated
160    * region cache if one is available, populated by any other HTable instances
161    * sharing this <code>conf</code> instance.  Recommended.
162    * @param conf Configuration object to use.
163    * @param tableName Name of the table.
164    * @throws IOException if a remote or network exception occurs
165    */
166   public HTable(Configuration conf, final byte [] tableName)
167   throws IOException {
168     this.tableName = tableName;
169     this.cleanupPoolOnClose = this.cleanupConnectionOnClose = true;
170     if (conf == null) {
171       this.connection = null;
172       return;
173     }
174     this.connection = HConnectionManager.getConnection(conf);
175     this.configuration = conf;
176 
177     int maxThreads = conf.getInt("hbase.htable.threads.max", Integer.MAX_VALUE);
178     if (maxThreads == 0) {
179       maxThreads = 1; // is there a better default?
180     }
181     long keepAliveTime = conf.getLong("hbase.htable.threads.keepalivetime", 60);
182 
183     // Using the "direct handoff" approach, new threads will only be created
184     // if it is necessary and will grow unbounded. This could be bad but in HCM
185     // we only create as many Runnables as there are region servers. It means
186     // it also scales when new region servers are added.
187     this.pool = new ThreadPoolExecutor(1, maxThreads, keepAliveTime, TimeUnit.SECONDS,
188         new SynchronousQueue<Runnable>(), Threads.newDaemonThreadFactory("hbase-table"));
189     ((ThreadPoolExecutor) this.pool).allowCoreThreadTimeOut(true);
190 
191     this.finishSetup();
192   }

218   /**
219    * Creates an object to access a HBase table.
220    * Shares zookeeper connection and other resources with other HTable instances
221    * created with the same <code>connection</code> instance.
222    * Use this constructor when the ExecutorService and HConnection instance are
223    * externally managed.
224    * @param tableName Name of the table.
225    * @param connection HConnection to be used.
226    * @param pool ExecutorService to be used.
227    * @throws IOException if a remote or network exception occurs
228    */
229   public HTable(final byte[] tableName, final HConnection connection,
230       final ExecutorService pool) throws IOException {
231     if (pool == null || pool.isShutdown()) {
232       throw new IllegalArgumentException("Pool is null or shut down.");
233     }
234     if (connection == null || connection.isClosed()) {
235       throw new IllegalArgumentException("Connection is null or closed.");
236     }
237     this.tableName = tableName;
238     this.cleanupPoolOnClose = this.cleanupConnectionOnClose = false;
239     this.connection = connection;
240     this.configuration = connection.getConfiguration();
241     this.pool = pool;
242 
243     this.finishSetup();
244   }

246   /**
247    * setup this HTable's parameter based on the passed configuration
248    * @param conf
249    */
250   private void finishSetup() throws IOException {
251     this.connection.locateRegion(tableName, HConstants.EMPTY_START_ROW);
252     this.operationTimeout = HTableDescriptor.isMetaTable(tableName) ? HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT
253         : this.configuration.getInt(HConstants.HBASE_CLIENT_OPERATION_TIMEOUT,
254             HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT);
255     this.writeBufferSize = this.configuration.getLong(
256         "hbase.client.write.buffer", 2097152);
257     this.clearBufferOnFail = true;
258     this.autoFlush = true;
259     this.currentWriteBufferSize = 0;
260     this.scannerCaching = this.configuration.getInt(
261         HConstants.HBASE_CLIENT_SCANNER_CACHING,
262         HConstants.DEFAULT_HBASE_CLIENT_SCANNER_CACHING);
263 
264     this.maxKeyValueSize = this.configuration.getInt(
265         "hbase.client.keyvalue.maxsize", -1);
266     this.closed = false;
267   }

453    * Gets the starting and ending row keys for every region in the currently
454    * open table.
455    * <p>
456    * This is mainly useful for the MapReduce integration.
457    * @return Pair of arrays of region starting and ending row keys
458    * @throws IOException if a remote or network exception occurs
459    */
460   public Pair<byte[][],byte[][]> getStartEndKeys() throws IOException {
461     NavigableMap<HRegionInfo, ServerName> regions = getRegionLocations();
462     final List<byte[]> startKeyList = new ArrayList<byte[]>(regions.size());
463     final List<byte[]> endKeyList = new ArrayList<byte[]>(regions.size());
464 
465     for (HRegionInfo region : regions.keySet()) {
466       startKeyList.add(region.getStartKey());
467       endKeyList.add(region.getEndKey());
468     }
469 
470     return new Pair<byte [][], byte [][]>(
471       startKeyList.toArray(new byte[startKeyList.size()][]),
472       endKeyList.toArray(new byte[endKeyList.size()][]));
473   }

487   /**
488    * Get the corresponding regions for an arbitrary range of keys.
489    * <p>
490    * @param startKey Starting row in range, inclusive
491    * @param endKey Ending row in range, exclusive
492    * @return A list of HRegionLocations corresponding to the regions that
493    * contain the specified range
494    * @throws IOException if a remote or network exception occurs
495    */
496   public List<HRegionLocation> getRegionsInRange(final byte [] startKey,
497     final byte [] endKey) throws IOException {
498     final boolean endKeyIsEndOfTable = Bytes.equals(endKey,
499                                                     HConstants.EMPTY_END_ROW);
500     if ((Bytes.compareTo(startKey, endKey) > 0) && !endKeyIsEndOfTable) {
501       throw new IllegalArgumentException(
502         "Invalid range: " + Bytes.toStringBinary(startKey) +
503         " > " + Bytes.toStringBinary(endKey));
504     }
505     final List<HRegionLocation> regionList = new ArrayList<HRegionLocation>();
506     byte [] currentKey = startKey;
507     do {
508       HRegionLocation regionLocation = getRegionLocation(currentKey, false);
509       regionList.add(regionLocation);
510       currentKey = regionLocation.getRegionInfo().getEndKey();
511     } while (!Bytes.equals(currentKey, HConstants.EMPTY_END_ROW) &&
512              (endKeyIsEndOfTable || Bytes.compareTo(currentKey, endKey) < 0));
513     return regionList;
514   }

652   public void delete(final List<Delete> deletes)
653   throws IOException {
654     Object[] results = new Object[deletes.size()];
655     try {
656       connection.processBatch((List) deletes, tableName, pool, results);
657     } catch (InterruptedException e) {
658       throw new IOException(e);
659     } finally {
660       // mutate list so that it is empty for complete success, or contains only failed records
661       // results are returned in the same order as the requests in list
662       // walk the list backwards, so we can remove from list without impacting the indexes of earlier members
663       for (int i = results.length - 1; i>=0; i--) {
664         // if result is not null, it succeeded
665         if (results[i] instanceof Result) {
666           deletes.remove(i);
667         }
668       }
669     }
670   }

688   private void doPut(final List<Put> puts) throws IOException {
689     int n = 0;
690     for (Put put : puts) {
691       validatePut(put);
692       writeBuffer.add(put);
693       currentWriteBufferSize += put.heapSize();
694 
695       // we need to periodically see if the writebuffer is full instead of waiting until the end of the List
696       n++;
697       if (n % DOPUT_WB_CHECK == 0 && currentWriteBufferSize > writeBufferSize) {
698         flushCommits();
699       }
700     }
701     if (autoFlush || currentWriteBufferSize > writeBufferSize) {
702       flushCommits();
703     }
704   }

884   /**
885    * {@inheritDoc}
886    */
887   @Override
888   public void flushCommits() throws IOException {
889     Object[] results = new Object[writeBuffer.size()];
890     boolean success = false;
891     try {
892       this.connection.processBatch(writeBuffer, tableName, pool, results);
893       success = true;
894     } catch (InterruptedException e) {
895       throw new InterruptedIOException(e.getMessage());
896     } finally {
897       // mutate list so that it is empty for complete success, or contains
898       // only failed records. Results are returned in the same order as the
899       // requests in list. Walk the list backwards, so we can remove from list
900       // without impacting the indexes of earlier members
901       currentWriteBufferSize = 0;
902       if (success || clearBufferOnFail) {
903         writeBuffer.clear();
904       } else {
905         for (int i = results.length - 1; i >= 0; i--) {
906           if (results[i] instanceof Result) {
907             writeBuffer.remove(i);
908           } else {
909             currentWriteBufferSize += writeBuffer.get(i).heapSize();
910           }
911         }
912       }
913     }
914   }

1036   /**
1037    * Turns 'auto-flush' on or off.
1038    * <p>
1039    * When enabled (default), {@link Put} operations don't get buffered/delayed
1040    * and are immediately executed. Failed operations are not retried. This is
1041    * slower but safer.
1042    * <p>
1043    * Turning off {@link #autoFlush} means that multiple {@link Put}s will be
1044    * accepted before any RPC is actually sent to do the write operations. If the
1045    * application dies before pending writes get flushed to HBase, data will be
1046    * lost.
1047    * <p>
1048    * When you turn {@link #autoFlush} off, you should also consider the
1049    * {@link #clearBufferOnFail} option. By default, asynchronous {@link Put}
1050    * requests will be retried on failure until successful. However, this can
1051    * pollute the writeBuffer and slow down batching performance. Additionally,
1052    * you may want to issue a number of Put requests and call
1053    * {@link #flushCommits()} as a barrier. In both use cases, consider setting
1054    * clearBufferOnFail to true to erase the buffer after {@link #flushCommits()}
1055    * has been called, regardless of success.
1056    *
1057    * @param autoFlush
1058    *          Whether or not to enable 'auto-flush'.
1059    * @param clearBufferOnFail
1060    *          Whether to keep Put failures in the writeBuffer
1061    * @see #flushCommits
1062    */
1063   public void setAutoFlush(boolean autoFlush, boolean clearBufferOnFail) {
1064     this.autoFlush = autoFlush;
1065     this.clearBufferOnFail = autoFlush || clearBufferOnFail;

1283       byte[] startKey, byte[] endKey, final Batch.Call<T,R> callable,
1284       final Batch.Callback<R> callback) throws ServiceException, Throwable {
1285 
1286     // get regions covered by the row range
1287     List<byte[]> keys = getStartKeysInRange(startKey, endKey);
1288

1295           new Callable<R>() {

1298               R result = callable.call(instance);
1299               byte[] region = channel.getLastRegion();
1300               if (callback != null) {
1301                 callback.update(region, r, result);
1302               }
1303               return result;
1304             }
1305           });
1306       futures.put(r, future);
1307     }

1309       try {

1313             + Bytes.toStringBinary(e.getKey()), ee);
1314         throw ee.getCause();

1316         Thread.currentThread().interrupt();
1317         throw new InterruptedIOException("Interrupted calling coprocessor service " + service.getName()
1318             + " for row " + Bytes.toStringBinary(e.getKey()))

1324   private List<byte[]> getStartKeysInRange(byte[] start, byte[] end)
1325   throws IOException {
1326     Pair<byte[][],byte[][]> startEndKeys = getStartEndKeys();
1327     byte[][] startKeys = startEndKeys.getFirst();
1328     byte[][] endKeys = startEndKeys.getSecond();
1329 
1330     if (start == null) {
1331       start = HConstants.EMPTY_START_ROW;
1332     }
1333     if (end == null) {
1334       end = HConstants.EMPTY_END_ROW;
1335     }
1336 
1337     List<byte[]> rangeKeys = new ArrayList<byte[]>();
1338     for (int i=0; i<startKeys.length; i++) {
1339       if (Bytes.compareTo(start, startKeys[i]) >= 0 ) {
1340         if (Bytes.equals(endKeys[i], HConstants.EMPTY_END_ROW) ||
1341             Bytes.compareTo(start, endKeys[i]) < 0) {
1342           rangeKeys.add(start);
