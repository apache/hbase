77     * This test uses a tgz file named "TestMetaMigrationConvertingToPB.tgz" under
78     * hbase-server/src/test/data which contains file data from a 0.92 cluster.
79     * The cluster has a table named "TestTable", which has 100 rows. 0.94 has same
80     * META structure, so it should be the same.
81     *
82     * hbase(main):001:0> create 'TestTable', 'f1'
83     * hbase(main):002:0> for i in 1..100
84     * hbase(main):003:1> put 'TestTable', "row#{i}", "f1:c1", i
85     * hbase(main):004:1> end
86     *
87     * There are 9 regions in the table

90    @BeforeClass
91    public static void setUpBeforeClass() throws Exception {
92      // Start up our mini cluster on top of an 0.92 root.dir that has data from
93      // a 0.92 hbase run -- it has a table with 100 rows in it  -- and see if
94      // we can migrate from 0.92
95      TEST_UTIL.startMiniZKCluster();
96      TEST_UTIL.startMiniDFSCluster(1);
97      Path testdir = TEST_UTIL.getDataTestDir("TestMetaMigrationConvertToPB");
98      // Untar our test dir.
99      File untar = untar(new File(testdir.toString()));
100     // Now copy the untar up into hdfs so when we start hbase, we'll run from it.
101     Configuration conf = TEST_UTIL.getConfiguration();
102     FsShell shell = new FsShell(conf);
103     FileSystem fs = FileSystem.get(conf);
104     // find where hbase will root itself, so we can copy filesystem there
105     Path hbaseRootDir = TEST_UTIL.getDefaultRootDirPath();
106     if (!fs.isDirectory(hbaseRootDir.getParent())) {
107       // mkdir at first
108       fs.mkdirs(hbaseRootDir.getParent());
109     }
110     doFsCommand(shell,
111       new String [] {"-put", untar.toURI().toString(), hbaseRootDir.toString()});
112     // See whats in minihdfs.
113     doFsCommand(shell, new String [] {"-lsr", "/"});
114     TEST_UTIL.startMiniHBaseCluster(1, 1);
115     // Assert we are running against the copied-up filesystem.  The copied-up
116     // rootdir should have had a table named 'TestTable' in it.  Assert it
117     // present.
118     HTable t = new HTable(TEST_UTIL.getConfiguration(), TESTTABLE);
119     ResultScanner scanner = t.getScanner(new Scan());
120     int count = 0;
121     while (scanner.next() != null) {
122       count++;
123     }
124     // Assert that we find all 100 rows that are in the data we loaded.  If
125     // so then we must have migrated it from 0.90 to 0.92.
126     Assert.assertEquals(ROW_COUNT, count);
127     scanner.close();
128     t.close();
129   }

131   private static File untar(final File testdir) throws IOException {
132     // Find the src data under src/test/data
133     final String datafile = "TestMetaMigrationConvertToPB";
134     String srcTarFile =
135       System.getProperty("project.build.testSourceDirectory", "src/test") +
136       File.separator + "data" + File.separator + datafile + ".tgz";
137     File homedir = new File(testdir.toString());
138     File tgtUntarDir = new File(homedir, datafile);
139     if (tgtUntarDir.exists()) {
140       if (!FileUtil.fullyDelete(tgtUntarDir)) {
141         throw new IOException("Failed delete of " + tgtUntarDir.toString());
142       }
143     }
144     LOG.info("Untarring " + srcTarFile + " into " + homedir.toString());
145     FileUtil.unTar(new File(srcTarFile), homedir);
146     Assert.assertTrue(tgtUntarDir.exists());
147     return tgtUntarDir;
148   }

176     LOG.info("Starting testMetaMigration");
177     final byte [] FAMILY = Bytes.toBytes("family");
178     HTableDescriptor htd = new HTableDescriptor("testMetaMigration");
179     HColumnDescriptor hcd = new HColumnDescriptor(FAMILY);

182     byte[][] regionNames = new byte[][]{
183         HConstants.EMPTY_START_ROW,

186     createMultiRegionsWithWritableSerialization(conf, htd.getName(), regionNames);
187     CatalogTracker ct =

189     // Erase the current version of root meta for this test.

192     LOG.info("Meta Print completed.testMetaMigration");
193 
194     long numMigratedRows = MetaMigrationConvertingToPB.updateMeta(
195         TEST_UTIL.getHBaseCluster().getMaster());
196     MetaReader.fullScanMetaAndPrint(ct);
197 
198     // Should be one entry only and it should be for the table we just added.
199     assertEquals(regionNames.length, numMigratedRows);
200 
201     // Assert that the flag in ROOT is updated to reflect the correct status
202     boolean metaUpdated =
203         MetaMigrationConvertingToPB.isMetaHRIUpdated(
204         TEST_UTIL.getMiniHBaseCluster().getMaster().getCatalogTracker());
205     assertEquals(true, metaUpdated);

209   /**
210    * This test assumes a master crash/failure during the meta migration process
211    * and attempts to continue the meta migration process when a new master takes over.
212    * When a master dies during the meta migration we will have some rows of
213    * META.CatalogFamily updated with PB serialization and some
214    * still hanging with writable serialization. When the backup master/ or
215    * fresh start of master attempts the migration it will encounter some rows of META
216    * already updated with new HRI and some still legacy. This test will simulate this
217    * scenario and validates that the migration process can safely skip the updated
218    * rows and migrate any pending rows at startup.
219    * @throws Exception
220    */
221   @Test
222   public void testMasterCrashDuringMetaMigration() throws Exception {
223     final byte[] FAMILY = Bytes.toBytes("family");
224     HTableDescriptor htd = new HTableDescriptor("testMasterCrashDuringMetaMigration");
225     HColumnDescriptor hcd = new HColumnDescriptor(FAMILY);
226       htd.addFamily(hcd);
227     Configuration conf = TEST_UTIL.getConfiguration();
228     // Create 10 New regions.
229     createMultiRegionsWithPBSerialization(conf, htd.getName(), 10);
230     // Create 10 Legacy regions.
231     createMultiRegionsWithWritableSerialization(conf, htd.getName(), 10);
232     CatalogTracker ct =
233       TEST_UTIL.getMiniHBaseCluster().getMaster().getCatalogTracker();
234     // Erase the current version of root meta for this test.
235     undoVersionInRoot(ct);
236 
237     MetaReader.fullScanMetaAndPrint(ct);
238     LOG.info("Meta Print completed.testUpdatesOnMetaWithLegacyHRI");
239 
240     long numMigratedRows =
241         MetaMigrationConvertingToPB.updateRootAndMetaIfNecessary(
242             TEST_UTIL.getHBaseCluster().getMaster());
243     assertEquals(numMigratedRows, 10);
244 
245     // Assert that the flag in ROOT is updated to reflect the correct status
246     boolean metaUpdated = MetaMigrationConvertingToPB.
247       isMetaHRIUpdated(TEST_UTIL.getMiniHBaseCluster().getMaster().getCatalogTracker());
248     assertEquals(true, metaUpdated);
249 
250     verifyMetaRowsAreUpdated(ct);
251 
252     LOG.info("END testMasterCrashDuringMetaMigration");
253   }

255   /**
256    * Verify that every META row is updated
257    */
258   void verifyMetaRowsAreUpdated(CatalogTracker catalogTracker)
259       throws IOException {
260     List<Result> results = MetaReader.fullScan(catalogTracker);
261     assertTrue(results.size() >= REGION_COUNT);
262 
263     for (Result result : results) {
264       byte[] hriBytes = result.getValue(HConstants.CATALOG_FAMILY,
265           HConstants.REGIONINFO_QUALIFIER);
266       assertTrue(hriBytes != null && hriBytes.length > 0);
267       assertTrue(MetaMigrationConvertingToPB.isMigrated(hriBytes));
268 
269       byte[] splitA = result.getValue(HConstants.CATALOG_FAMILY,
270           HConstants.SPLITA_QUALIFIER);
271       if (splitA != null && splitA.length > 0) {
272         assertTrue(MetaMigrationConvertingToPB.isMigrated(splitA));
273       }
274 
275       byte[] splitB = result.getValue(HConstants.CATALOG_FAMILY,
276           HConstants.SPLITB_QUALIFIER);
277       if (splitB != null && splitB.length > 0) {
278         assertTrue(MetaMigrationConvertingToPB.isMigrated(splitB));
279       }
280     }
281   }

283   /** Changes the version of META to 0 to simulate 0.92 and 0.94 clusters*/
284   private void undoVersionInRoot(CatalogTracker ct) throws IOException {
285     Put p = new Put(HRegionInfo.FIRST_META_REGIONINFO.getRegionName());
286 
287     p.add(HConstants.CATALOG_FAMILY, HConstants.META_VERSION_QUALIFIER,
288         Bytes.toBytes(META_VERSION_092));
289 
290     MetaEditor.putToRootTable(ct, p);
291     LOG.info("Downgraded -ROOT- meta version=" + META_VERSION_092);
292   }

297   public int createMultiRegionsWithWritableSerialization(final Configuration c,
298       final byte[] tableName, int numRegions) throws IOException {
299     if (numRegions < 3) throw new IOException("Must create at least 3 regions");
300     byte [] startKey = Bytes.toBytes("aaaaa");
301     byte [] endKey = Bytes.toBytes("zzzzz");
302     byte [][] splitKeys = Bytes.split(startKey, endKey, numRegions - 3);
303     byte [][] regionStartKeys = new byte[splitKeys.length+1][];
304     for (int i=0;i<splitKeys.length;i++) {
305       regionStartKeys[i+1] = splitKeys[i];
306     }
307     regionStartKeys[0] = HConstants.EMPTY_BYTE_ARRAY;
308     return createMultiRegionsWithWritableSerialization(c, tableName, regionStartKeys);
309   }

311   /**
312    * Inserts multiple regions into META using Writable serialization instead of PB
313    */
314   public int createMultiRegionsWithWritableSerialization(final Configuration c,
315       final byte[] tableName, byte [][] startKeys)
316   throws IOException {
317     Arrays.sort(startKeys, Bytes.BYTES_COMPARATOR);
318     HTable meta = new HTable(c, HConstants.META_TABLE_NAME);
319 
320     List<HRegionInfo> newRegions
321         = new ArrayList<HRegionInfo>(startKeys.length);
322     int count = 0;
323     for (int i = 0; i < startKeys.length; i++) {
324       int j = (i + 1) % startKeys.length;
325       HRegionInfo hri = new HRegionInfo(tableName, startKeys[i], startKeys[j]);
326       Put put = new Put(hri.getRegionName());
327       put.setWriteToWAL(false);
328       put.add(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER,
329         getBytes(hri)); //this is the old Writable serialization
330 
331       //also add the region as it's daughters
332       put.add(HConstants.CATALOG_FAMILY, HConstants.SPLITA_QUALIFIER,
333           getBytes(hri)); //this is the old Writable serialization
334 
335       put.add(HConstants.CATALOG_FAMILY, HConstants.SPLITB_QUALIFIER,
336           getBytes(hri)); //this is the old Writable serialization
337 
338       meta.put(put);
339       LOG.info("createMultiRegionsWithWritableSerialization: PUT inserted " + hri.toString());
340 
341       newRegions.add(hri);
342       count++;
343     }
344     meta.close();
345     return count;
346   }
