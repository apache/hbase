157   private void startAndWriteData() throws IOException {
158     // When the META table can be opened, the region servers are running
159     new HTable(TEST_UTIL.getConfiguration(), HConstants.META_TABLE_NAME);
160     this.server = cluster.getRegionServerThreads().get(0).getRegionServer();
161     this.log = server.getWAL();
162 
163     // Create the test table and open it
164     HTableDescriptor desc = new HTableDescriptor(tableName);
165     desc.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
166     admin.createTable(desc);
167     HTable table = new HTable(TEST_UTIL.getConfiguration(), tableName);
168 
169     server = TEST_UTIL.getRSForFirstRegionInTable(Bytes.toBytes(tableName));
170     this.log = server.getWAL();
171     for (int i = 1; i <= 256; i++) {    // 256 writes should cause 8 log rolls
172       Put put = new Put(Bytes.toBytes("row" + String.format("%1$04d", i)));
173       put.add(HConstants.CATALOG_FAMILY, null, value);
174       table.put(put);
175       if (i % 32 == 0) {
176         // After every 32 writes sleep to let the log roller run
177         try {
178           Thread.sleep(2000);
179         } catch (InterruptedException e) {
180           // continue
181         }
182       }
183     }
184   }

187    * Tests that logs are deleted
188    * @throws IOException
189    * @throws FailedLogCloseException

191   @Test
192   public void testLogRolling() throws FailedLogCloseException, IOException {
193     this.tableName = getName();
194       startAndWriteData();
195       LOG.info("after writing there are " + ((FSHLog) log).getNumLogFiles() + " log files");
196 
197       // flush all regions
198 
199       List<HRegion> regions =
200         new ArrayList<HRegion>(server.getOnlineRegionsLocalContext());
201       for (HRegion r: regions) {
202         r.flushcache();
203       }
204 
205       // Now roll the log
206       log.rollWriter();
207 
208       int count = ((FSHLog) log).getNumLogFiles();
209       LOG.info("after flushing all regions and rolling logs there are " +
210                                       ((FSHLog) log).getNumLogFiles() + " log files");
211       assertTrue(("actual count: " + count), count <= 2);
212   }

242   void batchWriteAndWait(HTable table, int start, boolean expect, int timeout)
243       throws IOException {
244     for (int i = 0; i < 10; i++) {
245       Put put = new Put(Bytes.toBytes("row"
246           + String.format("%1$04d", (start + i))));
247       put.add(HConstants.CATALOG_FAMILY, null, value);
248       table.put(put);
249     }
250     long startTime = System.currentTimeMillis();
251     long remaining = timeout;
252     while (remaining > 0) {
253       if (log.isLowReplicationRollEnabled() == expect) {
254         break;
255       } else {
256         try {
257           Thread.sleep(200);
258         } catch (InterruptedException e) {
259           // continue
260         }
261         remaining = timeout - (System.currentTimeMillis() - startTime);
262       }
263     }
264   }

269   DatanodeInfo[] getPipeline(HLog log) throws IllegalArgumentException,
270       IllegalAccessException, InvocationTargetException {
271     OutputStream stm = ((FSHLog) log).getOutputStream();
272     Method getPipeline = null;
273     for (Method m : stm.getClass().getDeclaredMethods()) {
274       if (m.getName().endsWith("getPipeline")) {
275         getPipeline = m;
276         getPipeline.setAccessible(true);
277         break;
278       }
279     }
280 
281     assertTrue("Need DFSOutputStream.getPipeline() for this test",
282         null != getPipeline);
283     Object repl = getPipeline.invoke(stm, new Object[] {} /* NO_ARGS */);
284     return (DatanodeInfo[]) repl;
285   }

288   /**
289    * Tests that logs are rolled upon detecting datanode death
290    * Requires an HDFS jar with HDFS-826 & syncFs() support (HDFS-200)
291    * @throws IOException
292    * @throws InterruptedException
293    * @throws InvocationTargetException
294    * @throws IllegalAccessException
295    * @throws IllegalArgumentException
296     */
297   @Test
298   public void testLogRollOnDatanodeDeath() throws Exception {
299     assertTrue("This test requires HLog file replication set to 2.",
300       fs.getDefaultReplication() == 2);
301     LOG.info("Replication=" + fs.getDefaultReplication());
302 
303     this.server = cluster.getRegionServer(0);
304     this.log = server.getWAL();
305 
306     // Create the test table and open it
307     String tableName = getName();
308     HTableDescriptor desc = new HTableDescriptor(tableName);
309     desc.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
310 
311     admin.createTable(desc);
312     HTable table = new HTable(TEST_UTIL.getConfiguration(), tableName);
313     assertTrue(table.isAutoFlush());
314 
315     server = TEST_UTIL.getRSForFirstRegionInTable(Bytes.toBytes(tableName));
316     this.log = server.getWAL();
317 
318     assertTrue("Need HDFS-826 for this test", ((FSHLog) log).canGetCurReplicas());
319     // don't run this test without append support (HDFS-200 & HDFS-142)
320     assertTrue("Need append support for this test", FSUtils
321         .isAppendSupported(TEST_UTIL.getConfiguration()));
322 
323     // add up the datanode count, to ensure proper replication when we kill 1
324     // This function is synchronous; when it returns, the dfs cluster is active
325     // We start 3 servers and then stop 2 to avoid a directory naming conflict
326     //  when we stop/start a namenode later, as mentioned in HBASE-5163
327     List<DataNode> existingNodes = dfsCluster.getDataNodes();
328     int numDataNodes = 3;
329     dfsCluster.startDataNodes(TEST_UTIL.getConfiguration(), numDataNodes, true,
330         null, null);
331     List<DataNode> allNodes = dfsCluster.getDataNodes();
332     for (int i = allNodes.size()-1; i >= 0; i--) {
333       if (existingNodes.contains(allNodes.get(i))) {
334         dfsCluster.stopDataNode( i );
335       }
336     }
337 
338     assertTrue("DataNodes " + dfsCluster.getDataNodes().size() +
339         " default replication " + fs.getDefaultReplication(),
340       dfsCluster.getDataNodes().size() >= fs.getDefaultReplication() + 1);
341 
342     writeData(table, 2);
343 
344     long curTime = System.currentTimeMillis();
345     long oldFilenum = ((FSHLog) log).getFilenum();
346     assertTrue("Log should have a timestamp older than now",
347         curTime > oldFilenum && oldFilenum != -1);
348 
349     assertTrue("The log shouldn't have rolled yet",
350       oldFilenum == ((FSHLog) log).getFilenum());
351     final DatanodeInfo[] pipeline = getPipeline(log);
352     assertTrue(pipeline.length == fs.getDefaultReplication());
353 
354     // kill a datanode in the pipeline to force a log roll on the next sync()
355     // This function is synchronous, when it returns the node is killed.
356     assertTrue(dfsCluster.stopDataNode(pipeline[0].getName()) != null);
357 
358     // this write should succeed, but trigger a log roll
359     writeData(table, 2);
360     long newFilenum = ((FSHLog) log).getFilenum();
361 
362     assertTrue("Missing datanode should've triggered a log roll",
363         newFilenum > oldFilenum && newFilenum > curTime);
364 
365     // write some more log data (this should use a new hdfs_out)
366     writeData(table, 3);
367     assertTrue("The log should not roll again.",
368       ((FSHLog) log).getFilenum() == newFilenum);
369     // kill another datanode in the pipeline, so the replicas will be lower than
370     // the configured value 2.
371     assertTrue(dfsCluster.stopDataNode(pipeline[1].getName()) != null);
372 
373     batchWriteAndWait(table, 3, false, 10000);
374     assertTrue("LowReplication Roller should've been disabled",
375         !log.isLowReplicationRollEnabled());
376 
377     dfsCluster
378         .startDataNodes(TEST_UTIL.getConfiguration(), 1, true, null, null);
379 
380     // Force roll writer. The new log file will have the default replications,
381     // and the LowReplication Roller will be enabled.
382     log.rollWriter(true);
383     batchWriteAndWait(table, 13, true, 10000);
384     assertTrue("New log file should have the default replication instead of " +
385       ((FSHLog) log).getLogReplication(),
386       ((FSHLog) log).getLogReplication() == fs.getDefaultReplication());
387     assertTrue("LowReplication Roller should've been enabled",
388         log.isLowReplicationRollEnabled());
389   }

448     // don't run this test without append support (HDFS-200 & HDFS-142)

540     }
