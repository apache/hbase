32  /**
33   * Used to perform Increment operations on a single row.
34   * <p>
35   * This operation does not appear atomic to readers.  Increments are done
36   * under a single row lock, so write operations to a row are synchronized, but
37   * readers do not take row locks so get and scan operations can see this
38   * operation partially completed.
39   * <p>
40   * To increment columns of a row, instantiate an Increment object with the row
41   * to increment.  At least one column to increment must be specified using the
42   * {@link #addColumn(byte[], byte[], long)} method.
43   */
44  @InterfaceAudience.Public
45  @InterfaceStability.Stable
46  public class Increment implements Row {
47    private byte [] row = null;
48    private long lockId = -1L;
49    private boolean writeToWAL = true;
50    private TimeRange tr = new TimeRange();
51    private Map<byte [], NavigableMap<byte [], Long>> familyMap =
52      new TreeMap<byte [], NavigableMap<byte [], Long>>(Bytes.BYTES_COMPARATOR);
53  
54    /** Constructor for Writable.  DO NOT USE */
55    public Increment() {}
56  
57    /**
58     * Create a Increment operation for the specified row.
59     * <p>
60     * At least one column must be incremented.
61     * @param row row key
62     */
63    public Increment(byte [] row) {
64      this(row, null);
65    }
66  
67    /**
68     * Create a Increment operation for the specified row, using an existing row
69     * lock.
70     * <p>
71     * At least one column must be incremented.
72     * @param row row key
73     * @param rowLock previously acquired row lock, or null
74     */
75    public Increment(byte [] row, RowLock rowLock) {
76      if (row == null) {
77        throw new IllegalArgumentException("Cannot increment a null row");
78      }
79      this.row = row;
80      if(rowLock != null) {
81        this.lockId = rowLock.getLockId();
82      }
83    }
84  
85    /**
86     * Increment the column from the specific family with the specified qualifier
87     * by the specified amount.
88     * <p>
89     * Overrides previous calls to addColumn for this family and qualifier.
90     * @param family family name
91     * @param qualifier column qualifier
92     * @param amount amount to increment by
93     * @return the Increment object
94     */
95    public Increment addColumn(byte [] family, byte [] qualifier, long amount) {
96      if (family == null) {
97        throw new IllegalArgumentException("family cannot be null");
98      }
99      if (qualifier == null) {
100       throw new IllegalArgumentException("qualifier cannot be null");
101     }
102     NavigableMap<byte [], Long> set = familyMap.get(family);
103     if(set == null) {
104       set = new TreeMap<byte [], Long>(Bytes.BYTES_COMPARATOR);
105     }
106     set.put(qualifier, amount);
107     familyMap.put(family, set);
108     return this;
109   }

199   /**
200    * Method for retrieving the number of columns to increment

203   public int numColumns() {
204     if (!hasFamilies()) return 0;
205     int num = 0;
206     for (NavigableMap<byte [], Long> family : familyMap.values()) {
207       num += family.size();
208     }
209     return num;

228   /**
229    * @return String
230    */
231   @Override
232   public String toString() {
233     StringBuilder sb = new StringBuilder();
234     sb.append("row=");
235     sb.append(Bytes.toStringBinary(this.row));
236     if(this.familyMap.size() == 0) {
237       sb.append(", no columns set to be incremented");
238       return sb.toString();
239     }
240     sb.append(", families=");
241     boolean moreThanOne = false;
242     for(Map.Entry<byte [], NavigableMap<byte[], Long>> entry :
243       this.familyMap.entrySet()) {
244       if(moreThanOne) {
245         sb.append("), ");
246       } else {
247         moreThanOne = true;
248         sb.append("{");
249       }
250       sb.append("(family=");
251       sb.append(Bytes.toString(entry.getKey()));
252       sb.append(", columns=");
253       if(entry.getValue() == null) {
254         sb.append("NONE");
255       } else {
256         sb.append("{");
257         boolean moreThanOneB = false;
258         for(Map.Entry<byte [], Long> column : entry.getValue().entrySet()) {
259           if(moreThanOneB) {
260             sb.append(", ");
261           } else {
262             moreThanOneB = true;
263           }
264           sb.append(Bytes.toStringBinary(column.getKey()) + "+=" + column.getValue());
265         }
266         sb.append("}");
267       }
268     }
269     sb.append("}");
270     return sb.toString();
271   }
