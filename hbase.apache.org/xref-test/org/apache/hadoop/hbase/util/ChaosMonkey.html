54  /**
55   * A utility to injects faults in a running cluster.
56   * <p>
57   * ChaosMonkey defines Action's and Policy's. Actions are sequences of events, like
58   *  - Select a random server to kill
59   *  - Sleep for 5 sec
60   *  - Start the server on the same host
61   * Actions can also be complex events, like rolling restart of all of the servers.
62   * <p>
63   * Policies on the other hand are responsible for executing the actions based on a strategy.
64   * The default policy is to execute a random action every minute based on predefined action
65   * weights. ChaosMonkey executes predefined named policies until it is stopped. More than one
66   * policy can be active at any time.
67   * <p>
68   * Chaos monkey can be run from the command line, or can be invoked from integration tests.
69   * See {@link IntegrationTestDataIngestWithChaosMonkey} or other integration tests that use
70   * chaos monkey for code examples.
71   * <p>
72   * ChaosMonkey class is indeed inspired by the Netflix's same-named tool:
73   * http://techblog.netflix.com/2012/07/chaos-monkey-released-into-wild.html
74   */
75  public class ChaosMonkey extends AbstractHBaseTool implements Stoppable {

135   public static class Action {
136     // TODO: interesting question - should actions be implemented inside
137     //       ChaosMonkey, or outside? If they are inside (initial), the class becomes
138     //       huge and all-encompassing; if they are outside ChaosMonkey becomes just
139     //       a random task scheduler. For now, keep inside.
140 
141     protected ActionContext context;
142     protected HBaseCluster cluster;
143     protected ClusterStatus initialStatus;
144     protected ServerName[] initialServers;
145     protected Random random = new Random();
146 
147     void init(ActionContext context) throws Exception {
148       this.context = context;
149       cluster = context.getHBaseCluster();
150       initialStatus = cluster.getInitialClusterStatus();
151       Collection<ServerName> regionServers = initialStatus.getServers();
152       initialServers = regionServers.toArray(new ServerName[regionServers.size()]);
153     }
154 
155     protected void perform() throws Exception { };
156 
157     // TODO: perhaps these methods should be elsewhere?
158     /** Returns current region servers */
159     protected ServerName[] getCurrentServers() throws IOException {
160       Collection<ServerName> regionServers = cluster.getClusterStatus().getServers();
161       return regionServers.toArray(new ServerName[regionServers.size()]);
162     }

164     protected void killMaster(ServerName server) throws IOException {
165       LOG.info("Killing master:" + server);
166       cluster.killMaster(server);
167       cluster.waitForMasterToStop(server, TIMEOUT);
168       LOG.info("Killed master server:" + server);
169     }
170 
171     protected void startMaster(ServerName server) throws IOException {
172       LOG.info("Starting master:" + server.getHostname());
173       cluster.startMaster(server.getHostname());
174       cluster.waitForActiveAndReadyMaster(TIMEOUT);
175       LOG.info("Started master: " + server);
176     }
177 
178     protected void killRs(ServerName server) throws IOException {
179       LOG.info("Killing region server:" + server);
180       cluster.killRegionServer(server);
181       cluster.waitForRegionServerToStop(server, TIMEOUT);
182       LOG.info("Killed region server:" + server + ". Reported num of rs:"
183           + cluster.getClusterStatus().getServersSize());
184     }
185 
186     protected void startRs(ServerName server) throws IOException {
187       LOG.info("Starting region server:" + server.getHostname());
188       cluster.startRegionServer(server.getHostname());
189       cluster.waitForRegionServerToStart(server.getHostname(), TIMEOUT);
190       LOG.info("Started region server:" + server + ". Reported num of rs:"
191           + cluster.getClusterStatus().getServersSize());
192     }
193 
194     protected void unbalanceRegions(ClusterStatus clusterStatus,
195         List<ServerName> fromServers, List<ServerName> toServers,
196         double fractionOfRegions) throws Exception {
197       List<byte[]> victimRegions = new LinkedList<byte[]>();
198       for (ServerName server : fromServers) {
199         ServerLoad serverLoad = clusterStatus.getLoad(server);
200         // Ugh.
201         List<byte[]> regions = new LinkedList<byte[]>(serverLoad.getRegionsLoad().keySet());
202         int victimRegionCount = (int)Math.ceil(fractionOfRegions * regions.size());
203         LOG.debug("Removing " + victimRegionCount + " regions from " + server.getServerName());
204         for (int i = 0; i < victimRegionCount; ++i) {
205           int victimIx = random.nextInt(regions.size());
206           String regionId = HRegionInfo.encodeRegionName(regions.remove(victimIx));
207           victimRegions.add(Bytes.toBytes(regionId));
208         }
209       }
210 
211       LOG.info("Moving " + victimRegions.size() + " regions from " + fromServers.size()
212           + " servers to " + toServers.size() + " different servers");
213       HBaseAdmin admin = this.context.getHaseIntegrationTestingUtility().getHBaseAdmin();
214       for (byte[] victimRegion : victimRegions) {
215         int targetIx = random.nextInt(toServers.size());
216         admin.move(victimRegion, Bytes.toBytes(toServers.get(targetIx).getServerName()));
217       }
218     }
219 
220     protected void forceBalancer() throws Exception {
221       HBaseAdmin admin = this.context.getHaseIntegrationTestingUtility().getHBaseAdmin();
222       boolean result = admin.balancer();
223       if (!result) {
224         LOG.error("Balancer didn't succeed");
225       }
226     }
227   }

314    * Restarts a ratio of the running regionservers at the same time
315    */
316   public static class BatchRestartRs extends RestartActionBase {
317     float ratio; //ratio of regionservers to restart
318 
319     public BatchRestartRs(long sleepTime, float ratio) {
320       super(sleepTime);
321       this.ratio = ratio;
322     }
323 
324     @Override
325     protected void perform() throws Exception {
326       LOG.info(String.format("Performing action: Batch restarting %d%% of region servers",
327           (int)(ratio * 100)));
328       List<ServerName> selectedServers = selectRandomItems(getCurrentServers(), ratio);
329 
330       for (ServerName server : selectedServers) {
331         LOG.info("Killing region server:" + server);
332         cluster.killRegionServer(server);
333       }
334 
335       for (ServerName server : selectedServers) {
336         cluster.waitForRegionServerToStop(server, TIMEOUT);
337       }
338 
339       LOG.info("Killed " + selectedServers.size() + " region servers. Reported num of rs:"
340           + cluster.getClusterStatus().getServersSize());
341 
342       sleep(sleepTime);
343 
344       for (ServerName server : selectedServers) {
345         LOG.info("Starting region server:" + server.getHostname());
346         cluster.startRegionServer(server.getHostname());
347 
348       }
349       for (ServerName server : selectedServers) {
350         cluster.waitForRegionServerToStart(server.getHostname(), TIMEOUT);
351       }
352       LOG.info("Started " + selectedServers.size() +" region servers. Reported num of rs:"
353           + cluster.getClusterStatus().getServersSize());
354     }

357   /**
358    * Restarts a ratio of the regionservers in a rolling fashion. At each step, either kills a
359    * server, or starts one, sleeping randomly (0-sleepTime) in between steps.
360    */
361   public static class RollingBatchRestartRs extends BatchRestartRs {
362     public RollingBatchRestartRs(long sleepTime, float ratio) {
363       super(sleepTime, ratio);
364     }
365 
366     @Override
367     protected void perform() throws Exception {
368       LOG.info(String.format("Performing action: Rolling batch restarting %d%% of region servers",
369           (int)(ratio * 100)));
370       List<ServerName> selectedServers = selectRandomItems(getCurrentServers(), ratio);
371 
372       Queue<ServerName> serversToBeKilled = new LinkedList<ServerName>(selectedServers);
373       Queue<ServerName> deadServers = new LinkedList<ServerName>();
374 
375       //
376       while (!serversToBeKilled.isEmpty() || !deadServers.isEmpty()) {
377         boolean action = true; //action true = kill server, false = start server
378 
379         if (serversToBeKilled.isEmpty() || deadServers.isEmpty()) {
380           action = deadServers.isEmpty();
381         } else {
382           action = random.nextBoolean();
383         }
384 
385         if (action) {
386           ServerName server = serversToBeKilled.remove();
387           killRs(server);
388           deadServers.add(server);
389         } else {
390           ServerName server = deadServers.remove();
391           startRs(server);
392         }
393 
394         sleep(random.nextInt((int)sleepTime));
395       }
396     }

400     private double fractionOfRegions;
401     private double fractionOfServers;
402 
403     /**
404      * Unbalances the regions on the cluster by choosing "target" servers, and moving
405      * some regions from each of the non-target servers to random target servers.
406      * @param fractionOfRegions Fraction of regions to move from each server.
407      * @param fractionOfServers Fraction of servers to be chosen as targets.
408      */
409     public UnbalanceRegionsAction(double fractionOfRegions, double fractionOfServers) {
410       this.fractionOfRegions = fractionOfRegions;
411       this.fractionOfServers = fractionOfServers;
412     }
413 
414     @Override
415     protected void perform() throws Exception {
416       LOG.info("Unbalancing regions");
417       ClusterStatus status = this.cluster.getClusterStatus();
418       List<ServerName> victimServers = new LinkedList<ServerName>(status.getServers());
419       int targetServerCount = (int)Math.ceil(fractionOfServers * victimServers.size());
420       List<ServerName> targetServers = new ArrayList<ServerName>(targetServerCount);
421       for (int i = 0; i < targetServerCount; ++i) {
422         int victimIx = random.nextInt(victimServers.size());
423         targetServers.add(victimServers.remove(victimIx));
424       }
425       unbalanceRegions(status, victimServers, targetServers, fractionOfRegions);
426     }
427   }

456   /**
457    * A policy, which picks a random action according to the given weights,
458    * and performs it every configurable period.
459    */
460   public static class PeriodicRandomActionPolicy extends Policy {
461     private long periodMs;
462     private List<Pair<Action, Integer>> actions;
463 
464     public PeriodicRandomActionPolicy(long periodMs, List<Pair<Action, Integer>> actions) {
465       this.periodMs = periodMs;
466       this.actions = actions;
467     }
468 
469     public PeriodicRandomActionPolicy(long periodMs, Pair<Action, Integer>... actions) {
470       // We don't expect it to be modified.
471       this(periodMs, Arrays.asList(actions));
472     }
473 
474     public PeriodicRandomActionPolicy(long periodMs, Action... actions) {
475       this.periodMs = periodMs;
476       this.actions = new ArrayList<Pair<Action, Integer>>(actions.length);
477       for (Action action : actions) {
478         this.actions.add(new Pair<Action, Integer>(action, 1));
479       }
480     }
481 
482     @Override
483     public void run() {
484       //add some jitter
485       int jitter = new Random().nextInt((int)periodMs);
486       LOG.info("Sleeping for " + jitter + " to add jitter");
487       Threads.sleep(jitter);
488 
489       while (!isStopped()) {
490         long start = System.currentTimeMillis();
491         Action action = selectWeightedRandomItem(actions);
492 
493         try {
494           action.perform();
495         } catch (Exception ex) {
496           LOG.warn("Exception occured during performing action: "
497               + StringUtils.stringifyException(ex));
498         }
499 
500         long sleepTime = periodMs - (System.currentTimeMillis() - start);
501         if (sleepTime > 0) {
502           LOG.info("Sleeping for:" + sleepTime);
503           Threads.sleep(sleepTime);
504         }
505       }
506     }
507 
508     @Override
509     public void init(PolicyContext context) throws Exception {
510       super.init(context);
511       LOG.info("Using ChaosMonkey Policy: " + this.getClass() + ", period:" + periodMs);
512       for (Pair<Action, Integer> action : actions) {
513         action.getFirst().init(this.context);
514       }
515     }
516   }

518   /** Selects a random item from the given items */
519   static <T> T selectRandomItem(T[] items) {
520     Random random = new Random();
521     return items[random.nextInt(items.length)];
522   }
523 
524   /** Selects a random item from the given items with weights*/
525   static <T> T selectWeightedRandomItem(List<Pair<T, Integer>> items) {
526     Random random = new Random();
527     int totalWeight = 0;
528     for (Pair<T, Integer> pair : items) {
529       totalWeight += pair.getSecond();
530     }
531 
532     int cutoff = random.nextInt(totalWeight);
533     int cummulative = 0;
534     T item = null;
535 
536     //warn: O(n)
537     for (int i=0; i<items.size(); i++) {
538       int curWeight = items.get(i).getSecond();
539       if ( cutoff < cummulative + curWeight) {
540         item = items.get(i).getFirst();
541         break;
542       }
543       cummulative += curWeight;
544     }
545 
546     return item;
547   }
548 
549   /** Selects and returns ceil(ratio * items.length) random items from the given array */
550   static <T> List<T> selectRandomItems(T[] items, float ratio) {
551     Random random = new Random();
552     int remaining = (int)Math.ceil(items.length * ratio);
553 
554     List<T> selectedItems = new ArrayList<T>(remaining);
555 
556     for (int i=0; i<items.length && remaining > 0; i++) {
557       if (random.nextFloat() < ((float)remaining/(items.length-i))) {
558         selectedItems.add(items[i]);
559         remaining--;
560       }
561     }
562 
563     return selectedItems;
564   }

589   private Policy[] policies;

592   public void start() throws Exception {
593     monkeyThreads = new Thread[policies.length];
594 
595     for (int i=0; i<policies.length; i++) {
596       policies[i].init(new PolicyContext(this.util));
597       Thread monkeyThread = new Thread(policies[i]);
598       monkeyThread.start();
599       monkeyThreads[i] = monkeyThread;
600     }
601   }

603   @Override
604   public void stop(String why) {
605     for (Policy policy : policies) {
606       policy.stop(why);
607     }
608   }

612     return policies[0].isStopped();
