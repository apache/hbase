64  /**
65   * Class that handles the source of a replication stream.
66   * Currently does not handle more than 1 slave
67   * For each slave cluster it selects a random number of peers
68   * using a replication ratio. For example, if replication ration = 0.1
69   * and slave cluster has 100 region servers, 10 will be selected.
70   * <p/>
71   * A stream is considered down when we cannot contact a region server on the
72   * peer cluster for more than 55 seconds by default.
73   * <p/>
74   *
75   */
76  @InterfaceAudience.Private
77  public class ReplicationSource extends Thread
78      implements ReplicationSourceInterface {

142   /**
143    * Instantiation method used by region servers
144    *
145    * @param conf configuration to use
146    * @param fs file system to use
147    * @param manager replication manager to ping to
148    * @param stopper     the atomic boolean to use to stop the regionserver
149    * @param replicating the atomic boolean that starts/stops replication
150    * @param peerClusterZnode the name of our znode
151    * @throws IOException
152    */
153   public void init(final Configuration conf,
154                    final FileSystem fs,
155                    final ReplicationSourceManager manager,
156                    final Stoppable stopper,
157                    final AtomicBoolean replicating,
158                    final String peerClusterZnode)
159       throws IOException {
160     this.stopper = stopper;
161     this.conf = conf;
162     this.replicationQueueSizeCapacity =
163         this.conf.getLong("replication.source.size.capacity", 1024*1024*64);
164     this.replicationQueueNbCapacity =
165         this.conf.getInt("replication.source.nb.capacity", 25000);
166     this.entriesArray = new HLog.Entry[this.replicationQueueNbCapacity];
167     for (int i = 0; i < this.replicationQueueNbCapacity; i++) {
168       this.entriesArray[i] = new HLog.Entry();
169     }
170     this.maxRetriesMultiplier =
171         this.conf.getInt("replication.source.maxretriesmultiplier", 10);
172     this.socketTimeoutMultiplier = maxRetriesMultiplier * maxRetriesMultiplier;
173     this.queue =
174         new PriorityBlockingQueue<Path>(
175             conf.getInt("hbase.regionserver.maxlogs", 32),
176             new LogsComparator());
177     this.conn = HConnectionManager.getConnection(conf);
178     this.zkHelper = manager.getRepZkWrapper();
179     this.ratio = this.conf.getFloat("replication.source.ratio", 0.1f);
180     this.currentPeers = new ArrayList<ServerName>();
181     this.random = new Random();
182     this.replicating = replicating;
183     this.manager = manager;
184     this.sleepForRetries =
185         this.conf.getLong("replication.source.sleepforretries", 1000);
186     this.fs = fs;
187     this.metrics = new MetricsSource(peerClusterZnode);
188 
189     try {
190       this.clusterId = zkHelper.getUUIDForCluster(zkHelper.getZookeeperWatcher());
191     } catch (KeeperException ke) {
192       throw new IOException("Could not read cluster id", ke);
193     }
194 
195     // Finally look if this is a recovered queue
196     this.checkIfQueueRecovered(peerClusterZnode);
197   }

199   // The passed znode will be either the id of the peer cluster or
200   // the handling story of that queue in the form of id-servername-*
201   private void checkIfQueueRecovered(String peerClusterZnode) {
202     String[] parts = peerClusterZnode.split("-");
203     this.queueRecovered = parts.length != 1;
204     this.peerId = this.queueRecovered ?
205         parts[0] : peerClusterZnode;
206     this.peerClusterZnode = peerClusterZnode;
207     this.deadRegionServers = new String[parts.length-1];
208     // Extract all the places where we could find the hlogs
209     for (int i = 1; i < parts.length; i++) {
210       this.deadRegionServers[i-1] = parts[i];
211     }
212   }

214   /**
215    * Select a number of peers at random using the ratio. Mininum 1.
216    */
217   private void chooseSinks() {
218     this.currentPeers.clear();
219     List<ServerName> addresses = this.zkHelper.getSlavesAddresses(peerId);
220     Set<ServerName> setOfAddr = new HashSet<ServerName>();
221     int nbPeers = (int) (Math.ceil(addresses.size() * ratio));
222     LOG.info("Getting " + nbPeers +
223         " rs from peer cluster # " + peerId);
224     for (int i = 0; i < nbPeers; i++) {
225       ServerName sn;
226       // Make sure we get one address that we don't already have
227       do {
228         sn = addresses.get(this.random.nextInt(addresses.size()));
229       } while (setOfAddr.contains(sn));
230       LOG.info("Choosing peer " + sn);
231       setOfAddr.add(sn);
232     }
233     this.currentPeers.addAll(setOfAddr);
234   }

239     this.metrics.setSizeOfLogQueue(queue.size());

242   @Override
243   public void run() {
244     connectToPeers();
245     // We were stopped while looping to connect to sinks, just abort
246     if (!this.isActive()) {
247       metrics.clear();
248       return;
249     }
250     int sleepMultiplier = 1;
251     // delay this until we are in an asynchronous thread
252     while (this.peerClusterId == null) {
253       this.peerClusterId = zkHelper.getPeerUUID(this.peerId);
254       if (this.peerClusterId == null) {
255         if (sleepForRetries("Cannot contact the peer's zk ensemble", sleepMultiplier)) {
256           sleepMultiplier++;
257         }
258       }
259     }
260     // resetting to 1 to reuse later
261     sleepMultiplier = 1;
262 
263     LOG.info("Replicating "+clusterId + " -> " + peerClusterId);
264 
265     // If this is recovered, the queue is already full and the first log
266     // normally has a position (unless the RS failed between 2 logs)
267     if (this.queueRecovered) {
268       try {
269         this.position = this.zkHelper.getHLogRepPosition(
270             this.peerClusterZnode, this.queue.peek().getName());
271       } catch (KeeperException e) {
272         this.terminate("Couldn't get the position of this recovered queue " +
273             peerClusterZnode, e);
274       }
275     }
276     // Loop until we close down
277     while (isActive()) {
278       // Sleep until replication is enabled again
279       if (!isPeerEnabled()) {
280         if (sleepForRetries("Replication is disabled", sleepMultiplier)) {
281           sleepMultiplier++;
282         }
283         continue;
284       }
285       Path oldPath = getCurrentPath(); //note that in the current scenario,
286                                        //oldPath will be null when a log roll
287                                        //happens.
288       // Get a new path
289       boolean hasCurrentPath = getNextPath();
290       if (getCurrentPath() != null && oldPath == null) {
291         sleepMultiplier = 1; //reset the sleepMultiplier on a path change
292       }
293       if (!hasCurrentPath) {
294         if (sleepForRetries("No log to process", sleepMultiplier)) {
295           sleepMultiplier++;
296         }
297         continue;
298       }
299       boolean currentWALisBeingWrittenTo = false;
300       //For WAL files we own (rather than recovered), take a snapshot of whether the
301       //current WAL file (this.currentPath) is in use (for writing) NOW!
302       //Since the new WAL paths are enqueued only after the prev WAL file
303       //is 'closed', presence of an element in the queue means that
304       //the previous WAL file was closed, else the file is in use (currentPath)
305       //We take the snapshot now so that we are protected against races
306       //where a new file gets enqueued while the current file is being processed
307       //(and where we just finished reading the current file).
308       if (!this.queueRecovered && queue.size() == 0) {
309         currentWALisBeingWrittenTo = true;
310       }
311       // Open a reader on it
312       if (!openReader(sleepMultiplier)) {
313         // Reset the sleep multiplier, else it'd be reused for the next file
314         sleepMultiplier = 1;
315         continue;
316       }
317 
318       // If we got a null reader but didn't continue, then sleep and continue
319       if (this.reader == null) {
320         if (sleepForRetries("Unable to open a reader", sleepMultiplier)) {
321           sleepMultiplier++;
322         }
323         continue;
324       }
325 
326       boolean gotIOE = false;
327       currentNbEntries = 0;
328       try {
329         if (readAllEntriesToReplicateOrNextFile(currentWALisBeingWrittenTo)) {
330           continue;
331         }
332       } catch (IOException ioe) {
333         LOG.warn(peerClusterZnode + " Got: ", ioe);
334         gotIOE = true;
335         if (ioe.getCause() instanceof EOFException) {
336 
337           boolean considerDumping = false;
338           if (this.queueRecovered) {
339             try {
340               FileStatus stat = this.fs.getFileStatus(this.currentPath);
341               if (stat.getLen() == 0) {
342                 LOG.warn(peerClusterZnode + " Got EOF and the file was empty");
343               }
344               considerDumping = true;
345             } catch (IOException e) {
346               LOG.warn(peerClusterZnode + " Got while getting file size: ", e);
347             }
348           } else if (currentNbEntries != 0) {
349             LOG.warn(peerClusterZnode + " Got EOF while reading, " +
350                 "looks like this file is broken? " + currentPath);
351             considerDumping = true;
352             currentNbEntries = 0;
353           }
354 
355           if (considerDumping &&
356               sleepMultiplier == this.maxRetriesMultiplier &&
357               processEndOfFile()) {
358             continue;
359           }
360         }
361       } finally {
362         try {
363           if (this.reader != null) {
364             this.reader.close();
365           }
366         } catch (IOException e) {
367           gotIOE = true;
368           LOG.warn("Unable to finalize the tailing of a file", e);
369         }
370       }
371 
372       // If we didn't get anything to replicate, or if we hit a IOE,
373       // wait a bit and retry.
374       // But if we need to stop, don't bother sleeping
375       if (this.isActive() && (gotIOE || currentNbEntries == 0)) {
376         if (this.lastLoggedPosition != this.position) {
377           this.manager.logPositionAndCleanOldLogs(this.currentPath,
378               this.peerClusterZnode, this.position, queueRecovered, currentWALisBeingWrittenTo);
379           this.lastLoggedPosition = this.position;
380         }
381         if (sleepForRetries("Nothing to replicate", sleepMultiplier)) {
382           sleepMultiplier++;
383         }
384         continue;
385       }
386       sleepMultiplier = 1;
387       shipEdits(currentWALisBeingWrittenTo);
388 
389     }
390     if (this.conn != null) {
391       try {
392         this.conn.close();
393       } catch (IOException e) {
394         LOG.debug("Attempt to close connection failed", e);
395       }
396     }
397     LOG.debug("Source exiting " + peerId);
398     metrics.clear();
399   }

401   /**
402    * Read all the entries from the current log files and retain those
403    * that need to be replicated. Else, process the end of the current file.
404    * @param currentWALisBeingWrittenTo is the current WAL being written to
405    * @return true if we got nothing and went to the next file, false if we got
406    * entries
407    * @throws IOException
408    */
409   protected boolean readAllEntriesToReplicateOrNextFile(boolean currentWALisBeingWrittenTo)
410       throws IOException{
411     long seenEntries = 0;
412     if (this.position != 0) {
413       this.reader.seek(this.position);
414     }
415     long startPosition = this.position;
416     HLog.Entry entry = readNextAndSetPosition(); 
417     while (entry != null) {
418       WALEdit edit = entry.getEdit();
419       this.metrics.incrLogEditsRead();
420       seenEntries++;
421       // Remove all KVs that should not be replicated
422       HLogKey logKey = entry.getKey();
423       // don't replicate if the log entries originated in the peer
424       if (!logKey.getClusterId().equals(peerClusterId)) {
425         removeNonReplicableEdits(edit);
426         // Don't replicate catalog entries, if the WALEdit wasn't
427         // containing anything to replicate and if we're currently not set to replicate
428         if (!(Bytes.equals(logKey.getTablename(), HConstants.ROOT_TABLE_NAME) ||
429             Bytes.equals(logKey.getTablename(), HConstants.META_TABLE_NAME)) &&
430             edit.size() != 0 && replicating.get()) {
431           // Only set the clusterId if is a local key.
432           // This ensures that the originator sets the cluster id
433           // and all replicas retain the initial cluster id.
434           // This is *only* place where a cluster id other than the default is set.
435           if (HConstants.DEFAULT_CLUSTER_ID == logKey.getClusterId()) {
436             logKey.setClusterId(this.clusterId);
437           }
438           currentNbOperations += countDistinctRowKeys(edit);
439           currentNbEntries++;
440         } else {
441           this.metrics.incrLogEditsFiltered();
442         }
443       }
444       // Stop if too many entries or too big
445       if ((this.reader.getPosition() - startPosition)
446           >= this.replicationQueueSizeCapacity ||
447           currentNbEntries >= this.replicationQueueNbCapacity) {
448         break;
449       }
450       try {
451         entry = readNextAndSetPosition();
452       } catch (IOException ie) {
453         LOG.debug("Break on IOE: " + ie.getMessage());
454         break;
455       }
456     }
457     if (currentWALisBeingWrittenTo) {
458       return false;
459     }
460     // If we didn't get anything and the queue has an object, it means we
461     // hit the end of the file for sure
462     return seenEntries == 0 && processEndOfFile();
463   }
464 
465   private HLog.Entry readNextAndSetPosition() throws IOException {
466     HLog.Entry entry = this.reader.next(entriesArray[currentNbEntries]);
467     // Store the position so that in the future the reader can start
468     // reading from here. If the above call to next() throws an
469     // exception, the position won't be changed and retry will happen
470     // from the last known good position
471     this.position = this.reader.getPosition();
472     return entry;
473   }

475   private void connectToPeers() {
476     // Connect to peer cluster first, unless we have to stop
477     while (this.isActive() && this.currentPeers.size() == 0) {
478 
479       try {
480         chooseSinks();
481         Thread.sleep(this.sleepForRetries);
482       } catch (InterruptedException e) {
483         LOG.error("Interrupted while trying to connect to sinks", e);
484       }
485     }
486   }

488   /**
489    * Poll for the next path
490    * @return true if a path was obtained, false if not
491    */
492   protected boolean getNextPath() {
493     try {
494       if (this.currentPath == null) {
495         this.currentPath = queue.poll(this.sleepForRetries, TimeUnit.MILLISECONDS);
496         this.metrics.setSizeOfLogQueue(queue.size());
497       }
498     } catch (InterruptedException e) {
499       LOG.warn("Interrupted while reading edits", e);
500     }
501     return this.currentPath != null;
502   }

504   /**
505    * Open a reader on the current path
506    *
507    * @param sleepMultiplier by how many times the default sleeping time is augmented
508    * @return true if we should continue with that file, false if we are over with it
509    */
510   protected boolean openReader(int sleepMultiplier) {
511     try {
512       try {
513        this.reader = null;
514        this.reader = HLogFactory.createReader(this.fs, 
515            this.currentPath, this.conf);
516       } catch (FileNotFoundException fnfe) {
517         if (this.queueRecovered) {
518           // We didn't find the log in the archive directory, look if it still
519           // exists in the dead RS folder (there could be a chain of failures
520           // to look at)
521           LOG.info("NB dead servers : " + deadRegionServers.length);
522           for (int i = this.deadRegionServers.length - 1; i >= 0; i--) {
523 
524             Path deadRsDirectory =
525                 new Path(manager.getLogDir().getParent(), this.deadRegionServers[i]);
526             Path[] locs = new Path[] {
527                 new Path(deadRsDirectory, currentPath.getName()),
528                 new Path(deadRsDirectory.suffix(HLog.SPLITTING_EXT),
529                                           currentPath.getName()),
530             };
531             for (Path possibleLogLocation : locs) {
532               LOG.info("Possible location " + possibleLogLocation.toUri().toString());
533               if (this.manager.getFs().exists(possibleLogLocation)) {
534                 // We found the right new location
535                 LOG.info("Log " + this.currentPath + " still exists at " +
536                     possibleLogLocation);
537                 // Breaking here will make us sleep since reader is null
538                 return true;
539               }
540             }
541           }
542           // TODO What happens if the log was missing from every single location?
543           // Although we need to check a couple of times as the log could have
544           // been moved by the master between the checks
545           // It can also happen if a recovered queue wasn't properly cleaned,
546           // such that the znode pointing to a log exists but the log was
547           // deleted a long time ago.
548           // For the moment, we'll throw the IO and processEndOfFile
549           throw new IOException("File from recovered queue is " +
550               "nowhere to be found", fnfe);
551         } else {
552           // If the log was archived, continue reading from there
553           Path archivedLogLocation =
554               new Path(manager.getOldLogDir(), currentPath.getName());
555           if (this.manager.getFs().exists(archivedLogLocation)) {
556             currentPath = archivedLogLocation;
557             LOG.info("Log " + this.currentPath + " was moved to " +
558                 archivedLogLocation);
559             // Open the log at the new location
560             this.openReader(sleepMultiplier);
561 
562           }
563           // TODO What happens the log is missing in both places?
564         }
565       }
566     } catch (IOException ioe) {
567       LOG.warn(peerClusterZnode + " Got: ", ioe);
568       // TODO Need a better way to determinate if a file is really gone but
569       // TODO without scanning all logs dir
570       if (sleepMultiplier == this.maxRetriesMultiplier) {
571         LOG.warn("Waited too long for this file, considering dumping");
572         return !processEndOfFile();
573       }
574     }
575     return true;
576   }

578   /**
579    * Do the sleeping logic
580    * @param msg Why we sleep
581    * @param sleepMultiplier by how many times the default sleeping time is augmented
582    * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
583    */
584   protected boolean sleepForRetries(String msg, int sleepMultiplier) {
585     try {
586       LOG.debug(msg + ", sleeping " + sleepForRetries + " times " + sleepMultiplier);
587       Thread.sleep(this.sleepForRetries * sleepMultiplier);
588     } catch (InterruptedException e) {
589       LOG.debug("Interrupted while sleeping between retries");
590     }
591     return sleepMultiplier < maxRetriesMultiplier;
592   }

594   /**
595    * We only want KVs that are scoped other than local
596    * @param edit The KV to check for replication
597    */
598   protected void removeNonReplicableEdits(WALEdit edit) {
599     NavigableMap<byte[], Integer> scopes = edit.getScopes();
600     List<KeyValue> kvs = edit.getKeyValues();
601     for (int i = edit.size()-1; i >= 0; i--) {
602       KeyValue kv = kvs.get(i);
603       // The scope will be null or empty if
604       // there's nothing to replicate in that WALEdit
605       if (scopes == null || !scopes.containsKey(kv.getFamily())) {
606         kvs.remove(i);
607       }
608     }
609   }

611   /**
612    * Count the number of different row keys in the given edit because of
613    * mini-batching. We assume that there's at least one KV in the WALEdit.
614    * @param edit edit to count row keys from
615    * @return number of different row keys
616    */
617   private int countDistinctRowKeys(WALEdit edit) {
618     List<KeyValue> kvs = edit.getKeyValues();
619     int distinctRowKeys = 1;
620     KeyValue lastKV = kvs.get(0);
621     for (int i = 0; i < edit.size(); i++) {
622       if (!kvs.get(i).matchingRow(lastKV)) {
623         distinctRowKeys++;
624       }
625     }
626     return distinctRowKeys;
627   }

629   /**
630    * Do the shipping logic
631    * @param currentWALisBeingWrittenTo was the current WAL being (seemingly) 
632    * written to when this method was called
633    */
634   protected void shipEdits(boolean currentWALisBeingWrittenTo) {
635     int sleepMultiplier = 1;
636     if (this.currentNbEntries == 0) {
637       LOG.warn("Was given 0 edits to ship");
638       return;
639     }
640     while (this.isActive()) {
641       if (!isPeerEnabled()) {
642         if (sleepForRetries("Replication is disabled", sleepMultiplier)) {
643           sleepMultiplier++;
644         }
645         continue;
646       }
647       try {
648         AdminProtocol rrs = getRS();
649         ProtobufUtil.replicateWALEntry(rrs,
650           Arrays.copyOf(this.entriesArray, currentNbEntries));
651         if (this.lastLoggedPosition != this.position) {
652           this.manager.logPositionAndCleanOldLogs(this.currentPath,
653               this.peerClusterZnode, this.position, queueRecovered, currentWALisBeingWrittenTo);
654           this.lastLoggedPosition = this.position;
655         }
656         this.totalReplicatedEdits += currentNbEntries;
657         this.metrics.shipBatch(this.currentNbOperations);
658         this.metrics.setAgeOfLastShippedOp(
659             this.entriesArray[currentNbEntries-1].getKey().getWriteTime());
660         break;
661 
662       } catch (IOException ioe) {
663         // Didn't ship anything, but must still age the last time we did
664         this.metrics.refreshAgeOfLastShippedOp();
665         if (ioe instanceof RemoteException) {
666           ioe = ((RemoteException) ioe).unwrapRemoteException();
667           LOG.warn("Can't replicate because of an error on the remote cluster: ", ioe);
668         } else {
669           if (ioe instanceof SocketTimeoutException) {
670             // This exception means we waited for more than 60s and nothing
671             // happened, the cluster is alive and calling it right away
672             // even for a test just makes things worse.
673             sleepForRetries("Encountered a SocketTimeoutException. Since the " +
674               "call to the remote cluster timed out, which is usually " +
675               "caused by a machine failure or a massive slowdown",
676               this.socketTimeoutMultiplier);
677           } else {
678             LOG.warn("Can't replicate because of a local or network error: ", ioe);
679           }
680         }
681 
682         try {
683           boolean down;
684           // Spin while the slave is down and we're not asked to shutdown/close
685           do {
686             down = isSlaveDown();
687             if (down) {
688               if (sleepForRetries("Since we are unable to replicate", sleepMultiplier)) {
689                 sleepMultiplier++;
690               } else {
691                 chooseSinks();
692               }
693             }
694           } while (this.isActive() && down );
695         } catch (InterruptedException e) {
696           LOG.debug("Interrupted while trying to contact the peer cluster");
697         }
698       }
699     }
700   }

776   /**
777    * Check if the slave is down by trying to establish a connection
778    * @return true if down, false if up
779    * @throws InterruptedException
780    */
781   public boolean isSlaveDown() throws InterruptedException {
782     final CountDownLatch latch = new CountDownLatch(1);
783     Thread pingThread = new Thread() {
784       public void run() {
785         try {
786           AdminProtocol rrs = getRS();
787           // Dummy call which should fail
788           ProtobufUtil.getServerInfo(rrs);
789           latch.countDown();
790         } catch (IOException ex) {
791           if (ex instanceof RemoteException) {
792             ex = ((RemoteException) ex).unwrapRemoteException();
793           }
794           LOG.info("Slave cluster looks down: " + ex.getMessage());
795         }
796       }
797     };
798     pingThread.start();
799     // awaits returns true if countDown happened
800     boolean down = ! latch.await(this.sleepForRetries, TimeUnit.MILLISECONDS);
801     pingThread.interrupt();
802     return down;
803   }
