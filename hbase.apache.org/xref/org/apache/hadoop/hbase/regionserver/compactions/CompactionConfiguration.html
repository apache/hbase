29  /**
30   * Compaction configuration for a particular instance of HStore.
31   * Takes into account both global settings and ones set on the column family/store.
32   * Control knobs for default compaction algorithm:
33   * <p/>
34   * maxCompactSize - upper bound on file size to be included in minor compactions
35   * minCompactSize - lower bound below which compaction is selected without ratio test
36   * minFilesToCompact - lower bound on number of files in any minor compaction
37   * maxFilesToCompact - upper bound on number of files in any minor compaction
38   * compactionRatio - Ratio used for compaction
39   * <p/>
40   * Set parameter as "hbase.hstore.compaction.<attribute>"
41   */
42  
43  //TODO: revisit this class for online parameter updating (both in xml and on the CF)
44  @InterfaceAudience.Private
45  public class CompactionConfiguration {
46  
47    static final Log LOG = LogFactory.getLog(CompactionConfiguration.class);
48  
49    private static final String CONFIG_PREFIX = "hbase.hstore.compaction.";
50  
51    Configuration conf;
52    StoreConfiguration storeConfig;
53  
54    long maxCompactSize;
55    long minCompactSize;
56    int minFilesToCompact;
57    int maxFilesToCompact;
58    double compactionRatio;
59    double offPeekCompactionRatio;
60    int offPeakStartHour;
61    int offPeakEndHour;
62    long throttlePoint;
63    boolean shouldDeleteExpired;
64    long majorCompactionPeriod;
65    float majorCompactionJitter;
66  
67    CompactionConfiguration(Configuration conf, StoreConfiguration storeConfig) {
68      this.conf = conf;
69      this.storeConfig = storeConfig;
70  
71      maxCompactSize = conf.getLong(CONFIG_PREFIX + "max.size", Long.MAX_VALUE);
72      minCompactSize = conf.getLong(CONFIG_PREFIX + "min.size",
73          storeConfig.getMemstoreFlushSize());
74      minFilesToCompact = Math.max(2, conf.getInt(CONFIG_PREFIX + "min",
75            /*old name*/ conf.getInt("hbase.hstore.compactionThreshold", 3)));
76      maxFilesToCompact = conf.getInt(CONFIG_PREFIX + "max", 10);
77      compactionRatio = conf.getFloat(CONFIG_PREFIX + "ratio", 1.2F);
78      offPeekCompactionRatio = conf.getFloat(CONFIG_PREFIX + "ratio.offpeak", 5.0F);
79      offPeakStartHour = conf.getInt("hbase.offpeak.start.hour", -1);
80      offPeakEndHour = conf.getInt("hbase.offpeak.end.hour", -1);
81  
82      if (!isValidHour(offPeakStartHour) || !isValidHour(offPeakEndHour)) {
83        if (!(offPeakStartHour == -1 && offPeakEndHour == -1)) {
84          LOG.warn("Ignoring invalid start/end hour for peak hour : start = " +
85              this.offPeakStartHour + " end = " + this.offPeakEndHour +
86              ". Valid numbers are [0-23]");
87        }
88        this.offPeakStartHour = this.offPeakEndHour = -1;
89      }
90  
91      throttlePoint =  conf.getLong("hbase.regionserver.thread.compaction.throttle",
92            2 * maxFilesToCompact * storeConfig.getMemstoreFlushSize());
93      shouldDeleteExpired = conf.getBoolean("hbase.store.delete.expired.storefile", true);
94      majorCompactionPeriod = conf.getLong(HConstants.MAJOR_COMPACTION_PERIOD, 1000*60*60*24);
95      majorCompactionJitter = conf.getFloat("hbase.hregion.majorcompaction.jitter", 0.20F);
96  
97      LOG.info("Compaction configuration " + this.toString());
98    }

100   @Override
101   public String toString() {
102     return String.format(
103       "size [%d, %d); files [%d, %d); ratio %f; off-peak ratio %f; off-peak hours %d-%d; "
104       + "throttle point %d;%s delete expired; major period %d, major jitter %f",
105       minCompactSize,
106       maxCompactSize,
107       minFilesToCompact,
108       maxFilesToCompact,
109       compactionRatio,
110       offPeekCompactionRatio,
111       offPeakStartHour,
112       offPeakEndHour,
113       throttlePoint,
114       shouldDeleteExpired ? "" : " don't",
115       majorCompactionPeriod,
116       majorCompactionJitter);
117   }
