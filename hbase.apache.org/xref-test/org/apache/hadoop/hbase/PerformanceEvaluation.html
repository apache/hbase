96  /**
97   * Script used evaluating HBase performance and scalability.  Runs a HBase
98   * client that steps through one of a set of hardcoded tests or 'experiments'
99   * (e.g. a random reads test, a random writes test, etc.). Pass on the
100  * command-line which test to run and how many clients are participating in
101  * this experiment. Run <code>java PerformanceEvaluation --help</code> to
102  * obtain usage.
103  *
104  * <p>This class sets up and runs the evaluation programs described in
105  * Section 7, <i>Performance Evaluation</i>, of the <a
106  * href="http://labs.google.com/papers/bigtable.html">Bigtable</a>
107  * paper, pages 8-10.
108  *
109  * <p>If number of clients > 1, we start up a MapReduce job. Each map task
110  * runs an individual client. Each client does about 1GB of data.
111  */

310   /**
311    *  InputFormat of Performance Evaluation MapReduce job.
312    *  It extends from FileInputFormat, want to use it's methods such as setInputPaths().
313    */
314   public static class PeInputFormat extends FileInputFormat<NullWritable, PeInputSplit> {
315 
316     @Override
317     public List<InputSplit> getSplits(JobContext job) throws IOException {
318       // generate splits
319       List<InputSplit> splitList = new ArrayList<InputSplit>();
320 
321       for (FileStatus file: listStatus(job)) {
322         if (file.isDir()) {
323           continue;
324         }
325         Path path = file.getPath();
326         FileSystem fs = path.getFileSystem(job.getConfiguration());
327         FSDataInputStream fileIn = fs.open(path);
328         LineReader in = new LineReader(fileIn, job.getConfiguration());
329         int lineLen = 0;
330         while(true) {
331           Text lineText = new Text();
332           lineLen = in.readLine(lineText);
333           if(lineLen <= 0) {
334           break;
335           }
336           Matcher m = LINE_PATTERN.matcher(lineText.toString());
337           if((m != null) && m.matches()) {
338             int startRow = Integer.parseInt(m.group(1));
339             int rows = Integer.parseInt(m.group(2));
340             int totalRows = Integer.parseInt(m.group(3));
341             int clients = Integer.parseInt(m.group(4));
342             boolean flushCommits = Boolean.parseBoolean(m.group(5));
343             boolean writeToWAL = Boolean.parseBoolean(m.group(6));
344 
345             LOG.debug("split["+ splitList.size() + "] " +
346                      " startRow=" + startRow +
347                      " rows=" + rows +
348                      " totalRows=" + totalRows +
349                      " clients=" + clients +
350                      " flushCommits=" + flushCommits +
351                      " writeToWAL=" + writeToWAL);
352 
353             PeInputSplit newSplit =
354               new PeInputSplit(startRow, rows, totalRows, clients,
355                 flushCommits, writeToWAL);
356             splitList.add(newSplit);
357           }
358         }
359         in.close();
360       }
361 
362       LOG.info("Total # of splits: " + splitList.size());
363       return splitList;
364     }

424   /**
425    * MapReduce job that runs a performance evaluation client in each map task.
426    */
427   public static class EvaluationMapTask
428       extends Mapper<NullWritable, PeInputSplit, LongWritable, LongWritable> {
429 
430     /** configuration parameter name that contains the command */
431     public final static String CMD_KEY = "EvaluationMapTask.command";
432     /** configuration parameter name that contains the PE impl */
433     public static final String PE_KEY = "EvaluationMapTask.performanceEvalImpl";
434 
435     private Class<? extends Test> cmd;
436     private PerformanceEvaluation pe;
437 
438     @Override
439     protected void setup(Context context) throws IOException, InterruptedException {
440       this.cmd = forName(context.getConfiguration().get(CMD_KEY), Test.class);
441 
442       // this is required so that extensions of PE are instantiated within the
443       // map reduce task...
444       Class<? extends PerformanceEvaluation> peClass =
445           forName(context.getConfiguration().get(PE_KEY), PerformanceEvaluation.class);
446       try {
447         this.pe = peClass.getConstructor(Configuration.class)
448             .newInstance(context.getConfiguration());
449       } catch (Exception e) {
450         throw new IllegalStateException("Could not instantiate PE instance", e);
451       }
452     }
453 
454     private <Type> Class<? extends Type> forName(String className, Class<Type> type) {
455       Class<? extends Type> clazz = null;
456       try {
457         clazz = Class.forName(className).asSubclass(type);
458       } catch (ClassNotFoundException e) {
459         throw new IllegalStateException("Could not find class for name: " + className, e);
460       }
461       return clazz;
462     }
463 
464     protected void map(NullWritable key, PeInputSplit value, final Context context)
465            throws IOException, InterruptedException {
466 
467       Status status = new Status() {
468         public void setStatus(String msg) {
469            context.setStatus(msg);
470         }
471       };
472 
473       // Evaluation task
474       long elapsedTime = this.pe.runOneClient(this.cmd, value.getStartRow(),
475                                   value.getRows(), value.getTotalRows(),
476                                   value.isFlushCommits(), value.isWriteToWAL(),
477                                   status);
478       // Collect how much time the thing took. Report as map output and
479       // to the ELAPSED_TIME counter.
480       context.getCounter(Counter.ELAPSED_TIME).increment(elapsedTime);
481       context.getCounter(Counter.ROWS).increment(value.rows);
482       context.write(new LongWritable(value.startRow), new LongWritable(elapsedTime));
483       context.progress();
484     }
485   }

487   /*
488    * If table does not already exist, create.
489    * @param c Client to use checking.
490    * @return True if we created the table.
491    * @throws IOException
492    */
493   private boolean checkTable(HBaseAdmin admin) throws IOException {
494     HTableDescriptor tableDescriptor = getTableDescriptor();
495     if (this.presplitRegions > 0) {
496       // presplit requested
497       if (admin.tableExists(tableDescriptor.getName())) {
498         admin.disableTable(tableDescriptor.getName());
499         admin.deleteTable(tableDescriptor.getName());
500       }
501 
502       byte[][] splits = getSplits();
503       for (int i=0; i < splits.length; i++) {
504         LOG.debug(" split " + i + ": " + Bytes.toStringBinary(splits[i]));
505       }
506       admin.createTable(tableDescriptor, splits);
507       LOG.info ("Table created with " + this.presplitRegions + " splits");
508     }
509     else {
510       boolean tableExists = admin.tableExists(tableDescriptor.getName());
511       if (!tableExists) {
512         admin.createTable(tableDescriptor);
513         LOG.info("Table " + tableDescriptor + " created");
514       }
515     }
516     boolean tableExists = admin.tableExists(tableDescriptor.getName());
517     return tableExists;
518   }
519 
520   protected HTableDescriptor getTableDescriptor() {
521     if (TABLE_DESCRIPTOR == null) {
522       TABLE_DESCRIPTOR = new HTableDescriptor(tableName);
523       HColumnDescriptor family = new HColumnDescriptor(FAMILY_NAME);
524       family.setDataBlockEncoding(blockEncoding);
525       family.setCompressionType(compression);
526       TABLE_DESCRIPTOR.addFamily(family);
527     }
528     return TABLE_DESCRIPTOR;
529   }

531   /**
532    * generates splits based on total number of rows and specified split regions
533    *
534    * @return splits : array of byte []
535    */
536   protected  byte[][] getSplits() {
537     if (this.presplitRegions == 0)
538       return new byte [0][];
539 
540     byte[][] splits = new byte[this.presplitRegions][];
541     int jump = this.R  / this.presplitRegions;
542     for (int i=0; i <this.presplitRegions; i++) {
543       int rowkey = jump * i;
544       splits[i] = format(rowkey);
545     }
546     return splits;
547   }

549   /*
550    * We're to run multiple clients concurrently.  Setup a mapreduce job.  Run
551    * one map per client.  Then run a single reduce to sum the elapsed times.
552    * @param cmd Command to run.
553    * @throws IOException
554    */
555   private void runNIsMoreThanOne(final Class<? extends Test> cmd)
556   throws IOException, InterruptedException, ClassNotFoundException {
557     checkTable(new HBaseAdmin(getConf()));
558     if (this.nomapred) {
559       doMultipleClients(cmd);
560     } else {
561       doMapReduce(cmd);
562     }
563   }

565   /*
566    * Run all clients in this vm each to its own thread.
567    * @param cmd Command to run.
568    * @throws IOException
569    */
570   private void doMultipleClients(final Class<? extends Test> cmd) throws IOException {
571     final List<Thread> threads = new ArrayList<Thread>(this.N);
572     final long[] timings = new long[this.N];
573     final int perClientRows = R/N;
574     for (int i = 0; i < this.N; i++) {
575       final int index = i;
576       Thread t = new Thread ("TestClient-" + i) {
577         @Override
578         public void run() {
579           super.run();
580           PerformanceEvaluation pe = new PerformanceEvaluation(getConf());
581           pe.N = N;
582           try {
583             long elapsedTime = pe.runOneClient(cmd, index * perClientRows,
584                perClientRows, R,
585                 flushCommits, writeToWAL, new Status() {
586                   public void setStatus(final String msg) throws IOException {
587                     LOG.info("client-" + getName() + " " + msg);
588                   }
589                 });
590             timings[index] = elapsedTime;
591             LOG.info("Finished " + getName() + " in " + elapsedTime +
592               "ms writing " + perClientRows + " rows");
593           } catch (IOException e) {
594             throw new RuntimeException(e);
595           }
596         }
597       };
598       threads.add(t);
599     }
600     for (Thread t: threads) {
601       t.start();
602     }
603     for (Thread t: threads) {
604       while(t.isAlive()) {
605         try {
606           t.join();
607         } catch (InterruptedException e) {
608           LOG.debug("Interrupted, continuing" + e.toString());
609         }
610       }
611     }
612     final String test = cmd.getSimpleName();
613     LOG.info("[" + test + "] Summary of timings (ms): "
614              + Arrays.toString(timings));
615     Arrays.sort(timings);
616     long total = 0;
617     for (int i = 0; i < this.N; i++) {
618       total += timings[i];
619     }
620     LOG.info("[" + test + "]"
621              + "\tMin: " + timings[0] + "ms"
622              + "\tMax: " + timings[this.N - 1] + "ms"
623              + "\tAvg: " + (total / this.N) + "ms");
624   }

626   /*
627    * Run a mapreduce job.  Run as many maps as asked-for clients.
628    * Before we start up the job, write out an input file with instruction
629    * per client regards which row they are to start on.
630    * @param cmd Command to run.
631    * @throws IOException
632    */
633   private void doMapReduce(final Class<? extends Test> cmd) throws IOException,
634         InterruptedException, ClassNotFoundException {
635     Configuration conf = getConf();
636     Path inputDir = writeInputFile(conf);
637     conf.set(EvaluationMapTask.CMD_KEY, cmd.getName());
638     conf.set(EvaluationMapTask.PE_KEY, getClass().getName());
639     Job job = new Job(conf);
640     job.setJarByClass(PerformanceEvaluation.class);
641     job.setJobName("HBase Performance Evaluation");
642 
643     job.setInputFormatClass(PeInputFormat.class);
644     PeInputFormat.setInputPaths(job, inputDir);
645 
646     job.setOutputKeyClass(LongWritable.class);
647     job.setOutputValueClass(LongWritable.class);
648 
649     job.setMapperClass(EvaluationMapTask.class);
650     job.setReducerClass(LongSumReducer.class);
651 
652     job.setNumReduceTasks(1);
653 
654     job.setOutputFormatClass(TextOutputFormat.class);
655     TextOutputFormat.setOutputPath(job, new Path(inputDir.getParent(), "outputs"));
656 
657     TableMapReduceUtil.addDependencyJars(job);
658     // Add a Class from the hbase.jar so it gets registered too.
659     TableMapReduceUtil.addDependencyJars(job.getConfiguration(),
660       org.apache.hadoop.hbase.util.Bytes.class);
661 
662     TableMapReduceUtil.initCredentials(job);
663 
664     job.waitForCompletion(true);
665   }

673   private Path writeInputFile(final Configuration c) throws IOException {
674     SimpleDateFormat formatter = new SimpleDateFormat("yyyyMMddHHmmss");
675     Path jobdir = new Path(PERF_EVAL_DIR, formatter.format(new Date()));
676     Path inputDir = new Path(jobdir, "inputs");
677 
678     FileSystem fs = FileSystem.get(c);
679     fs.mkdirs(inputDir);
680 
681     Path inputFile = new Path(inputDir, "input.txt");
682     PrintStream out = new PrintStream(fs.create(inputFile));
683     // Make input random.
684     Map<Integer, String> m = new TreeMap<Integer, String>();
685     Hash h = MurmurHash.getInstance();
686     int perClientRows = (this.R / this.N);
687     try {
688       for (int i = 0; i < 10; i++) {
689         for (int j = 0; j < N; j++) {
690           String s = "startRow=" + ((j * perClientRows) + (i * (perClientRows/10))) +
691           ", perClientRunRows=" + (perClientRows / 10) +
692           ", totalRows=" + this.R +
693           ", clients=" + this.N +
694           ", flushCommits=" + this.flushCommits +
695           ", writeToWAL=" + this.writeToWAL;
696           int hash = h.hash(Bytes.toBytes(s));
697           m.put(hash, s);
698         }
699       }
700       for (Map.Entry<Integer, String> e: m.entrySet()) {
701         out.println(e.getValue());
702       }
703     } finally {
704       out.close();
705     }
706     return inputDir;
707   }

896     private static final int MAX_OUTSTANDING_RPCS = 200000;  // Sized for 2G heap.
897 
898     private static HBaseClient client;  // Only one client regardless of number of threads.
899 
900     AsyncTest(final Configuration conf, final TestOptions options, final Status status) {
901       super(null, options, status);
902       final String zkquorum = conf.get(HConstants.ZOOKEEPER_QUORUM);
903       final String znode = conf.get(HConstants.ZOOKEEPER_ZNODE_PARENT,
904                                     HConstants.DEFAULT_ZOOKEEPER_ZNODE_PARENT);
905       synchronized (AsyncTest.class) {
906         if (client == null) {
907           client = new HBaseClient(zkquorum, znode);
908           // Sanity check.
909           try {
910             client.ensureTableFamilyExists(tableName, FAMILY_NAME).joinUninterruptibly();
911           } catch (Exception e) {
912             throw new RuntimeException("Missing test table/family?", e);
913           }
914         }
915       }
916       latch = new CountDownLatch(super.perClientRunRows);
917       final int maxrpcs = MAX_OUTSTANDING_RPCS / options.getNumClientThreads();
918       sem = new Semaphore(Math.max(100, maxrpcs));

921     /**
922      * If true, make sure that every read returns a valid-looking KeyValue.
923      */
924     private static final boolean CHECK_READS = false;
925 
926     /** Checks that the row retrieved from HBase looks valid.  */
927     protected static void check(final ArrayList<org.hbase.async.KeyValue> row) throws IOException {
928       if (!CHECK_READS) {
929         return;
930       }
931       if (row.size() != 1) {
932         throw new IOException((row.isEmpty() ? "No" : "Multiple (" + row.size() + ')')
933                               + " KeyValue found in row");
934       } else if (row.get(0).value().length != ROW_LENGTH) {
935         throw new IOException("Invalid value length (found: " + row.get(0).value().length
936                               + ", expected: " + ROW_LENGTH + ") in row \""
937                               + new String(row.get(0).key()) + '"');
938       }

941     private Exception error = null;  // Last exception caught asynchronously.
942     private volatile boolean failed = false;  // True if we caught an exception asynchronously.
943     /** Used by sub-classes to handle asynchronous exceptions.  */
944     protected final Callback<Exception, Exception> errback = new Callback<Exception, Exception>() {
945       public Exception call(final Exception e) throws Exception {
946         rpcCompleted();
947         if (e instanceof PleaseThrottleException) {
948           LOG.warn("Throttling thread " + Thread.currentThread().getName()
949                    + ", HBase isn't keeping up", e);
950           final int permits = sem.drainPermits();  // Prevent creation of further RPCs.
951           ((PleaseThrottleException) e).getDeferred().addBoth(new Callback<Object, Object>() {
952             public Object call(final Object arg) {
953               sem.release(permits);
954               LOG.warn("Done throttling thread " + Thread.currentThread().getName());
955               return arg;
956             }
957             public String toString() {
958               return "error recovery after " + e;
959             }
960           });
961           return null;
962         }
963         error = e;
964         failed = true;  // Volatile-write.
965         LOG.error(this + " caught an exception", e);
966         return e;
967       }
968 
969       private final String toString = "errback for " + AsyncTest.this + " in " + Thread.currentThread().getName();
970       public String toString() {
971         return toString;
972       }
973     };
974 
975     /**
976      * Latch to guarantee we have gotten a response for every single RPC sent.
977      * This latch is initialized up with the number of RPCs we intend to send.
978      * Every time an RPC completes successfully, we decrement its count down
979      * by one.  This way we guarantee that all RPCs have completed and their
980      * responses have been handled within the section of the code we're
981      * timing.
982      */
983     private final CountDownLatch latch;
984 
985     /**
986      * Semaphore to control the number of outstanding RPCs.
987      * Because the producer code is synchronous and asynchbase is
988      * non-blocking, the tests will try to create and send all RPCs at once,
989      * thus running the app out of memory.  In order to limit the number of
990      * RPCs in flight at the same time, we acquire a permit from this
991      * semaphore each time we access the client to send an RPC, and we release
992      * the permit each time the RPC completes.
993      */
994     private final Semaphore sem;
995 
996     /** Records the completion of an RPC.  */
997     protected final void rpcCompleted() {
998       sem.release();
999       latch.countDown();
1000     }
1001 
1002     /** Callback used on successful read RPCs.  */
1003     protected final Callback<Object, ArrayList<org.hbase.async.KeyValue>> readCallback = new Callback<Object, ArrayList<org.hbase.async.KeyValue>>() {
1004       public Object call(final ArrayList<org.hbase.async.KeyValue> row) throws IOException {
1005         rpcCompleted();
1006         check(row);
1007         return row;
1008       }
1009 
1010       private final String toString = "callback for " + AsyncTest.this + " in " + Thread.currentThread().getName();
1011       public String toString() {
1012         return toString;
1013       }
1014     };
1015 
1016     /** Callback used on other successful RPCs.  */
1017     protected final Callback<Object, Object> callback = new Callback<Object, Object>() {
1018       public Object call(final Object arg) {
1019         rpcCompleted();
1020         return arg;
1021       }
1022 
1023       private final String toString = "callback for " + AsyncTest.this + " in " + Thread.currentThread().getName();
1024       public String toString() {
1025         return toString;
1026       }
1027     };

1029     @Override
1030     final void testSetup() {
1031       // Nothing.
1032     }
1033 
1034     @Override
1035     final void testTakedown() throws IOException {
1036       try {
1037         // For tests with few writes, asking for a flush before waiting on the
1038         // latch tells asynchbase to start flushing writes instead of waiting
1039         // until the timer flushes them.
1040         client.flush().join();
1041         latch.await();  // Make sure the last RPC completed.
1042         if (failed) {  // Volatile-read
1043           throw error;
1044         }
1045       } catch (RuntimeException e) {
1046         throw e;
1047       } catch (IOException e) {
1048         throw e;
1049       } catch (Exception e) {
1050         throw new IOException("Uncaught exception from flush()", e);
1051       }
1052     }
1053 
1054     /** Returns the client to use to send an RPC.  Call once per RPC.  */
1055     protected final HBaseClient client() {
1056       try {
1057         sem.acquire();
1058       } catch (InterruptedException e) {
1059         LOG.error("Shouldn't happen!", e);
1060         return null;
1061       }
1062       return client;
1063     }

1422   public static byte [] format(final int number) {

1424     int d = Math.abs(number);
