36  @SuppressWarnings("deprecation")
37  @Category(SmallTests.class)
38  public class TestBlocksScanned extends HBaseTestCase {
39    private static byte [] TABLE = Bytes.toBytes("TestBlocksScanned");
40    private static byte [] FAMILY = Bytes.toBytes("family");
41    private static byte [] COL = Bytes.toBytes("col");
42    private static byte [] START_KEY = Bytes.toBytes("aaa");
43    private static byte [] END_KEY = Bytes.toBytes("zzz");
44    private static int BLOCK_SIZE = 70;
45  
46    private static HBaseTestingUtility TEST_UTIL = null;
47    private static HTableDescriptor TESTTABLEDESC = null;
48  
49     @Override
50     public void setUp() throws Exception {
51       super.setUp();
52  
53       TEST_UTIL = new HBaseTestingUtility();
54       TESTTABLEDESC = new HTableDescriptor(TABLE);
55  
56       TESTTABLEDESC.addFamily(
57           new HColumnDescriptor(FAMILY)
58           .setMaxVersions(10)
59           .setBlockCacheEnabled(true)
60           .setBlocksize(BLOCK_SIZE)
61           .setCompressionType(Compression.Algorithm.NONE)
62       );
63     }
64  
65     @Test
66    public void testBlocksScanned() throws Exception {
67      HRegion r = createNewHRegion(TESTTABLEDESC, START_KEY, END_KEY,
68          TEST_UTIL.getConfiguration());
69      addContent(r, FAMILY, COL);
70      r.flushcache();
71  
72      // Do simple test of getting one row only first.
73      Scan scan = new Scan(Bytes.toBytes("aaa"), Bytes.toBytes("aaz"));
74      scan.addColumn(FAMILY, COL);
75      scan.setMaxVersions(1);
76  
77      InternalScanner s = r.getScanner(scan);
78      List<KeyValue> results = new ArrayList<KeyValue>();
79      while (s.next(results));
80      s.close();
81  
82      int expectResultSize = 'z' - 'a';
83      Assert.assertEquals(expectResultSize, results.size());
84  
85      int kvPerBlock = (int) Math.ceil(BLOCK_SIZE / (double) results.get(0).getLength());
86      Assert.assertEquals(2, kvPerBlock);
87    }
