40   * Test for testing protocol buffer based RPC mechanism.
41   * This test depends on test.proto definition of types in 
42   * hbase-server/src/test/protobuf/test.proto

44   * hbase-server/src/test/protobuf/test_rpc_service.proto

46  @Category(MediumTests.class)
47  public class TestProtoBufRpc {
48    public final static String ADDRESS = "0.0.0.0";
49    public final static int PORT = 0;
50    private static InetSocketAddress addr;
51    private static Configuration conf;
52    private static RpcServer server;
53  
54    public interface TestRpcService
55        extends TestProtobufRpcProto.BlockingInterface, VersionedProtocol {
56      public long VERSION = 1;
57    }
58  
59    public static class PBServerImpl implements TestRpcService {
60  
61      @Override
62      public EmptyResponseProto ping(RpcController unused,
63          EmptyRequestProto request) throws ServiceException {
64        return EmptyResponseProto.newBuilder().build();
65      }
66  
67      @Override
68      public EchoResponseProto echo(RpcController unused, EchoRequestProto request)
69          throws ServiceException {
70        return EchoResponseProto.newBuilder().setMessage(request.getMessage())
71            .build();
72      }
73  
74      @Override
75      public EmptyResponseProto error(RpcController unused,
76          EmptyRequestProto request) throws ServiceException {
77        throw new ServiceException("error", new IOException("error"));
78      }
79  
80      @Override
81      public long getProtocolVersion(String protocol, long clientVersion)
82          throws IOException {
83        // TODO Auto-generated method stub
84        return 0;
85      }
86  
87      @Override
88      public ProtocolSignature getProtocolSignature(String protocol,
89          long clientVersion, int clientMethodsHash) throws IOException {
90        // TODO Auto-generated method stub
91        return null;
92      }
93    }
94  
95    @Before
96    public  void setUp() throws IOException { // Setup server for both protocols
97      conf = new Configuration();
98      // Set RPC engine to protobuf RPC engine
99      HBaseRPC.setProtocolEngine(conf, TestRpcService.class, ProtobufRpcEngine.class);
100 
101     // Create server side implementation
102     PBServerImpl serverImpl = new PBServerImpl();
103     // Get RPC server for server side implementation
104     server = HBaseRPC.getServer(TestRpcService.class,serverImpl, 
105         new Class[]{TestRpcService.class}, 
106         ADDRESS, PORT, 10, 10, true, conf, 0);
107     addr = server.getListenerAddress();
108     server.start();
109   }
110   
111   
112   @After
113   public void tearDown() throws Exception {
114     server.stop();
115   }

117   private static TestRpcService getClient() throws IOException {
118     // Set RPC engine to protobuf RPC engine
119     HBaseRPC.setProtocolEngine(conf, TestRpcService.class,
120         ProtobufRpcEngine.class);
121     return (TestRpcService) HBaseRPC.getProxy(TestRpcService.class, 0, 
122         addr, conf, 10000);
123   }

125   @Test
126   public void testProtoBufRpc() throws Exception {

128     testProtoBufRpc(client);
129   }

131   // separated test out so that other tests can call it.
132   public static void testProtoBufRpc(TestRpcService client) throws Exception {  
133     // Test ping method
134     EmptyRequestProto emptyRequest = EmptyRequestProto.newBuilder().build();
135     client.ping(null, emptyRequest);
136     
137     // Test echo method
138     EchoRequestProto echoRequest = EchoRequestProto.newBuilder()
139         .setMessage("hello").build();
140     EchoResponseProto echoResponse = client.echo(null, echoRequest);
141     Assert.assertEquals(echoResponse.getMessage(), "hello");
142     
143     // Test error method - error should be thrown as RemoteException
144     try {
145       client.error(null, emptyRequest);
146       Assert.fail("Expected exception is not thrown");
147     } catch (ServiceException e) {
148     }
149   }
