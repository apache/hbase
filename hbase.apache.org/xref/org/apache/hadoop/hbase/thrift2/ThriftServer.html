64  /**
65   * ThriftServer - this class starts up a Thrift server which implements the HBase API specified in the
66   * HbaseClient.thrift IDL file.
67   */
68  @InterfaceAudience.Private
69  public class ThriftServer {
70    private static final Log log = LogFactory.getLog(ThriftServer.class);
71  
72    public static final String DEFAULT_LISTEN_PORT = "9090";
73  
74    public ThriftServer() {
75    }

77    private static void printUsage() {
78      HelpFormatter formatter = new HelpFormatter();
79      formatter.printHelp("Thrift", null, getOptions(),
80          "To start the Thrift server run 'bin/hbase-daemon.sh start thrift2'\n" +
81              "To shutdown the thrift server run 'bin/hbase-daemon.sh stop thrift2' or" +
82              " send a kill signal to the thrift server pid",
83          true);
84    }

86    private static Options getOptions() {
87      Options options = new Options();
88      options.addOption("b", "bind", true,
89          "Address to bind the Thrift server to. [default: 0.0.0.0]");
90      options.addOption("p", "port", true, "Port to bind to [default: " + DEFAULT_LISTEN_PORT + "]");
91      options.addOption("f", "framed", false, "Use framed transport");
92      options.addOption("c", "compact", false, "Use the compact protocol");
93      options.addOption("h", "help", false, "Print help information");
94  
95      OptionGroup servers = new OptionGroup();
96      servers.addOption(
97          new Option("nonblocking", false, "Use the TNonblockingServer. This implies the framed transport."));
98      servers.addOption(new Option("hsha", false, "Use the THsHaServer. This implies the framed transport."));
99      servers.addOption(new Option("threadpool", false, "Use the TThreadPoolServer. This is the default."));
100     options.addOptionGroup(servers);
101     return options;
102   }

104   private static CommandLine parseArguments(Options options, String[] args) throws ParseException {
105     CommandLineParser parser = new PosixParser();
106     return parser.parse(options, args);
107   }

109   private static TProtocolFactory getTProtocolFactory(boolean isCompact) {
110     if (isCompact) {
111       log.debug("Using compact protocol");
112       return new TCompactProtocol.Factory();
113     } else {
114       log.debug("Using binary protocol");
115       return new TBinaryProtocol.Factory();
116     }
117   }
118 
119   private static TTransportFactory getTTransportFactory(boolean framed) {
120     if (framed) {
121       log.debug("Using framed transport");
122       return new TFramedTransport.Factory();
123     } else {
124       return new TTransportFactory();
125     }
126   }

128   /*
129    * If bindValue is null, we don't bind. 
130    */
131   private static InetSocketAddress bindToPort(String bindValue, int listenPort)
132       throws UnknownHostException {
133     try {
134       if (bindValue == null) {
135         return new InetSocketAddress(listenPort);
136       } else {
137         return new InetSocketAddress(InetAddress.getByName(bindValue), listenPort);
138       }
139     } catch (UnknownHostException e) {
140       throw new RuntimeException("Could not bind to provided ip address", e);
141     }
142   }

144   private static TServer getTNonBlockingServer(TProtocolFactory protocolFactory, THBaseService.Processor processor,
145       TTransportFactory transportFactory, InetSocketAddress inetSocketAddress) throws TTransportException {
146     TNonblockingServerTransport serverTransport = new TNonblockingServerSocket(inetSocketAddress);
147     log.info("starting HBase Nonblocking Thrift server on " + inetSocketAddress.toString());
148     TNonblockingServer.Args serverArgs = new TNonblockingServer.Args(serverTransport);
149     serverArgs.processor(processor);
150     serverArgs.transportFactory(transportFactory);
151     serverArgs.protocolFactory(protocolFactory);
152     return new TNonblockingServer(serverArgs);
153   }
154 
155   private static TServer getTHsHaServer(TProtocolFactory protocolFactory,
156       THBaseService.Processor processor, TTransportFactory transportFactory,
157       InetSocketAddress inetSocketAddress, ThriftMetrics metrics)
158       throws TTransportException {
159     TNonblockingServerTransport serverTransport = new TNonblockingServerSocket(inetSocketAddress);
160     log.info("starting HBase HsHA Thrift server on " + inetSocketAddress.toString());
161     THsHaServer.Args serverArgs = new THsHaServer.Args(serverTransport);
162     ExecutorService executorService = createExecutor(
163         serverArgs.getWorkerThreads(), metrics);
164     serverArgs.executorService(executorService);
165     serverArgs.processor(processor);
166     serverArgs.transportFactory(transportFactory);
167     serverArgs.protocolFactory(protocolFactory);
168     return new THsHaServer(serverArgs);
169   }
170 
171   private static ExecutorService createExecutor(
172       int workerThreads, ThriftMetrics metrics) {
173     CallQueue callQueue = new CallQueue(
174         new LinkedBlockingQueue<Call>(), metrics);
175     ThreadFactoryBuilder tfb = new ThreadFactoryBuilder();
176     tfb.setDaemon(true);
177     tfb.setNameFormat("thrift2-worker-%d");
178     return new ThreadPoolExecutor(workerThreads, workerThreads,
179             Long.MAX_VALUE, TimeUnit.SECONDS, callQueue, tfb.build());
180   }

182   private static TServer getTThreadPoolServer(TProtocolFactory protocolFactory, THBaseService.Processor processor,
183       TTransportFactory transportFactory, InetSocketAddress inetSocketAddress) throws TTransportException {
184     TServerTransport serverTransport = new TServerSocket(inetSocketAddress);
185     log.info("starting HBase ThreadPool Thrift server on " + inetSocketAddress.toString());
186     TThreadPoolServer.Args serverArgs = new TThreadPoolServer.Args(serverTransport);
187     serverArgs.processor(processor);
188     serverArgs.transportFactory(transportFactory);
189     serverArgs.protocolFactory(protocolFactory);
190     return new TThreadPoolServer(serverArgs);
191   }

193   /**
194    * Start up the Thrift2 server.
195    * 
196    * @param args
197    */
198   public static void main(String[] args) throws Exception {
199     TServer server = null;
200     Options options = getOptions();
201     try {
202       CommandLine cmd = parseArguments(options, args);
203 
204       /**
205        * This is to please both bin/hbase and bin/hbase-daemon. hbase-daemon provides "start" and "stop" arguments hbase
206        * should print the help if no argument is provided
207        */
208       List<?> argList = cmd.getArgList();
209       if (cmd.hasOption("help") || !argList.contains("start") || argList.contains("stop")) {
210         printUsage();
211         System.exit(1);
212       }
213 
214       // Get port to bind to
215       int listenPort = 0;
216       try {
217         listenPort = Integer.parseInt(cmd.getOptionValue("port", DEFAULT_LISTEN_PORT));
218       } catch (NumberFormatException e) {
219         throw new RuntimeException("Could not parse the value provided for the port option", e);
220       }
221 
222       boolean nonblocking = cmd.hasOption("nonblocking");
223       boolean hsha = cmd.hasOption("hsha");
224 
225       Configuration conf = HBaseConfiguration.create();
226       ThriftMetrics metrics = new ThriftMetrics(conf, ThriftMetrics.ThriftServerType.TWO);
227 
228       // Construct correct ProtocolFactory
229       TProtocolFactory protocolFactory = getTProtocolFactory(cmd.hasOption("compact"));
230       THBaseService.Iface handler =
231           ThriftHBaseServiceHandler.newInstance(conf, metrics);
232       THBaseService.Processor processor = new THBaseService.Processor(handler);
233 
234       boolean framed = cmd.hasOption("framed") || nonblocking || hsha;
235       TTransportFactory transportFactory = getTTransportFactory(framed);
236       InetSocketAddress inetSocketAddress = bindToPort(cmd.getOptionValue("bind"), listenPort);
237 
238       if (nonblocking) {
239         server = getTNonBlockingServer(protocolFactory, processor, transportFactory, inetSocketAddress);
240       } else if (hsha) {
241         server = getTHsHaServer(protocolFactory, processor, transportFactory, inetSocketAddress, metrics);
242       } else {
243         server = getTThreadPoolServer(protocolFactory, processor, transportFactory, inetSocketAddress);
244       }
245     } catch (Exception e) {
246       log.error(e.getMessage(), e);
247       printUsage();
248       System.exit(1);
249     }
250     server.serve();
251   }
