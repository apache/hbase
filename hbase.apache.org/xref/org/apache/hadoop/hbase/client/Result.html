38  /**
39   * Single row result of a {@link Get} or {@link Scan} query.<p>
40   *
41   * This class is NOT THREAD SAFE.<p>
42   *
43   * Convenience methods are available that return various {@link Map}
44   * structures and values directly.<p>
45   *
46   * To get a complete mapping of all cells in the Result, which can include
47   * multiple families and multiple versions, use {@link #getMap()}.<p>
48   *
49   * To get a mapping of each family to its columns (qualifiers and values),
50   * including only the latest version of each, use {@link #getNoVersionMap()}.
51   *
52   * To get a mapping of qualifiers to latest values for an individual family use
53   * {@link #getFamilyMap(byte[])}.<p>
54   *
55   * To get the latest value for a specific family and qualifier use {@link #getValue(byte[], byte[])}.
56   *
57   * A Result is backed by an array of {@link KeyValue} objects, each representing
58   * an HBase cell defined by the row, family, qualifier, timestamp, and value.<p>
59   *
60   * The underlying {@link KeyValue} objects can be accessed through the method {@link #list()}.
61   * Each KeyValue can then be accessed through
62   * {@link KeyValue#getRow()}, {@link KeyValue#getFamily()}, {@link KeyValue#getQualifier()},
63   * {@link KeyValue#getTimestamp()}, and {@link KeyValue#getValue()}.<p>
64   * 
65   * If you need to overwrite a Result with another Result instance -- as in the old 'mapred' RecordReader next
66   * invocations -- then create an empty Result with the null constructor and in then use {@link #copyFrom(Result)}
67   */
68  @InterfaceAudience.Public
69  @InterfaceStability.Stable
70  public class Result {

73    // that this is where we cache row if we're ever asked for it.

76    private transient NavigableMap<byte[], NavigableMap<byte[], NavigableMap<Long, byte[]>>> familyMap = null;

124   /**
125    * Return the array of KeyValues backing this Result instance.
126    *
127    * The array is sorted from smallest -> largest using the
128    * {@link KeyValue#COMPARATOR}.
129    *
130    * The array only contains what your Get or Scan specifies and no more.
131    * For example if you request column "A" 1 version you will have at most 1
132    * KeyValue in the array. If you request column "A" with 2 version you will
133    * have at most 2 KeyValues, with the first one being the newer timestamp and
134    * the second being the older timestamp (this is the sort order defined by
135    * {@link KeyValue#COMPARATOR}).  If columns don't exist, they won't be
136    * present in the result. Therefore if you ask for 1 version all columns,
137    * it is safe to iterate over this array and expect to see 1 KeyValue for
138    * each column and no more.
139    *
140    * This API is faster than using getFamilyMap() and getMap()
141    *
142    * @return array of KeyValues; can be null if nothing in the result
143    */
144   public KeyValue[] raw() {
145     return kvs;
146   }

159   /**
160    * Return the KeyValues for the specific column.  The KeyValues are sorted in
161    * the {@link KeyValue#COMPARATOR} order.  That implies the first entry in
162    * the list is the most recent column.  If the query (Scan or Get) only
163    * requested 1 version the list will contain at most 1 entry.  If the column
164    * did not exist in the result set (either the column does not exist
165    * or the column was not selected in the query) the list will be empty.
166    *
167    * Also see getColumnLatest which returns just a KeyValue
168    *
169    * @param family the family
170    * @param qualifier
171    * @return a list of KeyValues for this column or empty list if the column
172    * did not exist in the result set
173    */
174   public List<KeyValue> getColumn(byte [] family, byte [] qualifier) {
175     List<KeyValue> result = new ArrayList<KeyValue>();
176 
177     KeyValue [] kvs = raw();
178 
179     if (kvs == null || kvs.length == 0) {
180       return result;
181     }
182     int pos = binarySearch(kvs, family, qualifier);
183     if (pos == -1) {
184       return result; // cant find it
185     }
186 
187     for (int i = pos ; i < kvs.length ; i++ ) {
188       KeyValue kv = kvs[i];
189       if (kv.matchingColumn(family,qualifier)) {
190         result.add(kv);
191       } else {
192         break;
193       }
194     }
195 
196     return result;
197   }
198 
199   protected int binarySearch(final KeyValue [] kvs,
200                              final byte [] family,
201                              final byte [] qualifier) {
202     KeyValue searchTerm =
203         KeyValue.createFirstOnRow(kvs[0].getRow(),
204             family, qualifier);
205 
206     // pos === ( -(insertion point) - 1)
207     int pos = Arrays.binarySearch(kvs, searchTerm, KeyValue.COMPARATOR);
208     // never will exact match
209     if (pos < 0) {
210       pos = (pos+1) * -1;
211       // pos is now insertion point
212     }
213     if (pos == kvs.length) {
214       return -1; // doesn't exist
215     }
216     return pos;
217   }

219   /**
220    * Searches for the latest value for the specified column.
221    *
222    * @param kvs the array to search
223    * @param family family name
224    * @param foffset family offset
225    * @param flength family length
226    * @param qualifier column qualifier
227    * @param qoffset qualifier offset
228    * @param qlength qualifier length
229    *
230    * @return the index where the value was found, or -1 otherwise
231    */
232   protected int binarySearch(final KeyValue [] kvs,
233       final byte [] family, final int foffset, final int flength,
234       final byte [] qualifier, final int qoffset, final int qlength) {
235 
236     double keyValueSize = (double)
237         KeyValue.getKeyValueDataStructureSize(kvs[0].getRowLength(), flength, qlength, 0);
238 
239     if (buffer == null || keyValueSize > buffer.length) {
240       // pad to the smallest multiple of the pad width
241       buffer = new byte[(int) Math.ceil(keyValueSize / PAD_WIDTH) * PAD_WIDTH];
242     }
243 
244     KeyValue searchTerm = KeyValue.createFirstOnRow(buffer, 0,
245         kvs[0].getBuffer(), kvs[0].getRowOffset(), kvs[0].getRowLength(),
246         family, foffset, flength,
247         qualifier, qoffset, qlength);
248 
249     // pos === ( -(insertion point) - 1)
250     int pos = Arrays.binarySearch(kvs, searchTerm, KeyValue.COMPARATOR);
251     // never will exact match
252     if (pos < 0) {
253       pos = (pos+1) * -1;
254       // pos is now insertion point
255     }
256     if (pos == kvs.length) {
257       return -1; // doesn't exist
258     }
259     return pos;
260   }

262   /**
263    * The KeyValue for the most recent timestamp for a given column.
264    *
265    * @param family
266    * @param qualifier
267    *
268    * @return the KeyValue for the column, or null if no value exists in the row or none have been
269    * selected in the query (Get/Scan)
270    */
271   public KeyValue getColumnLatest(byte [] family, byte [] qualifier) {
272     KeyValue [] kvs = raw(); // side effect possibly.
273     if (kvs == null || kvs.length == 0) {
274       return null;
275     }
276     int pos = binarySearch(kvs, family, qualifier);
277     if (pos == -1) {
278       return null;
279     }
280     KeyValue kv = kvs[pos];
281     if (kv.matchingColumn(family, qualifier)) {
282       return kv;
283     }
284     return null;
285   }

516   /**
517    * Map of families to all versions of its qualifiers and values.
518    * <p>
519    * Returns a three level Map of the form:
520    * <code>Map&amp;family,Map&lt;qualifier,Map&lt;timestamp,value>>></code>
521    * <p>
522    * Note: All other map returning methods make use of this map internally.
523    * @return map from families to qualifiers to versions
524    */
525   public NavigableMap<byte[], NavigableMap<byte[], NavigableMap<Long, byte[]>>> getMap() {
526     if (this.familyMap != null) {
527       return this.familyMap;
528     }
529     if(isEmpty()) {
530       return null;
531     }
532     this.familyMap = new TreeMap<byte[], NavigableMap<byte[], NavigableMap<Long, byte[]>>>(Bytes.BYTES_COMPARATOR);
533     for(KeyValue kv : this.kvs) {
534       SplitKeyValue splitKV = kv.split();
535       byte [] family = splitKV.getFamily();
536       NavigableMap<byte[], NavigableMap<Long, byte[]>> columnMap =
537         familyMap.get(family);
538       if(columnMap == null) {
539         columnMap = new TreeMap<byte[], NavigableMap<Long, byte[]>>
540           (Bytes.BYTES_COMPARATOR);
541         familyMap.put(family, columnMap);
542       }
543       byte [] qualifier = splitKV.getQualifier();
544       NavigableMap<Long, byte[]> versionMap = columnMap.get(qualifier);
545       if(versionMap == null) {
546         versionMap = new TreeMap<Long, byte[]>(new Comparator<Long>() {
547           public int compare(Long l1, Long l2) {
548             return l2.compareTo(l1);
549           }
550         });
551         columnMap.put(qualifier, versionMap);
552       }
553       Long timestamp = Bytes.toLong(splitKV.getTimestamp());
554       byte [] value = splitKV.getValue();
555       versionMap.put(timestamp, value);
556     }
557     return this.familyMap;
558   }

560   /**
561    * Map of families to their most recent qualifiers and values.
562    * <p>
563    * Returns a two level Map of the form: <code>Map&amp;family,Map&lt;qualifier,value>></code>
564    * <p>
565    * The most recent version of each qualifier will be used.
566    * @return map from families to qualifiers and value
567    */
568   public NavigableMap<byte[], NavigableMap<byte[], byte[]>> getNoVersionMap() {
569     if(this.familyMap == null) {
570       getMap();
571     }
572     if(isEmpty()) {
573       return null;
574     }
575     NavigableMap<byte[], NavigableMap<byte[], byte[]>> returnMap =
576       new TreeMap<byte[], NavigableMap<byte[], byte[]>>(Bytes.BYTES_COMPARATOR);
577     for(Map.Entry<byte[], NavigableMap<byte[], NavigableMap<Long, byte[]>>>
578       familyEntry : familyMap.entrySet()) {
579       NavigableMap<byte[], byte[]> qualifierMap =
580         new TreeMap<byte[], byte[]>(Bytes.BYTES_COMPARATOR);
581       for(Map.Entry<byte[], NavigableMap<Long, byte[]>> qualifierEntry :
582         familyEntry.getValue().entrySet()) {
583         byte [] value =
584           qualifierEntry.getValue().get(qualifierEntry.getValue().firstKey());
585         qualifierMap.put(qualifierEntry.getKey(), value);
586       }
587       returnMap.put(familyEntry.getKey(), qualifierMap);
588     }
589     return returnMap;
590   }

592   /**
593    * Map of qualifiers to values.
594    * <p>
595    * Returns a Map of the form: <code>Map&lt;qualifier,value></code>
596    * @param family column family to get
597    * @return map of qualifiers to values
598    */
599   public NavigableMap<byte[], byte[]> getFamilyMap(byte [] family) {
600     if(this.familyMap == null) {
601       getMap();
602     }
603     if(isEmpty()) {
604       return null;
605     }
606     NavigableMap<byte[], byte[]> returnMap =
607       new TreeMap<byte[], byte[]>(Bytes.BYTES_COMPARATOR);
608     NavigableMap<byte[], NavigableMap<Long, byte[]>> qualifierMap =
609       familyMap.get(family);
610     if(qualifierMap == null) {
611       return returnMap;
612     }
613     for(Map.Entry<byte[], NavigableMap<Long, byte[]>> entry :
614       qualifierMap.entrySet()) {
615       byte [] value =
616         entry.getValue().get(entry.getValue().firstKey());
617       returnMap.put(entry.getKey(), value);
618     }
619     return returnMap;
620   }

648   /**
649    * @return String
650    */
651   @Override
652   public String toString() {
653     StringBuilder sb = new StringBuilder();
654     sb.append("keyvalues=");
655     if(isEmpty()) {
656       sb.append("NONE");
657       return sb.toString();
658     }
659     sb.append("{");
660     boolean moreThanOne = false;
661     for(KeyValue kv : this.kvs) {
662       if(moreThanOne) {
663         sb.append(", ");
664       } else {
665         moreThanOne = true;
666       }
667       sb.append(kv.toString());
668     }
669     sb.append("}");
670     return sb.toString();
671   }

673   /**
674    * Does a deep comparison of two Results, down to the byte arrays.
675    * @param res1 first result to compare
676    * @param res2 second result to compare
677    * @throws Exception Every difference is throwing an exception
678    */
679   public static void compareResults(Result res1, Result res2)
680       throws Exception {
681     if (res2 == null) {  
682       throw new Exception("There wasn't enough rows, we stopped at "
683           + Bytes.toStringBinary(res1.getRow()));
684     }
