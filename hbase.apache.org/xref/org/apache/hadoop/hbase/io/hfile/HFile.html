74  /**
75   * File format for hbase.
76   * A file of sorted key/value pairs. Both keys and values are byte arrays.
77   * <p>
78   * The memory footprint of a HFile includes the following (below is taken from the
79   * <a
80   * href=https://issues.apache.org/jira/browse/HADOOP-3315>TFile</a> documentation
81   * but applies also to HFile):
82   * <ul>
83   * <li>Some constant overhead of reading or writing a compressed block.
84   * <ul>
85   * <li>Each compressed block requires one compression/decompression codec for
86   * I/O.
87   * <li>Temporary space to buffer the key.
88   * <li>Temporary space to buffer the value.
89   * </ul>
90   * <li>HFile index, which is proportional to the total number of Data Blocks.
91   * The total amount of memory needed to hold the index can be estimated as
92   * (56+AvgKeySize)*NumBlocks.
93   * </ul>
94   * Suggestions on performance optimization.
95   * <ul>
96   * <li>Minimum block size. We recommend a setting of minimum block size between
97   * 8KB to 1MB for general usage. Larger block size is preferred if files are
98   * primarily for sequential access. However, it would lead to inefficient random
99   * access (because there are more data to decompress). Smaller blocks are good
100  * for random access, but require more memory to hold the block index, and may
101  * be slower to create (because we must flush the compressor stream at the
102  * conclusion of each data block, which leads to an FS I/O flush). Further, due
103  * to the internal caching in Compression codec, the smallest possible block
104  * size would be around 20KB-30KB.
105  * <li>The current implementation does not offer true multi-threading for
106  * reading. The implementation uses FSDataInputStream seek()+read(), which is
107  * shown to be much faster than positioned-read call in single thread mode.
108  * However, it also means that if multiple threads attempt to access the same
109  * HFile (using multiple scanners) simultaneously, the actual I/O is carried out
110  * sequentially even if they access different DFS blocks (Reexamine! pread seems
111  * to be 10% faster than seek+read in my testing -- stack).
112  * <li>Compression codec. Use "none" if the data is not very compressable (by
113  * compressable, I mean a compression ratio at least 2:1). Generally, use "lzo"
114  * as the starting point for experimenting. "gz" overs slightly better
115  * compression ratio over "lzo" but requires 4x CPU to compress and 2x CPU to
116  * decompress, comparing to "lzo".
117  * </ul>
118  *
119  * For more on the background behind HFile, see <a
120  * href=https://issues.apache.org/jira/browse/HBASE-61>HBASE-61</a>.
121  * <p>
122  * File is made of data blocks followed by meta data blocks (if any), a fileinfo
123  * block, data block index, meta data block index, and a fixed size trailer
124  * which records the offsets at which file changes content type.
125  * <pre>&lt;data blocks>&lt;meta blocks>&lt;fileinfo>&lt;data index>&lt;meta index>&lt;trailer></pre>
126  * Each block has a bit of magic at its start.  Block are comprised of
127  * key/values.  In data blocks, they are both byte arrays.  Metadata blocks are
128  * a String key and a byte array value.  An empty file looks like this:
129  * <pre>&lt;fileinfo>&lt;trailer></pre>.  That is, there are not data nor meta
130  * blocks present.
131  * <p>
132  * TODO: Do scanners need to be able to take a start and end row?
133  * TODO: Should BlockIndex know the name of its file?  Should it have a Path
134  * that points at its file say for the case where an index lives apart from
135  * an HFile instance?
136  */

195   // For getting more detailed stats on FS latencies
196   // If, for some reason, the metrics subsystem stops polling for latencies, 
197   // I don't want data to pile up in a memory leak
198   // so, after LATENCY_BUFFER_SIZE items have been enqueued for processing,
199   // fs latency stats will be dropped (and this behavior will be logged)
200   private static final int LATENCY_BUFFER_SIZE = 5000;
201   private static final BlockingQueue<Long> fsReadLatenciesNanos = 
202       new ArrayBlockingQueue<Long>(LATENCY_BUFFER_SIZE);
203   private static final BlockingQueue<Long> fsWriteLatenciesNanos = 
204       new ArrayBlockingQueue<Long>(LATENCY_BUFFER_SIZE);
205   private static final BlockingQueue<Long> fsPreadLatenciesNanos = 
206       new ArrayBlockingQueue<Long>(LATENCY_BUFFER_SIZE);
207   
208   public static final void offerReadLatency(long latencyNanos, boolean pread) {
209     if (pread) {
210       fsPreadLatenciesNanos.offer(latencyNanos); // might be silently dropped, if the queue is full
211       preadOps.incrementAndGet();
212       preadTimeNano.addAndGet(latencyNanos);
213     } else {
214       fsReadLatenciesNanos.offer(latencyNanos); // might be silently dropped, if the queue is full
215       readTimeNano.addAndGet(latencyNanos);
216       readOps.incrementAndGet();
217     }
218   }
219   
220   public static final void offerWriteLatency(long latencyNanos) {
221     fsWriteLatenciesNanos.offer(latencyNanos); // might be silently dropped, if the queue is full
222     
223     writeTimeNano.addAndGet(latencyNanos);
224     writeOps.incrementAndGet();
225   }
226   
227   public static final Collection<Long> getReadLatenciesNanos() {
228     final List<Long> latencies = 
229         Lists.newArrayListWithCapacity(fsReadLatenciesNanos.size());
230     fsReadLatenciesNanos.drainTo(latencies);
231     return latencies;
232   }

242     final List<Long> latencies =

533   /**
534    * Method returns the reader given the specified arguments.
535    * TODO This is a bad abstraction.  See HBASE-6635.
536    *
537    * @param path hfile's path
538    * @param fsdis an open checksummed stream of path's file
539    * @param fsdisNoFsChecksum an open unchecksummed stream of path's file
540    * @param size max size of the trailer.
541    * @param closeIStream boolean for closing file after the getting the reader version.
542    * @param cacheConf Cache configuation values, cannot be null.
543    * @param preferredEncodingInCache
544    * @param hfs
545    * @return an appropriate instance of HFileReader
546    * @throws IOException If file is invalid, will throw CorruptHFileException flavored IOException
547    */
548   private static Reader pickReaderVersion(Path path, FSDataInputStream fsdis,
549       FSDataInputStream fsdisNoFsChecksum,
550       long size, boolean closeIStream, CacheConfig cacheConf,
551       DataBlockEncoding preferredEncodingInCache, HFileSystem hfs)
552       throws IOException {
553     FixedFileTrailer trailer = null;
554     try {
555       trailer = FixedFileTrailer.readFromStream(fsdis, size);
556     } catch (IllegalArgumentException iae) {
557       throw new CorruptHFileException("Problem reading HFile Trailer from file " + path, iae);
558     }
559     switch (trailer.getMajorVersion()) {
560     case 1:
561       return new HFileReaderV1(path, trailer, fsdis, size, closeIStream,
562           cacheConf);
563     case 2:
564       return new HFileReaderV2(path, trailer, fsdis, fsdisNoFsChecksum,
565           size, closeIStream,
566           cacheConf, preferredEncodingInCache, hfs);
567     default:
568       throw new CorruptHFileException("Invalid HFile version " + trailer.getMajorVersion());
569     }
570   }

572   /**
573    * @param fs A file system
574    * @param path Path to HFile
575    * @param cacheConf Cache configuration for hfile's contents
576    * @param preferredEncodingInCache Preferred in-cache data encoding algorithm.
577    * @return A version specific Hfile Reader
578    * @throws IOException If file is invalid, will throw CorruptHFileException flavored IOException
579    */
580   public static Reader createReaderWithEncoding(
581       FileSystem fs, Path path, CacheConfig cacheConf,
582       DataBlockEncoding preferredEncodingInCache) throws IOException {
583     final boolean closeIStream = true;
584     HFileSystem hfs = null;
585     FSDataInputStream fsdis = fs.open(path);
586     FSDataInputStream fsdisNoFsChecksum = fsdis;
587     // If the fs is not an instance of HFileSystem, then create an 
588     // instance of HFileSystem that wraps over the specified fs.
589     // In this case, we will not be able to avoid checksumming inside
590     // the filesystem.
591     if (!(fs instanceof HFileSystem)) {
592       hfs = new HFileSystem(fs);
593     } else {
594       hfs = (HFileSystem)fs;
595       // open a stream to read data without checksum verification in
596       // the filesystem
597       fsdisNoFsChecksum = hfs.getNoChecksumFs().open(path);
598     }
599     return pickReaderVersion(path, fsdis, fsdisNoFsChecksum,
600         fs.getFileStatus(path).getLen(), closeIStream, cacheConf,
601         preferredEncodingInCache, hfs);
602   }

783     /**
784      * Write out this instance on the passed in <code>out</code> stream.
785      * We write it as a protobuf.
786      * @param out
787      * @throws IOException
788      * @see {@link #read(DataInputStream)}
789      */
790     void write(final DataOutputStream out) throws IOException {
791       HFileProtos.FileInfoProto.Builder builder = HFileProtos.FileInfoProto.newBuilder();
792       for (Map.Entry<byte [], byte[]> e: this.map.entrySet()) {
793         HBaseProtos.BytesBytesPair.Builder bbpBuilder = HBaseProtos.BytesBytesPair.newBuilder();
794         bbpBuilder.setFirst(ByteString.copyFrom(e.getKey()));
795         bbpBuilder.setSecond(ByteString.copyFrom(e.getValue()));
796         builder.addMapEntry(bbpBuilder.build());
797       }
798       out.write(ProtobufUtil.PB_MAGIC);
799       builder.build().writeDelimitedTo(out);
800     }

802     /**
803      * Populate this instance with what we find on the passed in <code>in</code> stream.
804      * Can deserialize protobuf of old Writables format.
805      * @param in
806      * @throws IOException
807      * @see {@link #write(DataOutputStream)}
808      */
809     void read(final DataInputStream in) throws IOException {
810       // This code is tested over in TestHFileReaderV1 where we read an old hfile w/ this new code.
811       int pblen = ProtobufUtil.lengthOfPBMagic();
812       byte [] pbuf = new byte[pblen];
813       if (in.markSupported()) in.mark(pblen);
814       int read = in.read(pbuf);
815       if (read != pblen) throw new IOException("read=" + read + ", wanted=" + pblen);
816       if (ProtobufUtil.isPBMagicPrefix(pbuf)) {
817         parsePB(HFileProtos.FileInfoProto.parseDelimitedFrom(in));
818       } else {
819         if (in.markSupported()) {
820           in.reset();
821           parseWritable(in);
822         } else {
823           // We cannot use BufferedInputStream, it consumes more than we read from the underlying IS
824           ByteArrayInputStream bais = new ByteArrayInputStream(pbuf);
825           SequenceInputStream sis = new SequenceInputStream(bais, in); // Concatenate input streams
826           // TODO: Am I leaking anything here wrapping the passed in stream?  We are not calling close on the wrapped
827           // streams but they should be let go after we leave this context?  I see that we keep a reference to the
828           // passed in inputstream but since we no longer have a reference to this after we leave, we should be ok.
829           parseWritable(new DataInputStream(sis));
830         }
831       }
832     }
833 
834     /** Now parse the old Writable format.  It was a list of Map entries.  Each map entry was a key and a value of
835      * a byte [].  The old map format had a byte before each entry that held a code which was short for the key or
836      * value type.  We know it was a byte [] so in below we just read and dump it.
837      * @throws IOException 
838      */
839     void parseWritable(final DataInputStream in) throws IOException {
840       // First clear the map.  Otherwise we will just accumulate entries every time this method is called.
841       this.map.clear();
842       // Read the number of entries in the map
843       int entries = in.readInt();
844       // Then read each key/value pair
845       for (int i = 0; i < entries; i++) {
846         byte [] key = Bytes.readByteArray(in);
847         // We used to read a byte that encoded the class type.  Read and ignore it because it is always byte [] in hfile
848         in.readByte();
849         byte [] value = Bytes.readByteArray(in);
850         this.map.put(key, value);
851       }
852     }

854     /**
855      * Fill our map with content of the pb we read off disk
856      * @param fip protobuf message to read
857      */
858     void parsePB(final HFileProtos.FileInfoProto fip) {
859       this.map.clear();
860       for (BytesBytesPair pair: fip.getMapEntryList()) {
861         this.map.put(pair.getFirst().toByteArray(), pair.getSecond().toByteArray());
862       }
