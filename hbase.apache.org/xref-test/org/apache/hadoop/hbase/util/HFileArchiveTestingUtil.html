51    public static boolean compareArchiveToOriginal(FileStatus[] previous, FileStatus[] archived,
52        FileSystem fs, boolean hasTimedBackup) {
53  
54      List<List<String>> lists = getFileLists(previous, archived);
55      List<String> original = lists.get(0);
56      Collections.sort(original);
57  
58      List<String> currentFiles = lists.get(1);
59      Collections.sort(currentFiles);
60  
61      List<String> backedup = lists.get(2);
62      Collections.sort(backedup);
63  
64      // check the backed up files versus the current (should match up, less the
65      // backup time in the name)
66      if (!hasTimedBackup == (backedup.size() > 0)) {
67        LOG.debug("backedup files doesn't match expected.");
68        return false;
69      }
70      String msg = null;
71      if (hasTimedBackup) {
72        msg = assertArchiveEquality(original, backedup);
73        if (msg != null) {
74          LOG.debug(msg);
75          return false;
76        }
77      }
78      msg = assertArchiveEquality(original, currentFiles);
79      if (msg != null) {
80        LOG.debug(msg);
81        return false;
82      }
83      return true;
84    }

98    /**
99     * Compare the archived files to the files in the original directory
100    * @param expected original files that should have been archived
101    * @param actual files that were archived
102    * @param fs {@link FileSystem} on which the archiving took place
103    * @param hasTimedBackup <tt>true</tt> if we expect to find an archive backup directory with a
104    *          copy of the files in the archive directory (and the original files).
105    * @throws IOException
106    */
107   public static void assertArchiveEqualToOriginal(FileStatus[] expected, FileStatus[] actual,
108       FileSystem fs, boolean hasTimedBackup) throws IOException {
109 
110     List<List<String>> lists = getFileLists(expected, actual);
111     List<String> original = lists.get(0);
112     Collections.sort(original);
113 
114     List<String> currentFiles = lists.get(1);
115     Collections.sort(currentFiles);
116 
117     List<String> backedup = lists.get(2);
118     Collections.sort(backedup);
119 
120     // check the backed up files versus the current (should match up, less the
121     // backup time in the name)
122     assertEquals("Didn't expect any backup files, but got: " + backedup, hasTimedBackup,
123       backedup.size() > 0);
124     String msg = null;
125     if (hasTimedBackup) {
126       assertArchiveEquality(original, backedup);
127       assertNull(msg, msg);
128     }
129 
130     // do the rest of the comparison
131     msg = assertArchiveEquality(original, currentFiles);
132     assertNull(msg, msg);
133   }

135   private static String assertArchiveEquality(List<String> expected, List<String> archived) {
136     String compare = compareFileLists(expected, archived);
137     if (!(expected.size() == archived.size())) return "Not the same number of current files\n"
138         + compare;
139     if (!expected.equals(archived)) return "Different backup files, but same amount\n" + compare;
140     return null;
141   }

143   /**
144    * @return <expected, gotten, backup>, where each is sorted
145    */
146   private static List<List<String>> getFileLists(FileStatus[] previous, FileStatus[] archived) {
147     List<List<String>> files = new ArrayList<List<String>>();
148 
149     // copy over the original files
150     List<String> originalFileNames = convertToString(previous);
151     files.add(originalFileNames);
152 
153     List<String> currentFiles = new ArrayList<String>(previous.length);
154     List<FileStatus> backedupFiles = new ArrayList<FileStatus>(previous.length);
155     for (FileStatus f : archived) {
156       String name = f.getPath().getName();
157       // if the file has been backed up
158       if (name.contains(".")) {
159         Path parent = f.getPath().getParent();
160         String shortName = name.split("[.]")[0];
161         Path modPath = new Path(parent, shortName);
162         FileStatus file = new FileStatus(f.getLen(), f.isDir(), f.getReplication(),
163             f.getBlockSize(), f.getModificationTime(), modPath);
164         backedupFiles.add(file);
165       } else {
166         // otherwise, add it to the list to compare to the original store files
167         currentFiles.add(name);
168       }
169     }
170 
171     files.add(currentFiles);
172     files.add(convertToString(backedupFiles));
173     return files;
174   }
175 
176   private static List<String> convertToString(FileStatus[] files) {
177     return convertToString(Arrays.asList(files));
178   }
179 
180   private static List<String> convertToString(List<FileStatus> files) {
181     List<String> originalFileNames = new ArrayList<String>(files.size());
182     for (FileStatus f : files) {
183       originalFileNames.add(f.getPath().getName());
184     }
185     return originalFileNames;
186   }

188   /* Get a pretty representation of the differences */
189   private static String compareFileLists(List<String> expected, List<String> gotten) {
190     StringBuilder sb = new StringBuilder("Expected (" + expected.size() + "): \t\t Gotten ("
191         + gotten.size() + "):\n");
192     List<String> notFound = new ArrayList<String>();
193     for (String s : expected) {
194       if (gotten.contains(s)) sb.append(s + "\t\t" + s + "\n");
195       else notFound.add(s);
196     }
197     sb.append("Not Found:\n");
198     for (String s : notFound) {
199       sb.append(s + "\n");
200     }
201     sb.append("\nExtra:\n");
202     for (String s : gotten) {
203       if (!expected.contains(s)) sb.append(s + "\n");
204     }
205     return sb.toString();
206   }
