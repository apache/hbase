215   @Test
216   public void testRegionScannerReseek() throws Exception {
217     // create new rows and column family to show how reseek works..
218     for (byte[] ROW : ROWS_THREE) {
219       Put p = new Put(ROW);
220       p.setWriteToWAL(true);
221       for (byte[] QUALIFIER : QUALIFIERS_THREE) {
222         p.add(FAMILIES[0], QUALIFIER, VALUES[0]);
223 
224       }
225       this.region.put(p);
226     }
227     for (byte[] ROW : ROWS_FOUR) {
228       Put p = new Put(ROW);
229       p.setWriteToWAL(false);
230       for (byte[] QUALIFIER : QUALIFIERS_FOUR) {
231         p.add(FAMILIES[1], QUALIFIER, VALUES[1]);
232       }
233       this.region.put(p);
234     }
235     // Flush
236     this.region.flushcache();
237 
238     // Insert second half (reverse families)
239     for (byte[] ROW : ROWS_THREE) {
240       Put p = new Put(ROW);
241       p.setWriteToWAL(false);
242       for (byte[] QUALIFIER : QUALIFIERS_THREE) {
243         p.add(FAMILIES[1], QUALIFIER, VALUES[0]);
244       }
245       this.region.put(p);
246     }
247     for (byte[] ROW : ROWS_FOUR) {
248       Put p = new Put(ROW);
249       p.setWriteToWAL(false);
250       for (byte[] QUALIFIER : QUALIFIERS_FOUR) {
251         p.add(FAMILIES[0], QUALIFIER, VALUES[1]);
252       }
253       this.region.put(p);
254     }
255 
256     Scan s = new Scan();
257     // set a start row
258     s.setStartRow(ROWS_FOUR[1]);
259     RegionScanner scanner = region.getScanner(s);
260 
261     // reseek to row three.
262     scanner.reseek(ROWS_THREE[1]);
263     List<KeyValue> results = new ArrayList<KeyValue>();
264 
265     // the results should belong to ROWS_THREE[1]
266     scanner.next(results);
267     for (KeyValue keyValue : results) {
268       assertEquals("The rows with ROWS_TWO as row key should be appearing.",
269           Bytes.toString(keyValue.getRow()), Bytes.toString(ROWS_THREE[1]));
270     }
271     // again try to reseek to a value before ROWS_THREE[1]
272     scanner.reseek(ROWS_ONE[1]);
273     results = new ArrayList<KeyValue>();
274     // This time no seek would have been done to ROWS_ONE[1]
275     scanner.next(results);
276     for (KeyValue keyValue : results) {
277       assertFalse("Cannot rewind back to a value less than previous reseek.",
278           Bytes.toString(keyValue.getRow()).contains("testRowOne"));
279     }
280   }

1376           KeyValue.COMPARATOR);

1386         " rows", expectedRows, i);
