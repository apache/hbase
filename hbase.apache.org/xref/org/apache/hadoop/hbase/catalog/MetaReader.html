54    static final byte [] META_REGION_PREFIX;
55    static {
56      // Copy the prefix from FIRST_META_REGIONINFO into META_REGION_PREFIX.
57      // FIRST_META_REGIONINFO == '.META.,,1'.  META_REGION_PREFIX == '.META.,'
58      int len = HRegionInfo.FIRST_META_REGIONINFO.getRegionName().length - 2;
59      META_REGION_PREFIX = new byte [len];
60      System.arraycopy(HRegionInfo.FIRST_META_REGIONINFO.getRegionName(), 0,
61        META_REGION_PREFIX, 0, len);
62    }

64    /**
65     * @param row
66     * @return True if <code>row</code> is row of <code>-ROOT-</code> table.
67     */
68    private static boolean isRootTableRow(final byte [] row) {
69      if (row.length < META_REGION_PREFIX.length + 2 /* ',', + '1' */) {
70        // Can't be meta table region.
71        return false;
72      }
73      // Compare the prefix of row.  If it matches META_REGION_PREFIX prefix,
74      // then this is row from -ROOT_ table.
75      return Bytes.equals(row, 0, META_REGION_PREFIX.length,
76        META_REGION_PREFIX, 0, META_REGION_PREFIX.length);
77    }

95    /**
96     * Performs a full scan of <code>.META.</code>, skipping regions from any
97     * tables in the specified set of disabled tables.
98     * @param catalogTracker
99     * @param disabledTables set of disabled tables that will not be returned
100    * @param excludeOfflinedSplitParents If true, do not include offlined split
101    * parents in the return.
102    * @return Returns a map of every region to it's currently assigned server,
103    * according to META.  If the region does not have an assignment it will have
104    * a null value in the map.
105    * @throws IOException
106    */
107   public static Map<HRegionInfo, ServerName> fullScan(
108       CatalogTracker catalogTracker, final Set<String> disabledTables,
109       final boolean excludeOfflinedSplitParents)
110   throws IOException {
111     final Map<HRegionInfo, ServerName> regions =
112       new TreeMap<HRegionInfo, ServerName>();
113     Visitor v = new Visitor() {
114       @Override
115       public boolean visit(Result r) throws IOException {
116         if (r ==  null || r.isEmpty()) return true;
117         Pair<HRegionInfo, ServerName> region = HRegionInfo.getHRegionInfoAndServerName(r);
118         if (region == null) return true;
119         HRegionInfo hri = region.getFirst();
120         if (hri  == null) return true;
121         if (hri.getTableNameAsString() == null) return true;
122         if (disabledTables.contains(
123             hri.getTableNameAsString())) return true;
124         // Are we to include split parents in the list?
125         if (excludeOfflinedSplitParents && hri.isSplitParent()) return true;
126         regions.put(hri, region.getSecond());
127         return true;
128       }
129     };
130     fullScan(catalogTracker, v);
131     return regions;
132   }

295   /**
296    * Checks if the specified table exists.  Looks at the META table hosted on
297    * the specified server.
298    * @param catalogTracker
299    * @param tableName table to check
300    * @return true if the table exists in meta, false if not
301    * @throws IOException
302    */
303   public static boolean tableExists(CatalogTracker catalogTracker,
304       String tableName)
305   throws IOException {
306     if (tableName.equals(HTableDescriptor.ROOT_TABLEDESC.getNameAsString()) ||
307         tableName.equals(HTableDescriptor.META_TABLEDESC.getNameAsString())) {
308       // Catalog tables always exist.
309       return true;
310     }
311     final byte [] tableNameBytes = Bytes.toBytes(tableName);
312     // Make a version of ResultCollectingVisitor that only collects the first
313     CollectingVisitor<HRegionInfo> visitor = new CollectingVisitor<HRegionInfo>() {
314       private HRegionInfo current = null;
315 
316       @Override
317       public boolean visit(Result r) throws IOException {
318         this.current =
319           HRegionInfo.getHRegionInfo(r, HConstants.REGIONINFO_QUALIFIER);
320         if (this.current == null) {
321           LOG.warn("No serialized HRegionInfo in " + r);
322           return true;
323         }
324         if (!isInsideTable(this.current, tableNameBytes)) return false;
325         // Else call super and add this Result to the collection.
326         super.visit(r);
327         // Stop collecting regions from table after we get one.
328         return false;
329       }
330 
331       @Override
332       void add(Result r) {
333         // Add the current HRI.
334         this.results.add(this.current);
335       }
336     };
337     fullScan(catalogTracker, visitor, getTableStartRowForMeta(tableNameBytes));
338     // If visitor has results >= 1 then table exists.
339     return visitor.getResults().size() >= 1;
340   }

403     System.arraycopy(tableName, 0, startRow, 0, tableName.length);

409   /**
410    * This method creates a Scan object that will only scan catalog rows that
411    * belong to the specified table. It doesn't specify any columns.
412    * This is a better alternative to just using a start row and scan until
413    * it hits a new table since that requires parsing the HRI to get the table
414    * name.
415    * @param tableName bytes of table's name
416    * @return configured Scan object
417    */
418   public static Scan getScanForTableName(byte[] tableName) {
419     String strName = Bytes.toString(tableName);
420     // Start key is just the table name with delimiters
421     byte[] startKey = Bytes.toBytes(strName + ",,");
422     // Stop key appends the smallest possible char to the table name
423     byte[] stopKey = Bytes.toBytes(strName + " ,,");
424 
425     Scan scan = new Scan(startKey);
426     scan.setStopRow(stopKey);
427     return scan;
428   }

444   /**
445    * @param catalogTracker
446    * @param tableName
447    * @return Return list of regioninfos and server addresses.
448    * @throws IOException
449    * @throws InterruptedException
450    */
451   public static List<Pair<HRegionInfo, ServerName>>
452   getTableRegionsAndLocations(final CatalogTracker catalogTracker,
453       final byte [] tableName, final boolean excludeOfflinedSplitParents)
454   throws IOException, InterruptedException {
455     if (Bytes.equals(tableName, HConstants.ROOT_TABLE_NAME)) {
456       // If root, do a bit of special handling.
457       ServerName serverName = catalogTracker.getRootLocation();
458       List<Pair<HRegionInfo, ServerName>> list =
459         new ArrayList<Pair<HRegionInfo, ServerName>>();
460       list.add(new Pair<HRegionInfo, ServerName>(HRegionInfo.ROOT_REGIONINFO,
461         serverName));
462       return list;
463     }
464     // Make a version of CollectingVisitor that collects HRegionInfo and ServerAddress
465     CollectingVisitor<Pair<HRegionInfo, ServerName>> visitor =
466         new CollectingVisitor<Pair<HRegionInfo, ServerName>>() {
467       private Pair<HRegionInfo, ServerName> current = null;
468 
469       @Override
470       public boolean visit(Result r) throws IOException {
471         HRegionInfo hri =
472           HRegionInfo.getHRegionInfo(r, HConstants.REGIONINFO_QUALIFIER);
473         if (hri == null) {
474           LOG.warn("No serialized HRegionInfo in " + r);
475           return true;
476         }
477         if (!isInsideTable(hri, tableName)) return false;
478         if (excludeOfflinedSplitParents && hri.isSplitParent()) return true;
479         ServerName sn = HRegionInfo.getServerName(r);
480         // Populate this.current so available when we call #add
481         this.current = new Pair<HRegionInfo, ServerName>(hri, sn);
482         // Else call super and add this Result to the collection.
483         return super.visit(r);
484       }
485 
486       @Override
487       void add(Result r) {
488         this.results.add(this.current);
489       }
490     };
491     fullScan(catalogTracker, visitor, getTableStartRowForMeta(tableName),
492       Bytes.equals(tableName, HConstants.META_TABLE_NAME));
493     return visitor.getResults();
494   }

496   /**
497    * @param catalogTracker
498    * @param serverName
499    * @return List of user regions installed on this server (does not include
500    * catalog regions).
501    * @throws IOException
502    */
503   public static NavigableMap<HRegionInfo, Result>
504   getServerUserRegions(CatalogTracker catalogTracker, final ServerName serverName)
505   throws IOException {
506     final NavigableMap<HRegionInfo, Result> hris = new TreeMap<HRegionInfo, Result>();
507     // Fill the above hris map with entries from .META. that have the passed
508     // servername.
509     CollectingVisitor<Result> v = new CollectingVisitor<Result>() {
510       @Override
511       void add(Result r) {
512         if (r == null || r.isEmpty()) return;
513         ServerName sn = HRegionInfo.getServerName(r);
514         if (sn != null && sn.equals(serverName)) this.results.add(r);
515       }
516     };
517     fullScan(catalogTracker, v);
518     List<Result> results = v.getResults();
519     if (results != null && !results.isEmpty()) {
520       // Convert results to Map keyed by HRI
521       for (Result r: results) {
522         Pair<HRegionInfo, ServerName> p = HRegionInfo.getHRegionInfoAndServerName(r);
523         if (p != null && p.getFirst() != null) hris.put(p.getFirst(), r);
524       }
525     }
526     return hris;
527   }

544   /**
545    * Performs a full scan of a catalog table.
546    * @param catalogTracker
547    * @param visitor Visitor invoked against each row.
548    * @param startrow Where to start the scan. Pass null if want to begin scan
549    * at first row.
550    * @param scanRoot True if we are to scan <code>-ROOT-</code> rather than
551    * <code>.META.</code>, the default (pass false to scan .META.)
552    * @throws IOException
553    */
554   static void fullScan(CatalogTracker catalogTracker,
555     final Visitor visitor, final byte [] startrow, final boolean scanRoot)
556   throws IOException {
557     Scan scan = new Scan();
558     if (startrow != null) scan.setStartRow(startrow);
559     if (startrow == null && !scanRoot) {
560       int caching = catalogTracker.getConnection().getConfiguration()
561           .getInt(HConstants.HBASE_META_SCANNER_CACHING, 100);
562       scan.setCaching(caching);
563     }
564     scan.addFamily(HConstants.CATALOG_FAMILY);
565     HTable metaTable = scanRoot?
566       getRootHTable(catalogTracker): getMetaHTable(catalogTracker);
567     ResultScanner scanner = metaTable.getScanner(scan);
568     try {
569       Result data;
570       while((data = scanner.next()) != null) {
571         if (data.isEmpty()) continue;
572         // Break if visit returns false.
573         if (!visitor.visit(data)) break;
574       }
575     } finally {
576       scanner.close();
577       metaTable.close();
578     }
579     return;
580   }

595   /**
596    * A {@link Visitor} that collects content out of passed {@link Result}.
597    */
598   static abstract class CollectingVisitor<T> implements Visitor {
599     final List<T> results = new ArrayList<T>();
600     @Override
601     public boolean visit(Result r) throws IOException {
602       if (r ==  null || r.isEmpty()) return true;
603       add(r);
604       return true;
605     }
606 
607     abstract void add(Result r);
608 
609     /**
610      * @return Collected results; wait till visits complete to collect all
611      * possible results
612      */
613     List<T> getResults() {
614       return this.results;
615     }
616   }
