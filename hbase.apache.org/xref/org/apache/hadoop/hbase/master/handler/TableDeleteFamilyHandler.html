34  /**
35   * Handles adding a new family to an existing table.
36   */
37  @InterfaceAudience.Private
38  public class TableDeleteFamilyHandler extends TableEventHandler {
39  
40    private final byte [] familyName;
41  
42    public TableDeleteFamilyHandler(byte[] tableName, byte [] familyName,
43        Server server, final MasterServices masterServices) throws IOException {
44      super(EventType.C_M_ADD_FAMILY, tableName, server, masterServices);
45      HTableDescriptor htd = getTableDescriptor();
46      this.familyName = hasColumnFamily(htd, familyName);
47    }
48  
49    @Override
50    protected void handleTableOperation(List<HRegionInfo> hris) throws IOException {
51      MasterCoprocessorHost cpHost = ((HMaster) this.server)
52          .getCoprocessorHost();
53      if (cpHost != null) {
54        cpHost.preDeleteColumnHandler(this.tableName, this.familyName);
55      }
56      // Update table descriptor in HDFS
57      HTableDescriptor htd =
58        this.masterServices.getMasterFileSystem().deleteColumn(tableName, familyName);
59      // Update in-memory descriptor cache
60      this.masterServices.getTableDescriptors().add(htd);
61      // Remove the column family from the file system
62      MasterFileSystem mfs = this.masterServices.getMasterFileSystem();
63      for (HRegionInfo hri : hris) {
64        // Delete the family directory in FS for all the regions one by one
65        mfs.deleteFamilyFromFS(hri, familyName);
66      }
67      if (cpHost != null) {
68        cpHost.postDeleteColumnHandler(this.tableName, this.familyName);
69      }
70    }
71  
72    @Override
73    public String toString() {
74      String name = "UnknownServerName";
75      if(server != null && server.getServerName() != null) {
76        name = server.getServerName().toString();
77      }
78      String family = "UnknownFamily";
79      if(familyName != null) {
80        family = Bytes.toString(familyName);
81      }
82      return getClass().getSimpleName() + "-" + name + "-" + getSeqid() + "-" + tableNameStr + "-" + family;
83    }
84  }
