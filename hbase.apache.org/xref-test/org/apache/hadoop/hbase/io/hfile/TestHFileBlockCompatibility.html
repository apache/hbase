100   public byte[] createTestV1Block(Compression.Algorithm algo)
101       throws IOException {
102     Compressor compressor = algo.getCompressor();
103     ByteArrayOutputStream baos = new ByteArrayOutputStream();
104     OutputStream os = algo.createCompressionStream(baos, compressor, 0);
105     DataOutputStream dos = new DataOutputStream(os);
106     BlockType.META.write(dos); // Let's make this a meta block.
107     TestHFileBlock.writeTestBlockContents(dos);
108     uncompressedSizeV1 = dos.size();
109     dos.flush();
110     algo.returnCompressor(compressor);
111     return baos.toByteArray();
112   }

114   private Writer createTestV2Block(Compression.Algorithm algo)
115       throws IOException {
116     final BlockType blockType = BlockType.DATA;
117     Writer hbw = new Writer(algo, null,
118         includesMemstoreTS);
119     DataOutputStream dos = hbw.startWriting(blockType);
120     TestHFileBlock.writeTestBlockContents(dos);
121     // make sure the block is ready by calling hbw.getHeaderAndData()
122     hbw.getHeaderAndData();
123     assertEquals(1000 * 4, hbw.getUncompressedSizeWithoutHeader());
124     hbw.releaseCompressor();
125     return hbw;
126   }

128  private String createTestBlockStr(Compression.Algorithm algo,
129       int correctLength) throws IOException {
130     Writer hbw = createTestV2Block(algo);
131     byte[] testV2Block = hbw.getHeaderAndData();
132     int osOffset = HFileBlock.HEADER_SIZE_NO_CHECKSUM + 9;
133     if (testV2Block.length == correctLength) {
134       // Force-set the "OS" field of the gzip header to 3 (Unix) to avoid
135       // variations across operating systems.
136       // See http://www.gzip.org/zlib/rfc-gzip.html for gzip format.
137       testV2Block[osOffset] = 3;
138     }
139     return Bytes.toStringBinary(testV2Block);
140   }

148   @Test
149   public void testGzipCompression() throws IOException {
150     final String correctTestBlockStr =
151         "DATABLK*\\x00\\x00\\x00:\\x00\\x00\\x0F\\xA0\\xFF\\xFF\\xFF\\xFF"
152             + "\\xFF\\xFF\\xFF\\xFF"
153             // gzip-compressed block: http://www.gzip.org/zlib/rfc-gzip.html
154             + "\\x1F\\x8B"  // gzip magic signature
155             + "\\x08"  // Compression method: 8 = "deflate"
156             + "\\x00"  // Flags
157             + "\\x00\\x00\\x00\\x00"  // mtime
158             + "\\x00"  // XFL (extra flags)
159             // OS (0 = FAT filesystems, 3 = Unix). However, this field
160             // sometimes gets set to 0 on Linux and Mac, so we reset it to 3.
161             + "\\x03"
162             + "\\xED\\xC3\\xC1\\x11\\x00 \\x08\\xC00DD\\xDD\\x7Fa"
163             + "\\xD6\\xE8\\xA3\\xB9K\\x84`\\x96Q\\xD3\\xA8\\xDB\\xA8e\\xD4c"
164             + "\\xD46\\xEA5\\xEA3\\xEA7\\xE7\\x00LI\\x5Cs\\xA0\\x0F\\x00\\x00";
165     final int correctGzipBlockLength = 82;
166 
167     String returnedStr = createTestBlockStr(GZ, correctGzipBlockLength);
168     assertEquals(correctTestBlockStr, returnedStr);
169   }

171   @Test
172   public void testReaderV1() throws IOException {
173     for (Compression.Algorithm algo : COMPRESSION_ALGORITHMS) {
174       for (boolean pread : new boolean[] { false, true }) {
175         byte[] block = createTestV1Block(algo);
176         Path path = new Path(TEST_UTIL.getDataTestDir(),
177           "blocks_v1_"+ algo);
178         LOG.info("Creating temporary file at " + path);
179         FSDataOutputStream os = fs.create(path);
180         int totalSize = 0;
181         int numBlocks = 50;
182         for (int i = 0; i < numBlocks; ++i) {
183           os.write(block);
184           totalSize += block.length;
185         }
186         os.close();
187 
188         FSDataInputStream is = fs.open(path);
189         HFileBlock.FSReader hbr = new HFileBlock.FSReaderV1(is, algo,
190             totalSize);
191         HFileBlock b;
192         int numBlocksRead = 0;
193         long pos = 0;
194         while (pos < totalSize) {
195           b = hbr.readBlockData(pos, block.length, uncompressedSizeV1, pread);
196           b.sanityCheck();
197           pos += block.length;
198           numBlocksRead++;
199         }
200         assertEquals(numBlocks, numBlocksRead);
201         is.close();
202       }
203     }
204   }

206   @Test
207   public void testReaderV2() throws IOException {
208     for (Compression.Algorithm algo : COMPRESSION_ALGORITHMS) {
209       for (boolean pread : new boolean[] { false, true }) {
210           LOG.info("testReaderV2: Compression algorithm: " + algo + 
211                    ", pread=" + pread);
212         Path path = new Path(TEST_UTIL.getDataTestDir(), "blocks_v2_"
213             + algo);
214         FSDataOutputStream os = fs.create(path);
215         Writer hbw = new Writer(algo, null,
216             includesMemstoreTS);
217         long totalSize = 0;
218         for (int blockId = 0; blockId < 2; ++blockId) {
219           DataOutputStream dos = hbw.startWriting(BlockType.DATA);
220           for (int i = 0; i < 1234; ++i)
221             dos.writeInt(i);
222           hbw.writeHeaderAndData(os);
223           totalSize += hbw.getOnDiskSizeWithHeader();
224         }
225         os.close();
226 
227         FSDataInputStream is = fs.open(path);
228         HFileBlock.FSReader hbr = new HFileBlock.FSReaderV2(is, is, algo,
229             totalSize, MINOR_VERSION, fs, path);
230         HFileBlock b = hbr.readBlockData(0, -1, -1, pread);
231         is.close();
232 
233         b.sanityCheck();
234         assertEquals(4936, b.getUncompressedSizeWithoutHeader());
235         assertEquals(algo == GZ ? 2173 : 4936, 
236                      b.getOnDiskSizeWithoutHeader() - b.totalChecksumBytes());
237         String blockStr = b.toString();
238 
239         if (algo == GZ) {
240           is = fs.open(path);
241           hbr = new HFileBlock.FSReaderV2(is, is, algo, totalSize, MINOR_VERSION,
242                                           fs, path);
243           b = hbr.readBlockData(0, 2173 + HFileBlock.HEADER_SIZE_NO_CHECKSUM +
244                                 b.totalChecksumBytes(), -1, pread);
245           assertEquals(blockStr, b.toString());
246           int wrongCompressedSize = 2172;
247           try {
248             b = hbr.readBlockData(0, wrongCompressedSize
249                 + HFileBlock.HEADER_SIZE_NO_CHECKSUM, -1, pread);
250             fail("Exception expected");
251           } catch (IOException ex) {
252             String expectedPrefix = "On-disk size without header provided is "
253                 + wrongCompressedSize + ", but block header contains "
254                 + b.getOnDiskSizeWithoutHeader() + ".";
255             assertTrue("Invalid exception message: '" + ex.getMessage()
256                 + "'.\nMessage is expected to start with: '" + expectedPrefix
257                 + "'", ex.getMessage().startsWith(expectedPrefix));
258           }
259           is.close();
260         }
261       }
262     }

267    * @throws IOException a bug or a problem with temporary files.

269   @Test
270   public void testDataBlockEncoding() throws IOException {
271     final int numBlocks = 5;
272     for (Compression.Algorithm algo : COMPRESSION_ALGORITHMS) {
273       for (boolean pread : new boolean[] { false, true }) {
274         for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
275           LOG.info("testDataBlockEncoding algo " + algo + 
276                    " pread = " + pread +
277                    " encoding " + encoding);
278           Path path = new Path(TEST_UTIL.getDataTestDir(), "blocks_v2_"
279               + algo + "_" + encoding.toString());
280           FSDataOutputStream os = fs.create(path);
281           HFileDataBlockEncoder dataBlockEncoder =
282               new HFileDataBlockEncoderImpl(encoding, encoding,
283                   TestHFileBlockCompatibility.Writer.DUMMY_HEADER);
284           TestHFileBlockCompatibility.Writer hbw =
285               new TestHFileBlockCompatibility.Writer(algo,
286                   dataBlockEncoder, includesMemstoreTS);
287           long totalSize = 0;
288           final List<Integer> encodedSizes = new ArrayList<Integer>();
289           final List<ByteBuffer> encodedBlocks = new ArrayList<ByteBuffer>();
290           for (int blockId = 0; blockId < numBlocks; ++blockId) {
291             DataOutputStream dos = hbw.startWriting(BlockType.DATA);
292             TestHFileBlock.writeEncodedBlock(algo, encoding, dos, encodedSizes,
293                 encodedBlocks, blockId, includesMemstoreTS,
294                 TestHFileBlockCompatibility.Writer.DUMMY_HEADER);
295 
296             hbw.writeHeaderAndData(os);
297             totalSize += hbw.getOnDiskSizeWithHeader();
298           }
299           os.close();
300 
301           FSDataInputStream is = fs.open(path);
302           HFileBlock.FSReaderV2 hbr = new HFileBlock.FSReaderV2(is, is, algo,
303               totalSize, MINOR_VERSION, fs, path);
304           hbr.setDataBlockEncoder(dataBlockEncoder);
305           hbr.setIncludesMemstoreTS(includesMemstoreTS);
306 
307           HFileBlock b;
308           int pos = 0;
309           for (int blockId = 0; blockId < numBlocks; ++blockId) {
310             b = hbr.readBlockData(pos, -1, -1, pread);
311             b.sanityCheck();
312             pos += b.getOnDiskSizeWithHeader();
313 
314             assertEquals((int) encodedSizes.get(blockId),
315                 b.getUncompressedSizeWithoutHeader());
316             ByteBuffer actualBuffer = b.getBufferWithoutHeader();
317             if (encoding != DataBlockEncoding.NONE) {
318               // We expect a two-byte big-endian encoding id.
319               assertEquals(0, actualBuffer.get(0));
320               assertEquals(encoding.getId(), actualBuffer.get(1));
321               actualBuffer.position(2);
322               actualBuffer = actualBuffer.slice();
323             }
324 
325             ByteBuffer expectedBuffer = encodedBlocks.get(blockId);
326             expectedBuffer.rewind();
327 
328             // test if content matches, produce nice message
329             TestHFileBlock.assertBuffersEqual(expectedBuffer, actualBuffer,
330               algo, encoding, pread);
331           }
332           is.close();
333         }
334       }
335     }

340   /**
341    * This is the version of the HFileBlock.Writer that is used to 
342    * create V2 blocks with minor version 0. These blocks do not 
343    * have hbase-level checksums. The code is here to test 
344    * backward compatibility. The reason we do not inherit from 
345    * HFileBlock.Writer is because we never ever want to change the code
346    * in this class but the code in HFileBlock.Writer will continually 
347    * evolve.
348    */
349   public static final class Writer {
350 
351     // These constants are as they were in minorVersion 0.
352     private static final int HEADER_SIZE = HFileBlock.HEADER_SIZE_NO_CHECKSUM;
353     private static final boolean DONT_FILL_HEADER = HFileBlock.DONT_FILL_HEADER;
354     private static final byte[] DUMMY_HEADER = 
355       HFileBlock.DUMMY_HEADER_NO_CHECKSUM;
356 
357     private enum State {
358       INIT,
359       WRITING,
360       BLOCK_READY
361     };
362 
363     /** Writer state. Used to ensure the correct usage protocol. */
364     private State state = State.INIT;
365 
366     /** Compression algorithm for all blocks this instance writes. */
367     private final Compression.Algorithm compressAlgo;
368 
369     /** Data block encoder used for data blocks */
370     private final HFileDataBlockEncoder dataBlockEncoder;
371 
372     private HFileBlockEncodingContext dataBlockEncodingCtx;
373     /** block encoding context for non-data blocks */
374     private HFileBlockDefaultEncodingContext defaultBlockEncodingCtx;
375 
376     /**
377      * The stream we use to accumulate data in uncompressed format for each
378      * block. We reset this stream at the end of each block and reuse it. The
379      * header is written as the first {@link #HEADER_SIZE} bytes into this
380      * stream.
381      */
382     private ByteArrayOutputStream baosInMemory;
383 
384     /** Compressor, which is also reused between consecutive blocks. */
385     private Compressor compressor;
386 
387     /**
388      * Current block type. Set in {@link #startWriting(BlockType)}. Could be
389      * changed in {@link #encodeDataBlockForDisk()} from {@link BlockType#DATA}
390      * to {@link BlockType#ENCODED_DATA}.
391      */
392     private BlockType blockType;
393 
394     /**
395      * A stream that we write uncompressed bytes to, which compresses them and
396      * writes them to {@link #baosInMemory}.
397      */
398     private DataOutputStream userDataStream;
399 
400     /**
401      * Bytes to be written to the file system, including the header. Compressed
402      * if compression is turned on.
403      */
404     private byte[] onDiskBytesWithHeader;
405 
406     /**
407      * Valid in the READY state. Contains the header and the uncompressed (but
408      * potentially encoded, if this is a data block) bytes, so the length is
409      * {@link #uncompressedSizeWithoutHeader} + {@link HFileBlock#HEADER_SIZE}.
410      */
411     private byte[] uncompressedBytesWithHeader;
412 
413     /**
414      * Current block's start offset in the {@link HFile}. Set in
415      * {@link #writeHeaderAndData(FSDataOutputStream)}.
416      */
417     private long startOffset;
418 
419     /**
420      * Offset of previous block by block type. Updated when the next block is
421      * started.
422      */
423     private long[] prevOffsetByType;
424 
425     /** The offset of the previous block of the same type */
426     private long prevOffset;
427 
428     /** Whether we are including memstore timestamp after every key/value */
429     private boolean includesMemstoreTS;
430 
431     /**
432      * @param compressionAlgorithm compression algorithm to use
433      * @param dataBlockEncoderAlgo data block encoding algorithm to use
434      */
435     public Writer(Compression.Algorithm compressionAlgorithm,
436           HFileDataBlockEncoder dataBlockEncoder, boolean includesMemstoreTS) {
437       compressAlgo = compressionAlgorithm == null ? NONE : compressionAlgorithm;
438       this.dataBlockEncoder = dataBlockEncoder != null
439           ? dataBlockEncoder : NoOpDataBlockEncoder.INSTANCE;
440 
441       defaultBlockEncodingCtx =
442           new HFileBlockDefaultEncodingContext(compressionAlgorithm,
443               null, DUMMY_HEADER);
444       dataBlockEncodingCtx =
445         this.dataBlockEncoder.newOnDiskDataBlockEncodingContext(
446             compressionAlgorithm, DUMMY_HEADER);
447 
448       baosInMemory = new ByteArrayOutputStream();
449 
450       prevOffsetByType = new long[BlockType.values().length];
451       for (int i = 0; i < prevOffsetByType.length; ++i)
452         prevOffsetByType[i] = -1;
453 
454       this.includesMemstoreTS = includesMemstoreTS;
455     }
456 
457     /**
458      * Starts writing into the block. The previous block's data is discarded.
459      *
460      * @return the stream the user can write their data into
461      * @throws IOException
462      */
463     public DataOutputStream startWriting(BlockType newBlockType)
464         throws IOException {
465       if (state == State.BLOCK_READY && startOffset != -1) {
466         // We had a previous block that was written to a stream at a specific
467         // offset. Save that offset as the last offset of a block of that type.
468         prevOffsetByType[blockType.getId()] = startOffset;
469       }
470 
471       startOffset = -1;
472       blockType = newBlockType;
473 
474       baosInMemory.reset();
475       baosInMemory.write(DUMMY_HEADER);
476 
477       state = State.WRITING;
478 
479       // We will compress it later in finishBlock()
480       userDataStream = new DataOutputStream(baosInMemory);
481       return userDataStream;
482     }
483 
484     /**
485      * Returns the stream for the user to write to. The block writer takes care
486      * of handling compression and buffering for caching on write. Can only be
487      * called in the "writing" state.
488      *
489      * @return the data output stream for the user to write to
490      */
491     DataOutputStream getUserDataStream() {
492       expectState(State.WRITING);
493       return userDataStream;
494     }
495 
496     /**
497      * Transitions the block writer from the "writing" state to the "block
498      * ready" state.  Does nothing if a block is already finished.
499      */
500     private void ensureBlockReady() throws IOException {
501       Preconditions.checkState(state != State.INIT,
502           "Unexpected state: " + state);
503 
504       if (state == State.BLOCK_READY)
505         return;
506 
507       // This will set state to BLOCK_READY.
508       finishBlock();
509     }
510 
511     /**
512      * An internal method that flushes the compressing stream (if using
513      * compression), serializes the header, and takes care of the separate
514      * uncompressed stream for caching on write, if applicable. Sets block
515      * write state to "block ready".
516      */
517     private void finishBlock() throws IOException {
518       userDataStream.flush();
519       // This does an array copy, so it is safe to cache this byte array.
520       uncompressedBytesWithHeader = baosInMemory.toByteArray();
521       prevOffset = prevOffsetByType[blockType.getId()];
522 
523       // We need to set state before we can package the block up for
524       // cache-on-write. In a way, the block is ready, but not yet encoded or
525       // compressed.
526       state = State.BLOCK_READY;
527       if (blockType == BlockType.DATA) {
528         encodeDataBlockForDisk();
529       } else {
530         defaultBlockEncodingCtx.compressAfterEncoding(
531             uncompressedBytesWithHeader, blockType);
532         onDiskBytesWithHeader =
533           defaultBlockEncodingCtx.getOnDiskBytesWithHeader();
534       }
535 
536       // put the header for on disk bytes
537       putHeader(onDiskBytesWithHeader, 0,
538           onDiskBytesWithHeader.length,
539           uncompressedBytesWithHeader.length);
540       //set the header for the uncompressed bytes (for cache-on-write)
541       putHeader(uncompressedBytesWithHeader, 0,
542           onDiskBytesWithHeader.length,
543         uncompressedBytesWithHeader.length);
544     }
545 
546     /**
547      * Encodes this block if it is a data block and encoding is turned on in
548      * {@link #dataBlockEncoder}.
549      */
550     private void encodeDataBlockForDisk() throws IOException {
551       // do data block encoding, if data block encoder is set
552       ByteBuffer rawKeyValues =
553           ByteBuffer.wrap(uncompressedBytesWithHeader, HEADER_SIZE,
554               uncompressedBytesWithHeader.length - HEADER_SIZE).slice();
555 
556       //do the encoding
557       dataBlockEncoder.beforeWriteToDisk(rawKeyValues,
558               includesMemstoreTS, dataBlockEncodingCtx, blockType);
559 
560       uncompressedBytesWithHeader =
561           dataBlockEncodingCtx.getUncompressedBytesWithHeader();
562       onDiskBytesWithHeader =
563           dataBlockEncodingCtx.getOnDiskBytesWithHeader();
564       blockType = dataBlockEncodingCtx.getBlockType();
565     }
566 
567     /**
568      * Put the header into the given byte array at the given offset.
569      * @param onDiskSize size of the block on disk
570      * @param uncompressedSize size of the block after decompression (but
571      *          before optional data block decoding)
572      */
573     private void putHeader(byte[] dest, int offset, int onDiskSize,
574         int uncompressedSize) {
575       offset = blockType.put(dest, offset);
576       offset = Bytes.putInt(dest, offset, onDiskSize - HEADER_SIZE);
577       offset = Bytes.putInt(dest, offset, uncompressedSize - HEADER_SIZE);
578       Bytes.putLong(dest, offset, prevOffset);
579     }
580 
581     /**
582      * Similar to {@link #writeHeaderAndData(FSDataOutputStream)}, but records
583      * the offset of this block so that it can be referenced in the next block
584      * of the same type.
585      *
586      * @param out
587      * @throws IOException
588      */
589     public void writeHeaderAndData(FSDataOutputStream out) throws IOException {
590       long offset = out.getPos();
591       if (startOffset != -1 && offset != startOffset) {
592         throw new IOException("A " + blockType + " block written to a "
593             + "stream twice, first at offset " + startOffset + ", then at "
594             + offset);
595       }
596       startOffset = offset;
597 
598       writeHeaderAndData((DataOutputStream) out);
599     }
600 
601     /**
602      * Writes the header and the compressed data of this block (or uncompressed
603      * data when not using compression) into the given stream. Can be called in
604      * the "writing" state or in the "block ready" state. If called in the
605      * "writing" state, transitions the writer to the "block ready" state.
606      *
607      * @param out the output stream to write the
608      * @throws IOException
609      */
610     private void writeHeaderAndData(DataOutputStream out) throws IOException {
611       ensureBlockReady();
612       out.write(onDiskBytesWithHeader);
613     }
614 
615     /**
616      * Returns the header or the compressed data (or uncompressed data when not
617      * using compression) as a byte array. Can be called in the "writing" state
618      * or in the "block ready" state. If called in the "writing" state,
619      * transitions the writer to the "block ready" state.
620      *
621      * @return header and data as they would be stored on disk in a byte array
622      * @throws IOException
623      */
624     public byte[] getHeaderAndData() throws IOException {
625       ensureBlockReady();
626       return onDiskBytesWithHeader;
627     }
628 
629     /**
630      * Releases the compressor this writer uses to compress blocks into the
631      * compressor pool. Needs to be called before the writer is discarded.
632      */
633     public void releaseCompressor() {
634       if (compressor != null) {
635         compressAlgo.returnCompressor(compressor);
636         compressor = null;
637       }
638     }
639 
640     /**
641      * Returns the on-disk size of the data portion of the block. This is the
642      * compressed size if compression is enabled. Can only be called in the
643      * "block ready" state. Header is not compressed, and its size is not
