43  /**
44   * Tests that need to spin up a cluster testing an {@link HRegion}.  Use
45   * {@link TestHRegion} if you don't need a cluster, if you can test w/ a
46   * standalone {@link HRegion}.
47   */
48  @Category(MediumTests.class)
49  public class TestHRegionOnCluster {
50    private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
51  
52    @Test (timeout=180000)
53    public void testDataCorrectnessReplayingRecoveredEdits() throws Exception {
54      final int NUM_MASTERS = 1;
55      final int NUM_RS = 3;
56      TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
57  
58      try {
59        final byte[] TABLENAME = Bytes
60            .toBytes("testDataCorrectnessReplayingRecoveredEdits");
61        final byte[] FAMILY = Bytes.toBytes("family");
62        MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
63        HMaster master = cluster.getMaster();
64  
65        // Create table
66        HTableDescriptor desc = new HTableDescriptor(TABLENAME);
67        desc.addFamily(new HColumnDescriptor(FAMILY));
68        HBaseAdmin hbaseAdmin = TEST_UTIL.getHBaseAdmin();
69        hbaseAdmin.createTable(desc);
70  
71        assertTrue(hbaseAdmin.isTableAvailable(TABLENAME));
72  
73        // Put data: r1->v1
74        HTable table = new HTable(TEST_UTIL.getConfiguration(), TABLENAME);
75        putDataAndVerify(table, "r1", FAMILY, "v1", 1);
76  
77        // Move region to target server
78        HRegionInfo regionInfo = table.getRegionLocation("r1").getRegionInfo();
79        int originServerNum = cluster.getServerWith(regionInfo.getRegionName());
80        HRegionServer originServer = cluster.getRegionServer(originServerNum);
81        int targetServerNum = (originServerNum + 1) % NUM_RS;
82        HRegionServer targetServer = cluster.getRegionServer(targetServerNum);
83        assertFalse(originServer.equals(targetServer));
84        hbaseAdmin.move(regionInfo.getEncodedNameAsBytes(),
85            Bytes.toBytes(targetServer.getServerName().getServerName()));
86        do {
87          Thread.sleep(1);
88        } while (cluster.getServerWith(regionInfo.getRegionName()) == originServerNum);
89  
90        // Put data: r2->v2
91        putDataAndVerify(table, "r2", FAMILY, "v2", 2);
92  
93        // Move region to origin server
94        hbaseAdmin.move(regionInfo.getEncodedNameAsBytes(),
95            Bytes.toBytes(originServer.getServerName().getServerName()));
96        do {
97          Thread.sleep(1);
98        } while (cluster.getServerWith(regionInfo.getRegionName()) == targetServerNum);
99  
100       // Put data: r3->v3
101       putDataAndVerify(table, "r3", FAMILY, "v3", 3);
102 
103       // Kill target server
104       targetServer.kill();
105       cluster.getRegionServerThreads().get(targetServerNum).join();
106       // Wait until finish processing of shutdown
107       while (master.getServerManager().areDeadServersInProgress()) {
108         Thread.sleep(5);
109       }
110       // Kill origin server
111       originServer.kill();
112       cluster.getRegionServerThreads().get(originServerNum).join();
113 
114       // Put data: r4->v4
115       putDataAndVerify(table, "r4", FAMILY, "v4", 4);
116 
117     } finally {
118       TEST_UTIL.shutdownMiniCluster();
119     }
120   }
121 
122   private void putDataAndVerify(HTable table, String row, byte[] family,
123       String value, int verifyNum) throws IOException {
124     System.out.println("=========Putting data :" + row);
125     Put put = new Put(Bytes.toBytes(row));
126     put.add(family, Bytes.toBytes("q1"), Bytes.toBytes(value));
127     table.put(put);
128     ResultScanner resultScanner = table.getScanner(new Scan());
129     List<Result> results = new ArrayList<Result>();
130     while (true) {
131       Result r = resultScanner.next();
132       if (r == null)
133         break;
134       results.add(r);
135     }
136     resultScanner.close();
137     if (results.size() != verifyNum) {
138       System.out.println(results);
139     }
140     assertEquals(verifyNum, results.size());
141   }
