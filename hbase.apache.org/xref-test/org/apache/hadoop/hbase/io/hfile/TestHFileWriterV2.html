84    @Test
85    public void testMidKeyInHFile() throws IOException{
86      Path hfilePath = new Path(TEST_UTIL.getDataTestDir(),
87      "testMidKeyInHFile");

89      int entryCount = 50000;

93    private void writeDataAndReadFromHFile(Path hfilePath,
94        Algorithm compressAlgo, int entryCount, boolean findMidKey) throws IOException {
95  
96      HFileWriterV2 writer = (HFileWriterV2)
97          new HFileWriterV2.WriterFactoryV2(conf, new CacheConfig(conf))
98              .withPath(fs, hfilePath)
99              .withBlockSize(4096)
100             .withCompression(compressAlgo)
101             .withComparator(KeyValue.KEY_COMPARATOR)
102             .create();
103 
104     Random rand = new Random(9713312); // Just a fixed seed.
105 
106     List<byte[]> keys = new ArrayList<byte[]>();
107     List<byte[]> values = new ArrayList<byte[]>();
108 
109     for (int i = 0; i < entryCount; ++i) {
110       byte[] keyBytes = randomOrderedKey(rand, i);
111 
112       // A random-length random value.
113       byte[] valueBytes = randomValue(rand);
114       writer.append(keyBytes, valueBytes);
115 
116       keys.add(keyBytes);
117       values.add(valueBytes);
118     }
119 
120     // Add in an arbitrary order. They will be sorted lexicographically by
121     // the key.
122     writer.appendMetaBlock("CAPITAL_OF_USA", new Text("Washington, D.C."));
123     writer.appendMetaBlock("CAPITAL_OF_RUSSIA", new Text("Moscow"));
124     writer.appendMetaBlock("CAPITAL_OF_FRANCE", new Text("Paris"));
125 
126     writer.close();
127     
128 
129     FSDataInputStream fsdis = fs.open(hfilePath);
130 
131     // A "manual" version of a new-format HFile reader. This unit test was
132     // written before the V2 reader was fully implemented.
133 
134     long fileSize = fs.getFileStatus(hfilePath).getLen();
135     FixedFileTrailer trailer =
136         FixedFileTrailer.readFromStream(fsdis, fileSize);
137 
138     assertEquals(2, trailer.getMajorVersion());
139     assertEquals(entryCount, trailer.getEntryCount());
140 
141     HFileBlock.FSReader blockReader =
142         new HFileBlock.FSReaderV2(fsdis, compressAlgo, fileSize);
143     // Comparator class name is stored in the trailer in version 2.
144     RawComparator<byte []> comparator = trailer.createComparator();
145     HFileBlockIndex.BlockIndexReader dataBlockIndexReader =
146         new HFileBlockIndex.BlockIndexReader(comparator,
147             trailer.getNumDataIndexLevels());
148     HFileBlockIndex.BlockIndexReader metaBlockIndexReader =
149         new HFileBlockIndex.BlockIndexReader(
150             Bytes.BYTES_RAWCOMPARATOR, 1);
151 
152     HFileBlock.BlockIterator blockIter = blockReader.blockRange(
153         trailer.getLoadOnOpenDataOffset(),
154         fileSize - trailer.getTrailerSize());
155     // Data index. We also read statistics about the block index written after
156     // the root level.
157     dataBlockIndexReader.readMultiLevelIndexRoot(
158         blockIter.nextBlockWithBlockType(BlockType.ROOT_INDEX),
159         trailer.getDataIndexCount());
160     
161     if (findMidKey) {
162       byte[] midkey = dataBlockIndexReader.midkey();
163       assertNotNull("Midkey should not be null", midkey);
164     }
165     
166     // Meta index.
167     metaBlockIndexReader.readRootIndex(
168         blockIter.nextBlockWithBlockType(BlockType.ROOT_INDEX).getByteStream(),
169         trailer.getMetaIndexCount());
170     // File info
171     FileInfo fileInfo = new FileInfo();
172     fileInfo.read(blockIter.nextBlockWithBlockType(BlockType.FILE_INFO).getByteStream());
173     byte [] keyValueFormatVersion = fileInfo.get(
174         HFileWriterV2.KEY_VALUE_VERSION);
175     boolean includeMemstoreTS = keyValueFormatVersion != null &&
176         Bytes.toInt(keyValueFormatVersion) > 0;
177 
178     // Counters for the number of key/value pairs and the number of blocks
179     int entriesRead = 0;
180     int blocksRead = 0;
181     long memstoreTS = 0;
182 
183     // Scan blocks the way the reader would scan them
184     fsdis.seek(0);
185     long curBlockPos = 0;
186     while (curBlockPos <= trailer.getLastDataBlockOffset()) {
187       HFileBlock block = blockReader.readBlockData(curBlockPos, -1, -1, false);
188       assertEquals(BlockType.DATA, block.getBlockType());
189       ByteBuffer buf = block.getBufferWithoutHeader();
190       while (buf.hasRemaining()) {
191         int keyLen = buf.getInt();
192         int valueLen = buf.getInt();
193 
194         byte[] key = new byte[keyLen];
195         buf.get(key);
196 
197         byte[] value = new byte[valueLen];
198         buf.get(value);
199 
200         if (includeMemstoreTS) {
201           ByteArrayInputStream byte_input = new ByteArrayInputStream(buf.array(),
202                                buf.arrayOffset() + buf.position(), buf.remaining());
203           DataInputStream data_input = new DataInputStream(byte_input);
204 
205           memstoreTS = WritableUtils.readVLong(data_input);
206           buf.position(buf.position() + WritableUtils.getVIntSize(memstoreTS));
207         }
208 
209         // A brute-force check to see that all keys and values are correct.
210         assertTrue(Bytes.compareTo(key, keys.get(entriesRead)) == 0);
211         assertTrue(Bytes.compareTo(value, values.get(entriesRead)) == 0);
212 
213         ++entriesRead;
214       }
215       ++blocksRead;
216       curBlockPos += block.getOnDiskSizeWithHeader();
217     }
218     LOG.info("Finished reading: entries=" + entriesRead + ", blocksRead="
219         + blocksRead);
220     assertEquals(entryCount, entriesRead);
221 
222     // Meta blocks. We can scan until the load-on-open data offset (which is
223     // the root block index offset in version 2) because we are not testing
224     // intermediate-level index blocks here.
225 
226     int metaCounter = 0;
227     while (fsdis.getPos() < trailer.getLoadOnOpenDataOffset()) {
228       LOG.info("Current offset: " + fsdis.getPos() + ", scanning until " +
229           trailer.getLoadOnOpenDataOffset());
230       HFileBlock block = blockReader.readBlockData(curBlockPos, -1, -1, false);
231       assertEquals(BlockType.META, block.getBlockType());
232       Text t = new Text();
233       ByteBuffer buf = block.getBufferWithoutHeader();
234       if (Writables.getWritable(buf.array(), buf.arrayOffset(), buf.limit(), t) == null) {
235         throw new IOException("Failed to deserialize block " + this + " into a " + t.getClass().getSimpleName());
236       }
237       Text expectedText =
238           (metaCounter == 0 ? new Text("Paris") : metaCounter == 1 ? new Text(
239               "Moscow") : new Text("Washington, D.C."));
240       assertEquals(expectedText, t);
241       LOG.info("Read meta block data: " + t);
242       ++metaCounter;
243       curBlockPos += block.getOnDiskSizeWithHeader();
244     }
245 
246     fsdis.close();
247   }

252   private static final String COLUMN_FAMILY_NAME = "_-myColumnFamily-_";
253   private static final int MIN_ROW_OR_QUALIFIER_LENGTH = 64;
254   private static final int MAX_ROW_OR_QUALIFIER_LENGTH = 128;

256   /**
257    * Generates a random key that is guaranteed to increase as the given index i
258    * increases. The result consists of a prefix, which is a deterministic
259    * increasing function of i, and a random suffix.
260    *
261    * @param rand
262    *          random number generator to use
263    * @param i
264    * @return
265    */
266   public static byte[] randomOrderedKey(Random rand, int i) {
267     StringBuilder k = new StringBuilder();
268 
269     // The fixed-length lexicographically increasing part of the key.
270     for (int bitIndex = 31; bitIndex >= 0; --bitIndex) {
271       if ((i & (1 << bitIndex)) == 0)
272         k.append("a");
273       else
274         k.append("b");
275     }
276 
277     // A random-length random suffix of the key.
278     for (int j = 0; j < rand.nextInt(50); ++j)
279       k.append(randomReadableChar(rand));
280 
281     byte[] keyBytes = k.toString().getBytes();
282     return keyBytes;
283   }

285   public static byte[] randomValue(Random rand) {
286     StringBuilder v = new StringBuilder();
287     for (int j = 0; j < 1 + rand.nextInt(2000); ++j) {
288       v.append((char) (32 + rand.nextInt(95)));
289     }
290 
291     byte[] valueBytes = v.toString().getBytes();
292     return valueBytes;
293   }

295   public static final char randomReadableChar(Random rand) {
296     int i = rand.nextInt(26 * 2 + 10 + 1);
297     if (i < 26)
298       return (char) ('A' + i);
299     i -= 26;
300 
301     if (i < 26)
302       return (char) ('a' + i);
303     i -= 26;
304 
305     if (i < 10)
306       return (char) ('0' + i);
307     i -= 10;
308 
309     assert i == 0;
310     return '_';
311   }

313   public static byte[] randomRowOrQualifier(Random rand) {
314     StringBuilder field = new StringBuilder();
315     int fieldLen = MIN_ROW_OR_QUALIFIER_LENGTH
316         + rand.nextInt(MAX_ROW_OR_QUALIFIER_LENGTH
317             - MIN_ROW_OR_QUALIFIER_LENGTH + 1);
318     for (int i = 0; i < fieldLen; ++i)
319       field.append(randomReadableChar(rand));
320     return field.toString().getBytes();
321   }

323   public static KeyValue randomKeyValue(Random rand) {
324     return new KeyValue(randomRowOrQualifier(rand),
325         COLUMN_FAMILY_NAME.getBytes(), randomRowOrQualifier(rand),
326         randomValue(rand));
327   }
