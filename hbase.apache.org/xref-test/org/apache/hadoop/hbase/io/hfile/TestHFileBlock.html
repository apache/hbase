120  static int writeTestKeyValues(OutputStream dos, int seed, boolean includesMemstoreTS)
121       throws IOException {
122     List<KeyValue> keyValues = new ArrayList<KeyValue>();
123     Random randomizer = new Random(42l + seed); // just any fixed number
124 
125     // generate keyValues
126     for (int i = 0; i < NUM_KEYVALUES; ++i) {
127       byte[] row;
128       long timestamp;
129       byte[] family;
130       byte[] qualifier;
131       byte[] value;
132 
133       // generate it or repeat, it should compress well
134       if (0 < i && randomizer.nextFloat() < CHANCE_TO_REPEAT) {
135         row = keyValues.get(randomizer.nextInt(keyValues.size())).getRow();
136       } else {
137         row = new byte[FIELD_LENGTH];
138         randomizer.nextBytes(row);
139       }
140       if (0 == i) {
141         family = new byte[FIELD_LENGTH];
142         randomizer.nextBytes(family);
143       } else {
144         family = keyValues.get(0).getFamily();
145       }
146       if (0 < i && randomizer.nextFloat() < CHANCE_TO_REPEAT) {
147         qualifier = keyValues.get(
148             randomizer.nextInt(keyValues.size())).getQualifier();
149       } else {
150         qualifier = new byte[FIELD_LENGTH];
151         randomizer.nextBytes(qualifier);
152       }
153       if (0 < i && randomizer.nextFloat() < CHANCE_TO_REPEAT) {
154         value = keyValues.get(randomizer.nextInt(keyValues.size())).getValue();
155       } else {
156         value = new byte[FIELD_LENGTH];
157         randomizer.nextBytes(value);
158       }
159       if (0 < i && randomizer.nextFloat() < CHANCE_TO_REPEAT) {
160         timestamp = keyValues.get(
161             randomizer.nextInt(keyValues.size())).getTimestamp();
162       } else {
163         timestamp = randomizer.nextLong();
164       }
165 
166       keyValues.add(new KeyValue(row, family, qualifier, timestamp, value));
167     }
168 
169     // sort it and write to stream
170     int totalSize = 0;
171     Collections.sort(keyValues, KeyValue.COMPARATOR);
172     DataOutputStream dataOutputStream = new DataOutputStream(dos);
173     for (KeyValue kv : keyValues) {
174       totalSize += kv.getLength();
175       dataOutputStream.write(kv.getBuffer(), kv.getOffset(), kv.getLength());
176       if (includesMemstoreTS) {
177         long memstoreTS = randomizer.nextLong();
178         WritableUtils.writeVLong(dataOutputStream, memstoreTS);
179         totalSize += WritableUtils.getVIntSize(memstoreTS);
180       }
181     }
182 
183     return totalSize;
184   }

186   public byte[] createTestV1Block(Compression.Algorithm algo)
187       throws IOException {
188     Compressor compressor = algo.getCompressor();
189     ByteArrayOutputStream baos = new ByteArrayOutputStream();
190     OutputStream os = algo.createCompressionStream(baos, compressor, 0);
191     DataOutputStream dos = new DataOutputStream(os);
192     BlockType.META.write(dos); // Let's make this a meta block.
193     writeTestBlockContents(dos);
194     uncompressedSizeV1 = dos.size();
195     dos.flush();
196     algo.returnCompressor(compressor);
197     return baos.toByteArray();
198   }

214   public String createTestBlockStr(Compression.Algorithm algo,
215       int correctLength) throws IOException {
216     HFileBlock.Writer hbw = createTestV2Block(algo, includesMemstoreTS);
217     byte[] testV2Block = hbw.getHeaderAndDataForTest();
218     int osOffset = HFileBlock.HEADER_SIZE + 9;
219     if (testV2Block.length == correctLength) {
220       // Force-set the "OS" field of the gzip header to 3 (Unix) to avoid
221       // variations across operating systems.
222       // See http://www.gzip.org/zlib/rfc-gzip.html for gzip format.
223       // We only make this change when the compressed block length matches.
224       // Otherwise, there are obviously other inconsistencies.
225       testV2Block[osOffset] = 3;
226     }
227     return Bytes.toStringBinary(testV2Block);
228   }

236   @Test
237   public void testGzipCompression() throws IOException {
238     final String correctTestBlockStr =
239         "DATABLK*\\x00\\x00\\x00>\\x00\\x00\\x0F\\xA0\\xFF\\xFF\\xFF\\xFF"
240             + "\\xFF\\xFF\\xFF\\xFF"
241             + "\\x01\\x00\\x00@\\x00\\x00\\x00\\x00["
242             // gzip-compressed block: http://www.gzip.org/zlib/rfc-gzip.html
243             + "\\x1F\\x8B"  // gzip magic signature
244             + "\\x08"  // Compression method: 8 = "deflate"
245             + "\\x00"  // Flags
246             + "\\x00\\x00\\x00\\x00"  // mtime
247             + "\\x00"  // XFL (extra flags)
248             // OS (0 = FAT filesystems, 3 = Unix). However, this field
249             // sometimes gets set to 0 on Linux and Mac, so we reset it to 3.
250             + "\\x03"
251             + "\\xED\\xC3\\xC1\\x11\\x00 \\x08\\xC00DD\\xDD\\x7Fa"
252             + "\\xD6\\xE8\\xA3\\xB9K\\x84`\\x96Q\\xD3\\xA8\\xDB\\xA8e\\xD4c"
253             + "\\xD46\\xEA5\\xEA3\\xEA7\\xE7\\x00LI\\x5Cs\\xA0\\x0F\\x00\\x00"
254             + "\\xAB\\x85g\\x91"; //  4 byte checksum
255     final int correctGzipBlockLength = 95;
256     assertEquals(correctTestBlockStr, createTestBlockStr(GZ,
257         correctGzipBlockLength));
258   }

359   @Test
360   public void testDataBlockEncoding() throws IOException {
361     final int numBlocks = 5;
362     for (Compression.Algorithm algo : COMPRESSION_ALGORITHMS) {
363       for (boolean pread : new boolean[] { false, true }) {
364         for (DataBlockEncoding encoding : DataBlockEncoding.values()) {
365           Path path = new Path(TEST_UTIL.getDataTestDir(), "blocks_v2_"
366               + algo + "_" + encoding.toString());
367           FSDataOutputStream os = fs.create(path);
368           HFileDataBlockEncoder dataBlockEncoder =
369               new HFileDataBlockEncoderImpl(encoding);
370           HFileBlock.Writer hbw = new HFileBlock.Writer(algo, dataBlockEncoder,
371               includesMemstoreTS, HFile.DEFAULT_CHECKSUM_TYPE,
372               HFile.DEFAULT_BYTES_PER_CHECKSUM);
373           long totalSize = 0;
374           final List<Integer> encodedSizes = new ArrayList<Integer>();
375           final List<ByteBuffer> encodedBlocks = new ArrayList<ByteBuffer>();
376           for (int blockId = 0; blockId < numBlocks; ++blockId) {
377             DataOutputStream dos = hbw.startWriting(BlockType.DATA);
378             writeEncodedBlock(algo, encoding, dos, encodedSizes, encodedBlocks,
379                 blockId, includesMemstoreTS, HFileBlock.DUMMY_HEADER);
380             hbw.writeHeaderAndData(os);
381             totalSize += hbw.getOnDiskSizeWithHeader();
382           }
383           os.close();
384 
385           FSDataInputStream is = fs.open(path);
386           HFileBlock.FSReaderV2 hbr = new HFileBlock.FSReaderV2(is, algo,
387               totalSize);
388           hbr.setDataBlockEncoder(dataBlockEncoder);
389           hbr.setIncludesMemstoreTS(includesMemstoreTS);
390 
391           HFileBlock b;
392           int pos = 0;
393           for (int blockId = 0; blockId < numBlocks; ++blockId) {
394             b = hbr.readBlockData(pos, -1, -1, pread);
395             assertEquals(0, HFile.getChecksumFailuresCount());
396             b.sanityCheck();
397             pos += b.getOnDiskSizeWithHeader();
398             assertEquals((int) encodedSizes.get(blockId),
399                 b.getUncompressedSizeWithoutHeader());
400             ByteBuffer actualBuffer = b.getBufferWithoutHeader();
401             if (encoding != DataBlockEncoding.NONE) {
402               // We expect a two-byte big-endian encoding id.
403               assertEquals(0, actualBuffer.get(0));
404               assertEquals(encoding.getId(), actualBuffer.get(1));
405               actualBuffer.position(2);
406               actualBuffer = actualBuffer.slice();
407             }
408 
409             ByteBuffer expectedBuffer = encodedBlocks.get(blockId);
410             expectedBuffer.rewind();
411 
412             // test if content matches, produce nice message
413             assertBuffersEqual(expectedBuffer, actualBuffer, algo, encoding,
414                 pread);
415           }
416           is.close();
417         }
418       }
419     }

422   static void writeEncodedBlock(Algorithm algo, DataBlockEncoding encoding,
423        DataOutputStream dos, final List<Integer> encodedSizes,
424       final List<ByteBuffer> encodedBlocks, int blockId, 
425       boolean includesMemstoreTS, byte[] dummyHeader) throws IOException {
426     ByteArrayOutputStream baos = new ByteArrayOutputStream();
427     DoubleOutputStream doubleOutputStream =
428         new DoubleOutputStream(dos, baos);
429     writeTestKeyValues(doubleOutputStream, blockId, includesMemstoreTS);
430     ByteBuffer rawBuf = ByteBuffer.wrap(baos.toByteArray());
431     rawBuf.rewind();
432 
433     DataBlockEncoder encoder = encoding.getEncoder();
434     int headerLen = dummyHeader.length;
435     byte[] encodedResultWithHeader = null;
436     if (encoder != null) {
437       HFileBlockEncodingContext encodingCtx =
438           encoder.newDataBlockEncodingContext(algo, encoding, dummyHeader);
439       encoder.encodeKeyValues(rawBuf, includesMemstoreTS,
440           encodingCtx);
441       encodedResultWithHeader =
442           encodingCtx.getUncompressedBytesWithHeader();
443     } else {
444       HFileBlockDefaultEncodingContext defaultEncodingCtx =
445         new HFileBlockDefaultEncodingContext(algo, encoding, dummyHeader);
446       byte[] rawBufWithHeader =
447           new byte[rawBuf.array().length + headerLen];
448       System.arraycopy(rawBuf.array(), 0, rawBufWithHeader,
449           headerLen, rawBuf.array().length);
450       defaultEncodingCtx.compressAfterEncoding(rawBufWithHeader,
451           BlockType.DATA);
452       encodedResultWithHeader =
453         defaultEncodingCtx.getUncompressedBytesWithHeader();
454     }
455     final int encodedSize =
456         encodedResultWithHeader.length - headerLen;
457     if (encoder != null) {
458       // We need to account for the two-byte encoding algorithm ID that
459       // comes after the 24-byte block header but before encoded KVs.
460       headerLen += DataBlockEncoding.ID_SIZE;
461     }
462     byte[] encodedDataSection =
463         new byte[encodedResultWithHeader.length - headerLen];
464     System.arraycopy(encodedResultWithHeader, headerLen,
465         encodedDataSection, 0, encodedDataSection.length);
466     final ByteBuffer encodedBuf =
467         ByteBuffer.wrap(encodedDataSection);
468     encodedSizes.add(encodedSize);
469     encodedBlocks.add(encodedBuf);
470   }

472   static void assertBuffersEqual(ByteBuffer expectedBuffer,
473       ByteBuffer actualBuffer, Compression.Algorithm compression,
474       DataBlockEncoding encoding, boolean pread) {
475     if (!actualBuffer.equals(expectedBuffer)) {
476       int prefix = 0;
477       int minLimit = Math.min(expectedBuffer.limit(), actualBuffer.limit());
478       while (prefix < minLimit &&
479           expectedBuffer.get(prefix) == actualBuffer.get(prefix)) {
480         prefix++;
481       }
482 
483       fail(String.format(
484           "Content mismath for compression %s, encoding %s, " +
485           "pread %s, commonPrefix %d, expected %s, got %s",
486           compression, encoding, pread, prefix,
487           nextBytesToStr(expectedBuffer, prefix),
488           nextBytesToStr(actualBuffer, prefix)));
489     }
490   }

492   /**
493    * Convert a few next bytes in the given buffer at the given position to
494    * string. Used for error messages.
495    */
496   private static String nextBytesToStr(ByteBuffer buf, int pos) {
497     int maxBytes = buf.limit() - pos;
498     int numBytes = Math.min(16, maxBytes);
499     return Bytes.toStringBinary(buf.array(), buf.arrayOffset() + pos,
500         numBytes) + (numBytes < maxBytes ? "..." : "");
501   }

503   @Test
504   public void testPreviousOffset() throws IOException {
505     for (Compression.Algorithm algo : COMPRESSION_ALGORITHMS) {
506       for (boolean pread : BOOLEAN_VALUES) {
507         for (boolean cacheOnWrite : BOOLEAN_VALUES) {
508           Random rand = defaultRandom();
509           LOG.info("testPreviousOffset:Compression algorithm: " + algo + 
510                    ", pread=" + pread +
511                    ", cacheOnWrite=" + cacheOnWrite);
512           Path path = new Path(TEST_UTIL.getDataTestDir(), "prev_offset");
513           List<Long> expectedOffsets = new ArrayList<Long>();
514           List<Long> expectedPrevOffsets = new ArrayList<Long>();
515           List<BlockType> expectedTypes = new ArrayList<BlockType>();
516           List<ByteBuffer> expectedContents = cacheOnWrite
517               ? new ArrayList<ByteBuffer>() : null;
518           long totalSize = writeBlocks(rand, algo, path, expectedOffsets,
519               expectedPrevOffsets, expectedTypes, expectedContents);
520 
521           FSDataInputStream is = fs.open(path);
522           HFileBlock.FSReader hbr = new HFileBlock.FSReaderV2(is, algo,
523               totalSize);
524           long curOffset = 0;
525           for (int i = 0; i < NUM_TEST_BLOCKS; ++i) {
526             if (!pread) {
527               assertEquals(is.getPos(), curOffset + (i == 0 ? 0 :
528                   HFileBlock.HEADER_SIZE));
529             }
530 
531             assertEquals(expectedOffsets.get(i).longValue(), curOffset);
532             if (detailedLogging) {
533               LOG.info("Reading block #" + i + " at offset " + curOffset);
534             }
535             HFileBlock b = hbr.readBlockData(curOffset, -1, -1, pread);
536             if (detailedLogging) {
537               LOG.info("Block #" + i + ": " + b);
538             }
539             assertEquals("Invalid block #" + i + "'s type:",
540                 expectedTypes.get(i), b.getBlockType());
541             assertEquals("Invalid previous block offset for block " + i
542                 + " of " + "type " + b.getBlockType() + ":",
543                 (long) expectedPrevOffsets.get(i), b.getPrevBlockOffset());
544             b.sanityCheck();
545             assertEquals(curOffset, b.getOffset());
546 
547             // Now re-load this block knowing the on-disk size. This tests a
548             // different branch in the loader.
549             HFileBlock b2 = hbr.readBlockData(curOffset,
550                 b.getOnDiskSizeWithHeader(), -1, pread);
551             b2.sanityCheck();
552 
553             assertEquals(b.getBlockType(), b2.getBlockType());
554             assertEquals(b.getOnDiskSizeWithoutHeader(),
555                 b2.getOnDiskSizeWithoutHeader());
556             assertEquals(b.getOnDiskSizeWithHeader(),
557                 b2.getOnDiskSizeWithHeader());
558             assertEquals(b.getUncompressedSizeWithoutHeader(),
559                 b2.getUncompressedSizeWithoutHeader());
560             assertEquals(b.getPrevBlockOffset(), b2.getPrevBlockOffset());
561             assertEquals(curOffset, b2.getOffset());
562             assertEquals(b.getBytesPerChecksum(), b2.getBytesPerChecksum());
563             assertEquals(b.getOnDiskDataSizeWithHeader(), 
564                          b2.getOnDiskDataSizeWithHeader());
565             assertEquals(0, HFile.getChecksumFailuresCount());
566 
567             curOffset += b.getOnDiskSizeWithHeader();
568 
569             if (cacheOnWrite) {
570               // In the cache-on-write mode we store uncompressed bytes so we
571               // can compare them to what was read by the block reader.
572               // b's buffer has header + data + checksum while 
573               // expectedContents have header + data only
574               ByteBuffer bufRead = b.getBufferWithHeader();
575               ByteBuffer bufExpected = expectedContents.get(i);
576               boolean bytesAreCorrect = Bytes.compareTo(bufRead.array(),
577                   bufRead.arrayOffset(), 
578                   bufRead.limit() - b.totalChecksumBytes(),
579                   bufExpected.array(), bufExpected.arrayOffset(),
580                   bufExpected.limit()) == 0;
581               String wrongBytesMsg = "";
582 
583               if (!bytesAreCorrect) {
584                 // Optimization: only construct an error message in case we
585                 // will need it.
586                 wrongBytesMsg = "Expected bytes in block #" + i + " (algo="
587                     + algo + ", pread=" + pread
588                     + ", cacheOnWrite=" + cacheOnWrite + "):\n";
589                 wrongBytesMsg += Bytes.toStringBinary(bufExpected.array(),
590                     bufExpected.arrayOffset(), Math.min(32,
591                         bufExpected.limit()))
592                     + ", actual:\n"
593                     + Bytes.toStringBinary(bufRead.array(),
594                         bufRead.arrayOffset(), Math.min(32, bufRead.limit()));
595                 if (detailedLogging) {
596                   LOG.warn("expected header" +

622   private class BlockReaderThread implements Callable<Boolean> {
623     private final String clientId;
624     private final HFileBlock.FSReader hbr;
625     private final List<Long> offsets;
626     private final List<BlockType> types;
627     private final long fileSize;
628 
629     public BlockReaderThread(String clientId,
630         HFileBlock.FSReader hbr, List<Long> offsets, List<BlockType> types,
631         long fileSize) {
632       this.clientId = clientId;
633       this.offsets = offsets;
634       this.hbr = hbr;
635       this.types = types;
636       this.fileSize = fileSize;
637     }
638 
639     @Override
640     public Boolean call() throws Exception {
641       Random rand = new Random(clientId.hashCode());
642       long endTime = System.currentTimeMillis() + 10000;
643       int numBlocksRead = 0;
644       int numPositionalRead = 0;
645       int numWithOnDiskSize = 0;
646       while (System.currentTimeMillis() < endTime) {
647         int blockId = rand.nextInt(NUM_TEST_BLOCKS);
648         long offset = offsets.get(blockId);
649         boolean pread = rand.nextBoolean();
650         boolean withOnDiskSize = rand.nextBoolean();
651         long expectedSize =
652           (blockId == NUM_TEST_BLOCKS - 1 ? fileSize
653               : offsets.get(blockId + 1)) - offset;
654 
655         HFileBlock b;
656         try {
657           long onDiskSizeArg = withOnDiskSize ? expectedSize : -1;
658           b = hbr.readBlockData(offset, onDiskSizeArg, -1, pread);
659         } catch (IOException ex) {
660           LOG.error("Error in client " + clientId + " trying to read block at "
661               + offset + ", pread=" + pread + ", withOnDiskSize=" +
662               withOnDiskSize, ex);
663           return false;
664         }
665 
666         assertEquals(types.get(blockId), b.getBlockType());
667         assertEquals(expectedSize, b.getOnDiskSizeWithHeader());
668         assertEquals(offset, b.getOffset());
669 
670         ++numBlocksRead;
671         if (pread)
672           ++numPositionalRead;
673         if (withOnDiskSize)
674           ++numWithOnDiskSize;
675       }
676       LOG.info("Client " + clientId + " successfully read " + numBlocksRead +
677         " blocks (with pread: " + numPositionalRead + ", with onDiskSize " +
678         "specified: " + numWithOnDiskSize + ")");
679 
680       return true;
681     }

722   private long writeBlocks(Random rand, Compression.Algorithm compressAlgo,
723       Path path, List<Long> expectedOffsets, List<Long> expectedPrevOffsets,
724       List<BlockType> expectedTypes, List<ByteBuffer> expectedContents
725   ) throws IOException {
726     boolean cacheOnWrite = expectedContents != null;
727     FSDataOutputStream os = fs.create(path);
728     HFileBlock.Writer hbw = new HFileBlock.Writer(compressAlgo, null,
729         includesMemstoreTS, HFile.DEFAULT_CHECKSUM_TYPE,
730         HFile.DEFAULT_BYTES_PER_CHECKSUM);
731     Map<BlockType, Long> prevOffsetByType = new HashMap<BlockType, Long>();
732     long totalSize = 0;
733     for (int i = 0; i < NUM_TEST_BLOCKS; ++i) {
734       long pos = os.getPos();
735       int blockTypeOrdinal = rand.nextInt(BlockType.values().length);
736       if (blockTypeOrdinal == BlockType.ENCODED_DATA.ordinal()) {
737         blockTypeOrdinal = BlockType.DATA.ordinal();
738       }
739       BlockType bt = BlockType.values()[blockTypeOrdinal];
740       DataOutputStream dos = hbw.startWriting(bt);
741       int size = rand.nextInt(500);
742       for (int j = 0; j < size; ++j) {
743         // This might compress well.
744         dos.writeShort(i + 1);
745         dos.writeInt(j + 1);
746       }
747 
748       if (expectedOffsets != null)
749         expectedOffsets.add(os.getPos());
750 
751       if (expectedPrevOffsets != null) {
752         Long prevOffset = prevOffsetByType.get(bt);
753         expectedPrevOffsets.add(prevOffset != null ? prevOffset : -1);
754         prevOffsetByType.put(bt, os.getPos());
755       }
756 
757       expectedTypes.add(bt);
758 
759       hbw.writeHeaderAndData(os);
760       totalSize += hbw.getOnDiskSizeWithHeader();
761 
762       if (cacheOnWrite)
763         expectedContents.add(hbw.getUncompressedBufferWithHeader());
764 
765       if (detailedLogging) {
766         LOG.info("Written block #" + i + " of type " + bt
767             + ", uncompressed size " + hbw.getUncompressedSizeWithoutHeader()
768             + " at offset " + pos);
769       }
770     }
771     os.close();
772     LOG.info("Created a temporary file at " + path + ", "
773         + fs.getFileStatus(path).getLen() + " byte, compression=" +
774         compressAlgo);
775     return totalSize;
776   }
