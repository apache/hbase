44  /**
45   * Handler to run disable of a table.
46   */
47  @InterfaceAudience.Private
48  public class DisableTableHandler extends EventHandler {
49    private static final Log LOG = LogFactory.getLog(DisableTableHandler.class);
50    private final byte [] tableName;
51    private final String tableNameStr;
52    private final AssignmentManager assignmentManager;
53  
54    public DisableTableHandler(Server server, byte [] tableName,
55        CatalogTracker catalogTracker, AssignmentManager assignmentManager,
56        boolean skipTableStateCheck)
57    throws TableNotFoundException, TableNotEnabledException, IOException {
58      super(server, EventType.C_M_DISABLE_TABLE);
59      this.tableName = tableName;
60      this.tableNameStr = Bytes.toString(this.tableName);
61      this.assignmentManager = assignmentManager;
62      // Check if table exists
63      // TODO: do we want to keep this in-memory as well?  i guess this is
64      //       part of old master rewrite, schema to zk to check for table
65      //       existence and such
66      if (!MetaReader.tableExists(catalogTracker, this.tableNameStr)) {
67        throw new TableNotFoundException(this.tableNameStr);
68      }
69  
70      // There could be multiple client requests trying to disable or enable
71      // the table at the same time. Ensure only the first request is honored
72      // After that, no other requests can be accepted until the table reaches
73      // DISABLED or ENABLED.
74      if (!skipTableStateCheck)
75      {
76        try {
77          if (!this.assignmentManager.getZKTable().checkEnabledAndSetDisablingTable
78            (this.tableNameStr)) {
79            LOG.info("Table " + tableNameStr + " isn't enabled; skipping disable");
80            throw new TableNotEnabledException(this.tableNameStr);
81          }
82        } catch (KeeperException e) {
83          throw new IOException("Unable to ensure that the table will be" +
84            " disabling because of a ZooKeeper issue", e);
85        }
86      }
87    }
88  
89    @Override
90    public String toString() {
91      String name = "UnknownServerName";
92      if(server != null && server.getServerName() != null) {
93        name = server.getServerName().toString();
94      }
95      return getClass().getSimpleName() + "-" + name + "-" + getSeqid() + "-" +
96        tableNameStr;
97    }
98  
99    @Override
100   public void process() {
101     try {
102       LOG.info("Attempting to disable table " + this.tableNameStr);
103       MasterCoprocessorHost cpHost = ((HMaster) this.server)
104           .getCoprocessorHost();
105       if (cpHost != null) {
106         cpHost.preDisableTableHandler(this.tableName);
107       }
108       handleDisableTable();
109       if (cpHost != null) {
110         cpHost.postDisableTableHandler(this.tableName);
111       }
112     } catch (IOException e) {
113       LOG.error("Error trying to disable table " + this.tableNameStr, e);
114     } catch (KeeperException e) {
115       LOG.error("Error trying to disable table " + this.tableNameStr, e);
116     }
117   }
118 
119   private void handleDisableTable() throws IOException, KeeperException {
120     // Set table disabling flag up in zk.
121     this.assignmentManager.getZKTable().setDisablingTable(this.tableNameStr);
122     boolean done = false;
123     while (true) {
124       // Get list of online regions that are of this table.  Regions that are
125       // already closed will not be included in this list; i.e. the returned
126       // list is not ALL regions in a table, its all online regions according
127       // to the in-memory state on this master.
128       final List<HRegionInfo> regions = this.assignmentManager
129         .getRegionStates().getRegionsOfTable(tableName);
130       if (regions.size() == 0) {
131         done = true;
132         break;
133       }
134       LOG.info("Offlining " + regions.size() + " regions.");
135       BulkDisabler bd = new BulkDisabler(this.server, regions);
136       try {
137         if (bd.bulkAssign()) {
138           done = true;
139           break;
140         }
141       } catch (InterruptedException e) {
142         LOG.warn("Disable was interrupted");
143         // Preserve the interrupt.
144         Thread.currentThread().interrupt();
145         break;
146       }
147     }
148     // Flip the table to disabled if success.
149     if (done) this.assignmentManager.getZKTable().setDisabledTable(this.tableNameStr);
150     LOG.info("Disabled table is done=" + done);
151   }

153   /**
154    * Run bulk disable.
155    */
156   class BulkDisabler extends BulkAssigner {
157     private final List<HRegionInfo> regions;
158 
159     BulkDisabler(final Server server, final List<HRegionInfo> regions) {
160       super(server);
161       this.regions = regions;
162     }
163 
164     @Override
165     protected void populatePool(ExecutorService pool) {
166       RegionStates regionStates = assignmentManager.getRegionStates();
167       for (HRegionInfo region: regions) {
168         if (regionStates.isRegionInTransition(region)) continue;
169         final HRegionInfo hri = region;
170         pool.execute(Trace.wrap(new Runnable() {
171           public void run() {
172             assignmentManager.unassign(hri);
173           }
174         }));
175       }
176     }
177 
178     @Override
179     protected boolean waitUntilDone(long timeout)
180     throws InterruptedException {
181       long startTime = System.currentTimeMillis();
182       long remaining = timeout;
183       List<HRegionInfo> regions = null;
184       while (!server.isStopped() && remaining > 0) {
185         Thread.sleep(waitingTimeForEvents);
186         regions = assignmentManager.getRegionStates().getRegionsOfTable(tableName);
187         if (regions.isEmpty()) break;
188         remaining = timeout - (System.currentTimeMillis() - startTime);
189       }
190       return regions != null && regions.isEmpty();
191     }
192   }
