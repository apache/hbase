164   /**
165    * Move aside a bad edits file.
166    * 
167    * @param fs
168    * @param edits
169    *          Edits file to move aside.
170    * @return The name of the moved aside file.
171    * @throws IOException
172    */
173   public static Path moveAsideBadEditsFile(final FileSystem fs, final Path edits)
174       throws IOException {
175     Path moveAsideName = new Path(edits.getParent(), edits.getName() + "."
176         + System.currentTimeMillis());
177     if (!fs.rename(edits, moveAsideName)) {
178       LOG.warn("Rename failed from " + edits + " to " + moveAsideName);
179     }
180     return moveAsideName;
181   }

183   /**
184    * @param path
185    *          - the path to analyze. Expected format, if it's in hlog directory:
186    *          / [base directory for hbase] / hbase / .logs / ServerName /
187    *          logfile
188    * @return null if it's not a log file. Returns the ServerName of the region
189    *         server that created this log file otherwise.
190    */
191   public static ServerName getServerNameFromHLogDirectoryName(
192       Configuration conf, String path) throws IOException {
193     if (path == null
194         || path.length() <= HConstants.HREGION_LOGDIR_NAME.length()) {
195       return null;
196     }
197 
198     if (conf == null) {
199       throw new IllegalArgumentException("parameter conf must be set");
200     }
201 
202     final String rootDir = conf.get(HConstants.HBASE_DIR);
203     if (rootDir == null || rootDir.isEmpty()) {
204       throw new IllegalArgumentException(HConstants.HBASE_DIR
205           + " key not found in conf.");
206     }
207 
208     final StringBuilder startPathSB = new StringBuilder(rootDir);
209     if (!rootDir.endsWith("/"))
210       startPathSB.append('/');
211     startPathSB.append(HConstants.HREGION_LOGDIR_NAME);
212     if (!HConstants.HREGION_LOGDIR_NAME.endsWith("/"))
213       startPathSB.append('/');
214     final String startPath = startPathSB.toString();
215 
216     String fullPath;
217     try {
218       fullPath = FileSystem.get(conf).makeQualified(new Path(path)).toString();
219     } catch (IllegalArgumentException e) {
220       LOG.info("Call to makeQualified failed on " + path + " " + e.getMessage());
221       return null;
222     }
223 
224     if (!fullPath.startsWith(startPath)) {
225       return null;
226     }
227 
228     final String serverNameAndFile = fullPath.substring(startPath.length());
229 
230     if (serverNameAndFile.indexOf('/') < "a,0,0".length()) {
231       // Either it's a file (not a directory) or it's not a ServerName format
232       return null;
233     }
234 
235     final String serverName = serverNameAndFile.substring(0,
236         serverNameAndFile.indexOf('/') - 1);
237 
238     if (!ServerName.isFullServerName(serverName)) {
239       return null;
240     }
241 
242     return ServerName.parseServerName(serverName);
243   }

245   /**
246    * Return regions (memstores) that have edits that are equal or less than the
247    * passed <code>oldestWALseqid</code>.
248    * 
249    * @param oldestWALseqid
250    * @param regionsToSeqids
251    *          Encoded region names to sequence ids
252    * @return All regions whose seqid is < than <code>oldestWALseqid</code> (Not
253    *         necessarily in order). Null if no regions found.
254    */
255   static byte[][] findMemstoresWithEditsEqualOrOlderThan(
256       final long oldestWALseqid, final Map<byte[], Long> regionsToSeqids) {
257     // This method is static so it can be unit tested the easier.
258     List<byte[]> regions = null;
259     for (Map.Entry<byte[], Long> e : regionsToSeqids.entrySet()) {
260       if (e.getValue().longValue() <= oldestWALseqid) {
261         if (regions == null)
262           regions = new ArrayList<byte[]>();
263         // Key is encoded region name.
264         regions.add(e.getKey());
265       }
266     }
267     return regions == null ? null : regions
268         .toArray(new byte[][] { HConstants.EMPTY_BYTE_ARRAY });
269   }

271   /**
272    * Returns sorted set of edit files made by wal-log splitter, excluding files
273    * with '.temp' suffix.
274    * 
275    * @param fs
276    * @param regiondir
277    * @return Files in passed <code>regiondir</code> as a sorted set.
278    * @throws IOException
279    */
280   public static NavigableSet<Path> getSplitEditFilesSorted(final FileSystem fs,
281       final Path regiondir) throws IOException {
282     NavigableSet<Path> filesSorted = new TreeSet<Path>();
283     Path editsdir = HLogUtil.getRegionDirRecoveredEditsDir(regiondir);
284     if (!fs.exists(editsdir))
285       return filesSorted;
286     FileStatus[] files = FSUtils.listStatus(fs, editsdir, new PathFilter() {
287       @Override
288       public boolean accept(Path p) {
289         boolean result = false;
290         try {
291           // Return files and only files that match the editfile names pattern.
292           // There can be other files in this directory other than edit files.
293           // In particular, on error, we'll move aside the bad edit file giving
294           // it a timestamp suffix. See moveAsideBadEditsFile.
295           Matcher m = HLog.EDITFILES_NAME_PATTERN.matcher(p.getName());
296           result = fs.isFile(p) && m.matches();
297           // Skip the file whose name ends with RECOVERED_LOG_TMPFILE_SUFFIX,
298           // because it means splithlog thread is writting this file.
299           if (p.getName().endsWith(HLog.RECOVERED_LOG_TMPFILE_SUFFIX)) {
300             result = false;
301           }
302         } catch (IOException e) {
303           LOG.warn("Failed isFile check on " + p);
304         }
305         return result;
306       }
307     });
308     if (files == null)
309       return filesSorted;
310     for (FileStatus status : files) {
311       filesSorted.add(status.getPath());
312     }
313     return filesSorted;
314   }
