111   /**
112    * Parses the filterString and constructs a filter using it
113    * <p>
114    * @param filterStringAsByteArray filter string given by the user
115    * @return filter object we constructed
116    */
117   public Filter parseFilterString (byte [] filterStringAsByteArray)
118     throws CharacterCodingException {
119     // stack for the operators and parenthesis
120     Stack <ByteBuffer> operatorStack = new Stack<ByteBuffer>();
121     // stack for the filter objects
122     Stack <Filter> filterStack = new Stack<Filter>();
123 
124     Filter filter = null;
125     for (int i=0; i<filterStringAsByteArray.length; i++) {
126       if (filterStringAsByteArray[i] == ParseConstants.LPAREN) {
127         // LPAREN found
128         operatorStack.push(ParseConstants.LPAREN_BUFFER);
129       } else if (filterStringAsByteArray[i] == ParseConstants.WHITESPACE ||
130                  filterStringAsByteArray[i] == ParseConstants.TAB) {
131         // WHITESPACE or TAB found
132         continue;
133       } else if (checkForOr(filterStringAsByteArray, i)) {
134         // OR found
135         i += ParseConstants.OR_ARRAY.length - 1;
136         reduce(operatorStack, filterStack, ParseConstants.OR_BUFFER);
137         operatorStack.push(ParseConstants.OR_BUFFER);
138       } else if (checkForAnd(filterStringAsByteArray, i)) {
139         // AND found
140         i += ParseConstants.AND_ARRAY.length - 1;
141         reduce(operatorStack, filterStack, ParseConstants.AND_BUFFER);
142         operatorStack.push(ParseConstants.AND_BUFFER);
143       } else if (checkForSkip(filterStringAsByteArray, i)) {
144         // SKIP found
145         i += ParseConstants.SKIP_ARRAY.length - 1;
146         reduce(operatorStack, filterStack, ParseConstants.SKIP_BUFFER);
147         operatorStack.push(ParseConstants.SKIP_BUFFER);
148       } else if (checkForWhile(filterStringAsByteArray, i)) {
149         // WHILE found
150         i += ParseConstants.WHILE_ARRAY.length - 1;
151         reduce(operatorStack, filterStack, ParseConstants.WHILE_BUFFER);
152         operatorStack.push(ParseConstants.WHILE_BUFFER);
153       } else if (filterStringAsByteArray[i] == ParseConstants.RPAREN) {
154         // RPAREN found
155         if (operatorStack.empty()) {
156           throw new IllegalArgumentException("Mismatched parenthesis");
157         }
158         ByteBuffer argumentOnTopOfStack = operatorStack.peek();
159         while (!(argumentOnTopOfStack.equals(ParseConstants.LPAREN_BUFFER))) {
160           filterStack.push(popArguments(operatorStack, filterStack));
161           if (operatorStack.empty()) {
162             throw new IllegalArgumentException("Mismatched parenthesis");
163           }
164           argumentOnTopOfStack = operatorStack.pop();
165         }
166       } else {
167         // SimpleFilterExpression found
168         byte [] filterSimpleExpression = extractFilterSimpleExpression(filterStringAsByteArray, i);
169         i+= (filterSimpleExpression.length - 1);
170         filter = parseSimpleFilterExpression(filterSimpleExpression);
171         filterStack.push(filter);
172       }
173     }
174 
175     // Finished parsing filterString
176     while (!operatorStack.empty()) {
177       filterStack.push(popArguments(operatorStack, filterStack));
178     }
179     filter = filterStack.pop();
180     if (!filterStack.empty()) {
181       throw new IllegalArgumentException("Incorrect Filter String");
182     }
183     return filter;
184   }

186 /**
187  * Extracts a simple filter expression from the filter string given by the user
188  * <p>
189  * A simpleFilterExpression is of the form: FilterName('arg', 'arg', 'arg')
190  * The user given filter string can have many simpleFilterExpressions combined
191  * using operators.
192  * <p>
193  * This function extracts a simpleFilterExpression from the
194  * larger filterString given the start offset of the simpler expression
195  * <p>
196  * @param filterStringAsByteArray filter string given by the user
197  * @param filterExpressionStartOffset start index of the simple filter expression
198  * @return byte array containing the simple filter expression
199  */
200   public byte [] extractFilterSimpleExpression (byte [] filterStringAsByteArray,
201                                                 int filterExpressionStartOffset)
202     throws CharacterCodingException {
203     int quoteCount = 0;
204     for (int i=filterExpressionStartOffset; i<filterStringAsByteArray.length; i++) {
205       if (filterStringAsByteArray[i] == ParseConstants.SINGLE_QUOTE) {
206         if (isQuoteUnescaped(filterStringAsByteArray, i)) {
207           quoteCount ++;
208         } else {
209           // To skip the next quote that has been escaped
210           i++;
211         }
212       }
213       if (filterStringAsByteArray[i] == ParseConstants.RPAREN && (quoteCount %2 ) == 0) {
214         byte [] filterSimpleExpression = new byte [i - filterExpressionStartOffset + 1];
215         Bytes.putBytes(filterSimpleExpression, 0, filterStringAsByteArray,
216                        filterExpressionStartOffset, i-filterExpressionStartOffset + 1);
217         return filterSimpleExpression;
218       }
219     }
220     throw new IllegalArgumentException("Incorrect Filter String");
221   }

223 /**
224  * Constructs a filter object given a simple filter expression
225  * <p>
226  * @param filterStringAsByteArray filter string given by the user
227  * @return filter object we constructed
228  */
229   public Filter parseSimpleFilterExpression (byte [] filterStringAsByteArray)
230     throws CharacterCodingException {
231 
232     String filterName = Bytes.toString(getFilterName(filterStringAsByteArray));
233     ArrayList<byte []> filterArguments = getFilterArguments(filterStringAsByteArray);
234     if (!filterHashMap.containsKey(filterName)) {
235       throw new IllegalArgumentException("Filter Name " + filterName + " not supported");
236     }
237     try {
238       filterName = filterHashMap.get(filterName);
239       Class c = Class.forName(filterName);
240       Class[] argTypes = new Class [] {ArrayList.class};
241       Method m = c.getDeclaredMethod("createFilterFromArguments", argTypes);
242       return (Filter) m.invoke(null,filterArguments);
243     } catch (ClassNotFoundException e) {
244       e.printStackTrace();
245     } catch (NoSuchMethodException e) {
246       e.printStackTrace();
247     } catch (IllegalAccessException e) {
248       e.printStackTrace();
249     } catch (InvocationTargetException e) {
250       e.printStackTrace();
251     }
252     throw new IllegalArgumentException("Incorrect filter string " +
253                                        new String(filterStringAsByteArray));
254   }

256 /**
257  * Returns the filter name given a simple filter expression
258  * <p>
259  * @param filterStringAsByteArray a simple filter expression
260  * @return name of filter in the simple filter expression
261  */
262   public static byte [] getFilterName (byte [] filterStringAsByteArray) {
263     int filterNameStartIndex = 0;
264     int filterNameEndIndex = 0;
265 
266     for (int i=filterNameStartIndex; i<filterStringAsByteArray.length; i++) {
267       if (filterStringAsByteArray[i] == ParseConstants.LPAREN ||
268           filterStringAsByteArray[i] == ParseConstants.WHITESPACE) {
269         filterNameEndIndex = i;
270         break;
271       }
272     }
273 
274     if (filterNameEndIndex == 0) {
275       throw new IllegalArgumentException("Incorrect Filter Name");
276     }
277 
278     byte [] filterName = new byte[filterNameEndIndex - filterNameStartIndex];
279     Bytes.putBytes(filterName, 0, filterStringAsByteArray, 0,
280                    filterNameEndIndex - filterNameStartIndex);
281     return filterName;
282   }

284 /**
285  * Returns the arguments of the filter from the filter string
286  * <p>
287  * @param filterStringAsByteArray filter string given by the user
288  * @return an ArrayList containing the arguments of the filter in the filter string
289  */
290   public static ArrayList<byte []> getFilterArguments (byte [] filterStringAsByteArray) {
291     int argumentListStartIndex = KeyValue.getDelimiter(filterStringAsByteArray, 0,
292                                                        filterStringAsByteArray.length,
293                                                        ParseConstants.LPAREN);
294     if (argumentListStartIndex == -1) {
295       throw new IllegalArgumentException("Incorrect argument list");
296     }
297 
298     int argumentStartIndex = 0;
299     int argumentEndIndex = 0;
300     ArrayList<byte []> filterArguments = new ArrayList<byte []>();
301 
302     for (int i = argumentListStartIndex + 1; i<filterStringAsByteArray.length; i++) {
303 
304       if (filterStringAsByteArray[i] == ParseConstants.WHITESPACE ||
305           filterStringAsByteArray[i] == ParseConstants.COMMA ||
306           filterStringAsByteArray[i] == ParseConstants.RPAREN) {
307         continue;
308       }
309 
310       // The argument is in single quotes - for example 'prefix'
311       if (filterStringAsByteArray[i] == ParseConstants.SINGLE_QUOTE) {
312         argumentStartIndex = i;
313         for (int j = argumentStartIndex+1; j < filterStringAsByteArray.length; j++) {
314           if (filterStringAsByteArray[j] == ParseConstants.SINGLE_QUOTE) {
315             if (isQuoteUnescaped(filterStringAsByteArray,j)) {
316               argumentEndIndex = j;
317               i = j+1;
318               byte [] filterArgument = createUnescapdArgument(filterStringAsByteArray,
319                                                               argumentStartIndex, argumentEndIndex);
320               filterArguments.add(filterArgument);
321               break;
322             } else {
323               // To jump over the second escaped quote
324               j++;
325             }
326           } else if (j == filterStringAsByteArray.length - 1) {
327             throw new IllegalArgumentException("Incorrect argument list");
328           }
329         }
330       } else {
331         // The argument is an integer, boolean, comparison operator like <, >, != etc
332         argumentStartIndex = i;
333         for (int j = argumentStartIndex; j < filterStringAsByteArray.length; j++) {
334           if (filterStringAsByteArray[j] == ParseConstants.WHITESPACE ||
335               filterStringAsByteArray[j] == ParseConstants.COMMA ||
336               filterStringAsByteArray[j] == ParseConstants.RPAREN) {
337             argumentEndIndex = j - 1;
338             i = j;
339             byte [] filterArgument = new byte [argumentEndIndex - argumentStartIndex + 1];
340             Bytes.putBytes(filterArgument, 0, filterStringAsByteArray,
341                            argumentStartIndex, argumentEndIndex - argumentStartIndex + 1);
342             filterArguments.add(filterArgument);
343             break;
344           } else if (j == filterStringAsByteArray.length - 1) {
345             throw new IllegalArgumentException("Incorrect argument list");
346           }
347         }
348       }
349     }
350     return filterArguments;
351   }

353 /**
354  * This function is called while parsing the filterString and an operator is parsed
355  * <p>
356  * @param operatorStack the stack containing the operators and parenthesis
357  * @param filterStack the stack containing the filters
358  * @param operator the operator found while parsing the filterString
359  */
360   public void reduce(Stack<ByteBuffer> operatorStack,
361                      Stack<Filter> filterStack,
362                      ByteBuffer operator) {
363     while (!operatorStack.empty() &&
364            !(ParseConstants.LPAREN_BUFFER.equals(operatorStack.peek())) &&
365            hasHigherPriority(operatorStack.peek(), operator)) {
366       filterStack.push(popArguments(operatorStack, filterStack));
367     }
368   }

370   /**
371    * Pops an argument from the operator stack and the number of arguments required by the operator
372    * from the filterStack and evaluates them
373    * <p>
374    * @param operatorStack the stack containing the operators
375    * @param filterStack the stack containing the filters
376    * @return the evaluated filter
377    */
378   public static Filter popArguments (Stack<ByteBuffer> operatorStack, Stack <Filter> filterStack) {
379     ByteBuffer argumentOnTopOfStack = operatorStack.peek();
380 
381     if (argumentOnTopOfStack.equals(ParseConstants.OR_BUFFER)) {
382       // The top of the stack is an OR
383       try {
384         ArrayList<Filter> listOfFilters = new ArrayList<Filter>();
385         while (!operatorStack.empty() && operatorStack.peek().equals(ParseConstants.OR_BUFFER)) {
386           Filter filter = filterStack.pop();
387           listOfFilters.add(0, filter);
388           operatorStack.pop();
389         }
390         Filter filter = filterStack.pop();
391         listOfFilters.add(0, filter);
392         Filter orFilter = new FilterList(FilterList.Operator.MUST_PASS_ONE, listOfFilters);
393         return orFilter;
394       } catch (EmptyStackException e) {
395         throw new IllegalArgumentException("Incorrect input string - an OR needs two filters");
396       }
397 
398     } else if (argumentOnTopOfStack.equals(ParseConstants.AND_BUFFER)) {
399       // The top of the stack is an AND
400       try {
401         ArrayList<Filter> listOfFilters = new ArrayList<Filter>();
402         while (!operatorStack.empty() && operatorStack.peek().equals(ParseConstants.AND_BUFFER)) {
403           Filter filter = filterStack.pop();
404           listOfFilters.add(0, filter);
405           operatorStack.pop();
406         }
407         Filter filter = filterStack.pop();
408         listOfFilters.add(0, filter);
409         Filter andFilter = new FilterList(FilterList.Operator.MUST_PASS_ALL, listOfFilters);
410         return andFilter;
411       } catch (EmptyStackException e) {
412         throw new IllegalArgumentException("Incorrect input string - an AND needs two filters");
413       }
414 
415     } else if (argumentOnTopOfStack.equals(ParseConstants.SKIP_BUFFER)) {
416       // The top of the stack is a SKIP
417       try {
418         Filter wrappedFilter = filterStack.pop();
419         Filter skipFilter = new SkipFilter(wrappedFilter);
420         operatorStack.pop();
421         return skipFilter;
422       } catch (EmptyStackException e) {
423         throw new IllegalArgumentException("Incorrect input string - a SKIP wraps a filter");
424       }
425 
426     } else if (argumentOnTopOfStack.equals(ParseConstants.WHILE_BUFFER)) {
427       // The top of the stack is a WHILE
428       try {
429         Filter wrappedFilter = filterStack.pop();
430         Filter whileMatchFilter = new WhileMatchFilter(wrappedFilter);
431         operatorStack.pop();
432         return whileMatchFilter;
433       } catch (EmptyStackException e) {
434         throw new IllegalArgumentException("Incorrect input string - a WHILE wraps a filter");
435       }
436 
437     } else if (argumentOnTopOfStack.equals(ParseConstants.LPAREN_BUFFER)) {
438       // The top of the stack is a LPAREN
439       try {
440         Filter filter  = filterStack.pop();
441         operatorStack.pop();
442         return filter;
443       } catch (EmptyStackException e) {
444         throw new IllegalArgumentException("Incorrect Filter String");
445       }
446 
447     } else {
448       throw new IllegalArgumentException("Incorrect arguments on operatorStack");
449     }
450   }

452 /**
453  * Returns which operator has higher precedence
454  * <p>
455  * If a has higher precedence than b, it returns true
456  * If they have the same precedence, it returns false
457  */
458   public boolean hasHigherPriority(ByteBuffer a, ByteBuffer b) {
459     if ((operatorPrecedenceHashMap.get(a) - operatorPrecedenceHashMap.get(b)) < 0) {
460       return true;
461     }
462     return false;
463   }

465 /**
466  * Removes the single quote escaping a single quote - thus it returns an unescaped argument
467  * <p>
468  * @param filterStringAsByteArray filter string given by user
469  * @param argumentStartIndex start index of the argument
470  * @param argumentEndIndex end index of the argument
471  * @return returns an unescaped argument
472  */
473   public static byte [] createUnescapdArgument (byte [] filterStringAsByteArray,
474                                                 int argumentStartIndex, int argumentEndIndex) {
475     int unescapedArgumentLength = 2;
476     for (int i = argumentStartIndex + 1; i <= argumentEndIndex - 1; i++) {
477       unescapedArgumentLength ++;
478       if (filterStringAsByteArray[i] == ParseConstants.SINGLE_QUOTE &&
479           i != (argumentEndIndex - 1) &&
480           filterStringAsByteArray[i+1] == ParseConstants.SINGLE_QUOTE) {
481         i++;
482         continue;
483       }
484     }
485 
486     byte [] unescapedArgument = new byte [unescapedArgumentLength];
487     int count = 1;
488     unescapedArgument[0] = '\'';
489     for (int i = argumentStartIndex + 1; i <= argumentEndIndex - 1; i++) {
490       if (filterStringAsByteArray [i] == ParseConstants.SINGLE_QUOTE &&
491           i != (argumentEndIndex - 1) &&
492           filterStringAsByteArray [i+1] == ParseConstants.SINGLE_QUOTE) {
493         unescapedArgument[count++] = filterStringAsByteArray [i+1];
494         i++;
495       }
496       else {
497         unescapedArgument[count++] = filterStringAsByteArray [i];
498       }
499     }
500     unescapedArgument[unescapedArgumentLength - 1] = '\'';
501     return unescapedArgument;
502   }

504 /**
505  * Checks if the current index of filter string we are on is the beginning of the keyword 'OR'
506  * <p>
507  * @param filterStringAsByteArray filter string given by the user
508  * @param indexOfOr index at which an 'O' was read
509  * @return true if the keyword 'OR' is at the current index
510  */
511   public static boolean checkForOr (byte [] filterStringAsByteArray, int indexOfOr)
512     throws CharacterCodingException, ArrayIndexOutOfBoundsException {
513 
514     try {
515       if (filterStringAsByteArray[indexOfOr] == ParseConstants.O &&
516           filterStringAsByteArray[indexOfOr+1] == ParseConstants.R &&
517           (filterStringAsByteArray[indexOfOr-1] == ParseConstants.WHITESPACE ||
518            filterStringAsByteArray[indexOfOr-1] == ParseConstants.RPAREN) &&
519           (filterStringAsByteArray[indexOfOr+2] == ParseConstants.WHITESPACE ||
520            filterStringAsByteArray[indexOfOr+2] == ParseConstants.LPAREN)) {
521         return true;
522       } else {
523         return false;
524       }
525     } catch (ArrayIndexOutOfBoundsException e) {
526       return false;
527     }
528   }

698

700       i++;
701       isNegative = true;
702     }

782 /**

784  * <p>

786  * @return the parsed comparator

788   public static ByteArrayComparable createComparator (byte [] comparator) {
789     if (comparator == null)

791     byte [][] parsedComparator = ParseFilter.parseComparator(comparator);
792     byte [] comparatorType = parsedComparator[0];
793     byte [] comparatorValue = parsedComparator[1];
794 
795

797       return new BinaryComparator(comparatorValue);
798     else if (Bytes.equals(comparatorType, ParseConstants.binaryPrefixType))

808 /**
809  * Splits a column in comparatorType:comparatorValue form into separate byte arrays
810  * <p>
811  * @param comparator the comparator
812  * @return the parsed arguments of the comparator as a 2D byte array
813  */
814   public static byte [][] parseComparator (byte [] comparator) {
815     final int index = KeyValue.getDelimiter(comparator, 0, comparator.length, ParseConstants.COLON);
816     if (index == -1) {
817       throw new IllegalArgumentException("Incorrect comparator");
818     }
819 
820     byte [][] result = new byte [2][0];
821     result[0] = new byte [index];
822     System.arraycopy(comparator, 0, result[0], 0, index);
823 
824     final int len = comparator.length - (index + 1);
825     result[1] = new byte[len];
826     System.arraycopy(comparator, index + 1, result[1], 0, len);
827 
828     return result;
829   }
