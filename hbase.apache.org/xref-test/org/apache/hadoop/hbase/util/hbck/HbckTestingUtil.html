32  public class HbckTestingUtil {
33    private static ExecutorService exec = new ScheduledThreadPoolExecutor(10);
34    public static HBaseFsck doFsck(
35        Configuration conf, boolean fix) throws Exception {
36      return doFsck(conf, fix, null);
37    }
38  
39    public static HBaseFsck doFsck(
40        Configuration conf, boolean fix, String table) throws Exception {
41      return doFsck(conf, fix, fix, fix, fix,fix, fix, fix, table);
42    }
43  
44    public static HBaseFsck doFsck(Configuration conf, boolean fixAssignments,
45        boolean fixMeta, boolean fixHdfsHoles, boolean fixHdfsOverlaps,
46        boolean fixHdfsOrphans, boolean fixTableOrphans, boolean fixVersionFile,
47        String table) throws Exception {
48      HBaseFsck fsck = new HBaseFsck(conf, exec);
49      fsck.connect();
50      fsck.setDisplayFullReport(); // i.e. -details
51      fsck.setTimeLag(0);
52      fsck.setFixAssignments(fixAssignments);
53      fsck.setFixMeta(fixMeta);
54      fsck.setFixHdfsHoles(fixHdfsHoles);
55      fsck.setFixHdfsOverlaps(fixHdfsOverlaps);
56      fsck.setFixHdfsOrphans(fixHdfsOrphans);
57      fsck.setFixTableOrphans(fixTableOrphans);
58      fsck.setFixVersionFile(fixVersionFile);
59      if (table != null) {
60        fsck.includeTable(table);
61      }
62      fsck.onlineHbck();
63      return fsck;
64    }
65  
66    /**
67     * Runs hbck with the -sidelineCorruptHFiles option
68     * @param conf
69     * @param table table constraint
70     * @return <returncode, hbckInstance>
71     * @throws Exception
72     */
73    public static HBaseFsck doHFileQuarantine(Configuration conf, String table) throws Exception {
74      String[] args = {"-sidelineCorruptHFiles", "-ignorePreCheckPermission", table};
75      HBaseFsck hbck = new HBaseFsck(conf, exec);
76      hbck.exec(exec, args);
77      return hbck;
78    }

80    public static void assertNoErrors(HBaseFsck fsck) throws Exception {
81      List<ERROR_CODE> errs = fsck.getErrors().getErrorList();
82      assertEquals(new ArrayList<ERROR_CODE>(), errs);
83    }
84  
85    public static void assertErrors(HBaseFsck fsck, ERROR_CODE[] expectedErrors) {
86      List<ERROR_CODE> errs = fsck.getErrors().getErrorList();
87      assertEquals(Arrays.asList(expectedErrors), errs);
88    }
