39  /**
40   * Compact passed set of files.
41   * Create an instance and then call {@ink #compact(Store, Collection, boolean, long)}.
42   */
43  @InterfaceAudience.Private
44  class Compactor extends Configured {
45    private static final Log LOG = LogFactory.getLog(Compactor.class);
46    private CompactionProgress progress;
47  
48    Compactor(final Configuration c) {
49      super(c);
50    }
51  
52    /**
53     * Do a minor/major compaction on an explicit set of storefiles from a Store.
54     *
55     * @param store Store the files belong to
56     * @param filesToCompact which files to compact
57     * @param majorCompaction true to major compact (prune all deletes, max versions, etc)
58     * @param maxId Readers maximum sequence id.
59     * @return Product of compaction or null if all cells expired or deleted and
60     * nothing made it through the compaction.
61     * @throws IOException
62     */
63    StoreFile.Writer compact(final HStore store,
64        final Collection<StoreFile> filesToCompact,
65        final boolean majorCompaction, final long maxId)
66    throws IOException {
67      // Calculate maximum key count after compaction (for blooms)
68      // Also calculate earliest put timestamp if major compaction
69      int maxKeyCount = 0;
70      long earliestPutTs = HConstants.LATEST_TIMESTAMP;
71      for (StoreFile file: filesToCompact) {
72        StoreFile.Reader r = file.getReader();
73        if (r == null) {
74          LOG.warn("Null reader for " + file.getPath());
75          continue;
76        }
77        // NOTE: getFilterEntries could cause under-sized blooms if the user
78        // switches bloom type (e.g. from ROW to ROWCOL)
79        long keyCount = (r.getBloomFilterType() == store.getFamily().getBloomFilterType())?
80          r.getFilterEntries() : r.getEntries();
81        maxKeyCount += keyCount;
82        // For major compactions calculate the earliest put timestamp of all
83        // involved storefiles. This is used to remove family delete marker during
84        // compaction.
85        if (majorCompaction) {
86          byte [] tmp = r.loadFileInfo().get(StoreFile.EARLIEST_PUT_TS);
87          if (tmp == null) {
88            // There's a file with no information, must be an old one
89            // assume we have very old puts
90            earliestPutTs = HConstants.OLDEST_TIMESTAMP;
91          } else {
92            earliestPutTs = Math.min(earliestPutTs, Bytes.toLong(tmp));
93          }
94        }
95        if (LOG.isDebugEnabled()) {
96          LOG.debug("Compacting " + file +
97            ", keycount=" + keyCount +
98            ", bloomtype=" + r.getBloomFilterType().toString() +
99            ", size=" + StringUtils.humanReadableInt(r.length()) +
100           ", encoding=" + r.getHFileReader().getEncodingOnDisk() +
101           (majorCompaction? ", earliestPutTs=" + earliestPutTs: ""));
102       }
103     }
104 
105     // keep track of compaction progress
106     this.progress = new CompactionProgress(maxKeyCount);
107 
108     // For each file, obtain a scanner:
109     List<StoreFileScanner> scanners = StoreFileScanner
110       .getScannersForStoreFiles(filesToCompact, false, false, true);
111 
112     // Get some configs
113     int compactionKVMax = getConf().getInt(HConstants.COMPACTION_KV_MAX, 10);
114     Compression.Algorithm compression = store.getFamily().getCompression();
115     // Avoid overriding compression setting for major compactions if the user
116     // has not specified it separately
117     Compression.Algorithm compactionCompression =
118       (store.getFamily().getCompactionCompression() != Compression.Algorithm.NONE) ?
119       store.getFamily().getCompactionCompression(): compression;
120     // Make the instantiation lazy in case compaction produces no product; i.e.
121     // where all source cells are expired or deleted.
122     StoreFile.Writer writer = null;
123     // Find the smallest read point across all the Scanners.
124     long smallestReadPoint = store.getHRegion().getSmallestReadPoint();
125     MultiVersionConsistencyControl.setThreadReadPoint(smallestReadPoint);
126     try {
127       InternalScanner scanner = null;
128       try {
129         if (store.getHRegion().getCoprocessorHost() != null) {
130           scanner = store
131               .getHRegion()
132               .getCoprocessorHost()
133               .preCompactScannerOpen(store, scanners,
134                   majorCompaction ? ScanType.MAJOR_COMPACT : ScanType.MINOR_COMPACT, earliestPutTs);
135         }
136         if (scanner == null) {
137           Scan scan = new Scan();
138           scan.setMaxVersions(store.getFamily().getMaxVersions());
139           /* Include deletes, unless we are doing a major compaction */
140           scanner = new StoreScanner(store, store.scanInfo, scan, scanners,
141             majorCompaction? ScanType.MAJOR_COMPACT : ScanType.MINOR_COMPACT,
142             smallestReadPoint, earliestPutTs);
143         }
144         if (store.getHRegion().getCoprocessorHost() != null) {
145           InternalScanner cpScanner =
146             store.getHRegion().getCoprocessorHost().preCompact(store, scanner);
147           // NULL scanner returned from coprocessor hooks means skip normal processing
148           if (cpScanner == null) {
149             return null;
150           }
151           scanner = cpScanner;
152         }
153 
154         int bytesWritten = 0;
155         // Since scanner.next() can return 'false' but still be delivering data,
156         // we have to use a do/while loop.
157         List<KeyValue> kvs = new ArrayList<KeyValue>();
158         // Limit to "hbase.hstore.compaction.kv.max" (default 10) to avoid OOME
159         boolean hasMore;
160         do {
161           hasMore = scanner.next(kvs, compactionKVMax);
162           // Create the writer even if no kv(Empty store file is also ok),
163           // because we need record the max seq id for the store file, see
164           // HBASE-6059
165           if (writer == null) {
166             writer = store.createWriterInTmp(maxKeyCount, compactionCompression, true);
167           }
168           if (writer != null) {
169             // output to writer:
170             for (KeyValue kv : kvs) {
171               if (kv.getMemstoreTS() <= smallestReadPoint) {
172                 kv.setMemstoreTS(0);
173               }
174               writer.append(kv);
175               // update progress per key
176               ++progress.currentCompactedKVs;
177 
178               // check periodically to see if a system stop is requested
179               if (HStore.closeCheckInterval > 0) {
180                 bytesWritten += kv.getLength();
181                 if (bytesWritten > HStore.closeCheckInterval) {
182                   bytesWritten = 0;
183                   isInterrupted(store, writer);
184                 }
185               }
186             }
187           }
188           kvs.clear();
189         } while (hasMore);
190       } finally {
191         if (scanner != null) {
192           scanner.close();
193         }
194       }
195     } finally {
196       if (writer != null) {
197         writer.appendMetadata(maxId, majorCompaction);
198         writer.close();
199       }
200     }
201     return writer;
202   }
203 
204   void isInterrupted(final HStore store, final StoreFile.Writer writer)
205   throws IOException {
206     if (store.getHRegion().areWritesEnabled()) return;
207     // Else cleanup.
208     writer.close();
209     store.getFileSystem().delete(writer.getPath(), false);
210     throw new InterruptedIOException( "Aborting compaction of store " + store +
211       " in region " + store.getHRegion() + " because it was interrupted.");
212   }
213 
214   CompactionProgress getProgress() {
215     return this.progress;
216   }
