84    private static byte[][] makeKeys() {
85      byte [][] starterKeys = HBaseTestingUtility.KEYS;
86      // Create a "non-uniform" test set with the following characteristics:
87      // a) Unequal number of keys per region
88  
89      // Don't use integer as a multiple, so that we have a number of keys that is
90      // not a multiple of the number of regions
91      int numKeys = (int) ((float) starterKeys.length * 10.33F);
92  
93      List<byte[]> keys = new ArrayList<byte[]>();
94      for (int i = 0; i < numKeys; i++) {
95        int kIdx = i % starterKeys.length;
96        byte[] k = starterKeys[kIdx];
97        byte[] cp = new byte[k.length + 1];
98        System.arraycopy(k, 0, cp, 0, k.length);
99        cp[k.length] = new Integer(i % 256).byteValue();
100       keys.add(cp);
101     }
102 
103     // b) Same duplicate keys (showing multiple Gets/Puts to the same row, which
104     // should work)
105     // c) keys are not in sorted order (within a region), to ensure that the
106     // sorting code and index mapping doesn't break the functionality
107     for (int i = 0; i < 100; i++) {
108       int kIdx = i % starterKeys.length;
109       byte[] k = starterKeys[kIdx];
110       byte[] cp = new byte[k.length + 1];
111       System.arraycopy(k, 0, cp, 0, k.length);
112       cp[k.length] = new Integer(i % 256).byteValue();
113       keys.add(cp);
114     }
115     return keys.toArray(new byte [][] {new byte [] {}});
116   }

141   @Test(timeout=300000) 
142   public void testBatchWithGet() throws Exception {
143     LOG.info("test=testBatchWithGet");
144     HTable table = new HTable(UTIL.getConfiguration(), TEST_TABLE);
145 
146     // load test data
147     List<Row> puts = constructPutRequests();
148     table.batch(puts);
149 
150     // create a list of gets and run it
151     List<Row> gets = new ArrayList<Row>();
152     for (byte[] k : KEYS) {
153       Get get = new Get(k);
154       get.addColumn(BYTES_FAMILY, QUALIFIER);
155       gets.add(get);
156     }
157     Result[] multiRes = new Result[gets.size()];
158     table.batch(gets, multiRes);
159 
160     // Same gets using individual call API
161     List<Result> singleRes = new ArrayList<Result>();
162     for (Row get : gets) {
163       singleRes.add(table.get((Get) get));
164     }
165 
166     // Compare results
167     Assert.assertEquals(singleRes.size(), multiRes.length);
168     for (int i = 0; i < singleRes.size(); i++) {
169       Assert.assertTrue(singleRes.get(i).containsColumn(BYTES_FAMILY, QUALIFIER));
170       KeyValue[] singleKvs = singleRes.get(i).raw();
171       KeyValue[] multiKvs = multiRes[i].raw();
172       for (int j = 0; j < singleKvs.length; j++) {
173         Assert.assertEquals(singleKvs[j], multiKvs[j]);
174         Assert.assertEquals(0, Bytes.compareTo(singleKvs[j].getValue(), multiKvs[j]
175             .getValue()));
176       }
177     }
178     table.close();
179   }

229   private void doTestFlushCommits(boolean doAbort) throws Exception {
230     // Load the data
231     LOG.info("get new table");
232     HTable table = new HTable(UTIL.getConfiguration(), TEST_TABLE);
233     table.setAutoFlush(false);
234     table.setWriteBufferSize(10 * 1024 * 1024);
235 
236     LOG.info("constructPutRequests");
237     List<Row> puts = constructPutRequests();
238     for (Row put : puts) {
239       table.put((Put) put);
240     }
241     LOG.info("puts");
242     table.flushCommits();
243     if (doAbort) {
244       LOG.info("Aborted=" + UTIL.getMiniHBaseCluster().abortRegionServer(0));
245 
246       // try putting more keys after the abort. same key/qual... just validating
247       // no exceptions thrown
248       puts = constructPutRequests();
249       for (Row put : puts) {
250         table.put((Put) put);
251       }
252 
253       table.flushCommits();
254     }
255 
256     LOG.info("validating loaded data");
257     validateLoadedData(table);
258 
259     // Validate server and region count
260     List<JVMClusterUtil.RegionServerThread> liveRSs = UTIL.getMiniHBaseCluster().getLiveRegionServerThreads();
261     int count = 0;
262     for (JVMClusterUtil.RegionServerThread t: liveRSs) {
263       count++;
264       LOG.info("Count=" + count + ", Alive=" + t.getRegionServer());
265     }
266     LOG.info("Count=" + count);
267     Assert.assertEquals("Server count=" + count + ", abort=" + doAbort, (doAbort? 1 : 2), count);
268     for (JVMClusterUtil.RegionServerThread t: liveRSs) {
269       int regions = ProtobufUtil.getOnlineRegions(t.getRegionServer()).size();
270       Assert.assertTrue("Count of regions=" + regions, regions > 10);
271     }
272     table.close();
273     LOG.info("done");
274   }

428   @Test(timeout=300000)
429   public void testBatchWithMixedActions() throws Exception {
430     LOG.info("test=testBatchWithMixedActions");
431     HTable table = new HTable(UTIL.getConfiguration(), TEST_TABLE);
432 
433     // Load some data to start
434     Object[] results = table.batch(constructPutRequests());
435     validateSizeAndEmpty(results, KEYS.length);
436 
437     // Batch: get, get, put(new col), delete, get, get of put, get of deleted,
438     // put
439     List<Row> actions = new ArrayList<Row>();
440 
441     byte[] qual2 = Bytes.toBytes("qual2");
442     byte[] val2 = Bytes.toBytes("putvalue2");
443 
444     // 0 get
445     Get get = new Get(KEYS[10]);
446     get.addColumn(BYTES_FAMILY, QUALIFIER);
447     actions.add(get);
448 
449     // 1 get
450     get = new Get(KEYS[11]);
451     get.addColumn(BYTES_FAMILY, QUALIFIER);
452     actions.add(get);
453 
454     // 2 put of new column
455     Put put = new Put(KEYS[10]);
456     put.add(BYTES_FAMILY, qual2, val2);
457     actions.add(put);
458 
459     // 3 delete
460     Delete delete = new Delete(KEYS[20]);
461     delete.deleteFamily(BYTES_FAMILY);
462     actions.add(delete);
463 
464     // 4 get
465     get = new Get(KEYS[30]);
466     get.addColumn(BYTES_FAMILY, QUALIFIER);
467     actions.add(get);
468 
469     // There used to be a 'get' of a previous put here, but removed
470     // since this API really cannot guarantee order in terms of mixed
471     // get/puts.
472 
473     // 5 put of new column
474     put = new Put(KEYS[40]);
475     put.add(BYTES_FAMILY, qual2, val2);
476     actions.add(put);
477 
478     results = table.batch(actions);
479 
480     // Validation
481 
482     validateResult(results[0]);
483     validateResult(results[1]);
484     validateEmpty(results[2]);
485     validateEmpty(results[3]);
486     validateResult(results[4]);
487     validateEmpty(results[5]);
488 
489     // validate last put, externally from the batch
490     get = new Get(KEYS[40]);
491     get.addColumn(BYTES_FAMILY, qual2);
492     Result r = table.get(get);
493     validateResult(r, qual2, val2);
494 
495     table.close();
496   }
