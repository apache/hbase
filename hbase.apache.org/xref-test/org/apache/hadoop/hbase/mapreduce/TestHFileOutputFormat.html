104   /**
105    * Simple mapper that makes KeyValue output.
106    */
107   static class RandomKVGeneratingMapper
108   extends Mapper<NullWritable, NullWritable,
109                  ImmutableBytesWritable, KeyValue> {
110 
111     private int keyLength;
112     private static final int KEYLEN_DEFAULT=10;
113     private static final String KEYLEN_CONF="randomkv.key.length";
114 
115     private int valLength;
116     private static final int VALLEN_DEFAULT=10;
117     private static final String VALLEN_CONF="randomkv.val.length";
118 
119     @Override
120     protected void setup(Context context) throws IOException,
121         InterruptedException {
122       super.setup(context);
123 
124       Configuration conf = context.getConfiguration();
125       keyLength = conf.getInt(KEYLEN_CONF, KEYLEN_DEFAULT);
126       valLength = conf.getInt(VALLEN_CONF, VALLEN_DEFAULT);
127     }
128 
129     protected void map(
130         NullWritable n1, NullWritable n2,
131         Mapper<NullWritable, NullWritable,
132                ImmutableBytesWritable,KeyValue>.Context context)
133         throws java.io.IOException ,InterruptedException
134     {
135 
136       byte keyBytes[] = new byte[keyLength];
137       byte valBytes[] = new byte[valLength];
138 
139       int taskId = context.getTaskAttemptID().getTaskID().getId();
140       assert taskId < Byte.MAX_VALUE : "Unit tests dont support > 127 tasks!";
141 
142       Random random = new Random();
143       for (int i = 0; i < ROWSPERSPLIT; i++) {
144 
145         random.nextBytes(keyBytes);
146         // Ensure that unique tasks generate unique keys
147         keyBytes[keyLength - 1] = (byte)(taskId & 0xFF);
148         random.nextBytes(valBytes);
149         ImmutableBytesWritable key = new ImmutableBytesWritable(keyBytes);
150 
151         for (byte[] family : TestHFileOutputFormat.FAMILIES) {
152           KeyValue kv = new KeyValue(keyBytes, family,
153               PerformanceEvaluation.QUALIFIER_NAME, valBytes);
154           context.write(key, kv);
155         }
156       }
157     }
158   }

160   private void setupRandomGeneratorMapper(Job job) {
161     job.setInputFormatClass(NMapInputFormat.class);
162     job.setMapperClass(RandomKVGeneratingMapper.class);
163     job.setMapOutputKeyClass(ImmutableBytesWritable.class);
164     job.setMapOutputValueClass(KeyValue.class);
165   }

167   /**
168    * Test that {@link HFileOutputFormat} RecordWriter amends timestamps if
169    * passed a keyvalue whose timestamp is {@link HConstants#LATEST_TIMESTAMP}.
170    * @see <a href="https://issues.apache.org/jira/browse/HBASE-2615">HBASE-2615</a>
171    */
172   @Test
173   public void test_LATEST_TIMESTAMP_isReplaced()
174   throws Exception {
175     Configuration conf = new Configuration(this.util.getConfiguration());
176     RecordWriter<ImmutableBytesWritable, KeyValue> writer = null;
177     TaskAttemptContext context = null;
178     Path dir =
179       util.getDataTestDir("test_LATEST_TIMESTAMP_isReplaced");
180     try {
181       Job job = new Job(conf);
182       FileOutputFormat.setOutputPath(job, dir);
183       context = createTestTaskAttemptContext(job);
184       HFileOutputFormat hof = new HFileOutputFormat();
185       writer = hof.getRecordWriter(context);
186       final byte [] b = Bytes.toBytes("b");
187 
188       // Test 1.  Pass a KV that has a ts of LATEST_TIMESTAMP.  It should be
189       // changed by call to write.  Check all in kv is same but ts.
190       KeyValue kv = new KeyValue(b, b, b);
191       KeyValue original = kv.clone();
192       writer.write(new ImmutableBytesWritable(), kv);
193       assertFalse(original.equals(kv));
194       assertTrue(Bytes.equals(original.getRow(), kv.getRow()));
195       assertTrue(original.matchingColumn(kv.getFamily(), kv.getQualifier()));
196       assertNotSame(original.getTimestamp(), kv.getTimestamp());
197       assertNotSame(HConstants.LATEST_TIMESTAMP, kv.getTimestamp());
198 
199       // Test 2. Now test passing a kv that has explicit ts.  It should not be
200       // changed by call to record write.
201       kv = new KeyValue(b, b, b, kv.getTimestamp() - 1, b);
202       original = kv.clone();
203       writer.write(new ImmutableBytesWritable(), kv);
204       assertTrue(original.equals(kv));
205     } finally {
206       if (writer != null && context != null) writer.close(context);
207       dir.getFileSystem(conf).delete(dir, true);
208     }
209   }

219    * Test that {@link HFileOutputFormat} creates an HFile with TIMERANGE
220    * metadata used by time-restricted scans.
221    */
222   @Test
223   public void test_TIMERANGE() throws Exception {
224     Configuration conf = new Configuration(this.util.getConfiguration());
225     RecordWriter<ImmutableBytesWritable, KeyValue> writer = null;
226     TaskAttemptContext context = null;
227     Path dir =
228       util.getDataTestDir("test_TIMERANGE_present");
229     LOG.info("Timerange dir writing to dir: "+ dir);
230     try {
231       // build a record writer using HFileOutputFormat
232       Job job = new Job(conf);
233       FileOutputFormat.setOutputPath(job, dir);
234       context = createTestTaskAttemptContext(job);
235       HFileOutputFormat hof = new HFileOutputFormat();
236       writer = hof.getRecordWriter(context);
237 
238       // Pass two key values with explicit times stamps
239       final byte [] b = Bytes.toBytes("b");
240 
241       // value 1 with timestamp 2000
242       KeyValue kv = new KeyValue(b, b, b, 2000, b);
243       KeyValue original = kv.clone();
244       writer.write(new ImmutableBytesWritable(), kv);
245       assertEquals(original,kv);
246 
247       // value 2 with timestamp 1000
248       kv = new KeyValue(b, b, b, 1000, b);
249       original = kv.clone();
250       writer.write(new ImmutableBytesWritable(), kv);
251       assertEquals(original, kv);
252 
253       // verify that the file has the proper FileInfo.
254       writer.close(context);
255 
256       // the generated file lives 1 directory down from the attempt directory
257       // and is the only file, e.g.
258       // _attempt__0000_r_000000_0/b/1979617994050536795
259       FileSystem fs = FileSystem.get(conf);
260       Path attemptDirectory = hof.getDefaultWorkFile(context, "").getParent();
261       FileStatus[] sub1 = fs.listStatus(attemptDirectory);
262       FileStatus[] file = fs.listStatus(sub1[0].getPath());
263 
264       // open as HFile Reader and pull out TIMERANGE FileInfo.
265       HFile.Reader rd = HFile.createReader(fs, file[0].getPath(),
266           new CacheConfig(conf));
267       Map<byte[],byte[]> finfo = rd.loadFileInfo();
268       byte[] range = finfo.get("TIMERANGE".getBytes());
269       assertNotNull(range);
270 
271       // unmarshall and check values.
272       TimeRangeTracker timeRangeTracker = new TimeRangeTracker();
273       Writables.copyWritable(range, timeRangeTracker);
274       LOG.info(timeRangeTracker.getMinimumTimestamp() +
275           "...." + timeRangeTracker.getMaximumTimestamp());
276       assertEquals(1000, timeRangeTracker.getMinimumTimestamp());
277       assertEquals(2000, timeRangeTracker.getMaximumTimestamp());
278       rd.close();
279     } finally {
280       if (writer != null && context != null) writer.close(context);
281       dir.getFileSystem(conf).delete(dir, true);
282     }
283   }

359   private void doIncrementalLoadTest(
360       boolean shouldChangeRegions) throws Exception {
361     util = new HBaseTestingUtility();
362     Configuration conf = util.getConfiguration();
363     byte[][] startKeys = generateRandomStartKeys(5);
364     HBaseAdmin admin = null;
365     try {
366       util.startMiniCluster();
367       Path testDir = util.getDataTestDirOnTestFS("testLocalMRIncrementalLoad");
368       admin = new HBaseAdmin(conf);
369       HTable table = util.createTable(TABLE_NAME, FAMILIES);
370       assertEquals("Should start with empty table",
371           0, util.countRows(table));
372       int numRegions = util.createMultiRegions(
373           util.getConfiguration(), table, FAMILIES[0], startKeys);
374       assertEquals("Should make 5 regions", numRegions, 5);
375 
376       // Generate the bulk load files
377       util.startMiniMapReduceCluster();
378       runIncrementalPELoad(conf, table, testDir);
379       // This doesn't write into the table, just makes files
380       assertEquals("HFOF should not touch actual table",
381           0, util.countRows(table));
382 
383 
384       // Make sure that a directory was created for every CF
385       int dir = 0;
386       for (FileStatus f : testDir.getFileSystem(conf).listStatus(testDir)) {
387         for (byte[] family : FAMILIES) {
388           if (Bytes.toString(family).equals(f.getPath().getName())) {
389             ++dir;
390           }
391         }
392       }
393       assertEquals("Column family not found in FS.", FAMILIES.length, dir);
394 
395       // handle the split case
396       if (shouldChangeRegions) {
397         LOG.info("Changing regions in table");
398         admin.disableTable(table.getTableName());
399         while(util.getMiniHBaseCluster().getMaster().getAssignmentManager().
400             getRegionStates().isRegionsInTransition()) {
401           Threads.sleep(200);
402           LOG.info("Waiting on table to finish disabling");
403         }
404         byte[][] newStartKeys = generateRandomStartKeys(15);
405         util.createMultiRegions(
406             util.getConfiguration(), table, FAMILIES[0], newStartKeys);
407         admin.enableTable(table.getTableName());
408         while (table.getRegionLocations().size() != 15 ||
409             !admin.isTableAvailable(table.getTableName())) {
410           Thread.sleep(200);
411           LOG.info("Waiting for new region assignment to happen");
412         }
413       }
414 
415       // Perform the actual load
416       new LoadIncrementalHFiles(conf).doBulkLoad(testDir, table);
417 
418       // Ensure data shows up
419       int expectedRows = NMapInputFormat.getNumMapTasks(conf) * ROWSPERSPLIT;
420       assertEquals("LoadIncrementalHFiles should put expected data in table",
421           expectedRows, util.countRows(table));
422       Scan scan = new Scan();
423       ResultScanner results = table.getScanner(scan);
424       for (Result res : results) {
425         assertEquals(FAMILIES.length, res.raw().length);
426         KeyValue first = res.raw()[0];
427         for (KeyValue kv : res.raw()) {
428           assertTrue(KeyValue.COMPARATOR.matchingRows(first, kv));
429           assertTrue(Bytes.equals(first.getValue(), kv.getValue()));
430         }
431       }
432       results.close();
433       String tableDigestBefore = util.checksumRows(table);
434 
435       // Cause regions to reopen
436       admin.disableTable(TABLE_NAME);
437       while (!admin.isTableDisabled(TABLE_NAME)) {
438         Thread.sleep(200);
439         LOG.info("Waiting for table to disable");
440       }
441       admin.enableTable(TABLE_NAME);
442       util.waitTableAvailable(TABLE_NAME, 30000);
443       assertEquals("Data should remain after reopening of regions",
444           tableDigestBefore, util.checksumRows(table));
445     } finally {
446       if (admin != null) admin.close();
447       util.shutdownMiniMapReduceCluster();
448       util.shutdownMiniCluster();
449     }
450   }

546    * Test that {@link HFileOutputFormat} RecordWriter uses compression settings
547    * from the column family descriptor
548    */
549   @Test
550   public void testColumnFamilyCompression() throws Exception {
551     Configuration conf = new Configuration(this.util.getConfiguration());
552     RecordWriter<ImmutableBytesWritable, KeyValue> writer = null;
553     TaskAttemptContext context = null;
554     Path dir =
555         util.getDataTestDirOnTestFS("testColumnFamilyCompression");
556 
557     HTable table = Mockito.mock(HTable.class);
558 
559     Map<String, Compression.Algorithm> configuredCompression =
560       new HashMap<String, Compression.Algorithm>();
561     Compression.Algorithm[] supportedAlgos = getSupportedCompressionAlgorithms();
562 
563     int familyIndex = 0;
564     for (byte[] family : FAMILIES) {
565       configuredCompression.put(Bytes.toString(family),
566                                 supportedAlgos[familyIndex++ % supportedAlgos.length]);
567     }
568     setupMockColumnFamilies(table, configuredCompression);
569 
570     // set up the table to return some mock keys
571     setupMockStartKeys(table);
572 
573     try {
574       // partial map red setup to get an operational writer for testing
575       // We turn off the sequence file compression, because DefaultCodec
576       // pollutes the GZip codec pool with an incompatible compressor.
577       conf.set("io.seqfile.compression.type", "NONE");
578       Job job = new Job(conf, "testLocalMRIncrementalLoad");
579       job.setWorkingDirectory(util.getDataTestDirOnTestFS("testColumnFamilyCompression"));
580       setupRandomGeneratorMapper(job);
581       HFileOutputFormat.configureIncrementalLoad(job, table);
582       FileOutputFormat.setOutputPath(job, dir);
583       context = createTestTaskAttemptContext(job);
584       HFileOutputFormat hof = new HFileOutputFormat();
585       writer = hof.getRecordWriter(context);
586 
587       // write out random rows
588       writeRandomKeyValues(writer, context, ROWSPERSPLIT);
589       writer.close(context);
590 
591       // Make sure that a directory was created for every CF
592       FileSystem fileSystem = dir.getFileSystem(conf);
593 
594       // commit so that the filesystem has one directory per column family
595       hof.getOutputCommitter(context).commitTask(context);
596       hof.getOutputCommitter(context).commitJob(context);
597       for (byte[] family : FAMILIES) {
598         String familyStr = new String(family);
599         boolean found = false;
600         for (FileStatus f : fileSystem.listStatus(dir)) {
601 
602           if (Bytes.toString(family).equals(f.getPath().getName())) {
603             // we found a matching directory
604             found = true;
605 
606             // verify that the compression on this file matches the configured
607             // compression
608             Path dataFilePath = fileSystem.listStatus(f.getPath())[0].getPath();
609             Reader reader = HFile.createReader(fileSystem, dataFilePath,
610                 new CacheConfig(conf));
611             reader.loadFileInfo();
612             assertEquals("Incorrect compression used for column family " + familyStr
613                          + "(reader: " + reader + ")",
614                          configuredCompression.get(familyStr), reader.getCompressionAlgorithm());
615             break;
616           }
617         }

619         if (!found) {

621         }

830   private void quickPoll(Callable<Boolean> c, int waitMs) throws Exception {
831     int sleepMs = 10;
832     int retries = (int) Math.ceil(((double) waitMs) / sleepMs);
833     while (retries-- > 0) {
834       if (c.call().booleanValue()) {
835         return;
836       }
837       Thread.sleep(sleepMs);
838     }
839     fail();
840   }
