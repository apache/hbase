49    /**
50     * Generates and returns a Put containing the region into for the catalog table
51     */
52    public static Put makePutFromRegionInfo(HRegionInfo regionInfo)
53    throws IOException {
54      Put put = new Put(regionInfo.getRegionName());
55      addRegionInfo(put, regionInfo);
56      return put;
57    }

59    /**
60     * Adds split daughters to the Put
61     */
62    public static Put addDaughtersToPut(Put put, HRegionInfo splitA, HRegionInfo splitB) {
63      if (splitA != null) {
64        put.add(HConstants.CATALOG_FAMILY, HConstants.SPLITA_QUALIFIER, splitA.toByteArray());
65      }
66      if (splitB != null) {
67        put.add(HConstants.CATALOG_FAMILY, HConstants.SPLITB_QUALIFIER, splitB.toByteArray());
68      }
69      return put;
70    }

163   /**
164    * Adds a META row for the specified new region to the given catalog table. The
165    * HTable is not flushed or closed.
166    * @param meta the HTable for META
167    * @param regionInfo region information
168    * @throws IOException if problem connecting or updating meta

170   public static void addRegionToMeta(HTable meta, HRegionInfo regionInfo) throws IOException {
171     addRegionToMeta(meta, regionInfo, null, null);
172   }

174   /**
175    * Adds a (single) META row for the specified new region and its daughters. Note that this does
176    * not add its daughter's as different rows, but adds information about the daughters
177    * in the same row as the parent. Use
178    * {@link #offlineParentInMeta(CatalogTracker, HRegionInfo, HRegionInfo, HRegionInfo)}
179    * and {@link #addDaughter(CatalogTracker, HRegionInfo, ServerName)}  if you want to do that.
180    * @param meta the HTable for META
181    * @param regionInfo region information
182    * @param splitA first split daughter of the parent regionInfo
183    * @param splitB second split daughter of the parent regionInfo
184    * @throws IOException if problem connecting or updating meta
185    */
186   public static void addRegionToMeta(HTable meta, HRegionInfo regionInfo,
187       HRegionInfo splitA, HRegionInfo splitB) throws IOException {
188     Put put = makePutFromRegionInfo(regionInfo);
189     addDaughtersToPut(put, splitA, splitB);
190     meta.put(put);
191     if (LOG.isDebugEnabled()) {
192       LOG.debug("Added region " + regionInfo.getRegionNameAsString() + " to META");
193     }
194   }

196   /**
197    * Adds a META row for each of the specified new regions.
198    * @param catalogTracker CatalogTracker
199    * @param regionInfos region information list
200    * @throws IOException if problem connecting or updating meta
201    */
202   public static void addRegionsToMeta(CatalogTracker catalogTracker,
203       List<HRegionInfo> regionInfos)
204   throws IOException {
205     List<Put> puts = new ArrayList<Put>();
206     for (HRegionInfo regionInfo : regionInfos) {
207       puts.add(makePutFromRegionInfo(regionInfo));
208     }
209     putsToMetaTable(catalogTracker, puts);
210     LOG.info("Added " + puts.size() + " regions in META");
211   }

213   /**
214    * Offline parent in meta.
215    * Used when splitting.
216    * @param catalogTracker
217    * @param parent
218    * @param a Split daughter region A
219    * @param b Split daughter region B
220    * @throws NotAllMetaRegionsOnlineException
221    * @throws IOException
222    */
223   public static void offlineParentInMeta(CatalogTracker catalogTracker,
224       HRegionInfo parent, final HRegionInfo a, final HRegionInfo b)
225   throws NotAllMetaRegionsOnlineException, IOException {
226     HRegionInfo copyOfParent = new HRegionInfo(parent);
227     copyOfParent.setOffline(true);
228     copyOfParent.setSplit(true);
229     HTable meta = MetaReader.getMetaHTable(catalogTracker);
230     try {
231       addRegionToMeta(meta, copyOfParent, a, b);
232       LOG.info("Offlined parent region " + parent.getRegionNameAsString() +
233           " in META");
234     } finally {
235       meta.close();
236     }
237   }

239   public static void addDaughter(final CatalogTracker catalogTracker,
240       final HRegionInfo regionInfo, final ServerName sn)
241   throws NotAllMetaRegionsOnlineException, IOException {
242     Put put = new Put(regionInfo.getRegionName());
243     addRegionInfo(put, regionInfo);
244     if (sn != null) addLocation(put, sn);
245     putToMetaTable(catalogTracker, put);
246     LOG.info("Added daughter " + regionInfo.getRegionNameAsString() +
247       (sn == null? ", serverName=null": ", serverName=" + sn.toString()));
248   }

289   /**
290    * Updates the location of the specified region to be the specified server.
291    * <p>
292    * Connects to the specified server which should be hosting the specified
293    * catalog region name to perform the edit.
294    *
295    * @param catalogTracker
296    * @param regionInfo region to update location of
297    * @param sn Server name
298    * @throws IOException In particular could throw {@link java.net.ConnectException}
299    * if the server is down on other end.
300    */
301   private static void updateLocation(final CatalogTracker catalogTracker,
302       HRegionInfo regionInfo, ServerName sn)
303   throws IOException {
304     Put put = new Put(regionInfo.getRegionName());
305     addLocation(put, sn);
306     putToCatalogTable(catalogTracker, put);
307     LOG.info("Updated row " + regionInfo.getRegionNameAsString() +
308       " with server=" + sn);
309   }

325   /**
326    * Deletes daughters references in offlined split parent.
327    * @param catalogTracker
328    * @param parent Parent row we're to remove daughter reference from
329    * @throws NotAllMetaRegionsOnlineException
330    * @throws IOException
331    */
332   public static void deleteDaughtersReferencesInParent(CatalogTracker catalogTracker,
333       final HRegionInfo parent)
334   throws NotAllMetaRegionsOnlineException, IOException {
335     Delete delete = new Delete(parent.getRegionName());
336     delete.deleteColumns(HConstants.CATALOG_FAMILY, HConstants.SPLITA_QUALIFIER);
337     delete.deleteColumns(HConstants.CATALOG_FAMILY, HConstants.SPLITB_QUALIFIER);
338     deleteMetaTable(catalogTracker, delete);
339     LOG.info("Deleted daughters references, qualifier=" + Bytes.toStringBinary(HConstants.SPLITA_QUALIFIER) +
340       " and qualifier=" + Bytes.toStringBinary(HConstants.SPLITB_QUALIFIER) +
341       ", from parent " + parent.getRegionNameAsString());
342   }

344   private static Put addRegionInfo(final Put p, final HRegionInfo hri)
345   throws IOException {
346     p.add(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER,
347         hri.toByteArray());
348     return p;
349   }
350 
351   private static Put addLocation(final Put p, final ServerName sn) {
352     p.add(HConstants.CATALOG_FAMILY, HConstants.SERVER_QUALIFIER,
353       Bytes.toBytes(sn.getHostAndPort()));
354     p.add(HConstants.CATALOG_FAMILY, HConstants.STARTCODE_QUALIFIER,
355       Bytes.toBytes(sn.getStartcode()));
356     return p;
357   }
