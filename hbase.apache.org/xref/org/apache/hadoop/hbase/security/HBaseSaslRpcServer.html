76    public static void init(Configuration conf) {
77      QualityOfProtection saslQOP = QualityOfProtection.AUTHENTICATION;
78      String rpcProtection = conf.get("hbase.rpc.protection",
79          QualityOfProtection.AUTHENTICATION.name().toLowerCase());
80      if (QualityOfProtection.INTEGRITY.name().toLowerCase()
81          .equals(rpcProtection)) {
82        saslQOP = QualityOfProtection.INTEGRITY;
83      } else if (QualityOfProtection.PRIVACY.name().toLowerCase().equals(
84          rpcProtection)) {
85        saslQOP = QualityOfProtection.PRIVACY;
86      }
87  
88      SASL_PROPS.put(Sasl.QOP, saslQOP.getSaslQop());
89      SASL_PROPS.put(Sasl.SERVER_AUTH, "true");
90    }

100   public static <T extends TokenIdentifier> T getIdentifier(String id,
101       SecretManager<T> secretManager) throws InvalidToken {
102     byte[] tokenId = decodeIdentifier(id);
103     T tokenIdentifier = secretManager.createIdentifier();
104     try {
105       tokenIdentifier.readFields(new DataInputStream(new ByteArrayInputStream(
106           tokenId)));
107     } catch (IOException e) {
108       throw (InvalidToken) new InvalidToken(
109           "Can't de-serialize tokenIdentifier").initCause(e);
110     }
111     return tokenIdentifier;
112   }

133   /** Authentication method */
134   public static enum AuthMethod {
135     SIMPLE((byte) 80, "", AuthenticationMethod.SIMPLE),
136     KERBEROS((byte) 81, "GSSAPI", AuthenticationMethod.KERBEROS),
137     DIGEST((byte) 82, "DIGEST-MD5", AuthenticationMethod.TOKEN);
138 
139     /** The code for this method. */
140     public final byte code;
141     public final String mechanismName;
142     public final AuthenticationMethod authenticationMethod;
143 
144     private AuthMethod(byte code, String mechanismName, 
145                        AuthenticationMethod authMethod) {
146       this.code = code;
147       this.mechanismName = mechanismName;
148       this.authenticationMethod = authMethod;
149     }
150 
151     private static final int FIRST_CODE = values()[0].code;
152 
153     /** Return the object represented by the code. */
154     private static AuthMethod valueOf(byte code) {
155       final int i = (code & 0xff) - FIRST_CODE;
156       return i < 0 || i >= values().length ? null : values()[i];
157     }
158 
159     /** Return the SASL mechanism name */
160     public String getMechanismName() {
161       return mechanismName;
162     }
163 
164     /** Read from in */
165     public static AuthMethod read(DataInput in) throws IOException {
166       return valueOf(in.readByte());
167     }
168 
169     /** Write to out */
170     public void write(DataOutput out) throws IOException {
171       out.write(code);
172     }

175   /** CallbackHandler for SASL DIGEST-MD5 mechanism */
176   public static class SaslDigestCallbackHandler implements CallbackHandler {
177     private SecretManager<TokenIdentifier> secretManager;
178     private HBaseServer.Connection connection;
179 
180     public SaslDigestCallbackHandler(
181         SecretManager<TokenIdentifier> secretManager,
182         HBaseServer.Connection connection) {
183       this.secretManager = secretManager;
184       this.connection = connection;
185     }
186 
187     private char[] getPassword(TokenIdentifier tokenid) throws InvalidToken {
188       return encodePassword(secretManager.retrievePassword(tokenid));
189     }
190 
191     /** {@inheritDoc} */
192     @Override
193     public void handle(Callback[] callbacks) throws InvalidToken,
194         UnsupportedCallbackException {
195       NameCallback nc = null;
196       PasswordCallback pc = null;
197       AuthorizeCallback ac = null;
198       for (Callback callback : callbacks) {
199         if (callback instanceof AuthorizeCallback) {
200           ac = (AuthorizeCallback) callback;
201         } else if (callback instanceof NameCallback) {
202           nc = (NameCallback) callback;
203         } else if (callback instanceof PasswordCallback) {
204           pc = (PasswordCallback) callback;
205         } else if (callback instanceof RealmCallback) {
206           continue; // realm is ignored
207         } else {
208           throw new UnsupportedCallbackException(callback,
209               "Unrecognized SASL DIGEST-MD5 Callback");
210         }
211       }
212       if (pc != null) {
213         TokenIdentifier tokenIdentifier = getIdentifier(nc.getDefaultName(), secretManager);
214         char[] password = getPassword(tokenIdentifier);
215         UserGroupInformation user = null;
216         user = tokenIdentifier.getUser(); // may throw exception
217         connection.attemptingUser = user;
218         if (LOG.isDebugEnabled()) {
219           LOG.debug("SASL server DIGEST-MD5 callback: setting password "
220               + "for client: " + tokenIdentifier.getUser());
221         }
222         pc.setPassword(password);
223       }
224       if (ac != null) {
225         String authid = ac.getAuthenticationID();
226         String authzid = ac.getAuthorizationID();
227         if (authid.equals(authzid)) {
228           ac.setAuthorized(true);
229         } else {
230           ac.setAuthorized(false);
231         }
232         if (ac.isAuthorized()) {
233           if (LOG.isDebugEnabled()) {
234             String username =
235               getIdentifier(authzid, secretManager).getUser().getUserName();
236             LOG.debug("SASL server DIGEST-MD5 callback: setting "
237                 + "canonicalized client ID: " + username);
238           }
239           ac.setAuthorizedID(authzid);
240         }
241       }
242     }
243   }
