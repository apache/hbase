150 /**
151  * HRegion stores data for a certain region of a table.  It stores all columns
152  * for each row. A given table consists of one or more HRegions.
153  *
154  * <p>We maintain multiple HStores for a single HRegion.
155  *
156  * <p>An Store is a set of rows with some column data; together,
157  * they make up all the data for the rows.
158  *
159  * <p>Each HRegion has a 'startKey' and 'endKey'.
160  * <p>The first is inclusive, the second is exclusive (except for
161  * the final region)  The endKey of region 0 is the same as
162  * startKey for region 1 (if it exists).  The startKey for the
163  * first region is null. The endKey for the final region is null.
164  *
165  * <p>Locking at the HRegion level serves only one purpose: preventing the
166  * region from being closed (and consequently split) while other operations
167  * are ongoing. Each row level operation obtains both a row lock and a region
168  * read lock for the duration of the operation. While a scanner is being
169  * constructed, getScanner holds a read lock. If the scanner is successfully
170  * constructed, it holds a read lock until it is closed. A close takes out a
171  * write lock and consequently will block for ongoing operations and will block
172  * new operations from starting while the close is in progress.
173  *
174  * <p>An HRegion is defined by its table and its key extent.
175  *
176  * <p>It consists of at least one Store.  The number of Stores should be
177  * configurable, so that data which is accessed together is stored in the same
178  * Store.  Right now, we approximate that by building a single Store for
179  * each column family.  (This config info will be communicated via the
180  * tabledesc.)
181  *
182  * <p>The HTableDescriptor contains metainfo about the HRegion's table.
183  * regionName is a unique identifier for this HRegion. (startKey, endKey]
184  * defines the keyspace for this HRegion.
185  */

294   /**
295    * @return The smallest mvcc readPoint across all the scanners in this
296    * region. Writes older than this readPoint, are included  in every
297    * read operation.
298    */
299   public long getSmallestReadPoint() {
300     long minimumReadPoint;
301     // We need to ensure that while we are calculating the smallestReadPoint
302     // no new RegionScanners can grab a readPoint that we are unaware of.
303     // We achieve this by synchronizing on the scannerReadPoints object.
304     synchronized(scannerReadPoints) {
305       minimumReadPoint = mvcc.memstoreReadPoint();
306 
307       for (Long readPoint: this.scannerReadPoints.values()) {
308         if (readPoint < minimumReadPoint) {
309           minimumReadPoint = readPoint;
310         }
311       }
312     }
313     return minimumReadPoint;
314   }

553   public long initialize(final CancelableProgressable reporter)
554   throws IOException {
555     MonitoredTask status = TaskMonitor.get().createStatus("Initializing region " + this);
556     long nextSeqId = -1;
557     try {
558       nextSeqId = initializeRegionInternals(reporter, status);
559       return nextSeqId;
560     } finally {
561       // nextSeqid will be -1 if the initialization fails.
562       // At least it will be 0 otherwise.
563       if (nextSeqId == -1) {
564         status
565             .abort("Exception during region " + this.getRegionNameAsString() + " initialization.");
566       }
567     }
568   }

570   private long initializeRegionInternals(final CancelableProgressable reporter, MonitoredTask status)
571       throws IOException, UnsupportedEncodingException {
572     if (coprocessorHost != null) {
573       status.setStatus("Running coprocessor pre-open hook");
574       coprocessorHost.preOpen();
575     }
576 
577     // Write HRI to a file in case we need to recover .META.
578     status.setStatus("Writing region info on filesystem");
579     checkRegioninfoOnFilesystem();
580 
581     // Remove temporary data left over from old regions
582     status.setStatus("Cleaning up temporary data from old regions");
583     cleanupTmpDir();
584 
585     // Load in all the HStores.
586     //
587     // Context: During replay we want to ensure that we do not lose any data. So, we
588     // have to be conservative in how we replay logs. For each store, we calculate
589     // the maxSeqId up to which the store was flushed. And, skip the edits which
590     // is equal to or lower than maxSeqId for each store.
591     Map<byte[], Long> maxSeqIdInStores = new TreeMap<byte[], Long>(
592         Bytes.BYTES_COMPARATOR);
593     long maxSeqId = -1;
594     // initialized to -1 so that we pick up MemstoreTS from column families
595     long maxMemstoreTS = -1;
596 
597     if (this.htableDescriptor != null &&
598         !htableDescriptor.getFamilies().isEmpty()) {
599       // initialize the thread pool for opening stores in parallel.
600       ThreadPoolExecutor storeOpenerThreadPool =
601         getStoreOpenAndCloseThreadPool(
602           "StoreOpenerThread-" + this.regionInfo.getRegionNameAsString());
603       CompletionService<HStore> completionService =
604         new ExecutorCompletionService<HStore>(storeOpenerThreadPool);
605 
606       // initialize each store in parallel
607       for (final HColumnDescriptor family : htableDescriptor.getFamilies()) {
608         status.setStatus("Instantiating store for column family " + family);
609         completionService.submit(new Callable<HStore>() {
610           public HStore call() throws IOException {
611             return instantiateHStore(tableDir, family);
612           }
613         });
614       }
615       try {
616         for (int i = 0; i < htableDescriptor.getFamilies().size(); i++) {
617           Future<HStore> future = completionService.take();
618           HStore store = future.get();
619 
620           this.stores.put(store.getColumnFamilyName().getBytes(), store);
621           // Do not include bulk loaded files when determining seqIdForReplay
622           long storeSeqIdForReplay = store.getMaxSequenceId(false);
623           maxSeqIdInStores.put(store.getColumnFamilyName().getBytes(),
624               storeSeqIdForReplay);
625           // Include bulk loaded files when determining seqIdForAssignment
626           long storeSeqIdForAssignment = store.getMaxSequenceId(true);
627           if (maxSeqId == -1 || storeSeqIdForAssignment > maxSeqId) {
628             maxSeqId = storeSeqIdForAssignment;
629           }
630           long maxStoreMemstoreTS = store.getMaxMemstoreTS();
631           if (maxStoreMemstoreTS > maxMemstoreTS) {
632             maxMemstoreTS = maxStoreMemstoreTS;
633           }
634         }
635       } catch (InterruptedException e) {
636         throw new IOException(e);
637       } catch (ExecutionException e) {
638         throw new IOException(e.getCause());
639       } finally {
640         storeOpenerThreadPool.shutdownNow();
641       }
642     }
643     mvcc.initialize(maxMemstoreTS + 1);
644     // Recover any edits if available.
645     maxSeqId = Math.max(maxSeqId, replayRecoveredEditsIfAny(
646         this.regiondir, maxSeqIdInStores, reporter, status));
647 
648     status.setStatus("Cleaning up detritus from prior splits");
649     // Get rid of any splits or merges that were lost in-progress.  Clean out
650     // these directories here on open.  We may be opening a region that was
651     // being split but we crashed in the middle of it all.
652     SplitTransaction.cleanupAnySplitDetritus(this);
653     FSUtils.deleteDirectory(this.fs, new Path(regiondir, MERGEDIR));
654     if (this.htableDescriptor != null) {
655       this.writestate.setReadOnly(this.htableDescriptor.isReadOnly());
656     }
657 
658     this.writestate.flushRequested = false;
659     this.writestate.compacting = 0;
660 
661     // Initialize split policy
662     this.splitPolicy = RegionSplitPolicy.create(this, conf);
663 
664     this.lastFlushTime = EnvironmentEdgeManager.currentTimeMillis();
665     // Use maximum of log sequenceid or that which was found in stores
666     // (particularly if no recovered edits, seqid will be -1).
667     long nextSeqid = maxSeqId + 1;
668     LOG.info("Onlined " + this.toString() + "; next sequenceid=" + nextSeqid);
669 
670     // A region can be reopened if failed a split; reset flags
671     this.closing.set(false);
672     this.closed.set(false);
673 
674     if (coprocessorHost != null) {
675       status.setStatus("Running coprocessor post-open hooks");
676       coprocessorHost.postOpen();
677     }
678 
679     status.markComplete("Region opened successfully");
680     return nextSeqid;
681   }

784   /*
785    * Write out an info file under the region directory.  Useful recovering
786    * mangled regions.
787    * @throws IOException
788    */
789   private void checkRegioninfoOnFilesystem() throws IOException {
790     Path regioninfoPath = new Path(this.regiondir, REGIONINFO_FILE);
791     // Compose the content of the file so we can compare to length in filesystem.  If not same,
792     // rewrite it (it may have been written in the old format using Writables instead of pb).  The
793     // pb version is much shorter -- we write now w/o the toString version -- so checking length
794     // only should be sufficient.  I don't want to read the file every time to check if it pb
795     // serialized.
796     byte [] content = getDotRegionInfoFileContent(this.getRegionInfo());
797     boolean exists = this.fs.exists(regioninfoPath);
798     FileStatus status = exists? this.fs.getFileStatus(regioninfoPath): null;
799     if (status != null && status.getLen() == content.length) {
800       // Then assume the content good and move on.
801       return;
802     }
803     // Create in tmpdir and then move into place in case we crash after
804     // create but before close.  If we don't successfully close the file,
805     // subsequent region reopens will fail the below because create is
806     // registered in NN.
807 
808     // First check to get the permissions
809     FsPermission perms = FSUtils.getFilePermissions(fs, conf, HConstants.DATA_FILE_UMASK_KEY);
810 
811     // And then create the file
812     Path tmpPath = new Path(getTmpDir(), REGIONINFO_FILE);
813 
814     // If datanode crashes or if the RS goes down just before the close is called while trying to
815     // close the created regioninfo file in the .tmp directory then on next
816     // creation we will be getting AlreadyCreatedException.
817     // Hence delete and create the file if exists.
818     if (FSUtils.isExists(fs, tmpPath)) {
819       FSUtils.delete(fs, tmpPath, true);
820     }
821 
822     FSDataOutputStream out = FSUtils.create(fs, tmpPath, perms);
823     try {
824       // We used to write out this file as serialized Writable followed by '\n\n' and then the
825       // toString of the HRegionInfo but now we just write out the pb serialized bytes so we can
826       // for sure tell whether the content has been pb'd or not just by looking at file length; the
827       // pb version will be shorter.
828       out.write(content);
829     } finally {
830       out.close();
831     }
832     if (exists) {
833       LOG.info("Rewriting .regioninfo file at " + regioninfoPath);
834       if (!fs.delete(regioninfoPath, false)) {
835         throw new IOException("Unable to remove existing " + regioninfoPath);
836       }
837     }
838     if (!fs.rename(tmpPath, regioninfoPath)) {
839       throw new IOException("Unable to rename " + tmpPath + " to " + regioninfoPath);
840     }
841   }

948   /**
949    * Close down this HRegion.  Flush the cache unless abort parameter is true,
950    * Shut down each HStore, don't service any more calls.
951    *
952    * This method could take some time to execute, so don't call it from a
953    * time-sensitive thread.
954    *
955    * @param abort true if server is aborting (only during testing)
956    * @return Vector of all the storage files that the HRegion's component
957    * HStores make use of.  It's a list of HStoreFile objects.  Can be null if
958    * we are not to close at this time or we are already closed.
959    *
960    * @throws IOException e
961    */
962   public List<StoreFile> close(final boolean abort) throws IOException {
963     // Only allow one thread to close at a time. Serialize them so dual
964     // threads attempting to close will run up against each other.
965     MonitoredTask status = TaskMonitor.get().createStatus(
966         "Closing region " + this +
967         (abort ? " due to abort" : ""));
968 
969     status.setStatus("Waiting for close lock");
970     try {
971       synchronized (closeLock) {
972         return doClose(abort, status);
973       }
974     } finally {
975       status.cleanup();
976     }
977   }

979   private List<StoreFile> doClose(
980       final boolean abort, MonitoredTask status)
981   throws IOException {
982     if (isClosed()) {
983       LOG.warn("Region " + this + " already closed");
984       return null;
985     }
986 
987     if (coprocessorHost != null) {
988       status.setStatus("Running coprocessor pre-close hooks");
989       this.coprocessorHost.preClose(abort);
990     }
991 
992     status.setStatus("Disabling compacts and flushes for region");
993     boolean wasFlushing = false;
994     synchronized (writestate) {
995       // Disable compacting and flushing by background threads for this
996       // region.
997       writestate.writesEnabled = false;
998       wasFlushing = writestate.flushing;
999       LOG.debug("Closing " + this + ": disabling compactions & flushes");
1000       waitForFlushesAndCompactions();
1001     }
1002     // If we were not just flushing, is it worth doing a preflush...one
1003     // that will clear out of the bulk of the memstore before we put up
1004     // the close flag?
1005     if (!abort && !wasFlushing && worthPreFlushing()) {
1006       status.setStatus("Pre-flushing region before close");
1007       LOG.info("Running close preflush of " + this.getRegionNameAsString());
1008       internalFlushcache(status);
1009     }
1010 
1011     this.closing.set(true);
1012     status.setStatus("Disabling writes for close");
1013     // block waiting for the lock for closing
1014     lock.writeLock().lock();
1015     try {
1016       if (this.isClosed()) {
1017         status.abort("Already got closed by another process");
1018         // SplitTransaction handles the null
1019         return null;
1020       }
1021       LOG.debug("Updates disabled for region " + this);
1022       // Don't flush the cache if we are aborting
1023       if (!abort) {
1024         internalFlushcache(status);
1025       }
1026 
1027       List<StoreFile> result = new ArrayList<StoreFile>();
1028       if (!stores.isEmpty()) {
1029         // initialize the thread pool for closing stores in parallel.
1030         ThreadPoolExecutor storeCloserThreadPool =
1031           getStoreOpenAndCloseThreadPool("StoreCloserThread-"
1032             + this.regionInfo.getRegionNameAsString());
1033         CompletionService<ImmutableList<StoreFile>> completionService =
1034           new ExecutorCompletionService<ImmutableList<StoreFile>>(
1035             storeCloserThreadPool);
1036 
1037         // close each store in parallel
1038         for (final Store store : stores.values()) {
1039           completionService
1040               .submit(new Callable<ImmutableList<StoreFile>>() {
1041                 public ImmutableList<StoreFile> call() throws IOException {
1042                   return store.close();
1043                 }
1044               });
1045         }
1046         try {
1047           for (int i = 0; i < stores.size(); i++) {
1048             Future<ImmutableList<StoreFile>> future = completionService
1049                 .take();
1050             ImmutableList<StoreFile> storeFileList = future.get();
1051             result.addAll(storeFileList);
1052           }
1053         } catch (InterruptedException e) {
1054           throw new IOException(e);
1055         } catch (ExecutionException e) {
1056           throw new IOException(e.getCause());
1057         } finally {
1058           storeCloserThreadPool.shutdownNow();
1059         }
1060       }
1061       this.closed.set(true);
1062 
1063       if (coprocessorHost != null) {
1064         status.setStatus("Running coprocessor post-close hooks");
1065         this.coprocessorHost.postClose(abort);
1066       }
1067       if ( this.metricsRegion != null) {
1068         this.metricsRegion.close();
1069       }
1070       if ( this.metricsRegionWrapper != null) {
1071         Closeables.closeQuietly(this.metricsRegionWrapper);
1072       }
1073       status.markComplete("Closed");
1074       LOG.info("Closed " + this);
1075       return result;
1076     } finally {
1077       lock.writeLock().unlock();
1078     }
1079   }

1082    * Wait for all current flushes and compactions of the region to complete.
1083    * <p>
1084    * Exposed for TESTING.
1085    */
1086   public void waitForFlushesAndCompactions() {

1094           // essentially ignore and propagate the interrupt back up
1095           Thread.currentThread().interrupt();

1297   /*
1298    * Called by compaction thread and after region is opened to compact the
1299    * HStores if necessary.
1300    *
1301    * <p>This operation could block for a long time, so don't call it from a
1302    * time-sensitive thread.
1303    *
1304    * Note that no locking is necessary at this level because compaction only
1305    * conflicts with a region split, and that cannot happen because the region
1306    * server does them sequentially and not in parallel.
1307    *
1308    * @param cr Compaction details, obtained by requestCompaction()
1309    * @return whether the compaction completed
1310    * @throws IOException e
1311    */
1312   public boolean compact(CompactionRequest cr)
1313   throws IOException {
1314     if (cr == null) {
1315       return false;
1316     }
1317     if (this.closing.get() || this.closed.get()) {
1318       LOG.debug("Skipping compaction on " + this + " because closing/closed");
1319       return false;
1320     }
1321     Preconditions.checkArgument(cr.getHRegion().equals(this));
1322     MonitoredTask status = null;
1323     // block waiting for the lock for compaction
1324     lock.readLock().lock();
1325     try {
1326       status = TaskMonitor.get().createStatus(
1327         "Compacting " + cr.getStore() + " in " + this);
1328       if (this.closed.get()) {
1329         LOG.debug("Skipping compaction on " + this + " because closed");
1330         return false;
1331       }
1332       boolean decr = true;
1333       try {
1334         synchronized (writestate) {
1335           if (writestate.writesEnabled) {
1336             ++writestate.compacting;
1337           } else {
1338             String msg = "NOT compacting region " + this + ". Writes disabled.";
1339             LOG.info(msg);
1340             status.abort(msg);
1341             decr = false;
1342             return false;
1343           }
1344         }
1345         LOG.info("Starting compaction on " + cr.getStore() + " in region "
1346             + this + (cr.getCompactSelection().isOffPeakCompaction()?" as an off-peak compaction":""));
1347         doRegionCompactionPrep();
1348         try {
1349           status.setStatus("Compacting store " + cr.getStore());
1350           cr.getStore().compact(cr);
1351         } catch (InterruptedIOException iioe) {
1352           String msg = "compaction interrupted";
1353           LOG.info(msg, iioe);
1354           status.abort(msg);
1355           return false;
1356         }
1357       } finally {
1358         if (decr) {
1359           synchronized (writestate) {
1360             --writestate.compacting;
1361             if (writestate.compacting <= 0) {
1362               writestate.notifyAll();
1363             }
1364           }
1365         }
1366       }
1367       status.markComplete("Compaction complete");
1368       return true;
1369     } finally {
1370       try {
1371         if (status != null) status.cleanup();
1372       } finally {
1373         lock.readLock().unlock();
1374       }
1375     }
1376   }

1378   /**
1379    * Flush the cache.
1380    *
1381    * When this method is called the cache will be flushed unless:
1382    * <ol>
1383    *   <li>the cache is empty</li>
1384    *   <li>the region is closed.</li>
1385    *   <li>a flush is already in progress</li>
1386    *   <li>writes are disabled</li>
1387    * </ol>
1388    *
1389    * <p>This method may block for some time, so it should not be called from a
1390    * time-sensitive thread.
1391    *
1392    * @return true if cache was flushed
1393    *
1394    * @throws IOException general io exceptions
1395    * @throws DroppedSnapshotException Thrown when replay of hlog is required
1396    * because a Snapshot was not properly persisted.
1397    */
1398   public boolean flushcache() throws IOException {
1399     // fail-fast instead of waiting on the lock
1400     if (this.closing.get()) {
1401       LOG.debug("Skipping flush on " + this + " because closing");
1402       return false;
1403     }
1404     MonitoredTask status = TaskMonitor.get().createStatus("Flushing " + this);
1405     status.setStatus("Acquiring readlock on region");
1406     // block waiting for the lock for flushing cache
1407     lock.readLock().lock();
1408     try {
1409       if (this.closed.get()) {
1410         LOG.debug("Skipping flush on " + this + " because closed");
1411         status.abort("Skipped: closed");
1412         return false;
1413       }
1414       if (coprocessorHost != null) {
1415         status.setStatus("Running coprocessor pre-flush hooks");
1416         coprocessorHost.preFlush();
1417       }
1418       if (numPutsWithoutWAL.get() > 0) {
1419         numPutsWithoutWAL.set(0);
1420         dataInMemoryWithoutWAL.set(0);
1421       }
1422       synchronized (writestate) {
1423         if (!writestate.flushing && writestate.writesEnabled) {
1424           this.writestate.flushing = true;
1425         } else {
1426           if (LOG.isDebugEnabled()) {
1427             LOG.debug("NOT flushing memstore for region " + this
1428                 + ", flushing=" + writestate.flushing + ", writesEnabled="
1429                 + writestate.writesEnabled);
1430           }
1431           status.abort("Not flushing since "
1432               + (writestate.flushing ? "already flushing"
1433                   : "writes not enabled"));
1434           return false;
1435         }
1436       }
1437       try {
1438         boolean result = internalFlushcache(status);
1439 
1440         if (coprocessorHost != null) {
1441           status.setStatus("Running post-flush coprocessor hooks");
1442           coprocessorHost.postFlush();
1443         }
1444 
1445         status.markComplete("Flush successful");
1446         return result;
1447       } finally {
1448         synchronized (writestate) {
1449           writestate.flushing = false;
1450           this.writestate.flushRequested = false;
1451           writestate.notifyAll();
1452         }
1453       }
1454     } finally {
1455       lock.readLock().unlock();
1456       status.cleanup();
1457     }
1458   }

1460   /**
1461    * Flush the memstore.
1462    *
1463    * Flushing the memstore is a little tricky. We have a lot of updates in the
1464    * memstore, all of which have also been written to the log. We need to
1465    * write those updates in the memstore out to disk, while being able to
1466    * process reads/writes as much as possible during the flush operation. Also,
1467    * the log has to state clearly the point in time at which the memstore was
1468    * flushed. (That way, during recovery, we know when we can rely on the
1469    * on-disk flushed structures and when we have to recover the memstore from
1470    * the log.)
1471    *
1472    * <p>So, we have a three-step process:
1473    *
1474    * <ul><li>A. Flush the memstore to the on-disk stores, noting the current
1475    * sequence ID for the log.<li>
1476    *
1477    * <li>B. Write a FLUSHCACHE-COMPLETE message to the log, using the sequence
1478    * ID that was current at the time of memstore-flush.</li>
1479    *
1480    * <li>C. Get rid of the memstore structures that are now redundant, as
1481    * they've been flushed to the on-disk HStores.</li>
1482    * </ul>
1483    * <p>This method is protected, but can be accessed via several public
1484    * routes.
1485    *
1486    * <p> This method may block for some time.
1487    * @param status
1488    *
1489    * @return true if the region needs compacting
1490    *
1491    * @throws IOException general io exceptions
1492    * @throws DroppedSnapshotException Thrown when replay of hlog is required
1493    * because a Snapshot was not properly persisted.
1494    */
1495   protected boolean internalFlushcache(MonitoredTask status)
1496       throws IOException {
1497     return internalFlushcache(this.log, -1, status);
1498   }

1500   /**
1501    * @param wal Null if we're NOT to go via hlog/wal.
1502    * @param myseqid The seqid to use if <code>wal</code> is null writing out
1503    * flush file.
1504    * @param status
1505    * @return true if the region needs compacting
1506    * @throws IOException
1507    * @see #internalFlushcache(MonitoredTask)
1508    */
1509   protected boolean internalFlushcache(
1510       final HLog wal, final long myseqid, MonitoredTask status)
1511   throws IOException {
1512     final long startTime = EnvironmentEdgeManager.currentTimeMillis();
1513     // Clear flush flag.
1514     // Record latest flush time
1515     this.lastFlushTime = startTime;
1516     // If nothing to flush, return and avoid logging start/stop flush.
1517     if (this.memstoreSize.get() <= 0) {
1518       return false;
1519     }
1520     if (LOG.isDebugEnabled()) {
1521       LOG.debug("Started memstore flush for " + this +
1522         ", current region memstore size " +
1523         StringUtils.humanReadableInt(this.memstoreSize.get()) +
1524         ((wal != null)? "": "; wal is null, using passed sequenceid=" + myseqid));
1525     }
1526 
1527     // Stop updates while we snapshot the memstore of all stores. We only have
1528     // to do this for a moment.  Its quick.  The subsequent sequence id that
1529     // goes into the HLog after we've flushed all these snapshots also goes
1530     // into the info file that sits beside the flushed files.
1531     // We also set the memstore size to zero here before we allow updates
1532     // again so its value will represent the size of the updates received
1533     // during the flush
1534     long sequenceId = -1L;
1535     MultiVersionConsistencyControl.WriteEntry w = null;
1536 
1537     // We have to take a write lock during snapshot, or else a write could
1538     // end up in both snapshot and memstore (makes it difficult to do atomic
1539     // rows then)
1540     status.setStatus("Obtaining lock to block concurrent updates");
1541     // block waiting for the lock for internal flush
1542     this.updatesLock.writeLock().lock();
1543     long flushsize = this.memstoreSize.get();
1544     status.setStatus("Preparing to flush by snapshotting stores");
1545     List<StoreFlusher> storeFlushers = new ArrayList<StoreFlusher>(stores.size());
1546     long completeSeqId = -1L;
1547     try {
1548       // Record the mvcc for all transactions in progress.
1549       w = mvcc.beginMemstoreInsert();
1550       mvcc.advanceMemstore(w);
1551 
1552       sequenceId = (wal == null)? myseqid:
1553         wal.startCacheFlush(this.regionInfo.getEncodedNameAsBytes());
1554       completeSeqId = this.getCompleteCacheFlushSequenceId(sequenceId);
1555       for (Store s : stores.values()) {
1556         storeFlushers.add(s.getStoreFlusher(completeSeqId));
1557       }
1558 
1559       // prepare flush (take a snapshot)
1560       for (StoreFlusher flusher : storeFlushers) {
1561         flusher.prepare();
1562       }
1563     } finally {
1564       this.updatesLock.writeLock().unlock();
1565     }
1566     String s = "Finished snapshotting " + this +
1567       ", commencing wait for mvcc, flushsize=" + flushsize;
1568     status.setStatus(s);
1569     LOG.debug(s);
1570 
1571     // wait for all in-progress transactions to commit to HLog before
1572     // we can start the flush. This prevents
1573     // uncommitted transactions from being written into HFiles.
1574     // We have to block before we start the flush, otherwise keys that
1575     // were removed via a rollbackMemstore could be written to Hfiles.
1576     mvcc.waitForRead(w);
1577 
1578     status.setStatus("Flushing stores");
1579     LOG.debug("Finished snapshotting, commencing flushing stores");
1580 
1581     // Any failure from here on out will be catastrophic requiring server
1582     // restart so hlog content can be replayed and put back into the memstore.
1583     // Otherwise, the snapshot content while backed up in the hlog, it will not
1584     // be part of the current running servers state.
1585     boolean compactionRequested = false;
1586     try {
1587       // A.  Flush memstore to all the HStores.
1588       // Keep running vector of all store files that includes both old and the
1589       // just-made new flush store file. The new flushed file is still in the
1590       // tmp directory.
1591 
1592       for (StoreFlusher flusher : storeFlushers) {
1593         flusher.flushCache(status);
1594       }
1595 
1596       // Switch snapshot (in memstore) -> new hfile (thus causing
1597       // all the store scanners to reset/reseek).
1598       for (StoreFlusher flusher : storeFlushers) {
1599         boolean needsCompaction = flusher.commit(status);
1600         if (needsCompaction) {
1601           compactionRequested = true;
1602         }
1603       }
1604       storeFlushers.clear();
1605 
1606       // Set down the memstore size by amount of flush.
1607       this.addAndGetGlobalMemstoreSize(-flushsize);
1608     } catch (Throwable t) {
1609       // An exception here means that the snapshot was not persisted.
1610       // The hlog needs to be replayed so its content is restored to memstore.
1611       // Currently, only a server restart will do this.
1612       // We used to only catch IOEs but its possible that we'd get other
1613       // exceptions -- e.g. HBASE-659 was about an NPE -- so now we catch
1614       // all and sundry.
1615       if (wal != null) {
1616         wal.abortCacheFlush(this.regionInfo.getEncodedNameAsBytes());
1617       }
1618       DroppedSnapshotException dse = new DroppedSnapshotException("region: " +
1619           Bytes.toStringBinary(getRegionName()));
1620       dse.initCause(t);
1621       status.abort("Flush failed: " + StringUtils.stringifyException(t));
1622       throw dse;
1623     }
1624 
1625     // If we get to here, the HStores have been written. If we get an
1626     // error in completeCacheFlush it will release the lock it is holding
1627 
1628     // B.  Write a FLUSHCACHE-COMPLETE message to the log.
1629     //     This tells future readers that the HStores were emitted correctly,
1630     //     and that all updates to the log for this regionName that have lower
1631     //     log-sequence-ids can be safely ignored.
1632     if (wal != null) {
1633       wal.completeCacheFlush(this.regionInfo.getEncodedNameAsBytes(),
1634         regionInfo.getTableName(), completeSeqId,
1635         this.getRegionInfo().isMetaRegion());
1636     }
1637 
1638     // Update the last flushed sequence id for region
1639     if (this.rsServices != null) {
1640       completeSequenceId = completeSeqId;
1641     }
1642 
1643     // C. Finally notify anyone waiting on memstore to clear:
1644     // e.g. checkResources().
1645     synchronized (this) {
1646       notifyAll(); // FindBugs NN_NAKED_NOTIFY
1647     }
1648 
1649     long time = EnvironmentEdgeManager.currentTimeMillis() - startTime;
1650     long memstoresize = this.memstoreSize.get();
1651     String msg = "Finished memstore flush of ~" +
1652       StringUtils.humanReadableInt(flushsize) + "/" + flushsize +
1653       ", currentsize=" +
1654       StringUtils.humanReadableInt(memstoresize) + "/" + memstoresize +
1655       " for region " + this + " in " + time + "ms, sequenceid=" + sequenceId +
1656       ", compaction requested=" + compactionRequested +
1657       ((wal == null)? "; wal=null": "");
1658     LOG.info(msg);
1659     status.setStatus(msg);
1660     this.recentFlushes.add(new Pair<Long,Long>(time/1000, flushsize));
1661 
1662     return compactionRequested;
1663   }

1858

1931    * @param put
1932    * @param lockid
1933    * @param writeToWAL
1934    * @throws IOException
1935    */
1936   public void put(Put put, Integer lockid, boolean writeToWAL)
1937   throws IOException {
1938     checkReadOnly();
1939 
1940     // Do a rough check that we have resources to accept a write.  The check is
1941     // 'rough' in that between the resource check and the call to obtain a
1942     // read lock, resources may run out.  For now, the thought is that this
1943     // will be extremely rare; we'll deal with it when it happens.
1944     checkResources();
1945     startRegionOperation();
1946     this.writeRequestsCount.increment();
1947     try {
1948       // We obtain a per-row lock, so other clients will block while one client
1949       // performs an update. The read lock is released by the client calling
1950       // #commit or #abort or if the HRegionServer lease on the lock expires.
1951       // See HRegionServer#RegionListener for how the expire on HRegionServer
1952       // invokes a HRegion#abort.
1953       byte [] row = put.getRow();
1954       // If we did not pass an existing row lock, obtain a new one
1955       Integer lid = getLock(lockid, row, true);
1956 
1957       try {
1958         // All edits for the given row (across all column families) must happen atomically.
1959         doBatchMutate(put, lid);
1960       } finally {
1961         if(lockid == null) releaseRowLock(lid);
1962       }
1963     } finally {
1964       closeRegionOperation();
1965     }
1966   }

2007    * It supports only Put and Delete mutations and will ignore other types passed.
2008    * @param mutationsAndLocks
2009    *          the list of mutations paired with their requested lock IDs.
2010    * @return an array of OperationStatus which internally contains the
2011    *         OperationStatusCode and the exceptionMessage if any.
2012    * @throws IOException
2013    */
2014   public OperationStatus[] batchMutate(
2015       Pair<Mutation, Integer>[] mutationsAndLocks) throws IOException {
2016     BatchOperationInProgress<Pair<Mutation, Integer>> batchOp =
2017       new BatchOperationInProgress<Pair<Mutation,Integer>>(mutationsAndLocks);
2018 
2019     boolean initialized = false;
2020 
2021     while (!batchOp.isDone()) {
2022       checkReadOnly();
2023       checkResources();
2024 
2025       long newSize;
2026       startRegionOperation();
2027 
2028       try {
2029         if (!initialized) {
2030           this.writeRequestsCount.increment();
2031           doPreMutationHook(batchOp);
2032           initialized = true;
2033         }
2034         long addedSize = doMiniBatchMutation(batchOp);
2035         newSize = this.addAndGetGlobalMemstoreSize(addedSize);
2036       } finally {
2037         closeRegionOperation();
2038       }
2039       if (isFlushSize(newSize)) {
2040         requestFlush();
2041       }
2042     }
2043     return batchOp.retCodeDetails;

2046   private void doPreMutationHook(BatchOperationInProgress<Pair<Mutation, Integer>> batchOp)
2047       throws IOException {
2048     /* Run coprocessor pre hook outside of locks to avoid deadlock */
2049     WALEdit walEdit = new WALEdit();
2050     if (coprocessorHost != null) {
2051       for (int i = 0 ; i < batchOp.operations.length; i++) {
2052         Pair<Mutation, Integer> nextPair = batchOp.operations[i];
2053         Mutation m = nextPair.getFirst();
2054         if (m instanceof Put) {
2055           if (coprocessorHost.prePut((Put) m, walEdit, m.getWriteToWAL())) {
2056             // pre hook says skip this Put
2057             // mark as success and skip in doMiniBatchMutation
2058             batchOp.retCodeDetails[i] = OperationStatus.SUCCESS;
2059           }
2060         } else if (m instanceof Delete) {
2061           if (coprocessorHost.preDelete((Delete) m, walEdit, m.getWriteToWAL())) {
2062             // pre hook says skip this Delete
2063             // mark as success and skip in doMiniBatchMutation
2064             batchOp.retCodeDetails[i] = OperationStatus.SUCCESS;
2065           }
2066         } else {
2067           // In case of passing Append mutations along with the Puts and Deletes in batchMutate
2068           // mark the operation return code as failure so that it will not be considered in
2069           // the doMiniBatchMutation
2070           batchOp.retCodeDetails[i] = new OperationStatus(OperationStatusCode.FAILURE,
2071               "Put/Delete mutations only supported in batchMutate() now");
2072         }
2073         if (!walEdit.isEmpty()) {
2074           batchOp.walEditsFromCoprocessors[i] = walEdit;
2075           walEdit = new WALEdit();
2076         }
2077       }
2078     }
2079   }

2083   private long doMiniBatchMutation(
2084     BatchOperationInProgress<Pair<Mutation, Integer>> batchOp) throws IOException {
2085 
2086     // variable to note if all Put items are for the same CF -- metrics related
2087     boolean putsCfSetConsistent = true;
2088     //The set of columnFamilies first seen for Put.
2089     Set<byte[]> putsCfSet = null;
2090     // variable to note if all Delete items are for the same CF -- metrics related
2091     boolean deletesCfSetConsistent = true;
2092     //The set of columnFamilies first seen for Delete.
2093     Set<byte[]> deletesCfSet = null;
2094 
2095     WALEdit walEdit = new WALEdit();
2096 
2097     long startTimeMs = EnvironmentEdgeManager.currentTimeMillis();
2098 
2099 
2100     MultiVersionConsistencyControl.WriteEntry w = null;
2101     long txid = 0;
2102     boolean walSyncSuccessful = false;
2103     boolean locked = false;
2104 
2105     /** Keep track of the locks we hold so we can release them in finally clause */
2106     List<Integer> acquiredLocks = Lists.newArrayListWithCapacity(batchOp.operations.length);
2107     // reference family maps directly so coprocessors can mutate them if desired
2108     Map<byte[],List<KeyValue>>[] familyMaps = new Map[batchOp.operations.length];
2109     // We try to set up a batch in the range [firstIndex,lastIndexExclusive)
2110     int firstIndex = batchOp.nextIndexToProcess;
2111     int lastIndexExclusive = firstIndex;
2112     boolean success = false;
2113     int noOfPuts = 0, noOfDeletes = 0;
2114     try {
2115       // ------------------------------------
2116       // STEP 1. Try to acquire as many locks as we can, and ensure
2117       // we acquire at least one.
2118       // ----------------------------------
2119       int numReadyToWrite = 0;
2120       long now = EnvironmentEdgeManager.currentTimeMillis();
2121       while (lastIndexExclusive < batchOp.operations.length) {
2122         Pair<Mutation, Integer> nextPair = batchOp.operations[lastIndexExclusive];
2123         Mutation mutation = nextPair.getFirst();
2124         boolean isPutMutation = mutation instanceof Put;
2125         Integer providedLockId = nextPair.getSecond();
2126 
2127         Map<byte[], List<KeyValue>> familyMap = mutation.getFamilyMap();
2128         // store the family map reference to allow for mutations
2129         familyMaps[lastIndexExclusive] = familyMap;
2130 
2131         // skip anything that "ran" already
2132         if (batchOp.retCodeDetails[lastIndexExclusive].getOperationStatusCode()
2133             != OperationStatusCode.NOT_RUN) {
2134           lastIndexExclusive++;
2135           continue;
2136         }
2137 
2138         try {
2139           if (isPutMutation) {
2140             // Check the families in the put. If bad, skip this one.
2141             checkFamilies(familyMap.keySet());
2142             checkTimestamps(mutation.getFamilyMap(), now);
2143           } else {
2144             prepareDelete((Delete) mutation);
2145           }
2146         } catch (NoSuchColumnFamilyException nscf) {
2147           LOG.warn("No such column family in batch mutation", nscf);
2148           batchOp.retCodeDetails[lastIndexExclusive] = new OperationStatus(
2149               OperationStatusCode.BAD_FAMILY, nscf.getMessage());
2150           lastIndexExclusive++;
2151           continue;
2152         } catch (FailedSanityCheckException fsce) {
2153           LOG.warn("Batch Mutation did not pass sanity check", fsce);
2154           batchOp.retCodeDetails[lastIndexExclusive] = new OperationStatus(
2155               OperationStatusCode.SANITY_CHECK_FAILURE, fsce.getMessage());
2156           lastIndexExclusive++;
2157           continue;
2158         }
2159         // If we haven't got any rows in our batch, we should block to
2160         // get the next one.
2161         boolean shouldBlock = numReadyToWrite == 0;
2162         Integer acquiredLockId = null;
2163         try {
2164           acquiredLockId = getLock(providedLockId, mutation.getRow(),
2165               shouldBlock);
2166         } catch (IOException ioe) {
2167           LOG.warn("Failed getting lock in batch put, row="
2168                   + Bytes.toStringBinary(mutation.getRow()), ioe);
2169         }
2170         if (acquiredLockId == null) {
2171           // We failed to grab another lock
2172           assert !shouldBlock : "Should never fail to get lock when blocking";
2173           break; // stop acquiring more rows for this batch
2174         }
2175         if (providedLockId == null) {
2176           acquiredLocks.add(acquiredLockId);
2177         }
2178         lastIndexExclusive++;
2179         numReadyToWrite++;
2180 
2181         if (isPutMutation) {
2182           // If Column Families stay consistent through out all of the
2183           // individual puts then metrics can be reported as a mutliput across
2184           // column families in the first put.
2185           if (putsCfSet == null) {
2186             putsCfSet = mutation.getFamilyMap().keySet();
2187           } else {
2188             putsCfSetConsistent = putsCfSetConsistent
2189                 && mutation.getFamilyMap().keySet().equals(putsCfSet);
2190           }
2191         } else {
2192           if (deletesCfSet == null) {
2193             deletesCfSet = mutation.getFamilyMap().keySet();
2194           } else {
2195             deletesCfSetConsistent = deletesCfSetConsistent
2196                 && mutation.getFamilyMap().keySet().equals(deletesCfSet);
2197           }
2198         }
2199       }
2200 
2201       // we should record the timestamp only after we have acquired the rowLock,
2202       // otherwise, newer puts/deletes are not guaranteed to have a newer timestamp
2203       now = EnvironmentEdgeManager.currentTimeMillis();
2204       byte[] byteNow = Bytes.toBytes(now);
2205 
2206       // Nothing to put/delete -- an exception in the above such as NoSuchColumnFamily?
2207       if (numReadyToWrite <= 0) return 0L;
2208 
2209       // We've now grabbed as many mutations off the list as we can
2210 
2211       // ------------------------------------
2212       // STEP 2. Update any LATEST_TIMESTAMP timestamps
2213       // ----------------------------------
2214       for (int i = firstIndex; i < lastIndexExclusive; i++) {
2215         // skip invalid
2216         if (batchOp.retCodeDetails[i].getOperationStatusCode()
2217             != OperationStatusCode.NOT_RUN) continue;
2218 
2219         Mutation mutation = batchOp.operations[i].getFirst();
2220         if (mutation instanceof Put) {
2221           updateKVTimestamps(familyMaps[i].values(), byteNow);
2222           noOfPuts++;
2223         } else {
2224           prepareDeleteTimestamps(familyMaps[i], byteNow);
2225           noOfDeletes++;
2226         }
2227       }
2228 
2229       lock(this.updatesLock.readLock(), numReadyToWrite);
2230       locked = true;
2231 
2232       //
2233       // ------------------------------------
2234       // Acquire the latest mvcc number
2235       // ----------------------------------
2236       w = mvcc.beginMemstoreInsert();
2237 
2238       // ------------------------------------
2239       // STEP 3. Write back to memstore
2240       // Write to memstore. It is ok to write to memstore
2241       // first without updating the HLog because we do not roll
2242       // forward the memstore MVCC. The MVCC will be moved up when
2243       // the complete operation is done. These changes are not yet
2244       // visible to scanners till we update the MVCC. The MVCC is
2245       // moved only when the sync is complete.
2246       // ----------------------------------
2247       long addedSize = 0;
2248       for (int i = firstIndex; i < lastIndexExclusive; i++) {
2249         if (batchOp.retCodeDetails[i].getOperationStatusCode()
2250             != OperationStatusCode.NOT_RUN) {
2251           continue;
2252         }
2253         addedSize += applyFamilyMapToMemstore(familyMaps[i], w);
2254       }
2255 
2256       // ------------------------------------
2257       // STEP 4. Build WAL edit
2258       // ----------------------------------
2259       for (int i = firstIndex; i < lastIndexExclusive; i++) {
2260         // Skip puts that were determined to be invalid during preprocessing
2261         if (batchOp.retCodeDetails[i].getOperationStatusCode()
2262             != OperationStatusCode.NOT_RUN) {
2263           continue;
2264         }
2265         batchOp.retCodeDetails[i] = OperationStatus.SUCCESS;
2266 
2267         Mutation m = batchOp.operations[i].getFirst();
2268         if (!m.getWriteToWAL()) {
2269           if (m instanceof Put) {
2270             recordPutWithoutWal(m.getFamilyMap());
2271           }
2272           continue;
2273         }
2274         // Add WAL edits by CP
2275         WALEdit fromCP = batchOp.walEditsFromCoprocessors[i];
2276         if (fromCP != null) {
2277           for (KeyValue kv : fromCP.getKeyValues()) {
2278             walEdit.add(kv);
2279           }
2280         }
2281         addFamilyMapToWALEdit(familyMaps[i], walEdit);
2282 
2283       }
2284 
2285       // -------------------------
2286       // STEP 5. Append the edit to WAL. Do not sync wal.
2287       // -------------------------
2288       Mutation first = batchOp.operations[firstIndex].getFirst();
2289       txid = this.log.appendNoSync(regionInfo, this.htableDescriptor.getName(),
2290                walEdit, first.getClusterId(), now, this.htableDescriptor);
2291 
2292       // -------------------------------
2293       // STEP 6. Release row locks, etc.
2294       // -------------------------------
2295       if (locked) {
2296         this.updatesLock.readLock().unlock();
2297         locked = false;
2298       }
2299       if (acquiredLocks != null) {
2300         for (Integer toRelease : acquiredLocks) {
2301           releaseRowLock(toRelease);
2302         }
2303         acquiredLocks = null;
2304       }
2305       // -------------------------
2306       // STEP 7. Sync wal.
2307       // -------------------------
2308       if (walEdit.size() > 0) {
2309         syncOrDefer(txid);
2310       }
2311       walSyncSuccessful = true;
2312       // ------------------------------------------------------------------
2313       // STEP 8. Advance mvcc. This will make this put visible to scanners and getters.
2314       // ------------------------------------------------------------------
2315       if (w != null) {
2316         mvcc.completeMemstoreInsert(w);
2317         w = null;
2318       }
2319 
2320       // ------------------------------------
2321       // STEP 9. Run coprocessor post hooks. This should be done after the wal is
2322       // synced so that the coprocessor contract is adhered to.
2323       // ------------------------------------
2324       if (coprocessorHost != null) {
2325         for (int i = firstIndex; i < lastIndexExclusive; i++) {
2326           // only for successful puts
2327           if (batchOp.retCodeDetails[i].getOperationStatusCode()
2328               != OperationStatusCode.SUCCESS) {
2329             continue;
2330           }
2331           Mutation m = batchOp.operations[i].getFirst();
2332           if (m instanceof Put) {
2333             coprocessorHost.postPut((Put) m, walEdit, m.getWriteToWAL());
2334           } else {
2335             coprocessorHost.postDelete((Delete) m, walEdit, m.getWriteToWAL());
2336           }
2337         }
2338       }
2339 
2340       success = true;
2341       return addedSize;
2342     } finally {
2343 
2344       // if the wal sync was unsuccessful, remove keys from memstore
2345       if (!walSyncSuccessful) {
2346         rollbackMemstore(batchOp, familyMaps, firstIndex, lastIndexExclusive);
2347       }
2348       if (w != null) mvcc.completeMemstoreInsert(w);
2349 
2350       if (locked) {
2351         this.updatesLock.readLock().unlock();
2352       }
2353 
2354       if (acquiredLocks != null) {
2355         for (Integer toRelease : acquiredLocks) {
2356           releaseRowLock(toRelease);
2357         }
2358       }
2359 
2360       // See if the column families were consistent through the whole thing.
2361       // if they were then keep them. If they were not then pass a null.
2362       // null will be treated as unknown.
2363       // Total time taken might be involving Puts and Deletes.
2364       // Split the time for puts and deletes based on the total number of Puts and Deletes.
2365 
2366       if (noOfPuts > 0) {
2367         // There were some Puts in the batch.
2368         double noOfMutations = noOfPuts + noOfDeletes;
2369         if (this.metricsRegion != null) {
2370           this.metricsRegion.updatePut();
2371         }
2372       }
2373       if (noOfDeletes > 0) {
2374         // There were some Deletes in the batch.
2375         if (this.metricsRegion != null) {
2376           this.metricsRegion.updateDelete();
2377         }
2378       }
2379       if (!success) {
2380         for (int i = firstIndex; i < lastIndexExclusive; i++) {
2381           if (batchOp.retCodeDetails[i].getOperationStatusCode() == OperationStatusCode.NOT_RUN) {
2382             batchOp.retCodeDetails[i] = OperationStatus.FAILURE;
2383           }
2384         }
2385       }
2386       batchOp.nextIndexToProcess = lastIndexExclusive;
2387     }
2388   }

2406   public boolean checkAndMutate(byte [] row, byte [] family, byte [] qualifier,
2407       CompareOp compareOp, ByteArrayComparable comparator, Mutation w,
2408       Integer lockId, boolean writeToWAL)
2409   throws IOException{
2410     checkReadOnly();
2411     //TODO, add check for value length or maybe even better move this to the

2413     checkResources();
2414     boolean isPut = w instanceof Put;
2415     if (!isPut && !(w instanceof Delete))
2416       throw new DoNotRetryIOException("Action must be Put or Delete");
2417     Row r = (Row)w;
2418     if (!Bytes.equals(row, r.getRow())) {
2419       throw new DoNotRetryIOException("Action's getRow must match the passed row");
2420     }
2421 
2422     startRegionOperation();
2423     try {
2424       RowLock lock = isPut ? ((Put)w).getRowLock() : ((Delete)w).getRowLock();
2425       Get get = new Get(row, lock);
2426       checkFamily(family);
2427       get.addColumn(family, qualifier);
2428 
2429       // Lock row
2430       Integer lid = getLock(lockId, get.getRow(), true);
2431       // wait for all previous transactions to complete (with lock held)
2432       mvcc.completeMemstoreInsert(mvcc.beginMemstoreInsert());
2433       List<KeyValue> result = null;
2434       try {
2435         result = get(get, false);
2436 
2437         boolean valueIsNull = comparator.getValue() == null ||
2438           comparator.getValue().length == 0;
2439         boolean matches = false;
2440         if (result.size() == 0 && valueIsNull) {
2441           matches = true;
2442         } else if (result.size() > 0 && result.get(0).getValue().length == 0 &&
2443             valueIsNull) {
2444           matches = true;
2445         } else if (result.size() == 1 && !valueIsNull) {
2446           KeyValue kv = result.get(0);
2447           int compareResult = comparator.compareTo(kv.getBuffer(),
2448               kv.getValueOffset(), kv.getValueLength());
2449           switch (compareOp) {
2450           case LESS:
2451             matches = compareResult <= 0;
2452             break;
2453           case LESS_OR_EQUAL:
2454             matches = compareResult < 0;
2455             break;
2456           case EQUAL:
2457             matches = compareResult == 0;
2458             break;
2459           case NOT_EQUAL:
2460             matches = compareResult != 0;
2461             break;
2462           case GREATER_OR_EQUAL:
2463             matches = compareResult > 0;
2464             break;
2465           case GREATER:
2466             matches = compareResult >= 0;
2467             break;
2468           default:
2469             throw new RuntimeException("Unknown Compare op " + compareOp.name());
2470           }
2471         }
2472         //If matches put the new put or delete the new delete
2473         if (matches) {
2474           // All edits for the given row (across all column families) must
2475           // happen atomically.
2476           doBatchMutate((Mutation)w, lid);
2477           this.checkAndMutateChecksPassed.increment();
2478           return true;
2479         }
2480         this.checkAndMutateChecksFailed.increment();
2481         return false;
2482       } finally {
2483         if(lockId == null) releaseRowLock(lid);
2484       }
2485     } finally {
2486       closeRegionOperation();
2487     }
2488   }

2516   /*
2517    * Check if resources to support an update.
2518    *
2519    * Here we synchronize on HRegion, a broad scoped lock.  Its appropriate
2520    * given we're figuring in here whether this region is able to take on
2521    * writes.  This is only method with a synchronize (at time of writing),
2522    * this and the synchronize on 'this' inside in internalFlushCache to send
2523    * the notify.
2524    */
2525   private void checkResources()
2526       throws RegionTooBusyException, InterruptedIOException {
2527 
2528     // If catalog region, do not impose resource constraints or block updates.
2529     if (this.getRegionInfo().isMetaRegion()) return;
2530 
2531     boolean blocked = false;
2532     long startTime = 0;
2533     while (this.memstoreSize.get() > this.blockingMemStoreSize) {
2534       requestFlush();
2535       if (!blocked) {
2536         startTime = EnvironmentEdgeManager.currentTimeMillis();
2537         LOG.info("Blocking updates for '" + Thread.currentThread().getName() +
2538           "' on region " + Bytes.toStringBinary(getRegionName()) +
2539           ": memstore size " +
2540           StringUtils.humanReadableInt(this.memstoreSize.get()) +
2541           " is >= than blocking " +
2542           StringUtils.humanReadableInt(this.blockingMemStoreSize) + " size");
2543       }

2570     }

2573       final long totalTime = EnvironmentEdgeManager.currentTimeMillis() - startTime;

2575         this.updatesBlockedMs.add(totalTime);
2576       }

2578           + Thread.currentThread().getName() + "'");

2657    * the wal. This method is then invoked to rollback the memstore.

2760    * the recovered edits back up into this region.
2761    *
2762    * <p>We can ignore any log message that has a sequence ID that's equal to or
2763    * lower than minSeqId.  (Because we know such log messages are already
2764    * reflected in the HFiles.)
2765    *
2766    * <p>While this is running we are putting pressure on memory yet we are
2767    * outside of our usual accounting because we are not yet an onlined region
2768    * (this stuff is being run as part of Region initialization).  This means
2769    * that if we're up against global memory limits, we'll not be flagged to flush
2770    * because we are not online. We can't be flushed by usual mechanisms anyways;
2771    * we're not yet online so our relative sequenceids are not yet aligned with
2772    * HLog sequenceids -- not till we come up online, post processing of split
2773    * edits.
2774    *
2775    * <p>But to help relieve memory pressure, at least manage our own heap size
2776    * flushing if are in excess of per-region limits.  Flushing, though, we have
2777    * to be careful and avoid using the regionserver/hlog sequenceid.  Its running
2778    * on a different line to whats going on in here in this region context so if we
2779    * crashed replaying these edits, but in the midst had a flush that used the
2780    * regionserver log with a sequenceid in excess of whats going on in here
2781    * in this region and with its split editlogs, then we could miss edits the
2782    * next time we go to recover. So, we have to flush inline, using seqids that
2783    * make sense in a this single region context only -- until we online.
2784    *
2785    * @param regiondir
2786    * @param maxSeqIdInStores Any edit found in split editlogs needs to be in excess of
2787    * the maxSeqId for the store to be applied, else its skipped.
2788    * @param reporter
2789    * @return the sequence id of the last edit added to this region out of the
2790    * recovered edits log or <code>minSeqId</code> if nothing added from editlogs.
2791    * @throws UnsupportedEncodingException
2792    * @throws IOException
2793    */

2795       Map<byte[], Long> maxSeqIdInStores,
2796       final CancelableProgressable reporter, final MonitoredTask status)

2799     for (Long maxSeqIdInStore : maxSeqIdInStores.values()) {
2800       if (maxSeqIdInStore < minSeqIdForTheRegion || minSeqIdForTheRegion == -1) {
2801         minSeqIdForTheRegion = maxSeqIdInStore;
2802       }
2803     }
2804     long seqid = minSeqIdForTheRegion;
2805     
2806     NavigableSet<Path> files = HLogUtil.getSplitEditFilesSorted(fs, regiondir);
2807     if (files == null || files.isEmpty()) return seqid;
2808 
2809     for (Path edits: files) {
2810       if (edits == null || !this.fs.exists(edits)) {
2811         LOG.warn("Null or non-existent edits file: " + edits);
2812         continue;
2813       }
2814       if (isZeroLengthThenDelete(this.fs, edits)) continue;
2815 
2816       long maxSeqId = Long.MAX_VALUE;
2817       String fileName = edits.getName();
2818       maxSeqId = Math.abs(Long.parseLong(fileName));
2819       if (maxSeqId <= minSeqIdForTheRegion) {
2820         String msg = "Maximum sequenceid for this log is " + maxSeqId
2821             + " and minimum sequenceid for the region is " + minSeqIdForTheRegion
2822             + ", skipped the whole file, path=" + edits;
2823         LOG.debug(msg);
2824         continue;
2825       }
2826 
2827       try {
2828         seqid = replayRecoveredEdits(edits, maxSeqIdInStores, reporter);
2829       } catch (IOException e) {
2830         boolean skipErrors = conf.getBoolean(
2831             HConstants.HREGION_EDITS_REPLAY_SKIP_ERRORS,
2832             conf.getBoolean(
2833                 "hbase.skip.errors",
2834                 HConstants.DEFAULT_HREGION_EDITS_REPLAY_SKIP_ERRORS));
2835         if (conf.get("hbase.skip.errors") != null) {
2836           LOG.warn(
2837               "The property 'hbase.skip.errors' has been deprecated. Please use " +
2838               HConstants.HREGION_EDITS_REPLAY_SKIP_ERRORS + " instead.");
2839         }
2840         if (skipErrors) {
2841           Path p = HLogUtil.moveAsideBadEditsFile(fs, edits);
2842           LOG.error(HConstants.HREGION_EDITS_REPLAY_SKIP_ERRORS
2843               + "=true so continuing. Renamed " + edits +
2844               " as " + p, e);
2845         } else {
2846           throw e;
2847         }
2848       }
2849       // The edits size added into rsAccounting during this replaying will not
2850       // be required any more. So just clear it.
2851       if (this.rsAccounting != null) {
2852         this.rsAccounting.clearRegionReplayEditsSize(this.regionInfo.getRegionName());
2853       }
2854     }
2855     if (seqid > minSeqIdForTheRegion) {
2856       // Then we added some edits to memory. Flush and cleanup split edit files.
2857       internalFlushcache(null, seqid, status);
2858     }
2859     // Now delete the content of recovered edits.  We're done w/ them.
2860     for (Path file: files) {
2861       if (!this.fs.delete(file, false)) {
2862         LOG.error("Failed delete of " + file);
2863       } else {
2864         LOG.debug("Deleted recovered.edits file=" + file);
2865       }
2866     }
2867     return seqid;
2868   }

2883     LOG.info(msg);
2884     MonitoredTask status = TaskMonitor.get().createStatus(msg);
2885 
2886     status.setStatus("Opening logs");

2888     try {
2889       reader = HLogFactory.createReader(this.fs, edits, conf);
2890       long currentEditSeqId = -1;
2891       long firstSeqIdInLog = -1;
2892       long skippedEdits = 0;
2893       long editsCount = 0;
2894       long intervalEdits = 0;

2896       Store store = null;
2897       boolean reported_once = false;
2898

2901         int interval = this.conf.getInt("hbase.hstore.report.interval.edits",

2904         int period = this.conf.getInt("hbase.hstore.report.period",
2905             this.conf.getInt("hbase.master.assignment.timeoutmonitor.timeout",
2906                 180000) / 2);

2908

2917               intervalEdits = 0;

2921                     " skipped=" + skippedEdits +
2922                     " edits=" + editsCount);
2923                 // Timeout reached
2924                 if(!reporter.progress()) {
2925                   msg = "Progressable reporter failed, stopping replay";
2926                   LOG.warn(msg);
2927                   status.abort(msg);
2928                   throw new IOException(msg);
2929                 }
2930                 reported_once = true;
2931                 lastReport = cur;
2932               }
2933             }
2934           }
2935 
2936           // Start coprocessor replay here. The coprocessor is for each WALEdit
2937           // instead of a KeyValue.
2938           if (coprocessorHost != null) {
2939             status.setStatus("Running pre-WAL-restore hook in coprocessors");
2940             if (coprocessorHost.preWALRestore(this.getRegionInfo(), key, val)) {
2941               // if bypass this log entry, ignore it ...

2994         status.abort(msg);

2997         // then this problem is idempotent and retrying won't help

2999           Path p = HLogUtil.moveAsideBadEditsFile(fs, edits);
3000           msg = "File corruption encountered!  " +
3001               "Continuing, but renaming " + edits + " as " + p;
3002           LOG.warn(msg, ioe);
3003           status.setStatus(msg);
3004         } else {
3005           status.abort(StringUtils.stringifyException(ioe));
3006           // other IO errors may be transient (bad network connection,
3007           // checksum exception on one datanode, etc).  throw & retry

4083       throw new IOException("Cannot merge non-adjacent regions");

4405     // 1. Run pre-process hook
4406     processor.preProcess(this, walEdit);
4407 
4408     // Short circuit the read only case
4409     if (processor.readOnly()) {
4410       try {
4411         long now = EnvironmentEdgeManager.currentTimeMillis();
4412         doProcessRowWithTimeout(
4413             processor, now, this, null, null, timeout);
4414         processor.postProcess(this, walEdit);
4415       } catch (IOException e) {
4416         throw e;
4417       } finally {
4418         closeRegionOperation();
4419       }
4420       return;
4421     }
4422 
4423     MultiVersionConsistencyControl.WriteEntry writeEntry = null;
4424     boolean locked = false;
4425     boolean walSyncSuccessful = false;
4426     List<Integer> acquiredLocks = null;
4427     long addedSize = 0;
4428     List<KeyValue> mutations = new ArrayList<KeyValue>();
4429     Collection<byte[]> rowsToLock = processor.getRowsToLock();
4430     try {
4431       // 2. Acquire the row lock(s)
4432       acquiredLocks = new ArrayList<Integer>(rowsToLock.size());
4433       for (byte[] row : rowsToLock) {
4434         // Attempt to lock all involved rows, fail if one lock times out
4435         Integer lid = getLock(null, row, true);
4436         if (lid == null) {
4437           throw new IOException("Failed to acquire lock on "
4438               + Bytes.toStringBinary(row));
4439         }
4440         acquiredLocks.add(lid);
4441       }
4442       // 3. Region lock
4443       lock(this.updatesLock.readLock(), acquiredLocks.size());
4444       locked = true;
4445 
4446       long now = EnvironmentEdgeManager.currentTimeMillis();
4447       try {
4448         // 4. Let the processor scan the rows, generate mutations and add
4449         //    waledits
4450         doProcessRowWithTimeout(
4451             processor, now, this, mutations, walEdit, timeout);
4452 
4453         if (!mutations.isEmpty()) {
4454           // 5. Get a mvcc write number
4455           writeEntry = mvcc.beginMemstoreInsert();
4456           // 6. Apply to memstore
4457           for (KeyValue kv : mutations) {
4458             kv.setMemstoreTS(writeEntry.getWriteNumber());
4459             byte[] family = kv.getFamily();
4460             checkFamily(family);
4461             addedSize += stores.get(family).add(kv);
4462           }
4463 
4464           long txid = 0;
4465           // 7. Append no sync
4466           if (!walEdit.isEmpty()) {
4467             txid = this.log.appendNoSync(this.regionInfo,
4468                 this.htableDescriptor.getName(), walEdit,
4469                 processor.getClusterId(), now, this.htableDescriptor);
4470           }
4471           // 8. Release region lock
4472           if (locked) {
4473             this.updatesLock.readLock().unlock();
4474             locked = false;
4475           }
4476 
4477           // 9. Release row lock(s)
4478           if (acquiredLocks != null) {
4479             for (Integer lid : acquiredLocks) {
4480               releaseRowLock(lid);
4481             }
4482             acquiredLocks = null;
4483           }
4484           // 10. Sync edit log
4485           if (txid != 0) {
4486             syncOrDefer(txid);
4487           }
4488           walSyncSuccessful = true;
4489         }
4490       } finally {
4491         if (!mutations.isEmpty() && !walSyncSuccessful) {
4492           LOG.warn("Wal sync failed. Roll back " + mutations.size() +
4493               " memstore keyvalues for row(s):" +
4494               processor.getRowsToLock().iterator().next() + "...");
4495           for (KeyValue kv : mutations) {
4496             stores.get(kv.getFamily()).rollback(kv);
4497           }
4498         }
4499         // 11. Roll mvcc forward
4500         if (writeEntry != null) {
4501           mvcc.completeMemstoreInsert(writeEntry);
4502           writeEntry = null;
4503         }
4504         if (locked) {
4505           this.updatesLock.readLock().unlock();
4506           locked = false;
4507         }
4508         if (acquiredLocks != null) {
4509           for (Integer lid : acquiredLocks) {
4510             releaseRowLock(lid);
4511           }
4512         }
4513 
4514       }
4515 
4516       // 12. Run post-process hook
4517       processor.postProcess(this, walEdit);
4518 
4519     } catch (IOException e) {
4520       throw e;
4521     } finally {
4522       closeRegionOperation();
4523       if (!mutations.isEmpty() &&
4524           isFlushSize(this.addAndGetGlobalMemstoreSize(addedSize))) {
4525         requestFlush();
4526       }
4527     }
4528   }

4646                   kv.getQualifierLength(), now, KeyValue.Type.Put,

4653                   newKV.getBuffer(),
4654                   newKV.getValueOffset() + oldKv.getValueLength(),

4658               // allocate an empty kv once
4659               newKV = new KeyValue(row.length, kv.getFamilyLength(),

4661                   kv.getValueLength());
4662               // copy in the value
4663               System.arraycopy(kv.getBuffer(), kv.getValueOffset(),
4664                   newKV.getBuffer(), newKV.getValueOffset(),
4665                   kv.getValueLength());

4670             System.arraycopy(kv.getBuffer(), kv.getFamilyOffset(),
4671                 newKV.getBuffer(), newKV.getFamilyOffset(),

4681             if (writeToWAL) {
4682               if (walEdits == null) {
4683                 walEdits = new WALEdit();
4684               }
4685               walEdits.add(newKV);
4686             }
4687           }

4689           //store the kvs to the temporary memstore before writing HLog
4690           tempMemstore.put(store, kvs);

4695           // Using default cluster id, as this can only happen in the orginating
4696           // cluster. A slave cluster receives the final value (not the delta)
4697           // as a Put.
4698           txid = this.log.appendNoSync(regionInfo,
4699               this.htableDescriptor.getName(), walEdits,
4700               HConstants.DEFAULT_CLUSTER_ID, EnvironmentEdgeManager.currentTimeMillis(),
4701               this.htableDescriptor);
4702         }
4703 
4704         //Actually write to Memstore now
4705         for (Map.Entry<Store, List<KeyValue>> entry : tempMemstore.entrySet()) {
4706           Store store = entry.getKey();

4708             // upsert if VERSIONS for this CF == 1
4709             size += store.upsert(entry.getValue(), getSmallestReadPoint());
4710           } else {
4711             // otherwise keep older versions around
4712             for (KeyValue kv : entry.getValue()) {
4713               size += store.add(kv);
4714             }
4715           }
4716           allKVs.addAll(entry.getValue());
4717         }
4718         size = this.addAndGetGlobalMemstoreSize(size);
4719         flush = isFlushSize(size);
4720       } finally {
4721         this.updatesLock.readLock().unlock();
4722         releaseRowLock(lid);
4723       }
4724       if (writeToWAL) {
4725         syncOrDefer(txid); // sync the transaction log outside the rowlock
4726       }
4727     } finally {
4728       if (w != null) {
4729         mvcc.completeMemstoreInsert(w);
4730       }
4731       closeRegionOperation();
4732     }
4733 
4734     if (this.metricsRegion != null) {
4735       this.metricsRegion.updateAppend();
4736     }
4737 
4738     if (flush) {
4739       // Request a cache flush. Do it outside update lock.
4740       requestFlush();
4741     }
4742
