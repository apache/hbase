18  package org.apache.hadoop.hbase;
19  
20  import org.apache.hadoop.conf.Configurable;
21  import org.apache.hadoop.conf.Configuration;
22  
23  /**
24   * HBase version of Hadoop's Configured class that doesn't initialize the
25   * configuration via {@link #setConf(Configuration)} in the constructor, but
26   * only sets the configuration through the {@link #setConf(Configuration)}
27   * method
28   */
29  public class BaseConfigurable implements Configurable {
30  
31    private Configuration conf;
32  
33    @Override
34    public void setConf(Configuration conf) {
35      this.conf = conf;
36    }
37  
38    @Override
39    public Configuration getConf() {
40      return this.conf;
41    }
42  
43  }
