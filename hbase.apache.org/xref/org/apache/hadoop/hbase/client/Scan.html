39  /**
40   * Used to perform Scan operations.
41   * <p>
42   * All operations are identical to {@link Get} with the exception of
43   * instantiation.  Rather than specifying a single row, an optional startRow
44   * and stopRow may be defined.  If rows are not specified, the Scanner will
45   * iterate over all rows.
46   * <p>
47   * To scan everything for each row, instantiate a Scan object.
48   * <p>
49   * To modify scanner caching for just this scan, use {@link #setCaching(int) setCaching}.
50   * If caching is NOT set, we will use the caching value of the hosting {@link HTable}.  See
51   * {@link HTable#setScannerCaching(int)}. In addition to row caching, it is possible to specify a
52   * maximum result size, using {@link #setMaxResultSize(long)}. When both are used,
53   * single server requests are limited by either number of rows or maximum result size, whichever
54   * limit comes first.
55   * <p>
56   * To further define the scope of what to get when scanning, perform additional
57   * methods as outlined below.
58   * <p>
59   * To get all columns from specific families, execute {@link #addFamily(byte[]) addFamily}
60   * for each family to retrieve.
61   * <p>
62   * To get specific columns, execute {@link #addColumn(byte[], byte[]) addColumn}
63   * for each column to retrieve.
64   * <p>
65   * To only retrieve columns within a specific range of version timestamps,
66   * execute {@link #setTimeRange(long, long) setTimeRange}.
67   * <p>
68   * To only retrieve columns with a specific timestamp, execute
69   * {@link #setTimeStamp(long) setTimestamp}.
70   * <p>
71   * To limit the number of versions of each column to be returned, execute
72   * {@link #setMaxVersions(int) setMaxVersions}.
73   * <p>
74   * To limit the maximum number of values returned for each call to next(),
75   * execute {@link #setBatch(int) setBatch}.
76   * <p>
77   * To add a filter, execute {@link #setFilter(org.apache.hadoop.hbase.filter.Filter) setFilter}.
78   * <p>
79   * Expert: To explicitly disable server-side block caching for this scan,
80   * execute {@link #setCacheBlocks(boolean)}.
81   */

145   /**
146    * Creates a new instance of this class while copying all values.
147    *
148    * @param scan  The scan instance to copy from.
149    * @throws IOException When copying the values fails.
150    */
151   public Scan(Scan scan) throws IOException {
152     startRow = scan.getStartRow();
153     stopRow  = scan.getStopRow();
154     maxVersions = scan.getMaxVersions();
155     batch = scan.getBatch();
156     storeLimit = scan.getMaxResultsPerColumnFamily();
157     storeOffset = scan.getRowOffsetPerColumnFamily();
158     caching = scan.getCaching();
159     maxResultSize = scan.getMaxResultSize();
160     cacheBlocks = scan.getCacheBlocks();
161     filter = scan.getFilter(); // clone?
162     TimeRange ctr = scan.getTimeRange();
163     tr = new TimeRange(ctr.getMin(), ctr.getMax());
164     Map<byte[], NavigableSet<byte[]>> fams = scan.getFamilyMap();
165     for (Map.Entry<byte[],NavigableSet<byte[]>> entry : fams.entrySet()) {
166       byte [] fam = entry.getKey();
167       NavigableSet<byte[]> cols = entry.getValue();
168       if (cols != null && cols.size() > 0) {
169         for (byte[] col : cols) {
170           addColumn(fam, col);
171         }
172       } else {
173         addFamily(fam);
174       }
175     }
176     for (Map.Entry<String, byte[]> attr : scan.getAttributesMap().entrySet()) {
177       setAttribute(attr.getKey(), attr.getValue());
178     }
179   }

220    * @param qualifier column qualifier
221    * @return this

223   public Scan addColumn(byte [] family, byte [] qualifier) {
224     NavigableSet<byte []> set = familyMap.get(family);
225     if(set == null) {
226       set = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
227     }
228     if (qualifier == null) {
229       qualifier = HConstants.EMPTY_BYTE_ARRAY;
230     }
231     set.add(qualifier);
232     familyMap.put(family, set);
233     return this;
234   }

521   /**
522    * Compile the table and column family (i.e. schema) information
523    * into a String. Useful for parsing and aggregation by debugging,
524    * logging, and administration tools.
525    * @return Map
526    */
527   @Override
528   public Map<String, Object> getFingerprint() {
529     Map<String, Object> map = new HashMap<String, Object>();
530     List<String> families = new ArrayList<String>();
531     if(this.familyMap.size() == 0) {
532       map.put("families", "ALL");
533       return map;
534     } else {
535       map.put("families", families);
536     }
537     for (Map.Entry<byte [], NavigableSet<byte[]>> entry :
538         this.familyMap.entrySet()) {
539       families.add(Bytes.toStringBinary(entry.getKey()));
540     }
541     return map;
542   }

544   /**
545    * Compile the details beyond the scope of getFingerprint (row, columns,
546    * timestamps, etc.) into a Map along with the fingerprinted information.
547    * Useful for debugging, logging, and administration tools.
548    * @param maxCols a limit on the number of columns output prior to truncation
549    * @return Map
550    */ 
551   @Override
552   public Map<String, Object> toMap(int maxCols) {
553     // start with the fingerpring map and build on top of it
554     Map<String, Object> map = getFingerprint();
555     // map from families to column list replaces fingerprint's list of families
556     Map<String, List<String>> familyColumns =
557       new HashMap<String, List<String>>();
558     map.put("families", familyColumns);
559     // add scalar information first
560     map.put("startRow", Bytes.toStringBinary(this.startRow));
561     map.put("stopRow", Bytes.toStringBinary(this.stopRow));
562     map.put("maxVersions", this.maxVersions);
563     map.put("batch", this.batch);
564     map.put("caching", this.caching);
565     map.put("maxResultSize", this.maxResultSize);
566     map.put("cacheBlocks", this.cacheBlocks);
567     List<Long> timeRange = new ArrayList<Long>();
568     timeRange.add(this.tr.getMin());
569     timeRange.add(this.tr.getMax());
570     map.put("timeRange", timeRange);
571     int colCount = 0;
572     // iterate through affected families and list out up to maxCols columns
573     for (Map.Entry<byte [], NavigableSet<byte[]>> entry :
574       this.familyMap.entrySet()) {
575       List<String> columns = new ArrayList<String>();
576       familyColumns.put(Bytes.toStringBinary(entry.getKey()), columns);
577       if(entry.getValue() == null) {
578         colCount++;
579         --maxCols;
580         columns.add("ALL");
581       } else {
582         colCount += entry.getValue().size();
583         if (maxCols <= 0) {
584           continue;
585         } 
586         for (byte [] column : entry.getValue()) {
587           if (--maxCols <= 0) {
588             continue;
589           }
590           columns.add(Bytes.toStringBinary(column));
591         }
592       } 
593     }       
594     map.put("totalColumns", colCount);
595     if (this.filter != null) {
596       map.put("filter", this.filter.toString());
597     }
598     // add the id if set
599     if (getId() != null) {
600       map.put("id", getId());
601     }
602     return map;
603   }
