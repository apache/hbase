53    public void testVersion() throws DeserializationException, IOException {
54      HBaseTestingUtility htu = new HBaseTestingUtility();
55      final FileSystem fs = htu.getTestFileSystem();
56      final Path rootdir = htu.getDataTestDir();
57      assertNull(FSUtils.getVersion(fs, rootdir));
58      // Write out old format version file.  See if we can read it in and convert.
59      Path versionFile = new Path(rootdir, HConstants.VERSION_FILE_NAME);
60      FSDataOutputStream s = fs.create(versionFile);
61      final String version = HConstants.FILE_SYSTEM_VERSION;
62      s.writeUTF(version);
63      s.close();
64      assertTrue(fs.exists(versionFile));
65      FileStatus [] status = fs.listStatus(versionFile);
66      assertNotNull(status);
67      assertTrue(status.length > 0);
68      String newVersion = FSUtils.getVersion(fs, rootdir);
69      assertEquals(version.length(), newVersion.length());
70      assertEquals(version, newVersion);
71      // File will have been converted. Exercise the pb format
72      assertEquals(version, FSUtils.getVersion(fs, rootdir));
73      FSUtils.checkVersion(fs, rootdir, true);
74    }

91    private void WriteDataToHDFS(FileSystem fs, Path file, int dataSize)
92      throws Exception {
93      FSDataOutputStream out = fs.create(file);

95      out.write(data, 0, dataSize);
96      out.close();

99    @Test public void testcomputeHDFSBlocksDistribution() throws Exception {
100     HBaseTestingUtility htu = new HBaseTestingUtility();
101     final int DEFAULT_BLOCK_SIZE = 1024;
102     htu.getConfiguration().setLong("dfs.block.size", DEFAULT_BLOCK_SIZE);
103     MiniDFSCluster cluster = null;
104     Path testFile = null;
105     
106     try {
107       // set up a cluster with 3 nodes
108       String hosts[] = new String[] { "host1", "host2", "host3" };
109       cluster = htu.startMiniDFSCluster(hosts);
110       cluster.waitActive();
111       FileSystem fs = cluster.getFileSystem();
112 
113       // create a file with two blocks
114       testFile = new Path("/test1.txt");
115       WriteDataToHDFS(fs, testFile, 2*DEFAULT_BLOCK_SIZE);
116       
117       // given the default replication factor is 3, the same as the number of
118       // datanodes; the locality index for each host should be 100%,
119       // or getWeight for each host should be the same as getUniqueBlocksWeights
120       final long maxTime = System.currentTimeMillis() + 2000;
121       boolean ok;
122       do {
123         ok = true;
124         FileStatus status = fs.getFileStatus(testFile);
125         HDFSBlocksDistribution blocksDistribution =
126           FSUtils.computeHDFSBlocksDistribution(fs, status, 0, status.getLen());
127         long uniqueBlocksTotalWeight =
128           blocksDistribution.getUniqueBlocksTotalWeight();
129         for (String host : hosts) {
130           long weight = blocksDistribution.getWeight(host);
131           ok = (ok && uniqueBlocksTotalWeight == weight);
132         }
133       } while (!ok && System.currentTimeMillis() < maxTime);
134       assertTrue(ok);
135       } finally {
136       htu.shutdownMiniDFSCluster();
137     }
138 
139 
140     try {
141       // set up a cluster with 4 nodes
142       String hosts[] = new String[] { "host1", "host2", "host3", "host4" };
143       cluster = htu.startMiniDFSCluster(hosts);
144       cluster.waitActive();
145       FileSystem fs = cluster.getFileSystem();
146 
147       // create a file with three blocks
148       testFile = new Path("/test2.txt");        
149       WriteDataToHDFS(fs, testFile, 3*DEFAULT_BLOCK_SIZE);
150               
151       // given the default replication factor is 3, we will have total of 9
152       // replica of blocks; thus the host with the highest weight should have
153       // weight == 3 * DEFAULT_BLOCK_SIZE
154       final long maxTime = System.currentTimeMillis() + 2000;
155       long weight;
156       long uniqueBlocksTotalWeight;
157       do {
158         FileStatus status = fs.getFileStatus(testFile);
159         HDFSBlocksDistribution blocksDistribution =
160           FSUtils.computeHDFSBlocksDistribution(fs, status, 0, status.getLen());
161         uniqueBlocksTotalWeight = blocksDistribution.getUniqueBlocksTotalWeight();
162 
163         String tophost = blocksDistribution.getTopHosts().get(0);
164         weight = blocksDistribution.getWeight(tophost);
165 
166         // NameNode is informed asynchronously, so we may have a delay. See HBASE-6175
167       } while (uniqueBlocksTotalWeight != weight && System.currentTimeMillis() < maxTime);
168       assertTrue(uniqueBlocksTotalWeight == weight);
169 
170     } finally {
171       htu.shutdownMiniDFSCluster();
172     }
173 
174     
175     try {
176       // set up a cluster with 4 nodes
177       String hosts[] = new String[] { "host1", "host2", "host3", "host4" };
178       cluster = htu.startMiniDFSCluster(hosts);
179       cluster.waitActive();
180       FileSystem fs = cluster.getFileSystem();
181 
182       // create a file with one block
183       testFile = new Path("/test3.txt");        
184       WriteDataToHDFS(fs, testFile, DEFAULT_BLOCK_SIZE);
185       
186       // given the default replication factor is 3, we will have total of 3
187       // replica of blocks; thus there is one host without weight
188       final long maxTime = System.currentTimeMillis() + 2000;
189       HDFSBlocksDistribution blocksDistribution;
190       do {
191         FileStatus status = fs.getFileStatus(testFile);
192         blocksDistribution = FSUtils.computeHDFSBlocksDistribution(fs, status, 0, status.getLen());
193         // NameNode is informed asynchronously, so we may have a delay. See HBASE-6175
194       }
195       while (blocksDistribution.getTopHosts().size() != 3 && System.currentTimeMillis() < maxTime);
196       assertEquals("Wrong number of hosts distributing blocks.", 3,
197         blocksDistribution.getTopHosts().size());
198     } finally {
199       htu.shutdownMiniDFSCluster();
200     }

203   @Test
204   public void testPermMask() throws Exception {
205 
206     Configuration conf = HBaseConfiguration.create();
207     conf.setBoolean(HConstants.ENABLE_DATA_FILE_UMASK, true);
208     FileSystem fs = FileSystem.get(conf);
209     // first check that we don't crash if we don't have perms set
210     FsPermission defaultPerms = FSUtils.getFilePermissions(fs, conf,
211         HConstants.DATA_FILE_UMASK_KEY);
212     assertEquals(FsPermission.getDefault(), defaultPerms);
213 
214     conf.setStrings(HConstants.DATA_FILE_UMASK_KEY, "077");
215     // now check that we get the right perms
216     FsPermission filePerm = FSUtils.getFilePermissions(fs, conf,
217         HConstants.DATA_FILE_UMASK_KEY);
218     assertEquals(new FsPermission("700"), filePerm);
219 
220     // then that the correct file is created
221     Path p = new Path("target" + File.separator + UUID.randomUUID().toString());
222     try {
223       FSDataOutputStream out = FSUtils.create(fs, p, filePerm);
224       out.close();
225       FileStatus stat = fs.getFileStatus(p);
226       assertEquals(new FsPermission("700"), stat.getPermission());
227       // and then cleanup
228     } finally {
229       fs.delete(p, true);
230     }
231   }

233   @Test
234   public void testDeleteAndExists() throws Exception {
235     HBaseTestingUtility htu = new HBaseTestingUtility();
236     Configuration conf = htu.getConfiguration();
237     conf.setBoolean(HConstants.ENABLE_DATA_FILE_UMASK, true);
238     FileSystem fs = FileSystem.get(conf);
239     FsPermission perms = FSUtils.getFilePermissions(fs, conf, HConstants.DATA_FILE_UMASK_KEY);
240     // then that the correct file is created
241     String file = UUID.randomUUID().toString();
242     Path p = new Path(htu.getDataTestDir(), "temptarget" + File.separator + file);
243     Path p1 = new Path(htu.getDataTestDir(), "temppath" + File.separator + file);
244     try {
245       FSDataOutputStream out = FSUtils.create(fs, p, perms);
246       out.close();
247       assertTrue("The created file should be present", FSUtils.isExists(fs, p));
248       // delete the file with recursion as false. Only the file will be deleted.
249       FSUtils.delete(fs, p, false);
250       // Create another file
251       FSDataOutputStream out1 = FSUtils.create(fs, p1, perms);
252       out1.close();
253       // delete the file with recursion as false. Still the file only will be deleted
254       FSUtils.delete(fs, p1, true);
255       assertFalse("The created file should be present", FSUtils.isExists(fs, p1));
256       // and then cleanup
257     } finally {
258       FSUtils.delete(fs, p, true);
259       FSUtils.delete(fs, p1, true);
260     }
261   }
