38  /**
39   * ThriftServer- this class starts up a Thrift server which implements the
40   * Hbase API specified in the Hbase.thrift IDL file. The server runs in an
41   * independent process.
42   */
43  @InterfaceAudience.Private
44  public class ThriftServer {
45  
46    private static final Log LOG = LogFactory.getLog(ThriftServer.class);
47  
48    private static final String MIN_WORKERS_OPTION = "minWorkers";
49    private static final String MAX_WORKERS_OPTION = "workers";
50    private static final String MAX_QUEUE_SIZE_OPTION = "queue";
51    private static final String KEEP_ALIVE_SEC_OPTION = "keepAliveSec";
52    static final String BIND_OPTION = "bind";
53    static final String COMPACT_OPTION = "compact";
54    static final String FRAMED_OPTION = "framed";
55    static final String PORT_OPTION = "port";
56  
57    private static final String DEFAULT_BIND_ADDR = "0.0.0.0";
58    private static final int DEFAULT_LISTEN_PORT = 9090;
59  
60    private Configuration conf;
61    ThriftServerRunner serverRunner;
62  
63    //
64    // Main program and support routines
65    //
66  
67    public ThriftServer(Configuration conf) {
68      this.conf = HBaseConfiguration.create(conf);
69    }
70  
71    private static void printUsageAndExit(Options options, int exitCode)
72        throws ExitCodeException {
73      HelpFormatter formatter = new HelpFormatter();
74      formatter.printHelp("Thrift", null, options,
75          "To start the Thrift server run 'bin/hbase-daemon.sh start thrift'\n" +
76          "To shutdown the thrift server run 'bin/hbase-daemon.sh stop " +
77          "thrift' or send a kill signal to the thrift server pid",
78          true);
79      throw new ExitCodeException(exitCode, "");
80    }
81  
82    /**
83     * Start up or shuts down the Thrift server, depending on the arguments.
84     * @param args
85     */
86     void doMain(final String[] args) throws Exception {
87       processOptions(args);
88       serverRunner = new ThriftServerRunner(conf);
89       serverRunner.run();
90    }
91  
92    /**
93     * Parse the command line options to set parameters the conf.
94     */
95    private void processOptions(final String[] args) throws Exception {
96      Options options = new Options();
97      options.addOption("b", BIND_OPTION, true, "Address to bind " +
98          "the Thrift server to. [default: " + DEFAULT_BIND_ADDR + "]");
99      options.addOption("p", PORT_OPTION, true, "Port to bind to [default: " +
100         DEFAULT_LISTEN_PORT + "]");
101     options.addOption("f", FRAMED_OPTION, false, "Use framed transport");
102     options.addOption("c", COMPACT_OPTION, false, "Use the compact protocol");
103     options.addOption("h", "help", false, "Print help information");
104 
105     options.addOption("m", MIN_WORKERS_OPTION, true,
106         "The minimum number of worker threads for " +
107         ImplType.THREAD_POOL.simpleClassName());
108 
109     options.addOption("w", MAX_WORKERS_OPTION, true,
110         "The maximum number of worker threads for " +
111         ImplType.THREAD_POOL.simpleClassName());
112 
113     options.addOption("q", MAX_QUEUE_SIZE_OPTION, true,
114         "The maximum number of queued requests in " +
115         ImplType.THREAD_POOL.simpleClassName());
116 
117     options.addOption("k", KEEP_ALIVE_SEC_OPTION, true,
118         "The amount of time in secods to keep a thread alive when idle in " +
119         ImplType.THREAD_POOL.simpleClassName());
120 
121     options.addOptionGroup(ImplType.createOptionGroup());
122 
123     CommandLineParser parser = new PosixParser();
124     CommandLine cmd = parser.parse(options, args);
125 
126     // This is so complicated to please both bin/hbase and bin/hbase-daemon.
127     // hbase-daemon provides "start" and "stop" arguments
128     // hbase should print the help if no argument is provided
129     List<String> commandLine = Arrays.asList(args);
130     boolean stop = commandLine.contains("stop");
131     boolean start = commandLine.contains("start");
132     boolean invalidStartStop = (start && stop) || (!start && !stop);
133     if (cmd.hasOption("help") || invalidStartStop) {
134       if (invalidStartStop) {
135         LOG.error("Exactly one of 'start' and 'stop' has to be specified");
136       }
137       printUsageAndExit(options, 1);
138     }
139 
140     // Get port to bind to
141     try {
142       int listenPort = Integer.parseInt(cmd.getOptionValue(PORT_OPTION,
143           String.valueOf(DEFAULT_LISTEN_PORT)));
144       conf.setInt(ThriftServerRunner.PORT_CONF_KEY, listenPort);
145     } catch (NumberFormatException e) {
146       LOG.error("Could not parse the value provided for the port option", e);
147       printUsageAndExit(options, -1);
148     }
149 
150     // Make optional changes to the configuration based on command-line options
151     optionToConf(cmd, MIN_WORKERS_OPTION,
152         conf, TBoundedThreadPoolServer.MIN_WORKER_THREADS_CONF_KEY);
153     optionToConf(cmd, MAX_WORKERS_OPTION,
154         conf, TBoundedThreadPoolServer.MAX_WORKER_THREADS_CONF_KEY);
155     optionToConf(cmd, MAX_QUEUE_SIZE_OPTION,
156         conf, TBoundedThreadPoolServer.MAX_QUEUED_REQUESTS_CONF_KEY);
157     optionToConf(cmd, KEEP_ALIVE_SEC_OPTION,
158         conf, TBoundedThreadPoolServer.THREAD_KEEP_ALIVE_TIME_SEC_CONF_KEY);
159 
160     // Set general thrift server options
161     conf.setBoolean(
162         ThriftServerRunner.COMPACT_CONF_KEY, cmd.hasOption(COMPACT_OPTION));
163     conf.setBoolean(
164         ThriftServerRunner.FRAMED_CONF_KEY, cmd.hasOption(FRAMED_OPTION));
165     if (cmd.hasOption(BIND_OPTION)) {
166       conf.set(
167           ThriftServerRunner.BIND_CONF_KEY, cmd.getOptionValue(BIND_OPTION));
168     }
169 
170     ImplType.setServerImpl(cmd, conf);
171   }
172 
173   public void stop() {
174     serverRunner.shutdown();
175   }
176 
177   private static void optionToConf(CommandLine cmd, String option,
178       Configuration conf, String destConfKey) {
179     if (cmd.hasOption(option)) {
180       String value = cmd.getOptionValue(option);
181       LOG.info("Set configuration key:" + destConfKey + " value:" + value);
182       conf.set(destConfKey, value);
183     }
184   }
185

187    * @param args
188    * @throws Exception
189    */
190   public static void main(String [] args) throws Exception {
191     VersionInfo.logVersion();
192     try {
193       new ThriftServer(HBaseConfiguration.create()).doMain(args);
194     } catch (ExitCodeException ex) {
195       System.exit(ex.getExitCode());
196     }
197   }
