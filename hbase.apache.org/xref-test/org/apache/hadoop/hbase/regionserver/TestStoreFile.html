108   /*
109    * Writes HStoreKey and ImmutableBytes data to passed writer and
110    * then closes it.
111    * @param writer
112    * @throws IOException
113    */
114   public static void writeStoreFile(final StoreFile.Writer writer, byte[] fam, byte[] qualifier)
115   throws IOException {
116     long now = System.currentTimeMillis();
117     try {
118       for (char d = FIRST_CHAR; d <= LAST_CHAR; d++) {
119         for (char e = FIRST_CHAR; e <= LAST_CHAR; e++) {
120           byte[] b = new byte[] { (byte) d, (byte) e };
121           writer.append(new KeyValue(b, fam, qualifier, now, b));
122         }
123       }
124     } finally {
125       writer.close();
126     }
127   }

134   public void testReference()
135   throws IOException {
136     Path storedir = new Path(new Path(this.testDir, "regionname"), "familyname");
137     Path dir = new Path(storedir, "1234567890");
138     // Make a store file and write data to it.
139     StoreFile.Writer writer = new StoreFile.WriterBuilder(conf, cacheConf,
140         this.fs, 8 * 1024)
141             .withOutputDir(dir)
142             .build();
143     writeStoreFile(writer);
144     StoreFile hsf = new StoreFile(this.fs, writer.getPath(), conf, cacheConf,
145         StoreFile.BloomType.NONE, NoOpDataBlockEncoder.INSTANCE);
146     StoreFile.Reader reader = hsf.createReader();
147     // Split on a row, not in middle of row.  Midkey returned by reader
148     // may be in middle of row.  Create new one with empty column and
149     // timestamp.
150     KeyValue kv = KeyValue.createKeyValueFromKey(reader.midkey());
151     byte [] midRow = kv.getRow();
152     kv = KeyValue.createKeyValueFromKey(reader.getLastKey());
153     byte [] finalRow = kv.getRow();
154     // Make a reference
155     Path refPath = StoreFile.split(fs, dir, hsf, midRow, true);
156     StoreFile refHsf = new StoreFile(this.fs, refPath, conf, cacheConf,
157         StoreFile.BloomType.NONE, NoOpDataBlockEncoder.INSTANCE);
158     // Now confirm that I can read from the reference and that it only gets
159     // keys from top half of the file.
160     HFileScanner s = refHsf.createReader().getScanner(false, false);
161     for(boolean first = true; (!s.isSeeked() && s.seekTo()) || s.next();) {
162       ByteBuffer bb = s.getKey();
163       kv = KeyValue.createKeyValueFromKey(bb);
164       if (first) {
165         assertTrue(Bytes.equals(kv.getRow(), midRow));
166         first = false;
167       }
168     }
169     assertTrue(Bytes.equals(kv.getRow(), finalRow));
170   }

173     final String columnFamily = "f";
174     HRegionInfo hri = new HRegionInfo(Bytes.toBytes("table-link"));
175     Path storedir = new Path(new Path(FSUtils.getRootDir(conf),
176       new Path(hri.getTableNameAsString(), hri.getEncodedName())), columnFamily);
177

179     StoreFile.Writer writer = new StoreFile.WriterBuilder(conf, cacheConf,
180          this.fs, 8 * 1024)
181             .withOutputDir(storedir)
182             .build();
183     Path storeFilePath = writer.getPath();
184     writeStoreFile(writer);
185     writer.close();
186 
187     Path dstPath = new Path(FSUtils.getRootDir(conf), new Path("test-region", columnFamily));
188     HFileLink.create(conf, this.fs, dstPath, hri, storeFilePath.getName());
189     Path linkFilePath = new Path(dstPath,
190                   HFileLink.createHFileLinkName(hri, storeFilePath.getName()));

193     StoreFile hsf = new StoreFile(this.fs, linkFilePath, conf, cacheConf,
194         StoreFile.BloomType.NONE, NoOpDataBlockEncoder.INSTANCE);

202       count++;
203     }
204     assertEquals((LAST_CHAR - FIRST_CHAR + 1) * (LAST_CHAR - FIRST_CHAR + 1), count);
205   }

207   private void checkHalfHFile(final StoreFile f)
208   throws IOException {
209     byte [] midkey = f.createReader().midkey();
210     KeyValue midKV = KeyValue.createKeyValueFromKey(midkey);
211     byte [] midRow = midKV.getRow();
212     // Create top split.
213     Path topDir = HStore.getStoreHomedir(this.testDir, "1",
214       Bytes.toBytes(f.getPath().getParent().getName()));
215     if (this.fs.exists(topDir)) {
216       this.fs.delete(topDir, true);
217     }
218     Path topPath = StoreFile.split(this.fs, topDir, f, midRow, true);
219     // Create bottom split.
220     Path bottomDir = HStore.getStoreHomedir(this.testDir, "2",
221       Bytes.toBytes(f.getPath().getParent().getName()));
222     if (this.fs.exists(bottomDir)) {
223       this.fs.delete(bottomDir, true);
224     }
225     Path bottomPath = StoreFile.split(this.fs, bottomDir, f, midRow, false);
226     // Make readers on top and bottom.
227     StoreFile.Reader top =
228         new StoreFile(this.fs, topPath, conf, cacheConf, BloomType.NONE,
229             NoOpDataBlockEncoder.INSTANCE).createReader();
230     StoreFile.Reader bottom = new StoreFile(this.fs, bottomPath,
231         conf, cacheConf, BloomType.NONE,
232         NoOpDataBlockEncoder.INSTANCE).createReader();
233     ByteBuffer previous = null;
234     LOG.info("Midkey: " + midKV.toString());
235     ByteBuffer bbMidkeyBytes = ByteBuffer.wrap(midkey);
236     try {
237       // Now make two HalfMapFiles and assert they can read the full backing
238       // file, one from the top and the other from the bottom.
239       // Test bottom half first.
240       // Now test reading from the top.
241       boolean first = true;
242       ByteBuffer key = null;
243       HFileScanner topScanner = top.getScanner(false, false);
244       while ((!topScanner.isSeeked() && topScanner.seekTo()) ||
245           (topScanner.isSeeked() && topScanner.next())) {
246         key = topScanner.getKey();
247 
248         if (topScanner.getReader().getComparator().compare(key.array(),
249           key.arrayOffset(), key.limit(), midkey, 0, midkey.length) < 0) {
250           fail("key=" + Bytes.toStringBinary(key) + " < midkey=" +
251               Bytes.toStringBinary(midkey));
252         }
253         if (first) {
254           first = false;
255           LOG.info("First in top: " + Bytes.toString(Bytes.toBytes(key)));
256         }
257       }
258       LOG.info("Last in top: " + Bytes.toString(Bytes.toBytes(key)));
259 
260       first = true;
261       HFileScanner bottomScanner = bottom.getScanner(false, false);
262       while ((!bottomScanner.isSeeked() && bottomScanner.seekTo()) ||
263           bottomScanner.next()) {
264         previous = bottomScanner.getKey();
265         key = bottomScanner.getKey();
266         if (first) {
267           first = false;
268           LOG.info("First in bottom: " +
269             Bytes.toString(Bytes.toBytes(previous)));
270         }
271         assertTrue(key.compareTo(bbMidkeyBytes) < 0);
272       }
273       if (previous != null) {
274         LOG.info("Last in bottom: " + Bytes.toString(Bytes.toBytes(previous)));
275       }
276       // Remove references.
277       this.fs.delete(topPath, false);
278       this.fs.delete(bottomPath, false);
279 
280       // Next test using a midkey that does not exist in the file.
281       // First, do a key that is < than first key. Ensure splits behave
282       // properly.
283       byte [] badmidkey = Bytes.toBytes("  .");
284       topPath = StoreFile.split(this.fs, topDir, f, badmidkey, true);
285       bottomPath = StoreFile.split(this.fs, bottomDir, f, badmidkey, false);
286       top = new StoreFile(this.fs, topPath, conf, cacheConf,
287           StoreFile.BloomType.NONE,
288           NoOpDataBlockEncoder.INSTANCE).createReader();
289       bottom = new StoreFile(this.fs, bottomPath, conf, cacheConf,
290           StoreFile.BloomType.NONE,
291           NoOpDataBlockEncoder.INSTANCE).createReader();
292       bottomScanner = bottom.getScanner(false, false);
293       int count = 0;
294       while ((!bottomScanner.isSeeked() && bottomScanner.seekTo()) ||
295           bottomScanner.next()) {
296         count++;
297       }
298       // When badkey is < than the bottom, should return no values.
299       assertTrue(count == 0);
300       // Now read from the top.
301       first = true;
302       topScanner = top.getScanner(false, false);
303       while ((!topScanner.isSeeked() && topScanner.seekTo()) ||
304           topScanner.next()) {
305         key = topScanner.getKey();
306         assertTrue(topScanner.getReader().getComparator().compare(key.array(),
307           key.arrayOffset(), key.limit(), badmidkey, 0, badmidkey.length) >= 0);
308         if (first) {
309           first = false;
310           KeyValue keyKV = KeyValue.createKeyValueFromKey(key);
311           LOG.info("First top when key < bottom: " + keyKV);
312           String tmp = Bytes.toString(keyKV.getRow());
313           for (int i = 0; i < tmp.length(); i++) {
314             assertTrue(tmp.charAt(i) == 'a');
315           }
316         }
317       }
318       KeyValue keyKV = KeyValue.createKeyValueFromKey(key);
319       LOG.info("Last top when key < bottom: " + keyKV);
320       String tmp = Bytes.toString(keyKV.getRow());
321       for (int i = 0; i < tmp.length(); i++) {
322         assertTrue(tmp.charAt(i) == 'z');
323       }
324       // Remove references.
325       this.fs.delete(topPath, false);
326       this.fs.delete(bottomPath, false);
327 
328       // Test when badkey is > than last key in file ('||' > 'zz').
329       badmidkey = Bytes.toBytes("|||");
330       topPath = StoreFile.split(this.fs, topDir, f, badmidkey, true);
331       bottomPath = StoreFile.split(this.fs, bottomDir, f, badmidkey, false);
332       top = new StoreFile(this.fs, topPath, conf, cacheConf,
333           StoreFile.BloomType.NONE,
334           NoOpDataBlockEncoder.INSTANCE).createReader();
335       bottom = new StoreFile(this.fs, bottomPath, conf, cacheConf,
336           StoreFile.BloomType.NONE,
337           NoOpDataBlockEncoder.INSTANCE).createReader();
338       first = true;
339       bottomScanner = bottom.getScanner(false, false);
340       while ((!bottomScanner.isSeeked() && bottomScanner.seekTo()) ||
341           bottomScanner.next()) {
342         key = bottomScanner.getKey();
343         if (first) {
344           first = false;
345           keyKV = KeyValue.createKeyValueFromKey(key);
346           LOG.info("First bottom when key > top: " + keyKV);
347           tmp = Bytes.toString(keyKV.getRow());
348           for (int i = 0; i < tmp.length(); i++) {
349             assertTrue(tmp.charAt(i) == 'a');
350           }
351         }
352       }
353       keyKV = KeyValue.createKeyValueFromKey(key);
354       LOG.info("Last bottom when key > top: " + keyKV);
355       for (int i = 0; i < tmp.length(); i++) {
356         assertTrue(Bytes.toString(keyKV.getRow()).charAt(i) == 'z');
357       }
358       count = 0;
359       topScanner = top.getScanner(false, false);
360       while ((!topScanner.isSeeked() && topScanner.seekTo()) ||
361           (topScanner.isSeeked() && topScanner.next())) {
362         count++;
363       }
364       // When badkey is < than the bottom, should return no values.
365       assertTrue(count == 0);
366     } finally {
367       if (top != null) {
368         top.close(true); // evict since we are about to delete the file
369       }
370       if (bottom != null) {
371         bottom.close(true); // evict since we are about to delete the file
372       }
373       fs.delete(f.getPath(), true);
374     }
375   }

379   private void bloomWriteRead(StoreFile.Writer writer, FileSystem fs)
380   throws Exception {
381     float err = conf.getFloat(
382         BloomFilterFactory.IO_STOREFILE_BLOOM_ERROR_RATE, 0);
383     Path f = writer.getPath();
384     long now = System.currentTimeMillis();
385     for (int i = 0; i < 2000; i += 2) {
386       String row = String.format(localFormatter, i);
387       KeyValue kv = new KeyValue(row.getBytes(), "family".getBytes(),
388         "col".getBytes(), now, "value".getBytes());
389       writer.append(kv);
390     }
391     writer.close();
392 
393     StoreFile.Reader reader = new StoreFile.Reader(fs, f, cacheConf,
394         DataBlockEncoding.NONE);
395     reader.loadFileInfo();
396     reader.loadBloomfilter();
397     StoreFileScanner scanner = reader.getStoreFileScanner(false, false);
398 
399     // check false positives rate
400     int falsePos = 0;
401     int falseNeg = 0;
402     for (int i = 0; i < 2000; i++) {
403       String row = String.format(localFormatter, i);
404       TreeSet<byte[]> columns = new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR);
405       columns.add("family:col".getBytes());
406 
407       Scan scan = new Scan(row.getBytes(),row.getBytes());
408       scan.addColumn("family".getBytes(), "family:col".getBytes());
409       boolean exists = scanner.shouldUseScanner(scan, columns, Long.MIN_VALUE);
410       if (i % 2 == 0) {
411         if (!exists) falseNeg++;
412       } else {
413         if (exists) falsePos++;
414       }
415     }
416     reader.close(true); // evict because we are about to delete the file
417     fs.delete(f, true);
418     assertEquals("False negatives: " + falseNeg, 0, falseNeg);
419     int maxFalsePos = (int) (2 * 2000 * err);
420     assertTrue("Too many false positives: " + falsePos + " (err=" + err
421         + ", expected no more than " + maxFalsePos + ")",
422         falsePos <= maxFalsePos);
423   }

502   public void testBloomTypes() throws Exception {
503     float err = (float) 0.01;
504     FileSystem fs = FileSystem.getLocal(conf);
505     conf.setFloat(BloomFilterFactory.IO_STOREFILE_BLOOM_ERROR_RATE, err);
506     conf.setBoolean(BloomFilterFactory.IO_STOREFILE_BLOOM_ENABLED, true);
507 
508     int rowCount = 50;
509     int colCount = 10;
510     int versions = 2;
511 
512     // run once using columns and once using rows
513     StoreFile.BloomType[] bt =
514       {StoreFile.BloomType.ROWCOL, StoreFile.BloomType.ROW};
515     int[] expKeys    = {rowCount*colCount, rowCount};
516     // below line deserves commentary.  it is expected bloom false positives
517     //  column = rowCount*2*colCount inserts
518     //  row-level = only rowCount*2 inserts, but failures will be magnified by
519     //              2nd for loop for every column (2*colCount)
520     float[] expErr   = {2*rowCount*colCount*err, 2*rowCount*2*colCount*err};
521 
522     for (int x : new int[]{0,1}) {
523       // write the file
524       Path f = new Path(ROOT_DIR, getName() + x);
525       StoreFile.Writer writer = new StoreFile.WriterBuilder(conf, cacheConf,
526           fs, StoreFile.DEFAULT_BLOCKSIZE_SMALL)
527               .withFilePath(f)
528               .withBloomType(bt[x])
529               .withMaxKeyCount(expKeys[x])
530               .withChecksumType(CKTYPE)
531               .withBytesPerChecksum(CKBYTES)
532               .build();
533 
534       long now = System.currentTimeMillis();
535       for (int i = 0; i < rowCount*2; i += 2) { // rows
536         for (int j = 0; j < colCount*2; j += 2) {   // column qualifiers
537           String row = String.format(localFormatter, i);
538           String col = String.format(localFormatter, j);
539           for (int k= 0; k < versions; ++k) { // versions
540             KeyValue kv = new KeyValue(row.getBytes(),
541               "family".getBytes(), ("col" + col).getBytes(),
542                 now-k, Bytes.toBytes((long)-1));
543             writer.append(kv);
544           }
545         }
546       }
547       writer.close();
548 
549       StoreFile.Reader reader = new StoreFile.Reader(fs, f, cacheConf,
550           DataBlockEncoding.NONE);
551       reader.loadFileInfo();
552       reader.loadBloomfilter();
553       StoreFileScanner scanner = reader.getStoreFileScanner(false, false);
554       assertEquals(expKeys[x], reader.generalBloomFilter.getKeyCount());
555 
556       // check false positives rate
557       int falsePos = 0;
558       int falseNeg = 0;
559       for (int i = 0; i < rowCount*2; ++i) { // rows
560         for (int j = 0; j < colCount*2; ++j) {   // column qualifiers
561           String row = String.format(localFormatter, i);
562           String col = String.format(localFormatter, j);
563           TreeSet<byte[]> columns = new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR);
564           columns.add(("col" + col).getBytes());
565 
566           Scan scan = new Scan(row.getBytes(),row.getBytes());
567           scan.addColumn("family".getBytes(), ("col"+col).getBytes());
568           boolean exists =
569               scanner.shouldUseScanner(scan, columns, Long.MIN_VALUE);
570           boolean shouldRowExist = i % 2 == 0;
571           boolean shouldColExist = j % 2 == 0;
572           shouldColExist = shouldColExist || bt[x] == StoreFile.BloomType.ROW;

577           }
578         }
579       }
580       reader.close(true); // evict because we are about to delete the file
581       fs.delete(f, true);
582       System.out.println(bt[x].toString());
583       System.out.println("  False negatives: " + falseNeg);
584       System.out.println("  False positives: " + falsePos);
585       assertEquals(0, falseNeg);
586       assertTrue(falsePos < 2*expErr[x]);
587     }

864     startEvicted += 3;

880   private StoreFile.Writer writeStoreFile(Configuration conf,
881       CacheConfig cacheConf, Path path, int numBlocks)
882   throws IOException {
883     // Let's put ~5 small KVs in each block, so let's make 5*numBlocks KVs
884     int numKVs = 5 * numBlocks;
885     List<KeyValue> kvs = new ArrayList<KeyValue>(numKVs);
886     byte [] b = Bytes.toBytes("x");
887     int totalSize = 0;
888     for (int i=numKVs;i>0;i--) {
889       KeyValue kv = new KeyValue(b, b, b, i, b);
890       kvs.add(kv);
891       // kv has memstoreTS 0, which takes 1 byte to store.
892       totalSize += kv.getLength() + 1;
893     }
894     int blockSize = totalSize / numBlocks;
895     StoreFile.Writer writer = new StoreFile.WriterBuilder(conf, cacheConf, fs,
896         blockSize)
897             .withFilePath(path)
898             .withMaxKeyCount(2000)
899             .withChecksumType(CKTYPE)
900             .withBytesPerChecksum(CKBYTES)
901             .build();
902     // We'll write N-1 KVs to ensure we don't write an extra block
903     kvs.remove(kvs.size()-1);
904     for (KeyValue kv : kvs) {
905       writer.append(kv);
906     }
907     writer.appendMetadata(0, false);
908     writer.close();
909     return writer;
910   }
