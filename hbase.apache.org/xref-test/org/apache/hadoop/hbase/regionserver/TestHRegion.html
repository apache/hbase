447     private final AtomicBoolean done;
448     private final AtomicInteger count;

453       super("getter." + i);
454       this.g = new Get(r);
455       this.done = d;

461       while (!this.done.get()) {

465         } catch (Exception e) {
466           this.e = e;
467           break;
468         }
469       }
470     }

540   private int getNumberOfRows(String keyPrefix, String value, HRegion r) throws Exception {

552       results.clear();

557   private InternalScanner buildScanner(String keyPrefix, String value, HRegion r)
558   throws IOException {
559     // Defaults FilterList.Operator.MUST_PASS_ALL.
560     FilterList allFilters = new FilterList();
561     allFilters.addFilter(new PrefixFilter(Bytes.toBytes(keyPrefix)));
562     // Only return rows where this column value exists in the row.
563     SingleColumnValueFilter filter =
564       new SingleColumnValueFilter(Bytes.toBytes("trans-tags"),
565         Bytes.toBytes("qual2"), CompareOp.EQUAL, Bytes.toBytes(value));
566     filter.setFilterIfMissing(true);
567     allFilters.addFilter(filter);
568     Scan scan = new Scan();
569     scan.addFamily(Bytes.toBytes("trans-blob"));
570     scan.addFamily(Bytes.toBytes("trans-type"));
571     scan.addFamily(Bytes.toBytes("trans-date"));
572     scan.addFamily(Bytes.toBytes("trans-tags"));
573     scan.addFamily(Bytes.toBytes("trans-group"));
574     scan.setFilter(allFilters);
575     return r.getScanner(scan);
576   }

1772    * @param parent Region to split.
1773    * @param midkey Key to split around.
1774    * @return The Regions we created.
1775    * @throws IOException
1776    */
1777   HRegion [] splitRegion(final HRegion parent, final byte [] midkey)
1778   throws IOException {
1779     PairOfSameType<HRegion> result = null;
1780     SplitTransaction st = new SplitTransaction(parent, midkey);
1781     // If prepare does not return true, for some reason -- logged inside in
1782     // the prepare call -- we are not ready to split just now.  Just return.
1783     if (!st.prepare()) return null;
1784     try {
1785       result = st.execute(null, null);
1786     } catch (IOException ioe) {
1787       try {
1788         LOG.info("Running rollback of failed split of " +
1789           parent.getRegionNameAsString() + "; " + ioe.getMessage());
1790         st.rollback(null, null);
1791         LOG.info("Successful rollback of failed split of " +
1792           parent.getRegionNameAsString());
1793         return null;
1794       } catch (RuntimeException e) {
1795         // If failed rollback, kill this server to avoid having a hole in table.
1796         LOG.info("Failed rollback of failed split of " +
1797           parent.getRegionNameAsString() + " -- aborting server", e);
1798       }
1799     }
