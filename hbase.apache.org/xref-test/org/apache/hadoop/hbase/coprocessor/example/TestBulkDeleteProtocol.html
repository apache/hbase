123   private long invokeBulkDeleteProtocol(byte[] tableName, final Scan scan, final int rowBatchSize,
124       final DeleteType deleteType, final Long timeStamp) throws Throwable {
125     HTable ht = new HTable(TEST_UTIL.getConfiguration(), tableName);
126     long noOfDeletedRows = 0L;
127     Batch.Call<BulkDeleteService, BulkDeleteResponse> callable = 
128       new Batch.Call<BulkDeleteService, BulkDeleteResponse>() {
129       ServerRpcController controller = new ServerRpcController();
130       BlockingRpcCallback<BulkDeleteResponse> rpcCallback = 
131         new BlockingRpcCallback<BulkDeleteResponse>();
132 
133       public BulkDeleteResponse call(BulkDeleteService service) throws IOException {
134         Builder builder = BulkDeleteRequest.newBuilder();
135         builder.setScan(ProtobufUtil.toScan(scan));
136         builder.setDeleteType(deleteType);
137         builder.setRowBatchSize(rowBatchSize);
138         if (timeStamp != null) {
139           builder.setTimestamp(timeStamp);
140         }
141         service.delete(controller, builder.build(), rpcCallback);
142         return rpcCallback.get();
143       }
144     };
145     Map<byte[], BulkDeleteResponse> result = ht.coprocessorService(BulkDeleteService.class, scan
146         .getStartRow(), scan.getStopRow(), callable);
147     for (BulkDeleteResponse response : result.values()) {
148       noOfDeletedRows += response.getRowsDeleted();
149     }
150     return noOfDeletedRows;
151   }

156     HTable ht = createTable(tableName);

161       puts.add(createPut(rowkey, value));

164     Scan scan = new Scan();
165     FilterList fl = new FilterList(Operator.MUST_PASS_ALL);

170     scan.setFilter(fl);
171     // Deleting all the rows where cf1:c1=v1
172     long noOfRowsDeleted = invokeBulkDeleteProtocol(tableName, scan, 500, DeleteType.ROW, null);
173     assertEquals(10, noOfRowsDeleted);
174

178     }

252       value = "v2".getBytes();
253       put.add(FAMILY1, QUALIFIER1, value);

266     int rows = 0;

268     scan.setMaxVersions();
269     for (Result result : ht.getScanner(scan)) {

274 
275       column = result.getColumn(FAMILY1, QUALIFIER2);
276       assertEquals(1, column.size());

278 
279       column = result.getColumn(FAMILY1, QUALIFIER3);

353       put.add(FAMILY1, QUALIFIER2, 2000L, value);
354       put.add(FAMILY1, QUALIFIER3, 2000L, value);
355       // Latest version values

357       put.add(FAMILY1, QUALIFIER1, value);

359       put.add(FAMILY1, QUALIFIER3, value);
360       puts.add(put);
361     }
362     ht.put(puts);
363 
364     // Delete all the versions of columns cf1:c1 and cf1:c2 falling with the time range

366     final Scan scan = new Scan();
367     scan.addColumn(FAMILY1, QUALIFIER1);

370     scan.setMaxVersions();
371 
372     long noOfDeletedRows = 0L;
373     long noOfVersionsDeleted = 0L;
374     Batch.Call<BulkDeleteService, BulkDeleteResponse> callable =

376       ServerRpcController controller = new ServerRpcController();
377       BlockingRpcCallback<BulkDeleteResponse> rpcCallback =

380       public BulkDeleteResponse call(BulkDeleteService service) throws IOException {
381         Builder builder = BulkDeleteRequest.newBuilder();
382         builder.setScan(ProtobufUtil.toScan(scan));
383         builder.setDeleteType(DeleteType.VERSION);
384         builder.setRowBatchSize(500);
385         service.delete(controller, builder.build(), rpcCallback);
386         return rpcCallback.get();
387       }
388     };
389     Map<byte[], BulkDeleteResponse> result = ht.coprocessorService(BulkDeleteService.class, scan
390         .getStartRow(), scan.getStopRow(), callable);
391     for (BulkDeleteResponse response : result.values()) {
392       noOfDeletedRows += response.getRowsDeleted();
393       noOfVersionsDeleted += response.getVersionsDeleted();
394     }
395     assertEquals(100, noOfDeletedRows);
396     assertEquals(400, noOfVersionsDeleted);
397 
398     int rows = 0;
399     Scan scan1 = new Scan();
400     scan1.setMaxVersions();
401     for (Result res : ht.getScanner(scan1)) {
402       assertEquals(3, res.getFamilyMap(FAMILY1).size());
403       List<KeyValue> column = res.getColumn(FAMILY1, QUALIFIER1);
404       assertEquals(2, column.size());
405       assertTrue(Bytes.equals("v4".getBytes(), column.get(0).getValue()));
406       assertTrue(Bytes.equals("v3".getBytes(), column.get(1).getValue()));
407       column = res.getColumn(FAMILY1, QUALIFIER2);
408       assertEquals(2, column.size());
409       assertTrue(Bytes.equals("v4".getBytes(), column.get(0).getValue()));
410       assertTrue(Bytes.equals("v3".getBytes(), column.get(1).getValue()));
411       assertEquals(4, res.getColumn(FAMILY1, QUALIFIER3).size());
412       rows++;
413     }
