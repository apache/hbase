300   /**
301    * Add content to region <code>r</code> on the passed column
302    * <code>column</code>.
303    * Adds data of the from 'aaa', 'aab', etc where key and value are the same.
304    * @param updater  An instance of {@link Incommon}.
305    * @param column
306    * @param startKeyBytes Where to start the rows inserted
307    * @param endKey Where to stop inserting rows.
308    * @param ts Timestamp to write the content with.
309    * @return count of what we added.
310    * @throws IOException
311    */
312   protected static long addContent(final Incommon updater,
313                                    final String columnFamily,
314                                    final String column,
315       final byte [] startKeyBytes, final byte [] endKey, final long ts)
316   throws IOException {
317     long count = 0;
318     // Add rows of three characters.  The first character starts with the
319     // 'a' character and runs up to 'z'.  Per first character, we run the
320     // second character over same range.  And same for the third so rows
321     // (and values) look like this: 'aaa', 'aab', 'aac', etc.
322     char secondCharStart = (char)startKeyBytes[1];
323     char thirdCharStart = (char)startKeyBytes[2];
324     EXIT: for (char c = (char)startKeyBytes[0]; c <= LAST_CHAR; c++) {
325       for (char d = secondCharStart; d <= LAST_CHAR; d++) {
326         for (char e = thirdCharStart; e <= LAST_CHAR; e++) {
327           byte [] t = new byte [] {(byte)c, (byte)d, (byte)e};
328           if (endKey != null && endKey.length > 0
329               && Bytes.compareTo(endKey, t) <= 0) {
330             break EXIT;
331           }
332           try {
333             Put put;
334             if(ts != -1) {
335               put = new Put(t, ts, null);
336             } else {
337               put = new Put(t);
338             }
339             try {
340               StringBuilder sb = new StringBuilder();
341               if (column != null && column.contains(":")) {
342                 sb.append(column);
343               } else {
344                 if (columnFamily != null) {
345                   sb.append(columnFamily);
346                   if (!columnFamily.endsWith(":")) {
347                     sb.append(":");
348                   }
349                   if (column != null) {
350                     sb.append(column);
351                   }
352                 }
353               }
354               byte[][] split =
355                 KeyValue.parseColumn(Bytes.toBytes(sb.toString()));
356               if(split.length == 1) {
357                 put.add(split[0], new byte[0], t);
358               } else {
359                 put.add(split[0], split[1], t);
360               }
361               updater.put(put);
362               count++;
363             } catch (RuntimeException ex) {
364               ex.printStackTrace();
365               throw ex;
366             } catch (IOException ex) {
367               ex.printStackTrace();
368               throw ex;
369             }
370           } catch (RuntimeException ex) {
371             ex.printStackTrace();
372             throw ex;
373           } catch (IOException ex) {
374             ex.printStackTrace();
375             throw ex;
376           }
377         }
378         // Set start character back to FIRST_CHAR after we've done first loop.
379         thirdCharStart = FIRST_CHAR;
380       }
381       secondCharStart = FIRST_CHAR;
382     }
383     return count;
384   }

610     throws IOException {

635   /**
636    * Common method to close down a MiniDFSCluster and the associated file system
637    *
638    * @param cluster
639    */
640   public static void shutdownDfs(MiniDFSCluster cluster) {
641     if (cluster != null) {
642       LOG.info("Shutting down Mini DFS ");
643       try {
644         cluster.shutdown();
645       } catch (Exception e) {
646         /// Can get a java.lang.reflect.UndeclaredThrowableException thrown
647         // here because of an InterruptedException. Don't let exceptions in
648         // here be cause of test failure.
649       }
650       try {
651         FileSystem fs = cluster.getFileSystem();
652         if (fs != null) {
653           LOG.info("Shutting down FileSystem");
654           fs.close();
655         }
656         FileSystem.closeAll();
657       } catch (IOException e) {
658         LOG.error("error closing file system", e);
659       }
660     }
661   }
