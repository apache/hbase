22  import org.apache.commons.logging.Log;
23  import org.apache.commons.logging.LogFactory;
24  import org.apache.hadoop.classification.InterfaceAudience;
25  import org.apache.hadoop.hbase.HRegionInfo;
26  import org.apache.hadoop.hbase.Server;
27  import org.apache.hadoop.hbase.ServerName;
28  import org.apache.hadoop.hbase.executor.EventHandler;
29  import org.apache.hadoop.hbase.master.AssignmentManager;
30  import org.apache.hadoop.hbase.master.RegionState;
31  import org.apache.hadoop.hbase.zookeeper.ZKAssign;
32  import org.apache.zookeeper.KeeperException;

34  /**
35   * Handles OPENED region event on Master.
36   */
37  @InterfaceAudience.Private
38  public class OpenedRegionHandler extends EventHandler implements TotesHRegionInfo {
39    private static final Log LOG = LogFactory.getLog(OpenedRegionHandler.class);
40    private final AssignmentManager assignmentManager;
41    private final HRegionInfo regionInfo;
42    private final ServerName sn;
43    private final OpenedPriority priority;
44    private final int expectedVersion;
45  
46    private enum OpenedPriority {
47      ROOT (1),
48      META (2),
49      USER (3);
50  
51      private final int value;
52      OpenedPriority(int value) {
53        this.value = value;
54      }
55      public int getValue() {
56        return value;
57      }
58    };
59  
60    public OpenedRegionHandler(Server server,
61        AssignmentManager assignmentManager, HRegionInfo regionInfo,
62        ServerName sn, int expectedVersion) {
63      super(server, EventType.RS_ZK_REGION_OPENED);
64      this.assignmentManager = assignmentManager;
65      this.regionInfo = regionInfo;
66      this.sn = sn;
67      this.expectedVersion = expectedVersion;
68      if(regionInfo.isRootRegion()) {
69        priority = OpenedPriority.ROOT;
70      } else if(regionInfo.isMetaRegion()) {
71        priority = OpenedPriority.META;
72      } else {
73        priority = OpenedPriority.USER;
74      }
75    }
76  
77    @Override
78    public int getPriority() {
79      return priority.getValue();
80    }
81  
82    @Override
83    public HRegionInfo getHRegionInfo() {
84      return this.regionInfo;
85    }
86    
87    @Override
88    public String toString() {
89      String name = "UnknownServerName";
90      if(server != null && server.getServerName() != null) {
91        name = server.getServerName().toString();
92      }
93      return getClass().getSimpleName() + "-" + name + "-" + getSeqid();
94    }
95  
96    @Override
97    public void process() {
98      // Code to defend against case where we get SPLIT before region open
99      // processing completes; temporary till we make SPLITs go via zk -- 0.92.
100     RegionState regionState = this.assignmentManager.getRegionStates()
101       .getRegionTransitionState(regionInfo.getEncodedName());
102     boolean openedNodeDeleted = false;
103     if (regionState != null && regionState.isOpened()) {
104       openedNodeDeleted = deleteOpenedNode(expectedVersion);
105       if (!openedNodeDeleted) {
106         LOG.error("The znode of region " + regionInfo.getRegionNameAsString()
107             + " could not be deleted.");
108       }
109     } else {
110       LOG.warn("Skipping the onlining of " + regionInfo.getRegionNameAsString() +
111         " because regions is NOT in RIT -- presuming this is because it SPLIT");
112     }
113     if (!openedNodeDeleted) {
114       if (this.assignmentManager.getZKTable().isDisablingOrDisabledTable(
115           regionInfo.getTableNameAsString())) {
116         debugLog(regionInfo, "Opened region "
117             + regionInfo.getRegionNameAsString() + " but "
118             + "this table is disabled, triggering close of region");
119         assignmentManager.unassign(regionInfo);
120       }
121     }
122   }
123 
124   private boolean deleteOpenedNode(int expectedVersion) {
125     debugLog(regionInfo, "Handling OPENED event for " +
126       this.regionInfo.getRegionNameAsString() + " from " + this.sn.toString() +
127       "; deleting unassigned node");
128     try {
129       // delete the opened znode only if the version matches.
130       return ZKAssign.deleteNode(server.getZooKeeper(),
131           regionInfo.getEncodedName(), EventType.RS_ZK_REGION_OPENED, expectedVersion);
132     } catch(KeeperException.NoNodeException e){
133       // Getting no node exception here means that already the region has been opened.
134       LOG.warn("The znode of the region " + regionInfo.getRegionNameAsString() +
135         " would have already been deleted");
136       return false;
137     } catch (KeeperException e) {
138       server.abort("Error deleting OPENED node in ZK (" +
139         regionInfo.getRegionNameAsString() + ")", e);
140     }
141     return false;
142   }
143 
144   private void debugLog(HRegionInfo region, String string) {
145     if (region.isMetaTable()) {
146       LOG.info(string);
147     } else {
148       LOG.debug(string);
149     }
150   }
