20  package org.apache.hadoop.hbase.zookeeper;
21  
22  import java.util.Properties;
23  import java.util.Map.Entry;
24  
25  import org.apache.hadoop.conf.Configuration;
26  import org.apache.hadoop.hbase.HBaseConfiguration;
27  
28  /**
29   * Tool for reading a ZooKeeper server from HBase XML configuration producing
30   * the '-server host:port' argument to pass ZooKeeperMain.  This program
31   * emits either '-server HOST:PORT" where HOST is one of the zk ensemble
32   * members plus zk client port OR it emits '' if no zk servers found (Yes,
33   * it emits '-server' too).
34   */
35  public class ZooKeeperMainServerArg {
36    public String parse(final Configuration c) {
37      // Note that we do not simply grab the property
38      // HConstants.ZOOKEEPER_QUORUM from the HBaseConfiguration because the
39      // user may be using a zoo.cfg file.
40      Properties zkProps = ZKConfig.makeZKProps(c);
41      String host = null;
42      String clientPort = null;
43      for (Entry<Object, Object> entry: zkProps.entrySet()) {
44        String key = entry.getKey().toString().trim();
45        String value = entry.getValue().toString().trim();
46        if (key.startsWith("server.") && host == null) {
47          String[] parts = value.split(":");
48          host = parts[0];
49        } else if (key.endsWith("clientPort")) {
50          clientPort = value;
51        }
52        if (host != null && clientPort != null) break;
53      }
54      return host != null && clientPort != null? host + ":" + clientPort: null;
55    }
56  
57    /**
58     * Run the tool.
59     * @param args Command line arguments. First arg is path to zookeepers file.
60     */
61    public static void main(String args[]) {
62      Configuration conf = HBaseConfiguration.create();
63      String hostport = new ZooKeeperMainServerArg().parse(conf);
64      System.out.println((hostport == null || hostport.length() == 0)? "":
65        "-server " + hostport);
66    }
67  }
