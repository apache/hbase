130     @Override
131     public void run() {
132       try {
133         this.user.runAs(new PrivilegedAction<Object>(){
134           public Object run() {
135             runRegionServer();
136             return null;
137           }
138         });
139       } catch (Throwable t) {
140         LOG.error("Exception in run", t);
141       } finally {
142         // Run this on the way out.
143         if (this.shutdownThread != null) {
144           this.shutdownThread.start();
145           Threads.shutdown(this.shutdownThread, 30000);
146         }
147       }
148     }

173   /**
174    * Alternate shutdown hook.
175    * Just shuts down the passed fs, not all as default filesystem hook does.
176    */
177   static class SingleFileSystemShutdownThread extends Thread {
178     private final FileSystem fs;
179     SingleFileSystemShutdownThread(final FileSystem fs) {
180       super("Shutdown of " + fs);
181       this.fs = fs;
182     }
183     @Override
184     public void run() {
185       try {
186         LOG.info("Hook closing fs=" + this.fs);
187         this.fs.close();
188       } catch (NullPointerException npe) {
189         LOG.debug("Need to fix these: " + npe.toString());
190       } catch (IOException e) {
191         LOG.warn("Running hook", e);
192       }
193     }
194   }

196   private void init(final int nMasterNodes, final int nRegionNodes,
197                  Class<? extends HMaster> masterClass,
198                  Class<? extends MiniHBaseCluster.MiniHBaseClusterRegionServer> regionserverClass)
199   throws IOException, InterruptedException {
200     try {
201       if (masterClass == null){
202         masterClass =  HMaster.class;
203       }
204       if (regionserverClass == null){
205         regionserverClass = MiniHBaseCluster.MiniHBaseClusterRegionServer.class;
206       }
207 
208       // start up a LocalHBaseCluster
209       hbaseCluster = new LocalHBaseCluster(conf, nMasterNodes, 0,
210           masterClass, regionserverClass);
211 
212       // manually add the regionservers as other users
213       for (int i=0; i<nRegionNodes; i++) {
214         Configuration rsConf = HBaseConfiguration.create(conf);
215         User user = HBaseTestingUtility.getDifferentUser(rsConf,
216             ".hfs."+index++);
217         hbaseCluster.addRegionServer(rsConf, i, user);
218       }
219 
220       hbaseCluster.startup();
221     } catch (IOException e) {
222       shutdown();
223       throw e;
224     } catch (Throwable t) {
225       LOG.error("Error starting cluster", t);
226       shutdown();
227       throw new IOException("Shutting down", t);
228     }
229   }

450   /**
451    * Blocks until there is an active master and that master has completed
452    * initialization.
453    *
454    * @return true if an active master becomes available.  false if there are no
455    *         masters left.
456    * @throws InterruptedException
457    */
458   public boolean waitForActiveAndReadyMaster(long timeout) throws IOException {
459     List<JVMClusterUtil.MasterThread> mts;
460     long start = System.currentTimeMillis();
461     while (!(mts = getMasterThreads()).isEmpty()
462         && (System.currentTimeMillis() - start) < timeout) {
463       for (JVMClusterUtil.MasterThread mt : mts) {
464         if (mt.getMaster().isActiveMaster() && mt.getMaster().isInitialized()) {
465           return true;
466         }
467       }
468 
469       Threads.sleep(100);
470     }
471     return false;
472   }
