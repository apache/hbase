53    @Test
54    public void testBackgroundEvictionThread() throws Exception {
55      long maxSize = 100000;
56      long blockSize = calculateBlockSizeDefault(maxSize, 9); // room for 9, will evict
57  
58      LruBlockCache cache = new LruBlockCache(maxSize,blockSize);
59  
60      CachedItem [] blocks = generateFixedBlocks(10, blockSize, "block");
61  
62      EvictionThread evictionThread = cache.getEvictionThread();
63      assertTrue(evictionThread != null);
64  
65      // Make sure eviction thread has entered run method
66      while (!evictionThread.isEnteringRun()) {
67        Thread.sleep(1);
68      }
69      
70  
71      // Add all the blocks
72      for (CachedItem block : blocks) {
73        cache.cacheBlock(block.cacheKey, block);
74      }
75  
76      // Let the eviction run
77      int n = 0;
78      while(cache.getEvictionCount() == 0) {
79        Thread.sleep(200);
80        assertTrue(n++ < 10);
81      }
82      System.out.println("Background Evictions run: " + cache.getEvictionCount());
83  
84      // A single eviction run should have occurred
85      assertEquals(cache.getEvictionCount(), 1);
86    }

190 
191     long maxSize = 100000;
192     long blockSize = calculateBlockSizeDefault(maxSize, 10);
193 
194     LruBlockCache cache = new LruBlockCache(maxSize,blockSize,false);
195 
196     CachedItem [] singleBlocks = generateFixedBlocks(5, 10000, "single");
197     CachedItem [] multiBlocks = generateFixedBlocks(5, 10000, "multi");
198 
199     long expectedCacheSize = cache.heapSize();
200 
201     // Add and get the multi blocks
202     for (CachedItem block : multiBlocks) {
203       cache.cacheBlock(block.cacheKey, block);
204       expectedCacheSize += block.cacheBlockHeapSize();
205       assertEquals(cache.getBlock(block.cacheKey, true, false), block);
206     }
207 
208     // Add the single blocks (no get)
209     for (CachedItem block : singleBlocks) {
210       cache.cacheBlock(block.cacheKey, block);
211       expectedCacheSize += block.heapSize();
212     }
213 
214     // A single eviction run should have occurred
215     assertEquals(cache.getEvictionCount(), 1);
216 
217     // We expect two entries evicted
218     assertEquals(cache.getEvictedCount(), 2);
219 
220     // Our expected size overruns acceptable limit
221     assertTrue(expectedCacheSize >
222       (maxSize * LruBlockCache.DEFAULT_ACCEPTABLE_FACTOR));
223 
224     // But the cache did not grow beyond max
225     assertTrue(cache.heapSize() <= maxSize);
226 
227     // And is now below the acceptable limit
228     assertTrue(cache.heapSize() <=
229         (maxSize * LruBlockCache.DEFAULT_ACCEPTABLE_FACTOR));
230 
231     // We expect fairness across the two priorities.
232     // This test makes multi go barely over its limit, in-memory
233     // empty, and the rest in single.  Two single evictions and
234     // one multi eviction expected.
235     assertTrue(cache.getBlock(singleBlocks[0].cacheKey, true, false) == null);
236     assertTrue(cache.getBlock(multiBlocks[0].cacheKey, true, false) == null);
237 
238     // And all others to be cached
239     for(int i=1;i<4;i++) {
240       assertEquals(cache.getBlock(singleBlocks[i].cacheKey, true, false),
241           singleBlocks[i]);
242       assertEquals(cache.getBlock(multiBlocks[i].cacheKey, true, false),
243           multiBlocks[i]);
244     }

247   @Test
248   public void testCacheEvictionThreePriorities() throws Exception {
249 
250     long maxSize = 100000;
251     long blockSize = calculateBlockSize(maxSize, 10);
252 
253     LruBlockCache cache = new LruBlockCache(maxSize, blockSize, false,
254         (int)Math.ceil(1.2*maxSize/blockSize),
255         LruBlockCache.DEFAULT_LOAD_FACTOR,
256         LruBlockCache.DEFAULT_CONCURRENCY_LEVEL,
257         0.98f, // min
258         0.99f, // acceptable
259         0.33f, // single
260         0.33f, // multi
261         0.34f);// memory
262 
263 
264     CachedItem [] singleBlocks = generateFixedBlocks(5, blockSize, "single");
265     CachedItem [] multiBlocks = generateFixedBlocks(5, blockSize, "multi");
266     CachedItem [] memoryBlocks = generateFixedBlocks(5, blockSize, "memory");
267 
268     long expectedCacheSize = cache.heapSize();
269 
270     // Add 3 blocks from each priority
271     for(int i=0;i<3;i++) {
272 
273       // Just add single blocks
274       cache.cacheBlock(singleBlocks[i].cacheKey, singleBlocks[i]);
275       expectedCacheSize += singleBlocks[i].cacheBlockHeapSize();
276 
277       // Add and get multi blocks
278       cache.cacheBlock(multiBlocks[i].cacheKey, multiBlocks[i]);
279       expectedCacheSize += multiBlocks[i].cacheBlockHeapSize();
280       cache.getBlock(multiBlocks[i].cacheKey, true, false);
281 
282       // Add memory blocks as such
283       cache.cacheBlock(memoryBlocks[i].cacheKey, memoryBlocks[i], true);
284       expectedCacheSize += memoryBlocks[i].cacheBlockHeapSize();
285 
286     }
287 
288     // Do not expect any evictions yet
289     assertEquals(0, cache.getEvictionCount());
290 
291     // Verify cache size
292     assertEquals(expectedCacheSize, cache.heapSize());
293 
294     // Insert a single block, oldest single should be evicted
295     cache.cacheBlock(singleBlocks[3].cacheKey, singleBlocks[3]);
296 
297     // Single eviction, one thing evicted
298     assertEquals(1, cache.getEvictionCount());
299     assertEquals(1, cache.getEvictedCount());
300 
301     // Verify oldest single block is the one evicted
302     assertEquals(null, cache.getBlock(singleBlocks[0].cacheKey, true, false));
303 
304     // Change the oldest remaining single block to a multi
305     cache.getBlock(singleBlocks[1].cacheKey, true, false);
306 
307     // Insert another single block
308     cache.cacheBlock(singleBlocks[4].cacheKey, singleBlocks[4]);
309 
310     // Two evictions, two evicted.
311     assertEquals(2, cache.getEvictionCount());
312     assertEquals(2, cache.getEvictedCount());
313 
314     // Oldest multi block should be evicted now
315     assertEquals(null, cache.getBlock(multiBlocks[0].cacheKey, true, false));
316 
317     // Insert another memory block
318     cache.cacheBlock(memoryBlocks[3].cacheKey, memoryBlocks[3], true);
319 
320     // Three evictions, three evicted.
321     assertEquals(3, cache.getEvictionCount());
322     assertEquals(3, cache.getEvictedCount());
323 
324     // Oldest memory block should be evicted now
325     assertEquals(null, cache.getBlock(memoryBlocks[0].cacheKey, true, false));
326 
327     // Add a block that is twice as big (should force two evictions)
328     CachedItem [] bigBlocks = generateFixedBlocks(3, blockSize*3, "big");
329     cache.cacheBlock(bigBlocks[0].cacheKey, bigBlocks[0]);
330 
331     // Four evictions, six evicted (inserted block 3X size, expect +3 evicted)
332     assertEquals(4, cache.getEvictionCount());
333     assertEquals(6, cache.getEvictedCount());
334 
335     // Expect three remaining singles to be evicted
336     assertEquals(null, cache.getBlock(singleBlocks[2].cacheKey, true, false));
337     assertEquals(null, cache.getBlock(singleBlocks[3].cacheKey, true, false));
338     assertEquals(null, cache.getBlock(singleBlocks[4].cacheKey, true, false));
339 
340     // Make the big block a multi block
341     cache.getBlock(bigBlocks[0].cacheKey, true, false);
342 
343     // Cache another single big block
344     cache.cacheBlock(bigBlocks[1].cacheKey, bigBlocks[1]);
345 
346     // Five evictions, nine evicted (3 new)
347     assertEquals(5, cache.getEvictionCount());
348     assertEquals(9, cache.getEvictedCount());
349 
350     // Expect three remaining multis to be evicted
351     assertEquals(null, cache.getBlock(singleBlocks[1].cacheKey, true, false));
352     assertEquals(null, cache.getBlock(multiBlocks[1].cacheKey, true, false));
353     assertEquals(null, cache.getBlock(multiBlocks[2].cacheKey, true, false));
354 
355     // Cache a big memory block
356     cache.cacheBlock(bigBlocks[2].cacheKey, bigBlocks[2], true);
357 
358     // Six evictions, twelve evicted (3 new)
359     assertEquals(6, cache.getEvictionCount());
360     assertEquals(12, cache.getEvictedCount());
361 
362     // Expect three remaining in-memory to be evicted
363     assertEquals(null, cache.getBlock(memoryBlocks[1].cacheKey, true, false));
364     assertEquals(null, cache.getBlock(memoryBlocks[2].cacheKey, true, false));
365     assertEquals(null, cache.getBlock(memoryBlocks[3].cacheKey, true, false));
366 
367 
368   }

370   // test scan resistance
371   @Test
372   public void testScanResistance() throws Exception {
373 
374     long maxSize = 100000;
375     long blockSize = calculateBlockSize(maxSize, 10);
376 
377     LruBlockCache cache = new LruBlockCache(maxSize, blockSize, false,
378         (int)Math.ceil(1.2*maxSize/blockSize),
379         LruBlockCache.DEFAULT_LOAD_FACTOR,
380         LruBlockCache.DEFAULT_CONCURRENCY_LEVEL,
381         0.66f, // min
382         0.99f, // acceptable
383         0.33f, // single
384         0.33f, // multi
385         0.34f);// memory
386 
387     CachedItem [] singleBlocks = generateFixedBlocks(20, blockSize, "single");
388     CachedItem [] multiBlocks = generateFixedBlocks(5, blockSize, "multi");
389 
390     // Add 5 multi blocks
391     for (CachedItem block : multiBlocks) {
392       cache.cacheBlock(block.cacheKey, block);
393       cache.getBlock(block.cacheKey, true, false);
394     }
395 
396     // Add 5 single blocks
397     for(int i=0;i<5;i++) {
398       cache.cacheBlock(singleBlocks[i].cacheKey, singleBlocks[i]);
399     }
400 
401     // An eviction ran
402     assertEquals(1, cache.getEvictionCount());
403 
404     // To drop down to 2/3 capacity, we'll need to evict 4 blocks
405     assertEquals(4, cache.getEvictedCount());
406 
407     // Should have been taken off equally from single and multi
408     assertEquals(null, cache.getBlock(singleBlocks[0].cacheKey, true, false));
409     assertEquals(null, cache.getBlock(singleBlocks[1].cacheKey, true, false));
410     assertEquals(null, cache.getBlock(multiBlocks[0].cacheKey, true, false));
411     assertEquals(null, cache.getBlock(multiBlocks[1].cacheKey, true, false));
412 
413     // Let's keep "scanning" by adding single blocks.  From here on we only
414     // expect evictions from the single bucket.
415 
416     // Every time we reach 10 total blocks (every 4 inserts) we get 4 single
417     // blocks evicted.  Inserting 13 blocks should yield 3 more evictions and
418     // 12 more evicted.
419 
420     for(int i=5;i<18;i++) {
421       cache.cacheBlock(singleBlocks[i].cacheKey, singleBlocks[i]);
422     }
423 
424     // 4 total evictions, 16 total evicted
425     assertEquals(4, cache.getEvictionCount());
426     assertEquals(16, cache.getEvictedCount());
427 
428     // Should now have 7 total blocks
429     assertEquals(7, cache.size());
430 
431   }

597   private CachedItem [] generateRandomBlocks(int numBlocks, long maxSize) {
598     CachedItem [] blocks = new CachedItem[numBlocks];
599     Random r = new Random();
600     for(int i=0;i<numBlocks;i++) {

606   private long calculateBlockSize(long maxSize, int numBlocks) {
607     long roughBlockSize = maxSize / numBlocks;
608     int numEntries = (int)Math.ceil((1.2)*maxSize/roughBlockSize);
609     long totalOverhead = LruBlockCache.CACHE_FIXED_OVERHEAD +
610         ClassSize.CONCURRENT_HASHMAP +
611         (numEntries * ClassSize.CONCURRENT_HASHMAP_ENTRY) +
612         (LruBlockCache.DEFAULT_CONCURRENCY_LEVEL * ClassSize.CONCURRENT_HASHMAP_SEGMENT);
613     long negateBlockSize = (long)(totalOverhead/numEntries);
614     negateBlockSize += CachedBlock.PER_BLOCK_OVERHEAD;
615     return ClassSize.align((long)Math.floor((roughBlockSize - negateBlockSize)*0.99f));
616   }
617 
618   private long calculateBlockSizeDefault(long maxSize, int numBlocks) {
619     long roughBlockSize = maxSize / numBlocks;
620     int numEntries = (int)Math.ceil((1.2)*maxSize/roughBlockSize);
621     long totalOverhead = LruBlockCache.CACHE_FIXED_OVERHEAD +
622         ClassSize.CONCURRENT_HASHMAP +
623         (numEntries * ClassSize.CONCURRENT_HASHMAP_ENTRY) +
624         (LruBlockCache.DEFAULT_CONCURRENCY_LEVEL * ClassSize.CONCURRENT_HASHMAP_SEGMENT);
625     long negateBlockSize = totalOverhead / numEntries;
626     negateBlockSize += CachedBlock.PER_BLOCK_OVERHEAD;
627     return ClassSize.align((long)Math.floor((roughBlockSize - negateBlockSize)*
628         LruBlockCache.DEFAULT_ACCEPTABLE_FACTOR));
629   }

631   private static class CachedItem implements Cacheable {
632     BlockCacheKey cacheKey;
633     int size;
634 
635     CachedItem(String blockName, int size) {
636       this.cacheKey = new BlockCacheKey(blockName, 0);
637       this.size = size;
638     }
639 
640     /** The size of this item reported to the block cache layer */
641     @Override
