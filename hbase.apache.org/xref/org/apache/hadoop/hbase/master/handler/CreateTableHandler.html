57  /**
58   * Handler to create a table.
59   */
60  @InterfaceAudience.Private
61  public class CreateTableHandler extends EventHandler {
62    private static final Log LOG = LogFactory.getLog(CreateTableHandler.class);
63    private MasterFileSystem fileSystemManager;
64    private final HTableDescriptor hTableDescriptor;
65    private Configuration conf;
66    private final AssignmentManager assignmentManager;
67    private final CatalogTracker catalogTracker;
68    private final HRegionInfo [] newRegions;
69  
70    public CreateTableHandler(Server server, MasterFileSystem fileSystemManager,
71        HTableDescriptor hTableDescriptor, Configuration conf, HRegionInfo [] newRegions,
72        CatalogTracker catalogTracker, AssignmentManager assignmentManager)
73            throws NotAllMetaRegionsOnlineException, TableExistsException, IOException {
74      super(server, EventType.C_M_CREATE_TABLE);
75  
76      this.fileSystemManager = fileSystemManager;
77      this.hTableDescriptor = hTableDescriptor;
78      this.conf = conf;
79      this.newRegions = newRegions;
80      this.catalogTracker = catalogTracker;
81      this.assignmentManager = assignmentManager;
82  
83      int timeout = conf.getInt("hbase.client.catalog.timeout", 10000);
84      // Need META availability to create a table
85      try {
86        if(catalogTracker.waitForMeta(timeout) == null) {
87          throw new NotAllMetaRegionsOnlineException();
88        }
89      } catch (InterruptedException e) {
90        LOG.warn("Interrupted waiting for meta availability", e);
91        throw new IOException(e);
92      }
93  
94      String tableName = this.hTableDescriptor.getNameAsString();
95      if (MetaReader.tableExists(catalogTracker, tableName)) {
96        throw new TableExistsException(tableName);
97      }
98  
99      // If we have multiple client threads trying to create the table at the
100     // same time, given the async nature of the operation, the table
101     // could be in a state where .META. table hasn't been updated yet in
102     // the process() function.
103     // Use enabling state to tell if there is already a request for the same
104     // table in progress. This will introduce a new zookeeper call. Given
105     // createTable isn't a frequent operation, that should be ok.
106     try {
107       if (!this.assignmentManager.getZKTable().checkAndSetEnablingTable(tableName))
108         throw new TableExistsException(tableName);
109     } catch (KeeperException e) {
110       throw new IOException("Unable to ensure that the table will be" +
111         " enabling because of a ZooKeeper issue", e);
112     }
113   }

116   @Override
117   public String toString() {
118     String name = "UnknownServerName";
119     if(server != null && server.getServerName() != null) {
120       name = server.getServerName().toString();
121     }
122     return getClass().getSimpleName() + "-" + name + "-" + getSeqid() + "-" +
123       this.hTableDescriptor.getNameAsString();
124   }

126   @Override
127   public void process() {
128     String tableName = this.hTableDescriptor.getNameAsString();
129     try {
130       LOG.info("Attempting to create the table " + tableName);
131       MasterCoprocessorHost cpHost = ((HMaster) this.server).getCoprocessorHost();
132       if (cpHost != null) {
133         cpHost.preCreateTableHandler(this.hTableDescriptor, this.newRegions);
134       }
135       handleCreateTable(tableName);
136       if (cpHost != null) {
137         cpHost.postCreateTableHandler(this.hTableDescriptor, this.newRegions);
138       }
139     } catch (IOException e) {
140       LOG.error("Error trying to create the table " + tableName, e);
141     } catch (KeeperException e) {
142       LOG.error("Error trying to create the table " + tableName, e);
143     }
144   }

146   private void handleCreateTable(String tableName) throws IOException,
147       KeeperException {
148     int regionNumber = newRegions.length;
149     ThreadPoolExecutor regionOpenAndInitThreadPool = getRegionOpenAndInitThreadPool(
150         "RegionOpenAndInitThread-" + tableName, regionNumber);
151     CompletionService<HRegion> completionService = new ExecutorCompletionService<HRegion>(
152         regionOpenAndInitThreadPool);
153     // TODO: Currently we make the table descriptor and as side-effect the
154     // tableDir is created.  Should we change below method to be createTable
155     // where we create table in tmp dir with its table descriptor file and then
156     // do rename to move it into place?
157     FSTableDescriptors.createTableDescriptor(this.hTableDescriptor, this.conf);
158     List<HRegionInfo> regionInfos = new ArrayList<HRegionInfo>();
159     for (final HRegionInfo newRegion : newRegions) {
160       completionService.submit(new Callable<HRegion>() {
161         public HRegion call() throws IOException {
162 
163           // 1. Create HRegion
164           HRegion region = HRegion.createHRegion(newRegion,
165               fileSystemManager.getRootDir(), conf, hTableDescriptor, null,
166               false, true);
167 
168           // 2. Close the new region to flush to disk. Close log file too.
169           region.close();
170           return region;
171         }
172       });
173     }
174     try {
175       // 3. wait for all regions to finish creation
176       for (int i = 0; i < regionNumber; i++) {
177         Future<HRegion> future = completionService.take();
178         HRegion region = future.get();
179         regionInfos.add(region.getRegionInfo());
180       }
181     } catch (InterruptedException e) {
182       throw new InterruptedIOException(e.getMessage());
183     } catch (ExecutionException e) {
184       throw new IOException(e.getCause());
185     } finally {
186       regionOpenAndInitThreadPool.shutdownNow();
187     }
188     if (regionInfos.size() > 0) {
189       MetaEditor.addRegionsToMeta(this.catalogTracker, regionInfos);
190     }
191 
192     // 4. Trigger immediate assignment of the regions in round-robin fashion
193     try {
194       List<HRegionInfo> regions = Arrays.asList(newRegions);
195       assignmentManager.getRegionStates().createRegionStates(regions);
196       assignmentManager.assign(regions);
197     } catch (InterruptedException ie) {
198       LOG.error("Caught " + ie + " during round-robin assignment");
199       throw new IOException(ie);
200     }
201 
202     // 5. Set table enabled flag up in zk.
203     try {
204       assignmentManager.getZKTable().
205         setEnabledTable(this.hTableDescriptor.getNameAsString());
206     } catch (KeeperException e) {
207       throw new IOException("Unable to ensure that the table will be" +
208         " enabled because of a ZooKeeper issue", e);
209     }
210   }

212   protected ThreadPoolExecutor getRegionOpenAndInitThreadPool(
213       final String threadNamePrefix, int regionNumber) {
214     int maxThreads = Math.min(regionNumber, conf.getInt(
215         "hbase.hregion.open.and.init.threads.max", 10));
216     ThreadPoolExecutor openAndInitializeThreadPool = Threads
217     .getBoundedCachedThreadPool(maxThreads, 30L, TimeUnit.SECONDS,
218         new ThreadFactory() {
219           private int count = 1;
220 
221           public Thread newThread(Runnable r) {
222             Thread t = new Thread(r, threadNamePrefix + "-" + count++);
223             return t;
224           }
225         });
226     return openAndInitializeThreadPool;
227   }
