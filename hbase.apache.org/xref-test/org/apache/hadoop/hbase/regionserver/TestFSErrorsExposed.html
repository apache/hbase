63    /**
64     * Injects errors into the pread calls of an on-disk file, and makes
65     * sure those bubble up to the HFile scanner
66     */
67    @Test
68    public void testHFileScannerThrowsErrors() throws IOException {
69      Path hfilePath = new Path(new Path(
70          util.getDataTestDir("internalScannerExposesErrors"),
71          "regionname"), "familyname");
72      HFileSystem hfs = (HFileSystem)util.getTestFileSystem();
73      FaultyFileSystem faultyfs = new FaultyFileSystem(hfs.getBackingFs());
74      FileSystem fs = new HFileSystem(faultyfs);
75      CacheConfig cacheConf = new CacheConfig(util.getConfiguration());
76      StoreFile.Writer writer = new StoreFile.WriterBuilder(
77          util.getConfiguration(), cacheConf, hfs, 2*1024)
78              .withOutputDir(hfilePath)
79              .build();
80      TestStoreFile.writeStoreFile(
81          writer, Bytes.toBytes("cf"), Bytes.toBytes("qual"));
82  
83      StoreFile sf = new StoreFile(fs, writer.getPath(),
84          util.getConfiguration(), cacheConf, StoreFile.BloomType.NONE,
85          NoOpDataBlockEncoder.INSTANCE);
86  
87      StoreFile.Reader reader = sf.createReader();
88      HFileScanner scanner = reader.getScanner(false, true);
89  
90      FaultyInputStream inStream = faultyfs.inStreams.get(0).get();
91      assertNotNull(inStream);
92  
93      scanner.seekTo();
94      // Do at least one successful read
95      assertTrue(scanner.next());
96  
97      faultyfs.startFaults();
98  
99      try {
100       int scanned=0;
101       while (scanner.next()) {
102         scanned++;
103       }
104       fail("Scanner didn't throw after faults injected");
105     } catch (IOException ioe) {
106       LOG.info("Got expected exception", ioe);
107       assertTrue(ioe.getMessage().contains("Fault"));
108     }
109     reader.close(true); // end of test so evictOnClose
110   }

138 
139     FaultyInputStream inStream = faultyfs.inStreams.get(0).get();
140     assertNotNull(inStream);
141 
142     scanner.seek(KeyValue.LOWESTKEY);
143     // Do at least one successful read
144     assertNotNull(scanner.next());
145     faultyfs.startFaults();
146 
147     try {
148       int scanned=0;
149       while (scanner.next() != null) {
150         scanned++;
151       }
152       fail("Scanner didn't throw after faults injected");
153     } catch (IOException ioe) {
154       LOG.info("Got expected exception", ioe);
155       assertTrue(ioe.getMessage().contains("Could not iterate"));

160   /**
161    * Cluster test which starts a region server with a region, then
162    * removes the data from HDFS underneath it, and ensures that
163    * errors are bubbled to the client.
164    */
165   @Test(timeout=5 * 60 * 1000)
166   public void testFullSystemBubblesFSErrors() throws Exception {
167     // We won't have an error if the datanode is not there if we use short circuit
168     //  it's a known 'feature'.
169     Assume.assumeTrue(!util.isReadShortCircuitOn());
170 
171     try {
172       // We set it not to run or it will trigger server shutdown while sync'ing
173       // because all the datanodes are bad
174       util.getConfiguration().setInt(
175           "hbase.regionserver.optionallogflushinterval", Integer.MAX_VALUE);
176 
177       util.getConfiguration().setInt("hbase.client.retries.number", 3);
178 
179       util.startMiniCluster(1);
180       byte[] tableName = Bytes.toBytes("table");
181       byte[] fam = Bytes.toBytes("fam");
182 
183       HBaseAdmin admin = new HBaseAdmin(util.getConfiguration());
184       HTableDescriptor desc = new HTableDescriptor(tableName);
185       desc.addFamily(new HColumnDescriptor(fam)
186           .setMaxVersions(1)
187           .setBlockCacheEnabled(false)
188       );
189       admin.createTable(desc);
190       // Make it fail faster.
191       util.getConfiguration().setInt("hbase.client.retries.number", 1);
192       // Make a new Configuration so it makes a new connection that has the
193       // above configuration on it; else we use the old one w/ 10 as default.
194       HTable table = new HTable(new Configuration(util.getConfiguration()), tableName);
195 
196       // Load some data
197       util.loadTable(table, fam);
198       table.flushCommits();
199       util.flush();
200       util.countRows(table);
201 
202       // Kill the DFS cluster
203       util.getDFSCluster().shutdownDataNodes();
204 
205       try {
206         util.countRows(table);
207         fail("Did not fail to count after removing data");
208       } catch (Exception e) {
209         LOG.info("Got expected error", e);
210         assertTrue(e.getMessage().contains("Could not seek"));
211       }
212 
213       // Restart data nodes so that HBase can shut down cleanly.
214       util.getDFSCluster().restartDataNodes();
215 
216     } finally {
217       util.getMiniHBaseCluster().killAll();
218       util.shutdownMiniCluster();
219     }
220   }

222   static class FaultyFileSystem extends FilterFileSystem {
223     List<SoftReference<FaultyInputStream>> inStreams =
224       new ArrayList<SoftReference<FaultyInputStream>>();
225 
226     public FaultyFileSystem(FileSystem testFileSystem) {
227       super(testFileSystem);
228     }
229 
230     @Override
231     public FSDataInputStream open(Path p, int bufferSize) throws IOException  {
232       FSDataInputStream orig = fs.open(p, bufferSize);
233       FaultyInputStream faulty = new FaultyInputStream(orig);
234       inStreams.add(new SoftReference<FaultyInputStream>(faulty));
235       return faulty;
236     }
237 
238     /**
239      * Starts to simulate faults on all streams opened so far
240      */
241     public void startFaults() {
242       for (SoftReference<FaultyInputStream> is: inStreams) {
243         is.get().startFaults();
244       }
245     }
246   }
247 
248   static class FaultyInputStream extends FSDataInputStream {
249     boolean faultsStarted = false;
250 
251     public FaultyInputStream(InputStream in) throws IOException {
252       super(in);
253     }
254 
255     public void startFaults() {
256       faultsStarted = true;
257     }
258 
259     public int read(long position, byte[] buffer, int offset, int length)
260       throws IOException {
261       injectFault();
262       return ((PositionedReadable)in).read(position, buffer, offset, length);
263     }
264 
265     private void injectFault() throws IOException {
266       if (faultsStarted) {
267         throw new IOException("Fault injected");
268       }
269     }
270   }
