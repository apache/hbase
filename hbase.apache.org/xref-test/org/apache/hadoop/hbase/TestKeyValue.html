37    public void testColumnCompare() throws Exception {
38      final byte [] a = Bytes.toBytes("aaa");
39      byte [] family1 = Bytes.toBytes("abc");
40      byte [] qualifier1 = Bytes.toBytes("def");
41      byte [] family2 = Bytes.toBytes("abcd");
42      byte [] qualifier2 = Bytes.toBytes("ef");
43  
44      KeyValue aaa = new KeyValue(a, family1, qualifier1, 0L, Type.Put, a);
45      assertFalse(aaa.matchingColumn(family2, qualifier2));
46      assertTrue(aaa.matchingColumn(family1, qualifier1));
47      aaa = new KeyValue(a, family2, qualifier2, 0L, Type.Put, a);
48      assertFalse(aaa.matchingColumn(family1, qualifier1));
49      assertTrue(aaa.matchingColumn(family2,qualifier2));
50      byte [] nullQualifier = new byte[0];
51      aaa = new KeyValue(a, family1, nullQualifier, 0L, Type.Put, a);
52      assertTrue(aaa.matchingColumn(family1,null));
53      assertFalse(aaa.matchingColumn(family2,qualifier2));
54    }

90    public void testPlainCompare() throws Exception {
91      final byte [] a = Bytes.toBytes("aaa");
92      final byte [] b = Bytes.toBytes("bbb");
93      final byte [] fam = Bytes.toBytes("col");
94      final byte [] qf = Bytes.toBytes("umn");
95  //    final byte [] column = Bytes.toBytes("col:umn");
96      KeyValue aaa = new KeyValue(a, fam, qf, a);
97      KeyValue bbb = new KeyValue(b, fam, qf, b);
98      byte [] keyabb = aaa.getKey();
99      byte [] keybbb = bbb.getKey();
100     assertTrue(KeyValue.COMPARATOR.compare(aaa, bbb) < 0);
101     assertTrue(KeyValue.KEY_COMPARATOR.compare(keyabb, 0, keyabb.length, keybbb,
102       0, keybbb.length) < 0);
103     assertTrue(KeyValue.COMPARATOR.compare(bbb, aaa) > 0);
104     assertTrue(KeyValue.KEY_COMPARATOR.compare(keybbb, 0, keybbb.length, keyabb,
105       0, keyabb.length) > 0);
106     // Compare breaks if passed same ByteBuffer as both left and right arguments.
107     assertTrue(KeyValue.COMPARATOR.compare(bbb, bbb) == 0);
108     assertTrue(KeyValue.KEY_COMPARATOR.compare(keybbb, 0, keybbb.length, keybbb,
109       0, keybbb.length) == 0);
110     assertTrue(KeyValue.COMPARATOR.compare(aaa, aaa) == 0);
111     assertTrue(KeyValue.KEY_COMPARATOR.compare(keyabb, 0, keyabb.length, keyabb,
112       0, keyabb.length) == 0);
113     // Do compare with different timestamps.
114     aaa = new KeyValue(a, fam, qf, 1, a);
115     bbb = new KeyValue(a, fam, qf, 2, a);
116     assertTrue(KeyValue.COMPARATOR.compare(aaa, bbb) > 0);
117     assertTrue(KeyValue.COMPARATOR.compare(bbb, aaa) < 0);
118     assertTrue(KeyValue.COMPARATOR.compare(aaa, aaa) == 0);
119     // Do compare with different types.  Higher numbered types -- Delete
120     // should sort ahead of lower numbers; i.e. Put
121     aaa = new KeyValue(a, fam, qf, 1, KeyValue.Type.Delete, a);
122     bbb = new KeyValue(a, fam, qf, 1, a);
123     assertTrue(KeyValue.COMPARATOR.compare(aaa, bbb) < 0);
124     assertTrue(KeyValue.COMPARATOR.compare(bbb, aaa) > 0);
125     assertTrue(KeyValue.COMPARATOR.compare(aaa, aaa) == 0);
126   }

207

209         Bytes.toBytes(""), Long.MAX_VALUE, (byte[])null);

211         Bytes.toBytes("fam"), Bytes.toBytes(""), Long.MAX_VALUE, (byte[])null);
212     assertTrue(KeyValue.META_COMPARATOR.compare(rowA, rowB) < 0);

243     Set<KeyValue> set = new TreeSet<KeyValue>(KeyValue.COMPARATOR);
244     final byte [] fam = Bytes.toBytes("col");

249       new KeyValue(Bytes.toBytes("aaaaa,,1"), fam, qf, 1, nb),
250       new KeyValue(Bytes.toBytes("aaaaa,\u1000,5"), fam, qf, 5, nb),
251       new KeyValue(Bytes.toBytes("aaaaa,a,4"), fam, qf, 4, nb),
252       new KeyValue(Bytes.toBytes("a,a,0"), fam, qf, 0, nb),
253     };
254     // Add to set with bad comparator
255     for (int i = 0; i < keys.length; i++) {
256       set.add(keys[i]);
257     }
258     // This will output the keys incorrectly.
259     boolean assertion = false;
260     int count = 0;
261     try {
262       for (KeyValue k: set) {
263         assertTrue(count++ == k.getTimestamp());
264       }
265     } catch (junit.framework.AssertionFailedError e) {
266       // Expected
267       assertion = true;
268     }
269     assertTrue(assertion);

286         new KeyValue(Bytes.toBytes(".META.,,0"), fam, qf, 0, nb),
287       };
288     // This will output the keys incorrectly.
289     set = new TreeSet<KeyValue>(new KeyValue.MetaComparator());
290     // Add to set with bad comparator

294     assertion = false;

296     try {
297       for (KeyValue k: set) {
298         assertTrue(count++ == k.getTimestamp());
299       }
300     } catch (junit.framework.AssertionFailedError e) {
301       // Expected
302       assertion = true;
303     }
304     // Now with right comparator
305     set = new TreeSet<KeyValue>(new KeyValue.RootComparator());
306     // Add to set with bad comparator
307     for (int i = 0; i < keys.length; i++) {
308       set.add(rootKeys[i]);
309     }
310     count = 0;
311     for (KeyValue k: set) {
312       assertTrue(count++ == k.getTimestamp());
313     }

338   private void assertKVLessWithoutRow(KeyValue.KeyComparator c, int common, KeyValue less,
339       KeyValue greater) {
340     int cmp = c.compareIgnoringPrefix(common, less.getBuffer(), less.getOffset()
341         + KeyValue.ROW_OFFSET, less.getKeyLength(), greater.getBuffer(),
342         greater.getOffset() + KeyValue.ROW_OFFSET, greater.getKeyLength());
343     assertTrue(cmp < 0);
344     cmp = c.compareIgnoringPrefix(common, greater.getBuffer(), greater.getOffset()
345         + KeyValue.ROW_OFFSET, greater.getKeyLength(), less.getBuffer(),
346         less.getOffset() + KeyValue.ROW_OFFSET, less.getKeyLength());
347     assertTrue(cmp > 0);
348   }

350   public void testCompareWithoutRow() {
351     final KeyValue.KeyComparator c = KeyValue.KEY_COMPARATOR;
352     byte[] row = Bytes.toBytes("row");
353 
354     byte[] fa = Bytes.toBytes("fa");
355     byte[] fami = Bytes.toBytes("fami");
356     byte[] fami1 = Bytes.toBytes("fami1");
357 
358     byte[] qual0 = Bytes.toBytes("");
359     byte[] qual1 = Bytes.toBytes("qf1");
360     byte[] qual2 = Bytes.toBytes("qf2");
361     long ts = 1;
362 
363     // 'fa:'
364     KeyValue kv_0 = new KeyValue(row, fa, qual0, ts, Type.Put);
365     // 'fami:'
366     KeyValue kv0_0 = new KeyValue(row, fami, qual0, ts, Type.Put);
367     // 'fami:qf1'
368     KeyValue kv0_1 = new KeyValue(row, fami, qual1, ts, Type.Put);
369     // 'fami:qf2'
370     KeyValue kv0_2 = new KeyValue(row, fami, qual2, ts, Type.Put);
371     // 'fami1:'
372     KeyValue kv1_0 = new KeyValue(row, fami1, qual0, ts, Type.Put);
373 
374     // 'fami:qf1' < 'fami:qf2'
375     assertKVLessWithoutRow(c, 0, kv0_1, kv0_2);
376     // 'fami:qf1' < 'fami1:'
377     assertKVLessWithoutRow(c, 0, kv0_1, kv1_0);
378 
379     // Test comparison by skipping the same prefix bytes.
380     /***
381      * KeyValue Format and commonLength:
382      * |_keyLen_|_valLen_|_rowLen_|_rowKey_|_famiLen_|_fami_|_Quali_|....
383      * ------------------|-------commonLength--------|--------------
384      */
385     int commonLength = KeyValue.ROW_LENGTH_SIZE + KeyValue.FAMILY_LENGTH_SIZE
386         + row.length;
387     // 'fa:' < 'fami:'. They have commonPrefix + 2 same prefix bytes.
388     assertKVLessWithoutRow(c, commonLength + 2, kv_0, kv0_0);
389     // 'fami:' < 'fami:qf1'. They have commonPrefix + 4 same prefix bytes.
390     assertKVLessWithoutRow(c, commonLength + 4, kv0_0, kv0_1);
391     // 'fami:qf1' < 'fami1:'. They have commonPrefix + 4 same prefix bytes.
392     assertKVLessWithoutRow(c, commonLength + 4, kv0_1, kv1_0);
393     // 'fami:qf1' < 'fami:qf2'. They have commonPrefix + 6 same prefix bytes.
394     assertKVLessWithoutRow(c, commonLength + 6, kv0_1, kv0_2);
395   }
