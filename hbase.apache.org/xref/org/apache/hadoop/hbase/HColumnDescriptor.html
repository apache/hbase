48  /**
49   * An HColumnDescriptor contains information about a column family such as the
50   * number of versions, compression settings, etc.
51   *
52   * It is used as input when creating a table or adding a column. Once set, the
53   * parameters that specify a column cannot be changed without deleting the
54   * column and recreating it. If there is data stored in the column, it will be
55   * deleted when the column is deleted.
56   */
57  @InterfaceAudience.Public
58  @InterfaceStability.Evolving
59  public class HColumnDescriptor implements WritableComparable<HColumnDescriptor> {

345   /**
346    * Constructor
347    * @param familyName Column family name. Must be 'printable' -- digit or
348    * letter -- and may not contain a <code>:<code>
349    * @param minVersions Minimum number of versions to keep
350    * @param maxVersions Maximum number of versions to keep
351    * @param keepDeletedCells Whether to retain deleted cells until they expire
352    *        up to maxVersions versions.
353    * @param compression Compression type
354    * @param encodeOnDisk whether to use the specified data block encoding
355    *        on disk. If false, the encoding will be used in cache only.
356    * @param dataBlockEncoding data block encoding
357    * @param inMemory If true, column data should be kept in an HRegionServer's
358    * cache
359    * @param blockCacheEnabled If true, MapFile blocks should be cached
360    * @param blocksize Block size to use when writing out storefiles.  Use
361    * smaller blocksizes for faster random-access at expense of larger indices
362    * (more memory consumption).  Default is usually 64k.
363    * @param timeToLive Time-to-live of cell contents, in seconds
364    * (use HConstants.FOREVER for unlimited TTL)
365    * @param bloomFilter Bloom filter type for this column
366    * @param scope The scope tag for this column
367    *
368    * @throws IllegalArgumentException if passed a family name that is made of
369    * other than 'word' characters: i.e. <code>[a-zA-Z_0-9]</code> or contains
370    * a <code>:</code>
371    * @throws IllegalArgumentException if the number of versions is &lt;= 0
372    * @deprecated use {@link #HColumnDescriptor(String)} and setters
373    */
374   @Deprecated
375   public HColumnDescriptor(final byte[] familyName, final int minVersions,
376       final int maxVersions, final boolean keepDeletedCells,
377       final String compression, final boolean encodeOnDisk,
378       final String dataBlockEncoding, final boolean inMemory,
379       final boolean blockCacheEnabled, final int blocksize,
380       final int timeToLive, final String bloomFilter, final int scope) {
381     isLegalFamilyName(familyName);
382     this.name = familyName;
383 
384     if (maxVersions <= 0) {
385       // TODO: Allow maxVersion of 0 to be the way you say "Keep all versions".
386       // Until there is support, consider 0 or < 0 -- a configuration error.
387       throw new IllegalArgumentException("Maximum versions must be positive");
388     }
389 
390     if (minVersions > 0) {
391       if (timeToLive == HConstants.FOREVER) {
392         throw new IllegalArgumentException("Minimum versions requires TTL.");
393       }
394       if (minVersions >= maxVersions) {
395         throw new IllegalArgumentException("Minimum versions must be < "
396             + "maximum versions.");
397       }
398     }
399 
400     setMaxVersions(maxVersions);
401     setMinVersions(minVersions);
402     setKeepDeletedCells(keepDeletedCells);
403     setInMemory(inMemory);
404     setBlockCacheEnabled(blockCacheEnabled);
405     setTimeToLive(timeToLive);
406     setCompressionType(Compression.Algorithm.
407       valueOf(compression.toUpperCase()));
408     setEncodeOnDisk(encodeOnDisk);
409     setDataBlockEncoding(DataBlockEncoding.
410         valueOf(dataBlockEncoding.toUpperCase()));
411     setBloomFilterType(StoreFile.BloomType.
412       valueOf(bloomFilter.toUpperCase()));
413     setBlocksize(blocksize);
414     setScope(scope);
415   }

417   /**
418    * @param b Family name.
419    * @return <code>b</code>
420    * @throws IllegalArgumentException If not null and not a legitimate family
421    * name: i.e. 'printable' and ends in a ':' (Null passes are allowed because
422    * <code>b</code> can be null when deserializing).  Cannot start with a '.'
423    * either. Also Family can not be an empty value.
424    */
425   public static byte [] isLegalFamilyName(final byte [] b) {
426     if (b == null) {
427       return b;
428     }
429     Preconditions.checkArgument(b.length != 0, "Family name can not be empty");
430     if (b[0] == '.') {
431       throw new IllegalArgumentException("Family names cannot start with a " +
432         "period: " + Bytes.toString(b));
433     }
434     for (int i = 0; i < b.length; i++) {
435       if (Character.isISOControl(b[i]) || b[i] == ':' || b[i] == '\\' || b[i] == '/') {
436         throw new IllegalArgumentException("Illegal character <" + b[i] +
437           ">. Family names cannot contain control characters or colons: " +
438           Bytes.toString(b));
439       }
440     }
441     return b;
442   }

600   /** @return data block encoding algorithm used on disk */
601   public DataBlockEncoding getDataBlockEncodingOnDisk() {
602     String encodeOnDiskStr = getValue(ENCODE_ON_DISK);
603     boolean encodeOnDisk;
604     if (encodeOnDiskStr == null) {
605       encodeOnDisk = DEFAULT_ENCODE_ON_DISK;
606     } else {
607       encodeOnDisk = Boolean.valueOf(encodeOnDiskStr);
608     }
609 
610     if (!encodeOnDisk) {
611       // No encoding on disk.
612       return DataBlockEncoding.NONE;
613     }
614     return getDataBlockEncoding();
615   }

907   private StringBuilder getValues(boolean printDefaults) {
908     StringBuilder s = new StringBuilder();
909 
910     boolean hasConfigKeys = false;
911 
912     // print all reserved keys first
913     for (ImmutableBytesWritable k : values.keySet()) {
914       if (!RESERVED_KEYWORDS.contains(k)) {
915         hasConfigKeys = true;
916         continue;
917       }
918       String key = Bytes.toString(k.get());
919       String value = Bytes.toString(values.get(k).get());
920       if (printDefaults
921           || !DEFAULT_VALUES.containsKey(key)
922           || !DEFAULT_VALUES.get(key).equalsIgnoreCase(value)) {
923         s.append(", ");
924         s.append(key);
925         s.append(" => ");
926         s.append('\'').append(value).append('\'');
927       }
928     }
929 
930     // print all non-reserved, advanced config keys as a separate subset
931     if (hasConfigKeys) {
932       s.append(", ");
933       s.append(HConstants.CONFIG).append(" => ");
934       s.append('{');
935       boolean printComma = false;
936       for (ImmutableBytesWritable k : values.keySet()) {
937         if (RESERVED_KEYWORDS.contains(k)) {
938           continue;
939         }
940         String key = Bytes.toString(k.get());
941         String value = Bytes.toString(values.get(k).get());
942         if (printComma) {
943           s.append(", ");
944         }
945         printComma = true;
946         s.append('\'').append(key).append('\'');
947         s.append(" => ");
948         s.append('\'').append(value).append('\'');
949       }
950       s.append('}');
951     }
952     return s;
953   }
