106   /* Are we running on jdk7? */
107   private static final boolean JDK7;
108   static {
109     final String version = System.getProperty("java.version");
110     // Verify String looks like this: 1.6.0_29
111     if (!version.matches("\\d\\.\\d\\..*")) {
112       throw new RuntimeException("Unexpected version format: " + version);
113     }
114     // Convert char to int
115     int major = (int)(version.charAt(0) - '0');
116     int minor = (int)(version.charAt(2) - '0');
117     JDK7 = major == 1 && minor == 7;
118   }

120   /**
121    * Method for reading the arc settings and setting overheads according
122    * to 32-bit or 64-bit architecture.
123    */
124   static {
125     //Default value is set to 8, covering the case when arcModel is unknown
126     if (is32BitJVM()) {
127       REFERENCE = 4;
128     } else {
129       REFERENCE = 8;
130     }
131 
132     OBJECT = 2 * REFERENCE;
133 
134     ARRAY = align(3 * REFERENCE);
135 
136     ARRAYLIST = align(OBJECT + align(REFERENCE) + align(ARRAY) +
137         (2 * Bytes.SIZEOF_INT));
138 
139     //noinspection PointlessArithmeticExpression
140     BYTE_BUFFER = align(OBJECT + align(REFERENCE) + align(ARRAY) +
141         (5 * Bytes.SIZEOF_INT) +
142         (3 * Bytes.SIZEOF_BOOLEAN) + Bytes.SIZEOF_LONG);
143 
144     INTEGER = align(OBJECT + Bytes.SIZEOF_INT);
145 
146     MAP_ENTRY = align(OBJECT + 5 * REFERENCE + Bytes.SIZEOF_BOOLEAN);
147 
148     TREEMAP = align(OBJECT + (2 * Bytes.SIZEOF_INT) + align(7 * REFERENCE));
149 
150     // STRING is different size in jdk6 and jdk7. Just use what we estimate as size rather than
151     // have a conditional on whether jdk7.
152     STRING = (int) estimateBase(String.class, false);
153 
154     // CONCURRENT_HASHMAP is different size in jdk6 and jdk7; it looks like its different between
155     // 23.6-b03 and 23.0-b21. Just use what we estimate as size rather than have a conditional on
156     // whether jdk7.
157     CONCURRENT_HASHMAP = (int) estimateBase(ConcurrentHashMap.class, false);
158 
159     CONCURRENT_HASHMAP_ENTRY = align(REFERENCE + OBJECT + (3 * REFERENCE) +
160         (2 * Bytes.SIZEOF_INT));
161 
162     CONCURRENT_HASHMAP_SEGMENT = align(REFERENCE + OBJECT +
163         (3 * Bytes.SIZEOF_INT) + Bytes.SIZEOF_FLOAT + ARRAY);
164 
165     CONCURRENT_SKIPLISTMAP = align(Bytes.SIZEOF_INT + OBJECT + (8 * REFERENCE));
166 
167     CONCURRENT_SKIPLISTMAP_ENTRY = align(
168         align(OBJECT + (3 * REFERENCE)) + /* one node per entry */
169         align((OBJECT + (3 * REFERENCE))/2)); /* one index per two entries */
170 
171     REENTRANT_LOCK = align(OBJECT + (3 * REFERENCE));
172 
173     ATOMIC_LONG = align(OBJECT + Bytes.SIZEOF_LONG);
174 
175     ATOMIC_INTEGER = align(OBJECT + Bytes.SIZEOF_INT);
176 
177     ATOMIC_BOOLEAN = align(OBJECT + Bytes.SIZEOF_BOOLEAN);
178 
179     COPYONWRITE_ARRAYSET = align(OBJECT + REFERENCE);
180 
181     COPYONWRITE_ARRAYLIST = align(OBJECT + (2 * REFERENCE) + ARRAY);
182   }

184   /**
185    * The estimate of the size of a class instance depends on whether the JVM
186    * uses 32 or 64 bit addresses, that is it depends on the size of an object
187    * reference. It is a linear function of the size of a reference, e.g.
188    * 24 + 5*r where r is the size of a reference (usually 4 or 8 bytes).
189    *
190    * This method returns the coefficients of the linear function, e.g. {24, 5}
191    * in the above example.
192    *
193    * @param cl A class whose instance size is to be estimated
194    * @param debug debug flag
195    * @return an array of 3 integers. The first integer is the size of the
196    * primitives, the second the number of arrays and the third the number of
197    * references.
198    */
199   @SuppressWarnings("unchecked")
200   private static int [] getSizeCoefficients(Class cl, boolean debug) {
201     int primitives = 0;
202     int arrays = 0;
203     //The number of references that a new object takes
204     int references = nrOfRefsPerObj;
205     int index = 0;
206 
207     for ( ; null != cl; cl = cl.getSuperclass()) {
208       Field[] field = cl.getDeclaredFields();
209       if (null != field) {
210         for (Field aField : field) {
211           if (Modifier.isStatic(aField.getModifiers())) continue;
212           Class fieldClass = aField.getType();
213           if (fieldClass.isArray()) {
214             arrays++;
215             references++;
216           } else if (!fieldClass.isPrimitive()) {
217             references++;
218           } else {// Is simple primitive
219             String name = fieldClass.getName();
220 
221             if (name.equals("int") || name.equals("I"))
222               primitives += Bytes.SIZEOF_INT;
223             else if (name.equals("long") || name.equals("J"))
224               primitives += Bytes.SIZEOF_LONG;
225             else if (name.equals("boolean") || name.equals("Z"))
226               primitives += Bytes.SIZEOF_BOOLEAN;
227             else if (name.equals("short") || name.equals("S"))
228               primitives += Bytes.SIZEOF_SHORT;
229             else if (name.equals("byte") || name.equals("B"))
230               primitives += Bytes.SIZEOF_BYTE;
231             else if (name.equals("char") || name.equals("C"))
232               primitives += Bytes.SIZEOF_CHAR;
233             else if (name.equals("float") || name.equals("F"))
234               primitives += Bytes.SIZEOF_FLOAT;
235             else if (name.equals("double") || name.equals("D"))
236               primitives += Bytes.SIZEOF_DOUBLE;
237           }
238           if (debug) {
239             if (LOG.isDebugEnabled()) {
240               // Write out region name as string and its encoded name.
241               LOG.debug("" + index + " " + aField.getName() + " " + aField.getType());
242             }
243           }
244           index++;
245         }
246       }
247     }
248     return new int [] {primitives, arrays, references};
249   }

251   /**
252    * Estimate the static space taken up by a class instance given the
253    * coefficients returned by getSizeCoefficients.
254    *
255    * @param coeff the coefficients
256    *
257    * @param debug debug flag
258    * @return the size estimate, in bytes
259    */
260   private static long estimateBaseFromCoefficients(int [] coeff, boolean debug) {
261     long prealign_size = coeff[0] + align(coeff[1] * ARRAY) + coeff[2] * REFERENCE;
262 
263     // Round up to a multiple of 8
264     long size = align(prealign_size);
265     if(debug) {
266       if (LOG.isDebugEnabled()) {
267         // Write out region name as string and its encoded name.
268         LOG.debug("Primitives=" + coeff[0] + ", arrays=" + coeff[1] +
269             ", references(includes " + nrOfRefsPerObj +
270             " for object overhead)=" + coeff[2] + ", refSize " + REFERENCE +
271             ", size=" + size + ", prealign_size=" + prealign_size);
272       }
273     }
274     return size;
275   }

277   /**
278    * Estimate the static space taken up by the fields of a class. This includes
279    * the space taken up by by references (the pointer) but not by the referenced
280    * object. So the estimated size of an array field does not depend on the size
281    * of the array. Similarly the size of an object (reference) field does not
282    * depend on the object.
283    *
284    * @param cl class
285    * @param debug debug flag
286    * @return the size estimate in bytes.
287    */
288   @SuppressWarnings("unchecked")
289   public static long estimateBase(Class cl, boolean debug) {
290     return estimateBaseFromCoefficients( getSizeCoefficients(cl, debug), debug);
291   }

293   /**
294    * Aligns a number to 8.
295    * @param num number to align to 8
296    * @return smallest number >= input that is a multiple of 8
297    */
298   public static int align(int num) {
299     return (int)(align((long)num));
300   }

302   /**
303    * Aligns a number to 8.
304    * @param num number to align to 8
305    * @return smallest number >= input that is a multiple of 8
306    */
307   public static long align(long num) {
308     //The 7 comes from that the alignSize is 8 which is the number of bytes
309     //stored and sent together
310     return  ((num + 7) >> 3) << 3;
311   }
