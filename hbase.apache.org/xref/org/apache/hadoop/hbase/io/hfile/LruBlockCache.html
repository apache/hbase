56  /**
57   * A block cache implementation that is memory-aware using {@link HeapSize},
58   * memory-bound using an LRU eviction algorithm, and concurrent: backed by a
59   * {@link ConcurrentHashMap} and with a non-blocking eviction thread giving
60   * constant-time {@link #cacheBlock} and {@link #getBlock} operations.<p>
61   *
62   * Contains three levels of block priority to allow for
63   * scan-resistance and in-memory families.  A block is added with an inMemory
64   * flag if necessary, otherwise a block becomes a single access priority.  Once
65   * a blocked is accessed again, it changes to multiple access.  This is used
66   * to prevent scans from thrashing the cache, adding a least-frequently-used
67   * element to the eviction algorithm.<p>
68   *
69   * Each priority is given its own chunk of the total cache to ensure
70   * fairness during eviction.  Each priority will retain close to its maximum
71   * size, however, if any priority is not using its entire chunk the others
72   * are able to grow beyond their chunk size.<p>
73   *
74   * Instantiated at a minimum with the total size and average block size.
75   * All sizes are in bytes.  The block size is not especially important as this
76   * cache is fully dynamic in its sizing of blocks.  It is only used for
77   * pre-allocating data structures and in initial heap estimation of the map.<p>
78   *
79   * The detailed constructor defines the sizes for the three priorities (they
80   * should total to the maximum size defined).  It also sets the levels that
81   * trigger and control the eviction thread.<p>
82   *
83   * The acceptable size is the cache size level which triggers the eviction
84   * process to start.  It evicts enough blocks to get the size below the
85   * minimum size specified.<p>
86   *
87   * Eviction happens in a separate thread and involves a single full-scan
88   * of the map.  It determines how many bytes must be freed to reach the minimum
89   * size, and then while scanning determines the fewest least-recently-used
90   * blocks necessary from each of the three priorities (would be 3 times bytes
91   * to free).  It then uses the priority chunk sizes to evict fairly according
92   * to the relative sizes and usage.
93   */
94  @InterfaceAudience.Private
95  public class LruBlockCache implements BlockCache, HeapSize {

217   /**
218    * Configurable constructor.  Use this constructor if not using defaults.
219    * @param maxSize maximum size of this cache, in bytes
220    * @param blockSize expected average size of blocks, in bytes
221    * @param evictionThread whether to run evictions in a bg thread or not
222    * @param mapInitialSize initial size of backing ConcurrentHashMap
223    * @param mapLoadFactor initial load factor of backing ConcurrentHashMap
224    * @param mapConcurrencyLevel initial concurrency factor for backing CHM
225    * @param minFactor percentage of total size that eviction will evict until
226    * @param acceptableFactor percentage of total size that triggers eviction
227    * @param singleFactor percentage of total size for single-access blocks
228    * @param multiFactor percentage of total size for multiple-access blocks
229    * @param memoryFactor percentage of total size for in-memory blocks
230    */
231   public LruBlockCache(long maxSize, long blockSize, boolean evictionThread,
232       int mapInitialSize, float mapLoadFactor, int mapConcurrencyLevel,
233       float minFactor, float acceptableFactor,
234       float singleFactor, float multiFactor, float memoryFactor) {
235     if(singleFactor + multiFactor + memoryFactor != 1) {
236       throw new IllegalArgumentException("Single, multi, and memory factors " +
237           " should total 1.0");
238     }
239     if(minFactor >= acceptableFactor) {
240       throw new IllegalArgumentException("minFactor must be smaller than acceptableFactor");
241     }
242     if(minFactor >= 1.0f || acceptableFactor >= 1.0f) {
243       throw new IllegalArgumentException("all factors must be < 1");
244     }
245     this.maxSize = maxSize;
246     this.blockSize = blockSize;
247     map = new ConcurrentHashMap<BlockCacheKey,CachedBlock>(mapInitialSize,
248         mapLoadFactor, mapConcurrencyLevel);
249     this.minFactor = minFactor;
250     this.acceptableFactor = acceptableFactor;
251     this.singleFactor = singleFactor;
252     this.multiFactor = multiFactor;
253     this.memoryFactor = memoryFactor;
254     this.stats = new CacheStats();
255     this.count = new AtomicLong(0);
256     this.elements = new AtomicLong(0);
257     this.overhead = calculateOverhead(maxSize, blockSize, mapConcurrencyLevel);
258     this.size = new AtomicLong(this.overhead);
259     if(evictionThread) {
260       this.evictionThread = new EvictionThread(this);
261       this.evictionThread.start(); // FindBugs SC_START_IN_CTOR
262     } else {
263       this.evictionThread = null;
264     }
265     this.scheduleThreadPool.scheduleAtFixedRate(new StatisticsThread(this),
266         statThreadPeriod, statThreadPeriod, TimeUnit.SECONDS);
267   }

278   /**
279    * Cache the block with the specified name and buffer.
280    * <p>
281    * It is assumed this will NEVER be called on an already cached block.  If
282    * that is done, an exception will be thrown.
283    * @param cacheKey block's cache key
284    * @param buf block buffer
285    * @param inMemory if block is in-memory
286    */
287   public void cacheBlock(BlockCacheKey cacheKey, Cacheable buf, boolean inMemory) {
288     CachedBlock cb = map.get(cacheKey);
289     if(cb != null) {
290       throw new RuntimeException("Cached an already cached block");
291     }
292     cb = new CachedBlock(cacheKey, buf, count.incrementAndGet(), inMemory);
293     long newSize = updateSizeMetrics(cb, false);
294     map.put(cacheKey, cb);
295     elements.incrementAndGet();
296     if(newSize > acceptableSize() && !evictionInProgress) {
297       runEviction();
298     }
299   }

318    * {@link CachedBlock}

320    * @param cb
321    * @param evict

323   protected long updateSizeMetrics(CachedBlock cb, boolean evict) {
324     long heapsize = cb.heapSize();
325     if (evict) {
326       heapsize *= -1;
327     }
328     return size.addAndGet(heapsize);
329   }

331   /**
332    * Get the buffer of the block with the specified name.
333    * @param cacheKey block's cache key
334    * @param caching true if the caller caches blocks on cache misses
335    * @param repeat Whether this is a repeat lookup for the same block
336    *        (used to avoid double counting cache misses when doing double-check locking)
337    * @return buffer of specified cache key, or null if not in cache
338    * @see HFileReaderV2#readBlock(long, long, boolean, boolean, boolean, BlockType)
339    */
340   @Override
341   public Cacheable getBlock(BlockCacheKey cacheKey, boolean caching, boolean repeat) {
342     CachedBlock cb = map.get(cacheKey);
343     if(cb == null) {
344       if (!repeat) stats.miss(caching);
345       return null;
346     }
347     stats.hit(caching);
348     cb.access(count.incrementAndGet());
349     return cb.getBuffer();
350   }

353   @Override
354   public boolean evictBlock(BlockCacheKey cacheKey) {
355     CachedBlock cb = map.get(cacheKey);
356     if (cb == null) return false;
357     evictBlock(cb);
358     return true;
359   }

361   /**
362    * Evicts all blocks for a specific HFile. This is an
363    * expensive operation implemented as a linear-time search through all blocks
364    * in the cache. Ideally this should be a search in a log-access-time map.
365    *
366    * <p>
367    * This is used for evict-on-close to remove all blocks of a specific HFile.
368    *
369    * @return the number of blocks evicted
370    */
371   @Override
372   public int evictBlocksByHfileName(String hfileName) {
373     int numEvicted = 0;
374     for (BlockCacheKey key : map.keySet()) {
375       if (key.getHfileName().equals(hfileName)) {
376         if (evictBlock(key))
377           ++numEvicted;
378       }
379     }
380     return numEvicted;
381   }
382 
383   protected long evictBlock(CachedBlock block) {
384     map.remove(block.getCacheKey());
385     updateSizeMetrics(block, true);
386     elements.decrementAndGet();
387     stats.evicted();
388     return block.heapSize();
389   }

391   /**
392    * Multi-threaded call to run the eviction process.
393    */
394   private void runEviction() {
395     if(evictionThread == null) {
396       evict();
397     } else {
398       evictionThread.evict();
399     }
400   }

402   /**
403    * Eviction method.
404    */
405   void evict() {
406 
407     // Ensure only one eviction at a time
408     if(!evictionLock.tryLock()) return;
409 
410     try {
411       evictionInProgress = true;
412       long currentSize = this.size.get();
413       long bytesToFree = currentSize - minSize();
414 
415       if (LOG.isDebugEnabled()) {
416         LOG.debug("Block cache LRU eviction started; Attempting to free " +
417           StringUtils.byteDesc(bytesToFree) + " of total=" +
418           StringUtils.byteDesc(currentSize));
419       }
420 
421       if(bytesToFree <= 0) return;
422 
423       // Instantiate priority buckets
424       BlockBucket bucketSingle = new BlockBucket(bytesToFree, blockSize,
425           singleSize());
426       BlockBucket bucketMulti = new BlockBucket(bytesToFree, blockSize,
427           multiSize());
428       BlockBucket bucketMemory = new BlockBucket(bytesToFree, blockSize,
429           memorySize());
430 
431       // Scan entire map putting into appropriate buckets
432       for(CachedBlock cachedBlock : map.values()) {
433         switch(cachedBlock.getPriority()) {
434           case SINGLE: {
435             bucketSingle.add(cachedBlock);
436             break;
437           }
438           case MULTI: {
439             bucketMulti.add(cachedBlock);
440             break;
441           }
442           case MEMORY: {
443             bucketMemory.add(cachedBlock);
444             break;
445           }
446         }
447       }
448 
449       PriorityQueue<BlockBucket> bucketQueue =
450         new PriorityQueue<BlockBucket>(3);
451 
452       bucketQueue.add(bucketSingle);
453       bucketQueue.add(bucketMulti);
454       bucketQueue.add(bucketMemory);
455 
456       int remainingBuckets = 3;
457       long bytesFreed = 0;
458 
459       BlockBucket bucket;
460       while((bucket = bucketQueue.poll()) != null) {
461         long overflow = bucket.overflow();
462         if(overflow > 0) {
463           long bucketBytesToFree = Math.min(overflow,
464             (bytesToFree - bytesFreed) / remainingBuckets);
465           bytesFreed += bucket.free(bucketBytesToFree);
466         }
467         remainingBuckets--;
468       }
469 
470       if (LOG.isDebugEnabled()) {
471         long single = bucketSingle.totalSize();
472         long multi = bucketMulti.totalSize();
473         long memory = bucketMemory.totalSize();
474         LOG.debug("Block cache LRU eviction completed; " +
475           "freed=" + StringUtils.byteDesc(bytesFreed) + ", " +
476           "total=" + StringUtils.byteDesc(this.size.get()) + ", " +
477           "single=" + StringUtils.byteDesc(single) + ", " +
478           "multi=" + StringUtils.byteDesc(multi) + ", " +
479           "memory=" + StringUtils.byteDesc(memory));
480       }
481     } finally {
482       stats.evict();
483       evictionInProgress = false;
484       evictionLock.unlock();
485     }
486   }

488   /**
489    * Used to group blocks into priority buckets.  There will be a BlockBucket
490    * for each priority (single, multi, memory).  Once bucketed, the eviction
491    * algorithm takes the appropriate number of elements out of each according
492    * to configuration parameters and their relatives sizes.
493    */
494   private class BlockBucket implements Comparable<BlockBucket> {
495 
496     private CachedBlockQueue queue;
497     private long totalSize = 0;
498     private long bucketSize;
499 
500     public BlockBucket(long bytesToFree, long blockSize, long bucketSize) {
501       this.bucketSize = bucketSize;
502       queue = new CachedBlockQueue(bytesToFree, blockSize);
503       totalSize = 0;
504     }
505 
506     public void add(CachedBlock block) {
507       totalSize += block.heapSize();
508       queue.add(block);
509     }
510 
511     public long free(long toFree) {
512       CachedBlock cb;
513       long freedBytes = 0;
514       while ((cb = queue.pollLast()) != null) {
515         freedBytes += evictBlock(cb);
516         if (freedBytes >= toFree) {
517           return freedBytes;
518         }
519       }
520       return freedBytes;
521     }
522 
523     public long overflow() {
524       return totalSize - bucketSize;
525     }
526 
527     public long totalSize() {
528       return totalSize;
529     }
530 
531     public int compareTo(BlockBucket that) {
532       if(this.overflow() == that.overflow()) return 0;
533       return this.overflow() > that.overflow() ? 1 : -1;
534     }
535   }

593   /*
594    * Eviction thread.  Sits in waiting state until an eviction is triggered
595    * when the cache size grows above the acceptable level.<p>
596    *
597    * Thread is triggered into action by {@link LruBlockCache#runEviction()}
598    */
599   static class EvictionThread extends HasThread {
600     private WeakReference<LruBlockCache> cache;
601     private boolean go = true;
602     // flag set after enter the run method, used for test
603     private boolean enteringRun = false;
604 
605     public EvictionThread(LruBlockCache cache) {
606       super(Thread.currentThread().getName() + ".LruBlockCache.EvictionThread");
607       setDaemon(true);
608       this.cache = new WeakReference<LruBlockCache>(cache);
609     }
610 
611     @Override
612     public void run() {
613       enteringRun = true;
614       while (this.go) {
615         synchronized(this) {
616           try {
617             this.wait();
618           } catch(InterruptedException e) {}
619         }
620         LruBlockCache cache = this.cache.get();
621         if(cache == null) break;
622         cache.evict();
623       }
624     }
625 
626     public void evict() {
627       synchronized(this) {
628         this.notify(); // FindBugs NN_NAKED_NOTIFY
629       }
630     }
631 
632     void shutdown() {
633       this.go = false;
634       interrupt();
635     }
636 
637     /**
638      * Used for the test.
639      */
640     boolean isEnteringRun() {
641       return this.enteringRun;
642     }
643   }

727       Path path = sfMap.get(sf);
728       if ( path != null) {
729         BlockCacheColumnFamilySummary lookup =
730           BlockCacheColumnFamilySummary.createFromStoreFilePath(path);
731         BlockCacheColumnFamilySummary bcse = bcs.get(lookup);
732         if (bcse == null) {
733           bcse = BlockCacheColumnFamilySummary.create(lookup);

738       }
739     }

741         new ArrayList<BlockCacheColumnFamilySummary>(bcs.values());
742     Collections.sort( list );
