77  /**
78   * Provides basic authorization checks for data access and administrative
79   * operations.
80   *
81   * <p>
82   * {@code AccessController} performs authorization checks for HBase operations
83   * based on:
84   * <ul>
85   *   <li>the identity of the user performing the operation</li>
86   *   <li>the scope over which the operation is performed, in increasing
87   *   specificity: global, table, column family, or qualifier</li>
88   *   <li>the type of action being performed (as mapped to
89   *   {@link Permission.Action} values)</li>
90   * </ul>
91   * If the authorization check fails, an {@link AccessDeniedException}
92   * will be thrown for the operation.
93   * </p>
94   *
95   * <p>
96   * To perform authorization checks, {@code AccessController} relies on the
97   * {@link org.apache.hadoop.hbase.ipc.SecureRpcEngine} being loaded to provide
98   * the user identities for remote requests.
99   * </p>
100  *
101  * <p>
102  * The access control lists used for authorization can be manipulated via the
103  * exposed {@link AccessControllerProtocol} implementation, and the associated
104  * {@code grant}, {@code revoke}, and {@code user_permission} HBase shell
105  * commands.
106  * </p>
107  */
108 public class AccessController extends BaseRegionObserver
109     implements MasterObserver, RegionServerObserver, AccessControllerProtocol,
110     AccessControlService.Interface, CoprocessorService {

197   void initialize(RegionCoprocessorEnvironment e) throws IOException {
198     final HRegion region = e.getRegion();
199 
200     Map<byte[],ListMultimap<String,TablePermission>> tables =
201         AccessControlLists.loadAll(region);
202     // For each table, write out the table's permissions to the respective
203     // znode for that table.
204     for (Map.Entry<byte[],ListMultimap<String,TablePermission>> t:
205       tables.entrySet()) {
206       byte[] table = t.getKey();
207       ListMultimap<String,TablePermission> perms = t.getValue();
208       byte[] serialized = AccessControlLists.writePermissionsAsBytes(perms, e.getConfiguration());
209       this.authManager.getZKPermissionWatcher().writeToZookeeper(table, serialized);
210     }
211   }

213   /**
214    * Writes all table ACLs for the tables in the given Map up into ZooKeeper
215    * znodes.  This is called to synchronize ACL changes following {@code _acl_}
216    * table updates.
217    */
218   void updateACL(RegionCoprocessorEnvironment e,
219       final Map<byte[], List<KeyValue>> familyMap) {
220     Set<byte[]> tableSet = new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR);
221     for (Map.Entry<byte[], List<KeyValue>> f : familyMap.entrySet()) {
222       List<KeyValue> kvs = f.getValue();
223       for (KeyValue kv: kvs) {
224         if (Bytes.equals(kv.getBuffer(), kv.getFamilyOffset(),
225             kv.getFamilyLength(), AccessControlLists.ACL_LIST_FAMILY, 0,
226             AccessControlLists.ACL_LIST_FAMILY.length)) {
227           tableSet.add(kv.getRow());
228         }
229       }
230     }
231 
232     ZKPermissionWatcher zkw = this.authManager.getZKPermissionWatcher();
233     Configuration conf = regionEnv.getConfiguration();
234     for (byte[] tableName: tableSet) {
235       try {
236         ListMultimap<String,TablePermission> perms =
237           AccessControlLists.getTablePermissions(conf, tableName);
238         byte[] serialized = AccessControlLists.writePermissionsAsBytes(perms, conf);
239         zkw.writeToZookeeper(tableName, serialized);
240       } catch (IOException ex) {
241         LOG.error("Failed updating permissions mirror for '" + tableName + "'", ex);
242       }
243     }
244   }

246   /**
247    * Check the current user for authorization to perform a specific action
248    * against the given set of row data.
249    *
250    * <p>Note: Ordering of the authorization checks
251    * has been carefully optimized to short-circuit the most common requests
252    * and minimize the amount of processing required.</p>
253    *
254    * @param permRequest the action being requested
255    * @param e the coprocessor environment
256    * @param families the map of column families to qualifiers present in
257    * the request
258    * @return
259    */
260   AuthResult permissionGranted(User user, Permission.Action permRequest,
261       RegionCoprocessorEnvironment e,
262       Map<byte [], ? extends Collection<?>> families) {
263     HRegionInfo hri = e.getRegion().getRegionInfo();
264     byte[] tableName = hri.getTableName();
265 
266     // 1. All users need read access to .META. and -ROOT- tables.
267     // this is a very common operation, so deal with it quickly.
268     if (hri.isRootRegion() || hri.isMetaRegion()) {
269       if (permRequest == Permission.Action.READ) {
270         return AuthResult.allow("All users allowed", user, permRequest, tableName);
271       }
272     }
273 
274     if (user == null) {
275       return AuthResult.deny("No user associated with request!", null, permRequest, tableName);
276     }
277 
278     // Users with CREATE/ADMIN rights need to modify .META. and _acl_ table
279     // e.g. When a new table is created a new entry in .META. is added,
280     // so the user need to be allowed to write on it.
281     // e.g. When a table is removed an entry is removed from .META. and _acl_
282     // and the user need to be allowed to write on both tables.
283     if (permRequest == Permission.Action.WRITE &&
284        (hri.isRootRegion() || hri.isMetaRegion() ||
285         Bytes.equals(tableName, AccessControlLists.ACL_GLOBAL_NAME)) &&
286        (authManager.authorize(user, Permission.Action.CREATE) ||
287         authManager.authorize(user, Permission.Action.ADMIN)))
288     {
289        return AuthResult.allow("Table permission granted", user, permRequest, tableName);
290     }
291 
292     // 2. check for the table-level, if successful we can short-circuit
293     if (authManager.authorize(user, tableName, (byte[])null, permRequest)) {
294       return AuthResult.allow("Table permission granted", user, permRequest, tableName);
295     }
296 
297     // 3. check permissions against the requested families
298     if (families != null && families.size() > 0) {
299       // all families must pass
300       for (Map.Entry<byte [], ? extends Collection<?>> family : families.entrySet()) {
301         // a) check for family level access
302         if (authManager.authorize(user, tableName, family.getKey(),
303             permRequest)) {
304           continue;  // family-level permission overrides per-qualifier
305         }
306 
307         // b) qualifier level access can still succeed
308         if ((family.getValue() != null) && (family.getValue().size() > 0)) {
309           if (family.getValue() instanceof Set) {
310             // for each qualifier of the family
311             Set<byte[]> familySet = (Set<byte[]>)family.getValue();
312             for (byte[] qualifier : familySet) {
313               if (!authManager.authorize(user, tableName, family.getKey(),
314                                          qualifier, permRequest)) {
315                 return AuthResult.deny("Failed qualifier check", user,
316                     permRequest, tableName, family.getKey(), qualifier);
317               }
318             }
319           } else if (family.getValue() instanceof List) { // List<KeyValue>
320             List<KeyValue> kvList = (List<KeyValue>)family.getValue();
321             for (KeyValue kv : kvList) {
322               if (!authManager.authorize(user, tableName, family.getKey(),
323                       kv.getQualifier(), permRequest)) {
324                 return AuthResult.deny("Failed qualifier check", user,
325                     permRequest, tableName, family.getKey(), kv.getQualifier());
326               }
327             }
328           }
329         } else {
330           // no qualifiers and family-level check already failed
331           return AuthResult.deny("Failed family check", user, permRequest,
332               tableName, family.getKey(), null);
333         }
334       }
335 
336       // all family checks passed
337       return AuthResult.allow("All family checks passed", user, permRequest,
338           tableName);
339     }
340 
341     // 4. no families to check and table level access failed
342     return AuthResult.deny("No families to check and table permission failed",
343         user, permRequest, tableName);
344   }

346   private void logResult(AuthResult result) {
347     if (AUDITLOG.isTraceEnabled()) {
348       InetAddress remoteAddr = null;
349       RequestContext ctx = RequestContext.get();
350       if (ctx != null) {
351         remoteAddr = ctx.getRemoteAddress();
352       }
353       AUDITLOG.trace("Access " + (result.isAllowed() ? "allowed" : "denied") +
354           " for user " + (result.getUser() != null ? result.getUser().getShortName() : "UNKNOWN") +
355           "; reason: " + result.getReason() +
356           "; remote address: " + (remoteAddr != null ? remoteAddr : "") +
357           "; context: " + result.toContextString());
358     }

362    * Returns the active user to which authorization checks should be applied.
363    * If we are in the context of an RPC call, the remote user is used,
364    * otherwise the currently logged in user is used.

366   private User getActiveUser() throws IOException {
367     User user = RequestContext.getRequestUser();

370       user = User.getCurrent();
371     }
372     return user;

375   /**
376    * Authorizes that the current user has any of the given permissions for the
377    * given table, column family and column qualifier.
378    * @param tableName Table requested
379    * @param family Column family requested
380    * @param qualifier Column qualifier requested
381    * @throws IOException if obtaining the current user fails
382    * @throws AccessDeniedException if user has no authorization
383    */
384   private void requirePermission(byte[] tableName, byte[] family, byte[] qualifier,
385       Action... permissions) throws IOException {
386     User user = getActiveUser();
387     AuthResult result = null;
388 
389     for (Action permission : permissions) {
390       if (authManager.authorize(user, tableName, family, qualifier, permission)) {
391         result = AuthResult.allow("Table permission granted", user, permission, tableName, family, qualifier);
392         break;
393       } else {
394         // rest of the world
395         result = AuthResult.deny("Insufficient permissions", user, permission, tableName, family, qualifier);
396       }
397     }
398     logResult(result);
399     if (!result.isAllowed()) {
400       throw new AccessDeniedException("Insufficient permissions " + result.toContextString());
401     }
402   }

404   /**
405    * Authorizes that the current user has global privileges for the given action.
406    * @param perm The action being requested
407    * @throws IOException if obtaining the current user fails
408    * @throws AccessDeniedException if authorization is denied
409    */
410   private void requirePermission(Permission.Action perm) throws IOException {
411     User user = getActiveUser();
412     if (authManager.authorize(user, perm)) {
413       logResult(AuthResult.allow("Global check allowed", user, perm, null));
414     } else {
415       logResult(AuthResult.deny("Global check failed", user, perm, null));
416       throw new AccessDeniedException("Insufficient permissions for user '" +
417           (user != null ? user.getShortName() : "null") +"' (global, action=" +
418           perm.toString() + ")");
419     }
420   }

422   /**
423    * Authorizes that the current user has permission to perform the given
424    * action on the set of table column families.
425    * @param perm Action that is required
426    * @param env The current coprocessor environment
427    * @param families The set of column families present/required in the request
428    * @throws AccessDeniedException if the authorization check failed
429    */
430   private void requirePermission(Permission.Action perm,
431         RegionCoprocessorEnvironment env, Collection<byte[]> families)
432       throws IOException {
433     // create a map of family-qualifier
434     HashMap<byte[], Set<byte[]>> familyMap = new HashMap<byte[], Set<byte[]>>();
435     for (byte[] family : families) {
436       familyMap.put(family, null);
437     }
438     requirePermission(perm, env, familyMap);
439   }

441   /**
442    * Authorizes that the current user has permission to perform the given
443    * action on the set of table column families.
444    * @param perm Action that is required
445    * @param env The current coprocessor environment
446    * @param families The map of column families-qualifiers.
447    * @throws AccessDeniedException if the authorization check failed
448    */
449   private void requirePermission(Permission.Action perm,
450         RegionCoprocessorEnvironment env,
451         Map<byte[], ? extends Collection<?>> families)
452       throws IOException {
453     User user = getActiveUser();
454     AuthResult result = permissionGranted(user, perm, env, families);
455     logResult(result);
456 
457     if (!result.isAllowed()) {
458       StringBuffer sb = new StringBuffer("");
459       if ((families != null && families.size() > 0)) {
460         for (byte[] familyName : families.keySet()) {
461           if (sb.length() != 0) {
462             sb.append(", ");
463           }
464           sb.append(Bytes.toString(familyName));
465         }
466       }
467       throw new AccessDeniedException("Insufficient permissions (table=" +
468         env.getRegion().getTableDesc().getNameAsString()+
469         ((families != null && families.size() > 0) ? ", family: " +
470         sb.toString() : "") + ", action=" +
471         perm.toString() + ")");
472     }
473   }

475   /**
476    * Returns <code>true</code> if the current user is allowed the given action
477    * over at least one of the column qualifiers in the given column families.
478    */
479   private boolean hasFamilyQualifierPermission(User user,
480       Permission.Action perm,
481       RegionCoprocessorEnvironment env,
482       Map<byte[], ? extends Set<byte[]>> familyMap)
483     throws IOException {
484     HRegionInfo hri = env.getRegion().getRegionInfo();
485     byte[] tableName = hri.getTableName();
486 
487     if (user == null) {
488       return false;
489     }
490 
491     if (familyMap != null && familyMap.size() > 0) {
492       // at least one family must be allowed
493       for (Map.Entry<byte[], ? extends Set<byte[]>> family :
494           familyMap.entrySet()) {
495         if (family.getValue() != null && !family.getValue().isEmpty()) {
496           for (byte[] qualifier : family.getValue()) {
497             if (authManager.matchPermission(user, tableName,
498                 family.getKey(), qualifier, perm)) {
499               return true;
500             }
501           }
502         } else {
503           if (authManager.matchPermission(user, tableName, family.getKey(),
504               perm)) {
505             return true;
506           }
507         }
508       }
509     } else if (LOG.isDebugEnabled()) {
510       LOG.debug("Empty family map passed for permission check");
511     }
512 
513     return false;
514   }

516   /* ---- MasterObserver implementation ---- */
517   public void start(CoprocessorEnvironment env) throws IOException {
518 
519     ZooKeeperWatcher zk = null;
520     if (env instanceof MasterCoprocessorEnvironment) {
521       // if running on HMaster
522       MasterCoprocessorEnvironment mEnv = (MasterCoprocessorEnvironment) env;
523       zk = mEnv.getMasterServices().getZooKeeper();      
524     } else if (env instanceof RegionServerCoprocessorEnvironment) {      
525       RegionServerCoprocessorEnvironment rsEnv = (RegionServerCoprocessorEnvironment) env;
526       zk = rsEnv.getRegionServerServices().getZooKeeper();      
527     } else if (env instanceof RegionCoprocessorEnvironment) {
528       // if running at region
529       regionEnv = (RegionCoprocessorEnvironment) env;
530       zk = regionEnv.getRegionServerServices().getZooKeeper();
531     }
532 
533     // If zk is null or IOException while obtaining auth manager,
534     // throw RuntimeException so that the coprocessor is unloaded.
535     if (zk != null) {
536       try {
537         this.authManager = TableAuthManager.get(zk, env.getConfiguration());
538       } catch (IOException ioe) {
539         throw new RuntimeException("Error obtaining TableAuthManager", ioe);
540       }
541     } else {
542       throw new RuntimeException("Error obtaining TableAuthManager, zk found null.");
543     }
544   }

856   @Override
857   public void preGet(final ObserverContext<RegionCoprocessorEnvironment> c,
858       final Get get, final List<KeyValue> result) throws IOException {
859     /*
860      if column family level checks fail, check for a qualifier level permission
861      in one of the families.  If it is present, then continue with the AccessControlFilter.
862       */
863     RegionCoprocessorEnvironment e = c.getEnvironment();
864     User requestUser = getActiveUser();
865     AuthResult authResult = permissionGranted(requestUser,
866         Permission.Action.READ, e, get.getFamilyMap());
867     if (!authResult.isAllowed()) {
868       if (hasFamilyQualifierPermission(requestUser,
869           Permission.Action.READ, e, get.getFamilyMap())) {
870         byte[] table = getTableName(e);
871         AccessControlFilter filter = new AccessControlFilter(authManager,
872             requestUser, table);
873 
874         // wrap any existing filter
875         if (get.getFilter() != null) {
876           FilterList wrapper = new FilterList(FilterList.Operator.MUST_PASS_ALL,
877               Lists.newArrayList(filter, get.getFilter()));
878           get.setFilter(wrapper);
879         } else {
880           get.setFilter(filter);
881         }
882         logResult(AuthResult.allow("Access allowed with filter", requestUser,
883             Permission.Action.READ, authResult.table));
884       } else {
885         logResult(authResult);
886         throw new AccessDeniedException("Insufficient permissions (table=" +
887           e.getRegion().getTableDesc().getNameAsString() + ", action=READ)");
888       }
889     } else {
890       // log auth success
891       logResult(authResult);
892     }
893   }

1016         logResult(authResult);
