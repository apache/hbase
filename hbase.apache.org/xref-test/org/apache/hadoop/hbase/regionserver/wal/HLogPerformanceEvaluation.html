54   * This class runs performance benchmarks for {@link HLog}.
55   * See usage for this tool by running:
56   * <code>$ hbase org.apache.hadoop.hbase.regionserver.wal.HLogPerformanceEvaluation -h</code>

69    private int valueSize = 512;

72    /**
73     * Perform HLog.append() of Put object, for the number of iterations requested.
74     * Keys and Vaues are generated randomly, the number of column familes,
75     * qualifiers and key/value size is tunable by the user.
76     */
77    class HLogPutBenchmark implements Runnable {
78      private final long numIterations;
79      private final int numFamilies;
80      private final boolean noSync;
81      private final HRegion region;
82      private final HTableDescriptor htd;
83  
84      HLogPutBenchmark(final HRegion region, final HTableDescriptor htd,
85          final long numIterations, final boolean noSync) {
86        this.numIterations = numIterations;
87        this.noSync = noSync;
88        this.numFamilies = htd.getColumnFamilies().length;
89        this.region = region;
90        this.htd = htd;
91      }
92  
93      public void run() {
94        byte[] key = new byte[keySize];
95        byte[] value = new byte[valueSize];
96        Random rand = new Random(Thread.currentThread().getId());
97        HLog hlog = region.getLog();
98  
99        try {
100         long startTime = System.currentTimeMillis();
101         for (int i = 0; i < numIterations; ++i) {
102           Put put = setupPut(rand, key, value, numFamilies);
103           long now = System.currentTimeMillis();
104           WALEdit walEdit = new WALEdit();
105           addFamilyMapToWALEdit(put.getFamilyMap(), walEdit);
106           HRegionInfo hri = region.getRegionInfo();
107           if (this.noSync) {
108             hlog.appendNoSync(hri, hri.getTableName(), walEdit,
109                               HConstants.DEFAULT_CLUSTER_ID, now, htd);
110           } else {
111             hlog.append(hri, hri.getTableName(), walEdit, now, htd);
112           }
113         }
114         long totalTime = (System.currentTimeMillis() - startTime);
115         logBenchmarkResult(Thread.currentThread().getName(), numIterations, totalTime);
116       } catch (Exception e) {
117         LOG.error(getClass().getSimpleName() + " Thread failed", e);
118       }
119     }
120   }

122   @Override
123   public int run(String[] args) throws Exception {
124     Path rootRegionDir = null;
125     int numThreads = 1;
126     long numIterations = 10000;
127     int numFamilies = 1;
128     boolean noSync = false;
129     boolean verify = false;
130     boolean verbose = false;
131     long roll = Long.MAX_VALUE;
132     // Process command line args
133     for (int i = 0; i < args.length; i++) {
134       String cmd = args[i];
135       try {
136         if (cmd.equals("-threads")) {
137           numThreads = Integer.parseInt(args[++i]);
138         } else if (cmd.equals("-iterations")) {
139           numIterations = Long.parseLong(args[++i]);
140         } else if (cmd.equals("-path")) {
141           rootRegionDir = new Path(args[++i]);
142         } else if (cmd.equals("-families")) {
143           numFamilies = Integer.parseInt(args[++i]);
144         } else if (cmd.equals("-qualifiers")) {
145           numQualifiers = Integer.parseInt(args[++i]);
146         } else if (cmd.equals("-keySize")) {
147           keySize = Integer.parseInt(args[++i]);
148         } else if (cmd.equals("-valueSize")) {
149           valueSize = Integer.parseInt(args[++i]);
150         } else if (cmd.equals("-nosync")) {
151           noSync = true;
152         } else if (cmd.equals("-verify")) {
153           verify = true;
154         } else if (cmd.equals("-verbose")) {
155           verbose = true;
156         } else if (cmd.equals("-roll")) {
157           roll = Long.parseLong(args[++i]);
158         } else if (cmd.equals("-h")) {
159           printUsageAndExit();
160         } else if (cmd.equals("--help")) {
161           printUsageAndExit();
162         } else {
163           System.err.println("UNEXPECTED: " + cmd);
164           printUsageAndExit();
165         }
166       } catch (Exception e) {
167         printUsageAndExit();
168       }
169     }
170 
171     // Run HLog Performance Evaluation
172     FileSystem fs = FileSystem.get(getConf());
173     LOG.info("" + fs);
174     try {
175       if (rootRegionDir == null) {
176         rootRegionDir = TEST_UTIL.getDataTestDir("HLogPerformanceEvaluation");
177       }
178       rootRegionDir = rootRegionDir.makeQualified(fs);
179       cleanRegionRootDir(fs, rootRegionDir);
180       // Initialize Table Descriptor
181       HTableDescriptor htd = createHTableDescriptor(numFamilies);
182       final long whenToRoll = roll;
183       HLog hlog = new FSHLog(fs, rootRegionDir, "wals", getConf()) {
184         int appends = 0;
185         protected void doWrite(HRegionInfo info, HLogKey logKey, WALEdit logEdit,
186             HTableDescriptor htd)
187         throws IOException {
188           this.appends++;
189           if (this.appends % whenToRoll == 0) {
190             LOG.info("Rolling after " + appends + " edits");
191             rollWriter();
192           }
193           super.doWrite(info, logKey, logEdit, htd);
194         };
195       };
196       hlog.rollWriter();
197       HRegion region = null;
198       try {
199         region = openRegion(fs, rootRegionDir, htd, hlog);
200         long putTime = runBenchmark(new HLogPutBenchmark(region, htd, numIterations, noSync), numThreads);
201         logBenchmarkResult("Summary: threads=" + numThreads + ", iterations=" + numIterations,
202           numIterations * numThreads, putTime);
203         if (region != null) {
204           closeRegion(region);
205           region = null;
206         }
207         if (verify) {
208           Path dir = ((FSHLog) hlog).getDir();
209           long editCount = 0;
210           for (FileStatus fss: fs.listStatus(dir)) {
211             editCount += verify(fss.getPath(), verbose);
212           }
213           long expected = numIterations * numThreads;
214           if (editCount != expected) {
215             throw new IllegalStateException("Counted=" + editCount + ", expected=" + expected);
216           }
217         }
218       } finally {
219         if (region != null) closeRegion(region);
220         // Remove the root dir for this test region
221         cleanRegionRootDir(fs, rootRegionDir);
222       }
223     } finally {
224       fs.close();
225     }
226 
227     return(0);
228   }

230   private static HTableDescriptor createHTableDescriptor(final int numFamilies) {
231     HTableDescriptor htd = new HTableDescriptor(TABLE_NAME);
232     for (int i = 0; i < numFamilies; ++i) {
233       HColumnDescriptor colDef = new HColumnDescriptor(FAMILY_PREFIX + i);
234       htd.addFamily(colDef);
235     }
236     return htd;
237   }

239   /**
240    * Verify the content of the WAL file.
241    * Verify that sequenceids are ascending and that the file has expected number
242    * of edits.
243    * @param wal
244    * @return Count of edits.
245    * @throws IOException
246    */
247   private long verify(final Path wal, final boolean verbose) throws IOException {
248     HLog.Reader reader = HLogFactory.createReader(wal.getFileSystem(getConf()), 
249         wal, getConf());
250     long previousSeqid = -1;
251     long count = 0;
252     try {
253       while (true) {
254         Entry e = reader.next();
255         if (e == null) break;
256         count++;
257         long seqid = e.getKey().getLogSeqNum();
258         if (verbose) LOG.info("seqid=" + seqid);
259         if (previousSeqid >= seqid) {
260           throw new IllegalStateException("wal=" + wal.getName() +
261             ", previousSeqid=" + previousSeqid + ", seqid=" + seqid);
262         }
263         previousSeqid = seqid;
264       }
265     } finally {
266       reader.close();
267     }
268     return count;
269   }

271   private static void logBenchmarkResult(String testName, long numTests, long totalTime) {

273     LOG.info(String.format("%s took %.3fs %.3fops/s", testName, tsec, numTests / tsec));

321   private Put setupPut(Random rand, byte[] key, byte[] value, final int numFamilies) {
322     rand.nextBytes(key);
323     Put put = new Put(key);
324     for (int cf = 0; cf < numFamilies; ++cf) {
325       for (int q = 0; q < numQualifiers; ++q) {
326         rand.nextBytes(value);
327         put.add(Bytes.toBytes(FAMILY_PREFIX + cf), Bytes.toBytes(QUALIFIER_PREFIX + q), value);
328       }
329     }
330     return put;
331   }

333   private void addFamilyMapToWALEdit(Map<byte[], List<KeyValue>> familyMap, WALEdit walEdit) {
334     for (List<KeyValue> edits : familyMap.values()) {
335       for (KeyValue kv : edits) {
336         walEdit.add(kv);
337       }
338     }
339   }

341   private long runBenchmark(Runnable runnable, final int numThreads) throws InterruptedException {
342     Thread[] threads = new Thread[numThreads];
343     long startTime = System.currentTimeMillis();
344     for (int i = 0; i < numThreads; ++i) {
345       threads[i] = new Thread(runnable);
346       threads[i].start();
347     }
348     for (Thread t : threads) t.join();
349     long endTime = System.currentTimeMillis();
350     return(endTime - startTime);
351   }
