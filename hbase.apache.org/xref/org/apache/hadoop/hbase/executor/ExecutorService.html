45  /**
46   * This is a generic executor service. This component abstracts a
47   * threadpool, a queue to which {@link EventHandler.EventType}s can be submitted,
48   * and a <code>Runnable</code> that handles the object that is added to the queue.
49   *
50   * <p>In order to create a new service, create an instance of this class and
51   * then do: <code>instance.startExecutorService("myService");</code>.  When done
52   * call {@link #shutdown()}.
53   *
54   * <p>In order to use the service created above, call
55   * {@link #submit(EventHandler)}. Register pre- and post- processing listeners
56   * by registering your implementation of {@link EventHandler.EventHandlerListener}
57   * with {@link #registerListener(EventHandler.EventType, EventHandler.EventHandlerListener)}.  Be sure
58   * to deregister your listener when done via {@link #unregisterListener(EventHandler.EventType)}.
59   */
60  @InterfaceAudience.Private
61  public class ExecutorService {
62    private static final Log LOG = LogFactory.getLog(ExecutorService.class);
63  
64    // hold the all the executors created in a map addressable by their names
65    private final ConcurrentHashMap<String, Executor> executorMap =
66      new ConcurrentHashMap<String, Executor>();

69    private ConcurrentHashMap<EventHandler.EventType, EventHandlerListener> eventHandlerListeners =
70      new ConcurrentHashMap<EventHandler.EventType, EventHandlerListener>();

117   /**
118    * Start an executor service with a given name. If there was a service already
119    * started with the same name, this throws a RuntimeException.
120    * @param name Name of the service to start.
121    */
122   void startExecutorService(String name, int maxThreads) {
123     if (this.executorMap.get(name) != null) {
124       throw new RuntimeException("An executor service with the name " + name +
125         " is already running!");
126     }
127     Executor hbes = new Executor(name, maxThreads, this.eventHandlerListeners);
128     if (this.executorMap.putIfAbsent(name, hbes) != null) {
129       throw new RuntimeException("An executor service with the name " + name +
130       " is already running (2)!");
131     }
132     LOG.debug("Starting executor service name=" + name +
133       ", corePoolSize=" + hbes.threadPoolExecutor.getCorePoolSize() +
134       ", maxPoolSize=" + hbes.threadPoolExecutor.getMaximumPoolSize());
135   }
136 
137   boolean isExecutorServiceRunning(String name) {
138     return this.executorMap.containsKey(name);
139   }

141   public void shutdown() {
142     for(Entry<String, Executor> entry: this.executorMap.entrySet()) {
143       List<Runnable> wasRunning =
144         entry.getValue().threadPoolExecutor.shutdownNow();
145       if (!wasRunning.isEmpty()) {
146         LOG.info(entry.getValue() + " had " + wasRunning + " on shutdown");
147       }
148     }
149     this.executorMap.clear();
150   }

162   public void startExecutorService(final ExecutorType type, final int maxThreads) {
163     String name = type.getExecutorName(this.servername);
164     if (isExecutorServiceRunning(name)) {
165       LOG.debug("Executor service " + toString() + " already running on " +
166         this.servername);
167       return;
168     }
169     startExecutorService(name, maxThreads);
170   }

172   public void submit(final EventHandler eh) {
173     Executor executor = getExecutor(eh.getEventType().getExecutorServiceType());
174     if (executor == null) {
175       // This happens only when events are submitted after shutdown() was
176       // called, so dropping them should be "ok" since it means we're
177       // shutting down.
178       LOG.error("Cannot submit [" + eh + "] because the executor is missing." +
179         " Is this process shutting down?");
180     } else {
181       executor.submit(eh);
182     }
183   }

207   public Map<String, ExecutorStatus> getAllExecutorStatuses() {

215   /**
216    * Executor instance.
217    */
218   static class Executor {
219     // how long to retain excess threads
220     static final long keepAliveTimeInMillis = 1000;
221     // the thread pool executor that services the requests
222     final TrackingThreadPoolExecutor threadPoolExecutor;
223     // work queue to use - unbounded queue
224     final BlockingQueue<Runnable> q = new LinkedBlockingQueue<Runnable>();
225     private final String name;
226     private final Map<EventHandler.EventType, EventHandlerListener> eventHandlerListeners;
227     private static final AtomicLong seqids = new AtomicLong(0);
228     private final long id;
229 
230     protected Executor(String name, int maxThreads,
231         final Map<EventHandler.EventType, EventHandlerListener> eventHandlerListeners) {
232       this.id = seqids.incrementAndGet();
233       this.name = name;
234       this.eventHandlerListeners = eventHandlerListeners;
235       // create the thread pool executor
236       this.threadPoolExecutor = new TrackingThreadPoolExecutor(
237           maxThreads, maxThreads,
238           keepAliveTimeInMillis, TimeUnit.MILLISECONDS, q);
239       // name the threads for this threadpool
240       ThreadFactoryBuilder tfb = new ThreadFactoryBuilder();
241       tfb.setNameFormat(this.name + "-%d");
242       this.threadPoolExecutor.setThreadFactory(tfb.build());
243     }
244 
245     /**
246      * Submit the event to the queue for handling.
247      * @param event
248      */
249     void submit(final EventHandler event) {
250       // If there is a listener for this type, make sure we call the before
251       // and after process methods.
252       EventHandlerListener listener =
253         this.eventHandlerListeners.get(event.getEventType());
254       if (listener != null) {
255         event.setListener(listener);
256       }
257       this.threadPoolExecutor.execute(event);
258     }
259     
260     public String toString() {
261       return getClass().getSimpleName() + "-" + id + "-" + name;
262     }
263 
264     public ExecutorStatus getStatus() {
265       List<EventHandler> queuedEvents = Lists.newArrayList();
266       for (Runnable r : q) {
267         if (!(r instanceof EventHandler)) {
268           LOG.warn("Non-EventHandler " + r + " queued in " + name);
269           continue;
270         }
271         queuedEvents.add((EventHandler)r);
272       }
273       
274       List<RunningEventStatus> running = Lists.newArrayList();
275       for (Map.Entry<Thread, Runnable> e :
276           threadPoolExecutor.getRunningTasks().entrySet()) {
277         Runnable r = e.getValue();
278         if (!(r instanceof EventHandler)) {
279           LOG.warn("Non-EventHandler " + r + " running in " + name);
280           continue;
281         }
282         running.add(new RunningEventStatus(e.getKey(), (EventHandler)r));
283       }
284       
285       return new ExecutorStatus(this, queuedEvents, running);
286     }
287   }

289   /**
290    * A subclass of ThreadPoolExecutor that keeps track of the Runnables that
291    * are executing at any given point in time.
292    */
293   static class TrackingThreadPoolExecutor extends ThreadPoolExecutor {
294     private ConcurrentMap<Thread, Runnable> running = Maps.newConcurrentMap(); 
295       
296     public TrackingThreadPoolExecutor(int corePoolSize, int maximumPoolSize,
297         long keepAliveTime, TimeUnit unit, BlockingQueue<Runnable> workQueue) {
298       super(corePoolSize, maximumPoolSize, keepAliveTime, unit, workQueue);
299     }
300 
301     @Override
302     protected void afterExecute(Runnable r, Throwable t) {
303       super.afterExecute(r, t);
304       running.remove(Thread.currentThread());
305     }
306 
307     @Override
308     protected void beforeExecute(Thread t, Runnable r) {
309       Runnable oldPut = running.put(t, r);
310       assert oldPut == null : "inconsistency for thread " + t;
311       super.beforeExecute(t, r);
312     }
313    
314     /**
315      * @return a map of the threads currently running tasks
316      * inside this executor. Each key is an active thread,
317      * and the value is the task that is currently running.
318      * Note that this is not a stable snapshot of the map.
319      */
320     public ConcurrentMap<Thread, Runnable> getRunningTasks() {
321       return running;
322     }
323   }

325   /**
326    * A snapshot of the status of a particular executor. This includes
327    * the contents of the executor's pending queue, as well as the
328    * threads and events currently being processed.
329    *
330    * This is a consistent snapshot that is immutable once constructed.
331    */
332   public static class ExecutorStatus {
333     final Executor executor;
334     final List<EventHandler> queuedEvents;
335     final List<RunningEventStatus> running;
336 
337     ExecutorStatus(Executor executor,
338         List<EventHandler> queuedEvents,
339         List<RunningEventStatus> running) {
340       this.executor = executor;
341       this.queuedEvents = queuedEvents;
342       this.running = running;
343     }
344    
345     /**
346      * Dump a textual representation of the executor's status
347      * to the given writer.
348      *
349      * @param out the stream to write to
350      * @param indent a string prefix for each line, used for indentation
351      */
352     public void dumpTo(Writer out, String indent) throws IOException {
353       out.write(indent + "Status for executor: " + executor + "\n");
354       out.write(indent + "=======================================\n");
355       out.write(indent + queuedEvents.size() + " events queued, " +
356           running.size() + " running\n");
357       if (!queuedEvents.isEmpty()) {
358         out.write(indent + "Queued:\n");
359         for (EventHandler e : queuedEvents) {
360           out.write(indent + "  " + e + "\n");
361         }
362         out.write("\n");
363       }
364       if (!running.isEmpty()) {
365         out.write(indent + "Running:\n");
366         for (RunningEventStatus stat : running) {
367           out.write(indent + "  Running on thread '" +
368               stat.threadInfo.getThreadName() +
369               "': " + stat.event + "\n");
370           out.write(ThreadMonitoring.formatThreadInfo(
371               stat.threadInfo, indent + "  "));
372           out.write("\n");
373         }
374       }
375       out.flush();
376     }
377   }
