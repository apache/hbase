141   private interface Expr {
142     public long eval();
143   }
144 
145   private void waitForCounter(final AtomicLong ctr, long oldval, long newval, long timems) {
146     Expr e = new Expr() {
147       public long eval() {
148         return ctr.get();
149       }
150     };
151     waitForCounter(e, oldval, newval, timems);
152     return;
153   }
154 
155   private void waitForCounter(Expr e, long oldval, long newval,
156       long timems) {
157     long curt = System.currentTimeMillis();
158     long endt = curt + timems;
159     while (curt < endt) {
160       if (e.eval() == oldval) {
161         try {
162           Thread.sleep(10);
163         } catch (InterruptedException eintr) {
164         }
165         curt = System.currentTimeMillis();
166       } else {
167         assertEquals(newval, e.eval());
168         return;
169       }
170     }
171     assertTrue(false);
172   }

174   private String submitTaskAndWait(TaskBatch batch, String name)
175   throws KeeperException, InterruptedException {
176     String tasknode = ZKSplitLog.getEncodedNodeName(zkw, name);
177     NodeCreationListener listener = new NodeCreationListener(zkw, tasknode);
178     zkw.registerListener(listener);
179     ZKUtil.watchAndCheckExists(zkw, tasknode);
180 
181     slm.enqueueSplitTask(name, batch);
182     assertEquals(1, batch.installed);
183     assertTrue(slm.findOrCreateOrphanTask(tasknode).batch == batch);
184     assertEquals(1L, tot_mgr_node_create_queued.get());
185 
186     LOG.debug("waiting for task node creation");
187     listener.waitForCreation();
188     LOG.debug("task created");
189     return tasknode;
190   }

403   public void testUnassignedTimeout() throws Exception {
404     LOG.info("TestUnassignedTimeout - iff all tasks are unassigned then" +
405         " resubmit");
406 
407     // create an orphan task in OWNED state
408     String tasknode1 = ZKSplitLog.getEncodedNodeName(zkw, "orphan/1");
409     final ServerName worker1 = new ServerName("worker1,1,1");
410     SplitLogTask slt = new SplitLogTask.Owned(worker1);
411     zkw.getRecoverableZooKeeper().create(tasknode1, slt.toByteArray(), Ids.OPEN_ACL_UNSAFE,
412         CreateMode.PERSISTENT);
413 
414     slm = new SplitLogManager(zkw, conf, stopper, master, DUMMY_MASTER, null);
415     slm.finishInitialization();
416     waitForCounter(tot_mgr_orphan_task_acquired, 0, 1, to/2);
417 
418     // submit another task which will stay in unassigned mode
419     TaskBatch batch = new TaskBatch();
420     submitTaskAndWait(batch, "foo/1");
421 
422     // keep updating the orphan owned node every to/2 seconds
423     for (int i = 0; i < (3 * to)/100; i++) {
424       Thread.sleep(100);
425       final ServerName worker2 = new ServerName("worker1,1,1");
426       slt = new SplitLogTask.Owned(worker2);
427       ZKUtil.setData(zkw, tasknode1, slt.toByteArray());
428     }
429 
430     // since we have stopped heartbeating the owned node therefore it should
431     // get resubmitted
432     LOG.info("waiting for manager to resubmit the orphan task");
433     waitForCounter(tot_mgr_resubmit, 0, 1, to + to/2);
434 
435     // now all the nodes are unassigned. manager should post another rescan
436     waitForCounter(tot_mgr_resubmit_unassigned, 0, 1, 2 * to + to/2);
437   }

506   @Test(timeout=45000)
507   public void testVanishingTaskZNode() throws Exception {
508     LOG.info("testVanishingTaskZNode");
509 
510     conf.setInt("hbase.splitlog.manager.unassigned.timeout", 0);
511 
512     slm = new SplitLogManager(zkw, conf, stopper, master, DUMMY_MASTER, null);
513     slm.finishInitialization();
514     FileSystem fs = TEST_UTIL.getTestFileSystem();
515     final Path logDir = new Path(fs.getWorkingDirectory(),
516         UUID.randomUUID().toString());
517     fs.mkdirs(logDir);
518     Thread thread = null;
519     try {
520       Path logFile = new Path(logDir, UUID.randomUUID().toString());
521       fs.createNewFile(logFile);
522       thread = new Thread() {
523         public void run() {
524           try {
525             // this call will block because there are no SplitLogWorkers,
526             // until the task znode is deleted below. Then the call will
527             // complete successfully, assuming the log is split.
528             slm.splitLogDistributed(logDir);
529           } catch (Exception e) {
530             LOG.warn("splitLogDistributed failed", e);
531           }
532         }
533       };
534       thread.start();
535       waitForCounter(tot_mgr_node_create_result, 0, 1, 10000);
