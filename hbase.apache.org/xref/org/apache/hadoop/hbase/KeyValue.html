44  /**
45   * An HBase Key/Value. This is the fundamental HBase Type.
46   * <p>
47   * If being used client-side, the primary methods to access individual fields are {@link #getRow()},
48   * {@link #getFamily()}, {@link #getQualifier()}, {@link #getTimestamp()}, and {@link #getValue()}.
49   * These methods allocate new byte arrays and return copies. Avoid their use server-side.
50   * <p>
51   * Instances of this class are immutable. They do not implement Comparable but Comparators are
52   * provided. Comparators change with context, whether user table or a catalog table comparison. Its
53   * critical you use the appropriate comparator. There are Comparators for KeyValue instances and
54   * then for just the Key portion of a KeyValue used mostly by HFile.
55   * <p>
56   * KeyValue wraps a byte array and takes offsets and lengths into passed array at where to start
57   * interpreting the content as KeyValue. The KeyValue format inside a byte array is:
58   * <code>&lt;keylength> &lt;valuelength> &lt;key> &lt;value></code> Key is further decomposed as:
59   * <code>&lt;rowlength> &lt;row> &lt;columnfamilylength> &lt;columnfamily> &lt;columnqualifier> &lt;timestamp> &lt;keytype></code>
60   * The <code>rowlength</code> maximum is <code>Short.MAX_SIZE</code>, column family length maximum
61   * is <code>Byte.MAX_SIZE</code>, and column qualifier + key length must be <
62   * <code>Integer.MAX_SIZE</code>. The column does not contain the family/qualifier delimiter,
63   * {@link #COLUMN_FAMILY_DELIMITER}
64   */

507   /**
508    * Create an empty byte[] representing a KeyValue
509    * All lengths are preset and can be filled in later.
510    * @param rlength
511    * @param flength
512    * @param qlength
513    * @param timestamp
514    * @param type
515    * @param vlength
516    * @return The newly created byte array.
517    */
518   static byte[] createEmptyByteArray(final int rlength, int flength,
519       int qlength, final long timestamp, final Type type, int vlength) {
520     if (rlength > Short.MAX_VALUE) {
521       throw new IllegalArgumentException("Row > " + Short.MAX_VALUE);
522     }
523     if (flength > Byte.MAX_VALUE) {
524       throw new IllegalArgumentException("Family > " + Byte.MAX_VALUE);
525     }
526     // Qualifier length
527     if (qlength > Integer.MAX_VALUE - rlength - flength) {
528       throw new IllegalArgumentException("Qualifier > " + Integer.MAX_VALUE);
529     }
530     // Key length
531     long longkeylength = getKeyDataStructureSize(rlength, flength, qlength);
532     if (longkeylength > Integer.MAX_VALUE) {
533       throw new IllegalArgumentException("keylength " + longkeylength + " > " +
534         Integer.MAX_VALUE);
535     }
536     int keylength = (int)longkeylength;
537     // Value length
538     if (vlength > HConstants.MAXIMUM_VALUE_LENGTH) { // FindBugs INT_VACUOUS_COMPARISON
539       throw new IllegalArgumentException("Valuer > " +
540           HConstants.MAXIMUM_VALUE_LENGTH);
541     }
542 
543     // Allocate right-sized byte array.
544     byte [] bytes =
545         new byte[(int) getKeyValueDataStructureSize(rlength, flength, qlength, vlength)];
546     // Write the correct size markers
547     int pos = 0;
548     pos = Bytes.putInt(bytes, pos, keylength);
549     pos = Bytes.putInt(bytes, pos, vlength);
550     pos = Bytes.putShort(bytes, pos, (short)(rlength & 0x0000ffff));
551     pos += rlength;
552     pos = Bytes.putByte(bytes, pos, (byte)(flength & 0x0000ff));
553     pos += flength + qlength;
554     pos = Bytes.putLong(bytes, pos, timestamp);
555     pos = Bytes.putByte(bytes, pos, type.getCode());
556     return bytes;
557   }

638   /**
639    * Checks the parameters passed to a constructor.
640    *
641    * @param row row key
642    * @param rlength row length
643    * @param family family name
644    * @param flength family length
645    * @param qualifier column qualifier
646    * @param qlength qualifier length
647    * @param value column value
648    * @param vlength value length
649    *
650    * @throws IllegalArgumentException an illegal value was passed
651    */
652   private static void checkParameters(final byte [] row, final int rlength,
653       final byte [] family, int flength,
654       final byte [] qualifier, int qlength,
655       final byte [] value, int vlength)
656           throws IllegalArgumentException {
657 
658     if (rlength > Short.MAX_VALUE) {
659       throw new IllegalArgumentException("Row > " + Short.MAX_VALUE);
660     }
661     if (row == null) {
662       throw new IllegalArgumentException("Row is null");
663     }
664     // Family length
665     flength = family == null ? 0 : flength;
666     if (flength > Byte.MAX_VALUE) {
667       throw new IllegalArgumentException("Family > " + Byte.MAX_VALUE);
668     }
669     // Qualifier length
670     qlength = qualifier == null ? 0 : qlength;
671     if (qlength > Integer.MAX_VALUE - rlength - flength) {
672       throw new IllegalArgumentException("Qualifier > " + Integer.MAX_VALUE);
673     }
674     // Key length
675     long longKeyLength = getKeyDataStructureSize(rlength, flength, qlength);

689    * Write KeyValue format into the provided byte array.
690    *
691    * @param buffer the bytes buffer to use
692    * @param boffset buffer offset
693    * @param row row key
694    * @param roffset row offset
695    * @param rlength row length
696    * @param family family name
697    * @param foffset family offset
698    * @param flength family length
699    * @param qualifier column qualifier
700    * @param qoffset qualifier offset
701    * @param qlength qualifier length
702    * @param timestamp version timestamp
703    * @param type key type
704    * @param value column value
705    * @param voffset value offset
706    * @param vlength value length
707    *
708    * @return The number of useful bytes in the buffer.
709    *
710    * @throws IllegalArgumentException an illegal value was passed or there is insufficient space
711    * remaining in the buffer
712    */
713   static int writeByteArray(byte [] buffer, final int boffset,
714       final byte [] row, final int roffset, final int rlength,
715       final byte [] family, final int foffset, int flength,
716       final byte [] qualifier, final int qoffset, int qlength,
717       final long timestamp, final Type type,
718       final byte [] value, final int voffset, int vlength) {
719 
720     checkParameters(row, rlength, family, flength, qualifier, qlength, value, vlength);
721 
722     int keyLength = (int) getKeyDataStructureSize(rlength, flength, qlength);
723     int keyValueLength = (int) getKeyValueDataStructureSize(rlength, flength, qlength, vlength);
724     if (keyValueLength > buffer.length - boffset) {
725       throw new IllegalArgumentException("Buffer size " + (buffer.length - boffset) + " < " +
726           keyValueLength);
727     }
728 
729     // Write key, value and key row length.
730     int pos = boffset;
731     pos = Bytes.putInt(buffer, pos, keyLength);
732     pos = Bytes.putInt(buffer, pos, vlength);
733     pos = Bytes.putShort(buffer, pos, (short)(rlength & 0x0000ffff));
734     pos = Bytes.putBytes(buffer, pos, row, roffset, rlength);
735     pos = Bytes.putByte(buffer, pos, (byte) (flength & 0x0000ff));
736     if (flength != 0) {
737       pos = Bytes.putBytes(buffer, pos, family, foffset, flength);
738     }
739     if (qlength != 0) {
740       pos = Bytes.putBytes(buffer, pos, qualifier, qoffset, qlength);
741     }
742     pos = Bytes.putLong(buffer, pos, timestamp);
743     pos = Bytes.putByte(buffer, pos, type.getCode());
744     if (value != null && value.length > 0) {
745       pos = Bytes.putBytes(buffer, pos, value, voffset, vlength);
746     }
747 
748     return keyValueLength;
749   }

957   public static String keyToString(final byte [] b, final int o, final int l) {
958     if (b == null) return "";
959     int rowlength = Bytes.toShort(b, o);
960     String row = Bytes.toStringBinary(b, o + Bytes.SIZEOF_SHORT, rowlength);
961     int columnoffset = o + Bytes.SIZEOF_SHORT + 1 + rowlength;
962     int familylength = b[columnoffset - 1];
963     int columnlength = l - ((columnoffset - o) + TIMESTAMP_TYPE_SIZE);
964     String family = familylength == 0? "":
965       Bytes.toStringBinary(b, columnoffset, familylength);
966     String qualifier = columnlength == 0? "":
967       Bytes.toStringBinary(b, columnoffset + familylength,
968       columnlength - familylength);
969     long timestamp = Bytes.toLong(b, o + (l - TIMESTAMP_TYPE_SIZE));
970     String timestampStr = humanReadableTimestamp(timestamp);
971     byte type = b[o + l - 1];
972     return row + "/" + family +
973       (family != null && family.length() > 0? ":" :"") +
974       qualifier + "/" + timestampStr + "/" + Type.codeToType(type);
975   }

1454   public SplitKeyValue split() {
1455     SplitKeyValue split = new SplitKeyValue();
1456     int splitOffset = this.offset;
1457     int keyLen = Bytes.toInt(bytes, splitOffset);
1458     splitOffset += Bytes.SIZEOF_INT;
1459     int valLen = Bytes.toInt(bytes, splitOffset);
1460     splitOffset += Bytes.SIZEOF_INT;
1461     short rowLen = Bytes.toShort(bytes, splitOffset);
1462     splitOffset += Bytes.SIZEOF_SHORT;
1463     byte [] row = new byte[rowLen];
1464     System.arraycopy(bytes, splitOffset, row, 0, rowLen);
1465     splitOffset += rowLen;
1466     split.setRow(row);
1467     byte famLen = bytes[splitOffset];
1468     splitOffset += Bytes.SIZEOF_BYTE;
1469     byte [] family = new byte[famLen];
1470     System.arraycopy(bytes, splitOffset, family, 0, famLen);
1471     splitOffset += famLen;
1472     split.setFamily(family);
1473     int colLen = keyLen -
1474       (rowLen + famLen + Bytes.SIZEOF_SHORT + Bytes.SIZEOF_BYTE +
1475       Bytes.SIZEOF_LONG + Bytes.SIZEOF_BYTE);
1476     byte [] qualifier = new byte[colLen];
1477     System.arraycopy(bytes, splitOffset, qualifier, 0, colLen);
1478     splitOffset += colLen;
1479     split.setQualifier(qualifier);
1480     byte [] timestamp = new byte[Bytes.SIZEOF_LONG];
1481     System.arraycopy(bytes, splitOffset, timestamp, 0, Bytes.SIZEOF_LONG);
1482     splitOffset += Bytes.SIZEOF_LONG;
1483     split.setTimestamp(timestamp);
1484     byte [] type = new byte[1];
1485     type[0] = bytes[splitOffset];
1486     splitOffset += Bytes.SIZEOF_BYTE;
1487     split.setType(type);
1488     byte [] value = new byte[valLen];
1489     System.arraycopy(bytes, splitOffset, value, 0, valLen);
1490     split.setValue(value);

2464   public static class KeyComparator
2465       implements RawComparator<byte []>, SamePrefixComparator<byte[]> {
2466     volatile boolean ignoreTimestamp = false;
2467     volatile boolean ignoreType = false;
2468 
2469     public int compare(byte[] left, int loffset, int llength, byte[] right,
2470         int roffset, int rlength) {
2471       // Compare row
2472       short lrowlength = Bytes.toShort(left, loffset);
2473       short rrowlength = Bytes.toShort(right, roffset);
2474       int compare = compareRows(left, loffset + Bytes.SIZEOF_SHORT,
2475           lrowlength, right, roffset + Bytes.SIZEOF_SHORT, rrowlength);
2476       if (compare != 0) {
2477         return compare;
2478       }
2479 
2480       // Compare the rest of the two KVs without making any assumptions about
2481       // the common prefix. This function will not compare rows anyway, so we
2482       // don't need to tell it that the common prefix includes the row.
2483       return compareWithoutRow(0, left, loffset, llength, right, roffset,
2484           rlength, rrowlength);

2487     /**
2488      * Compare the two key-values, ignoring the prefix of the given length
2489      * that is known to be the same between the two.
2490      * @param commonPrefix the prefix length to ignore
2491      */
2492     @Override
2493     public int compareIgnoringPrefix(int commonPrefix, byte[] left,
2494         int loffset, int llength, byte[] right, int roffset, int rlength) {
2495       // Compare row
2496       short lrowlength = Bytes.toShort(left, loffset);
2497       short rrowlength;
2498 
2499       int comparisonResult = 0;
2500       if (commonPrefix < ROW_LENGTH_SIZE) {
2501         // almost nothing in common
2502         rrowlength = Bytes.toShort(right, roffset);
2503         comparisonResult = compareRows(left, loffset + ROW_LENGTH_SIZE,
2504             lrowlength, right, roffset + ROW_LENGTH_SIZE, rrowlength);
2505       } else { // the row length is the same
2506         rrowlength = lrowlength;
2507         if (commonPrefix < ROW_LENGTH_SIZE + rrowlength) {
2508           // The rows are not the same. Exclude the common prefix and compare
2509           // the rest of the two rows.
2510           int common = commonPrefix - ROW_LENGTH_SIZE;
2511           comparisonResult = compareRows(
2512               left, loffset + common + ROW_LENGTH_SIZE, lrowlength - common,
2513               right, roffset + common + ROW_LENGTH_SIZE, rrowlength - common);
2514         }
2515       }
2516       if (comparisonResult != 0) {
2517         return comparisonResult;
2518       }
2519 
2520       assert lrowlength == rrowlength;
2521 
2522       return compareWithoutRow(commonPrefix, left, loffset, llength, right,
2523           roffset, rlength, lrowlength);
2524     }

2526     /**
2527      * Compare columnFamily, qualifier, timestamp, and key type (everything
2528      * except the row). This method is used both in the normal comparator and
2529      * the "same-prefix" comparator. Note that we are assuming that row portions
2530      * of both KVs have already been parsed and found identical, and we don't
2531      * validate that assumption here.
2532      * @param commonPrefix
2533      *          the length of the common prefix of the two key-values being
2534      *          compared, including row length and row
2535      */
2536     private int compareWithoutRow(int commonPrefix, byte[] left, int loffset,
2537         int llength, byte[] right, int roffset, int rlength, short rowlength) {
2538       /***
2539        * KeyValue Format and commonLength:
2540        * |_keyLen_|_valLen_|_rowLen_|_rowKey_|_famiLen_|_fami_|_Quali_|....
2541        * ------------------|-------commonLength--------|--------------
2542        */
2543       int commonLength = ROW_LENGTH_SIZE + FAMILY_LENGTH_SIZE + rowlength;
2544 
2545       // commonLength + TIMESTAMP_TYPE_SIZE
2546       int commonLengthWithTSAndType = TIMESTAMP_TYPE_SIZE + commonLength;
2547       // ColumnFamily + Qualifier length.
2548       int lcolumnlength = llength - commonLengthWithTSAndType;
2549       int rcolumnlength = rlength - commonLengthWithTSAndType;
2550 
2551       byte ltype = left[loffset + (llength - 1)];
2552       byte rtype = right[roffset + (rlength - 1)];
2553 
2554       // If the column is not specified, the "minimum" key type appears the
2555       // latest in the sorted order, regardless of the timestamp. This is used
2556       // for specifying the last key/value in a given row, because there is no
2557       // "lexicographically last column" (it would be infinitely long). The
2558       // "maximum" key type does not need this behavior.
2559       if (lcolumnlength == 0 && ltype == Type.Minimum.getCode()) {
2560         // left is "bigger", i.e. it appears later in the sorted order
2561         return 1;
2562       }
2563       if (rcolumnlength == 0 && rtype == Type.Minimum.getCode()) {
2564         return -1;
2565       }
2566 
2567       int lfamilyoffset = commonLength + loffset;
2568       int rfamilyoffset = commonLength + roffset;
2569 
2570       // Column family length.
2571       int lfamilylength = left[lfamilyoffset - 1];
2572       int rfamilylength = right[rfamilyoffset - 1];
2573       // If left family size is not equal to right family size, we need not
2574       // compare the qualifiers.
2575       boolean sameFamilySize = (lfamilylength == rfamilylength);
2576       int common = 0;
2577       if (commonPrefix > 0) {
2578         common = Math.max(0, commonPrefix - commonLength);
2579         if (!sameFamilySize) {
2580           // Common should not be larger than Math.min(lfamilylength,
2581           // rfamilylength).
2582           common = Math.min(common, Math.min(lfamilylength, rfamilylength));
2583         } else {
2584           common = Math.min(common, Math.min(lcolumnlength, rcolumnlength));
2585         }
2586       }
