19  package org.apache.hadoop.hbase.io;
20  
21  import java.io.DataInput;
22  import java.io.IOException;
23  import java.io.InputStream;
24  
25  import org.apache.hadoop.classification.InterfaceAudience;
26  
27  /**
28   * An InputStream that wraps a DataInput.
29   * @see DataOutputOutputStream
30   */
31  @InterfaceAudience.Private
32  public class DataInputInputStream extends InputStream {
33  
34    private DataInput in;
35  
36    /**
37     * Construct an InputStream from the given DataInput. If 'in'
38     * is already an InputStream, simply returns it. Otherwise, wraps
39     * it in an InputStream.
40     * @param in the DataInput to wrap
41     * @return an InputStream instance that reads from 'in'
42     */
43    public static InputStream constructInputStream(DataInput in) {
44      if (in instanceof InputStream) {
45        return (InputStream)in;
46      } else {
47        return new DataInputInputStream(in);
48      }
49    }
50  
51  
52    public DataInputInputStream(DataInput in) {
53      this.in = in;
54    }
55  
56    @Override
57    public int read() throws IOException {
58      return in.readUnsignedByte();
59    }
60  }
