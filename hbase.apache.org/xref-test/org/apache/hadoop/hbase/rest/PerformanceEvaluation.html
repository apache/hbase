81  /**
82   * Script used evaluating Stargate performance and scalability.  Runs a SG
83   * client that steps through one of a set of hardcoded tests or 'experiments'
84   * (e.g. a random reads test, a random writes test, etc.). Pass on the
85   * command-line which test to run and how many clients are participating in
86   * this experiment. Run <code>java PerformanceEvaluation --help</code> to
87   * obtain usage.
88   * 
89   * <p>This class sets up and runs the evaluation programs described in
90   * Section 7, <i>Performance Evaluation</i>, of the <a
91   * href="http://labs.google.com/papers/bigtable.html">Bigtable</a>
92   * paper, pages 8-10.
93   * 
94   * <p>If number of clients > 1, we start up a MapReduce job. Each map task
95   * runs an individual client. Each client does about 1GB of data.
96   */
97  public class PerformanceEvaluation  {

273   /**
274    *  InputFormat of Performance Evaluation MapReduce job.
275    *  It extends from FileInputFormat, want to use it's methods such as setInputPaths(). 
276    */
277   public static class PeInputFormat extends FileInputFormat<NullWritable, PeInputSplit> {
278 
279     @Override
280     public List<InputSplit> getSplits(JobContext job) throws IOException {
281       // generate splits
282       List<InputSplit> splitList = new ArrayList<InputSplit>();
283       
284       for (FileStatus file: listStatus(job)) {
285         Path path = file.getPath();
286         FileSystem fs = path.getFileSystem(job.getConfiguration());
287         FSDataInputStream fileIn = fs.open(path);
288         LineReader in = new LineReader(fileIn, job.getConfiguration());
289         int lineLen = 0;
290         while(true) {
291           Text lineText = new Text();
292           lineLen = in.readLine(lineText);
293           if(lineLen <= 0) {
294           break;
295           }
296           Matcher m = LINE_PATTERN.matcher(lineText.toString());
297           if((m != null) && m.matches()) {
298             int startRow = Integer.parseInt(m.group(1));
299             int rows = Integer.parseInt(m.group(2));
300             int totalRows = Integer.parseInt(m.group(3));
301             int clients = Integer.parseInt(m.group(4));
302             int rowsPerPut = Integer.parseInt(m.group(5));
303 
304             LOG.debug("split["+ splitList.size() + "] " + 
305                      " startRow=" + startRow +
306                      " rows=" + rows +
307                      " totalRows=" + totalRows +
308                      " clients=" + clients +
309                      " rowsPerPut=" + rowsPerPut);
310 
311             PeInputSplit newSplit =
312               new PeInputSplit(startRow, rows, totalRows, clients, rowsPerPut);
313             splitList.add(newSplit);
314           }
315         }
316         in.close();
317       }
318       
319       LOG.info("Total # of splits: " + splitList.size());
320       return splitList;
321     }

324     public RecordReader<NullWritable, PeInputSplit> createRecordReader(InputSplit split,

344         if(readOver) {

359

363       }
364

367         if(readOver) {
368           return 1.0f;
369         } else {
370           return 0.0f;
371         }
372       }
373       
374       @Override
375       public void close() throws IOException {
376         // do nothing
377       }
378     }

381   /**
382    * MapReduce job that runs a performance evaluation client in each map task.
383    */
384   public static class EvaluationMapTask 
385       extends Mapper<NullWritable, PeInputSplit, LongWritable, LongWritable> {
386 
387     /** configuration parameter name that contains the command */
388     public final static String CMD_KEY = "EvaluationMapTask.command";
389     /** configuration parameter name that contains the PE impl */
390     public static final String PE_KEY = "EvaluationMapTask.performanceEvalImpl";
391 
392     private Class<? extends Test> cmd;
393     private PerformanceEvaluation pe;
394 
395     @Override
396     protected void setup(Context context) throws IOException, InterruptedException {
397       this.cmd = forName(context.getConfiguration().get(CMD_KEY), Test.class);
398 
399       // this is required so that extensions of PE are instantiated within the
400       // map reduce task...
401       Class<? extends PerformanceEvaluation> peClass =
402           forName(context.getConfiguration().get(PE_KEY), PerformanceEvaluation.class);
403       try {
404         this.pe = peClass.getConstructor(Configuration.class)
405             .newInstance(context.getConfiguration());
406       } catch (Exception e) {
407         throw new IllegalStateException("Could not instantiate PE instance", e);
408       }
409     }
410 
411     private <Type> Class<? extends Type> forName(String className, Class<Type> type) {
412       Class<? extends Type> clazz = null;
413       try {
414         clazz = Class.forName(className).asSubclass(type);
415       } catch (ClassNotFoundException e) {
416         throw new IllegalStateException("Could not find class for name: " + className, e);
417       }
418       return clazz;
419     }
420 
421     protected void map(NullWritable key, PeInputSplit value, final Context context) 
422            throws IOException, InterruptedException {
423       
424       Status status = new Status() {
425         public void setStatus(String msg) {
426            context.setStatus(msg); 
427         }
428       };
429       
430       // Evaluation task
431       long elapsedTime = this.pe.runOneClient(this.cmd, value.getStartRow(),
432         value.getRows(), value.getTotalRows(), value.getRowsPerPut(), status);
433       // Collect how much time the thing took. Report as map output and
434       // to the ELAPSED_TIME counter.
435       context.getCounter(Counter.ELAPSED_TIME).increment(elapsedTime);
436       context.getCounter(Counter.ROWS).increment(value.rows);
437       context.write(new LongWritable(value.startRow), new LongWritable(elapsedTime));
438       context.progress();
439     }
440   }

462   /*
463    * We're to run multiple clients concurrently.  Setup a mapreduce job.  Run
464    * one map per client.  Then run a single reduce to sum the elapsed times.
465    * @param cmd Command to run.
466    * @throws IOException
467    */
468   private void runNIsMoreThanOne(final Class<? extends Test> cmd)
469   throws IOException, InterruptedException, ClassNotFoundException {
470     checkTable();
471     if (nomapred) {
472       doMultipleClients(cmd);
473     } else {
474       doMapReduce(cmd);
475     }
476   }

478   /*
479    * Run all clients in this vm each to its own thread.
480    * @param cmd Command to run.
481    * @throws IOException
482    */
483   private void doMultipleClients(final Class<? extends Test> cmd) throws IOException {
484     final List<Thread> threads = new ArrayList<Thread>(N);
485     final int perClientRows = R/N;
486     for (int i = 0; i < N; i++) {
487       Thread t = new Thread (Integer.toString(i)) {
488         @Override
489         public void run() {
490           super.run();
491           PerformanceEvaluation pe = new PerformanceEvaluation(conf);
492           int index = Integer.parseInt(getName());
493           try {
494             long elapsedTime = pe.runOneClient(cmd, index * perClientRows,
495               perClientRows, R, B, new Status() {
496                   public void setStatus(final String msg) throws IOException {
497                     LOG.info("client-" + getName() + " " + msg);
498                   }
499                 });
500             LOG.info("Finished " + getName() + " in " + elapsedTime +
501               "ms writing " + perClientRows + " rows");
502           } catch (IOException e) {
503             throw new RuntimeException(e);
504           }
505         }
506       };
507       threads.add(t);
508     }
509     for (Thread t: threads) {
510       t.start();
511     }
512     for (Thread t: threads) {
513       while(t.isAlive()) {
514         try {
515           t.join();
516         } catch (InterruptedException e) {
517           LOG.debug("Interrupted, continuing" + e.toString());
518         }
519       }
520     }
521   }

523   /*
524    * Run a mapreduce job.  Run as many maps as asked-for clients.
525    * Before we start up the job, write out an input file with instruction
526    * per client regards which row they are to start on.
527    * @param cmd Command to run.
528    * @throws IOException
529    */
530   private void doMapReduce(final Class<? extends Test> cmd) throws IOException,
531         InterruptedException, ClassNotFoundException {
532     Path inputDir = writeInputFile(this.conf);
533     this.conf.set(EvaluationMapTask.CMD_KEY, cmd.getName());
534     this.conf.set(EvaluationMapTask.PE_KEY, getClass().getName());
535     Job job = new Job(this.conf);
536     job.setJarByClass(PerformanceEvaluation.class);
537     job.setJobName("HBase Performance Evaluation");
538     
539     job.setInputFormatClass(PeInputFormat.class);
540     PeInputFormat.setInputPaths(job, inputDir);
541     
542     job.setOutputKeyClass(LongWritable.class);
543     job.setOutputValueClass(LongWritable.class);
544     
545     job.setMapperClass(EvaluationMapTask.class);
546     job.setReducerClass(LongSumReducer.class);
547         
548     job.setNumReduceTasks(1);
549     
550     job.setOutputFormatClass(TextOutputFormat.class);
551     TextOutputFormat.setOutputPath(job, new Path(inputDir,"outputs"));
552     
553     job.waitForCompletion(true);
554   }

556   /*
557    * Write input file of offsets-per-client for the mapreduce job.
558    * @param c Configuration
559    * @return Directory that contains file written.
560    * @throws IOException
561    */
562   private Path writeInputFile(final Configuration c) throws IOException {
563     FileSystem fs = FileSystem.get(c);
564     if (!fs.exists(PERF_EVAL_DIR)) {
565       fs.mkdirs(PERF_EVAL_DIR);
566     }
567     SimpleDateFormat formatter = new SimpleDateFormat("yyyyMMddHHmmss");
568     Path subdir = new Path(PERF_EVAL_DIR, formatter.format(new Date()));
569     fs.mkdirs(subdir);
570     Path inputFile = new Path(subdir, "input.txt");
571     PrintStream out = new PrintStream(fs.create(inputFile));
572     // Make input random.
573     Map<Integer, String> m = new TreeMap<Integer, String>();
574     Hash h = MurmurHash.getInstance();
575     int perClientRows = (R / N);
576     try {
577       for (int i = 0; i < 10; i++) {
578         for (int j = 0; j < N; j++) {
579           String s = "startRow=" + ((j * perClientRows) + (i * (perClientRows/10))) +
580           ", perClientRunRows=" + (perClientRows / 10) +
581           ", totalRows=" + R +
582           ", clients=" + N +
583           ", rowsPerPut=" + B;
584           int hash = h.hash(Bytes.toBytes(s));
585           m.put(hash, s);
586         }
587       }
588       for (Map.Entry<Integer, String> e: m.entrySet()) {
589         out.println(e.getValue());
590       }
591     } finally {
592       out.close();
593     }
594     return subdir;
595   }

667   /*
668    * A test.
669    * Subclass to particularize what happens per row.
670    */
671   static abstract class Test {
672     // Below is make it so when Tests are all running in the one
673     // jvm, that they each have a differently seeded Random. 
674     private static final Random randomSeed =
675       new Random(System.currentTimeMillis());
676     private static long nextRandomSeed() {
677       return randomSeed.nextLong();
678     }
679     protected final Random rand = new Random(nextRandomSeed());
680 
681     protected final int startRow;
682     protected final int perClientRunRows;
683     protected final int totalRows;
684     protected final Status status;
685     protected byte[] tableName;
686     protected RemoteHTable table;
687     protected volatile Configuration conf;
688 
689     /**
690      * Note that all subclasses of this class must provide a public contructor
691      * that has the exact same list of arguments.
692      */
693     Test(final Configuration conf, final TestOptions options, final Status status) {
694       super();
695       this.startRow = options.getStartRow();
696       this.perClientRunRows = options.getPerClientRunRows();
697       this.totalRows = options.getTotalRows();
698       this.status = status;
699       this.tableName = options.getTableName();
700       this.table = null;
701       this.conf = conf;
702     }
703     
704     protected String generateStatus(final int sr, final int i, final int lr) {
705       return sr + "/" + i + "/" + lr;

1031    * Format passed integer.
1032    * @param number
1033    * @return Returns zero-prefixed 10-byte wide decimal version of passed
1034    * number (Does absolute in case number is negative).
1035    */
1036   public static byte [] format(final int number) {
1037     byte [] b = new byte[10];
1038     int d = Math.abs(number);
1039     for (int i = b.length - 1; i >= 0; i--) {
1040       b[i] = (byte)((d % 10) + '0');
1041       d /= 10;
1042     }
1043     return b;
1044   }

1164   public int doCommandLine(final String[] args) {
1165     // Process command-line args. TODO: Better cmd-line processing
1166     // (but hopefully something not as painful as cli options).    
1167     int errCode = -1;
1168     if (args.length < 1) {
1169       printUsage();
1170       return errCode;
1171     }
1172 
1173     try {
1174       for (int i = 0; i < args.length; i++) {
1175         String cmd = args[i];
1176         if (cmd.equals("-h")) {
1177           printUsage();
1178           errCode = 0;
1179           break;
1180         }
1181        
1182         final String nmr = "--nomapred";
1183         if (cmd.startsWith(nmr)) {
1184           nomapred = true;
1185           continue;
1186         }
1187         
1188         final String rows = "--rows=";
1189         if (cmd.startsWith(rows)) {
1190           R = Integer.parseInt(cmd.substring(rows.length()));
1191           continue;
1192         }
1193

1195         if (cmd.startsWith(rowsPerPut)) {
1196           this.B = Integer.parseInt(cmd.substring(rowsPerPut.length()));
1197           continue;
1198         }
1199 
1200         final String host = "--host=";
1201         if (cmd.startsWith(host)) {
1202           cluster.add(cmd.substring(host.length()));
1203           continue;
1204         }
1205 
1206         Class<? extends Test> cmdClass = determineCommandClass(cmd);
1207         if (cmdClass != null) {
1208           getArgs(i + 1, args);

1210             String s = conf.get("stargate.hostname", "localhost");
1211             if (s.contains(":")) {
1212               cluster.add(s);
1213             } else {
1214               cluster.add(s, conf.getInt("stargate.port", 8080));
1215             }
1216           }

1219           break;
