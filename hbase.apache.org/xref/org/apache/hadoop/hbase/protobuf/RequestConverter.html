127 /**
128    * Create a new protocol buffer GetRequest to get a row, all columns in a family.
129    * If there is no such row, return the closest row before it.
130    *
131    * @param regionName the name of the region to get
132    * @param row the row to get
133    * @param family the column family to get
134    * should return the immediate row before
135    * @return a protocol buffer GetReuqest
136    */
137   public static GetRequest buildGetRowOrBeforeRequest(
138       final byte[] regionName, final byte[] row, final byte[] family) {
139     GetRequest.Builder builder = GetRequest.newBuilder();
140     RegionSpecifier region = buildRegionSpecifier(
141       RegionSpecifierType.REGION_NAME, regionName);
142     builder.setClosestRowBefore(true);
143     builder.setRegion(region);
144 
145     Column.Builder columnBuilder = Column.newBuilder();
146     columnBuilder.setFamily(ByteString.copyFrom(family));
147     ClientProtos.Get.Builder getBuilder =
148       ClientProtos.Get.newBuilder();
149     getBuilder.setRow(ByteString.copyFrom(row));
150     getBuilder.addColumn(columnBuilder.build());
151     builder.setGet(getBuilder.build());
152     return builder.build();
153   }

187    * Create a protocol buffer MutateRequest for a client increment
188    *
189    * @param regionName
190    * @param row
191    * @param family
192    * @param qualifier
193    * @param amount
194    * @param writeToWAL
195    * @return a mutate request
196    */
197   public static MutateRequest buildMutateRequest(
198       final byte[] regionName, final byte[] row, final byte[] family,
199       final byte [] qualifier, final long amount, final boolean writeToWAL) {
200     MutateRequest.Builder builder = MutateRequest.newBuilder();
201     RegionSpecifier region = buildRegionSpecifier(
202       RegionSpecifierType.REGION_NAME, regionName);
203     builder.setRegion(region);
204 
205     Mutate.Builder mutateBuilder = Mutate.newBuilder();
206     mutateBuilder.setRow(ByteString.copyFrom(row));
207     mutateBuilder.setMutateType(MutateType.INCREMENT);
208     mutateBuilder.setWriteToWAL(writeToWAL);
209     ColumnValue.Builder columnBuilder = ColumnValue.newBuilder();
210     columnBuilder.setFamily(ByteString.copyFrom(family));
211     QualifierValue.Builder valueBuilder = QualifierValue.newBuilder();
212     valueBuilder.setValue(ByteString.copyFrom(Bytes.toBytes(amount)));
213     valueBuilder.setQualifier(ByteString.copyFrom(qualifier));
214     columnBuilder.addQualifierValue(valueBuilder.build());
215     mutateBuilder.addColumnValue(columnBuilder.build());
216 
217     builder.setMutate(mutateBuilder.build());
218     return builder.build();
219   }

348   /**
349    * Create a protocol buffer MultiRequest for a row mutations
350    *
351    * @param regionName
352    * @param rowMutations
353    * @return a multi request
354    * @throws IOException
355    */
356   public static MultiRequest buildMultiRequest(final byte[] regionName,
357       final RowMutations rowMutations) throws IOException {
358     MultiRequest.Builder builder = MultiRequest.newBuilder();
359     RegionSpecifier region = buildRegionSpecifier(
360       RegionSpecifierType.REGION_NAME, regionName);
361     builder.setRegion(region);
362     builder.setAtomic(true);
363     for (Mutation mutation: rowMutations.getMutations()) {
364       MutateType mutateType = null;
365       if (mutation instanceof Put) {
366         mutateType = MutateType.PUT;
367       } else if (mutation instanceof Delete) {
368         mutateType = MutateType.DELETE;
369       } else {
370         throw new DoNotRetryIOException(
371           "RowMutations supports only put and delete, not "
372             + mutation.getClass().getName());
373       }
374       Mutate mutate = ProtobufUtil.toMutate(mutateType, mutation);
375       builder.addAction(MultiAction.newBuilder().setMutate(mutate).build());
376     }
377     return builder.build();
378   }

516   /**
517    * Create a protocol buffer multi request for a list of actions.
518    * RowMutations in the list (if any) will be ignored.
519    *
520    * @param regionName
521    * @param actions
522    * @return a multi request
523    * @throws IOException
524    */
525   public static <R> MultiRequest buildMultiRequest(final byte[] regionName,
526       final List<Action<R>> actions) throws IOException {
527     MultiRequest.Builder builder = MultiRequest.newBuilder();
528     RegionSpecifier region = buildRegionSpecifier(
529       RegionSpecifierType.REGION_NAME, regionName);
530     builder.setRegion(region);
531     for (Action<R> action: actions) {
532       MultiAction.Builder protoAction = MultiAction.newBuilder();
533 
534       Row row = action.getAction();
535       if (row instanceof Get) {
536         protoAction.setGet(ProtobufUtil.toGet((Get)row));
537       } else if (row instanceof Put) {
538         protoAction.setMutate(ProtobufUtil.toMutate(MutateType.PUT, (Put)row));
539       } else if (row instanceof Delete) {
540         protoAction.setMutate(ProtobufUtil.toMutate(MutateType.DELETE, (Delete)row));
541       } else if (row instanceof Exec) {
542         protoAction.setExec(ProtobufUtil.toExec((Exec)row));
543       } else if (row instanceof Append) {
544         protoAction.setMutate(ProtobufUtil.toMutate(MutateType.APPEND, (Append)row));
545       } else if (row instanceof Increment) {
546         protoAction.setMutate(ProtobufUtil.toMutate((Increment)row));
547       } else if (row instanceof RowMutations) {
548         continue; // ignore RowMutations
549       } else {
550         throw new DoNotRetryIOException(
551           "multi doesn't support " + row.getClass().getName());
552       }
553       builder.addAction(protoAction.build());
554     }
555     return builder.build();
556   }

756   * Create a new ReplicateWALEntryRequest from a list of HLog entries
757   *
758   * @param entries the HLog entries to be replicated
759   * @return a ReplicateWALEntryRequest
760   */
761  public static ReplicateWALEntryRequest
762      buildReplicateWALEntryRequest(final HLog.Entry[] entries) {
763    FamilyScope.Builder scopeBuilder = FamilyScope.newBuilder();
764    WALEntry.Builder entryBuilder = WALEntry.newBuilder();
765    ReplicateWALEntryRequest.Builder builder =
766      ReplicateWALEntryRequest.newBuilder();
767    for (HLog.Entry entry: entries) {
768      entryBuilder.clear();
769      WALKey.Builder keyBuilder = entryBuilder.getKeyBuilder();
770      HLogKey key = entry.getKey();
771      keyBuilder.setEncodedRegionName(
772        ByteString.copyFrom(key.getEncodedRegionName()));
773      keyBuilder.setTableName(ByteString.copyFrom(key.getTablename()));
774      keyBuilder.setLogSequenceNumber(key.getLogSeqNum());
775      keyBuilder.setWriteTime(key.getWriteTime());
776      UUID clusterId = key.getClusterId();
777      if (clusterId != null) {
778        AdminProtos.UUID.Builder uuidBuilder = keyBuilder.getClusterIdBuilder();
779        uuidBuilder.setLeastSigBits(clusterId.getLeastSignificantBits());
780        uuidBuilder.setMostSigBits(clusterId.getMostSignificantBits());
781      }
782      WALEdit edit = entry.getEdit();
783      WALEntry.WALEdit.Builder editBuilder = entryBuilder.getEditBuilder();
784      NavigableMap<byte[], Integer> scopes = edit.getScopes();
785      if (scopes != null && !scopes.isEmpty()) {
786        for (Map.Entry<byte[], Integer> scope: scopes.entrySet()) {
787          scopeBuilder.setFamily(ByteString.copyFrom(scope.getKey()));
788          ScopeType scopeType = ScopeType.valueOf(scope.getValue().intValue());
789          scopeBuilder.setScopeType(scopeType);
790          editBuilder.addFamilyScope(scopeBuilder.build());
791        }
792      }
793      List<KeyValue> keyValues = edit.getKeyValues();
794      for (KeyValue value: keyValues) {
795        editBuilder.addKeyValueBytes(ByteString.copyFrom(
796          value.getBuffer(), value.getOffset(), value.getLength()));
797      }
798      builder.addEntry(entryBuilder.build());
799    }
800    return builder.build();
801  }
