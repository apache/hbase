19  package org.apache.hadoop.hbase.regionserver;
20  
21  import static org.junit.Assert.assertEquals;
22  import static org.junit.Assert.assertTrue;
23  
24  import java.io.IOException;
25  
26  import org.apache.hadoop.conf.Configuration;
27  import org.apache.hadoop.fs.FileSystem;
28  import org.apache.hadoop.fs.Path;
29  import org.apache.hadoop.hbase.*;
30  import org.apache.hadoop.hbase.client.Increment;
31  import org.apache.hadoop.hbase.client.Result;
32  import org.apache.hadoop.hbase.util.Bytes;
33  import org.junit.Test;
34  import org.junit.experimental.categories.Category;
35  
36  @Category(SmallTests.class)
37  public class TestResettingCounters {
38  
39    @Test
40    public void testResettingCounters() throws Exception {
41  
42      HBaseTestingUtility htu = new HBaseTestingUtility();
43      Configuration conf = htu.getConfiguration();
44      FileSystem fs = FileSystem.get(conf);
45      byte [] table = Bytes.toBytes("table");
46      byte [][] families = new byte [][] {
47          Bytes.toBytes("family1"),
48          Bytes.toBytes("family2"),
49          Bytes.toBytes("family3")
50      };
51      int numQualifiers = 10;
52      byte [][] qualifiers = new byte [numQualifiers][];
53      for (int i=0; i<numQualifiers; i++) qualifiers[i] = Bytes.toBytes("qf" + i);
54      int numRows = 10;
55      byte [][] rows = new byte [numRows][];
56      for (int i=0; i<numRows; i++) rows[i] = Bytes.toBytes("r" + i);
57  
58      HTableDescriptor htd = new HTableDescriptor(table);
59      for (byte [] family : families) htd.addFamily(new HColumnDescriptor(family));
60  
61      HRegionInfo hri = new HRegionInfo(htd.getName(), null, null, false);
62      String testDir = htu.getDataTestDir() + "/TestResettingCounters/";
63      Path path = new Path(testDir);
64      if (fs.exists(path)) {
65        if (!fs.delete(path, true)) {
66          throw new IOException("Failed delete of " + path);
67        }
68      }
69      HRegion region = HRegion.createHRegion(hri, path, conf, htd);
70      try {
71        Increment odd = new Increment(rows[0]);
72        Increment even = new Increment(rows[0]);
73        Increment all = new Increment(rows[0]);
74        for (int i=0;i<numQualifiers;i++) {
75          if (i % 2 == 0) even.addColumn(families[0], qualifiers[i], 1);
76          else odd.addColumn(families[0], qualifiers[i], 1);
77          all.addColumn(families[0], qualifiers[i], 1);
78        }
79  
80        // increment odd qualifiers 5 times and flush
81        for (int i=0;i<5;i++) region.increment(odd, null, false);
82        region.flushcache();
83  
84        // increment even qualifiers 5 times
85        for (int i=0;i<5;i++) region.increment(even, null, false);
86  
87        // increment all qualifiers, should have value=6 for all
88        Result result = region.increment(all, null, false);
89        assertEquals(numQualifiers, result.size());
90        KeyValue [] kvs = result.raw();
91        for (int i=0;i<kvs.length;i++) {
92          System.out.println(kvs[i].toString());
93          assertTrue(Bytes.equals(kvs[i].getQualifier(), qualifiers[i]));
94          assertEquals(6, Bytes.toLong(kvs[i].getValue()));
95        }
96      } finally {
97        HRegion.closeHRegion(region);
98      }
99      HRegion.closeHRegion(region);
100   }
101 
102 }
