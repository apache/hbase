38  /**
39   * Implementation of {@link Filter} that represents an ordered List of Filters
40   * which will be evaluated with a specified boolean operator {@link Operator#MUST_PASS_ALL}
41   * (<code>!AND</code>) or {@link Operator#MUST_PASS_ONE} (<code>!OR</code>).
42   * Since you can use Filter Lists as children of Filter Lists, you can create a
43   * hierarchy of filters to be evaluated.
44   * Defaults to {@link Operator#MUST_PASS_ALL}.
45   * <p>TODO: Fix creation of Configuration on serialization and deserialization.
46   */

52      /** !AND */
53      MUST_PASS_ALL,
54      /** !OR */
55      MUST_PASS_ONE

151   @Override
152   public boolean filterRowKey(byte[] rowKey, int offset, int length) {
153     for (Filter filter : filters) {
154       if (this.operator == Operator.MUST_PASS_ALL) {
155         if (filter.filterAllRemaining() ||
156             filter.filterRowKey(rowKey, offset, length)) {
157           return true;
158         }
159       } else if (this.operator == Operator.MUST_PASS_ONE) {
160         if (!filter.filterAllRemaining() &&
161             !filter.filterRowKey(rowKey, offset, length)) {
162           return false;
163         }
164       }
165     }
166     return this.operator == Operator.MUST_PASS_ONE;
167   }

169   @Override
170   public boolean filterAllRemaining() {
171     for (Filter filter : filters) {
172       if (filter.filterAllRemaining()) {
173         if (operator == Operator.MUST_PASS_ALL) {
174           return true;

176       } else {
177         if (operator == Operator.MUST_PASS_ONE) {
178           return false;
179         }
180       }
181     }
182     return operator == Operator.MUST_PASS_ONE;

194   @Override
195   public ReturnCode filterKeyValue(KeyValue v) {
196     ReturnCode rc = operator == Operator.MUST_PASS_ONE?
197         ReturnCode.SKIP: ReturnCode.INCLUDE;
198     for (Filter filter : filters) {
199       if (operator == Operator.MUST_PASS_ALL) {
200         if (filter.filterAllRemaining()) {
201           return ReturnCode.NEXT_ROW;
202         }
203         ReturnCode code = filter.filterKeyValue(v);
204         switch (code) {
205         // Override INCLUDE and continue to evaluate.
206         case INCLUDE_AND_NEXT_COL:
207           rc = ReturnCode.INCLUDE_AND_NEXT_COL;
208         case INCLUDE:
209           continue;
210         default:
211           return code;
212         }
213       } else if (operator == Operator.MUST_PASS_ONE) {
214         if (filter.filterAllRemaining()) {
215           continue;
216         }
217 
218         switch (filter.filterKeyValue(v)) {
219         case INCLUDE:
220           if (rc != ReturnCode.INCLUDE_AND_NEXT_COL) {
221             rc = ReturnCode.INCLUDE;
222           }
223           break;
224         case INCLUDE_AND_NEXT_COL:
225           rc = ReturnCode.INCLUDE_AND_NEXT_COL;
226           // must continue here to evaluate all filters
227           break;
228         case NEXT_ROW:
229           break;
230         case SKIP:
231           // continue;
232           break;
233         case NEXT_COL:
234           break;
235         case SEEK_NEXT_USING_HINT:
236           break;
237         default:
238           throw new IllegalStateException("Received code is not valid.");
239         }
240       }
241     }
242     return rc;
243   }

262   @Override
263   public boolean filterRow() {
264     for (Filter filter : filters) {
265       if (operator == Operator.MUST_PASS_ALL) {
266         if (filter.filterAllRemaining() || filter.filterRow()) {
267           return true;
268         }
269       } else if (operator == Operator.MUST_PASS_ONE) {
270         if (!filter.filterAllRemaining()
271             && !filter.filterRow()) {
272           return false;
273         }
274       }
275     }
276     return  operator == Operator.MUST_PASS_ONE;
277   }

292   /**
293    * @param pbBytes A pb serialized {@link FilterList} instance
294    * @return An instance of {@link FilterList} made from <code>bytes</code>
295    * @throws DeserializationException
296    * @see #toByteArray
297    */
298   public static FilterList parseFrom(final byte [] pbBytes)
299   throws DeserializationException {
300     FilterProtos.FilterList proto;
301     try {
302       proto = FilterProtos.FilterList.parseFrom(pbBytes);
303     } catch (InvalidProtocolBufferException e) {
304       throw new DeserializationException(e);
305     }
306 
307     List<Filter> rowFilters = new ArrayList<Filter>(proto.getFiltersCount());
308     try {
309       for (HBaseProtos.Filter filter : proto.getFiltersList()) {
310         rowFilters.add(ProtobufUtil.toFilter(filter));
311       }
312     } catch (IOException ioe) {
313       throw new DeserializationException(ioe);
314     }
315     return new FilterList(Operator.valueOf(proto.getOperator().name()),rowFilters);
316   }

333   @Override
334   public KeyValue getNextKeyHint(KeyValue currentKV) {
335     KeyValue keyHint = null;
336     for (Filter filter : filters) {
337       KeyValue curKeyHint = filter.getNextKeyHint(currentKV);
338       if (curKeyHint == null && operator == Operator.MUST_PASS_ONE) {
339         // If we ever don't have a hint and this is must-pass-one, then no hint
340         return null;
341       }
342       if (curKeyHint != null) {
343         // If this is the first hint we find, set it
344         if (keyHint == null) {
345           keyHint = curKeyHint;
346           continue;
347         }
348         // There is an existing hint
349         if (operator == Operator.MUST_PASS_ALL &&
350             KeyValue.COMPARATOR.compare(keyHint, curKeyHint) < 0) {
351           // If all conditions must pass, we can keep the max hint
352           keyHint = curKeyHint;
353         } else if (operator == Operator.MUST_PASS_ONE &&
354             KeyValue.COMPARATOR.compare(keyHint, curKeyHint) > 0) {
355           // If any condition can pass, we need to keep the min hint
356           keyHint = curKeyHint;
357         }
358       }
359     }
360     return keyHint;
361   }

368   protected String toString(int maxFilters) {
369     int endIndex = this.filters.size() < maxFilters
370         ? this.filters.size() : maxFilters;
371     return String.format("%s %s (%d/%d): %s",
372         this.getClass().getSimpleName(),
373         this.operator == Operator.MUST_PASS_ALL ? "AND" : "OR",
374         endIndex,
375         this.filters.size(),
376         this.filters.subList(0, endIndex).toString());
377   }
