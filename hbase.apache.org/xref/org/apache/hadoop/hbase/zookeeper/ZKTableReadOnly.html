34  /**
35   * Non-instantiable class that provides helper functions for
36   * clients other than {@link AssignmentManager} for reading the
37   * state of a table in ZK.
38   *
39   * <p>Does not cache state like {@link ZKTable}, actually reads from ZK each call.
40   */
41  public class ZKTableReadOnly {
42  
43    private ZKTableReadOnly() {}
44    
45    /**
46     * Go to zookeeper and see if state of table is {@code ZooKeeperProtos.Table.State#DISABLED}.
47     * This method does not use cache.
48     * This method is for clients other than {@link AssignmentManager}
49     * @param zkw
50     * @param tableName
51     * @return True if table is enabled.
52     * @throws KeeperException
53     */
54    public static boolean isDisabledTable(final ZooKeeperWatcher zkw,
55        final String tableName)
56    throws KeeperException {
57      ZooKeeperProtos.Table.State state = getTableState(zkw, tableName);
58      return isTableState(ZooKeeperProtos.Table.State.DISABLED, state);
59    }
60  
61    /**
62     * Go to zookeeper and see if state of table is {@code ZooKeeperProtos.Table.State#ENABLED}.
63     * This method does not use cache.
64     * This method is for clients other than {@link AssignmentManager}
65     * @param zkw
66     * @param tableName
67     * @return True if table is enabled.
68     * @throws KeeperException
69     */
70    public static boolean isEnabledTable(final ZooKeeperWatcher zkw,
71        final String tableName)
72    throws KeeperException {
73      return getTableState(zkw, tableName) == ZooKeeperProtos.Table.State.ENABLED;
74    }
75  
76    /**
77     * Go to zookeeper and see if state of table is {@code ZooKeeperProtos.Table.State#DISABLING}
78     * of {@code ZooKeeperProtos.Table.State#DISABLED}.
79     * This method does not use cache.
80     * This method is for clients other than {@link AssignmentManager}.
81     * @param zkw
82     * @param tableName
83     * @return True if table is enabled.
84     * @throws KeeperException
85     */
86    public static boolean isDisablingOrDisabledTable(final ZooKeeperWatcher zkw,
87        final String tableName)
88    throws KeeperException {
89      ZooKeeperProtos.Table.State state = getTableState(zkw, tableName);
90      return isTableState(ZooKeeperProtos.Table.State.DISABLING, state) ||
91        isTableState(ZooKeeperProtos.Table.State.DISABLED, state);
92    }
93  
94    /**
95     * Gets a list of all the tables set as disabled in zookeeper.
96     * @return Set of disabled tables, empty Set if none
97     * @throws KeeperException
98     */
99    public static Set<String> getDisabledTables(ZooKeeperWatcher zkw)
100   throws KeeperException {
101     Set<String> disabledTables = new HashSet<String>();
102     List<String> children =
103       ZKUtil.listChildrenNoWatch(zkw, zkw.tableZNode);
104     for (String child: children) {
105       ZooKeeperProtos.Table.State state = getTableState(zkw, child);
106       if (state == ZooKeeperProtos.Table.State.DISABLED) disabledTables.add(child);
107     }
108     return disabledTables;
109   }
110 
111   /**
112    * Gets a list of all the tables set as disabled in zookeeper.
113    * @return Set of disabled tables, empty Set if none
114    * @throws KeeperException
115    */
116   public static Set<String> getDisabledOrDisablingTables(ZooKeeperWatcher zkw)
117   throws KeeperException {
118     Set<String> disabledTables = new HashSet<String>();
119     List<String> children =
120       ZKUtil.listChildrenNoWatch(zkw, zkw.tableZNode);
121     for (String child: children) {
122       ZooKeeperProtos.Table.State state = getTableState(zkw, child);
123       if (state == ZooKeeperProtos.Table.State.DISABLED ||
124           state == ZooKeeperProtos.Table.State.DISABLING)
125         disabledTables.add(child);
126     }
127     return disabledTables;
128   }
129 
130   static boolean isTableState(final ZooKeeperProtos.Table.State expectedState,
131       final ZooKeeperProtos.Table.State currentState) {
132     return currentState != null && currentState.equals(expectedState);
133   }
134 
135   /**
136    * @param zkw
137    * @param child
138    * @return Null or {@link ZooKeeperProtos.Table.State} found in znode.
139    * @throws KeeperException
140    */
141   static ZooKeeperProtos.Table.State getTableState(final ZooKeeperWatcher zkw,
142       final String child)
143   throws KeeperException {
144     String znode = ZKUtil.joinZNode(zkw.tableZNode, child);
145     byte [] data = ZKUtil.getData(zkw, znode);
146     if (data == null || data.length <= 0) return ZooKeeperProtos.Table.State.ENABLED;
147     try {
148       ProtobufUtil.expectPBMagicPrefix(data);
149       ZooKeeperProtos.Table.Builder builder = ZooKeeperProtos.Table.newBuilder();
150       int magicLen = ProtobufUtil.lengthOfPBMagic();
151       ZooKeeperProtos.Table t = builder.mergeFrom(data, magicLen, data.length - magicLen).build();
152       return t.getState();
153     } catch (InvalidProtocolBufferException e) {
154       KeeperException ke = new KeeperException.DataInconsistencyException();
155       ke.initCause(e);
156       throw ke;
157     } catch (DeserializationException e) {
158       throw ZKUtil.convert(e);
159     }
160   }
