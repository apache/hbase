295   /**
296    * Compiles the test class with bogus code into a .class file.
297    * Unfortunately it's very tedious.
298    * @param counter Unique test counter.
299    * @param packageNameSuffix Package name suffix (e.g. ".suffix") for nesting, or "".
300    * @return The resulting .class file and the location in jar it is supposed to go to.
301    */
302   private static FileAndPath compileTestClass(long counter,
303       String packageNameSuffix, String classNamePrefix) throws Exception {
304     classNamePrefix = classNamePrefix + counter;
305     String packageName = makePackageName(packageNameSuffix, counter);
306     String javaPath = basePath + classNamePrefix + ".java";
307     String classPath = basePath + classNamePrefix + ".class";
308     PrintStream source = new PrintStream(javaPath);
309     source.println("package " + packageName + ";");
310     source.println("public class " + classNamePrefix
311         + " { public static void main(String[] args) { } };");
312     source.close();
313     JavaCompiler jc = ToolProvider.getSystemJavaCompiler();
314     int result = jc.run(null, null, null, javaPath);
315     assertEquals(0, result);
316     File classFile = new File(classPath);
317     assertTrue(classFile.exists());
318     return new FileAndPath(packageName.replace('.', '/') + '/', classFile);
319   }

321   /**
322    * Makes a jar out of some class files. Unfortunately it's very tedious.
323    * @param filesInJar Files created via compileTestClass.
324    * @return path to the resulting jar file.
325    */
326   private static String packageAndLoadJar(FileAndPath... filesInJar) throws Exception {
327     // First, write the bogus jar file.
328     String path = basePath + "jar" + jarCounter.incrementAndGet() + ".jar";
329     Manifest manifest = new Manifest();
330     manifest.getMainAttributes().put(Attributes.Name.MANIFEST_VERSION, "1.0");
331     FileOutputStream fos = new FileOutputStream(path);
332     JarOutputStream jarOutputStream = new JarOutputStream(fos, manifest);
333     // Directory entries for all packages have to be added explicitly for
334     // resources to be findable via ClassLoader. Directory entries must end
335     // with "/"; the initial one is expected to, also.
336     Set<String> pathsInJar = new HashSet<String>();
337     for (FileAndPath fileAndPath : filesInJar) {
338       String pathToAdd = fileAndPath.path;
339       while (pathsInJar.add(pathToAdd)) {
340         int ix = pathToAdd.lastIndexOf('/', pathToAdd.length() - 2);
341         if (ix < 0) {
342           break;
343         }
344         pathToAdd = pathToAdd.substring(0, ix);
345       }
346     }
347     for (String pathInJar : pathsInJar) {
348       jarOutputStream.putNextEntry(new JarEntry(pathInJar));
349       jarOutputStream.closeEntry();
350     }
351     for (FileAndPath fileAndPath : filesInJar) {
352       File file = fileAndPath.file;
353       jarOutputStream.putNextEntry(
354           new JarEntry(fileAndPath.path + file.getName()));
355       byte[] allBytes = new byte[(int)file.length()];
356       FileInputStream fis = new FileInputStream(file);
357       fis.read(allBytes);
358       fis.close();
359       jarOutputStream.write(allBytes);
360       jarOutputStream.closeEntry();
361     }
362     jarOutputStream.close();
363     fos.close();
364 
365     // Add the file to classpath.
366     File jarFile = new File(path);
367     assertTrue(jarFile.exists());
368     URLClassLoader urlClassLoader = (URLClassLoader)ClassLoader.getSystemClassLoader();
369     Method method = URLClassLoader.class
370         .getDeclaredMethod("addURL", new Class[] { URL.class });
371     method.setAccessible(true);
372     method.invoke(urlClassLoader, new Object[] { jarFile.toURI().toURL() });
373     return jarFile.getAbsolutePath();
374   }
