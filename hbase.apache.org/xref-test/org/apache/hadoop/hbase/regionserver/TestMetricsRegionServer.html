86      MetricsRegionServerSource serverSource = rsm.getMetricsSource();
87      for (int i=0; i < 12; i ++) {
88        rsm.updateAppend(12);

90      }
91      for (int i=0; i < 13; i ++) {
92        rsm.updateDelete(13);
93        rsm.updateDelete(1003);
94      }
95      for (int i=0; i < 14; i ++) {
96        rsm.updateGet(14);
97        rsm.updateGet(1004);
98      }
99      for (int i=0; i < 15; i ++) {
100       rsm.updateIncrement(15);
101       rsm.updateIncrement(1005);
102     }

104       rsm.updatePut(16);
105       rsm.updatePut(1006);
106     }
