125   private void startCmdLineThread(final String[] args) {
126     LOG.info("Starting HBase Thrift server with command line: " +
127         Joiner.on(" ").join(args));
128 
129     cmdLineException = null;
130     cmdLineThread = new Thread(new Runnable() {
131       @Override
132       public void run() {
133         try {
134           thriftServer.doMain(args);
135         } catch (Exception e) {
136           cmdLineException = e;
137         }
138       }
139     });
140     cmdLineThread.setName(ThriftServer.class.getSimpleName() +
141         "-cmdline");
142     cmdLineThread.start();
143   }

145   @Test(timeout=60 * 1000)
146   public void testRunThriftServer() throws Exception {
147     List<String> args = new ArrayList<String>();
148     if (implType != null) {
149       String serverTypeOption = implType.toString();
150       assertTrue(serverTypeOption.startsWith("-"));
151       args.add(serverTypeOption);
152     }
153     port = HBaseTestingUtility.randomFreePort();
154     args.add("-" + ThriftServer.PORT_OPTION);
155     args.add(String.valueOf(port));
156     if (specifyFramed) {
157       args.add("-" + ThriftServer.FRAMED_OPTION);
158     }
159     if (specifyBindIP) {
160       args.add("-" + ThriftServer.BIND_OPTION);
161       args.add(InetAddress.getLocalHost().getHostName());
162     }
163     if (specifyCompact) {
164       args.add("-" + ThriftServer.COMPACT_OPTION);
165     }
166     args.add("start");
167 
168     thriftServer = new ThriftServer(TEST_UTIL.getConfiguration());
169     startCmdLineThread(args.toArray(new String[0]));
170 
171     while ( thriftServer.serverRunner == null || thriftServer.serverRunner.tserver == null ){
172       Thread.sleep(1);
173     }
174 
175     Class<? extends TServer> expectedClass = implType != null ?
176         implType.serverClass : TBoundedThreadPoolServer.class;
177     assertEquals(expectedClass,
178                  thriftServer.serverRunner.tserver.getClass());
179 
180     try {
181       talkToThriftServer();
182     } catch (Exception ex) {
183       clientSideException = ex;
184     } finally {
185       stopCmdLineThread();
186     }
187 
188     if (clientSideException != null) {
189       LOG.error("Thrift client threw an exception. Parameters:" +
190           getParametersString(), clientSideException);
191       throw new Exception(clientSideException);
192     }
193   }

195   private static volatile boolean tableCreated = false;
196 
197   private void talkToThriftServer() throws Exception {
198     TSocket sock = new TSocket(InetAddress.getLocalHost().getHostName(),
199         port);
200     TTransport transport = sock;
201     if (specifyFramed || implType.isAlwaysFramed) {
202       transport = new TFramedTransport(transport);
203     }
204 
205     sock.open();
206     try {
207       TProtocol prot;
208       if (specifyCompact) {
209         prot = new TCompactProtocol(transport);
210       } else {
211         prot = new TBinaryProtocol(transport);
212       }
213       Hbase.Client client = new Hbase.Client(prot);
214       if (!tableCreated){
215         TestThriftServer.createTestTables(client);
216         tableCreated = true;
217       }
218       TestThriftServer.checkTableList(client);
219 
220     } finally {
221       sock.close();
222     }
223   }
224 
225   private void stopCmdLineThread() throws Exception {
226     LOG.debug("Stopping " + implType.simpleClassName() + " Thrift server");
227     thriftServer.stop();
228     cmdLineThread.join();
229     if (cmdLineException != null) {
230       LOG.error("Command-line invocation of HBase Thrift server threw an " +
231           "exception", cmdLineException);
232       throw new Exception(cmdLineException);
233     }
234   }
