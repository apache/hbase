19  package org.apache.hadoop.hbase.mapreduce;
20  
21  import org.apache.hadoop.classification.InterfaceAudience;
22  import org.apache.hadoop.classification.InterfaceStability;
23  import org.apache.hadoop.hbase.mapreduce.replication.VerifyReplication;
24  import org.apache.hadoop.util.ProgramDriver;
25  
26  /**
27   * Driver for hbase mapreduce jobs. Select which to run by passing
28   * name of job to this main.
29   */
30  @InterfaceAudience.Public
31  @InterfaceStability.Stable
32  public class Driver {
33    /**
34     * @param args
35     * @throws Throwable
36     */
37    public static void main(String[] args) throws Throwable {
38      ProgramDriver pgd = new ProgramDriver();
39      pgd.addClass(RowCounter.NAME, RowCounter.class,
40        "Count rows in HBase table");
41      pgd.addClass(CellCounter.NAME, CellCounter.class,
42        "Count cells in HBase table");
43      pgd.addClass(Export.NAME, Export.class, "Write table data to HDFS.");
44      pgd.addClass(Import.NAME, Import.class, "Import data written by Export.");
45      pgd.addClass(ImportTsv.NAME, ImportTsv.class, "Import data in TSV format.");
46      pgd.addClass(LoadIncrementalHFiles.NAME, LoadIncrementalHFiles.class,
47                   "Complete a bulk data load.");
48      pgd.addClass(CopyTable.NAME, CopyTable.class,
49          "Export a table from local cluster to peer cluster");
50      pgd.addClass(VerifyReplication.NAME, VerifyReplication.class, "Compare" +
51          " the data from tables in two different clusters. WARNING: It" +
52          " doesn't work for incrementColumnValues'd cells since the" +
53          " timestamp is changed after being appended to the log.");
54      ProgramDriver.class.getMethod("driver", new Class [] {String[].class}).
55        invoke(pgd, new Object[]{args});
56    }
57  }
