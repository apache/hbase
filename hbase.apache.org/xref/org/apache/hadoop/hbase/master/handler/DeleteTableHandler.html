21  import java.io.IOException;
22  import java.util.List;
23  
24  import org.apache.commons.logging.Log;
25  import org.apache.commons.logging.LogFactory;
26  import org.apache.hadoop.classification.InterfaceAudience;
27  import org.apache.hadoop.hbase.HRegionInfo;
28  import org.apache.hadoop.hbase.Server;
29  import org.apache.hadoop.hbase.catalog.MetaEditor;
30  import org.apache.hadoop.hbase.master.AssignmentManager;
31  import org.apache.hadoop.hbase.master.HMaster;
32  import org.apache.hadoop.hbase.master.MasterCoprocessorHost;
33  import org.apache.hadoop.hbase.master.MasterServices;
34  import org.apache.hadoop.hbase.util.Bytes;
35  import org.apache.hadoop.hbase.util.Threads;
36  import org.apache.zookeeper.KeeperException;
37  
38  @InterfaceAudience.Private
39  public class DeleteTableHandler extends TableEventHandler {
40    private static final Log LOG = LogFactory.getLog(DeleteTableHandler.class);
41  
42    public DeleteTableHandler(byte [] tableName, Server server,
43        final MasterServices masterServices)
44    throws IOException {
45      super(EventType.C_M_DELETE_TABLE, tableName, server, masterServices);
46      // The next call fails if no such table.
47      getTableDescriptor();
48    }
49  
50    @Override
51    protected void handleTableOperation(List<HRegionInfo> regions)
52    throws IOException, KeeperException {
53      MasterCoprocessorHost cpHost = ((HMaster) this.server)
54          .getCoprocessorHost();
55      if (cpHost != null) {
56        cpHost.preDeleteTableHandler(this.tableName);
57      }
58      AssignmentManager am = this.masterServices.getAssignmentManager();
59      long waitTime = server.getConfiguration().
60        getLong("hbase.master.wait.on.region", 5 * 60 * 1000);
61      for (HRegionInfo region : regions) {
62        long done = System.currentTimeMillis() + waitTime;
63        while (System.currentTimeMillis() < done) {
64          if (!am.getRegionStates().isRegionInTransition(region)) break;
65          Threads.sleep(waitingTimeForEvents);
66          LOG.debug("Waiting on region to clear regions in transition; "
67            + am.getRegionStates().getRegionTransitionState(region));
68        }
69        if (am.getRegionStates().isRegionInTransition(region)) {
70          throw new IOException("Waited hbase.master.wait.on.region (" +
71            waitTime + "ms) for region to leave region " +
72            region.getRegionNameAsString() + " in transitions");
73        }
74        LOG.debug("Deleting region " + region.getRegionNameAsString() +
75          " from META and FS");
76        // Remove region from META
77        MetaEditor.deleteRegion(this.server.getCatalogTracker(), region);
78        // Delete region from FS
79        this.masterServices.getMasterFileSystem().deleteRegion(region);
80      }
81      // Delete table from FS
82      this.masterServices.getMasterFileSystem().deleteTable(tableName);
83      // Update table descriptor cache
84      this.masterServices.getTableDescriptors().remove(Bytes.toString(tableName));
85  
86      // If entry for this table in zk, and up in AssignmentManager, remove it.
87  
88      am.getZKTable().setDeletedTable(Bytes.toString(tableName));
89      if (cpHost != null) {
90        cpHost.postDeleteTableHandler(this.tableName);
91      }
92    }
93  
94    @Override
95    public String toString() {
96      String name = "UnknownServerName";
97      if(server != null && server.getServerName() != null) {
98        name = server.getServerName().toString();
99      }
100     return getClass().getSimpleName() + "-" + name + "-" + getSeqid() + "-" + tableNameStr;
101   }
