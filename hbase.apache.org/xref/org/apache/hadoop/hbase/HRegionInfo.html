56   * HRegion information.
57   * Contains HRegion id, start and end keys, a reference to this HRegions' table descriptor, etc.

59  @InterfaceAudience.Public

62    /*
63     * There are two versions associated with HRegionInfo: HRegionInfo.VERSION and
64     * HConstants.META_VERSION. HRegionInfo.VERSION indicates the data structure's versioning
65     * while HConstants.META_VERSION indicates the versioning of the serialized HRIs stored in
66     * the META table.
67     *
68     * Pre-0.92:
69     *   HRI.VERSION == 0 and HConstants.META_VERSION does not exist (is not stored at META table)
70     *   HRegionInfo had an HTableDescriptor reference inside it.
71     *   HRegionInfo is serialized as Writable to META table.
72     * For 0.92.x and 0.94.x:
73     *   HRI.VERSION == 1 and HConstants.META_VERSION == 0
74     *   HRI no longer has HTableDescriptor in it.
75     *   HRI is serialized as Writable to META table.
76     * For 0.96.x:
77     *   HRI.VERSION == 1 and HConstants.META_VERSION == 1
78     *   HRI data structure is the same as 0.92 and 0.94
79     *   HRI is serialized as PB to META table.
80     *
81     * Versioning of HRegionInfo is deprecated. HRegionInfo does protobuf
82     * serialization using RegionInfo class, which has it's own versioning.
83     */
84    @Deprecated

88    /**
89     * The new format for a region name contains its encodedName at the end.
90     * The encoded name also serves as the directory name for the region
91     * in the filesystem.
92     *
93     * New region name format:
94     *    &lt;tablename>,,&lt;startkey>,&lt;regionIdTimestamp>.&lt;encodedName>.
95     * where,
96     *    &lt;encodedName> is a hex version of the MD5 hash of
97     *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
98     *
99     * The old region name format:
100    *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
101    * For region names in the old format, the encoded name is a 32-bit
102    * JenkinsHash integer value (in its decimal notation, string form).
103    *<p>
104    * **NOTE**
105    *
106    * ROOT, the first META region, and regions created by an older
107    * version of HBase (0.20 or prior) will continue to use the
108    * old region name format.
109    */
110 
111   /** Separator used to demarcate the encodedName in a region name
112    * in the new format. See description on new format above.
113    */
114   private static final int ENC_SEPARATOR = '.';
115   public  static final int MD5_HEX_LENGTH   = 32;
116 
117   /**
118    * Does region name contain its encoded name?
119    * @param regionName region name
120    * @return boolean indicating if this a new format region
121    *         name which contains its encoded name.
122    */
123   private static boolean hasEncodedName(final byte[] regionName) {
124     // check if region name ends in ENC_SEPARATOR
125     if ((regionName.length >= 1)
126         && (regionName[regionName.length - 1] == ENC_SEPARATOR)) {
127       // region name is new format. it contains the encoded name.
128       return true;
129     }
130     return false;
131   }
132 
133   /**
134    * @param regionName
135    * @return the encodedName
136    */
137   public static String encodeRegionName(final byte [] regionName) {
138     String encodedName;
139     if (hasEncodedName(regionName)) {
140       // region is in new format:
141       // <tableName>,<startKey>,<regionIdTimeStamp>/encodedName/
142       encodedName = Bytes.toString(regionName,
143           regionName.length - MD5_HEX_LENGTH - 1,
144           MD5_HEX_LENGTH);
145     } else {
146       // old format region name. ROOT and first META region also
147       // use this format.EncodedName is the JenkinsHash value.
148       int hashVal = Math.abs(JenkinsHash.getInstance().hash(regionName,
149         regionName.length, 0));
150       encodedName = String.valueOf(hashVal);
151     }
152     return encodedName;
153   }

349   /**
350    * Make a region name of passed parameters.
351    * @param tableName
352    * @param startKey Can be null
353    * @param id Region id (Usually timestamp from when region was created).
354    * @param newFormat should we create the region name in the new format
355    *                  (such that it contains its encoded name?).
356    * @return Region name made of passed tableName, startKey and id
357    */
358   public static byte [] createRegionName(final byte [] tableName,
359       final byte [] startKey, final byte [] id, boolean newFormat) {
360     byte [] b = new byte [tableName.length + 2 + id.length +
361        (startKey == null? 0: startKey.length) +
362        (newFormat ? (MD5_HEX_LENGTH + 2) : 0)];
363 
364     int offset = tableName.length;
365     System.arraycopy(tableName, 0, b, 0, offset);
366     b[offset++] = HConstants.DELIMITER;
367     if (startKey != null && startKey.length > 0) {
368       System.arraycopy(startKey, 0, b, offset, startKey.length);
369       offset += startKey.length;
370     }
371     b[offset++] = HConstants.DELIMITER;
372     System.arraycopy(id, 0, b, offset, id.length);
373     offset += id.length;
374 
375     if (newFormat) {
376       //
377       // Encoded name should be built into the region name.
378       //
379       // Use the region name thus far (namely, <tablename>,<startKey>,<id>)
380       // to compute a MD5 hash to be used as the encoded name, and append
381       // it to the byte buffer.
382       //
383       String md5Hash = MD5Hash.getMD5AsHex(b, 0, offset);
384       byte [] md5HashBytes = Bytes.toBytes(md5Hash);
385 
386       if (md5HashBytes.length != MD5_HEX_LENGTH) {
387         LOG.error("MD5-hash length mismatch: Expected=" + MD5_HEX_LENGTH +
388                   "; Got=" + md5HashBytes.length);
389       }
390 
391       // now append the bytes '.<encodedName>.' to the end
392       b[offset++] = ENC_SEPARATOR;
393       System.arraycopy(md5HashBytes, 0, b, offset, MD5_HEX_LENGTH);
394       offset += MD5_HEX_LENGTH;
395       b[offset++] = ENC_SEPARATOR;
396     }
397 
398     return b;
399   }

401   /**
402    * Gets the table name from the specified region name.
403    * @param regionName
404    * @return Table name.
405    */
406   public static byte [] getTableName(byte [] regionName) {
407     int offset = -1;
408     for (int i = 0; i < regionName.length; i++) {
409       if (regionName[i] == HConstants.DELIMITER) {
410         offset = i;
411         break;
412       }
413     }
414     byte [] tableName = new byte[offset];
415     System.arraycopy(regionName, 0, tableName, 0, offset);
416     return tableName;
417   }

419   /**
420    * Separate elements of a regionName.
421    * @param regionName
422    * @return Array of byte[] containing tableName, startKey and id
423    * @throws IOException
424    */
425   public static byte [][] parseRegionName(final byte [] regionName)
426   throws IOException {
427     int offset = -1;
428     for (int i = 0; i < regionName.length; i++) {
429       if (regionName[i] == HConstants.DELIMITER) {
430         offset = i;
431         break;
432       }
433     }
434     if(offset == -1) throw new IOException("Invalid regionName format");
435     byte [] tableName = new byte[offset];
436     System.arraycopy(regionName, 0, tableName, 0, offset);
437     offset = -1;
438     for (int i = regionName.length - 1; i > 0; i--) {
439       if(regionName[i] == HConstants.DELIMITER) {
440         offset = i;
441         break;
442       }
443     }
444     if(offset == -1) throw new IOException("Invalid regionName format");
445     byte [] startKey = HConstants.EMPTY_BYTE_ARRAY;
446     if(offset != tableName.length + 1) {
447       startKey = new byte[offset - tableName.length - 1];
448       System.arraycopy(regionName, tableName.length + 1, startKey, 0,
449           offset - tableName.length - 1);
450     }
451     byte [] id = new byte[regionName.length - offset - 1];
452     System.arraycopy(regionName, offset + 1, id, 0,
453         regionName.length - offset - 1);
454     byte [][] elements = new byte[3][];
455     elements[0] = tableName;
456     elements[1] = startKey;
457     elements[2] = id;
458     return elements;
459   }

533   /**
534    * Returns true if the given inclusive range of rows is fully contained
535    * by this region. For example, if the region is foo,a,g and this is
536    * passed ["b","c"] or ["a","c"] it will return true, but if this is passed
537    * ["b","z"] it will return false.
538    * @throws IllegalArgumentException if the range passed is invalid (ie end < start)
539    */
540   public boolean containsRange(byte[] rangeStartKey, byte[] rangeEndKey) {
541     if (Bytes.compareTo(rangeStartKey, rangeEndKey) > 0) {
542       throw new IllegalArgumentException(
543       "Invalid range: " + Bytes.toStringBinary(rangeStartKey) +
544       " > " + Bytes.toStringBinary(rangeEndKey));
545     }
546 
547     boolean firstKeyInRange = Bytes.compareTo(rangeStartKey, startKey) >= 0;
548     boolean lastKeyInRange =
549       Bytes.compareTo(rangeEndKey, endKey) < 0 ||
550       Bytes.equals(endKey, HConstants.EMPTY_BYTE_ARRAY);
551     return firstKeyInRange && lastKeyInRange;
552   }
553 
554   /**
555    * Return true if the given row falls in this region.
556    */
557   public boolean containsRow(byte[] row) {
558     return Bytes.compareTo(row, startKey) >= 0 &&
559       (Bytes.compareTo(row, endKey) < 0 ||
560        Bytes.equals(endKey, HConstants.EMPTY_BYTE_ARRAY));
561   }

796   public int compareTo(HRegionInfo o) {
797     if (o == null) {
798       return 1;
799     }
800 
801     // Are regions of same table?
802     int result = Bytes.compareTo(this.tableName, o.tableName);
803     if (result != 0) {
804       return result;
805     }
806 
807     // Compare start keys.
808     result = Bytes.compareTo(this.startKey, o.startKey);
809     if (result != 0) {
810       return result;
811     }
812 
813     // Compare end keys.
814     result = Bytes.compareTo(this.endKey, o.endKey);
815 
816     if (result != 0) {
817       if (this.getStartKey().length != 0
818               && this.getEndKey().length == 0) {
819           return 1; // this is last region
820       }
821       if (o.getStartKey().length != 0
822               && o.getEndKey().length == 0) {
823           return -1; // o is the last region
824       }
825       return result;
826     }
827 
828     // regionId is usually milli timestamp -- this defines older stamps
829     // to be "smaller" than newer stamps in sort order.
830     if (this.regionId > o.regionId) {
831       return 1;
832     } else if (this.regionId < o.regionId) {
833       return -1;
834     }
835 
836     if (this.offLine == o.offLine)
837       return 0;
838     if (this.offLine == true) return -1;
839 
840     return 1;
841   }

888   public static HRegionInfo convert(final RegionInfo proto) {

1055   /**
1056    * Parses an HRegionInfo instance from the passed in stream.  Presumes the HRegionInfo was
1057    * serialized to the stream with {@link #toDelimitedByteArray()}
1058    * @param in
1059    * @return An instance of HRegionInfo.
1060    * @throws IOException
1061    */
1062   public static HRegionInfo parseFrom(final DataInputStream in) throws IOException {
1063     // I need to be able to move back in the stream if this is not a pb serialization so I can
1064     // do the Writable decoding instead.
1065     int pblen = ProtobufUtil.lengthOfPBMagic();
1066     byte [] pbuf = new byte[pblen];
1067     if (in.markSupported()) { //read it with mark()
1068       in.mark(pblen);
1069     }
1070     int read = in.read(pbuf); //assumption: if Writable serialization, it should be longer than pblen.
1071     if (read != pblen) throw new IOException("read=" + read + ", wanted=" + pblen);
1072     if (ProtobufUtil.isPBMagicPrefix(pbuf)) {
1073       return convert(HBaseProtos.RegionInfo.parseDelimitedFrom(in));
1074     } else {
1075         // Presume Writables.  Need to reset the stream since it didn't start w/ pb.
1076       if (in.markSupported()) {
1077         in.reset();
1078         HRegionInfo hri = new HRegionInfo();
1079         hri.readFields(in);
1080         return hri;
1081       } else {
1082         //we cannot use BufferedInputStream, it consumes more than we read from the underlying IS
1083         ByteArrayInputStream bais = new ByteArrayInputStream(pbuf);
1084         SequenceInputStream sis = new SequenceInputStream(bais, in); //concatenate input streams
1085         HRegionInfo hri = new HRegionInfo();
1086         hri.readFields(new DataInputStream(sis));
1087         return hri;
1088       }
1089     }
1090   }

1095    * to EOF which may not be what you want). {@link #parseDelimitedFrom(byte[], int, int)} can
1096    * be used to read back the instances.

1104     int size = 0;

1107       size += bytes[i].length;
