84      /**
85       * Create a new ClientScanner for the specified table
86       * Note that the passed {@link Scan}'s start row maybe changed changed. 
87       * 
88       * @param conf The {@link Configuration} to use.
89       * @param scan {@link Scan} to use in this scanner
90       * @param tableName The table that we wish to scan
91       * @param connection Connection identifying the cluster
92       * @throws IOException
93       */
94      public ClientScanner(final Configuration conf, final Scan scan,
95        final byte[] tableName, HConnection connection) throws IOException {
96        if (LOG.isDebugEnabled()) {
97          LOG.debug("Creating scanner over "
98              + Bytes.toString(tableName)
99              + " starting at key '" + Bytes.toStringBinary(scan.getStartRow()) + "'");
100       }
101       this.scan = scan;
102       this.tableName = tableName;
103       this.lastNext = System.currentTimeMillis();
104       this.connection = connection;
105       if (scan.getMaxResultSize() > 0) {
106         this.maxScannerResultSize = scan.getMaxResultSize();
107       } else {
108         this.maxScannerResultSize = conf.getLong(
109           HConstants.HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE_KEY,
110           HConstants.DEFAULT_HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE);
111       }
112       this.scannerTimeout = conf.getInt(HConstants.HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD,
113         HConstants.DEFAULT_HBASE_CLIENT_SCANNER_TIMEOUT_PERIOD);
114 
115       // check if application wants to collect scan metrics
116       byte[] enableMetrics = scan.getAttribute(
117         Scan.SCAN_ATTRIBUTES_METRICS_ENABLE);
118       if (enableMetrics != null && Bytes.toBoolean(enableMetrics)) {
119         scanMetrics = new ScanMetrics();
120       }
121 
122       // Use the caching from the Scan.  If not set, use the default cache setting for this table.
123       if (this.scan.getCaching() > 0) {
124         this.caching = this.scan.getCaching();
125       } else {
126         this.caching = conf.getInt(
127             HConstants.HBASE_CLIENT_SCANNER_CACHING,
128             HConstants.DEFAULT_HBASE_CLIENT_SCANNER_CACHING);
129       }
130 
131       // initialize the scanner
132       nextScanner(this.caching, false);
133     }

151     // returns true if the passed region endKey
152     private boolean checkScanStopRow(final byte [] endKey) {
153       if (this.scan.getStopRow().length > 0) {
154         // there is a stop row, check to see if we are past it.
155         byte [] stopRow = scan.getStopRow();
156         int cmp = Bytes.compareTo(stopRow, 0, stopRow.length,
157           endKey, 0, endKey.length);
158         if (cmp <= 0) {
159           // stopRow <= endKey (endKey is equals to or larger than stopRow)
160           // This is a stop.
161           return true;
162         }
163       }
164       return false; //unlikely.
165     }

167     /*
168      * Gets a scanner for the next region.  If this.currentRegion != null, then
169      * we will move to the endrow of this.currentRegion.  Else we will get
170      * scanner at the scan.getStartRow().  We will go no further, just tidy
171      * up outstanding scanners, if <code>currentRegion != null</code> and
172      * <code>done</code> is true.
173      * @param nbRows
174      * @param done Server-side says we're done scanning.
175      */
176     private boolean nextScanner(int nbRows, final boolean done)
177     throws IOException {
178       // Close the previous scanner if it's open
179       if (this.callable != null) {
180         this.callable.setClose();
181         callable.withRetries();
182         this.callable = null;
183       }
184 
185       // Where to start the next scanner
186       byte [] localStartKey;
187 
188       // if we're at end of table, close and return false to stop iterating
189       if (this.currentRegion != null) {
190         byte [] endKey = this.currentRegion.getEndKey();
191         if (endKey == null ||
192             Bytes.equals(endKey, HConstants.EMPTY_BYTE_ARRAY) ||
193             checkScanStopRow(endKey) ||
194             done) {
195           close();
196           if (LOG.isDebugEnabled()) {
197             LOG.debug("Finished with scanning at " + this.currentRegion);
198           }
199           return false;
200         }
201         localStartKey = endKey;
202         if (LOG.isDebugEnabled()) {
203           LOG.debug("Finished with region " + this.currentRegion);
204         }
205       } else {
206         localStartKey = this.scan.getStartRow();
207       }
208 
209       if (LOG.isDebugEnabled()) {
210         LOG.debug("Advancing internal scanner to startKey at '" +
211           Bytes.toStringBinary(localStartKey) + "'");
212       }
213       try {
214         callable = getScannerCallable(localStartKey, nbRows);
215         // Open a scanner on the region server starting at the
216         // beginning of the region
217         callable.withRetries();
218         this.currentRegion = callable.getHRegionInfo();
219         if (this.scanMetrics != null) {
220           this.scanMetrics.countOfRegions.incrementAndGet();
221         }
222       } catch (IOException e) {
223         close();
224         throw e;
225       }
226       return true;
227     }
228 
229     protected ScannerCallable getScannerCallable(byte [] localStartKey,
230         int nbRows) {
231       scan.setStartRow(localStartKey);
232       ScannerCallable s = new ScannerCallable(getConnection(),
233         getTableName(), scan, this.scanMetrics);
234       s.setCaching(nbRows);
235       return s;
236     }

238     /**
239      * Publish the scan metrics. For now, we use scan.setAttribute to pass the metrics back to the
240      * application or TableInputFormat.Later, we could push it to other systems. We don't use metrics
241      * framework because it doesn't support multi-instances of the same metrics on the same machine;
242      * for scan/map reduce scenarios, we will have multiple scans running at the same time.
243      *
244      * By default, scan metrics are disabled; if the application wants to collect them, this behavior
245      * can be turned on by calling calling:
246      *
247      * scan.setAttribute(SCAN_ATTRIBUTES_METRICS_ENABLE, Bytes.toBytes(Boolean.TRUE))
248      */
249     private void writeScanMetrics() throws IOException {
250       if (this.scanMetrics == null) {
251         return;
252       }
253       final DataOutputBuffer d = new DataOutputBuffer();
254       MapReduceProtos.ScanMetrics pScanMetrics = ProtobufUtil.toScanMetrics(scanMetrics);
255       scan.setAttribute(Scan.SCAN_ATTRIBUTES_METRICS_DATA, pScanMetrics.toByteArray());
256     }

258     public Result next() throws IOException {
259       // If the scanner is closed and there's nothing left in the cache, next is a no-op.
260       if (cache.size() == 0 && this.closed) {
261         return null;
262       }
263       if (cache.size() == 0) {
264         Result [] values = null;
265         long remainingResultSize = maxScannerResultSize;
266         int countdown = this.caching;
267         // We need to reset it if it's a new callable that was created
268         // with a countdown in nextScanner
269         callable.setCaching(this.caching);
270         // This flag is set when we want to skip the result returned.  We do
271         // this when we reset scanner because it split under us.
272         boolean skipFirst = false;
273         boolean retryAfterOutOfOrderException  = true;
274         do {
275           try {
276             if (skipFirst) {
277               // Skip only the first row (which was the last row of the last
278               // already-processed batch).
279               callable.setCaching(1);
280               values = callable.withRetries();
281               callable.setCaching(this.caching);
282               skipFirst = false;
283             }
284             // Server returns a null values if scanning is to stop.  Else,
285             // returns an empty array if scanning is to go on and we've just
286             // exhausted current region.
287             values = callable.withRetries();
288             retryAfterOutOfOrderException  = true;
289           } catch (DoNotRetryIOException e) {
290             if (e instanceof UnknownScannerException) {
291               long timeout = lastNext + scannerTimeout;
292               // If we are over the timeout, throw this exception to the client
293               // Else, it's because the region moved and we used the old id
294               // against the new region server; reset the scanner.
295               if (timeout < System.currentTimeMillis()) {
296                 long elapsed = System.currentTimeMillis() - lastNext;
297                 ScannerTimeoutException ex = new ScannerTimeoutException(
298                     elapsed + "ms passed since the last invocation, " +
299                         "timeout is currently set to " + scannerTimeout);
300                 ex.initCause(e);
301                 throw ex;
302               }
303             } else {
304               Throwable cause = e.getCause();
305               if ((cause == null || (!(cause instanceof NotServingRegionException)
306                   && !(cause instanceof RegionServerStoppedException)))
307                   && !(e instanceof OutOfOrderScannerNextException)) {
308                 throw e;
309               }
310             }
311             // Else, its signal from depths of ScannerCallable that we got an
312             // NSRE on a next and that we need to reset the scanner.
313             if (this.lastResult != null) {
314               this.scan.setStartRow(this.lastResult.getRow());
315               // Skip first row returned.  We already let it out on previous
316               // invocation.
317               skipFirst = true;
318             }
319             if (e instanceof OutOfOrderScannerNextException) {
320               if (retryAfterOutOfOrderException) {
321                 retryAfterOutOfOrderException = false;
322               } else {
323                 throw new DoNotRetryIOException("Failed after retry"
324                     + ", it could be cause by rpc timeout", e);
325               }
326             }
327             // Clear region
328             this.currentRegion = null;
329             callable = null;
330             continue;
331           }
332           long currentTime = System.currentTimeMillis();
333           if (this.scanMetrics != null ) {
334             this.scanMetrics.sumOfMillisSecBetweenNexts.addAndGet(currentTime-lastNext);
335           }
336           lastNext = currentTime;
337           if (values != null && values.length > 0) {
338             for (Result rs : values) {
339               cache.add(rs);
340               for (KeyValue kv : rs.raw()) {
341                   remainingResultSize -= kv.heapSize();
342               }
343               countdown--;
344               this.lastResult = rs;
345             }
346           }
347           // Values == null means server-side filter has determined we must STOP
348         } while (remainingResultSize > 0 && countdown > 0 && nextScanner(countdown, values == null));
349       }
350 
351       if (cache.size() > 0) {
352         return cache.poll();
353       }
354 
355       // if we exhausted this scanner before calling close, write out the scan metrics
356       writeScanMetrics();
357       return null;
358     }

360     /**
361      * Get <param>nbRows</param> rows.
362      * How many RPCs are made is determined by the {@link Scan#setCaching(int)}
363      * setting (or hbase.client.scanner.caching in hbase-site.xml).
364      * @param nbRows number of rows to return
365      * @return Between zero and <param>nbRows</param> RowResults.  Scan is done
366      * if returned array is of zero-length (We never return null).
367      * @throws IOException
368      */
369     public Result [] next(int nbRows) throws IOException {
370       // Collect values to be returned here
371       ArrayList<Result> resultSets = new ArrayList<Result>(nbRows);
372       for(int i = 0; i < nbRows; i++) {
373         Result next = next();
374         if (next != null) {
375           resultSets.add(next);
376         } else {
377           break;
378         }
379       }
380       return resultSets.toArray(new Result[resultSets.size()]);
381     }

383     public void close() {
384       if (callable != null) {
385         callable.setClose();
386         try {
387           callable.withRetries();
388         } catch (IOException e) {
389           // We used to catch this error, interpret, and rethrow. However, we
390           // have since decided that it's not nice for a scanner's close to
391           // throw exceptions. Chances are it was just an UnknownScanner
392           // exception due to lease time out.
393         } finally {
394           // we want to output the scan metrics even if an error occurred on close
395           try {
396             writeScanMetrics();
397           } catch (IOException e) {
398             // As above, we still don't want the scanner close() method to throw.
399           }
400         }
401         callable = null;
402       }
403       closed = true;
404     }
