19  package org.apache.hadoop.hbase.regionserver;
20  
21  import java.io.IOException;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  import org.apache.hadoop.classification.InterfaceAudience;
26  import org.apache.hadoop.hbase.RemoteExceptionHandler;
27  import org.apache.hadoop.hbase.util.Bytes;
28  import org.apache.hadoop.util.StringUtils;
29  
30  import com.google.common.base.Preconditions;
31  
32  /**
33   * Handles processing region splits. Put in a queue, owned by HRegionServer.
34   */
35  @InterfaceAudience.Private
36  class SplitRequest implements Runnable {
37    static final Log LOG = LogFactory.getLog(SplitRequest.class);
38    private final HRegion parent;
39    private final byte[] midKey;
40    private final HRegionServer server;
41  
42    SplitRequest(HRegion region, byte[] midKey, HRegionServer hrs) {
43      Preconditions.checkNotNull(hrs);
44      this.parent = region;
45      this.midKey = midKey;
46      this.server = hrs;
47    }
48  
49    @Override
50    public String toString() {
51      return "regionName=" + parent + ", midKey=" + Bytes.toStringBinary(midKey);
52    }
53  
54    @Override
55    public void run() {
56      if (this.server.isStopping() || this.server.isStopped()) {
57        LOG.debug("Skipping split because server is stopping=" +
58          this.server.isStopping() + " or stopped=" + this.server.isStopped());
59        return;
60      }
61      try {
62        final long startTime = System.currentTimeMillis();
63        SplitTransaction st = new SplitTransaction(parent, midKey);
64        // If prepare does not return true, for some reason -- logged inside in
65        // the prepare call -- we are not ready to split just now. Just return.
66        if (!st.prepare()) return;
67        try {
68          st.execute(this.server, this.server);
69        } catch (Exception e) {
70          if (this.server.isStopping() || this.server.isStopped()) {
71            LOG.info(
72                "Skip rollback/cleanup of failed split of "
73                    + parent.getRegionNameAsString() + " because server is"
74                    + (this.server.isStopping() ? " stopping" : " stopped"), e);
75            return;
76          }
77          try {
78            LOG.info("Running rollback/cleanup of failed split of " +
79              parent.getRegionNameAsString() + "; " + e.getMessage(), e);
80            if (st.rollback(this.server, this.server)) {
81              LOG.info("Successful rollback of failed split of " +
82                parent.getRegionNameAsString());
83            } else {
84              this.server.abort("Abort; we got an error after point-of-no-return");
85            }
86          } catch (RuntimeException ee) {
87            String msg = "Failed rollback of failed split of " +
88              parent.getRegionNameAsString() + " -- aborting server";
89            // If failed rollback, kill this server to avoid having a hole in table.
90            LOG.info(msg, ee);
91            this.server.abort(msg);
92          }
93          return;
94        }
95        LOG.info("Region split, META updated, and report to master. Parent="
96            + parent.getRegionInfo().getRegionNameAsString() + ", new regions: "
97            + st.getFirstDaughter().getRegionNameAsString() + ", "
98            + st.getSecondDaughter().getRegionNameAsString() + ". Split took "
99            + StringUtils.formatTimeDiff(System.currentTimeMillis(), startTime));
100     } catch (IOException ex) {
101       LOG.error("Split failed " + this, RemoteExceptionHandler
102           .checkIOException(ex));
103       server.checkFileSystem();
104     } finally {
105       if (this.parent.getCoprocessorHost() != null) {
106         try {
107           this.parent.getCoprocessorHost().postCompleteSplit();
108         } catch (IOException io) {
109           LOG.error("Split failed " + this,
110               RemoteExceptionHandler.checkIOException(io));
111         }
112       }
113     }
114   }
115 }
