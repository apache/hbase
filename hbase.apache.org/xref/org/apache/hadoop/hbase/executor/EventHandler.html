34  /**
35   * Abstract base class for all HBase event handlers. Subclasses should
36   * implement the {@link #process()} method.  Subclasses should also do all
37   * necessary checks up in their constructor if possible -- check table exists,
38   * is disabled, etc. -- so they fail fast rather than later when process is
39   * running.  Do it this way because process be invoked directly but event
40   * handlers are also
41   * run in an executor context -- i.e. asynchronously -- and in this case,
42   * exceptions thrown at process time will not be seen by the invoker, not till
43   * we implement a call-back mechanism so the client can pick them up later.
44   * <p>
45   * Event handlers have an {@link EventType}.
46   * {@link EventType} is a list of ALL handler event types.  We need to keep
47   * a full list in one place -- and as enums is a good shorthand for an
48   * implemenations -- because event handlers can be passed to executors when
49   * they are to be run asynchronously. The
50   * hbase executor, see {@link ExecutorService}, has a switch for passing
51   * event type to executor.
52   * <p>
53   * Event listeners can be installed and will be called pre- and post- process if
54   * this EventHandler is run in a Thread (its a Runnable so if its {@link #run()}
55   * method gets called).  Implement
56   * {@link EventHandlerListener}s, and registering using
57   * {@link #setListener(EventHandlerListener)}.
58   * @see ExecutorService
59   */
60  @InterfaceAudience.Private
61  public abstract class EventHandler implements Runnable, Comparable<Runnable> {
62    private static final Log LOG = LogFactory.getLog(EventHandler.class);
63

65    protected EventType eventType;
66

70    protected static AtomicLong seqids = new AtomicLong(0);

72    // sequence id for this event
73    private final long seqid;

75    // Listener to call pre- and post- processing.  May be null.
76    private EventHandlerListener listener;
77  
78    // Time to wait for events to happen, should be kept short
79    protected final int waitingTimeForEvents;

99    /**
100    * List of all HBase event handler types.  Event types are named by a
101    * convention: event type names specify the component from which the event
102    * originated and then where its destined -- e.g. RS2ZK_ prefix means the
103    * event came from a regionserver destined for zookeeper -- and then what
104    * the even is; e.g. REGION_OPENING.
105    * 
106    * <p>We give the enums indices so we can add types later and keep them
107    * grouped together rather than have to add them always to the end as we
108    * would have to if we used raw enum ordinals.
109    */
110   public enum EventType {
111     // Messages originating from RS (NOTE: there is NO direct communication from
112     // RS to Master). These are a result of RS updates into ZK.
113     // RS_ZK_REGION_CLOSING    (1),   // It is replaced by M_ZK_REGION_CLOSING(HBASE-4739)
114     RS_ZK_REGION_CLOSED       (2, ExecutorType.MASTER_CLOSE_REGION),   // RS has finished closing a region
115     RS_ZK_REGION_OPENING      (3, null), // RS is in process of opening a region
116     RS_ZK_REGION_OPENED       (4, ExecutorType.MASTER_OPEN_REGION), // RS has finished opening a region
117     RS_ZK_REGION_SPLITTING    (5, null), // RS has started a region split
118     RS_ZK_REGION_SPLIT        (6, ExecutorType.MASTER_SERVER_OPERATIONS),   // RS split has completed.
119     RS_ZK_REGION_FAILED_OPEN  (7, ExecutorType.MASTER_CLOSE_REGION),   // RS failed to open a region
120 
121     // Messages originating from Master to RS
122     M_RS_OPEN_REGION          (20, ExecutorType.RS_OPEN_REGION),  // Master asking RS to open a region
123     M_RS_OPEN_ROOT            (21, ExecutorType.RS_OPEN_REGION),  // Master asking RS to open root
124     M_RS_OPEN_META            (22, ExecutorType.RS_OPEN_META),  // Master asking RS to open meta
125     M_RS_CLOSE_REGION         (23, ExecutorType.RS_CLOSE_REGION),  // Master asking RS to close a region
126     M_RS_CLOSE_ROOT           (24, ExecutorType.RS_CLOSE_ROOT),  // Master asking RS to close root
127     M_RS_CLOSE_META           (25, ExecutorType.RS_CLOSE_META),  // Master asking RS to close meta
128 
129     // Messages originating from Client to Master
130     C_M_DELETE_TABLE          (40, ExecutorType.MASTER_TABLE_OPERATIONS),   // Client asking Master to delete a table
131     C_M_DISABLE_TABLE         (41, ExecutorType.MASTER_TABLE_OPERATIONS),   // Client asking Master to disable a table
132     C_M_ENABLE_TABLE          (42, ExecutorType.MASTER_TABLE_OPERATIONS),   // Client asking Master to enable a table
133     C_M_MODIFY_TABLE          (43, ExecutorType.MASTER_TABLE_OPERATIONS),   // Client asking Master to modify a table
134     C_M_ADD_FAMILY            (44, null), // Client asking Master to add family to table
135     C_M_DELETE_FAMILY         (45, null), // Client asking Master to delete family of table
136     C_M_MODIFY_FAMILY         (46, null), // Client asking Master to modify family of table
137     C_M_CREATE_TABLE          (47, ExecutorType.MASTER_TABLE_OPERATIONS),   // Client asking Master to create a table
138 
139     // Updates from master to ZK. This is done by the master and there is
140     // nothing to process by either Master or RS
141     M_ZK_REGION_OFFLINE       (50, null), // Master adds this region as offline in ZK
142     M_ZK_REGION_CLOSING       (51, null), // Master adds this region as closing in ZK
143 
144     // Master controlled events to be executed on the master
145     M_SERVER_SHUTDOWN         (70, ExecutorType.MASTER_SERVER_OPERATIONS),  // Master is processing shutdown of a RS
146     M_META_SERVER_SHUTDOWN    (72, ExecutorType.MASTER_META_SERVER_OPERATIONS);  // Master is processing shutdown of RS hosting a meta region (-ROOT- or .META.).
147 
148     private final int code;
149     private final ExecutorService.ExecutorType executor;

152      * Constructor

154     EventType(final int code, final ExecutorType executor) {
155       this.code = code;
156       this.executor = executor;
157     }

159     public int getCode() {
160       return this.code;
161     }

163     public static EventType get(final int code) {
164       // Is this going to be slow?  Its used rare but still...
165       for (EventType et: EventType.values()) {
166         if (et.getCode() == code) return et;
167       }
168       throw new IllegalArgumentException("Unknown code " + code);
169     }

171     public boolean isOnlineSchemaChangeSupported() {
172       return (
173         this.equals(EventType.C_M_ADD_FAMILY) ||
174         this.equals(EventType.C_M_DELETE_FAMILY) ||
175         this.equals(EventType.C_M_MODIFY_FAMILY) ||
176         this.equals(EventType.C_M_MODIFY_TABLE)
177       );
178     }

180     ExecutorType getExecutorServiceType() {
181       return this.executor;
182     }

185   /**
186    * Default base class constructor.
187    */
188   public EventHandler(Server server, EventType eventType) {
189     this.parent = Trace.currentTrace();
190     this.server = server;
191     this.eventType = eventType;
192     seqid = seqids.incrementAndGet();
193     this.waitingTimeForEvents = server.getConfiguration().
194         getInt("hbase.master.event.waiting.time", 1000);
195   }
196 
197   public void run() {
198     Span chunk = Trace.startSpan(Thread.currentThread().getName(), parent,
199           Sampler.ALWAYS);
200     try {
201       if (getListener() != null) getListener().beforeProcess(this);
202       process();
203       if (getListener() != null) getListener().afterProcess(this);
204     } catch(Throwable t) {
205       LOG.error("Caught throwable while processing event " + eventType, t);
206     } finally {
207       chunk.stop();
208     }
209   }

226   /**
227    * Get the priority level for this handler instance.  This uses natural
228    * ordering so lower numbers are higher priority.
229    * <p>
230    * Lowest priority is Integer.MAX_VALUE.  Highest priority is 0.
231    * <p>
232    * Subclasses should override this method to allow prioritizing handlers.
233    * <p>
234    * Handlers with the same priority are handled in FIFO order.
235    * <p>
236    * @return Integer.MAX_VALUE by default, override to set higher priorities
237    */
238   public int getPriority() {
239     return Integer.MAX_VALUE;
240   }

246     return this.seqid;

249   /**
250    * Default prioritized runnable comparator which implements a FIFO ordering.
251    * <p>
252    * Subclasses should not override this.  Instead, if they want to implement
253    * priority beyond FIFO, they should override {@link #getPriority()}.
254    */
255   @Override
256   public int compareTo(Runnable o) {
257     EventHandler eh = (EventHandler)o;
258     if(getPriority() != eh.getPriority()) {
259       return (getPriority() < eh.getPriority()) ? -1 : 1;
260     }
261     return (this.seqid < eh.seqid) ? -1 : 1;
262   }
