36  /**
37   * The filter looks for the given columns in KeyValue. Once there is a match for
38   * any one of the columns, it returns ReturnCode.NEXT_ROW for remaining
39   * KeyValues in the row.
40   * <p>
41   * Note : It may emit KVs which do not have the given columns in them, if
42   * these KVs happen to occur before a KV which does have a match. Given this
43   * caveat, this filter is only useful for special cases like {@link RowCounter}.
44   * <p>
45   */
46  @InterfaceAudience.Public
47  @InterfaceStability.Stable
48  public class FirstKeyValueMatchingQualifiersFilter extends FirstKeyOnlyFilter {
49  
50    private Set<byte []> qualifiers;
51  
52    /**
53     * Constructor which takes a set of columns. As soon as first KeyValue
54     * matching any of these columns is found, filter moves to next row.
55     * 
56     * @param qualifiers the set of columns to me matched.
57     */
58    public FirstKeyValueMatchingQualifiersFilter(Set<byte []> qualifiers) {
59      this.qualifiers = qualifiers;
60    }
61  
62    public ReturnCode filterKeyValue(KeyValue v) {
63      if (hasFoundKV()) {
64        return ReturnCode.NEXT_ROW;
65      } else if (hasOneMatchingQualifier(v)) {
66        setFoundKV(true);
67      }
68      return ReturnCode.INCLUDE;
69    }
70  
71    private boolean hasOneMatchingQualifier(KeyValue v) {
72      for (byte[] q : qualifiers) {
73        if (v.matchingQualifier(q)) {
74          return true;
75        }
76      }
77      return false;
78    }
79  
80    /**
81     * @return The filter serialized using pb
82     */
83    public byte [] toByteArray() {
84      FilterProtos.FirstKeyValueMatchingQualifiersFilter.Builder builder =
85        FilterProtos.FirstKeyValueMatchingQualifiersFilter.newBuilder();
86      for (byte[] qualifier : qualifiers) {
87        if (qualifier != null) builder.addQualifiers(ByteString.copyFrom(qualifier));
88      }
89      return builder.build().toByteArray();
90    }
91  
92    /**
93     * @param pbBytes A pb serialized {@link FirstKeyValueMatchingQualifiersFilter} instance
94     * @return An instance of {@link FirstKeyValueMatchingQualifiersFilter} made from <code>bytes</code>
95     * @throws DeserializationException
96     * @see #toByteArray
97     */
98    public static FirstKeyValueMatchingQualifiersFilter parseFrom(final byte [] pbBytes)
99    throws DeserializationException {
100     FilterProtos.FirstKeyValueMatchingQualifiersFilter proto;
101     try {
102       proto = FilterProtos.FirstKeyValueMatchingQualifiersFilter.parseFrom(pbBytes);
103     } catch (InvalidProtocolBufferException e) {
104       throw new DeserializationException(e);
105     }
106 
107     TreeSet<byte []> qualifiers = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
108     for (ByteString qualifier : proto.getQualifiersList()) {
109       qualifiers.add(qualifier.toByteArray());
110     }
111     return new FirstKeyValueMatchingQualifiersFilter(qualifiers);
112   }
113 
114   /**
115    * @param other
116    * @return true if and only if the fields of the filter that are serialized
117    * are equal to the corresponding fields in other.  Used for testing.
118    */
119   boolean areSerializedFieldsEqual(Filter o) {
120     if (o == this) return true;
121     if (!(o instanceof FirstKeyValueMatchingQualifiersFilter)) return false;
122 
123     FirstKeyValueMatchingQualifiersFilter other = (FirstKeyValueMatchingQualifiersFilter)o;
124     return this.qualifiers.equals(other.qualifiers);
125   }
