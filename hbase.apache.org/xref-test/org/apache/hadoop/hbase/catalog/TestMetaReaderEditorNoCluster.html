68      boolean aborted = false;
69      @Override

77        return this.aborted;

91    @Test
92    public void testGetHRegionInfo() throws IOException {
93      assertNull(HRegionInfo.getHRegionInfo(new Result()));
94  
95      List<KeyValue> kvs = new ArrayList<KeyValue>();
96      Result r = new Result(kvs);
97      assertNull(HRegionInfo.getHRegionInfo(r));
98  
99      byte [] f = HConstants.CATALOG_FAMILY;
100     // Make a key value that doesn't have the expected qualifier.
101     kvs.add(new KeyValue(HConstants.EMPTY_BYTE_ARRAY, f,
102       HConstants.SERVER_QUALIFIER, f));
103     r = new Result(kvs);
104     assertNull(HRegionInfo.getHRegionInfo(r));
105     // Make a key that does not have a regioninfo value.
106     kvs.add(new KeyValue(HConstants.EMPTY_BYTE_ARRAY, f,
107       HConstants.REGIONINFO_QUALIFIER, f));
108     HRegionInfo hri = HRegionInfo.getHRegionInfo(new Result(kvs));
109     assertTrue(hri == null);
110     // OK, give it what it expects
111     kvs.clear();
112     kvs.add(new KeyValue(HConstants.EMPTY_BYTE_ARRAY, f,
113       HConstants.REGIONINFO_QUALIFIER,
114       HRegionInfo.FIRST_META_REGIONINFO.toByteArray()));
115     hri = HRegionInfo.getHRegionInfo(new Result(kvs));
116     assertNotNull(hri);
117     assertTrue(hri.equals(HRegionInfo.FIRST_META_REGIONINFO));
118   }

120   /**
121    * Test that MetaReader will ride over server throwing
122    * "Server not running" IOEs.
123    * @see https://issues.apache.org/jira/browse/HBASE-3446
124    * @throws IOException
125    * @throws InterruptedException
126    */
127   @Test
128   public void testRideOverServerNotRunning()
129       throws IOException, InterruptedException, ServiceException {
130     // Need a zk watcher.
131     ZooKeeperWatcher zkw = new ZooKeeperWatcher(UTIL.getConfiguration(),
132       this.getClass().getSimpleName(), ABORTABLE, true);
133     // This is a servername we use in a few places below.
134     ServerName sn = new ServerName("example.com", 1234, System.currentTimeMillis());
135 
136     HConnection connection = null;
137     CatalogTracker ct = null;
138     try {
139       // Mock an ClientProtocol. Our mock implementation will fail a few
140       // times when we go to open a scanner.
141       final ClientProtocol implementation = Mockito.mock(ClientProtocol.class);
142       // When scan called throw IOE 'Server not running' a few times
143       // before we return a scanner id.  Whats WEIRD is that these
144       // exceptions do not show in the log because they are caught and only
145       // printed if we FAIL.  We eventually succeed after retry so these don't
146       // show.  We will know if they happened or not because we will ask
147       // mockito at the end of this test to verify that scan was indeed
148       // called the wanted number of times.
149       List<KeyValue> kvs = new ArrayList<KeyValue>();
150       final byte [] rowToVerify = Bytes.toBytes("rowToVerify");
151       kvs.add(new KeyValue(rowToVerify,
152         HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER,
153         HRegionInfo.FIRST_META_REGIONINFO.toByteArray()));
154       kvs.add(new KeyValue(rowToVerify,
155         HConstants.CATALOG_FAMILY, HConstants.SERVER_QUALIFIER,
156         Bytes.toBytes(sn.getHostAndPort())));
157       kvs.add(new KeyValue(rowToVerify,
158         HConstants.CATALOG_FAMILY, HConstants.STARTCODE_QUALIFIER,
159         Bytes.toBytes(sn.getStartcode())));
160       final Result [] results = new Result [] {new Result(kvs)};
161       ScanResponse.Builder builder = ScanResponse.newBuilder();
162       for (Result result: results) {
163         builder.addResult(ProtobufUtil.toResult(result));
164       }
165       Mockito.when(implementation.scan(
166         (RpcController)Mockito.any(), (ScanRequest)Mockito.any())).
167           thenThrow(new ServiceException("Server not running (1 of 3)")).
168           thenThrow(new ServiceException("Server not running (2 of 3)")).
169           thenThrow(new ServiceException("Server not running (3 of 3)")).
170           thenReturn(ScanResponse.newBuilder().setScannerId(1234567890L).build())
171             .thenReturn(builder.build()).thenReturn(
172               ScanResponse.newBuilder().setMoreResults(false).build());
173 
174       // Associate a spied-upon HConnection with UTIL.getConfiguration.  Need
175       // to shove this in here first so it gets picked up all over; e.g. by
176       // HTable.
177       connection = HConnectionTestingUtility.getSpiedConnection(UTIL.getConfiguration());
178       // Fix the location lookup so it 'works' though no network.  First
179       // make an 'any location' object.
180       final HRegionLocation anyLocation =
181         new HRegionLocation(HRegionInfo.FIRST_META_REGIONINFO, sn.getHostname(),
182           sn.getPort());
183       // Return the any location object when locateRegion is called in HTable
184       // constructor and when its called by ServerCallable (it uses getRegionLocation).
185       // The ugly format below comes of 'Important gotcha on spying real objects!' from
186       // http://mockito.googlecode.com/svn/branches/1.6/javadoc/org/mockito/Mockito.html
187       Mockito.doReturn(anyLocation).
188         when(connection).locateRegion((byte[]) Mockito.any(), (byte[]) Mockito.any());
189       Mockito.doReturn(anyLocation).
190         when(connection).getRegionLocation((byte[]) Mockito.any(),
191           (byte[]) Mockito.any(), Mockito.anyBoolean());
192 
193       // Now shove our HRI implementation into the spied-upon connection.
194       Mockito.doReturn(implementation).
195         when(connection).getClient(Mockito.anyString(), Mockito.anyInt());
196 
197       // Now start up the catalogtracker with our doctored Connection.
198       ct = new CatalogTracker(zkw, null, connection, ABORTABLE, 0);
199       ct.start();
200       // Scan meta for user tables and verify we got back expected answer.
201       NavigableMap<HRegionInfo, Result> hris = MetaReader.getServerUserRegions(ct, sn);
202       assertEquals(1, hris.size());
203       assertTrue(hris.firstEntry().getKey().equals(HRegionInfo.FIRST_META_REGIONINFO));
204       assertTrue(Bytes.equals(rowToVerify, hris.firstEntry().getValue().getRow()));
205       // Finally verify that scan was called four times -- three times
206       // with exception and then on 4th, 5th and 6th attempt we succeed
207       Mockito.verify(implementation, Mockito.times(6)).
208         scan((RpcController)Mockito.any(), (ScanRequest)Mockito.any());
209     } finally {
210       if (ct != null) ct.stop();
211       HConnectionManager.deleteConnection(UTIL.getConfiguration(), true);
212       zkw.close();
213     }
214   }
