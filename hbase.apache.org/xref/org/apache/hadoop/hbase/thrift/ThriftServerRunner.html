165     String getDescription() {
166       StringBuilder sb = new StringBuilder("Use the " +
167           serverClass.getSimpleName());
168       if (isAlwaysFramed) {

185     static ImplType getServerImpl(Configuration conf) {
186       String confType = conf.get(SERVER_TYPE_CONF_KEY, THREAD_POOL.option);
187       for (ImplType t : values()) {
188         if (confType.equals(t.option)) {
189           return t;
190         }
191       }
192       throw new AssertionError("Unknown server ImplType.option:" + confType);
193     }
194 
195     static void setServerImpl(CommandLine cmd, Configuration conf) {
196       ImplType chosenType = null;
197       int numChosen = 0;
198       for (ImplType t : values()) {
199         if (cmd.hasOption(t.option)) {
200           chosenType = t;
201           ++numChosen;
202         }
203       }
204       if (numChosen < 1) {
205         LOG.info("Using default thrift server type");
206         chosenType = DEFAULT;
207       } else if (numChosen > 1) {
208         throw new AssertionError("Exactly one option out of " +
209           Arrays.toString(values()) + " has to be specified");
210       }
211       LOG.info("Using thrift server type " + chosenType.option);
212       conf.set(SERVER_TYPE_CONF_KEY, chosenType.option);
213     }

244    * Runs the Thrift server

246   @Override
247   public void run() {
248     try {
249       setupServer();
250       tserver.serve();
251     } catch (Exception e) {
252       LOG.fatal("Cannot run ThriftServer", e);
253       // Crash the process if the ThriftServer is not running
254       System.exit(-1);
255     }
256   }

258   public void shutdown() {
259     if (tserver != null) {
260       tserver.stop();
261       tserver = null;
262     }

265   /**
266    * Setting up the thrift TServer
267    */
268   private void setupServer() throws Exception {
269     // Construct correct ProtocolFactory
270     TProtocolFactory protocolFactory;
271     if (conf.getBoolean(COMPACT_CONF_KEY, false)) {
272       LOG.debug("Using compact protocol");
273       protocolFactory = new TCompactProtocol.Factory();
274     } else {
275       LOG.debug("Using binary protocol");
276       protocolFactory = new TBinaryProtocol.Factory();
277     }
278 
279     Hbase.Processor<Hbase.Iface> processor =
280         new Hbase.Processor<Hbase.Iface>(handler);
281     ImplType implType = ImplType.getServerImpl(conf);
282 
283     // Construct correct TransportFactory
284     TTransportFactory transportFactory;
285     if (conf.getBoolean(FRAMED_CONF_KEY, false) || implType.isAlwaysFramed) {
286       transportFactory = new TFramedTransport.Factory();
287       LOG.debug("Using framed transport");
288     } else {
289       transportFactory = new TTransportFactory();
290     }
291 
292     if (conf.get(BIND_CONF_KEY) != null && !implType.canSpecifyBindIP) {
293       LOG.error("Server types " + Joiner.on(", ").join(
294           ImplType.serversThatCannotSpecifyBindIP()) + " don't support IP " +
295           "address binding at the moment. See " +
296           "https://issues.apache.org/jira/browse/HBASE-2155 for details.");
297       throw new RuntimeException(
298           "-" + BIND_CONF_KEY + " not supported with " + implType);
299     }
300 
301     if (implType == ImplType.HS_HA || implType == ImplType.NONBLOCKING ||
302         implType == ImplType.THREADED_SELECTOR) {
303 
304       InetAddress listenAddress = getBindAddress(conf); 
305       TNonblockingServerTransport serverTransport = new TNonblockingServerSocket(
306           new InetSocketAddress(listenAddress, listenPort));
307 
308       if (implType == ImplType.NONBLOCKING) {
309         TNonblockingServer.Args serverArgs =
310             new TNonblockingServer.Args(serverTransport);
311         serverArgs.processor(processor)
312                   .transportFactory(transportFactory)
313                   .protocolFactory(protocolFactory);
314         tserver = new TNonblockingServer(serverArgs);
315       } else if (implType == ImplType.HS_HA) {
316         THsHaServer.Args serverArgs = new THsHaServer.Args(serverTransport);
317         CallQueue callQueue =
318             new CallQueue(new LinkedBlockingQueue<Call>(), metrics);
319         ExecutorService executorService = createExecutor(
320             callQueue, serverArgs.getWorkerThreads());
321         serverArgs.executorService(executorService)
322                   .processor(processor)
323                   .transportFactory(transportFactory)
324                   .protocolFactory(protocolFactory);
325         tserver = new THsHaServer(serverArgs);
326       } else { // THREADED_SELECTOR
327         TThreadedSelectorServer.Args serverArgs =
328             new HThreadedSelectorServerArgs(serverTransport, conf);
329         CallQueue callQueue =
330             new CallQueue(new LinkedBlockingQueue<Call>(), metrics);
331         ExecutorService executorService = createExecutor(
332             callQueue, serverArgs.getWorkerThreads());
333         serverArgs.executorService(executorService)
334                   .processor(processor)
335                   .transportFactory(transportFactory)
336                   .protocolFactory(protocolFactory);
337         tserver = new TThreadedSelectorServer(serverArgs);
338       }
339       LOG.info("starting HBase " + implType.simpleClassName() +
340           " server on " + Integer.toString(listenPort));
341     } else if (implType == ImplType.THREAD_POOL) {
342       // Thread pool server. Get the IP address to bind to.
343       InetAddress listenAddress = getBindAddress(conf);
344 
345       TServerTransport serverTransport = new TServerSocket(
346           new InetSocketAddress(listenAddress, listenPort));
347 
348       TBoundedThreadPoolServer.Args serverArgs =
349           new TBoundedThreadPoolServer.Args(serverTransport, conf);
350       serverArgs.processor(processor)
351                 .transportFactory(transportFactory)
352                 .protocolFactory(protocolFactory);
353       LOG.info("starting " + ImplType.THREAD_POOL.simpleClassName() + " on "
354           + listenAddress + ":" + Integer.toString(listenPort)
355           + "; " + serverArgs);
356       TBoundedThreadPoolServer tserver =
357           new TBoundedThreadPoolServer(serverArgs, metrics);
358       this.tserver = tserver;
359     } else {
360       throw new AssertionError("Unsupported Thrift server implementation: " +
361           implType.simpleClassName());
362     }
363 
364     // A sanity check that we instantiated the right type of server.
365     if (tserver.getClass() != implType.serverClass) {
366       throw new AssertionError("Expected to create Thrift server class " +
367           implType.serverClass.getName() + " but got " +
368           tserver.getClass().getName());
369     }
370 
371     // login the server principal (if using secure Hadoop)
372     if (User.isSecurityEnabled() && User.isHBaseSecurityEnabled(conf)) {
373       String machineName = Strings.domainNamePointerToHostName(
374         DNS.getDefaultHost(conf.get("hbase.thrift.dns.interface", "default"),
375           conf.get("hbase.thrift.dns.nameserver", "default")));
376       User.login(conf, "hbase.thrift.keytab.file",
377           "hbase.thrift.kerberos.principal", machineName);
378     }
379 
380     registerFilters(conf);
381   }

383   ExecutorService createExecutor(BlockingQueue<Runnable> callQueue,
384                                  int workerThreads) {
385     ThreadFactoryBuilder tfb = new ThreadFactoryBuilder();
386     tfb.setDaemon(true);
387     tfb.setNameFormat("thrift-worker-%d");
388     return new ThreadPoolExecutor(workerThreads, workerThreads,
389             Long.MAX_VALUE, TimeUnit.SECONDS, callQueue, tfb.build());
390   }

395     return InetAddress.getByName(bindAddressStr);
396   }

422     /**
423      * Returns a list of all the column families for a given htable.
424      *
425      * @param table
426      * @return
427      * @throws IOException
428      */
429     byte[][] getAllColumns(HTable table) throws IOException {
430       HColumnDescriptor[] cds = table.getTableDescriptor().getColumnFamilies();
431       byte[][] columns = new byte[cds.length][];
432       for (int i = 0; i < cds.length; i++) {
433         columns[i] = Bytes.add(cds[i].getName(),
434             KeyValue.COLUMN_FAMILY_DELIM_ARRAY);
435       }
436       return columns;
437     }

439     /**
440      * Creates and returns an HTable instance from a given table name.
441      *
442      * @param tableName
443      *          name of table
444      * @return HTable object
445      * @throws IOException
446      * @throws IOError
447      */
448     public HTable getTable(final byte[] tableName) throws
449         IOException {
450       String table = new String(tableName);
451       Map<String, HTable> tables = threadLocalTables.get();
452       if (!tables.containsKey(table)) {
453         tables.put(table, new HTable(conf, tableName));
454       }
455       return tables.get(table);
456     }

595     /**
596      * @return the list of regions in the given table, or an empty list if the table does not exist
597      */
598     @Override
599     public List<TRegionInfo> getTableRegions(ByteBuffer tableName)
600     throws IOError {
601       try {
602         HTable table;
603         try {
604           table = getTable(tableName);
605         } catch (TableNotFoundException ex) {
606           return new ArrayList<TRegionInfo>();
607         }
608         Map<HRegionInfo, ServerName> regionLocations =
609             table.getRegionLocations();
610         List<TRegionInfo> results = new ArrayList<TRegionInfo>();
611         for (Map.Entry<HRegionInfo, ServerName> entry :
612             regionLocations.entrySet()) {
613           HRegionInfo info = entry.getKey();
614           ServerName serverName = entry.getValue();
615           TRegionInfo region = new TRegionInfo();
616           region.serverName = ByteBuffer.wrap(
617               Bytes.toBytes(serverName.getHostname()));
618           region.port = serverName.getPort();
619           region.startKey = ByteBuffer.wrap(info.getStartKey());
620           region.endKey = ByteBuffer.wrap(info.getEndKey());
621           region.id = info.getRegionId();
622           region.name = ByteBuffer.wrap(info.getRegionName());
623           region.version = info.getVersion();
624           results.add(region);
625         }
626         return results;
627       } catch (TableNotFoundException e) {
628         // Return empty list for non-existing table
629         return Collections.emptyList();
630       } catch (IOException e){
631         LOG.warn(e.getMessage(), e);
632         throw new IOError(e.getMessage());
633       }
634     }

649     protected List<TCell> get(ByteBuffer tableName,
650                               ByteBuffer row,
651                               byte[] family,
652                               byte[] qualifier,
653                               Map<ByteBuffer, ByteBuffer> attributes) throws IOError {
654       try {
655         HTable table = getTable(tableName);
656         Get get = new Get(getBytes(row));
657         addAttributes(get, attributes);
658         if (qualifier == null || qualifier.length == 0) {
