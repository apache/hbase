45  /**
46   * A write/read/verify load test on a mini HBase cluster. Tests reading
47   * and then writing.
48   */
49  @Category(LargeTests.class)
50  @RunWith(Parameterized.class)
51  public class TestMiniClusterLoadSequential {
52  
53    private static final Log LOG = LogFactory.getLog(
54        TestMiniClusterLoadSequential.class);
55  
56    protected static final byte[] TABLE = Bytes.toBytes("load_test_tbl");
57    protected static final byte[] CF = Bytes.toBytes("load_test_cf");
58    protected static final int NUM_THREADS = 8;
59    protected static final int NUM_RS = 2;
60    protected static final int TIMEOUT_MS = 120000;
61    protected static final HBaseTestingUtility TEST_UTIL =
62        new HBaseTestingUtility();
63  
64    protected final Configuration conf = TEST_UTIL.getConfiguration();
65    protected final boolean isMultiPut;
66    protected final DataBlockEncoding dataBlockEncoding;
67  
68    protected MultiThreadedWriter writerThreads;
69    protected MultiThreadedReader readerThreads;
70    protected int numKeys;
71  
72    protected Compression.Algorithm compression = Compression.Algorithm.NONE;
73  
74    public TestMiniClusterLoadSequential(boolean isMultiPut,
75        DataBlockEncoding dataBlockEncoding) {
76      this.isMultiPut = isMultiPut;
77      this.dataBlockEncoding = dataBlockEncoding;
78      conf.setInt(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 1024 * 1024);
79  
80      // We don't want any region reassignments by the load balancer during the test.
81      conf.setFloat(HConstants.LOAD_BALANCER_SLOP_KEY, 10.0f);
82    }
83  
84    @Parameters
85    public static Collection<Object[]> parameters() {
86      List<Object[]> parameters = new ArrayList<Object[]>();
87      for (boolean multiPut : new boolean[]{false, true}) {
88        for (DataBlockEncoding dataBlockEncoding : new DataBlockEncoding[] {
89            DataBlockEncoding.NONE, DataBlockEncoding.PREFIX }) {
90          parameters.add(new Object[]{multiPut, dataBlockEncoding});
91        }
92      }
93      return parameters;
94    }
95  
96    @Before
97    public void setUp() throws Exception {
98      LOG.debug("Test setup: isMultiPut=" + isMultiPut);
99      TEST_UTIL.startMiniCluster(1, NUM_RS);
100   }
101 
102   @After
103   public void tearDown() throws Exception {
104     LOG.debug("Test teardown: isMultiPut=" + isMultiPut);
105     TEST_UTIL.shutdownMiniCluster();
106   }
107 
108   @Test(timeout=TIMEOUT_MS)
109   public void loadTest() throws Exception {
110     prepareForLoadTest();
111     runLoadTestOnExistingTable();
112   }
113 
114   protected void runLoadTestOnExistingTable() throws IOException {
115     writerThreads.start(0, numKeys, NUM_THREADS);
116     writerThreads.waitForFinish();
117     assertEquals(0, writerThreads.getNumWriteFailures());
118 
119     readerThreads.start(0, numKeys, NUM_THREADS);
120     readerThreads.waitForFinish();
121     assertEquals(0, readerThreads.getNumReadFailures());
122     assertEquals(0, readerThreads.getNumReadErrors());
123     assertEquals(numKeys, readerThreads.getNumKeysVerified());
124   }
125 
126   protected void prepareForLoadTest() throws IOException {
127     LOG.info("Starting load test: dataBlockEncoding=" + dataBlockEncoding +
128         ", isMultiPut=" + isMultiPut);
129     numKeys = numKeys();
130     HBaseAdmin admin = new HBaseAdmin(conf);
131     while (admin.getClusterStatus().getServers().size() < NUM_RS) {
132       LOG.info("Sleeping until " + NUM_RS + " RSs are online");
133       Threads.sleepWithoutInterrupt(1000);
134     }
135     admin.close();
136 
137     int numRegions = HBaseTestingUtility.createPreSplitLoadTestTable(conf,
138         TABLE, CF, compression, dataBlockEncoding);
139 
140     TEST_UTIL.waitUntilAllRegionsAssigned(numRegions);
141 
142     writerThreads = new MultiThreadedWriter(conf, TABLE, CF);
143     writerThreads.setMultiPut(isMultiPut);
144     readerThreads = new MultiThreadedReader(conf, TABLE, CF, 100);
145   }
146 
147   protected int numKeys() {
148     return 10000;
149   }
150 
151   protected HColumnDescriptor getColumnDesc(HBaseAdmin admin)
152       throws TableNotFoundException, IOException {
153     return admin.getTableDescriptor(TABLE).getFamily(CF);
154   }
