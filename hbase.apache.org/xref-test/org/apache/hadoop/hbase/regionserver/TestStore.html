186   public void testDeleteExpiredStoreFiles() throws Exception {
187     int storeFileNum = 4;
188     int ttl = 4;
189     
190     Configuration conf = HBaseConfiguration.create();
191     // Enable the expired store file deletion
192     conf.setBoolean("hbase.store.delete.expired.storefile", true);
193     HColumnDescriptor hcd = new HColumnDescriptor(family);
194     hcd.setTimeToLive(ttl);
195     init(getName(), conf, hcd);
196 
197     long sleepTime = this.store.scanInfo.getTtl() / storeFileNum;
198     long timeStamp;
199     // There are 4 store files and the max time stamp difference among these
200     // store files will be (this.store.ttl / storeFileNum)
201     for (int i = 1; i <= storeFileNum; i++) {
202       LOG.info("Adding some data for the store file #" + i);
203       timeStamp = EnvironmentEdgeManager.currentTimeMillis();
204       this.store.add(new KeyValue(row, family, qf1, timeStamp, (byte[]) null));
205       this.store.add(new KeyValue(row, family, qf2, timeStamp, (byte[]) null));
206       this.store.add(new KeyValue(row, family, qf3, timeStamp, (byte[]) null));
207       flush(i);
208       Thread.sleep(sleepTime);
209     }
210 
211     // Verify the total number of store files
212     assertEquals(storeFileNum, this.store.getStorefiles().size());
213 
214     // Each compaction request will find one expired store file and delete it
215     // by the compaction.
216     for (int i = 1; i <= storeFileNum; i++) {
217       // verify the expired store file.
218       CompactionRequest cr = this.store.requestCompaction();
219       // the first is expired normally.
220       // If not the first compaction, there is another empty store file,
221       assertEquals(Math.min(i, 2), cr.getFiles().size());
222       for (int j = 0; i < cr.getFiles().size(); j++) {
223         assertTrue(cr.getFiles().get(j).getReader().getMaxTimestamp() < (System
224             .currentTimeMillis() - this.store.scanInfo.getTtl()));
225       }
226       // Verify that the expired store file is compacted to an empty store file.
227       StoreFile compactedFile = this.store.compact(cr);
228       // It is an empty store file.
229       assertEquals(0, compactedFile.getReader().getEntries());
230 
231       // Let the next store file expired.
232       Thread.sleep(sleepTime);
233     }
234   }

236   public void testLowestModificationTime() throws Exception {
237     Configuration conf = HBaseConfiguration.create();
238     FileSystem fs = FileSystem.get(conf);
239     // Initialize region
240     init(getName(), conf);
241     
242     int storeFileNum = 4;
243     for (int i = 1; i <= storeFileNum; i++) {
244       LOG.info("Adding some data for the store file #"+i);
245       this.store.add(new KeyValue(row, family, qf1, i, (byte[])null));
246       this.store.add(new KeyValue(row, family, qf2, i, (byte[])null));
247       this.store.add(new KeyValue(row, family, qf3, i, (byte[])null));
248       flush(i);
249     }
250     // after flush; check the lowest time stamp
251     long lowestTimeStampFromManager = StoreUtils.getLowestTimestamp(store.getStorefiles());
252     long lowestTimeStampFromFS = getLowestTimeStampFromFS(fs, store.getStorefiles());
253     assertEquals(lowestTimeStampFromManager,lowestTimeStampFromFS);
254 
255     // after compact; check the lowest time stamp
256     store.compact(store.requestCompaction());
257     lowestTimeStampFromManager = StoreUtils.getLowestTimestamp(store.getStorefiles());
258     lowestTimeStampFromFS = getLowestTimeStampFromFS(fs, store.getStorefiles());
259     assertEquals(lowestTimeStampFromManager, lowestTimeStampFromFS);
260   }
261   
262   private static long getLowestTimeStampFromFS(FileSystem fs, 
263       final List<StoreFile> candidates) throws IOException {
264     long minTs = Long.MAX_VALUE;
265     if (candidates.isEmpty()) {
266       return minTs; 
267     }
268     Path[] p = new Path[candidates.size()];
269     for (int i = 0; i < candidates.size(); ++i) {
270       p[i] = candidates.get(i).getPath();
271     }
272     
273     FileStatus[] stats = fs.listStatus(p);
274     if (stats == null || stats.length == 0) {
275       return minTs;
276     }
277     for (FileStatus s : stats) {
278       minTs = Math.min(minTs, s.getModificationTime());
279     }
280     return minTs;
281   }
