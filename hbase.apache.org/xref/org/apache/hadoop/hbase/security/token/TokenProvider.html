19  package org.apache.hadoop.hbase.security.token;
20  
21  import java.io.IOException;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  import org.apache.hadoop.hbase.CoprocessorEnvironment;
26  import org.apache.hadoop.hbase.coprocessor.BaseEndpointCoprocessor;
27  import org.apache.hadoop.hbase.coprocessor.RegionCoprocessorEnvironment;
28  import org.apache.hadoop.hbase.ipc.HBaseServer;
29  import org.apache.hadoop.hbase.ipc.RequestContext;
30  import org.apache.hadoop.hbase.ipc.RpcServer;
31  import org.apache.hadoop.hbase.security.AccessDeniedException;
32  import org.apache.hadoop.hbase.security.User;
33  import org.apache.hadoop.security.UserGroupInformation;
34  import org.apache.hadoop.security.UserGroupInformation.AuthenticationMethod;
35  import org.apache.hadoop.security.token.SecretManager;
36  import org.apache.hadoop.security.token.Token;
37  
38  /**
39   * Provides a service for obtaining authentication tokens via the
40   * {@link AuthenticationProtocol} coprocessor protocol.
41   */
42  public class TokenProvider extends BaseEndpointCoprocessor
43      implements AuthenticationProtocol {
44  
45    public static final long VERSION = 0L;
46    private static Log LOG = LogFactory.getLog(TokenProvider.class);
47  
48    private AuthenticationTokenSecretManager secretManager;
49  
50  
51    @Override
52    public void start(CoprocessorEnvironment env) {
53      super.start(env);
54  
55      // if running at region
56      if (env instanceof RegionCoprocessorEnvironment) {
57        RegionCoprocessorEnvironment regionEnv =
58            (RegionCoprocessorEnvironment)env;
59        RpcServer server = regionEnv.getRegionServerServices().getRpcServer();
60        SecretManager<?> mgr = ((HBaseServer)server).getSecretManager();
61        if (mgr instanceof AuthenticationTokenSecretManager) {
62          secretManager = (AuthenticationTokenSecretManager)mgr;
63        }
64      }
65    }
66  
67    @Override
68    public Token<AuthenticationTokenIdentifier> getAuthenticationToken()
69        throws IOException {
70      if (secretManager == null) {
71        throw new IOException(
72            "No secret manager configured for token authentication");
73      }
74  
75      User currentUser = RequestContext.getRequestUser();
76      UserGroupInformation ugi = null;
77      if (currentUser != null) {
78        ugi = currentUser.getUGI();
79      }
80      if (currentUser == null) {
81        throw new AccessDeniedException("No authenticated user for request!");
82      } else if (!isAllowedDelegationTokenOp(ugi)) {
83        LOG.warn("Token generation denied for user="+currentUser.getName()
84            +", authMethod="+ugi.getAuthenticationMethod());
85        throw new AccessDeniedException(
86            "Token generation only allowed for Kerberos authenticated clients");
87      }
88  
89      return secretManager.generateToken(currentUser.getName());
90    }
91  
92    /**
93     * @param ugi
94     * @return true if delegation token operation is allowed
95     */
96    private boolean isAllowedDelegationTokenOp(UserGroupInformation ugi) throws IOException {
97      AuthenticationMethod authMethod = ugi.getAuthenticationMethod();
98      if (authMethod == AuthenticationMethod.PROXY) {
99        authMethod = ugi.getRealUser().getAuthenticationMethod();
100     }
101     if (authMethod != AuthenticationMethod.KERBEROS
102         && authMethod != AuthenticationMethod.KERBEROS_SSL
103         && authMethod != AuthenticationMethod.CERTIFICATE) {
104       return false;
105     }
106     return true;
107   }
108 
109   @Override
110   public String whoami() {
111     return RequestContext.getRequestUserName();
112   }
113 
114   @Override
115   public long getProtocolVersion(String protocol, long clientVersion)
116       throws IOException {
117     if (AuthenticationProtocol.class.getName().equals(protocol)) {
118       return TokenProvider.VERSION;
119     }
120     LOG.warn("Unknown protocol requested: "+protocol);
121     return -1;
122   }
