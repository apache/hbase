19  package org.apache.hadoop.hbase;
20  
21  import org.apache.hadoop.classification.InterfaceAudience;
22  import org.apache.hadoop.classification.InterfaceStability;
23  import org.apache.hadoop.hbase.util.Addressing;
24  
25  /**
26   * Data structure to hold HRegionInfo and the address for the hosting
27   * HRegionServer.  Immutable.  Comparable, but we compare the 'location' only:
28   * i.e. the hostname and port, and *not* the regioninfo.  This means two
29   * instances are the same if they refer to the same 'location' (the same
30   * hostname and port), though they may be carrying different regions.
31   */
32  @InterfaceAudience.Public
33  @InterfaceStability.Evolving
34  public class HRegionLocation implements Comparable<HRegionLocation> {
35    private final HRegionInfo regionInfo;
36    private final String hostname;
37    private final int port;
38    // Cache of the 'toString' result.
39    private String cachedString = null;
40    // Cache of the hostname + port
41    private String cachedHostnamePort;
42  
43    /**
44     * Constructor
45     * @param regionInfo the HRegionInfo for the region
46     * @param hostname Hostname
47     * @param port port
48     */
49    public HRegionLocation(HRegionInfo regionInfo, final String hostname,
50        final int port) {
51      this.regionInfo = regionInfo;
52      this.hostname = hostname;
53      this.port = port;
54    }
55  
56    /**
57     * @see java.lang.Object#toString()
58     */
59    @Override
60    public synchronized String toString() {
61      if (this.cachedString == null) {
62        this.cachedString = "region=" + this.regionInfo.getRegionNameAsString() +
63        ", hostname=" + this.hostname + ", port=" + this.port;
64      }
65      return this.cachedString;
66    }
67  
68    /**
69     * @see java.lang.Object#equals(java.lang.Object)
70     */
71    @Override
72    public boolean equals(Object o) {
73      if (this == o) {
74        return true;
75      }
76      if (o == null) {
77        return false;
78      }
79      if (!(o instanceof HRegionLocation)) {
80        return false;
81      }
82      return this.compareTo((HRegionLocation)o) == 0;
83    }
84  
85    /**
86     * @see java.lang.Object#hashCode()
87     */
88    @Override
89    public int hashCode() {
90      int result = this.hostname.hashCode();
91      result ^= this.port;
92      return result;
93    }
94  
95    /** @return HRegionInfo */
96    public HRegionInfo getRegionInfo(){
97      return regionInfo;
98    }
99  
100   public String getHostname() {
101     return this.hostname;
102   }
103 
104   public int getPort() {
105     return this.port;
106   }
107 
108   /**
109    * @return String made of hostname and port formatted as per {@link Addressing#createHostAndPortStr(String, int)}
110    */
111   public synchronized String getHostnamePort() {
112     if (this.cachedHostnamePort == null) {
113       this.cachedHostnamePort =
114         Addressing.createHostAndPortStr(this.hostname, this.port);
115     }
116     return this.cachedHostnamePort;
117   }
118 
119   //
120   // Comparable
121   //
122 
123   public int compareTo(HRegionLocation o) {
124     int result = this.hostname.compareTo(o.getHostname());
125     if (result != 0) return result;
126     return this.port - o.getPort();
127   }
128 }
