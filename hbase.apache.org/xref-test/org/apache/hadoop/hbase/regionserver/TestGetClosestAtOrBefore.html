64    public void testUsingMetaAndBinary() throws IOException {
65      FileSystem filesystem = FileSystem.get(conf);
66      Path rootdir = testDir;
67      // Up flush size else we bind up when we use default catalog flush of 16k.
68      HTableDescriptor.META_TABLEDESC.setMemStoreFlushSize(64 * 1024 * 1024);
69  
70      HRegion mr = HRegion.createHRegion(HRegionInfo.FIRST_META_REGIONINFO,
71        rootdir, this.conf, HTableDescriptor.META_TABLEDESC);
72      try {
73      // Write rows for three tables 'A', 'B', and 'C'.
74      for (char c = 'A'; c < 'D'; c++) {
75        HTableDescriptor htd = new HTableDescriptor("" + c);
76        final int last = 128;
77        final int interval = 2;
78        for (int i = 0; i <= last; i += interval) {
79          HRegionInfo hri = new HRegionInfo(htd.getName(),
80            i == 0? HConstants.EMPTY_BYTE_ARRAY: Bytes.toBytes((byte)i),
81            i == last? HConstants.EMPTY_BYTE_ARRAY: Bytes.toBytes((byte)i + interval));
82  
83          Put put = MetaEditor.makePutFromRegionInfo(hri);
84          put.setWriteToWAL(false);
85          mr.put(put, false);
86        }
87      }
88      InternalScanner s = mr.getScanner(new Scan());
89      try {
90        List<KeyValue> keys = new ArrayList<KeyValue>();
91        while(s.next(keys)) {
92          LOG.info(keys);
93          keys.clear();
94        }
95      } finally {
96        s.close();
97      }
98      findRow(mr, 'C', 44, 44);
99      findRow(mr, 'C', 45, 44);
100     findRow(mr, 'C', 46, 46);
101     findRow(mr, 'C', 43, 42);
102     mr.flushcache();
103     findRow(mr, 'C', 44, 44);
104     findRow(mr, 'C', 45, 44);
105     findRow(mr, 'C', 46, 46);
106     findRow(mr, 'C', 43, 42);
107     // Now delete 'C' and make sure I don't get entries from 'B'.
108     byte [] firstRowInC = HRegionInfo.createRegionName(Bytes.toBytes("" + 'C'),
109       HConstants.EMPTY_BYTE_ARRAY, HConstants.ZEROES, false);
110     Scan scan = new Scan(firstRowInC);
111     s = mr.getScanner(scan);
112     try {
113       List<KeyValue> keys = new ArrayList<KeyValue>();
114       while (s.next(keys)) {
115         mr.delete(new Delete(keys.get(0).getRow()), null, false);
116         keys.clear();
117       }
118     } finally {
119       s.close();
120     }
121     // Assert we get null back (pass -1).
122     findRow(mr, 'C', 44, -1);
123     findRow(mr, 'C', 45, -1);
124     findRow(mr, 'C', 46, -1);
125     findRow(mr, 'C', 43, -1);
126     mr.flushcache();
127     findRow(mr, 'C', 44, -1);
128     findRow(mr, 'C', 45, -1);
129     findRow(mr, 'C', 46, -1);
130     findRow(mr, 'C', 43, -1);
131     } finally {
132       if (mr != null) {
133         try {
134           mr.close();
135         } catch (Exception e) {
136           e.printStackTrace();
137         }
138         mr.getLog().closeAndDelete();
139       }
140     }
141   }

143   /*
144    * @param mr
145    * @param table
146    * @param rowToFind
147    * @param answer Pass -1 if we're not to find anything.
148    * @return Row found.
149    * @throws IOException
150    */
151   private byte [] findRow(final HRegion mr, final char table,
152     final int rowToFind, final int answer)
153   throws IOException {
154     byte [] tableb = Bytes.toBytes("" + table);
155     // Find the row.
156     byte [] tofindBytes = Bytes.toBytes((short)rowToFind);
157     byte [] metaKey = HRegionInfo.createRegionName(tableb, tofindBytes,
158       HConstants.NINES, false);
159     LOG.info("find=" + new String(metaKey));
160     Result r = mr.getClosestRowBefore(metaKey);
161     if (answer == -1) {
162       assertNull(r);
163       return null;
164     }
165     assertTrue(Bytes.compareTo(Bytes.toBytes((short)answer),
166       extractRowFromMetaRow(r.getRow())) == 0);
167     return r.getRow();
168   }
169 
170   private byte [] extractRowFromMetaRow(final byte [] b) {
171     int firstDelimiter = KeyValue.getDelimiter(b, 0, b.length,
172       HConstants.DELIMITER);
173     int lastDelimiter = KeyValue.getDelimiterInReverse(b, 0, b.length,
174       HConstants.DELIMITER);
175     int length = lastDelimiter - firstDelimiter - 1;
176     byte [] row = new byte[length];
177     System.arraycopy(b, firstDelimiter + 1, row, 0, length);
178     return row;
179   }

182    * Test file of multiple deletes and with deletes as final key.
183    * @see <a href="https://issues.apache.org/jira/browse/HBASE-751">HBASE-751</a>
184    */
185   public void testGetClosestRowBefore3() throws IOException{
186     HRegion region = null;
187     byte [] c0 = COLUMNS[0];
188     byte [] c1 = COLUMNS[1];
189     try {
190       HTableDescriptor htd = createTableDescriptor(getName());
191       region = createNewHRegion(htd, null, null);
192 
193       Put p = new Put(T00);
194       p.add(c0, c0, T00);
195       region.put(p);
196 
197       p = new Put(T10);
198       p.add(c0, c0, T10);
199       region.put(p);
200 
201       p = new Put(T20);
202       p.add(c0, c0, T20);
203       region.put(p);
204 
205       Result r = region.getClosestRowBefore(T20, c0);
206       assertTrue(Bytes.equals(T20, r.getRow()));
207 
208       Delete d = new Delete(T20);
209       d.deleteColumn(c0, c0);
210       region.delete(d, null, false);
211 
212       r = region.getClosestRowBefore(T20, c0);
213       assertTrue(Bytes.equals(T10, r.getRow()));
214 
215       p = new Put(T30);
216       p.add(c0, c0, T30);
217       region.put(p);
218 
219       r = region.getClosestRowBefore(T30, c0);
220       assertTrue(Bytes.equals(T30, r.getRow()));
221 
222       d = new Delete(T30);
223       d.deleteColumn(c0, c0);
224       region.delete(d, null, false);
225 
226       r = region.getClosestRowBefore(T30, c0);
227       assertTrue(Bytes.equals(T10, r.getRow()));
228       r = region.getClosestRowBefore(T31, c0);
229       assertTrue(Bytes.equals(T10, r.getRow()));
230 
231       region.flushcache();
232 
233       // try finding "010" after flush
234       r = region.getClosestRowBefore(T30, c0);
235       assertTrue(Bytes.equals(T10, r.getRow()));
236       r = region.getClosestRowBefore(T31, c0);
237       assertTrue(Bytes.equals(T10, r.getRow()));
238 
239       // Put into a different column family.  Should make it so I still get t10
240       p = new Put(T20);
241       p.add(c1, c1, T20);
242       region.put(p);
243 
244       r = region.getClosestRowBefore(T30, c0);
245       assertTrue(Bytes.equals(T10, r.getRow()));
246       r = region.getClosestRowBefore(T31, c0);
247       assertTrue(Bytes.equals(T10, r.getRow()));
248 
249       region.flushcache();
250 
251       r = region.getClosestRowBefore(T30, c0);
252       assertTrue(Bytes.equals(T10, r.getRow()));
253       r = region.getClosestRowBefore(T31, c0);
254       assertTrue(Bytes.equals(T10, r.getRow()));
255 
256       // Now try combo of memcache and mapfiles.  Delete the t20 COLUMS[1]
257       // in memory; make sure we get back t10 again.
258       d = new Delete(T20);
259       d.deleteColumn(c1, c1);
260       region.delete(d, null, false);
261       r = region.getClosestRowBefore(T30, c0);
262       assertTrue(Bytes.equals(T10, r.getRow()));
263 
264       // Ask for a value off the end of the file.  Should return t10.
265       r = region.getClosestRowBefore(T31, c0);
266       assertTrue(Bytes.equals(T10, r.getRow()));
267       region.flushcache();
268       r = region.getClosestRowBefore(T31, c0);
269       assertTrue(Bytes.equals(T10, r.getRow()));
270 
271       // Ok.  Let the candidate come out of hfile but have delete of
272       // the candidate be in memory.
273       p = new Put(T11);
274       p.add(c0, c0, T11);
275       region.put(p);
276       d = new Delete(T10);
277       d.deleteColumn(c1, c1);
278       r = region.getClosestRowBefore(T12, c0);
279       assertTrue(Bytes.equals(T11, r.getRow()));
280     } finally {
281       if (region != null) {
282         try {
283           region.close();
284         } catch (Exception e) {
285           e.printStackTrace();
286         }
287         region.getLog().closeAndDelete();
288       }
289     }
290   }
