97    /**
98     * The reader will always hold the root level index in the memory. Index
99     * blocks at all other levels will be cached in the LRU cache in practice,
100    * although this API does not enforce that.
101    *
102    * All non-root (leaf and intermediate) index blocks contain what we call a
103    * "secondary index": an array of offsets to the entries within the block.
104    * This allows us to do binary search for the entry corresponding to the
105    * given key without having to deserialize the block.
106    */

164     /**
165      * Return the data block which contains this key. This function will only
166      * be called when the HFile version is larger than 1.
167      *
168      * @param key the key we are looking for
169      * @param keyOffset the offset of the key in its byte array
170      * @param keyLength the length of the key
171      * @param currentBlock the current block, to avoid re-reading the same
172      *          block
173      * @return reader a basic way to load blocks
174      * @throws IOException
175      */
176     public HFileBlock seekToDataBlock(final byte[] key, int keyOffset,
177         int keyLength, HFileBlock currentBlock, boolean cacheBlocks,
178         boolean pread, boolean isCompaction)
179         throws IOException {
180       BlockWithScanInfo blockWithScanInfo = loadDataBlockWithScanInfo(key, keyOffset, keyLength,
181           currentBlock, cacheBlocks, pread, isCompaction);
182       if (blockWithScanInfo == null) {
183         return null;
184       } else {
185         return blockWithScanInfo.getHFileBlock();
186       }
187     }

189     /**
190      * Return the BlockWithScanInfo which contains the DataBlock with other scan info
191      * such as nextIndexedKey.
192      * This function will only be called when the HFile version is larger than 1.
193      *
194      * @param key the key we are looking for
195      * @param keyOffset the offset of the key in its byte array
196      * @param keyLength the length of the key
197      * @param currentBlock the current block, to avoid re-reading the same
198      *          block
199      * @param cacheBlocks
200      * @param pread
201      * @param isCompaction
202      * @return the BlockWithScanInfo which contains the DataBlock with other scan info
203      *         such as nextIndexedKey.
204      * @throws IOException
205      */
206     public BlockWithScanInfo loadDataBlockWithScanInfo(final byte[] key, int keyOffset,
207         int keyLength, HFileBlock currentBlock, boolean cacheBlocks,
208         boolean pread, boolean isCompaction)
209         throws IOException {
210       int rootLevelIndex = rootBlockContainingKey(key, keyOffset, keyLength);
211       if (rootLevelIndex < 0 || rootLevelIndex >= blockOffsets.length) {
212         return null;
213       }
214 
215       // the next indexed key
216       byte[] nextIndexedKey = null;
217 
218       // Read the next-level (intermediate or leaf) index block.
219       long currentOffset = blockOffsets[rootLevelIndex];
220       int currentOnDiskSize = blockDataSizes[rootLevelIndex];
221 
222       if (rootLevelIndex < blockKeys.length - 1) {
223         nextIndexedKey = blockKeys[rootLevelIndex + 1];
224       } else {
225         nextIndexedKey = HConstants.NO_NEXT_INDEXED_KEY;
226       }
227 
228       int lookupLevel = 1; // How many levels deep we are in our lookup.
229       int index = -1;
230 
231       HFileBlock block;
232       while (true) {
233 
234         if (currentBlock != null && currentBlock.getOffset() == currentOffset)
235         {
236           // Avoid reading the same block again, even with caching turned off.
237           // This is crucial for compaction-type workload which might have
238           // caching turned off. This is like a one-block cache inside the
239           // scanner.
240           block = currentBlock;
241         } else {
242           // Call HFile's caching block reader API. We always cache index
243           // blocks, otherwise we might get terrible performance.
244           boolean shouldCache = cacheBlocks || (lookupLevel < searchTreeLevel);
245           BlockType expectedBlockType;
246           if (lookupLevel < searchTreeLevel - 1) {
247             expectedBlockType = BlockType.INTERMEDIATE_INDEX;
248           } else if (lookupLevel == searchTreeLevel - 1) {
249             expectedBlockType = BlockType.LEAF_INDEX;
250           } else {
251             // this also accounts for ENCODED_DATA
252             expectedBlockType = BlockType.DATA;
253           }
254           block = cachingBlockReader.readBlock(currentOffset,
255               currentOnDiskSize, shouldCache, pread, isCompaction,
256               expectedBlockType);
257         }
258 
259         if (block == null) {
260           throw new IOException("Failed to read block at offset " +
261               currentOffset + ", onDiskSize=" + currentOnDiskSize);
262         }
263 
264         // Found a data block, break the loop and check our level in the tree.
265         if (block.getBlockType().equals(BlockType.DATA) ||
266             block.getBlockType().equals(BlockType.ENCODED_DATA)) {
267           break;
268         }
269 
270         // Not a data block. This must be a leaf-level or intermediate-level
271         // index block. We don't allow going deeper than searchTreeLevel.
272         if (++lookupLevel > searchTreeLevel) {
273           throw new IOException("Search Tree Level overflow: lookupLevel="+
274               lookupLevel + ", searchTreeLevel=" + searchTreeLevel);
275         }
276 
277         // Locate the entry corresponding to the given key in the non-root
278         // (leaf or intermediate-level) index block.
279         ByteBuffer buffer = block.getBufferWithoutHeader();
280         index = locateNonRootIndexEntry(buffer, key, keyOffset, keyLength, comparator);
281         if (index == -1) {
282           throw new IOException("The key "
283               + Bytes.toStringBinary(key, keyOffset, keyLength)
284               + " is before the" + " first key of the non-root index block "
285               + block);
286         }
287 
288         currentOffset = buffer.getLong();
289         currentOnDiskSize = buffer.getInt();
290 
291         // Only update next indexed key if there is a next indexed key in the current level
292         byte[] tmpNextIndexedKey = getNonRootIndexedKey(buffer, index + 1);
293         if (tmpNextIndexedKey != null) {
294           nextIndexedKey = tmpNextIndexedKey;
295         }
296       }
297 
298       if (lookupLevel != searchTreeLevel) {
299         throw new IOException("Reached a data block at level " + lookupLevel +
300             " but the number of levels is " + searchTreeLevel);
301       }
302 
303       // set the next indexed key for the current block.
304       BlockWithScanInfo blockWithScanInfo = new BlockWithScanInfo(block, nextIndexedKey);
305       return blockWithScanInfo;
306     }

308     /**
309      * An approximation to the {@link HFile}'s mid-key. Operates on block
310      * boundaries, and does not go inside blocks. In other words, returns the
311      * first key of the middle block of the file.
312      *
313      * @return the first key of the middle block
314      */
315     public byte[] midkey() throws IOException {
316       if (rootCount == 0)
317         throw new IOException("HFile empty");
318 
319       byte[] midKey = this.midKey.get();
320       if (midKey != null)
321         return midKey;
322 
323       if (midLeafBlockOffset >= 0) {
324         if (cachingBlockReader == null) {
325           throw new IOException("Have to read the middle leaf block but " +
326               "no block reader available");
327         }
328 
329         // Caching, using pread, assuming this is not a compaction.
330         HFileBlock midLeafBlock = cachingBlockReader.readBlock(
331             midLeafBlockOffset, midLeafBlockOnDiskSize, true, true, false,
332             BlockType.LEAF_INDEX);
333 
334         ByteBuffer b = midLeafBlock.getBufferWithoutHeader();
335         int numDataBlocks = b.getInt();
336         int keyRelOffset = b.getInt(Bytes.SIZEOF_INT * (midKeyEntry + 1));
337         int keyLen = b.getInt(Bytes.SIZEOF_INT * (midKeyEntry + 2)) -
338             keyRelOffset;
339         int keyOffset = b.arrayOffset() +
340             Bytes.SIZEOF_INT * (numDataBlocks + 2) + keyRelOffset +
341             SECONDARY_INDEX_ENTRY_OVERHEAD;
342         midKey = Arrays.copyOfRange(b.array(), keyOffset, keyOffset + keyLen);
343       } else {
344         // The middle of the root-level index.
345         midKey = blockKeys[rootCount / 2];
346       }
347 
348       this.midKey.set(midKey);
349       return midKey;
350     }

382     /**
383      * Finds the root-level index block containing the given key.
384      *
385      * @param key
386      *          Key to find
387      * @return Offset of block containing <code>key</code> (between 0 and the
388      *         number of blocks - 1) or -1 if this file does not contain the
389      *         request.
390      */
391     public int rootBlockContainingKey(final byte[] key, int offset,
392         int length) {
393       int pos = Bytes.binarySearch(blockKeys, key, offset, length,
394           comparator);
395       // pos is between -(blockKeys.length + 1) to blockKeys.length - 1, see
396       // binarySearch's javadoc.
397 
398       if (pos >= 0) {
399         // This means this is an exact match with an element of blockKeys.
400         assert pos < blockKeys.length;
401         return pos;
402       }
403 
404       // Otherwise, pos = -(i + 1), where blockKeys[i - 1] < key < blockKeys[i],
405       // and i is in [0, blockKeys.length]. We are returning j = i - 1 such that
406       // blockKeys[j] <= key < blockKeys[j + 1]. In particular, j = -1 if
407       // key < blockKeys[0], meaning the file does not contain the given key.
408 
409       int i = -pos - 1;
410       assert 0 <= i && i <= blockKeys.length;
411       return i - 1;
412     }

414     /**
415      * Adds a new entry in the root block index. Only used when reading.
416      *
417      * @param key Last key in the block
418      * @param offset file offset where the block is stored
419      * @param dataSize the uncompressed data size
420      */
421     private void add(final byte[] key, final long offset, final int dataSize) {
422       blockOffsets[rootCount] = offset;
423       blockKeys[rootCount] = key;
424       blockDataSizes[rootCount] = dataSize;
425 
426       rootCount++;
427       rootByteSize += SECONDARY_INDEX_ENTRY_OVERHEAD + key.length;
428     }

430     /**
431      * The indexed key at the ith position in the nonRootIndex. The position starts at 0.
432      * @param nonRootIndex
433      * @param i the ith position
434      * @return The indexed key at the ith position in the nonRootIndex.
435      */
436     private byte[] getNonRootIndexedKey(ByteBuffer nonRootIndex, int i) {
437       int numEntries = nonRootIndex.getInt(0);
438       if (i < 0 || i >= numEntries) {
439         return null;
440       }
441 
442       // Entries start after the number of entries and the secondary index.
443       // The secondary index takes numEntries + 1 ints.
444       int entriesOffset = Bytes.SIZEOF_INT * (numEntries + 2);
445       // Targetkey's offset relative to the end of secondary index
446       int targetKeyRelOffset = nonRootIndex.getInt(
447           Bytes.SIZEOF_INT * (i + 1));
448 
449       // The offset of the target key in the blockIndex buffer
450       int targetKeyOffset = entriesOffset     // Skip secondary index
451           + targetKeyRelOffset               // Skip all entries until mid
452           + SECONDARY_INDEX_ENTRY_OVERHEAD;  // Skip offset and on-disk-size
453 
454       // We subtract the two consecutive secondary index elements, which
455       // gives us the size of the whole (offset, onDiskSize, key) tuple. We
456       // then need to subtract the overhead of offset and onDiskSize.
457       int targetKeyLength = nonRootIndex.getInt(Bytes.SIZEOF_INT * (i + 2)) -
458         targetKeyRelOffset - SECONDARY_INDEX_ENTRY_OVERHEAD;
459 
460       int from = nonRootIndex.arrayOffset() + targetKeyOffset;
461       int to = from + targetKeyLength;
462       return Arrays.copyOfRange(nonRootIndex.array(), from, to);
463     }

465     /**
466      * Performs a binary search over a non-root level index block. Utilizes the
467      * secondary index, which records the offsets of (offset, onDiskSize,
468      * firstKey) tuples of all entries.
469      *
470      * @param key the key we are searching for offsets to individual entries in
471      *          the blockIndex buffer
472      * @param keyOffset the offset of the key in its byte array
473      * @param keyLength the length of the key
474      * @param nonRootIndex the non-root index block buffer, starting with the
475      *          secondary index. The position is ignored.
476      * @return the index i in [0, numEntries - 1] such that keys[i] <= key <
477      *         keys[i + 1], if keys is the array of all keys being searched, or
478      *         -1 otherwise
479      * @throws IOException
480      */
481     static int binarySearchNonRootIndex(byte[] key, int keyOffset,
482         int keyLength, ByteBuffer nonRootIndex,
483         RawComparator<byte[]> comparator) {
484 
485       int numEntries = nonRootIndex.getInt(0);
486       int low = 0;
487       int high = numEntries - 1;
488       int mid = 0;
489 
490       // Entries start after the number of entries and the secondary index.
491       // The secondary index takes numEntries + 1 ints.
492       int entriesOffset = Bytes.SIZEOF_INT * (numEntries + 2);
493 
494       // If we imagine that keys[-1] = -Infinity and
495       // keys[numEntries] = Infinity, then we are maintaining an invariant that
496       // keys[low - 1] < key < keys[high + 1] while narrowing down the range.
497 
498       while (low <= high) {
499         mid = (low + high) >>> 1;
500 
501         // Midkey's offset relative to the end of secondary index
502         int midKeyRelOffset = nonRootIndex.getInt(
503             Bytes.SIZEOF_INT * (mid + 1));
504 
505         // The offset of the middle key in the blockIndex buffer
506         int midKeyOffset = entriesOffset       // Skip secondary index
507             + midKeyRelOffset                  // Skip all entries until mid
508             + SECONDARY_INDEX_ENTRY_OVERHEAD;  // Skip offset and on-disk-size
509 
510         // We subtract the two consecutive secondary index elements, which
511         // gives us the size of the whole (offset, onDiskSize, key) tuple. We
512         // then need to subtract the overhead of offset and onDiskSize.
513         int midLength = nonRootIndex.getInt(Bytes.SIZEOF_INT * (mid + 2)) -
514             midKeyRelOffset - SECONDARY_INDEX_ENTRY_OVERHEAD;
515 
516         // we have to compare in this order, because the comparator order
517         // has special logic when the 'left side' is a special key.
518         int cmp = comparator.compare(key, keyOffset, keyLength,
519             nonRootIndex.array(), nonRootIndex.arrayOffset() + midKeyOffset,
520             midLength);
521 
522         // key lives above the midpoint
523         if (cmp > 0)
524           low = mid + 1; // Maintain the invariant that keys[low - 1] < key
525         // key lives below the midpoint
526         else if (cmp < 0)
527           high = mid - 1; // Maintain the invariant that key < keys[high + 1]
528         else
529           return mid; // exact match
530       }
531 
532       // As per our invariant, keys[low - 1] < key < keys[high + 1], meaning
533       // that low - 1 < high + 1 and (low - high) <= 1. As per the loop break
534       // condition, low >= high + 1. Therefore, low = high + 1.
535 
536       if (low != high + 1) {
537         throw new IllegalStateException("Binary search broken: low=" + low
538             + " " + "instead of " + (high + 1));
539       }
540 
541       // OK, our invariant says that keys[low - 1] < key < keys[low]. We need to
542       // return i such that keys[i] <= key < keys[i + 1]. Therefore i = low - 1.
543       int i = low - 1;
544 
545       // Some extra validation on the result.
546       if (i < -1 || i >= numEntries) {
547         throw new IllegalStateException("Binary search broken: result is " +
548             i + " but expected to be between -1 and (numEntries - 1) = " +
549             (numEntries - 1));
550       }
551 
552       return i;
553     }

555     /**
556      * Search for one key using the secondary index in a non-root block. In case
557      * of success, positions the provided buffer at the entry of interest, where
558      * the file offset and the on-disk-size can be read.
559      *
560      * @param nonRootBlock a non-root block without header. Initial position
561      *          does not matter.
562      * @param key the byte array containing the key
563      * @param keyOffset the offset of the key in its byte array
564      * @param keyLength the length of the key
565      * @return the index position where the given key was found,
566      *         otherwise return -1 in the case the given key is before the first key.
567      *
568      */
569     static int locateNonRootIndexEntry(ByteBuffer nonRootBlock, byte[] key,
570         int keyOffset, int keyLength, RawComparator<byte[]> comparator) {
571       int entryIndex = binarySearchNonRootIndex(key, keyOffset, keyLength,
572           nonRootBlock, comparator);
573 
574       if (entryIndex != -1) {
575         int numEntries = nonRootBlock.getInt(0);
576 
577         // The end of secondary index and the beginning of entries themselves.
578         int entriesOffset = Bytes.SIZEOF_INT * (numEntries + 2);
579 
580         // The offset of the entry we are interested in relative to the end of
581         // the secondary index.
582         int entryRelOffset = nonRootBlock.getInt(Bytes.SIZEOF_INT
583             * (1 + entryIndex));
584 
585         nonRootBlock.position(entriesOffset + entryRelOffset);
586       }
587 
588       return entryIndex;
589     }

591     /**
592      * Read in the root-level index from the given input stream. Must match
593      * what was written into the root level by
594      * {@link BlockIndexWriter#writeIndexBlocks(FSDataOutputStream)} at the
595      * offset that function returned.
596      *
597      * @param in the buffered input stream or wrapped byte input stream
598      * @param numEntries the number of root-level index entries
599      * @throws IOException
600      */
601     public void readRootIndex(DataInput in, final int numEntries)
602         throws IOException {
603       blockOffsets = new long[numEntries];
604       blockKeys = new byte[numEntries][];
605       blockDataSizes = new int[numEntries];
606 
607       // If index size is zero, no index was written.
608       if (numEntries > 0) {
609         for (int i = 0; i < numEntries; ++i) {
610           long offset = in.readLong();
611           int dataSize = in.readInt();
612           byte[] key = Bytes.readByteArray(in);
613           add(key, offset, dataSize);
614         }
615       }
616     }

644     public void readMultiLevelIndexRoot(HFileBlock blk,
645         final int numEntries) throws IOException {
646       DataInputStream in = readRootIndex(blk, numEntries);
647       // after reading the root index the checksum bytes have to
648       // be subtracted to know if the mid key exists.
649       int checkSumBytes = blk.totalChecksumBytes();
650       if ((in.available() - checkSumBytes) < MID_KEY_METADATA_SIZE) {
651         // No mid-key metadata available.
652         return;
653       }
654       midLeafBlockOffset = in.readLong();
655       midLeafBlockOnDiskSize = in.readInt();
656       midKeyEntry = in.readInt();
657     }

659     @Override
660     public String toString() {
661       StringBuilder sb = new StringBuilder();
662       sb.append("size=" + rootCount).append("\n");
663       for (int i = 0; i < rootCount; i++) {
664         sb.append("key=").append(KeyValue.keyToString(blockKeys[i]))
665             .append("\n  offset=").append(blockOffsets[i])
666             .append(", dataSize=" + blockDataSizes[i]).append("\n");
667       }
668       return sb.toString();
669     }

671     @Override
672     public long heapSize() {
673       long heapSize = ClassSize.align(6 * ClassSize.REFERENCE +
674           3 * Bytes.SIZEOF_INT + ClassSize.OBJECT);
675 
676       // Mid-key metadata.
677       heapSize += MID_KEY_METADATA_SIZE;
678 
679       // Calculating the size of blockKeys
680       if (blockKeys != null) {
681         // Adding array + references overhead
682         heapSize += ClassSize.align(ClassSize.ARRAY + blockKeys.length
683             * ClassSize.REFERENCE);
684 
685         // Adding bytes
686         for (byte[] key : blockKeys) {
687           heapSize += ClassSize.align(ClassSize.ARRAY + key.length);
688         }
689       }
690 
691       if (blockOffsets != null) {
692         heapSize += ClassSize.align(ClassSize.ARRAY + blockOffsets.length
693             * Bytes.SIZEOF_LONG);
694       }
695 
696       if (blockDataSizes != null) {
697         heapSize += ClassSize.align(ClassSize.ARRAY + blockDataSizes.length
698             * Bytes.SIZEOF_INT);
699       }
700 
701       return ClassSize.align(heapSize);
702     }

706   /**
707    * Writes the block index into the output stream. Generate the tree from
708    * bottom up. The leaf level is written to disk as a sequence of inline
709    * blocks, if it is larger than a certain number of bytes. If the leaf level
710    * is not large enough, we write all entries to the root level instead.
711    *
712    * After all leaf blocks have been written, we end up with an index
713    * referencing the resulting leaf index blocks. If that index is larger than
714    * the allowed root index size, the writer will break it up into
715    * reasonable-size intermediate-level index block chunks write those chunks
716    * out, and create another index referencing those chunks. This will be
717    * repeated until the remaining index is small enough to become the root
718    * index. However, in most practical cases we will only have leaf-level
719    * blocks and the root index, or just the root index.
720    */
721   public static class BlockIndexWriter implements InlineBlockWriter {
722     /**
723      * While the index is being written, this represents the current block
724      * index referencing all leaf blocks, with one exception. If the file is
725      * being closed and there are not enough blocks to complete even a single
726      * leaf block, no leaf blocks get written and this contains the entire
727      * block index. After all levels of the index were written by
728      * {@link #writeIndexBlocks(FSDataOutputStream)}, this contains the final
729      * root-level index.
730      */
731     private BlockIndexChunk rootChunk = new BlockIndexChunk();
732 
733     /**
734      * Current leaf-level chunk. New entries referencing data blocks get added
735      * to this chunk until it grows large enough to be written to disk.
736      */
737     private BlockIndexChunk curInlineChunk = new BlockIndexChunk();
738 
739     /**
740      * The number of block index levels. This is one if there is only root
741      * level (even empty), two if there a leaf level and root level, and is
742      * higher if there are intermediate levels. This is only final after
743      * {@link #writeIndexBlocks(FSDataOutputStream)} has been called. The
744      * initial value accounts for the root level, and will be increased to two
745      * as soon as we find out there is a leaf-level in
746      * {@link #blockWritten(long, int)}.
747      */
748     private int numLevels = 1;
749 
750     private HFileBlock.Writer blockWriter;
751     private byte[] firstKey = null;
752 
753     /**
754      * The total number of leaf-level entries, i.e. entries referenced by
755      * leaf-level blocks. For the data block index this is equal to the number
756      * of data blocks.
757      */
758     private long totalNumEntries;
759 
760     /** Total compressed size of all index blocks. */
761     private long totalBlockOnDiskSize;
762 
763     /** Total uncompressed size of all index blocks. */
764     private long totalBlockUncompressedSize;
765 
766     /** The maximum size guideline of all multi-level index blocks. */
767     private int maxChunkSize;
768 
769     /** Whether we require this block index to always be single-level. */
770     private boolean singleLevelOnly;
771 
772     /** Block cache, or null if cache-on-write is disabled */
773     private BlockCache blockCache;
774 
775     /** Name to use for computing cache keys */
776     private String nameForCaching;

778     /** Creates a single-level block index writer */
779     public BlockIndexWriter() {
780       this(null, null, null);
781       singleLevelOnly = true;
782     }

784     /**
785      * Creates a multi-level block index writer.
786      *
787      * @param blockWriter the block writer to use to write index blocks
788      * @param blockCache if this is not null, index blocks will be cached
789      *    on write into this block cache.
790      */
791     public BlockIndexWriter(HFileBlock.Writer blockWriter,
792         BlockCache blockCache, String nameForCaching) {
793       if ((blockCache == null) != (nameForCaching == null)) {
794         throw new IllegalArgumentException("Block cache and file name for " +
795             "caching must be both specified or both null");
796       }
797 
798       this.blockWriter = blockWriter;
799       this.blockCache = blockCache;
800       this.nameForCaching = nameForCaching;
801       this.maxChunkSize = HFileBlockIndex.DEFAULT_MAX_CHUNK_SIZE;
802     }

804     public void setMaxChunkSize(int maxChunkSize) {
805       if (maxChunkSize <= 0) {
806         throw new IllegalArgumentException("Invald maximum index block size");
807       }
808       this.maxChunkSize = maxChunkSize;
809     }

811     /**
812      * Writes the root level and intermediate levels of the block index into
813      * the output stream, generating the tree from bottom up. Assumes that the
814      * leaf level has been inline-written to the disk if there is enough data
815      * for more than one leaf block. We iterate by breaking the current level
816      * of the block index, starting with the index of all leaf-level blocks,
817      * into chunks small enough to be written to disk, and generate its parent
818      * level, until we end up with a level small enough to become the root
819      * level.
820      *
821      * If the leaf level is not large enough, there is no inline block index
822      * anymore, so we only write that level of block index to disk as the root
823      * level.
824      *
825      * @param out FSDataOutputStream
826      * @return position at which we entered the root-level index.
827      * @throws IOException
828      */
829     public long writeIndexBlocks(FSDataOutputStream out) throws IOException {
830       if (curInlineChunk != null && curInlineChunk.getNumEntries() != 0) {
831         throw new IOException("Trying to write a multi-level block index, " +
832             "but are " + curInlineChunk.getNumEntries() + " entries in the " +
833             "last inline chunk.");
834       }
835 
836       // We need to get mid-key metadata before we create intermediate
837       // indexes and overwrite the root chunk.
838       byte[] midKeyMetadata = numLevels > 1 ? rootChunk.getMidKeyMetadata()
839           : null;
840 
841       if (curInlineChunk != null) {
842         while (rootChunk.getRootSize() > maxChunkSize) {
843           rootChunk = writeIntermediateLevel(out, rootChunk);
844           numLevels += 1;
845         }
846       }
847 
848       // write the root level
849       long rootLevelIndexPos = out.getPos();
850 
851       {
852         DataOutput blockStream =
853             blockWriter.startWriting(BlockType.ROOT_INDEX);
854         rootChunk.writeRoot(blockStream);
855         if (midKeyMetadata != null)
856           blockStream.write(midKeyMetadata);
857         blockWriter.writeHeaderAndData(out);
858       }
859 
860       // Add root index block size
861       totalBlockOnDiskSize += blockWriter.getOnDiskSizeWithoutHeader();
862       totalBlockUncompressedSize +=
863           blockWriter.getUncompressedSizeWithoutHeader();
864 
865       if (LOG.isTraceEnabled()) {
866         LOG.trace("Wrote a " + numLevels + "-level index with root level at pos "
867           + rootLevelIndexPos + ", " + rootChunk.getNumEntries()
868           + " root-level entries, " + totalNumEntries + " total entries, "
869           + StringUtils.humanReadableInt(this.totalBlockOnDiskSize) +
870           " on-disk size, "
871           + StringUtils.humanReadableInt(totalBlockUncompressedSize) +
872           " total uncompressed size.");
873       }
874       return rootLevelIndexPos;
875     }

877     /**
878      * Writes the block index data as a single level only. Does not do any
879      * block framing.
880      *
881      * @param out the buffered output stream to write the index to. Typically a
882      *          stream writing into an {@link HFile} block.
883      * @param description a short description of the index being written. Used
884      *          in a log message.
885      * @throws IOException
886      */
887     public void writeSingleLevelIndex(DataOutput out, String description)
888         throws IOException {
889       expectNumLevels(1);
890 
891       if (!singleLevelOnly)
892         throw new IOException("Single-level mode is turned off");
893 
894       if (rootChunk.getNumEntries() > 0)
895         throw new IOException("Root-level entries already added in " +
896             "single-level mode");
897 
898       rootChunk = curInlineChunk;
899       curInlineChunk = new BlockIndexChunk();
900 
901       if (LOG.isTraceEnabled()) {
902         LOG.trace("Wrote a single-level " + description + " index with "
903           + rootChunk.getNumEntries() + " entries, " + rootChunk.getRootSize()
904           + " bytes");
905       }
906       rootChunk.writeRoot(out);
907     }

909     /**
910      * Split the current level of the block index into intermediate index
911      * blocks of permitted size and write those blocks to disk. Return the next
912      * level of the block index referencing those intermediate-level blocks.
913      *
914      * @param out
915      * @param currentLevel the current level of the block index, such as the a
916      *          chunk referencing all leaf-level index blocks
917      * @return the parent level block index, which becomes the root index after
918      *         a few (usually zero) iterations
919      * @throws IOException
920      */
921     private BlockIndexChunk writeIntermediateLevel(FSDataOutputStream out,
922         BlockIndexChunk currentLevel) throws IOException {
923       // Entries referencing intermediate-level blocks we are about to create.
924       BlockIndexChunk parent = new BlockIndexChunk();
925 
926       // The current intermediate-level block index chunk.
927       BlockIndexChunk curChunk = new BlockIndexChunk();
928 
929       for (int i = 0; i < currentLevel.getNumEntries(); ++i) {
930         curChunk.add(currentLevel.getBlockKey(i),
931             currentLevel.getBlockOffset(i), currentLevel.getOnDiskDataSize(i));
932 
933         if (curChunk.getRootSize() >= maxChunkSize)
934           writeIntermediateBlock(out, parent, curChunk);
935       }
936 
937       if (curChunk.getNumEntries() > 0) {
938         writeIntermediateBlock(out, parent, curChunk);
939       }
940 
941       return parent;
942     }
943 
944     private void writeIntermediateBlock(FSDataOutputStream out,
945         BlockIndexChunk parent, BlockIndexChunk curChunk) throws IOException {
946       long beginOffset = out.getPos();
947       DataOutputStream dos = blockWriter.startWriting(
948           BlockType.INTERMEDIATE_INDEX);
949       curChunk.writeNonRoot(dos);
950       byte[] curFirstKey = curChunk.getBlockKey(0);
951       blockWriter.writeHeaderAndData(out);
952 
953       if (blockCache != null) {
954         HFileBlock blockForCaching = blockWriter.getBlockForCaching();
955         blockCache.cacheBlock(new BlockCacheKey(nameForCaching,
956             beginOffset, DataBlockEncoding.NONE, 
957             blockForCaching.getBlockType()), blockForCaching);
958       }
959 
960       // Add intermediate index block size
961       totalBlockOnDiskSize += blockWriter.getOnDiskSizeWithoutHeader();
962       totalBlockUncompressedSize +=
963           blockWriter.getUncompressedSizeWithoutHeader();
964 
965       // OFFSET is the beginning offset the chunk of block index entries.
966       // SIZE is the total byte size of the chunk of block index entries
967       // + the secondary index size
968       // FIRST_KEY is the first key in the chunk of block index
969       // entries.
970       parent.add(curFirstKey, beginOffset,
971           blockWriter.getOnDiskSizeWithHeader());
972 
973       // clear current block index chunk
974       curChunk.clear();
975       curFirstKey = null;
976     }

999     /**
1000      * Whether there is an inline block ready to be written. In general, we
1001      * write an leaf-level index block as an inline block as soon as its size
1002      * as serialized in the non-root format reaches a certain threshold.
1003      */
1004     @Override
1005     public boolean shouldWriteBlock(boolean closing) {
1006       if (singleLevelOnly) {
1007         throw new UnsupportedOperationException(INLINE_BLOCKS_NOT_ALLOWED);
1008       }
1009 
1010       if (curInlineChunk == null) {
1011         throw new IllegalStateException("curInlineChunk is null; has shouldWriteBlock been " +
1012             "called with closing=true and then called again?");
1013       }
1014 
1015       if (curInlineChunk.getNumEntries() == 0) {
1016         return false;
1017       }
1018 
1019       // We do have some entries in the current inline chunk.
1020       if (closing) {
1021         if (rootChunk.getNumEntries() == 0) {
1022           // We did not add any leaf-level blocks yet. Instead of creating a
1023           // leaf level with one block, move these entries to the root level.
1024 
1025           expectNumLevels(1);
1026           rootChunk = curInlineChunk;
1027           curInlineChunk = null;  // Disallow adding any more index entries.
1028           return false;
1029         }
1030 
1031         return true;
1032       } else {
1033         return curInlineChunk.getNonRootSize() >= maxChunkSize;
1034       }
1035     }

1038      * Write out the current inline index block. Inline blocks are non-root
1039      * blocks, so the non-root index format is used.
1040      *
1041      * @param out
1042      */
1043     @Override
1044     public void writeInlineBlock(DataOutput out) throws IOException {
1045       if (singleLevelOnly)
1046         throw new UnsupportedOperationException(INLINE_BLOCKS_NOT_ALLOWED);
1047 
1048       // Write the inline block index to the output stream in the non-root
1049       // index block format.
1050       curInlineChunk.writeNonRoot(out);
1051 
1052       // Save the first key of the inline block so that we can add it to the
1053       // parent-level index.
1054       firstKey = curInlineChunk.getBlockKey(0);
1055 
1056       // Start a new inline index block
1057       curInlineChunk.clear();
1058     }

1060     /**
1061      * Called after an inline block has been written so that we can add an
1062      * entry referring to that block to the parent-level index.
1063      */
1064     @Override
1065     public void blockWritten(long offset, int onDiskSize, int uncompressedSize)
1066     {
1067       // Add leaf index block size
1068       totalBlockOnDiskSize += onDiskSize;
1069       totalBlockUncompressedSize += uncompressedSize;
1070 
1071       if (singleLevelOnly)
1072         throw new UnsupportedOperationException(INLINE_BLOCKS_NOT_ALLOWED);
1073 
1074       if (firstKey == null) {
1075         throw new IllegalStateException("Trying to add second-level index " +
1076             "entry with offset=" + offset + " and onDiskSize=" + onDiskSize +
1077             "but the first key was not set in writeInlineBlock");
1078       }
1079 
1080       if (rootChunk.getNumEntries() == 0) {
1081         // We are writing the first leaf block, so increase index level.
1082         expectNumLevels(1);
1083         numLevels = 2;
1084       }
1085 
1086       // Add another entry to the second-level index. Include the number of
1087       // entries in all previous leaf-level chunks for mid-key calculation.
1088       rootChunk.add(firstKey, offset, onDiskSize, totalNumEntries);
1089       firstKey = null;
1090     }

1146   /**
1147    * A single chunk of the block index in the process of writing. The data in
1148    * this chunk can become a leaf-level, intermediate-level, or root index
1149    * block.
1150    */
1151   static class BlockIndexChunk {
1152 
1153     /** First keys of the key range corresponding to each index entry. */
1154     private final List<byte[]> blockKeys = new ArrayList<byte[]>();
1155 
1156     /** Block offset in backing stream. */
1157     private final List<Long> blockOffsets = new ArrayList<Long>();
1158 
1159     /** On-disk data sizes of lower-level data or index blocks. */
1160     private final List<Integer> onDiskDataSizes = new ArrayList<Integer>();
1161 
1162     /**
1163      * The cumulative number of sub-entries, i.e. entries on deeper-level block
1164      * index entries. numSubEntriesAt[i] is the number of sub-entries in the
1165      * blocks corresponding to this chunk's entries #0 through #i inclusively.
1166      */
1167     private final List<Long> numSubEntriesAt = new ArrayList<Long>();
1168 
1169     /**
1170      * The offset of the next entry to be added, relative to the end of the
1171      * "secondary index" in the "non-root" format representation of this index
1172      * chunk. This is the next value to be added to the secondary index.
1173      */
1174     private int curTotalNonRootEntrySize = 0;
1175 
1176     /**
1177      * The accumulated size of this chunk if stored in the root index format.
1178      */
1179     private int curTotalRootSize = 0;
1180 
1181     /**
1182      * The "secondary index" used for binary search over variable-length
1183      * records in a "non-root" format block. These offsets are relative to the
1184      * end of this secondary index.
1185      */
1186     private final List<Integer> secondaryIndexOffsetMarks =
1187         new ArrayList<Integer>();
1188 
1189     /**
1190      * Adds a new entry to this block index chunk.
1191      *
1192      * @param firstKey the first key in the block pointed to by this entry
1193      * @param blockOffset the offset of the next-level block pointed to by this
1194      *          entry
1195      * @param onDiskDataSize the on-disk data of the block pointed to by this
1196      *          entry, including header size
1197      * @param curTotalNumSubEntries if this chunk is the root index chunk under
1198      *          construction, this specifies the current total number of
1199      *          sub-entries in all leaf-level chunks, including the one
1200      *          corresponding to the second-level entry being added.
1201      */
1202     void add(byte[] firstKey, long blockOffset, int onDiskDataSize,
1203         long curTotalNumSubEntries) {
1204       // Record the offset for the secondary index
1205       secondaryIndexOffsetMarks.add(curTotalNonRootEntrySize);
1206       curTotalNonRootEntrySize += SECONDARY_INDEX_ENTRY_OVERHEAD
1207           + firstKey.length;
1208 
1209       curTotalRootSize += Bytes.SIZEOF_LONG + Bytes.SIZEOF_INT
1210           + WritableUtils.getVIntSize(firstKey.length) + firstKey.length;
1211 
1212       blockKeys.add(firstKey);
1213       blockOffsets.add(blockOffset);
1214       onDiskDataSizes.add(onDiskDataSize);
1215 
1216       if (curTotalNumSubEntries != -1) {
1217         numSubEntriesAt.add(curTotalNumSubEntries);
1218 
1219         // Make sure the parallel arrays are in sync.
1220         if (numSubEntriesAt.size() != blockKeys.size()) {
1221           throw new IllegalStateException("Only have key/value count " +
1222               "stats for " + numSubEntriesAt.size() + " block index " +
1223               "entries out of " + blockKeys.size());
1224         }
1225       }
1226     }
1227 
1228     /**
1229      * The same as {@link #add(byte[], long, int, long)} but does not take the
1230      * key/value into account. Used for single-level indexes.
1231      *
1232      * @see {@link #add(byte[], long, int, long)}
1233      */
1234     public void add(byte[] firstKey, long blockOffset, int onDiskDataSize) {
1235       add(firstKey, blockOffset, onDiskDataSize, -1);
1236     }
1237 
1238     public void clear() {
1239       blockKeys.clear();
1240       blockOffsets.clear();
1241       onDiskDataSizes.clear();
1242       secondaryIndexOffsetMarks.clear();
1243       numSubEntriesAt.clear();
1244       curTotalNonRootEntrySize = 0;
1245       curTotalRootSize = 0;
1246     }
1247 
1248     /**
1249      * Finds the entry corresponding to the deeper-level index block containing
1250      * the given deeper-level entry (a "sub-entry"), assuming a global 0-based
1251      * ordering of sub-entries.
1252      *
1253      * <p>
1254      * <i> Implementation note. </i> We are looking for i such that
1255      * numSubEntriesAt[i - 1] <= k < numSubEntriesAt[i], because a deeper-level
1256      * block #i (0-based) contains sub-entries # numSubEntriesAt[i - 1]'th
1257      * through numSubEntriesAt[i] - 1, assuming a global 0-based ordering of
1258      * sub-entries. i is by definition the insertion point of k in
1259      * numSubEntriesAt.
1260      *
1261      * @param k sub-entry index, from 0 to the total number sub-entries - 1
1262      * @return the 0-based index of the entry corresponding to the given
1263      *         sub-entry
1264      */
1265     public int getEntryBySubEntry(long k) {
1266       // We define mid-key as the key corresponding to k'th sub-entry
1267       // (0-based).
1268 
1269       int i = Collections.binarySearch(numSubEntriesAt, k);
1270 
1271       // Exact match: cumulativeWeight[i] = k. This means chunks #0 through
1272       // #i contain exactly k sub-entries, and the sub-entry #k (0-based)
1273       // is in the (i + 1)'th chunk.
1274       if (i >= 0)
1275         return i + 1;
1276 
1277       // Inexact match. Return the insertion point.
1278       return -i - 1;
1279     }
1280 
1281     /**
1282      * Used when writing the root block index of a multi-level block index.
1283      * Serializes additional information allowing to efficiently identify the
1284      * mid-key.
1285      *
1286      * @return a few serialized fields for finding the mid-key
1287      * @throws IOException if could not create metadata for computing mid-key
1288      */
1289     public byte[] getMidKeyMetadata() throws IOException {
1290       ByteArrayOutputStream baos = new ByteArrayOutputStream(
1291           MID_KEY_METADATA_SIZE);
1292       DataOutputStream baosDos = new DataOutputStream(baos);
1293       long totalNumSubEntries = numSubEntriesAt.get(blockKeys.size() - 1);
1294       if (totalNumSubEntries == 0) {
1295         throw new IOException("No leaf-level entries, mid-key unavailable");
1296       }
1297       long midKeySubEntry = (totalNumSubEntries - 1) / 2;
1298       int midKeyEntry = getEntryBySubEntry(midKeySubEntry);
1299 
1300       baosDos.writeLong(blockOffsets.get(midKeyEntry));
1301       baosDos.writeInt(onDiskDataSizes.get(midKeyEntry));
1302 
1303       long numSubEntriesBefore = midKeyEntry > 0
1304           ? numSubEntriesAt.get(midKeyEntry - 1) : 0;
1305       long subEntryWithinEntry = midKeySubEntry - numSubEntriesBefore;
1306       if (subEntryWithinEntry < 0 || subEntryWithinEntry > Integer.MAX_VALUE)
1307       {
1308         throw new IOException("Could not identify mid-key index within the "
1309             + "leaf-level block containing mid-key: out of range ("
1310             + subEntryWithinEntry + ", numSubEntriesBefore="
1311             + numSubEntriesBefore + ", midKeySubEntry=" + midKeySubEntry
1312             + ")");
1313       }
1314 
1315       baosDos.writeInt((int) subEntryWithinEntry);
1316 
1317       if (baosDos.size() != MID_KEY_METADATA_SIZE) {
1318         throw new IOException("Could not write mid-key metadata: size=" +
1319             baosDos.size() + ", correct size: " + MID_KEY_METADATA_SIZE);
1320       }
1321 
1322       // Close just to be good citizens, although this has no effect.
1323       baos.close();
1324 
1325       return baos.toByteArray();
1326     }

1328     /**
1329      * Writes the block index chunk in the non-root index block format. This
1330      * format contains the number of entries, an index of integer offsets
1331      * for quick binary search on variable-length records, and tuples of
1332      * block offset, on-disk block size, and the first key for each entry.
1333      *
1334      * @param out
1335      * @throws IOException
1336      */
1337     void writeNonRoot(DataOutput out) throws IOException {
1338       // The number of entries in the block.
1339       out.writeInt(blockKeys.size());
1340 
1341       if (secondaryIndexOffsetMarks.size() != blockKeys.size()) {
1342         throw new IOException("Corrupted block index chunk writer: " +
1343             blockKeys.size() + " entries but " +
1344             secondaryIndexOffsetMarks.size() + " secondary index items");
1345       }
1346 
1347       // For each entry, write a "secondary index" of relative offsets to the
1348       // entries from the end of the secondary index. This works, because at
1349       // read time we read the number of entries and know where the secondary
1350       // index ends.
1351       for (int currentSecondaryIndex : secondaryIndexOffsetMarks)
1352         out.writeInt(currentSecondaryIndex);
1353 
1354       // We include one other element in the secondary index to calculate the
1355       // size of each entry more easily by subtracting secondary index elements.
1356       out.writeInt(curTotalNonRootEntrySize);
1357 
1358       for (int i = 0; i < blockKeys.size(); ++i) {
1359         out.writeLong(blockOffsets.get(i));
1360         out.writeInt(onDiskDataSizes.get(i));
1361         out.write(blockKeys.get(i));
1362       }
1363     }
