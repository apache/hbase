105   public ProcessBasedLocalHBaseCluster(Configuration conf,
106       int numDataNodes, int numRegionServers) {
107     this.conf = conf;
108     this.hbaseHome = HBaseHomePath.getHomePath();
109     this.numMasters = 1;
110     this.numRegionServers = numRegionServers;
111     this.workDir = hbaseHome + "/target/local_cluster";
112     this.numDataNodes = numDataNodes;
113 
114     hbaseDaemonScript = hbaseHome + "/bin/hbase-daemon.sh";
115     zkClientPort = HBaseTestingUtility.randomFreePort();
116 
117     this.rsPorts = sortedPorts(numRegionServers);
118     this.masterPorts = sortedPorts(numMasters);
119 
120     conf.set(HConstants.ZOOKEEPER_QUORUM, HConstants.LOCALHOST);
121     conf.setInt(HConstants.ZOOKEEPER_CLIENT_PORT, zkClientPort);
122   }

136   /**
137    * Generates a list of random port numbers in the sorted order. A sorted
138    * order makes sense if we ever want to refer to these servers by their index
139    * in the returned array, e.g. server #0, #1, etc.
140    */
141   private static List<Integer> sortedPorts(int n) {
142     List<Integer> ports = new ArrayList<Integer>(n);
143     for (int i = 0; i < n; ++i) {
144       ports.add(HBaseTestingUtility.randomFreePort());
145     }
146     Collections.sort(ports);
147     return ports;
148   }

150   public void startHBase() throws IOException {
151     startDaemonLogTailer();
152     cleanupOldState();
153 
154     // start ZK
155     LOG.info("Starting ZooKeeper on port " + zkClientPort);
156     startZK();
157 
158     HBaseTestingUtility.waitForHostPort(HConstants.LOCALHOST, zkClientPort);
159 
160     for (int masterPort : masterPorts) {
161       startMaster(masterPort);
162     }
163 
164     ZKUtil.waitForBaseZNode(conf);
165 
166     for (int rsPort : rsPorts) {
167       startRegionServer(rsPort);
168     }
169 
170     LOG.info("Waiting for HBase startup by scanning META");
171     int attemptsLeft = 10;
172     while (attemptsLeft-- > 0) {
173       try {
174         new HTable(conf, HConstants.META_TABLE_NAME);
175       } catch (Exception e) {
176         LOG.info("Waiting for HBase to startup. Retries left: " + attemptsLeft,
177             e);
178         Threads.sleep(1000);
179       }
180     }
181 
182     LOG.info("Process-based HBase Cluster with " + numRegionServers +
183         " region servers up and running... \n\n");
184   }

210   private void executeCommand(String command, Map<String,
211       String> envOverrides) {
212     ensureShutdownHookInstalled();
213     LOG.debug("Command : " + command);
214 
215     try {
216       String [] envp = null;
217       if (envOverrides != null) {
218         Map<String, String> map = new HashMap<String, String>(
219             System.getenv());
220         map.putAll(envOverrides);
221         envp = new String[map.size()];
222         int idx = 0;
223         for (Map.Entry<String, String> e: map.entrySet()) {
224           envp[idx++] = e.getKey() + "=" + e.getValue();
225         }
226       }
227 
228       Process p = Runtime.getRuntime().exec(command, envp);
229 
230       BufferedReader stdInput = new BufferedReader(
231           new InputStreamReader(p.getInputStream()));
232       BufferedReader stdError = new BufferedReader(
233           new InputStreamReader(p.getErrorStream()));
234 
235       // read the output from the command
236       String s = null;
237       while ((s = stdInput.readLine()) != null) {
238         System.out.println(s);
239       }
240 
241       // read any errors from the attempted command
242       while ((s = stdError.readLine()) != null) {
243         System.out.println(s);
244       }
245     } catch (IOException e) {
246       LOG.error("Error running: " + command, e);
247     }
248   }

250   private void shutdownAllProcesses() {
251     LOG.info("Killing daemons using pid files");
252     final List<String> pidFiles = new ArrayList<String>(daemonPidFiles);
253     for (String pidFile : pidFiles) {
254       int pid = 0;
255       try {
256         pid = readPidFromFile(pidFile);
257       } catch (IOException ex) {
258         LOG.error("Could not read pid from file " + pidFile);
259       }
260 
261       if (pid > 0) {
262         LOG.info("Killing pid " + pid + " (" + pidFile + ")");
263         killProcess(pid);
264       }
265     }
266   }
267 
268   private void ensureShutdownHookInstalled() {
269     if (shutdownHookInstalled) {
270       return;
271     }
272 
273     Runtime.getRuntime().addShutdownHook(new Thread(new Runnable() {
274       @Override
275       public void run() {
276         shutdownAllProcesses();
277       }
278     }));
279 
280     shutdownHookInstalled = true;
281   }

306   private static int readPidFromFile(String pidFile) throws IOException {
307     Scanner scanner = new Scanner(new File(pidFile));
308     try {
309       return scanner.nextInt();
310     } finally {
311       scanner.close();
312     }
313   }

315   private String pidFilePath(ServerType serverType, int port) {
316     String dir = serverWorkingDir(serverType, port);
317     String user = System.getenv("USER");
318     String pidFile = String.format("%s/hbase-%s-%s.pid",
319                                    dir, user, serverType.fullName);
320     return pidFile;
321   }

323   private void killServer(ServerType serverType, int port) throws IOException {
324     int pid = getServerPID(serverType, port);
325     if (pid > 0) {

327       killProcess(pid);

336   private void startServer(ServerType serverType, int rsPort) {
337     // create working directory for this region server.
338     String dir = serverWorkingDir(serverType, rsPort);
339     String confStr = generateConfig(serverType, rsPort, dir);
340     LOG.debug("Creating directory " + dir);
341     new File(dir).mkdirs();
342 
343     writeStringToFile(confStr, dir + "/hbase-site.xml");
344 
345     // Set debug options to an empty string so that hbase-config.sh does not configure them
346     // using default ports. If we want to run remote debugging on process-based local cluster's
347     // daemons, we can automatically choose non-conflicting JDWP and JMX ports for each daemon
348     // and specify them here.
349     writeStringToFile(
350         "unset HBASE_MASTER_OPTS\n" +
351         "unset HBASE_REGIONSERVER_OPTS\n" +
352         "unset HBASE_ZOOKEEPER_OPTS\n" +
353         "HBASE_MASTER_DBG_OPTS=' '\n" +
354         "HBASE_REGIONSERVER_DBG_OPTS=' '\n" +
355         "HBASE_ZOOKEEPER_DBG_OPTS=' '\n" +
356         "HBASE_MASTER_JMX_OPTS=' '\n" +
357         "HBASE_REGIONSERVER_JMX_OPTS=' '\n" +
358         "HBASE_ZOOKEEPER_JMX_OPTS=' '\n",
359         dir + "/hbase-env.sh");
360 
361     Map<String, String> envOverrides = new HashMap<String, String>();
362     envOverrides.put("HBASE_LOG_DIR", dir);
363     envOverrides.put("HBASE_PID_DIR", dir);
364     try {
365       FileUtils.copyFile(
366           new File(hbaseHome, "conf/log4j.properties"),
367           new File(dir, "log4j.properties"));
368     } catch (IOException ex) {
369       LOG.error("Could not install log4j.properties into " + dir);
370     }
371 
372     executeCommand(hbaseDaemonScript + " --config " + dir +
373                    " start " + serverType.fullName, envOverrides);
374     daemonPidFiles.add(pidFilePath(serverType, rsPort));
375     logTailDirs.add(dir);
376   }

378   private final String generateConfig(ServerType serverType, int rpcPort,
379       String daemonDir) {
380     StringBuilder sb = new StringBuilder();
381     Map<String, Object> confMap = new TreeMap<String, Object>();
382     confMap.put(HConstants.CLUSTER_DISTRIBUTED, true);
383 
384     if (serverType == ServerType.MASTER) {
385       confMap.put(HConstants.MASTER_PORT, rpcPort);
386 
387       int masterInfoPort = HBaseTestingUtility.randomFreePort();
388       reportWebUIPort("master", masterInfoPort);
389       confMap.put(HConstants.MASTER_INFO_PORT, masterInfoPort);
390     } else if (serverType == ServerType.RS) {
391       confMap.put(HConstants.REGIONSERVER_PORT, rpcPort);
392 
393       int rsInfoPort = HBaseTestingUtility.randomFreePort();
394       reportWebUIPort("region server", rsInfoPort);
395       confMap.put(HConstants.REGIONSERVER_INFO_PORT, rsInfoPort);
396     } else {
397       confMap.put(HConstants.ZOOKEEPER_DATA_DIR, daemonDir);
398     }
399 
400     confMap.put(HConstants.ZOOKEEPER_CLIENT_PORT, zkClientPort);
401     confMap.put(HConstants.HREGION_MAX_FILESIZE, MAX_FILE_SIZE_OVERRIDE);
402 
403     if (dfsCluster != null) {
404       String fsURL = "hdfs://" + HConstants.LOCALHOST + ":" + dfsCluster.getNameNodePort();
405       confMap.put("fs.default.name", fsURL);
406       confMap.put("fs.defaultFS", fsURL);
407       confMap.put("hbase.rootdir", fsURL + "/hbase_test");
408     }
409 
410     sb.append("<configuration>\n");
411     for (Map.Entry<String, Object> entry : confMap.entrySet()) {
412       sb.append("  <property>\n");
413       sb.append("    <name>" + entry.getKey() + "</name>\n");
414       sb.append("    <value>" + entry.getValue() + "</value>\n");
415       sb.append("  </property>\n");
416     }
417     sb.append("</configuration>\n");
418     return sb.toString();
419   }

437   private static final Pattern TO_REMOVE_FROM_LOG_LINES_RE =
438       Pattern.compile("org\\.apache\\.hadoop\\.hbase\\.");

440   private static final Pattern LOG_PATH_FORMAT_RE =
441       Pattern.compile("^.*/([A-Z]+)-(\\d+)/[^/]+$");

443   private static String processLine(String line) {
444     Matcher m = TO_REMOVE_FROM_LOG_LINES_RE.matcher(line);
445     return m.replaceAll("");

448   private final class LocalDaemonLogTailer implements Runnable {
449     private final Set<String> tailedFiles = new HashSet<String>();
450     private final List<String> dirList = new ArrayList<String>();
451     private final Object printLock = new Object();
452 
453     private final FilenameFilter LOG_FILES = new FilenameFilter() {
454       @Override
455       public boolean accept(File dir, String name) {
456         return name.endsWith(".out") || name.endsWith(".log");
457       }
458     };
459 
460     @Override
461     public void run() {
462       try {
463         runInternal();
464       } catch (IOException ex) {
465         LOG.error(ex);
466       }
467     }
468 
469     private void runInternal() throws IOException {
470       Thread.currentThread().setName(getClass().getSimpleName());
471       while (true) {
472         scanDirs();
473         try {
474           Thread.sleep(500);
475         } catch (InterruptedException e) {
476           LOG.error("Log tailer thread interrupted", e);
477           break;
478         }
479       }
480     }
481 
482     private void scanDirs() throws FileNotFoundException {
483       dirList.clear();
484       dirList.addAll(logTailDirs);
485       for (String d : dirList) {
486         for (File f : new File(d).listFiles(LOG_FILES)) {
487           String filePath = f.getAbsolutePath();
488           if (!tailedFiles.contains(filePath)) {
489             tailedFiles.add(filePath);
490             startTailingFile(filePath);
491           }
492         }
493       }
494     }
495 
496     private void startTailingFile(final String filePath) throws FileNotFoundException {
497       final PrintStream dest = filePath.endsWith(".log") ? System.err : System.out;
498       final ServerType serverType;
499       final int serverPort;
500       Matcher m = LOG_PATH_FORMAT_RE.matcher(filePath);
501       if (m.matches()) {
502         serverType = ServerType.valueOf(m.group(1));
503         serverPort = Integer.valueOf(m.group(2));
504       } else {
505         LOG.error("Unrecognized log path format: " + filePath);
506         return;
507       }
508       final String logMsgPrefix =
509           "[" + serverType + (serverPort != 0 ? ":" + serverPort : "") + "] ";
510 
511       LOG.debug("Tailing " + filePath);
512       Thread t = new Thread(new Runnable() {
513         @Override
514         public void run() {
515           try {
516             FileInputStream fis = new FileInputStream(filePath);
517             BufferedReader br = new BufferedReader(new InputStreamReader(fis));
518             String line;
519             while (true) {
520               try {
521                 Thread.sleep(200);
522               } catch (InterruptedException e) {
523                 LOG.error("Tailer for " + filePath + " interrupted");
524                 break;
525               }
526               while ((line = br.readLine()) != null) {
527                 line = logMsgPrefix + processLine(line);
528                 synchronized (printLock) {
529                   if (line.endsWith("\n")) {
530                     dest.print(line);
531                   } else {
532                     dest.println(line);
533                   }
534                   dest.flush();
535                 }
536               }
537             }
538           } catch (IOException ex) {
539             LOG.error("Failed tailing " + filePath, ex);
540           }
541         }
542       });
543       t.setDaemon(true);
544       t.setName("Tailer for " + filePath);
545       t.start();
546     }
547 
548   }

550   private void startDaemonLogTailer() {
551     logTailerThread = new Thread(new LocalDaemonLogTailer());
552     logTailerThread.setDaemon(true);
553     logTailerThread.start();
