66  /**
67   * HLog stores all the edits to the HStore.  Its the hbase write-ahead-log
68   * implementation.
69   *
70   * It performs logfile-rolling, so external callers are not aware that the
71   * underlying file is being rolled.
72   *
73   * <p>
74   * There is one HLog per RegionServer.  All edits for all Regions carried by
75   * a particular RegionServer are entered first in the HLog.
76   *
77   * <p>
78   * Each HRegion is identified by a unique long <code>int</code>. HRegions do
79   * not need to declare themselves before using the HLog; they simply include
80   * their HRegion-id in the <code>append</code> or
81   * <code>completeCacheFlush</code> calls.
82   *
83   * <p>
84   * An HLog consists of multiple on-disk files, which have a chronological order.
85   * As data is flushed to other (better) on-disk structures, the log becomes
86   * obsolete. We can destroy all the log messages for a given HRegion-id up to
87   * the most-recent CACHEFLUSH message from that HRegion.
88   *
89   * <p>
90   * It's only practical to delete entire files. Thus, we delete an entire on-disk
91   * file F when all of the messages in F have a log-sequence-id that's older
92   * (smaller) than the most-recent CACHEFLUSH message for every HRegion that has
93   * a message in F.
94   *
95   * <p>
96   * Synchronized methods can never execute in parallel. However, between the
97   * start of a cache flush and the completion point, appends are allowed but log
98   * rolling is not. To prevent log rolling taking place during this period, a
99   * separate reentrant lock is used.
100  *
101  * <p>To read an HLog, call {@link HLogFactory#createReader(org.apache.hadoop.fs.FileSystem,
102  * org.apache.hadoop.fs.Path, org.apache.hadoop.conf.Configuration)}.
103  *
104  */
105 @InterfaceAudience.Private

268   /**
269    * Create an edit log at the given <code>dir</code> location.
270    *
271    * You should never have to load an existing log. If there is a log at
272    * startup, it should have already been processed and deleted by the time the
273    * HLog object is started up.
274    *
275    * @param fs filesystem handle
276    * @param root path to where logs and oldlogs
277    * @param oldLogName path to where hlogs are archived
278    * @param conf configuration to use
279    * @param listeners Listeners on WAL events. Listeners passed here will
280    * be registered before we do anything else; e.g. the
281    * Constructor {@link #rollWriter()}.
282    * @param failIfLogDirExists If true IOException will be thrown if dir already exists.
283    * @param prefix should always be hostname and port in distributed env and
284    *        it will be URL encoded before being used.
285    *        If prefix is null, "hlog" will be used
286    * @throws IOException
287    */
288   private FSHLog(final FileSystem fs, final Path root, final String logName,
289       final String oldLogName, final Configuration conf, 
290       final List<WALActionsListener> listeners,
291       final boolean failIfLogDirExists, final String prefix)
292   throws IOException {
293     super();
294     this.fs = fs;
295     this.rootDir = root;
296     this.dir = new Path(this.rootDir, logName);
297     this.oldLogDir = new Path(this.rootDir, oldLogName);
298     this.conf = conf;
299    
300     if (listeners != null) {
301       for (WALActionsListener i: listeners) {
302         registerWALActionsListener(i);
303       }
304     }
305     
306     this.failIfLogDirExists = failIfLogDirExists;
307     
308     this.blocksize = this.conf.getLong("hbase.regionserver.hlog.blocksize",
309         getDefaultBlockSize());
310     // Roll at 95% of block size.
311     float multi = conf.getFloat("hbase.regionserver.logroll.multiplier", 0.95f);
312     this.logrollsize = (long)(this.blocksize * multi);
313     this.optionalFlushInterval =
314       conf.getLong("hbase.regionserver.optionallogflushinterval", 1 * 1000);
315     
316     this.maxLogs = conf.getInt("hbase.regionserver.maxlogs", 32);
317     this.minTolerableReplication = conf.getInt(
318         "hbase.regionserver.hlog.tolerable.lowreplication",
319         this.fs.getDefaultReplication());
320     this.lowReplicationRollLimit = conf.getInt(
321         "hbase.regionserver.hlog.lowreplication.rolllimit", 5);
322     this.enabled = conf.getBoolean("hbase.regionserver.hlog.enabled", true);
323     this.closeErrorsTolerated = conf.getInt(
324         "hbase.regionserver.logroll.errors.tolerated", 0);
325     
326     this.logSyncerThread = new LogSyncer(this.optionalFlushInterval);
327     
328     LOG.info("HLog configuration: blocksize=" +
329       StringUtils.byteDesc(this.blocksize) +
330       ", rollsize=" + StringUtils.byteDesc(this.logrollsize) +
331       ", enabled=" + this.enabled +
332       ", optionallogflushinternal=" + this.optionalFlushInterval + "ms");
333     // If prefix is null||empty then just name it hlog
334     this.prefix = prefix == null || prefix.isEmpty() ?
335         "hlog" : URLEncoder.encode(prefix, "UTF8");
336    
337     if (failIfLogDirExists && this.fs.exists(dir)) {
338       throw new IOException("Target HLog directory already exists: " + dir);
339     }
340     if (!fs.mkdirs(dir)) {
341       throw new IOException("Unable to mkdir " + dir);
342     }
343 
344     if (!fs.exists(oldLogDir)) {
345       if (!fs.mkdirs(this.oldLogDir)) {
346         throw new IOException("Unable to mkdir " + this.oldLogDir);
347       }
348     }
349     // rollWriter sets this.hdfs_out if it can.
350     rollWriter();
351     
352     // handle the reflection necessary to call getNumCurrentReplicas()
353     this.getNumCurrentReplicas = getGetNumCurrentReplicas(this.hdfs_out);
354 
355     Threads.setDaemonThreadRunning(logSyncerThread.getThread(),
356         Thread.currentThread().getName() + ".logSyncer");
357     coprocessorHost = new WALCoprocessorHost(this, conf);
358 
359     this.metrics = new MetricsWAL();
360   }

362   // use reflection to search for getDefaultBlockSize(Path f)
363   // if the method doesn't exist, fall back to using getDefaultBlockSize()
364   private long getDefaultBlockSize() throws IOException {
365     Method m = null;
366     Class<? extends FileSystem> cls = this.fs.getClass();
367     try {
368       m = cls.getMethod("getDefaultBlockSize",
369           new Class<?>[] { Path.class });
370     } catch (NoSuchMethodException e) {
371       LOG.info("FileSystem doesn't support getDefaultBlockSize");
372     } catch (SecurityException e) {
373       LOG.info("Doesn't have access to getDefaultBlockSize on "
374           + "FileSystems", e);
375       m = null; // could happen on setAccessible()
376     }
377     if (null == m) {
378       return this.fs.getDefaultBlockSize();
379     } else {
380       try {
381         Object ret = m.invoke(this.fs, this.dir);
382         return ((Long)ret).longValue();
383       } catch (Exception e) {
384         throw new IOException(e);
385       }
386     }
387   }

389   /**
390    * Find the 'getNumCurrentReplicas' on the passed <code>os</code> stream.
391    * @return Method or null.
392    */
393   private Method getGetNumCurrentReplicas(final FSDataOutputStream os) {
394     Method m = null;
395     if (os != null) {
396       Class<? extends OutputStream> wrappedStreamClass = os.getWrappedStream()
397           .getClass();
398       try {
399         m = wrappedStreamClass.getDeclaredMethod("getNumCurrentReplicas",
400             new Class<?>[] {});
401         m.setAccessible(true);
402       } catch (NoSuchMethodException e) {
403         LOG.info("FileSystem's output stream doesn't support"
404             + " getNumCurrentReplicas; --HDFS-826 not available; fsOut="
405             + wrappedStreamClass.getName());
406       } catch (SecurityException e) {
407         LOG.info("Doesn't have access to getNumCurrentReplicas on "
408             + "FileSystems's output stream --HDFS-826 not available; fsOut="
409             + wrappedStreamClass.getName(), e);
410         m = null; // could happen on setAccessible()
411       }
412     }
413     if (m != null) {
414       LOG.info("Using getNumCurrentReplicas--HDFS-826");
415     }
416     return m;
417   }

466   @Override
467   public byte [][] rollWriter(boolean force)
468       throws FailedLogCloseException, IOException {
469     // Return if nothing to flush.
470     if (!force && this.writer != null && this.numEntries.get() <= 0) {
471       return null;
472     }
473     byte [][] regionsToFlush = null;
474     this.cacheFlushLock.lock();
475     try {
476       this.logRollRunning = true;
477       if (closed) {
478         LOG.debug("HLog closed.  Skipping rolling of writer");
479         return regionsToFlush;
480       }
481       // Do all the preparation outside of the updateLock to block
482       // as less as possible the incoming writes
483       long currentFilenum = this.filenum;
484       Path oldPath = null;
485       if (currentFilenum > 0) {
486         oldPath = computeFilename(currentFilenum);
487       }
488       this.filenum = System.currentTimeMillis();
489       Path newPath = computeFilename();
490 
491       // Tell our listeners that a new log is about to be created
492       if (!this.listeners.isEmpty()) {
493         for (WALActionsListener i : this.listeners) {
494           i.preLogRoll(oldPath, newPath);
495         }
496       }
497       FSHLog.Writer nextWriter = this.createWriterInstance(fs, newPath, conf);
498       // Can we get at the dfsclient outputstream?  If an instance of
499       // SFLW, it'll have done the necessary reflection to get at the
500       // protected field name.
501       FSDataOutputStream nextHdfsOut = null;
502       if (nextWriter instanceof SequenceFileLogWriter) {
503         nextHdfsOut = ((SequenceFileLogWriter)nextWriter).getWriterFSDataOutputStream();
504       }
505 
506       synchronized (updateLock) {
507         // Clean up current writer.
508         Path oldFile = cleanupCurrentWriter(currentFilenum);
509         this.writer = nextWriter;
510         this.hdfs_out = nextHdfsOut;
511 
512         LOG.info((oldFile != null?
513             "Roll " + FSUtils.getPath(oldFile) + ", entries=" +
514             this.numEntries.get() +
515             ", filesize=" +
516             this.fs.getFileStatus(oldFile).getLen() + ". ": "") +
517           " for " + FSUtils.getPath(newPath));
518         this.numEntries.set(0);
519       }
520       // Tell our listeners that a new log was created
521       if (!this.listeners.isEmpty()) {
522         for (WALActionsListener i : this.listeners) {
523           i.postLogRoll(oldPath, newPath);
524         }
525       }
526 
527       // Can we delete any of the old log files?
528       if (this.outputfiles.size() > 0) {
529         if (this.lastSeqWritten.isEmpty()) {
530           LOG.debug("Last sequenceid written is empty. Deleting all old hlogs");
531           // If so, then no new writes have come in since all regions were
532           // flushed (and removed from the lastSeqWritten map). Means can
533           // remove all but currently open log file.
534           for (Map.Entry<Long, Path> e : this.outputfiles.entrySet()) {
535             archiveLogFile(e.getValue(), e.getKey());
536           }
537           this.outputfiles.clear();
538         } else {
539           regionsToFlush = cleanOldLogs();
540         }
541       }
542     } finally {
543       try {
544         this.logRollRunning = false;
545       } finally {
546         this.cacheFlushLock.unlock();
547       }
548     }
549     return regionsToFlush;
550   }

567   /*
568    * Clean up old commit logs.
569    * @return If lots of logs, flush the returned region so next time through
570    * we can clean logs. Returns null if nothing to flush.  Returns array of
571    * encoded region names to flush.
572    * @throws IOException
573    */
574   private byte [][] cleanOldLogs() throws IOException {
575     Long oldestOutstandingSeqNum = getOldestOutstandingSeqNum();
576     // Get the set of all log files whose last sequence number is smaller than
577     // the oldest edit's sequence number.
578     TreeSet<Long> sequenceNumbers = new TreeSet<Long>(this.outputfiles.headMap(
579         oldestOutstandingSeqNum).keySet());
580     // Now remove old log files (if any)
581     int logsToRemove = sequenceNumbers.size();
582     if (logsToRemove > 0) {
583       if (LOG.isDebugEnabled()) {
584         // Find associated region; helps debugging.
585         byte [] oldestRegion = getOldestRegion(oldestOutstandingSeqNum);
586         LOG.debug("Found " + logsToRemove + " hlogs to remove" +
587           " out of total " + this.outputfiles.size() + ";" +
588           " oldest outstanding sequenceid is " + oldestOutstandingSeqNum +
589           " from region " + Bytes.toStringBinary(oldestRegion));
590       }
591       for (Long seq : sequenceNumbers) {
592         archiveLogFile(this.outputfiles.remove(seq), seq);
593       }
594     }
595 
596     // If too many log files, figure which regions we need to flush.
597     // Array is an array of encoded region names.
598     byte [][] regions = null;
599     int logCount = this.outputfiles.size();
600     if (logCount > this.maxLogs && logCount > 0) {
601       // This is an array of encoded region names.
602       regions = HLogUtil.findMemstoresWithEditsEqualOrOlderThan(this.outputfiles.firstKey(),
603         this.lastSeqWritten);
604       if (regions != null) {
605         StringBuilder sb = new StringBuilder();
606         for (int i = 0; i < regions.length; i++) {
607           if (i > 0) sb.append(", ");
608           sb.append(Bytes.toStringBinary(regions[i]));
609         }
610         LOG.info("Too many hlogs: logs=" + logCount + ", maxlogs=" +
611            this.maxLogs + "; forcing flush of " + regions.length + " regions(s): " +
612            sb.toString());
613       }
614     }
615     return regions;
616   }

618   /*
619    * @return Logs older than this id are safe to remove.
620    */
621   private Long getOldestOutstandingSeqNum() {
622     return Collections.min(this.lastSeqWritten.values());
623   }

625   /**
626    * @param oldestOutstandingSeqNum
627    * @return (Encoded) name of oldest outstanding region.
628    */
629   private byte [] getOldestRegion(final Long oldestOutstandingSeqNum) {
630     byte [] oldestRegion = null;
631     for (Map.Entry<byte [], Long> e: this.lastSeqWritten.entrySet()) {
632       if (e.getValue().longValue() == oldestOutstandingSeqNum.longValue()) {
633         // Key is encoded region name.
634         oldestRegion = e.getKey();
635         break;
636       }
637     }
638     return oldestRegion;
639   }

641   /*
642    * Cleans up current writer closing and adding to outputfiles.
643    * Presumes we're operating inside an updateLock scope.
644    * @return Path to current writer or null if none.
645    * @throws IOException
646    */
647   Path cleanupCurrentWriter(final long currentfilenum) throws IOException {
648     Path oldFile = null;
649     if (this.writer != null) {
650       // Close the current writer, get a new one.
651       try {
652         // Wait till all current transactions are written to the hlog.
653         // No new transactions can occur because we have the updatelock.
654         if (this.unflushedEntries.get() != this.syncedTillHere) {
655           LOG.debug("cleanupCurrentWriter " +
656                    " waiting for transactions to get synced " +
657                    " total " + this.unflushedEntries.get() +
658                    " synced till here " + syncedTillHere);
659           sync();
660         }
661         this.writer.close();
662         this.writer = null;
663         closeErrorCount.set(0);
664       } catch (IOException e) {
665         LOG.error("Failed close of HLog writer", e);
666         int errors = closeErrorCount.incrementAndGet();
667         if (errors <= closeErrorsTolerated && !hasDeferredEntries()) {
668           LOG.warn("Riding over HLog close failure! error count="+errors);
669         } else {
670           if (hasDeferredEntries()) {
671             LOG.error("Aborting due to unflushed edits in HLog");
672           }
673           // Failed close of log file.  Means we're losing edits.  For now,
674           // shut ourselves down to minimize loss.  Alternative is to try and
675           // keep going.  See HBASE-930.
676           FailedLogCloseException flce =
677             new FailedLogCloseException("#" + currentfilenum);
678           flce.initCause(e);
679           throw flce;
680         }
681       }
682       if (currentfilenum >= 0) {
683         oldFile = computeFilename(currentfilenum);
684         this.outputfiles.put(Long.valueOf(this.logSeqNum.get()), oldFile);
685       }
686     }
687     return oldFile;
688   }

858    *
859    * Later, if we sort by these keys, we obtain all the relevant edits for a
860    * given key-range of the HRegion (TODO). Any edits that do not have a
861    * matching COMPLETE_CACHEFLUSH message can be discarded.
862    *
863    * <p>
864    * Logs cannot be restarted once closed, or once the HLog process dies. Each
865    * time the HLog starts, it must create a new log. This means that other
866    * systems should process the log appropriately upon each startup (and prior
867    * to initializing HLog).
868    *
869    * synchronized prevents appends during the completion of a cache flush or for
870    * the duration of a log roll.
871    *
872    * @param info
873    * @param tableName
874    * @param edits
875    * @param clusterId The originating clusterId for this edit (for replication)
876    * @param now
877    * @param doSync shall we sync?
878    * @return txid of this transaction
879    * @throws IOException
880    */
881   private long append(HRegionInfo info, byte [] tableName, WALEdit edits, UUID clusterId,
882       final long now, HTableDescriptor htd, boolean doSync)
883     throws IOException {
884       if (edits.isEmpty()) return this.unflushedEntries.get();;
885       if (this.closed) {
886         throw new IOException("Cannot append; log is closed");
887       }

890         long seqNum = obtainSeqNum();
891         // The 'lastSeqWritten' map holds the sequence number of the oldest
892         // write for each region (i.e. the first edit added to the particular
893         // memstore). . When the cache is flushed, the entry for the
894         // region being flushed is removed if the sequence number of the flush
895         // is greater than or equal to the value in lastSeqWritten.
896         // Use encoded name.  Its shorter, guaranteed unique and a subset of
897         // actual  name.
898         byte [] encodedRegionName = info.getEncodedNameAsBytes();
899         this.lastSeqWritten.putIfAbsent(encodedRegionName, seqNum);
900         HLogKey logKey = makeKey(encodedRegionName, tableName, seqNum, now, clusterId);
901         doWrite(info, logKey, edits, htd);

904         if (htd.isDeferredLogFlush()) {
905           lastDeferredTxid = txid;
906         }
907       }
908       // Sync if catalog region, and if not then check if that table supports
909       // deferred log flushing
910       if (doSync && 
911           (info.isMetaRegion() ||
912           !htd.isDeferredLogFlush())) {
913         // sync txn to file system
914         this.sync(txid);
915       }
916       return txid;
917     }

933   /**
934    * This class is responsible to hold the HLog's appended Entry list
935    * and to sync them according to a configurable interval.
936    *
937    * Deferred log flushing works first by piggy backing on this process by
938    * simply not sync'ing the appended Entry. It can also be sync'd by other
939    * non-deferred log flushed entries outside of this thread.
940    */
941   class LogSyncer extends HasThread {
942 
943     private final long optionalFlushInterval;
944 
945     private final AtomicBoolean closeLogSyncer = new AtomicBoolean(false);
946 
947     // List of pending writes to the HLog. There corresponds to transactions
948     // that have not yet returned to the client. We keep them cached here
949     // instead of writing them to HDFS piecemeal, because the HDFS write 
950     // method is pretty heavyweight as far as locking is concerned. The 
951     // goal is to increase the batchsize for writing-to-hdfs as well as
952     // sync-to-hdfs, so that we can get better system throughput.
953     private List<Entry> pendingWrites = new LinkedList<Entry>();
954 
955     LogSyncer(long optionalFlushInterval) {
956       this.optionalFlushInterval = optionalFlushInterval;
957     }
958 
959     @Override
960     public void run() {
961       try {
962         // awaiting with a timeout doesn't always
963         // throw exceptions on interrupt
964         while(!this.isInterrupted() && !closeLogSyncer.get()) {
965 
966           try {
967             if (unflushedEntries.get() <= syncedTillHere) {
968               synchronized (closeLogSyncer) {
969                 closeLogSyncer.wait(this.optionalFlushInterval);
970               }
971             }
972             // Calling sync since we waited or had unflushed entries.
973             // Entries appended but not sync'd are taken care of here AKA
974             // deferred log flush
975             sync();
976           } catch (IOException e) {
977             LOG.error("Error while syncing, requesting close of hlog ", e);
978             requestLogRoll();
979           }
980         }
981       } catch (InterruptedException e) {
982         LOG.debug(getName() + " interrupted while waiting for sync requests");
983       } finally {
984         LOG.info(getName() + " exiting");
985       }
986     }
987 
988     // appends new writes to the pendingWrites. It is better to keep it in
989     // our own queue rather than writing it to the HDFS output stream because
990     // HDFSOutputStream.writeChunk is not lightweight at all.
991     synchronized void append(Entry e) throws IOException {
992       pendingWrites.add(e);
993     }
994 
995     // Returns all currently pending writes. New writes
996     // will accumulate in a new list.
997     synchronized List<Entry> getPendingWrites() {
998       List<Entry> save = this.pendingWrites;
999       this.pendingWrites = new LinkedList<Entry>();
1000       return save;
1001     }
1002 
1003     // writes out pending entries to the HLog
1004     void hlogFlush(Writer writer, List<Entry> pending) throws IOException {
1005       if (pending == null) return;
1006 
1007       // write out all accumulated Entries to hdfs.
1008       for (Entry e : pending) {
1009         writer.append(e);
1010       }
1011     }
1012 
1013     void close() {
1014       synchronized (closeLogSyncer) {
1015         closeLogSyncer.set(true);
1016         closeLogSyncer.notifyAll();
1017       }
1018     }
1019   }

1026   // sync all transactions upto the specified txid
1027   private void syncer(long txid) throws IOException {
1028     Writer tempWriter;
1029     synchronized (this.updateLock) {
1030       if (this.closed) return;
1031       tempWriter = this.writer; // guaranteed non-null
1032     }
1033     // if the transaction that we are interested in is already 
1034     // synced, then return immediately.
1035     if (txid <= this.syncedTillHere) {
1036       return;
1037     }
1038     try {
1039       long doneUpto;
1040       long now = EnvironmentEdgeManager.currentTimeMillis();
1041       // First flush all the pending writes to HDFS. Then 
1042       // issue the sync to HDFS. If sync is successful, then update
1043       // syncedTillHere to indicate that transactions till this
1044       // number has been successfully synced.
1045       synchronized (flushLock) {
1046         if (txid <= this.syncedTillHere) {
1047           return;
1048         }
1049         doneUpto = this.unflushedEntries.get();
1050         List<Entry> pending = logSyncerThread.getPendingWrites();
1051         try {
1052           logSyncerThread.hlogFlush(tempWriter, pending);
1053         } catch(IOException io) {
1054           synchronized (this.updateLock) {
1055             // HBASE-4387, HBASE-5623, retry with updateLock held
1056             tempWriter = this.writer;
1057             logSyncerThread.hlogFlush(tempWriter, pending);
1058           }
1059         }
1060       }
1061       // another thread might have sync'ed avoid double-sync'ing
1062       if (txid <= this.syncedTillHere) {
1063         return;
1064       }
1065       try {
1066         tempWriter.sync();
1067       } catch(IOException io) {
1068         synchronized (this.updateLock) {
1069           // HBASE-4387, HBASE-5623, retry with updateLock held
1070           tempWriter = this.writer;
1071           tempWriter.sync();
1072         }
1073       }
1074       this.syncedTillHere = Math.max(this.syncedTillHere, doneUpto);
1075 
1076       this.metrics.finishSync(EnvironmentEdgeManager.currentTimeMillis() - now);
1077       if (!this.logRollRunning) {
1078         checkLowReplication();
1079         try {
1080           if (tempWriter.getLength() > this.logrollsize) {
1081             requestLogRoll();
1082           }
1083         } catch (IOException x) {
1084           LOG.debug("Log roll failed and will be retried. (This is not an error)");
1085         }
1086       }
1087     } catch (IOException e) {
1088       LOG.fatal("Could not sync. Requesting close of hlog", e);
1089       requestLogRoll();
1090       throw e;
1091     }
1092   }

1094   private void checkLowReplication() {
1095     // if the number of replicas in HDFS has fallen below the configured
1096     // value, then roll logs.
1097     try {
1098       int numCurrentReplicas = getLogReplication();
1099       if (numCurrentReplicas != 0
1100           && numCurrentReplicas < this.minTolerableReplication) {
1101         if (this.lowReplicationRollEnabled) {
1102           if (this.consecutiveLogRolls.get() < this.lowReplicationRollLimit) {
1103             LOG.warn("HDFS pipeline error detected. " + "Found "
1104                 + numCurrentReplicas + " replicas but expecting no less than "
1105                 + this.minTolerableReplication + " replicas. "
1106                 + " Requesting close of hlog.");
1107             requestLogRoll();
1108             // If rollWriter is requested, increase consecutiveLogRolls. Once it
1109             // is larger than lowReplicationRollLimit, disable the
1110             // LowReplication-Roller
1111             this.consecutiveLogRolls.getAndIncrement();
1112           } else {
1113             LOG.warn("Too many consecutive RollWriter requests, it's a sign of "
1114                 + "the total number of live datanodes is lower than the tolerable replicas.");
1115             this.consecutiveLogRolls.set(0);
1116             this.lowReplicationRollEnabled = false;
1117           }
1118         }
1119       } else if (numCurrentReplicas >= this.minTolerableReplication) {
1120 
1121         if (!this.lowReplicationRollEnabled) {
1122           // The new writer's log replicas is always the default value.
1123           // So we should not enable LowReplication-Roller. If numEntries
1124           // is lower than or equals 1, we consider it as a new writer.
1125           if (this.numEntries.get() <= 1) {
1126             return;
1127           }
1128           // Once the live datanode number and the replicas return to normal,
1129           // enable the LowReplication-Roller.
1130           this.lowReplicationRollEnabled = true;
1131           LOG.info("LowReplication-Roller was enabled.");
1132         }
1133       }
1134     } catch (Exception e) {
1135       LOG.warn("Unable to invoke DFSOutputStream.getNumCurrentReplicas" + e +
1136           " still proceeding ahead...");
1137     }
1138   }

1191   protected void doWrite(HRegionInfo info, HLogKey logKey, WALEdit logEdit,
1192                            HTableDescriptor htd)
1193   throws IOException {
1194     if (!this.enabled) {
1195       return;
1196     }
1197     if (!this.listeners.isEmpty()) {
1198       for (WALActionsListener i: this.listeners) {
1199         i.visitLogEntryBeforeWrite(htd, logKey, logEdit);
1200       }
1201     }
1202     try {
1203       long now = EnvironmentEdgeManager.currentTimeMillis();
1204       // coprocessor hook:
1205       if (!coprocessorHost.preWALWrite(info, logKey, logEdit)) {
1206         // write to our buffer for the Hlog file.
1207         logSyncerThread.append(new FSHLog.Entry(logKey, logEdit));
1208       }
1209       long took = EnvironmentEdgeManager.currentTimeMillis() - now;
1210       coprocessorHost.postWALWrite(info, logKey, logEdit);
1211       long len = 0;
1212       for (KeyValue kv : logEdit.getKeyValues()) {
1213         len += kv.getLength();
1214       }
1215       this.metrics.finishAppend(took, len);
1216     } catch (IOException e) {
1217       LOG.fatal("Could not append. Requesting close of hlog", e);
1218       requestLogRoll();
1219       throw e;
1220     }

1239   private byte[] getSnapshotName(byte[] encodedRegionName) {
1240     byte snp[] = new byte[encodedRegionName.length + 3];
1241     // an encoded region name has only hex digits. s, n or p are not hex
1242     // and therefore snapshot-names will never collide with
1243     // encoded-region-names
1244     snp[0] = 's'; snp[1] = 'n'; snp[2] = 'p';
1245     for (int i = 0; i < encodedRegionName.length; i++) {
1246       snp[i+3] = encodedRegionName[i];
1247     }
1248     return snp;
1249   }

1251   @Override
1252   public long startCacheFlush(final byte[] encodedRegionName) {
1253     this.cacheFlushLock.lock();
1254     Long seq = this.lastSeqWritten.remove(encodedRegionName);
1255     // seq is the lsn of the oldest edit associated with this region. If a
1256     // snapshot already exists - because the last flush failed - then seq will
1257     // be the lsn of the oldest edit in the snapshot
1258     if (seq != null) {
1259       // keeping the earliest sequence number of the snapshot in
1260       // lastSeqWritten maintains the correctness of
1261       // getOldestOutstandingSeqNum(). But it doesn't matter really because
1262       // everything is being done inside of cacheFlush lock.
1263       Long oldseq =
1264         lastSeqWritten.put(getSnapshotName(encodedRegionName), seq);
1265       if (oldseq != null) {
1266         LOG.error("Logic Error Snapshot seq id from earlier flush still" +
1267             " present! for region " + Bytes.toString(encodedRegionName) +
1268             " overwritten oldseq=" + oldseq + "with new seq=" + seq);
1269         Runtime.getRuntime().halt(1);
1270       }
1271     }
1272     return obtainSeqNum();
1273   }

1275   @Override
1276   public void completeCacheFlush(final byte [] encodedRegionName,
1277       final byte [] tableName, final long logSeqId, final boolean isMetaRegion)
1278   throws IOException {
1279     try {
1280       if (this.closed) {
1281         return;
1282       }
1283       long txid = 0;
1284       synchronized (updateLock) {
1285         long now = EnvironmentEdgeManager.currentTimeMillis();
1286         WALEdit edit = completeCacheFlushLogEdit();
1287         HLogKey key = makeKey(encodedRegionName, tableName, logSeqId,
1288             System.currentTimeMillis(), HConstants.DEFAULT_CLUSTER_ID);
1289         logSyncerThread.append(new Entry(key, edit));
1290         txid = this.unflushedEntries.incrementAndGet();
1291         long took = EnvironmentEdgeManager.currentTimeMillis() - now;
1292         long len = 0;
1293         for (KeyValue kv : edit.getKeyValues()) {
1294           len += kv.getLength();
1295         }
1296         this.metrics.finishAppend(took, len);
1297         this.numEntries.incrementAndGet();
1298       }
1299       // sync txn to file system
1300       this.sync(txid);
1301 
1302     } finally {
1303       // updateLock not needed for removing snapshot's entry
1304       // Cleaning up of lastSeqWritten is in the finally clause because we
1305       // don't want to confuse getOldestOutstandingSeqNum()
1306       this.lastSeqWritten.remove(getSnapshotName(encodedRegionName));
1307       this.cacheFlushLock.unlock();
1308     }
1309   }

1320   public void abortCacheFlush(byte[] encodedRegionName) {
1321     Long snapshot_seq =
1322       this.lastSeqWritten.remove(getSnapshotName(encodedRegionName));
1323     if (snapshot_seq != null) {
1324       // updateLock not necessary because we are racing against
1325       // lastSeqWritten.putIfAbsent() in append() and we will always win
1326       // before releasing cacheFlushLock make sure that the region's entry in
1327       // lastSeqWritten points to the earliest edit in the region
1328       Long current_memstore_earliest_seq =
1329         this.lastSeqWritten.put(encodedRegionName, snapshot_seq);
1330       if (current_memstore_earliest_seq != null &&
1331           (current_memstore_earliest_seq.longValue() <=
1332             snapshot_seq.longValue())) {
1333         LOG.error("Logic Error region " + Bytes.toString(encodedRegionName) +
1334             "acquired edits out of order current memstore seq=" +
1335             current_memstore_earliest_seq + " snapshot seq=" + snapshot_seq);
1336         Runtime.getRuntime().halt(1);
1337       }
1338     }
1339     this.cacheFlushLock.unlock();
