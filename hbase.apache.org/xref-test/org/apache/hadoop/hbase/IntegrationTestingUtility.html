19  package org.apache.hadoop.hbase;
20  
21  import java.io.IOException;
22  
23  import org.apache.hadoop.conf.Configuration;
24  
25  /**
26   * Facility for <strong>integration/system</strong> tests. This extends {@link HBaseTestingUtility}
27   * and adds-in the functionality needed by integration and system tests. This class understands
28   * distributed and pseudo-distributed/local cluster deployments, and abstracts those from the tests
29   * in this module.
30   * <p>
31   * IntegrationTestingUtility is constructed and used by the integration tests, but the tests
32   * themselves should not assume a particular deployment. They can rely on the methods in this
33   * class and HBaseCluster. Before the testing begins, the test should initialize the cluster by
34   * calling {@link #initializeCluster(int)}.
35   * <p>
36   * The cluster that is used defaults to a mini cluster, but it can be forced to use a distributed
37   * cluster by calling {@link #setUseDistributedCluster(Configuration)}. This method is invoked by
38   * test drivers (maven, IntegrationTestsDriver, etc) before initializing the cluster
39   * via {@link #initializeCluster(int)}. Individual tests should not directly call
40   * {@link #setUseDistributedCluster(Configuration)}.
41   */
42  public class IntegrationTestingUtility extends HBaseTestingUtility {
43  
44    public IntegrationTestingUtility() {
45      this(HBaseConfiguration.create());
46    }
47  
48    public IntegrationTestingUtility(Configuration conf) {
49      super(conf);
50    }
51  
52    /**
53     * Configuration that controls whether this utility assumes a running/deployed cluster.
54     * This is different than "hbase.cluster.distributed" since that parameter indicates whether the
55     * cluster is in an actual distributed environment, while this shows that there is a
56     * deployed (distributed or pseudo-distributed) cluster running, and we do not need to
57     * start a mini-cluster for tests.
58     */
59    public static final String IS_DISTRIBUTED_CLUSTER = "hbase.test.cluster.distributed";
60  
61    /**
62     * Initializes the state of the cluster. It starts a new in-process mini cluster, OR
63     * if we are given an already deployed distributed cluster it initializes the state.
64     * @param numSlaves Number of slaves to start up if we are booting a mini cluster. Otherwise
65     * we check whether this many nodes are available and throw an exception if not.
66     */
67    public void initializeCluster(int numSlaves) throws Exception {
68      if (isDistributedCluster()) {
69        createDistributedHBaseCluster();
70        checkNodeCount(numSlaves);
71      } else {
72        startMiniCluster(numSlaves);
73      }
74    }
75  
76    /**
77     * Checks whether we have more than numSlaves nodes. Throws an
78     * exception otherwise.
79     */
80    public void checkNodeCount(int numSlaves) throws Exception {
81      HBaseCluster cluster = getHBaseClusterInterface();
82      if (cluster.getClusterStatus().getServers().size() < numSlaves) {
83        throw new Exception("Cluster does not have enough nodes:" + numSlaves);
84      }
85    }
86  
87    /**
88     * Restores the cluster to the initial state if it is a distributed cluster, otherwise, shutdowns the
89     * mini cluster.
90     */
91    public void restoreCluster() throws IOException {
92      if (isDistributedCluster()) {
93        getHBaseClusterInterface().restoreInitialStatus();
94      } else {
95        getMiniHBaseCluster().shutdown();
96      }
97    }
98  
99    /**
100    * Sets the configuration property to use a distributed cluster for the integration tests. Test drivers
101    * should use this to enforce cluster deployment.
102    */
103   public static void setUseDistributedCluster(Configuration conf) {
104     conf.setBoolean(IS_DISTRIBUTED_CLUSTER, true);
105     System.setProperty(IS_DISTRIBUTED_CLUSTER, "true");
106   }
107 
108   /**
109    * @return whether we are interacting with a distributed cluster as opposed to and in-process mini
110    * cluster or a local cluster.
111    * @see IntegrationTestingUtility#setUseDistributedCluster(Configuration)
112    */
113   private boolean isDistributedCluster() {
114     Configuration conf = getConfiguration();
115     boolean isDistributedCluster = false;
116     isDistributedCluster = Boolean.parseBoolean(System.getProperty(IS_DISTRIBUTED_CLUSTER, "false"));
117     if (!isDistributedCluster) {
118       isDistributedCluster = conf.getBoolean(IS_DISTRIBUTED_CLUSTER, false);
119     }
120     return isDistributedCluster;
121   }
122 
123   private void createDistributedHBaseCluster() throws IOException {
124     Configuration conf = getConfiguration();
125     ClusterManager clusterManager = new HBaseClusterManager();
126     setHBaseCluster(new DistributedHBaseCluster(conf, clusterManager));
127     getHBaseAdmin();
128   }
