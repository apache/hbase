25  import org.apache.hadoop.classification.InterfaceStability;

31  /**
32   * Defines how value for specific column is interpreted and provides utility
33   * methods like compare, add, multiply etc for them. Takes column family, column
34   * qualifier and return the cell value. Its concrete implementation should
35   * handle null case gracefully. Refer to {@link LongColumnInterpreter} for an
36   * example.
37   * <p>
38   * Takes two generic parameters. The cell value type of the interpreter is <T>.
39   * During some computations like sum, average, the return type can be different
40   * than the cell value data type, for eg, sum of int cell values might overflow
41   * in case of a int result, we should use Long for its result. Therefore, this
42   * class mandates to use a different (promoted) data type for result of these
43   * computations <S>. All computations are performed on the promoted data type
44   * <S>. There is a conversion method
45   * {@link ColumnInterpreter#castToReturnType(Object)} which takes a <T> type and
46   * returns a <S> type.
47   * @param <T> Cell value data type
48   * @param <S> Promoted data type
49   */
50  @InterfaceAudience.Public
51  @InterfaceStability.Evolving
52  public interface ColumnInterpreter<T, S> {
53  
54    /**
55     * @param colFamily
56     * @param colQualifier
57     * @param kv
58     * @return value of type T
59     * @throws IOException
60     */
61    T getValue(byte[] colFamily, byte[] colQualifier, KeyValue kv)
62        throws IOException;
63  
64    /**
65     * @param l1
66     * @param l2
67     * @return sum or non null value among (if either of them is null); otherwise
68     * returns a null.
69     */
70    public S add(S l1, S l2);
71  
72    /**
73     * returns the maximum value for this type T
74     * @return max
75     */
76  
77    T getMaxValue();
78  
79    T getMinValue();
80  
81    /**
82     * @param o1
83     * @param o2
84     * @return multiplication
85     */
86    S multiply(S o1, S o2);
87  
88    /**
89     * @param o
90     * @return increment
91     */
92    S increment(S o);
93  
94    /**
95     * provides casting opportunity between the data types.
96     * @param o
97     * @return cast
98     */
99    S castToReturnType(T o);
100 
101   /**
102    * This takes care if either of arguments are null. returns 0 if they are
103    * equal or both are null;
104    * <ul>
105    * <li>>0 if l1 > l2 or l1 is not null and l2 is null.
106    * <li>< 0 if l1 < l2 or l1 is null and l2 is not null.
107    */
108   int compare(final T l1, final T l2);
109 
110   /**
111    * used for computing average of <S> data values. Not providing the divide
112    * method that takes two <S> values as it is not needed as of now.
113    * @param o
114    * @param l
115    * @return Average
116    */
117   double divideForAvg(S o, Long l);
118 
119   /**
120    * This method should return any additional data that is needed on the
121    * server side to construct the ColumnInterpreter. The server
122    * will pass this to the {@link #initialize(ByteString)}
123    * method. If there is no ColumnInterpreter specific data (for e.g.,
124    * {@link LongColumnInterpreter}) then null should be returned.
125    * @return the PB message
126    */
127   ByteString columnInterpreterSpecificData();
128 
129   /**
130    * Return the PB for type T
131    * @param t
132    * @return PB-message
133    */
134   ByteString getProtoForCellType(T t);
135 
136   /**
137    * Return the PB for type S
138    * @param s
139    * @return PB-message
140    */
141   ByteString getProtoForPromotedType(S s);
142 
143   /**
144    * This method should initialize any field(s) of the ColumnInterpreter with
145    * a parsing of the passed message bytes (used on the server side).
146    * @param bytes
147    */
148   void initialize(ByteString bytes);
149   
150   /**
151    * Converts the bytes in the server's response to the expected type S
152    * @param response
153    * @return response of type S constructed from the message
154    */
155   S parseResponseAsPromotedType(byte[] response);
156   
157   /**
158    * The response message comes as type S. This will convert/cast it to T.
159    * In some sense, performs the opposite of {@link #castToReturnType(Object)}
160    * @param response
161    * @return cast
162    */
163   T castToCellType(S response);
164 }
