51  /**
52   * A base for {@link TableInputFormat}s. Receives a {@link HTable}, an
53   * {@link Scan} instance that defines the input columns etc. Subclasses may use
54   * other TableRecordReader implementations.
55   * <p>
56   * An example of a subclass:
57   * <pre>
58   *   class ExampleTIF extends TableInputFormatBase implements JobConfigurable {
59   *
60   *     public void configure(JobConf job) {
61   *       HTable exampleTable = new HTable(HBaseConfiguration.create(job),
62   *         Bytes.toBytes("exampleTable"));
63   *       // mandatory
64   *       setHTable(exampleTable);
65   *       Text[] inputColumns = new byte [][] { Bytes.toBytes("columnA"),
66   *         Bytes.toBytes("columnB") };
67   *       // mandatory
68   *       setInputColumns(inputColumns);
69   *       RowFilterInterface exampleFilter = new RegExpRowFilter("keyPrefix.*");
70   *       // optional
71   *       setRowFilter(exampleFilter);
72   *     }
73   *
74   *     public void validateInput(JobConf job) throws IOException {
75   *     }
76   *  }
77   * </pre>
78   */
79  @InterfaceAudience.Public
80  @InterfaceStability.Stable
81  public abstract class TableInputFormatBase
82  extends InputFormat<ImmutableBytesWritable, Result> {

101   /**
102    * Builds a TableRecordReader. If no TableRecordReader was provided, uses
103    * the default.
104    *
105    * @param split  The split to work with.
106    * @param context  The current context.
107    * @return The newly created record reader.
108    * @throws IOException When creating the reader fails.
109    * @see org.apache.hadoop.mapreduce.InputFormat#createRecordReader(
110    *   org.apache.hadoop.mapreduce.InputSplit,
111    *   org.apache.hadoop.mapreduce.TaskAttemptContext)
112    */
113   @Override
114   public RecordReader<ImmutableBytesWritable, Result> createRecordReader(
115       InputSplit split, TaskAttemptContext context)
116   throws IOException {
117     if (table == null) {
118       throw new IOException("Cannot create a record reader because of a" +
119           " previous error. Please look at the previous logs lines from" +
120           " the task's full log for more details.");
121     }
122     TableSplit tSplit = (TableSplit) split;
123     TableRecordReader trr = this.tableRecordReader;
124     // if no table record reader was provided use default
125     if (trr == null) {
126       trr = new TableRecordReader();
127     }
128     Scan sc = new Scan(this.scan);
129     sc.setStartRow(tSplit.getStartRow());
130     sc.setStopRow(tSplit.getEndRow());
131     trr.setScan(sc);
132     trr.setHTable(table);
133     try {
134       trr.initialize(tSplit, context);
135     } catch (InterruptedException e) {
136       throw new InterruptedIOException(e.getMessage());
137     }
138     return trr;
139   }

141   /**
142    * Calculates the splits that will serve as input for the map tasks. The
143    * number of splits matches the number of regions in a table.
144    *
145    * @param context  The current job context.
146    * @return The list of input splits.
147    * @throws IOException When creating the list of splits fails.
148    * @see org.apache.hadoop.mapreduce.InputFormat#getSplits(
149    *   org.apache.hadoop.mapreduce.JobContext)
150    */
151   @Override
152   public List<InputSplit> getSplits(JobContext context) throws IOException {
153 	if (table == null) {
154 	    throw new IOException("No table was provided.");
155 	}
156     // Get the name server address and the default value is null.
157     this.nameServer =
158       context.getConfiguration().get("hbase.nameserver.address", null);
159     
160     Pair<byte[][], byte[][]> keys = table.getStartEndKeys();
161     if (keys == null || keys.getFirst() == null ||
162         keys.getFirst().length == 0) {
163       HRegionLocation regLoc = table.getRegionLocation(HConstants.EMPTY_BYTE_ARRAY, false);
164       if (null == regLoc) {
165         throw new IOException("Expecting at least one region.");
166       }
167       List<InputSplit> splits = new ArrayList<InputSplit>(1);
168       InputSplit split = new TableSplit(table.getTableName(),
169           HConstants.EMPTY_BYTE_ARRAY, HConstants.EMPTY_BYTE_ARRAY, regLoc
170               .getHostnamePort().split(Addressing.HOSTNAME_PORT_SEPARATOR)[0]);
171       splits.add(split);
172       return splits;
173     }
174     List<InputSplit> splits = new ArrayList<InputSplit>(keys.getFirst().length);
175     for (int i = 0; i < keys.getFirst().length; i++) {
176       if ( !includeRegionInSplit(keys.getFirst()[i], keys.getSecond()[i])) {
177         continue;
178       }
179       HRegionLocation location = table.getRegionLocation(keys.getFirst()[i], false);
180       // The below InetSocketAddress creation does a name resolution.
181       InetSocketAddress isa = new InetSocketAddress(location.getHostname(), location.getPort());
182       if (isa.isUnresolved()) {
183         LOG.warn("Failed resolve " + isa);
184       }
185       InetAddress regionAddress = isa.getAddress();
186       String regionLocation;
187       try {
188         regionLocation = reverseDNS(regionAddress);
189       } catch (NamingException e) {
190         LOG.error("Cannot resolve the host name for " + regionAddress + " because of " + e);
191         regionLocation = location.getHostname();
192       }
193 
194       byte[] startRow = scan.getStartRow();
195       byte[] stopRow = scan.getStopRow();
196       // determine if the given start an stop key fall into the region
197       if ((startRow.length == 0 || keys.getSecond()[i].length == 0 ||
198           Bytes.compareTo(startRow, keys.getSecond()[i]) < 0) &&
199           (stopRow.length == 0 ||
200            Bytes.compareTo(stopRow, keys.getFirst()[i]) > 0)) {
201         byte[] splitStart = startRow.length == 0 ||
202           Bytes.compareTo(keys.getFirst()[i], startRow) >= 0 ?
203             keys.getFirst()[i] : startRow;
204         byte[] splitStop = (stopRow.length == 0 ||
205           Bytes.compareTo(keys.getSecond()[i], stopRow) <= 0) &&
206           keys.getSecond()[i].length > 0 ?
207             keys.getSecond()[i] : stopRow;
208         InputSplit split = new TableSplit(table.getTableName(),
209           splitStart, splitStop, regionLocation);
210         splits.add(split);
211         if (LOG.isDebugEnabled()) {
212           LOG.debug("getSplits: split -> " + i + " -> " + split);
213         }
214       }
215     }
216     return splits;
217   }
218   
219   private String reverseDNS(InetAddress ipAddress) throws NamingException {
220     String hostName = this.reverseDNSCacheMap.get(ipAddress);
221     if (hostName == null) {
222       hostName = DNS.reverseDns(ipAddress, this.nameServer);
223       this.reverseDNSCacheMap.put(ipAddress, hostName);
224     }
225     return hostName;
226   }

228   /**
229    *
230    *
231    * Test if the given region is to be included in the InputSplit while splitting
232    * the regions of a table.
233    * <p>
234    * This optimization is effective when there is a specific reasoning to exclude an entire region from the M-R job,
235    * (and hence, not contributing to the InputSplit), given the start and end keys of the same. <br>
236    * Useful when we need to remember the last-processed top record and revisit the [last, current) interval for M-R processing,
237    * continuously. In addition to reducing InputSplits, reduces the load on the region server as well, due to the ordering of the keys.
238    * <br>
239    * <br>
240    * Note: It is possible that <code>endKey.length() == 0 </code> , for the last (recent) region.
241    * <br>
242    * Override this method, if you want to bulk exclude regions altogether from M-R. By default, no region is excluded( i.e. all regions are included).
243    *
244    *
245    * @param startKey Start key of the region
246    * @param endKey End key of the region
247    * @return true, if this region needs to be included as part of the input (default).
248    *
249    */
250   protected boolean includeRegionInSplit(final byte[] startKey, final byte [] endKey) {
251     return true;
252   }
