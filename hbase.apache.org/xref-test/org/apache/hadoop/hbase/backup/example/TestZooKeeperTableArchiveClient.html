153     createArchiveDirectory();

157     toCleanup.add(archiveDir);

161     // setup the delegate
162     Stoppable stop = new StoppableImplementation();
163     HFileCleaner cleaner = setupAndCreateCleaner(conf, fs, archiveDir, stop);
164     List<BaseHFileCleanerDelegate> cleaners = turnOnArchiving(STRING_TABLE_NAME, cleaner);
165     final LongTermArchivingHFileCleaner delegate = (LongTermArchivingHFileCleaner) cleaners.get(0);
166 
167     // create the region
168     HColumnDescriptor hcd = new HColumnDescriptor(TEST_FAM);
169     HRegion region = UTIL.createTestRegion(STRING_TABLE_NAME, hcd);
170 
171     loadFlushAndCompact(region, TEST_FAM);
172 
173     // get the current hfiles in the archive directory
174     List<Path> files = getAllFiles(fs, archiveDir);
175     if (files == null) {

177       throw new RuntimeException("Didn't archive any files!");
178     }
179     CountDownLatch finished = setupCleanerWatching(delegate, cleaners, files.size());
180 
181     runCleaner(cleaner, finished, stop);
182 
183     // know the cleaner ran, so now check all the files again to make sure they are still there
184     List<Path> archivedFiles = getAllFiles(fs, archiveDir);
185     assertEquals("Archived files changed after running archive cleaner.", files, archivedFiles);

187     // but we still have the archive directory

191   /**
192    * Test archiving/cleaning across multiple tables, where some are retained, and others aren't
193    * @throws Exception on failure
194    */
195   @Test
196   public void testMultipleTables() throws Exception {
197     createArchiveDirectory();
198     String otherTable = "otherTable";
199 
200     FileSystem fs = UTIL.getTestFileSystem();
201     Path archiveDir = getArchiveDir();
202     Path tableDir = getTableDir(STRING_TABLE_NAME);
203     Path otherTableDir = getTableDir(otherTable);
204 
205     // register cleanup for the created directories
206     toCleanup.add(archiveDir);
207     toCleanup.add(tableDir);
208     toCleanup.add(otherTableDir);
209     Configuration conf = UTIL.getConfiguration();
210     // setup the delegate
211     Stoppable stop = new StoppableImplementation();
212     HFileCleaner cleaner = setupAndCreateCleaner(conf, fs, archiveDir, stop);
213     List<BaseHFileCleanerDelegate> cleaners = turnOnArchiving(STRING_TABLE_NAME, cleaner);
214     final LongTermArchivingHFileCleaner delegate = (LongTermArchivingHFileCleaner) cleaners.get(0);
215 
216     // create the region
217     HColumnDescriptor hcd = new HColumnDescriptor(TEST_FAM);
218     HRegion region = UTIL.createTestRegion(STRING_TABLE_NAME, hcd);
219     loadFlushAndCompact(region, TEST_FAM);
220 
221     // create the another table that we don't archive
222     hcd = new HColumnDescriptor(TEST_FAM);
223     HRegion otherRegion = UTIL.createTestRegion(otherTable, hcd);
224     loadFlushAndCompact(otherRegion, TEST_FAM);
225 
226     // get the current hfiles in the archive directory
227     List<Path> files = getAllFiles(fs, archiveDir);
228     if (files == null) {
229       FSUtils.logFileSystemState(fs, archiveDir, LOG);
230       throw new RuntimeException("Didn't load archive any files!");
231     }
232 
233     // make sure we have files from both tables
234     int initialCountForPrimary = 0;
235     int initialCountForOtherTable = 0;
236     for (Path file : files) {
237       String tableName = file.getParent().getParent().getParent().getName();
238       // check to which table this file belongs
239       if (tableName.equals(otherTable)) initialCountForOtherTable++;
240       else if (tableName.equals(STRING_TABLE_NAME)) initialCountForPrimary++;
241     }
242 
243     assertTrue("Didn't archive files for:" + STRING_TABLE_NAME, initialCountForPrimary > 0);
244     assertTrue("Didn't archive files for:" + otherTable, initialCountForOtherTable > 0);
245 
246     // run the cleaners, checking for each of the directories + files (both should be deleted and
247     // need to be checked) in 'otherTable' and the files (which should be retained) in the 'table'
248     CountDownLatch finished = setupCleanerWatching(delegate, cleaners, files.size() + 3);
249     // run the cleaner
250     cleaner.start();
251     // wait for the cleaner to check all the files
252     finished.await();
253     // stop the cleaner
254     stop.stop("");
255 
256     // know the cleaner ran, so now check all the files again to make sure they are still there
257     List<Path> archivedFiles = getAllFiles(fs, archiveDir);
258     int archivedForPrimary = 0;
259     for(Path file: archivedFiles) {
260       String tableName = file.getParent().getParent().getParent().getName();
261       // ensure we don't have files from the non-archived table
262       assertFalse("Have a file from the non-archived table: " + file, tableName.equals(otherTable));
263       if (tableName.equals(STRING_TABLE_NAME)) archivedForPrimary++;
264     }
265 
266     assertEquals("Not all archived files for the primary table were retained.", initialCountForPrimary,
267       archivedForPrimary);
268 
269     // but we still have the archive directory
270     assertTrue("Archive directory was deleted via archiver", fs.exists(archiveDir));
271   }

298   /**
299    * Start archiving table for given hfile cleaner
300    * @param tableName table to archive
301    * @param cleaner cleaner to check to make sure change propagated
302    * @return underlying {@link LongTermArchivingHFileCleaner} that is managing archiving
303    * @throws IOException on failure
304    * @throws KeeperException on failure
305    */
306   private List<BaseHFileCleanerDelegate> turnOnArchiving(String tableName, HFileCleaner cleaner)
307       throws IOException, KeeperException {
308     // turn on hfile retention
309     LOG.debug("----Starting archiving for table:" + tableName);
310     archivingClient.enableHFileBackupAsync(Bytes.toBytes(tableName));
311     assertTrue("Archving didn't get turned on", archivingClient.getArchivingEnabled(tableName));
312 
313     // wait for the archiver to get the notification
314     List<BaseHFileCleanerDelegate> cleaners = cleaner.getDelegatesForTesting();
315     LongTermArchivingHFileCleaner delegate = (LongTermArchivingHFileCleaner) cleaners.get(0);
316     while (!delegate.archiveTracker.keepHFiles(STRING_TABLE_NAME)) {
317       // spin until propagation - should be fast
318     }
319     return cleaners;
320   }

322   /**
323    * Spy on the {@link LongTermArchivingHFileCleaner} to ensure we can catch when the cleaner has
324    * seen all the files
325    * @return a {@link CountDownLatch} to wait on that releases when the cleaner has been called at
326    *         least the expected number of times.
327    */
328   private CountDownLatch setupCleanerWatching(LongTermArchivingHFileCleaner cleaner,
329       List<BaseHFileCleanerDelegate> cleaners, final int expected) {
330     // replace the cleaner with one that we can can check
331     BaseHFileCleanerDelegate delegateSpy = Mockito.spy(cleaner);
332     final int[] counter = new int[] { 0 };
333     final CountDownLatch finished = new CountDownLatch(1);
334     Mockito.doAnswer(new Answer<Boolean>() {
335 
336       @Override
337       public Boolean answer(InvocationOnMock invocation) throws Throwable {
338         counter[0]++;
339         LOG.debug(counter[0] + "/ " + expected + ") Wrapping call to isFileDeletable for file: "
340             + invocation.getArguments()[0]);
341 
342         Boolean ret = (Boolean) invocation.callRealMethod();
343         if (counter[0] >= expected) finished.countDown();
344         return ret;
345       }
346     }).when(delegateSpy).isFileDeletable(Mockito.any(Path.class));
347     cleaners.set(0, delegateSpy);
348 
349     return finished;
350   }

352   /**
353    * Get all the files (non-directory entries) in the file system under the passed directory
354    * @param dir directory to investigate
355    * @return all files under the directory
356    */
357   private List<Path> getAllFiles(FileSystem fs, Path dir) throws IOException {
358     FileStatus[] files = FSUtils.listStatus(fs, dir, null);
359     if (files == null) {
360       LOG.warn("No files under:" + dir);
361       return null;
362     }
363 
364     List<Path> allFiles = new ArrayList<Path>();
365     for (FileStatus file : files) {
366       if (file.isDir()) {
367         List<Path> subFiles = getAllFiles(fs, file.getPath());
368         if (subFiles != null) allFiles.addAll(subFiles);
369         continue;
370       }
371       allFiles.add(file.getPath());
372     }
373     return allFiles;
374   }

376   private void loadFlushAndCompact(HRegion region, byte[] family) throws IOException {
377     // create two hfiles in the region
378     createHFileInRegion(region, family);
379     createHFileInRegion(region, family);
380 
381     Store s = region.getStore(family);
382     int count = s.getStorefilesCount();
383     assertTrue("Don't have the expected store files, wanted >= 2 store files, but was:" + count,
384       count >= 2);
385 
386     // compact the two files into one file to get files in the archive
387     LOG.debug("Compacting stores");
388     region.compactStores(true);
389   }

391   /**
392    * Create a new hfile in the passed region
393    * @param region region to operate on
394    * @param columnFamily family for which to add data
395    * @throws IOException
396    */
397   private void createHFileInRegion(HRegion region, byte[] columnFamily) throws IOException {
398     // put one row in the region
399     Put p = new Put(Bytes.toBytes("row"));
400     p.add(columnFamily, Bytes.toBytes("Qual"), Bytes.toBytes("v1"));
401     region.put(p);
402     // flush the region to make a store file
403     region.flushcache();
404   }
