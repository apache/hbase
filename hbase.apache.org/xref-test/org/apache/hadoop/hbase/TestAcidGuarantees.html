46  /**
47   * Test case that uses multiple threads to read and write multifamily rows
48   * into a table, verifying that reads never see partially-complete writes.
49   * 
50   * This can run as a junit test, or with a main() function which runs against
51   * a real cluster (eg for testing with failures, region movement, etc)
52   */
53  @Category(MediumTests.class)
54  public class TestAcidGuarantees implements Tool {

87    /**
88     * Thread that does random full-row writes into a table.
89     */
90    public static class AtomicityWriter extends RepeatingTestThread {
91      Random rand = new Random();
92      byte data[] = new byte[10];
93      byte targetRows[][];
94      byte targetFamilies[][];
95      HTable table;
96      AtomicLong numWritten = new AtomicLong();
97      
98      public AtomicityWriter(TestContext ctx, byte targetRows[][],
99                             byte targetFamilies[][]) throws IOException {
100       super(ctx);
101       this.targetRows = targetRows;
102       this.targetFamilies = targetFamilies;
103       table = new HTable(ctx.getConf(), TABLE_NAME);
104     }
105     public void doAnAction() throws Exception {
106       // Pick a random row to write into
107       byte[] targetRow = targetRows[rand.nextInt(targetRows.length)];
108       Put p = new Put(targetRow); 
109       rand.nextBytes(data);
110 
111       for (byte[] family : targetFamilies) {
112         for (int i = 0; i < NUM_COLS_TO_CHECK; i++) {
113           byte qualifier[] = Bytes.toBytes("col" + i);
114           p.add(family, qualifier, data);
115         }
116       }
117       table.put(p);
118       numWritten.getAndIncrement();
119     }
120   }

122   /**
123    * Thread that does single-row reads in a table, looking for partially
124    * completed rows.
125    */
126   public static class AtomicGetReader extends RepeatingTestThread {
127     byte targetRow[];
128     byte targetFamilies[][];
129     HTable table;
130     int numVerified = 0;
131     AtomicLong numRead = new AtomicLong();
132 
133     public AtomicGetReader(TestContext ctx, byte targetRow[],
134                            byte targetFamilies[][]) throws IOException {
135       super(ctx);
136       this.targetRow = targetRow;
137       this.targetFamilies = targetFamilies;
138       table = new HTable(ctx.getConf(), TABLE_NAME);
139     }
140 
141     public void doAnAction() throws Exception {
142       Get g = new Get(targetRow);
143       Result res = table.get(g);
144       byte[] gotValue = null;
145       if (res.getRow() == null) {
146         // Trying to verify but we didn't find the row - the writing
147         // thread probably just hasn't started writing yet, so we can
148         // ignore this action
149         return;
150       }
151       
152       for (byte[] family : targetFamilies) {
153         for (int i = 0; i < NUM_COLS_TO_CHECK; i++) {
154           byte qualifier[] = Bytes.toBytes("col" + i);
155           byte thisValue[] = res.getValue(family, qualifier);
156           if (gotValue != null && !Bytes.equals(gotValue, thisValue)) {
157             gotFailure(gotValue, res);
158           }
159           numVerified++;
160           gotValue = thisValue;
161         }
162       }
163       numRead.getAndIncrement();
164     }
165 
166     private void gotFailure(byte[] expected, Result res) {
167       StringBuilder msg = new StringBuilder();
168       msg.append("Failed after ").append(numVerified).append("!");
169       msg.append("Expected=").append(Bytes.toStringBinary(expected));
170       msg.append("Got:\n");
171       for (KeyValue kv : res.list()) {
172         msg.append(kv.toString());
173         msg.append(" val= ");
174         msg.append(Bytes.toStringBinary(kv.getValue()));
175         msg.append("\n");
176       }
177       throw new RuntimeException(msg.toString());
178     }
179   }

181   /**
182    * Thread that does full scans of the table looking for any partially completed
183    * rows.
184    */
185   public static class AtomicScanReader extends RepeatingTestThread {
186     byte targetFamilies[][];
187     HTable table;
188     AtomicLong numScans = new AtomicLong();
189     AtomicLong numRowsScanned = new AtomicLong();
190 
191     public AtomicScanReader(TestContext ctx,
192                            byte targetFamilies[][]) throws IOException {
193       super(ctx);
194       this.targetFamilies = targetFamilies;
195       table = new HTable(ctx.getConf(), TABLE_NAME);
196     }
197 
198     public void doAnAction() throws Exception {
199       Scan s = new Scan();
200       for (byte[] family : targetFamilies) {
201         s.addFamily(family);
202       }
203       ResultScanner scanner = table.getScanner(s);
204       
205       for (Result res : scanner) {
206         byte[] gotValue = null;
207   
208         for (byte[] family : targetFamilies) {
209           for (int i = 0; i < NUM_COLS_TO_CHECK; i++) {
210             byte qualifier[] = Bytes.toBytes("col" + i);
211             byte thisValue[] = res.getValue(family, qualifier);
212             if (gotValue != null && !Bytes.equals(gotValue, thisValue)) {
213               gotFailure(gotValue, res);
214             }
215             gotValue = thisValue;
216           }
217         }
218         numRowsScanned.getAndIncrement();
219       }
220       numScans.getAndIncrement();
221     }
222 
223     private void gotFailure(byte[] expected, Result res) {
224       StringBuilder msg = new StringBuilder();
225       msg.append("Failed after ").append(numRowsScanned).append("!");
226       msg.append("Expected=").append(Bytes.toStringBinary(expected));
227       msg.append("Got:\n");
228       for (KeyValue kv : res.list()) {
229         msg.append(kv.toString());
230         msg.append(" val= ");
231         msg.append(Bytes.toStringBinary(kv.getValue()));
232         msg.append("\n");
233       }
234       throw new RuntimeException(msg.toString());
235     }
236   }

246   public void runTestAtomicity(long millisToRun,
247       int numWriters,
248       int numGetters,
249       int numScanners,
250       int numUniqueRows,
251       final boolean systemTest) throws Exception {
252     createTableIfMissing();
253     TestContext ctx = new TestContext(util.getConfiguration());
254     
255     byte rows[][] = new byte[numUniqueRows][];
256     for (int i = 0; i < numUniqueRows; i++) {
257       rows[i] = Bytes.toBytes("test_row_" + i);
258     }
259     
260     List<AtomicityWriter> writers = Lists.newArrayList();
261     for (int i = 0; i < numWriters; i++) {
262       AtomicityWriter writer = new AtomicityWriter(
263           ctx, rows, FAMILIES);
264       writers.add(writer);
265       ctx.addThread(writer);
266     }
267     // Add a flusher
268     ctx.addThread(new RepeatingTestThread(ctx) {
269       HBaseAdmin admin = new HBaseAdmin(util.getConfiguration());
270       public void doAnAction() throws Exception {
271         admin.flush(TABLE_NAME);
272         // Flushing has been a source of ACID violations previously (see HBASE-2856), so ideally,
273         // we would flush as often as possible.  On a running cluster, this isn't practical:
274         // (1) we will cause a lot of load due to all the flushing and compacting
275         // (2) we cannot change the flushing/compacting related Configuration options to try to
276         // alleviate this
277         // (3) it is an unrealistic workload, since no one would actually flush that often.
278         // Therefore, let's flush every minute to have more flushes than usual, but not overload
279         // the running cluster.
280         if (systemTest) Thread.sleep(60000);
281       }
282     });
283 
284     List<AtomicGetReader> getters = Lists.newArrayList();
285     for (int i = 0; i < numGetters; i++) {
286       AtomicGetReader getter = new AtomicGetReader(
287           ctx, rows[i % numUniqueRows], FAMILIES);
288       getters.add(getter);
289       ctx.addThread(getter);
290     }
291     
292     List<AtomicScanReader> scanners = Lists.newArrayList();
293     for (int i = 0; i < numScanners; i++) {
294       AtomicScanReader scanner = new AtomicScanReader(ctx, FAMILIES);
295       scanners.add(scanner);
296       ctx.addThread(scanner);
297     }
298     
299     ctx.startThreads();
300     ctx.waitFor(millisToRun);
301     ctx.stop();
302     
303     LOG.info("Finished test. Writers:");
304     for (AtomicityWriter writer : writers) {
305       LOG.info("  wrote " + writer.numWritten.get());
306     }
307     LOG.info("Readers:");
308     for (AtomicGetReader reader : getters) {
309       LOG.info("  read " + reader.numRead.get());
310     }
311     LOG.info("Scanners:");
312     for (AtomicScanReader scanner : scanners) {
313       LOG.info("  scanned " + scanner.numScans.get());
314       LOG.info("  verified " + scanner.numRowsScanned.get() + " rows");
315     }
316   }

362   @Override
363   public int run(String[] arg0) throws Exception {
364     Configuration c = getConf();
365     int millis = c.getInt("millis", 5000);
366     int numWriters = c.getInt("numWriters", 50);
367     int numGetters = c.getInt("numGetters", 2);
368     int numScanners = c.getInt("numScanners", 2);
369     int numUniqueRows = c.getInt("numUniqueRows", 3);
370     runTestAtomicity(millis, numWriters, numGetters, numScanners, numUniqueRows, true);
371     return 0;
372   }
