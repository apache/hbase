94        static enum ComparatorType {
95          BinaryComparator,
96          BinaryPrefixComparator,
97          BitComparator,
98          NullComparator,
99          RegexStringComparator,
100         SubstringComparator    
101       }
102 
103       public ByteArrayComparableModel() { }
104 
105       public ByteArrayComparableModel(
106           ByteArrayComparable comparator) {
107         String typeName = comparator.getClass().getSimpleName();
108         ComparatorType type = ComparatorType.valueOf(typeName);
109         this.type = typeName;
110         switch (type) {
111           case BinaryComparator:
112           case BinaryPrefixComparator:
113             this.value = Base64.encodeBytes(comparator.getValue());
114             break;
115           case BitComparator:
116             this.value = Base64.encodeBytes(comparator.getValue());
117             this.op = ((BitComparator)comparator).getOperator().toString();
118             break;
119           case NullComparator:
120             break;
121           case RegexStringComparator:
122           case SubstringComparator:
123             this.value = Bytes.toString(comparator.getValue());
124             break;
125           default:
126             throw new RuntimeException("unhandled filter type: " + type);
127         }
128       }
129 
130       public ByteArrayComparable build() {
131         ByteArrayComparable comparator;
132         switch (ComparatorType.valueOf(type)) {
133           case BinaryComparator:
134             comparator = new BinaryComparator(Base64.decode(value));
135             break;
136           case BinaryPrefixComparator:
137             comparator = new BinaryPrefixComparator(Base64.decode(value));
138             break;
139           case BitComparator:
140             comparator = new BitComparator(Base64.decode(value),
141                 BitComparator.BitwiseOp.valueOf(op));
142             break;
143           case NullComparator:
144             comparator = new NullComparator();
145             break;
146           case RegexStringComparator:
147             comparator = new RegexStringComparator(value);
148             break;
149           case SubstringComparator:
150             comparator = new SubstringComparator(value);
151             break;
152           default:
153             throw new RuntimeException("unhandled comparator type: " + type);
154         }
155         return comparator;
156       }

209     public FilterModel(Filter filter) { 
210       String typeName = filter.getClass().getSimpleName();
211       FilterType type = FilterType.valueOf(typeName);
212       this.type = typeName;
213       switch (type) {
214         case ColumnCountGetFilter:
215           this.limit = ((ColumnCountGetFilter)filter).getLimit();
216           break;
217         case ColumnPaginationFilter:
218           this.limit = ((ColumnPaginationFilter)filter).getLimit();
219           this.offset = ((ColumnPaginationFilter)filter).getOffset();
220           break;
221         case ColumnPrefixFilter:
222           this.value = Base64.encodeBytes(((ColumnPrefixFilter)filter).getPrefix());
223           break;
224         case ColumnRangeFilter:
225           this.minColumn = Base64.encodeBytes(((ColumnRangeFilter)filter).getMinColumn());
226           this.minColumnInclusive = ((ColumnRangeFilter)filter).getMinColumnInclusive();
227           this.maxColumn = Base64.encodeBytes(((ColumnRangeFilter)filter).getMaxColumn());
228           this.maxColumnInclusive = ((ColumnRangeFilter)filter).getMaxColumnInclusive();
229           break;
230         case DependentColumnFilter: {
231           DependentColumnFilter dcf = (DependentColumnFilter)filter;
232           this.family = Base64.encodeBytes(dcf.getFamily());
233           byte[] qualifier = dcf.getQualifier();
234           if (qualifier != null) {
235             this.qualifier = Base64.encodeBytes(qualifier);
236           }
237           this.op = dcf.getOperator().toString();
238           this.comparator = new ByteArrayComparableModel(dcf.getComparator());
239           this.dropDependentColumn = dcf.dropDependentColumn();
240         } break;
241         case FilterList:
242           this.op = ((FilterList)filter).getOperator().toString();
243           this.filters = new ArrayList<FilterModel>();
244           for (Filter child: ((FilterList)filter).getFilters()) {
245             this.filters.add(new FilterModel(child));
246           }
247           break;
248         case FirstKeyOnlyFilter:
249         case KeyOnlyFilter:
250           break;
251         case InclusiveStopFilter:
252           this.value = 
253             Base64.encodeBytes(((InclusiveStopFilter)filter).getStopRowKey());
254           break;
255         case MultipleColumnPrefixFilter:
256           this.prefixes = new ArrayList<String>();
257           for (byte[] prefix: ((MultipleColumnPrefixFilter)filter).getPrefix()) {
258             this.prefixes.add(Base64.encodeBytes(prefix));
259           }
260           break;
261         case PageFilter:
262           this.value = Long.toString(((PageFilter)filter).getPageSize());
263           break;
264         case PrefixFilter:
265           this.value = Base64.encodeBytes(((PrefixFilter)filter).getPrefix());
266           break;
267         case FamilyFilter:
268         case QualifierFilter:
269         case RowFilter:
270         case ValueFilter:
271           this.op = ((CompareFilter)filter).getOperator().toString();
272           this.comparator = 
273             new ByteArrayComparableModel(
274               ((CompareFilter)filter).getComparator());
275           break;
276         case RandomRowFilter:
277           this.chance = ((RandomRowFilter)filter).getChance();
278           break;
279         case SingleColumnValueExcludeFilter:
280         case SingleColumnValueFilter: {
281           SingleColumnValueFilter scvf = (SingleColumnValueFilter) filter;
282           this.family = Base64.encodeBytes(scvf.getFamily());
283           byte[] qualifier = scvf.getQualifier();
284           if (qualifier != null) {
285             this.qualifier = Base64.encodeBytes(qualifier);
286           }
287           this.op = scvf.getOperator().toString();
288           this.comparator = 
289             new ByteArrayComparableModel(scvf.getComparator());
290           if (scvf.getFilterIfMissing()) {
291             this.ifMissing = true;
292           }
293           if (scvf.getLatestVersionOnly()) {
294             this.latestVersion = true;
295           }
296         } break;
297         case SkipFilter:
298           this.filters = new ArrayList<FilterModel>();
299           this.filters.add(new FilterModel(((SkipFilter)filter).getFilter()));
300           break;
301         case TimestampsFilter:
302           this.timestamps = ((TimestampsFilter)filter).getTimestamps();
303           break;
304         case WhileMatchFilter:
305           this.filters = new ArrayList<FilterModel>();
306           this.filters.add(
307             new FilterModel(((WhileMatchFilter)filter).getFilter()));
308           break;
309         default:
310           throw new RuntimeException("unhandled filter type " + type);
311       }
312     }

314     public Filter build() {
315       Filter filter;
316       switch (FilterType.valueOf(type)) {
317       case ColumnCountGetFilter:
318         filter = new ColumnCountGetFilter(limit);
319         break;
320       case ColumnPaginationFilter:
321         filter = new ColumnPaginationFilter(limit, offset);
322         break;
323       case ColumnPrefixFilter:
324         filter = new ColumnPrefixFilter(Base64.decode(value));
325         break;
326       case ColumnRangeFilter:
327         filter = new ColumnRangeFilter(Base64.decode(minColumn),
328             minColumnInclusive, Base64.decode(maxColumn),
329             maxColumnInclusive);
330         break;
331       case DependentColumnFilter:
332         filter = new DependentColumnFilter(Base64.decode(family),
333             qualifier != null ? Base64.decode(qualifier) : null,
334             dropDependentColumn, CompareOp.valueOf(op), comparator.build());
335         break;
336       case FamilyFilter:
337         filter = new FamilyFilter(CompareOp.valueOf(op), comparator.build());
338         break;
339       case FilterList: {
340         List<Filter> list = new ArrayList<Filter>();
341         for (FilterModel model: filters) {
342           list.add(model.build());
343         }
344         filter = new FilterList(FilterList.Operator.valueOf(op), list);
345       } break;
346       case FirstKeyOnlyFilter:
347         filter = new FirstKeyOnlyFilter();
348         break;
349       case InclusiveStopFilter:
350         filter = new InclusiveStopFilter(Base64.decode(value));
351         break;
352       case KeyOnlyFilter:
353         filter = new KeyOnlyFilter();
354         break;
355       case MultipleColumnPrefixFilter: {
356         byte[][] values = new byte[prefixes.size()][];
357         for (int i = 0; i < prefixes.size(); i++) {
358           values[i] = Base64.decode(prefixes.get(i));
359         }
360         filter = new MultipleColumnPrefixFilter(values);
361       } break;
362       case PageFilter:
363         filter = new PageFilter(Long.valueOf(value));
364         break;
365       case PrefixFilter:
366         filter = new PrefixFilter(Base64.decode(value));
367         break;
368       case QualifierFilter:
369         filter = new QualifierFilter(CompareOp.valueOf(op), comparator.build());
370         break;
371       case RandomRowFilter:
372         filter = new RandomRowFilter(chance);
373         break;
374       case RowFilter:
375         filter = new RowFilter(CompareOp.valueOf(op), comparator.build());
376         break;
377       case SingleColumnValueFilter:
378         filter = new SingleColumnValueFilter(Base64.decode(family),
379           qualifier != null ? Base64.decode(qualifier) : null,
380           CompareOp.valueOf(op), comparator.build());
381         if (ifMissing != null) {
382           ((SingleColumnValueFilter)filter).setFilterIfMissing(ifMissing);
383         }
384         if (latestVersion != null) {
385           ((SingleColumnValueFilter)filter).setLatestVersionOnly(latestVersion);
386         }
387         break;
388       case SingleColumnValueExcludeFilter:
389         filter = new SingleColumnValueExcludeFilter(Base64.decode(family),
390           qualifier != null ? Base64.decode(qualifier) : null,
391           CompareOp.valueOf(op), comparator.build());
392         if (ifMissing != null) {
393           ((SingleColumnValueExcludeFilter)filter).setFilterIfMissing(ifMissing);
394         }
395         if (latestVersion != null) {
396           ((SingleColumnValueExcludeFilter)filter).setLatestVersionOnly(latestVersion);
397         }
398         break;
399       case SkipFilter:
400         filter = new SkipFilter(filters.get(0).build());
401         break;
402       case TimestampsFilter:
403         filter = new TimestampsFilter(timestamps);
404         break;
405       case ValueFilter:
406         filter = new ValueFilter(CompareOp.valueOf(op), comparator.build());
407         break;
408       case WhileMatchFilter:
409         filter = new WhileMatchFilter(filters.get(0).build());
410         break;
411       default:
412         throw new RuntimeException("unhandled filter type: " + type);
413       }
414       return filter;
415     }

449   private static final byte[] COLUMN_DIVIDER = Bytes.toBytes(":");
450 
451   /**
452    * @param scan the scan specification
453    * @throws Exception 
454    */
455   public static ScannerModel fromScan(Scan scan) throws Exception {
456     ScannerModel model = new ScannerModel();
457     model.setStartRow(scan.getStartRow());
458     model.setEndRow(scan.getStopRow());
459     Map<byte [], NavigableSet<byte []>> families = scan.getFamilyMap();
460     if (families != null) {
461       for (Map.Entry<byte [], NavigableSet<byte []>> entry : families.entrySet()) {
462         if (entry.getValue() != null) {
463           for (byte[] qualifier: entry.getValue()) {
464             model.addColumn(Bytes.add(entry.getKey(), COLUMN_DIVIDER, qualifier));
465           }
466         } else {
467           model.addColumn(entry.getKey());
468         }
469       }
470     }
471     model.setStartTime(scan.getTimeRange().getMin());
472     model.setEndTime(scan.getTimeRange().getMax());
473     int caching = scan.getCaching();
474     if (caching > 0) {
475       model.setBatch(caching);
476     }
477     int maxVersions = scan.getMaxVersions();
478     if (maxVersions > 0) {
479       model.setMaxVersions(maxVersions);
480     }
481     Filter filter = scan.getFilter();
482     if (filter != null) {
483       model.setFilter(stringifyFilter(filter));
484     }
485     return model;
486   }

682   @Override
683   public byte[] createProtobufOutput() {
684     Scanner.Builder builder = Scanner.newBuilder();
685     if (!Bytes.equals(startRow, HConstants.EMPTY_START_ROW)) {
686       builder.setStartRow(ByteString.copyFrom(startRow));
687     }
688     if (!Bytes.equals(endRow, HConstants.EMPTY_START_ROW)) {
689       builder.setEndRow(ByteString.copyFrom(endRow));
690     }
691     for (byte[] column: columns) {
692       builder.addColumns(ByteString.copyFrom(column));
693     }
694     if (startTime != 0) {
695       builder.setStartTime(startTime);
696     }
697     if (endTime != 0) {
698       builder.setEndTime(endTime);
699     }
700     builder.setBatch(getBatch());
701     builder.setMaxVersions(maxVersions);
702     if (filter != null) {
703       builder.setFilter(filter);
704     }
705     return builder.build().toByteArray();
706   }

708   @Override
709   public ProtobufMessageHandler getObjectFromMessage(byte[] message)
710       throws IOException {
711     Scanner.Builder builder = Scanner.newBuilder();
712     builder.mergeFrom(message);
713     if (builder.hasStartRow()) {
714       startRow = builder.getStartRow().toByteArray();
715     }
716     if (builder.hasEndRow()) {
717       endRow = builder.getEndRow().toByteArray();
718     }
719     for (ByteString column: builder.getColumnsList()) {
720       addColumn(column.toByteArray());
721     }
722     if (builder.hasBatch()) {
723       batch = builder.getBatch();
724     }
725     if (builder.hasStartTime()) {
726       startTime = builder.getStartTime();
727     }
