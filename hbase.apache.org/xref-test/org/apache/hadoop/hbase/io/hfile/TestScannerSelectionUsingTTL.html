50  /**
51   * Test the optimization that does not scan files where all timestamps are
52   * expired.
53   */
54  @RunWith(Parameterized.class)
55  @Category(MediumTests.class)
56  public class TestScannerSelectionUsingTTL {
57  
58    private static final Log LOG =
59        LogFactory.getLog(TestScannerSelectionUsingTTL.class);
60  
61    private static final HBaseTestingUtility TEST_UTIL =
62        new HBaseTestingUtility();
63    private static String TABLE = "myTable";
64    private static String FAMILY = "myCF";
65    private static byte[] FAMILY_BYTES = Bytes.toBytes(FAMILY);
66  
67    private static final int TTL_SECONDS = 10;
68    private static final int TTL_MS = TTL_SECONDS * 1000;
69  
70    private static final int NUM_EXPIRED_FILES = 2;
71    private static final int NUM_ROWS = 8;
72    private static final int NUM_COLS_PER_ROW = 5;
73  
74    public final int numFreshFiles, totalNumFiles;
75  
76    /** Whether we are specifying the exact files to compact */
77    private final boolean explicitCompaction;
78  
79    @Parameters
80    public static Collection<Object[]> parameters() {
81      List<Object[]> params = new ArrayList<Object[]>();
82      for (int numFreshFiles = 1; numFreshFiles <= 3; ++numFreshFiles) {
83        for (boolean explicitCompaction : new boolean[] { false, true }) {
84          params.add(new Object[] { numFreshFiles, explicitCompaction });
85        }
86      }
87      return params;
88    }
89  
90    public TestScannerSelectionUsingTTL(int numFreshFiles,
91        boolean explicitCompaction) {
92      this.numFreshFiles = numFreshFiles;
93      this.totalNumFiles = numFreshFiles + NUM_EXPIRED_FILES;
94      this.explicitCompaction = explicitCompaction;
95    }
96  
97    @Test
98    public void testScannerSelection() throws IOException {
99      Configuration conf = TEST_UTIL.getConfiguration();
100     conf.setBoolean("hbase.store.delete.expired.storefile", false);
101     HColumnDescriptor hcd =
102       new HColumnDescriptor(FAMILY_BYTES)
103           .setMaxVersions(Integer.MAX_VALUE)
104           .setTimeToLive(TTL_SECONDS);
105     HTableDescriptor htd = new HTableDescriptor(TABLE);
106     htd.addFamily(hcd);
107     HRegionInfo info = new HRegionInfo(Bytes.toBytes(TABLE));
108     HRegion region =
109         HRegion.createHRegion(info, TEST_UTIL.getDataTestDir(info.getEncodedName()),
110             conf, htd);
111 
112     for (int iFile = 0; iFile < totalNumFiles; ++iFile) {
113       if (iFile == NUM_EXPIRED_FILES) {
114         Threads.sleepWithoutInterrupt(TTL_MS);
115       }
116 
117       for (int iRow = 0; iRow < NUM_ROWS; ++iRow) {
118         Put put = new Put(Bytes.toBytes("row" + iRow));
119         for (int iCol = 0; iCol < NUM_COLS_PER_ROW; ++iCol) {
120           put.add(FAMILY_BYTES, Bytes.toBytes("col" + iCol),
121               Bytes.toBytes("value" + iFile + "_" + iRow + "_" + iCol));
122         }
123         region.put(put);
124       }
125       region.flushcache();
126     }
127 
128     Scan scan = new Scan();
129     scan.setMaxVersions(Integer.MAX_VALUE);
130     CacheConfig cacheConf = new CacheConfig(conf);
131     LruBlockCache cache = (LruBlockCache) cacheConf.getBlockCache();
132     cache.clearCache();
133     InternalScanner scanner = region.getScanner(scan);
134     List<KeyValue> results = new ArrayList<KeyValue>();
135     final int expectedKVsPerRow = numFreshFiles * NUM_COLS_PER_ROW;
136     int numReturnedRows = 0;
137     LOG.info("Scanning the entire table");
138     while (scanner.next(results) || results.size() > 0) {
139       assertEquals(expectedKVsPerRow, results.size());
140       ++numReturnedRows;
141       results.clear();
142     }
143     assertEquals(NUM_ROWS, numReturnedRows);
144     Set<String> accessedFiles = cache.getCachedFileNamesForTest();
145     LOG.debug("Files accessed during scan: " + accessedFiles);
146 
147     // Exercise both compaction codepaths.
148     if (explicitCompaction) {
149       region.getStore(FAMILY_BYTES).compactRecentForTesting(totalNumFiles);
150     } else {
151       region.compactStores();
152     }
153 
154     region.close();
155   }
