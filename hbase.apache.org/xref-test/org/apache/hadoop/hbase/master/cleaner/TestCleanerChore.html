178   /**
179    * While cleaning a directory, all the files in the directory may be deleted, but there may be
180    * another file added, in which case the directory shouldn't be deleted.
181    * @throws IOException on failure
182    */
183   @Test
184   public void testCleanerDoesNotDeleteDirectoryWithLateAddedFiles() throws IOException {
185     Stoppable stop = new StoppableImplementation();
186     Configuration conf = UTIL.getConfiguration();
187     final Path testDir = UTIL.getDataTestDir();
188     final FileSystem fs = UTIL.getTestFileSystem();
189     String confKey = "hbase.test.cleaner.delegates";
190     conf.set(confKey, AlwaysDelete.class.getName());
191 
192     AllValidPaths chore = new AllValidPaths("test-file-cleaner", stop, conf, fs, testDir, confKey);
193     // spy on the delegate to ensure that we don't check for directories
194     AlwaysDelete delegate = (AlwaysDelete) chore.cleanersChain.get(0);
195     AlwaysDelete spy = Mockito.spy(delegate);
196     chore.cleanersChain.set(0, spy);
197 
198     // create the directory layout in the directory to clean
199     final Path parent = new Path(testDir, "parent");
200     Path file = new Path(parent, "someFile");
201     fs.mkdirs(parent);
202     // touch a new file
203     fs.create(file).close();
204     assertTrue("Test file didn't get created.", fs.exists(file));
205     final Path addedFile = new Path(parent, "addedFile");
206 
207     // when we attempt to delete the original file, add another file in the same directory
208     Mockito.doAnswer(new Answer<Boolean>() {
209       @Override
210       public Boolean answer(InvocationOnMock invocation) throws Throwable {
211         fs.create(addedFile).close();
212         FSUtils.logFileSystemState(fs, testDir, LOG);
213         return (Boolean) invocation.callRealMethod();
214       }
215     }).when(spy).isFileDeletable(Mockito.any(Path.class));
216 
217     // run the chore
218     chore.chore();
219 
220     // make sure all the directories + added file exist, but the original file is deleted
221     assertTrue("Added file unexpectedly deleted", fs.exists(addedFile));
222     assertTrue("Parent directory deleted unexpectedly", fs.exists(parent));
223     assertFalse("Original file unexpectedly retained", fs.exists(file));
224     Mockito.verify(spy, Mockito.times(1)).isFileDeletable(Mockito.any(Path.class));
225     Mockito.reset(spy);
226   }
