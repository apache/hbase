35  /**
36   * Handles closing of a region on a region server.
37   */
38  @InterfaceAudience.Private
39  public class CloseRegionHandler extends EventHandler {
40    // NOTE on priorities shutting down.  There are none for close. There are some
41    // for open.  I think that is right.  On shutdown, we want the meta to close
42    // before root and both to close after the user regions have closed.  What
43    // about the case where master tells us to shutdown a catalog region and we
44    // have a running queue of user regions to close?
45    private static final Log LOG = LogFactory.getLog(CloseRegionHandler.class);
46  
47    private final int FAILED = -1;
48    int expectedVersion = FAILED;
49  
50    private final RegionServerServices rsServices;
51  
52    private final HRegionInfo regionInfo;
53  
54    // If true, the hosting server is aborting.  Region close process is different
55    // when we are aborting.
56    private final boolean abort;
57  
58    // Update zk on closing transitions. Usually true.  Its false if cluster
59    // is going down.  In this case, its the rs that initiates the region
60    // close -- not the master process so state up in zk will unlikely be
61    // CLOSING.
62    private final boolean zk;
63    private ServerName destination;
64  
65    // This is executed after receiving an CLOSE RPC from the master.
66    public CloseRegionHandler(final Server server,
67        final RegionServerServices rsServices, HRegionInfo regionInfo) {
68      this(server, rsServices, regionInfo, false, true, -1, EventType.M_RS_CLOSE_REGION, null);
69    }

102     protected CloseRegionHandler(final Server server,
103       final RegionServerServices rsServices, HRegionInfo regionInfo,
104       boolean abort, final boolean zk, final int versionOfClosingNode,
105       EventType eventType, ServerName destination) {
106     super(server, eventType);
107     this.server = server;
108     this.rsServices = rsServices;
109     this.regionInfo = regionInfo;
110     this.abort = abort;
111     this.zk = zk;
112     this.expectedVersion = versionOfClosingNode;
113     this.destination = destination;
114   }
115 
116   public HRegionInfo getRegionInfo() {
117     return regionInfo;
118   }
119 
120   @Override
121   public void process() {
122     try {
123       String name = regionInfo.getRegionNameAsString();
124       LOG.debug("Processing close of " + name);
125       String encodedRegionName = regionInfo.getEncodedName();
126       // Check that this region is being served here
127       HRegion region = this.rsServices.getFromOnlineRegions(encodedRegionName);
128       if (region == null) {
129         LOG.warn("Received CLOSE for region " + name +
130             " but currently not serving");
131         return;
132       }
133 
134       // Close the region
135       try {
136         // TODO: If we need to keep updating CLOSING stamp to prevent against
137         // a timeout if this is long-running, need to spin up a thread?
138         if (region.close(abort) == null) {
139           // This region got closed.  Most likely due to a split. So instead
140           // of doing the setClosedState() below, let's just ignore cont
141           // The split message will clean up the master state.
142           LOG.warn("Can't close region: was already closed during close(): " +
143             regionInfo.getRegionNameAsString());
144           return;
145         }
146       } catch (Throwable t) {
147         // A throwable here indicates that we couldn't successfully flush the
148         // memstore before closing. So, we need to abort the server and allow
149         // the master to split our logs in order to recover the data.
150         server.abort("Unrecoverable exception while closing region " +
151           regionInfo.getRegionNameAsString() + ", still finishing close", t);
152         throw new RuntimeException(t);
153       }
154 
155       this.rsServices.removeFromOnlineRegions(regionInfo.getEncodedName(), destination);
156 
157       if (this.zk) {
158         if (setClosedState(this.expectedVersion, region)) {
159           LOG.debug("set region closed state in zk successfully for region " +
160             name + " sn name: " + this.server.getServerName());
161         } else {
162           LOG.debug("set region closed state in zk unsuccessfully for region " +
163             name + " sn name: " + this.server.getServerName());
164         }
165       }
166 
167       // Done!  Region is closed on this RS
168       LOG.debug("Closed region " + region.getRegionNameAsString());
169     } finally {
170       this.rsServices.getRegionsInTransitionInRS().
171           remove(this.regionInfo.getEncodedNameAsBytes());
172     }
173   }
174 
175   /**
176    * Transition ZK node to CLOSED
177    * @param expectedVersion
178    * @return If the state is set successfully
179    */
180   private boolean setClosedState(final int expectedVersion, final HRegion region) {
181     try {
182       if (ZKAssign.transitionNodeClosed(server.getZooKeeper(), regionInfo,
183           server.getServerName(), expectedVersion) == FAILED) {
184         LOG.warn("Completed the CLOSE of a region but when transitioning from " +
185             " CLOSING to CLOSED got a version mismatch, someone else clashed " +
186             "so now unassigning");
187         region.close();
188         return false;
189       }
190     } catch (NullPointerException e) {
191       // I've seen NPE when table was deleted while close was running in unit tests.
192       LOG.warn("NPE during close -- catching and continuing...", e);
193       return false;
194     } catch (KeeperException e) {
195       LOG.error("Failed transitioning node from CLOSING to CLOSED", e);
196       return false;
197     } catch (IOException e) {
198       LOG.error("Failed to close region after failing to transition", e);
199       return false;
200     }
201     return true;
202   }
