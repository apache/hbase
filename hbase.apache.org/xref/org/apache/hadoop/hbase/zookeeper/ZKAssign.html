42  /**
43   * Utility class for doing region assignment in ZooKeeper.  This class extends
44   * stuff done in {@link ZKUtil} to cover specific assignment operations.
45   * <p>
46   * Contains only static methods and constants.
47   * <p>
48   * Used by both the Master and RegionServer.
49   * <p>
50   * All valid transitions outlined below:
51   * <p>
52   * <b>MASTER</b>
53   * <ol>
54   *   <li>
55   *     Master creates an unassigned node as OFFLINE.
56   *     - Cluster startup and table enabling.
57   *   </li>
58   *   <li>
59   *     Master forces an existing unassigned node to OFFLINE.
60   *     - RegionServer failure.
61   *     - Allows transitions from all states to OFFLINE.
62   *   </li>
63   *   <li>
64   *     Master deletes an unassigned node that was in a OPENED state.
65   *     - Normal region transitions.  Besides cluster startup, no other deletions
66   *     of unassigned nodes is allowed.
67   *   </li>
68   *   <li>
69   *     Master deletes all unassigned nodes regardless of state.
70   *     - Cluster startup before any assignment happens.
71   *   </li>
72   * </ol>
73   * <p>
74   * <b>REGIONSERVER</b>
75   * <ol>
76   *   <li>
77   *     RegionServer creates an unassigned node as CLOSING.
78   *     - All region closes will do this in response to a CLOSE RPC from Master.
79   *     - A node can never be transitioned to CLOSING, only created.
80   *   </li>
81   *   <li>
82   *     RegionServer transitions an unassigned node from CLOSING to CLOSED.
83   *     - Normal region closes.  CAS operation.
84   *   </li>
85   *   <li>
86   *     RegionServer transitions an unassigned node from OFFLINE to OPENING.
87   *     - All region opens will do this in response to an OPEN RPC from the Master.
88   *     - Normal region opens.  CAS operation.
89   *   </li>
90   *   <li>
91   *     RegionServer transitions an unassigned node from OPENING to OPENED.
92   *     - Normal region opens.  CAS operation.
93   *   </li>
94   * </ol>
95   */
96  @InterfaceAudience.Public

146   public static void createNodeOffline(ZooKeeperWatcher zkw, HRegionInfo region,
147       ServerName serverName, final EventType event)

184   /**
185    * Creates or force updates an unassigned node to the OFFLINE state for the
186    * specified region.
187    * <p>
188    * Attempts to create the node but if it exists will force it to transition to
189    * and OFFLINE state.
190    *
191    * <p>Sets a watcher on the unassigned region node if the method is
192    * successful.
193    *
194    * <p>This method should be used when assigning a region.
195    *
196    * @param zkw zk reference
197    * @param region region to be created as offline
198    * @param serverName server transition will happen on
199    * @return the version of the znode created in OFFLINE state, -1 if
200    *         unsuccessful.
201    * @throws KeeperException if unexpected zookeeper exception
202    * @throws KeeperException.NodeExistsException if node already exists
203    */
204   public static int createOrForceNodeOffline(ZooKeeperWatcher zkw,
205       HRegionInfo region, ServerName serverName) throws KeeperException {
206     LOG.debug(zkw.prefix("Creating (or updating) unassigned node for " +
207       region.getEncodedName() + " with OFFLINE state"));
208     RegionTransition rt = RegionTransition.createRegionTransition(EventType.M_ZK_REGION_OFFLINE,
209       region.getRegionName(), serverName, HConstants.EMPTY_BYTE_ARRAY);
210     byte [] data = rt.toByteArray();
211     String node = getNodeName(zkw, region.getEncodedName());
212     zkw.sync(node);
213     int version = ZKUtil.checkExists(zkw, node);
214     if (version == -1) {
215       return ZKUtil.createAndWatch(zkw, node, data);
216     } else {
217       boolean setData = false;
218       try {
219         setData = ZKUtil.setData(zkw, node, data, version);
220         // Setdata throws KeeperException which aborts the Master. So we are
221         // catching it here.
222         // If just before setting the znode to OFFLINE if the RS has made any
223         // change to the
224         // znode state then we need to return -1.
225       } catch (KeeperException kpe) {
226         LOG.info("Version mismatch while setting the node to OFFLINE state.");
227         return -1;
228       }
229       if (!setData) {
230         return -1;
231       } else {
232         // We successfully forced to OFFLINE, reset watch and handle if
233         // the state changed in between our set and the watch
234         byte [] bytes = ZKAssign.getData(zkw, region.getEncodedName());
235         rt = getRegionTransition(bytes);
236         if (rt.getEventType() != EventType.M_ZK_REGION_OFFLINE) {
237           // state changed, need to process
238           return -1;
239         }
240       }
241     }
242     return version + 1;
243   }

376   /**
377    * Deletes an existing unassigned node that is in the specified state for the
378    * specified region.
379    *
380    * <p>If a node does not already exist for this region, a
381    * {@link NoNodeException} will be thrown.
382    *
383    * <p>No watcher is set whether this succeeds or not.
384    *
385    * <p>Returns false if the node was not in the proper state but did exist.
386    *
387    * <p>This method is used when a region finishes opening/closing.
388    * The Master acknowledges completion
389    * of the specified regions transition to being closed/opened.
390    *
391    * @param zkw zk reference
392    * @param regionName region to be deleted from zk
393    * @param expectedState state region must be in for delete to complete
394    * @param expectedVersion of the znode that is to be deleted.
395    *        If expectedVersion need not be compared while deleting the znode
396    *        pass -1
397    * @throws KeeperException if unexpected zookeeper exception
398    * @throws KeeperException.NoNodeException if node does not exist
399    */
400   public static boolean deleteNode(ZooKeeperWatcher zkw, String regionName,
401       EventType expectedState, int expectedVersion)
402   throws KeeperException, KeeperException.NoNodeException {
403     LOG.debug(zkw.prefix("Deleting existing unassigned " +
404       "node for " + regionName + " that is in expected state " + expectedState));
405     String node = getNodeName(zkw, regionName);
406     zkw.sync(node);
407     Stat stat = new Stat();
408     byte [] bytes = ZKUtil.getDataNoWatch(zkw, node, stat);
409     if (bytes == null) {
410       // If it came back null, node does not exist.
411       throw KeeperException.create(Code.NONODE);
412     }
413     RegionTransition rt = getRegionTransition(bytes);
414     EventType et = rt.getEventType();
415     if (!et.equals(expectedState)) {
416       LOG.warn(zkw.prefix("Attempting to delete unassigned node " + regionName + " in " +
417         expectedState + " state but node is in " + et + " state"));
418       return false;
419     }
420     if (expectedVersion != -1
421         && stat.getVersion() != expectedVersion) {
422       LOG.warn("The node " + regionName + " we are trying to delete is not" +
423         " the expected one. Got a version mismatch");
424       return false;
425     }
426     if(!ZKUtil.deleteNode(zkw, node, stat.getVersion())) {
427       LOG.warn(zkw.prefix("Attempting to delete " +
428           "unassigned node " + regionName + " in " + expectedState +
429           " state but after verifying state, we got a version mismatch"));
430       return false;
431     }
432     LOG.debug(zkw.prefix("Successfully deleted unassigned node for region " +
433         regionName + " in expected state " + expectedState));
434     return true;
435   }

636    * <p>Attempts to transition the unassigned node for the specified region
637    * from the expected state to the state in the specified transition data.

668   public static int transitionNode(ZooKeeperWatcher zkw, HRegionInfo region,
669       ServerName serverName, EventType beginState, EventType endState,
670       int expectedVersion, final byte [] payload)
671   throws KeeperException {
672     String encoded = region.getEncodedName();
673     if(LOG.isDebugEnabled()) {
674       LOG.debug(zkw.prefix("Attempting to transition node " +
675         HRegionInfo.prettyPrint(encoded) +
676         " from " + beginState.toString() + " to " + endState.toString()));
677     }
678 
679     String node = getNodeName(zkw, encoded);
680     zkw.sync(node);
681 
682     // Read existing data of the node
683     Stat stat = new Stat();
684     byte [] existingBytes = ZKUtil.getDataNoWatch(zkw, node, stat);
685     if (existingBytes == null) {
686       // Node no longer exists.  Return -1. It means unsuccessful transition.
687       return -1;
688     }
689     RegionTransition rt = getRegionTransition(existingBytes);
690 
691     // Verify it is the expected version
692     if (expectedVersion != -1 && stat.getVersion() != expectedVersion) {
693       LOG.warn(zkw.prefix("Attempt to transition the " +
694         "unassigned node for " + encoded +
695         " from " + beginState + " to " + endState + " failed, " +
696         "the node existed but was version " + stat.getVersion() +
697         " not the expected version " + expectedVersion));
698         return -1;
699     } else if (beginState.equals(EventType.M_ZK_REGION_OFFLINE)
700         && endState.equals(EventType.RS_ZK_REGION_OPENING)
701         && expectedVersion == -1 && stat.getVersion() != 0) {
702       // the below check ensures that double assignment doesnot happen.
703       // When the node is created for the first time then the expected version
704       // that is passed will be -1 and the version in znode will be 0.
705       // In all other cases the version in znode will be > 0.
706       LOG.warn(zkw.prefix("Attempt to transition the " + "unassigned node for "
707           + encoded + " from " + beginState + " to " + endState + " failed, "
708           + "the node existed but was version " + stat.getVersion()
709           + " not the expected version " + expectedVersion));
710       return -1;
711     }
712 
713     // Verify it is in expected state
714     EventType et = rt.getEventType();
715     if (!et.equals(beginState)) {
716       LOG.warn(zkw.prefix("Attempt to transition the " +
717         "unassigned node for " + encoded +
718         " from " + beginState + " to " + endState + " failed, " +
719         "the node existed but was in the state " + et + " set by the server " + serverName));
720       return -1;
721     }
722 
723     // Write new data, ensuring data has not changed since we last read it
724     try {
725       rt = RegionTransition.createRegionTransition(endState, region.getRegionName(), serverName, payload);
726       if(!ZKUtil.setData(zkw, node, rt.toByteArray(), stat.getVersion())) {
727         LOG.warn(zkw.prefix("Attempt to transition the " +
728         "unassigned node for " + encoded +
729         " from " + beginState + " to " + endState + " failed, " +
730         "the node existed and was in the expected state but then when " +
731         "setting data we got a version mismatch"));
732         return -1;
733       }
734       if(LOG.isDebugEnabled()) {
735         LOG.debug(zkw.prefix("Successfully transitioned node " + encoded +
736           " from " + beginState + " to " + endState));
737       }
738       return stat.getVersion() + 1;
739     } catch (KeeperException.NoNodeException nne) {
740       LOG.warn(zkw.prefix("Attempt to transition the " +
741         "unassigned node for " + encoded +
742         " from " + beginState + " to " + endState + " failed, " +
743         "the node existed and was in the expected state but then when " +
744         "setting data it no longer existed"));
745       return -1;
746     }
747   }
