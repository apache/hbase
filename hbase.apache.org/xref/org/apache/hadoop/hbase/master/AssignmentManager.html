345   /**
346    * Process all regions that are in transition in zookeeper and also
347    * processes the list of dead servers by scanning the META.
348    * Used by master joining an cluster.  If we figure this is a clean cluster
349    * startup, will assign all user regions.
350    * @param deadServers
351    *          Map of dead servers and their regions. Can be null.
352    * @throws KeeperException
353    * @throws IOException
354    * @throws InterruptedException
355    */
356   void processDeadServersAndRegionsInTransition(
357       final Map<ServerName, List<HRegionInfo>> deadServers)
358           throws KeeperException, IOException, InterruptedException {
359     List<String> nodes = ZKUtil.listChildrenNoWatch(watcher,
360       watcher.assignmentZNode);
361 
362     if (nodes == null) {
363       String errorMessage = "Failed to get the children from ZK";
364       server.abort(errorMessage, new IOException(errorMessage));
365       return;
366     }
367 
368     boolean failover = !serverManager.getDeadServers().isEmpty();
369 
370     if (!failover) {
371       // Run through all regions.  If they are not assigned and not in RIT, then
372       // its a clean cluster startup, else its a failover.
373       Map<HRegionInfo, ServerName> regions = regionStates.getRegionAssignments();
374       for (Map.Entry<HRegionInfo, ServerName> e: regions.entrySet()) {
375         if (!e.getKey().isMetaTable() && e.getValue() != null) {
376           LOG.debug("Found " + e + " out on cluster");
377           failover = true;
378           break;
379         }
380         if (nodes.contains(e.getKey().getEncodedName())) {
381           LOG.debug("Found " + e.getKey().getRegionNameAsString() + " in RITs");
382           // Could be a meta region.
383           failover = true;
384           break;
385         }
386       }
387     }
388 
389     // If we found user regions out on cluster, its a failover.
390     if (failover) {
391       LOG.info("Found regions out on cluster or in RIT; failover");
392       // Process list of dead servers and regions in RIT.
393       // See HBASE-4580 for more information.
394       processDeadServersAndRecoverLostRegions(deadServers, nodes);
395     } else {
396       // Fresh cluster startup.
397       LOG.info("Clean cluster startup. Assigning userregions");
398       assignAllUserRegions();
399     }
400   }

402   /**
403    * If region is up in zk in transition, then do fixup and block and wait until
404    * the region is assigned and out of transition.  Used on startup for
405    * catalog regions.
406    * @param hri Region to look for.
407    * @return True if we processed a region in transition else false if region
408    * was not up in zk in transition.
409    * @throws InterruptedException
410    * @throws KeeperException
411    * @throws IOException
412    */
413   boolean processRegionInTransitionAndBlockUntilAssigned(final HRegionInfo hri)
414       throws InterruptedException, KeeperException, IOException {
415     boolean intransistion = processRegionInTransition(hri.getEncodedName(), hri);
416     if (!intransistion) return intransistion;
417     LOG.debug("Waiting on " + HRegionInfo.prettyPrint(hri.getEncodedName()));
418     while (!this.server.isStopped() &&
419       this.regionStates.isRegionInTransition(hri.getEncodedName())) {
420       // We put a timeout because we may have the region getting in just between the test
421       //  and the waitForUpdate
422       this.regionStates.waitForUpdate(100);
423     }
424     return intransistion;
425   }

427   /**
428    * Process failover of new master for region <code>encodedRegionName</code>
429    * up in zookeeper.
430    * @param encodedRegionName Region to process failover for.
431    * @param regionInfo If null we'll go get it from meta table.
432    * @return True if we processed <code>regionInfo</code> as a RIT.
433    * @throws KeeperException
434    * @throws IOException
435    */
436   boolean processRegionInTransition(final String encodedRegionName,
437       final HRegionInfo regionInfo) throws KeeperException, IOException {
438     // We need a lock here to ensure that we will not put the same region twice
439     // It has no reason to be a lock shared with the other operations.
440     // We can do the lock on the region only, instead of a global lock: what we want to ensure
441     // is that we don't have two threads working on the same region.
442     Lock lock = locker.acquireLock(encodedRegionName);
443     try {
444       Stat stat = new Stat();
445       byte [] data = ZKAssign.getDataAndWatch(watcher, encodedRegionName, stat);
446       if (data == null) return false;
447       RegionTransition rt;
448       try {
449         rt = RegionTransition.parseFrom(data);
450       } catch (DeserializationException e) {
451         LOG.warn("Failed parse znode data", e);
452         return false;
453       }
454       HRegionInfo hri = regionInfo;
455       if (hri == null) {
456         hri = regionStates.getRegionInfo(rt.getRegionName());
457         if (hri == null) return false;
458       }
459       processRegionsInTransition(rt, hri, stat.getVersion());
460       return true;
461     } finally {
462       lock.unlock();
463     }
464   }

466   /**
467    * This call is invoked only (1) master assign root and meta;
468    * (2) during failover mode startup, zk assignment node processing.
469    * The locker is set in the caller.
470    *
471    * It should be private but it is used by some test too.
472    */
473   void processRegionsInTransition(
474       final RegionTransition rt, final HRegionInfo regionInfo,
475       int expectedVersion) throws KeeperException {
476     EventType et = rt.getEventType();
477     // Get ServerName.  Could not be null.
478     ServerName sn = rt.getServerName();
479     String encodedRegionName = regionInfo.getEncodedName();
480     LOG.info("Processing region " + regionInfo.getRegionNameAsString() + " in state " + et);
481 
482     if (regionStates.isRegionInTransition(encodedRegionName)) {
483       // Just return
484       return;
485     }
486     switch (et) {
487     case M_ZK_REGION_CLOSING:
488       // If zk node of the region was updated by a live server skip this
489       // region and just add it into RIT.
490       if (!serverManager.isServerOnline(sn)) {
491         // If was not online, its closed now. Force to OFFLINE and this
492         // will get it reassigned if appropriate
493         forceOffline(regionInfo, rt);
494       } else {
495         // Just insert region into RIT.
496         // If this never updates the timeout will trigger new assignment
497         regionStates.updateRegionState(rt, RegionState.State.CLOSING);
498       }
499       break;
500 
501     case RS_ZK_REGION_CLOSED:
502     case RS_ZK_REGION_FAILED_OPEN:
503       // Region is closed, insert into RIT and handle it
504       addToRITandCallClose(regionInfo, RegionState.State.CLOSED, rt);
505       break;
506 
507     case M_ZK_REGION_OFFLINE:
508       // If zk node of the region was updated by a live server skip this
509       // region and just add it into RIT.
510       if (!serverManager.isServerOnline(sn)) {
511         // Region is offline, insert into RIT and handle it like a closed
512         addToRITandCallClose(regionInfo, RegionState.State.OFFLINE, rt);
513       } else {
514         // Just insert region into RIT.
515         // If this never updates the timeout will trigger new assignment
516         regionStates.updateRegionState(rt, RegionState.State.PENDING_OPEN);
517       }
518       break;
519 
520     case RS_ZK_REGION_OPENING:
521       regionStates.updateRegionState(rt, RegionState.State.OPENING);
522       if (regionInfo.isMetaTable() || !serverManager.isServerOnline(sn)) {
523         // If ROOT or .META. table is waiting for timeout monitor to assign
524         // it may take lot of time when the assignment.timeout.period is
525         // the default value which may be very long.  We will not be able
526         // to serve any request during this time.
527         // So we will assign the ROOT and .META. region immediately.
528         // For a user region, if the server is not online, it takes
529         // some time for timeout monitor to kick in.  We know the region
530         // won't open. So we will assign the opening
531         // region immediately too.
532         //
533         // Otherwise, just insert region into RIT. If the state never
534         // updates, the timeout will trigger new assignment
535         processOpeningState(regionInfo);
536       }
537       break;
538 
539     case RS_ZK_REGION_OPENED:
540       if (!serverManager.isServerOnline(sn)) {
541         forceOffline(regionInfo, rt);
542       } else {
543         // Region is opened, insert into RIT and handle it
544         regionStates.updateRegionState(rt, RegionState.State.OPEN);
545         new OpenedRegionHandler(server, this, regionInfo, sn, expectedVersion).process();
546       }
547       break;
548     case RS_ZK_REGION_SPLITTING:
549       LOG.debug("Processed region in state : " + et);
550       break;
551     case RS_ZK_REGION_SPLIT:
552       LOG.debug("Processed region in state : " + et);
553       break;
554     default:
555       throw new IllegalStateException("Received region in state :" + et + " is not valid");
556     }
557   }

601   /**
602    * Handles various states an unassigned node can be in.
603    * <p>
604    * Method is called when a state change is suspected for an unassigned node.
605    * <p>
606    * This deals with skipped transitions (we got a CLOSED but didn't see CLOSING
607    * yet).
608    * @param rt
609    * @param expectedVersion
610    */
611   private void handleRegion(final RegionTransition rt, int expectedVersion) {
612     if (rt == null) {
613       LOG.warn("Unexpected NULL input " + rt);
614       return;
615     }
616     final ServerName sn = rt.getServerName();
617     // Check if this is a special HBCK transition
618     if (sn.equals(HBCK_CODE_SERVERNAME)) {
619       handleHBCK(rt);
620       return;
621     }
622     final long createTime = rt.getCreateTime();
623     final byte[] regionName = rt.getRegionName();
624     String encodedName = HRegionInfo.encodeRegionName(regionName);
625     String prettyPrintedRegionName = HRegionInfo.prettyPrint(encodedName);
626     // Verify this is a known server
627     if (!serverManager.isServerOnline(sn)
628       && !ignoreStatesRSOffline.contains(rt.getEventType())) {
629       LOG.warn("Attempted to handle region transition for server but " +
630         "server is not online: " + prettyPrintedRegionName);
631       return;
632     }
633 
634     RegionState regionState =
635       regionStates.getRegionTransitionState(encodedName);
636     long startTime = System.currentTimeMillis();
637     if (LOG.isDebugEnabled()) {
638       boolean lateEvent = createTime < (startTime - 15000);
639       LOG.debug("Handling transition=" + rt.getEventType() +
640         ", server=" + sn + ", region=" +
641         (prettyPrintedRegionName == null ? "null" : prettyPrintedRegionName) +
642         (lateEvent ? ", which is more than 15 seconds late" : "") +
643         ", current state from region state map =" + regionState);
644     }
645     // We don't do anything for this event,
646     // so separate it out, no need to lock/unlock anything
647     if (rt.getEventType() == EventType.M_ZK_REGION_OFFLINE) {
648       return;
649     }
650 
651     // We need a lock on the region as we could update it
652     Lock lock = locker.acquireLock(encodedName);
653     try {
654       RegionState latestState =
655         regionStates.getRegionTransitionState(encodedName);
656       if ((regionState == null && latestState != null)
657           || (regionState != null && latestState == null)
658           || (regionState != null && latestState != null
659             && latestState.getState() != regionState.getState())) {
660         LOG.warn("Region state changed from " + regionState + " to "
661           + latestState + ", while acquiring lock");
662       }
663       long waitedTime = System.currentTimeMillis() - startTime;
664       if (waitedTime > 5000) {
665         LOG.warn("Took " + waitedTime + "ms to acquire the lock");
666       }
667       regionState = latestState;
668       switch (rt.getEventType()) {
669         case RS_ZK_REGION_SPLITTING:
670           if (!isInStateForSplitting(regionState)) break;
671           regionStates.updateRegionState(rt, RegionState.State.SPLITTING);
672           break;
673 
674         case RS_ZK_REGION_SPLIT:
675           // RegionState must be null, or SPLITTING or PENDING_CLOSE.
676           if (!isInStateForSplitting(regionState)) break;
677           // If null, add SPLITTING state before going to SPLIT
678           if (regionState == null) {
679             regionState = regionStates.updateRegionState(rt,
680               RegionState.State.SPLITTING);
681 
682             String message = "Received SPLIT for region " + prettyPrintedRegionName +
683               " from server " + sn;
684             // If still null, it means we cannot find it and it was already processed
685             if (regionState == null) {
686               LOG.warn(message + " but it doesn't exist anymore," +
687                   " probably already processed its split");
688               break;
689             }
690             LOG.info(message +
691                 " but region was not first in SPLITTING state; continuing");
692           }
693           // Check it has daughters.
694           byte [] payload = rt.getPayload();
695           List<HRegionInfo> daughters = null;
696           try {
697             daughters = HRegionInfo.parseDelimitedFrom(payload, 0, payload.length);
698           } catch (IOException e) {
699             LOG.error("Dropped split! Failed reading split payload for " +
700               prettyPrintedRegionName);
701             break;
702           }
703           assert daughters.size() == 2;
704           // Assert that we can get a serverinfo for this server.
705           if (!this.serverManager.isServerOnline(sn)) {
706             LOG.error("Dropped split! ServerName=" + sn + " unknown.");
707             break;
708           }
709           // Run handler to do the rest of the SPLIT handling.
710           this.executorService.submit(new SplitRegionHandler(server, this,
711             regionState.getRegion(), sn, daughters));
712           break;
713 
714         case M_ZK_REGION_CLOSING:
715           // Should see CLOSING after we have asked it to CLOSE or additional
716           // times after already being in state of CLOSING
717           if (regionState != null
718               && !regionState.isPendingCloseOrClosingOnServer(sn)) {
719             LOG.warn("Received CLOSING for region " + prettyPrintedRegionName
720               + " from server " + sn + " but region was in the state " + regionState
721               + " and not in expected PENDING_CLOSE or CLOSING states,"
722               + " or not on the expected server");
723             return;
724           }
725           // Transition to CLOSING (or update stamp if already CLOSING)
726           regionStates.updateRegionState(rt, RegionState.State.CLOSING);
727           break;
728 
729         case RS_ZK_REGION_CLOSED:
730           // Should see CLOSED after CLOSING but possible after PENDING_CLOSE
731           if (regionState != null
732               && !regionState.isPendingCloseOrClosingOnServer(sn)) {
733             LOG.warn("Received CLOSED for region " + prettyPrintedRegionName
734               + " from server " + sn + " but region was in the state " + regionState
735               + " and not in expected PENDING_CLOSE or CLOSING states,"
736               + " or not on the expected server");
737             return;
738           }
739           // Handle CLOSED by assigning elsewhere or stopping if a disable
740           // If we got here all is good.  Need to update RegionState -- else
741           // what follows will fail because not in expected state.
742           regionState = regionStates.updateRegionState(rt, RegionState.State.CLOSED);
743           if (regionState != null) {
744             removeClosedRegion(regionState.getRegion());
745             this.executorService.submit(new ClosedRegionHandler(server,
746               this, regionState.getRegion()));
747           }
748           break;
749 
750         case RS_ZK_REGION_FAILED_OPEN:
751           if (regionState != null
752               && !regionState.isPendingOpenOrOpeningOnServer(sn)) {
753             LOG.warn("Received FAILED_OPEN for region " + prettyPrintedRegionName
754               + " from server " + sn + " but region was in the state " + regionState
755               + " and not in expected PENDING_OPEN or OPENING states,"
756               + " or not on the expected server");
757             return;
758           }
759           // Handle this the same as if it were opened and then closed.
760           regionState = regionStates.updateRegionState(rt, RegionState.State.CLOSED);
761           // When there are more than one region server a new RS is selected as the
762           // destination and the same is updated in the regionplan. (HBASE-5546)
763           if (regionState != null) {
764             getRegionPlan(regionState.getRegion(), sn, true);
765             this.executorService.submit(new ClosedRegionHandler(server,
766               this, regionState.getRegion()));
767           }
768           break;
769 
770         case RS_ZK_REGION_OPENING:
771           // Should see OPENING after we have asked it to OPEN or additional
772           // times after already being in state of OPENING
773           if (regionState != null
774               && !regionState.isPendingOpenOrOpeningOnServer(sn)) {
775             LOG.warn("Received OPENING for region " + prettyPrintedRegionName
776               + " from server " + sn + " but region was in the state " + regionState
777               + " and not in expected PENDING_OPEN or OPENING states,"
778               + " or not on the expected server");
779             return;
780           }
781           // Transition to OPENING (or update stamp if already OPENING)
782           regionStates.updateRegionState(rt, RegionState.State.OPENING);
783           break;
784 
785         case RS_ZK_REGION_OPENED:
786           // Should see OPENED after OPENING but possible after PENDING_OPEN
787           if (regionState != null
788               && !regionState.isPendingOpenOrOpeningOnServer(sn)) {
789             LOG.warn("Received OPENED for region " + prettyPrintedRegionName
790               + " from server " + sn + " but region was in the state " + regionState
791               + " and not in expected PENDING_OPEN or OPENING states,"
792               + " or not on the expected server");
793             return;
794           }
795           // Handle OPENED by removing from transition and deleted zk node
796           regionState = regionStates.updateRegionState(rt, RegionState.State.OPEN);
797           if (regionState != null) {
798             this.executorService.submit(new OpenedRegionHandler(
799               server, this, regionState.getRegion(), sn, expectedVersion));
800           }
801           break;
802 
803         default:
804           throw new IllegalStateException("Received event is not valid.");
805       }
806     } finally {
807       lock.unlock();
808     }
809   }

921   // We  don't want to have two events on the same region managed simultaneously.
922   // For this reason, we need to wait if an event on the same region is currently in progress.
923   // So we track the region names of the events in progress, and we keep a waiting list.
924   private final Set<String> regionsInProgress = new HashSet<String>();
925   // In a LinkedHashMultimap, the put order is kept when we retrieve the collection back. We need
926   //  this as we want the events to be managed in the same order as we received them.
927   private final LinkedHashMultimap <String, RegionRunnable>
928       zkEventWorkerWaitingList = LinkedHashMultimap.create();
929 
930   /**
931    * A specific runnable that works only on a region.
932    */
933   private static interface RegionRunnable extends Runnable{
934     /**
935      * @return - the name of the region it works on.
936      */
937     public String getRegionName();
938   }

940   /**
941    * Submit a task, ensuring that there is only one task at a time that working on a given region.
942    * Order is respected.
943    */
944   protected void zkEventWorkersSubmit(final RegionRunnable regRunnable) {
945 
946     synchronized (regionsInProgress) {
947       // If we're there is already a task with this region, we add it to the
948       //  waiting list and return.
949       if (regionsInProgress.contains(regRunnable.getRegionName())) {
950         synchronized (zkEventWorkerWaitingList){
951           zkEventWorkerWaitingList.put(regRunnable.getRegionName(), regRunnable);
952         }
953         return;
954       }
955 
956       // No event in progress on this region => we can submit a new task immediately.
957       regionsInProgress.add(regRunnable.getRegionName());
958       zkEventWorkers.submit(new Runnable() {
959         @Override
960         public void run() {
961           try {
962             regRunnable.run();
963           } finally {
964             // now that we have finished, let's see if there is an event for the same region in the
965             //  waiting list. If it's the case, we can now submit it to the pool.
966             synchronized (regionsInProgress) {
967               regionsInProgress.remove(regRunnable.getRegionName());
968               synchronized (zkEventWorkerWaitingList) {
969                 java.util.Set<RegionRunnable> waiting = zkEventWorkerWaitingList.get(
970                     regRunnable.getRegionName());
971                 if (!waiting.isEmpty()) {
972                   // We want the first object only. The only way to get it is through an iterator.
973                   RegionRunnable toSubmit = waiting.iterator().next();
974                   zkEventWorkerWaitingList.remove(toSubmit.getRegionName(), toSubmit);
975                   zkEventWorkersSubmit(toSubmit);
976                 }
977               }
978             }
979           }
980         }
981       });
982     }
983   }

1148   /**

1150    * <code>sn</code> in common.
1151    * Call this method whenever a server checks in.  Doing so helps the case where
1152    * a new regionserver has joined the cluster and its been given 1k regions to
1153    * open.  If this method is tickled every time the region reports in a
1154    * successful open then the 1k-th region won't be timed out just because its
1155    * sitting behind the open of 999 other regions.  This method is NOT used
1156    * as part of bulk assign -- there we have a different mechanism for extending
1157    * the regions in transition timer (we turn it off temporarily -- because
1158    * there is no regionplan involved when bulk assigning.
1159    * @param sn
1160    */
1161   private void updateTimers(final ServerName sn) {
1162     if (sn == null) return;
1163 
1164     // This loop could be expensive.
1165     // First make a copy of current regionPlan rather than hold sync while
1166     // looping because holding sync can cause deadlock.  Its ok in this loop
1167     // if the Map we're going against is a little stale
1168     List<Map.Entry<String, RegionPlan>> rps;
1169     synchronized(this.regionPlans) {
1170       rps = new ArrayList<Map.Entry<String, RegionPlan>>(regionPlans.entrySet());
1171     }
1172 
1173     for (Map.Entry<String, RegionPlan> e : rps) {
1174       if (e.getValue() != null && e.getKey() != null && sn.equals(e.getValue().getDestination())) {
1175         RegionState regionState = regionStates.getRegionTransitionState(e.getKey());
1176         if (regionState != null) {
1177           regionState.updateTimestampToNow();
1178         }
1179       }
1180     }

1245     if (!setOfflineInZK && isDisabledorDisablingRegionInRIT(region)) {
1246       return;
1247     }
1248     if (this.serverManager.isClusterShutdown()) {
1249       LOG.info("Cluster shutdown is set; skipping assign of " +
1250         region.getRegionNameAsString());
1251       return;
1252     }
1253     String encodedName = region.getEncodedName();
1254     Lock lock = locker.acquireLock(encodedName);
1255     try {
1256       RegionState state = forceRegionStateToOffline(region, forceNewPlan);
1257       if (state != null) {
1258         assign(state, setOfflineInZK, forceNewPlan);
1259       }
1260     } finally {
1261       lock.unlock();
1262     }
1263   }

1265   /**
1266    * Bulk assign regions to <code>destination</code>.
1267    * @param destination
1268    * @param regions Regions to assign.
1269    * @return true if successful
1270    */
1271   boolean assign(final ServerName destination,
1272       final List<HRegionInfo> regions) {
1273     int regionCount = regions.size();
1274     if (regionCount == 0) {
1275       return true;
1276     }
1277     LOG.debug("Bulk assigning " + regionCount + " region(s) to " +
1278       destination.toString());
1279 
1280     Set<String> encodedNames = new HashSet<String>(regionCount);
1281     for (HRegionInfo region : regions) {
1282       encodedNames.add(region.getEncodedName());
1283     }
1284 
1285     List<HRegionInfo> failedToOpenRegions = new ArrayList<HRegionInfo>();
1286     Map<String, Lock> locks = locker.acquireLocks(encodedNames);
1287     try {
1288       AtomicInteger counter = new AtomicInteger(0);
1289       Map<String, Integer> offlineNodesVersions = new ConcurrentHashMap<String, Integer>();
1290       OfflineCallback cb = new OfflineCallback(
1291         watcher, destination, counter, offlineNodesVersions);
1292       Map<String, RegionPlan> plans = new HashMap<String, RegionPlan>(regions.size());
1293       List<RegionState> states = new ArrayList<RegionState>(regions.size());
1294       for (HRegionInfo region : regions) {
1295         String encodedRegionName = region.getEncodedName();
1296         RegionState state = forceRegionStateToOffline(region, true);
1297         if (state != null && asyncSetOfflineInZooKeeper(state, cb, destination)) {
1298           RegionPlan plan = new RegionPlan(region, state.getServerName(), destination);
1299           plans.put(encodedRegionName, plan);
1300           states.add(state);
1301         } else {
1302           LOG.warn("failed to force region state to offline or "
1303             + "failed to set it offline in ZK, will reassign later: " + region);
1304           failedToOpenRegions.add(region); // assign individually later
1305           Lock lock = locks.remove(encodedRegionName);
1306           lock.unlock();
1307         }
1308       }
1309 
1310       // Wait until all unassigned nodes have been put up and watchers set.
1311       int total = states.size();
1312       for (int oldCounter = 0; !server.isStopped();) {
1313         int count = counter.get();
1314         if (oldCounter != count) {
1315           LOG.info(destination.toString() + " unassigned znodes=" + count +
1316             " of total=" + total);
1317           oldCounter = count;
1318         }
1319         if (count >= total) break;
1320         Threads.sleep(5);
1321       }
1322 
1323       if (server.isStopped()) {
1324         return false;
1325       }
1326 
1327       // Add region plans, so we can updateTimers when one region is opened so
1328       // that unnecessary timeout on RIT is reduced.
1329       this.addPlans(plans);
1330 
1331       List<Pair<HRegionInfo, Integer>> regionOpenInfos =
1332         new ArrayList<Pair<HRegionInfo, Integer>>(states.size());
1333       for (RegionState state: states) {
1334         HRegionInfo region = state.getRegion();
1335         String encodedRegionName = region.getEncodedName();
1336         Integer nodeVersion = offlineNodesVersions.get(encodedRegionName);
1337         if (nodeVersion == null || nodeVersion.intValue() == -1) {
1338           LOG.warn("failed to offline in zookeeper: " + region);
1339           failedToOpenRegions.add(region); // assign individually later
1340           Lock lock = locks.remove(encodedRegionName);
1341           lock.unlock();
1342         } else {
1343           regionStates.updateRegionState(region,
1344             RegionState.State.PENDING_OPEN, destination);
1345           regionOpenInfos.add(new Pair<HRegionInfo, Integer>(
1346             region, nodeVersion));
1347         }
1348       }
1349 
1350       // Move on to open regions.
1351       try {
1352         // Send OPEN RPC. If it fails on a IOE or RemoteException, the
1353         // TimeoutMonitor will pick up the pieces.
1354         long maxWaitTime = System.currentTimeMillis() +
1355           this.server.getConfiguration().
1356             getLong("hbase.regionserver.rpc.startup.waittime", 60000);
1357         for (int i = 1; i <= maximumAttempts && !server.isStopped(); i++) {
1358           try {
1359             List<RegionOpeningState> regionOpeningStateList = serverManager
1360               .sendRegionOpen(destination, regionOpenInfos);
1361             if (regionOpeningStateList == null) {
1362               // Failed getting RPC connection to this server
1363               return false;
1364             }
1365             for (int k = 0, n = regionOpeningStateList.size(); k < n; k++) {
1366               RegionOpeningState openingState = regionOpeningStateList.get(k);
1367               if (openingState != RegionOpeningState.OPENED) {
1368                 HRegionInfo region = regionOpenInfos.get(k).getFirst();
1369                 if (openingState == RegionOpeningState.ALREADY_OPENED) {
1370                   processAlreadyOpenedRegion(region, destination);
1371                 } else if (openingState == RegionOpeningState.FAILED_OPENING) {
1372                   // Failed opening this region, reassign it later
1373                   failedToOpenRegions.add(region);
1374                 } else {
1375                   LOG.warn("THIS SHOULD NOT HAPPEN: unknown opening state "
1376                     + openingState + " in assigning region " + region);
1377                 }
1378               }
1379             }
1380             break;
1381           } catch (IOException e) {
1382             if (e instanceof RemoteException) {
1383               e = ((RemoteException)e).unwrapRemoteException();
1384             }
1385             if (e instanceof RegionServerStoppedException) {
1386               LOG.warn("The region server was shut down, ", e);
1387               // No need to retry, the region server is a goner.
1388               return false;
1389             } else if (e instanceof ServerNotRunningYetException) {
1390               long now = System.currentTimeMillis();
1391               if (now < maxWaitTime) {
1392                 LOG.debug("Server is not yet up; waiting up to " +
1393                   (maxWaitTime - now) + "ms", e);
1394                 Thread.sleep(100);
1395                 i--; // reset the try count
1396                 continue;
1397               }
1398             } else if (e instanceof java.net.SocketTimeoutException
1399                 && this.serverManager.isServerOnline(destination)) {
1400               // In case socket is timed out and the region server is still online,
1401               // the openRegion RPC could have been accepted by the server and
1402               // just the response didn't go through.  So we will retry to
1403               // open the region on the same server.
1404               if (LOG.isDebugEnabled()) {
1405                 LOG.debug("Bulk assigner openRegion() to " + destination
1406                   + " has timed out, but the regions might"
1407                   + " already be opened on it.", e);
1408               }
1409               continue;
1410             }
1411             throw e;
1412           }
1413         }
1414       } catch (IOException e) {
1415         // Can be a socket timeout, EOF, NoRouteToHost, etc
1416         LOG.info("Unable to communicate with the region server in order" +
1417           " to assign regions", e);
1418         return false;
1419       } catch (InterruptedException e) {
1420         throw new RuntimeException(e);
1421       }
1422     } finally {
1423       for (Lock lock : locks.values()) {
1424         lock.unlock();
1425       }
1426     }
1427 
1428     if (!failedToOpenRegions.isEmpty()) {
1429       for (HRegionInfo region : failedToOpenRegions) {
1430         invokeAssign(region);
1431       }
1432     }
1433     LOG.debug("Bulk assigning done for " + destination.toString());
1434     return true;
1435   }

1437   /**
1438    * Send CLOSE RPC if the server is online, otherwise, offline the region
1439    */
1440   private void unassign(final HRegionInfo region,
1441       final RegionState state, final int versionOfClosingNode,
1442       final ServerName dest, final boolean transitionInZK) {
1443     // Send CLOSE RPC
1444     ServerName server = state.getServerName();
1445     // ClosedRegionhandler can remove the server from this.regions
1446     if (!serverManager.isServerOnline(server)) {
1447       if (transitionInZK) {
1448         // delete the node. if no node exists need not bother.
1449         deleteClosingOrClosedNode(region);
1450       }
1451       regionOffline(region);
1452       return;
1453     }
1454 
1455     for (int i = 1; i <= this.maximumAttempts; i++) {
1456       try {
1457         if (serverManager.sendRegionClose(server, region,
1458           versionOfClosingNode, dest, transitionInZK)) {
1459           LOG.debug("Sent CLOSE to " + server + " for region " +
1460             region.getRegionNameAsString());
1461           return;
1462         }
1463         // This never happens. Currently regionserver close always return true.
1464         LOG.warn("Server " + server + " region CLOSE RPC returned false for " +
1465           region.getRegionNameAsString());
1466       } catch (Throwable t) {
1467         if (t instanceof RemoteException) {
1468           t = ((RemoteException)t).unwrapRemoteException();
1469         }
1470         if (t instanceof NotServingRegionException) {
1471           if (transitionInZK) {
1472             deleteClosingOrClosedNode(region);
1473           }
1474           regionOffline(region);
1475           return;
1476         } else if (t instanceof RegionAlreadyInTransitionException) {
1477           // RS is already processing this region, only need to update the timestamp
1478           LOG.debug("update " + state + " the timestamp.");
1479           state.updateTimestampToNow();
1480         }
1481         LOG.info("Server " + server + " returned " + t + " for "
1482           + region.getRegionNameAsString() + ", try=" + i
1483           + " of " + this.maximumAttempts, t);
1484         // Presume retry or server will expire.
1485       }
1486     }
1487   }

1489   /**
1490    * Set region to OFFLINE unless it is opening and forceNewPlan is false.
1491    */
1492   private RegionState forceRegionStateToOffline(
1493       final HRegionInfo region, final boolean forceNewPlan) {
1494     RegionState state = regionStates.getRegionState(region);
1495     if (state == null) {
1496       LOG.warn("Assigning a region not in region states: " + region);
1497       state = regionStates.createRegionState(region);
1498     } else {
1499       switch (state.getState()) {
1500       case OPEN:
1501       case OPENING:
1502       case PENDING_OPEN:
1503         if (!forceNewPlan) {
1504           LOG.debug("Attempting to assign region " +
1505             region + " but it is already in transition: " + state);
1506           return null;
1507         }
1508       case CLOSING:
1509       case PENDING_CLOSE:
1510         unassign(region, state, -1, null, false);
1511       case CLOSED:
1512         if (!state.isOffline()) {
1513           LOG.debug("Forcing OFFLINE; was=" + state);
1514           state = regionStates.updateRegionState(
1515             region, RegionState.State.OFFLINE);
1516         }
1517       case OFFLINE:
1518         break;
1519       default:
1520         LOG.error("Trying to assign region " + region
1521           + ", which is in state " + state);
1522         return null;
1523       }
1524     }
1525     return state;
1526   }

1528   /**
1529    * Caller must hold lock on the passed <code>state</code> object.
1530    * @param state
1531    * @param setOfflineInZK
1532    * @param forceNewPlan
1533    */
1534   private void assign(RegionState state,
1535       final boolean setOfflineInZK, final boolean forceNewPlan) {
1536     RegionState currentState = state;
1537     int versionOfOfflineNode = -1;
1538     RegionPlan plan = null;
1539     long maxRegionServerStartupWaitTime = -1;
1540     HRegionInfo region = state.getRegion();
1541     for (int i = 1; i <= maximumAttempts && !server.isStopped(); i++) {
1542       if (plan == null) { // Get a server for the region at first
1543         plan = getRegionPlan(region, forceNewPlan);
1544       }
1545       if (plan == null) {
1546         LOG.debug("Unable to determine a plan to assign " + region);
1547         this.timeoutMonitor.setAllRegionServersOffline(true);
1548         return; // Should get reassigned later when RIT times out.
1549       }
1550       if (setOfflineInZK && versionOfOfflineNode == -1) {
1551         // get the version of the znode after setting it to OFFLINE.
1552         // versionOfOfflineNode will be -1 if the znode was not set to OFFLINE
1553         versionOfOfflineNode = setOfflineInZooKeeper(currentState, plan.getDestination());
1554         if (versionOfOfflineNode != -1) {
1555           if (isDisabledorDisablingRegionInRIT(region)) {
1556             return;
1557           }
1558           // In case of assignment from EnableTableHandler table state is ENABLING. Any how
1559           // EnableTableHandler will set ENABLED after assigning all the table regions. If we
1560           // try to set to ENABLED directly then client API may think table is enabled.
1561           // When we have a case such as all the regions are added directly into .META. and we call
1562           // assignRegion then we need to make the table ENABLED. Hence in such case the table
1563           // will not be in ENABLING or ENABLED state.
1564           String tableName = region.getTableNameAsString();
1565           if (!zkTable.isEnablingTable(tableName) && !zkTable.isEnabledTable(tableName)) {
1566             LOG.debug("Setting table " + tableName + " to ENABLED state.");
1567             setEnabledTable(tableName);
1568           }
1569         }
1570       }
1571       if (setOfflineInZK && versionOfOfflineNode == -1) {
1572         return;
1573       }
1574       if (this.server.isStopped()) {
1575         LOG.debug("Server stopped; skipping assign of " + region);
1576         return;
1577       }
1578       try {
1579         LOG.info("Assigning region " + region.getRegionNameAsString() +
1580           " to " + plan.getDestination().toString());
1581         // Transition RegionState to PENDING_OPEN
1582         currentState = regionStates.updateRegionState(region,
1583           RegionState.State.PENDING_OPEN, plan.getDestination());
1584         // Send OPEN RPC. This can fail if the server on other end is is not up.
1585         // Pass the version that was obtained while setting the node to OFFLINE.
1586         RegionOpeningState regionOpenState = serverManager.sendRegionOpen(plan
1587             .getDestination(), region, versionOfOfflineNode);
1588         if (regionOpenState == RegionOpeningState.ALREADY_OPENED) {
1589           processAlreadyOpenedRegion(region, plan.getDestination());
1590         } else if (regionOpenState == RegionOpeningState.FAILED_OPENING) {
1591           // Failed opening this region
1592           throw new Exception("Get regionOpeningState=" + regionOpenState);
1593         }
1594         break;
1595       } catch (Throwable t) {
1596         if (t instanceof RemoteException) {
1597           t = ((RemoteException) t).unwrapRemoteException();
1598         }
1599         boolean regionAlreadyInTransitionException = false;
1600         boolean serverNotRunningYet = false;
1601         boolean socketTimedOut = false;
1602         if (t instanceof RegionAlreadyInTransitionException) {
1603           regionAlreadyInTransitionException = true;
1604           if (LOG.isDebugEnabled()) {
1605             LOG.debug("Failed assignment in: " + plan.getDestination() + " due to "
1606               + t.getMessage());
1607           }
1608         } else if (t instanceof ServerNotRunningYetException) {
1609           if (maxRegionServerStartupWaitTime < 0) {
1610             maxRegionServerStartupWaitTime = System.currentTimeMillis() +
1611               this.server.getConfiguration().
1612                 getLong("hbase.regionserver.rpc.startup.waittime", 60000);
1613           }
1614           try {
1615             long now = System.currentTimeMillis();
1616             if (now < maxRegionServerStartupWaitTime) {
1617               LOG.debug("Server is not yet up; waiting up to " +
1618                 (maxRegionServerStartupWaitTime - now) + "ms", t);
1619               serverNotRunningYet = true;
1620               Thread.sleep(100);
1621               i--; // reset the try count
1622             } else {
1623               LOG.debug("Server is not up for a while; try a new one", t);
1624             }
1625           } catch (InterruptedException ie) {
1626             LOG.warn("Failed to assign "
1627               + region.getRegionNameAsString() + " since interrupted", ie);
1628             Thread.currentThread().interrupt();
1629             return;
1630           }
1631         } else if (t instanceof java.net.SocketTimeoutException
1632             && this.serverManager.isServerOnline(plan.getDestination())) {
1633           // In case socket is timed out and the region server is still online,
1634           // the openRegion RPC could have been accepted by the server and
1635           // just the response didn't go through.  So we will retry to
1636           // open the region on the same server to avoid possible
1637           // double assignment.
1638           socketTimedOut = true;
1639           if (LOG.isDebugEnabled()) {
1640             LOG.debug("Call openRegion() to " + plan.getDestination()
1641               + " has timed out when trying to assign "
1642               + region.getRegionNameAsString()
1643               + ", but the region might already be opened on "
1644               + plan.getDestination() + ".", t);
1645           }
1646         }
1647 
1648         LOG.warn("Failed assignment of "
1649           + region.getRegionNameAsString()
1650           + " to "
1651           + plan.getDestination()
1652           + ", trying to assign "
1653           + (regionAlreadyInTransitionException || serverNotRunningYet || socketTimedOut
1654             ? "to the same region server because of RegionAlreadyInTransitionException"
1655               + "/ServerNotRunningYetException/SocketTimeoutException;"
1656               : "elsewhere instead; ")
1657           + "try=" + i + " of " + this.maximumAttempts, t);
1658 
1659         if (i == this.maximumAttempts) {
1660           // Don't reset the region state or get a new plan any more.
1661           // This is the last try.
1662           continue;
1663         }
1664 
1665         // If region opened on destination of present plan, reassigning to new
1666         // RS may cause double assignments. In case of RegionAlreadyInTransitionException
1667         // reassigning to same RS.
1668         RegionPlan newPlan = plan;
1669         if (!(regionAlreadyInTransitionException
1670             || serverNotRunningYet || socketTimedOut)) {
1671           // Force a new plan and reassign. Will return null if no servers.
1672           // The new plan could be the same as the existing plan since we don't
1673           // exclude the server of the original plan, which should not be
1674           // excluded since it could be the only server up now.
1675           newPlan = getRegionPlan(region, true);
1676         }
1677         if (newPlan == null) {
1678           this.timeoutMonitor.setAllRegionServersOffline(true);
1679           LOG.warn("Unable to find a viable location to assign region " +
1680             region.getRegionNameAsString());
1681           return;
1682         }
1683         if (plan != newPlan
1684             && !plan.getDestination().equals(newPlan.getDestination())) {
1685           // Clean out plan we failed execute and one that doesn't look like it'll
1686           // succeed anyways; we need a new plan!
1687           // Transition back to OFFLINE
1688           currentState = regionStates.updateRegionState(
1689             region, RegionState.State.OFFLINE);
1690           versionOfOfflineNode = -1;
1691           plan = newPlan;
1692         }
1693       }
1694     }
1695   }

1732   /**
1733    * Set region as OFFLINED up in zookeeper
1734    *
1735    * @param state
1736    * @return the version of the offline node if setting of the OFFLINE node was
1737    *         successful, -1 otherwise.
1738    */
1739   private int setOfflineInZooKeeper(final RegionState state, final ServerName destination) {
1740     if (!state.isClosed() && !state.isOffline()) {
1741       String msg = "Unexpected state : " + state + " .. Cannot transit it to OFFLINE.";
1742       this.server.abort(msg, new IllegalStateException(msg));
1743       return -1;
1744     }
1745     regionStates.updateRegionState(state.getRegion(),
1746       RegionState.State.OFFLINE);
1747     int versionOfOfflineNode = -1;
1748     try {
1749       // get the version after setting the znode to OFFLINE
1750       versionOfOfflineNode = ZKAssign.createOrForceNodeOffline(watcher,
1751         state.getRegion(), destination);
1752       if (versionOfOfflineNode == -1) {
1753         LOG.warn("Attempted to create/force node into OFFLINE state before "
1754             + "completing assignment but failed to do so for " + state);
1755         return -1;
1756       }
1757     } catch (KeeperException e) {
1758       server.abort("Unexpected ZK exception creating/setting node OFFLINE", e);
1759       return -1;
1760     }
1761     return versionOfOfflineNode;
1762   }

1774   /**
1775    * @param region the region to assign
1776    * @param serverToExclude Server to exclude (we know its bad). Pass null if
1777    * all servers are thought to be assignable.
1778    * @param forceNewPlan If true, then if an existing plan exists, a new plan
1779    * will be generated.
1780    * @return Plan for passed <code>region</code> (If none currently, it creates one or
1781    * if no servers to assign, it returns null).
1782    */
1783   private RegionPlan getRegionPlan(final HRegionInfo region,
1784       final ServerName serverToExclude, final boolean forceNewPlan) {
1785     // Pickup existing plan or make a new one
1786     final String encodedName = region.getEncodedName();
1787     final List<ServerName> destServers =
1788       serverManager.createDestinationServersList(serverToExclude);
1789 
1790     if (destServers.isEmpty()){
1791       LOG.warn("Can't move the region " + encodedName +
1792         ", there is no destination server available.");
1793       return null;
1794     }
1795 
1796     RegionPlan randomPlan = null;
1797     boolean newPlan = false;
1798     RegionPlan existingPlan = null;
1799 
1800     synchronized (this.regionPlans) {
1801       existingPlan = this.regionPlans.get(encodedName);
1802 
1803       if (existingPlan != null && existingPlan.getDestination() != null) {
1804         LOG.debug("Found an existing plan for " + region.getRegionNameAsString()
1805           + " destination server is " + existingPlan.getDestination());
1806       }
1807 
1808       if (forceNewPlan
1809           || existingPlan == null
1810           || existingPlan.getDestination() == null
1811           || !destServers.contains(existingPlan.getDestination())) {
1812         newPlan = true;
1813         randomPlan = new RegionPlan(region, null,
1814             balancer.randomAssignment(region, destServers));
1815         this.regionPlans.put(encodedName, randomPlan);
1816       }
1817     }
1818 
1819     if (newPlan) {
1820       LOG.debug("No previous transition plan was found (or we are ignoring " +
1821         "an existing plan) for " + region.getRegionNameAsString() +
1822         " so generated a random one; " + randomPlan + "; " +
1823         serverManager.countOfRegionServers() +
1824                " (online=" + serverManager.getOnlineServers().size() +
1825                ", available=" + destServers.size() + ") available servers");
1826         return randomPlan;
1827       }
1828     LOG.debug("Using pre-existing plan for region " +
1829       region.getRegionNameAsString() + "; plan=" + existingPlan);
1830     return existingPlan;
1831   }

1887    * presume the region being unassigned no longer exists (its been split out
1888    * of existence). TODO: What to do if split fails and is rolled back and
1889    * parent is revivified?
1890    * <p>
1891    * If a RegionPlan is already set, it will remain.
1892    *
1893    * @param region server to be unassigned
1894    * @param force if region should be closed even if already closing
1895    */
1896   public void unassign(HRegionInfo region, boolean force, ServerName dest) {
1897     // TODO: Method needs refactoring.  Ugly buried returns throughout.  Beware!
1898     LOG.debug("Starting unassignment of region " +
1899       region.getRegionNameAsString() + " (offlining)");
1900 
1901     String encodedName = region.getEncodedName();
1902     // Grab the state of this region and synchronize on it
1903     int versionOfClosingNode = -1;
1904     // We need a lock here as we're going to do a put later and we don't want multiple states
1905     //  creation
1906     ReentrantLock lock = locker.acquireLock(encodedName);
1907     RegionState state = regionStates.getRegionTransitionState(encodedName);
1908     try {
1909       if (state == null) {
1910         // Create the znode in CLOSING state
1911         try {
1912           state = regionStates.getRegionState(region);
1913           if (state == null || state.getServerName() == null) {
1914             // We don't know where the region is, offline it.
1915             // No need to send CLOSE RPC
1916             regionOffline(region);
1917             return;
1918           }
1919           versionOfClosingNode = ZKAssign.createNodeClosing(
1920             watcher, region, state.getServerName());
1921           if (versionOfClosingNode == -1) {
1922             LOG.debug("Attempting to unassign region " +
1923                 region.getRegionNameAsString() + " but ZK closing node "
1924                 + "can't be created.");
1925             return;
1926           }
1927         } catch (KeeperException e) {
1928           if (e instanceof NodeExistsException) {
1929             // Handle race between master initiated close and regionserver
1930             // orchestrated splitting. See if existing node is in a
1931             // SPLITTING or SPLIT state.  If so, the regionserver started
1932             // an op on node before we could get our CLOSING in.  Deal.
1933             NodeExistsException nee = (NodeExistsException)e;
1934             String path = nee.getPath();
1935             try {
1936               if (isSplitOrSplitting(path)) {
1937                 LOG.debug(path + " is SPLIT or SPLITTING; " +
1938                   "skipping unassign because region no longer exists -- its split");
1939                 return;
1940               }
1941             } catch (KeeperException.NoNodeException ke) {
1942               LOG.warn("Failed getData on SPLITTING/SPLIT at " + path +
1943                 "; presuming split and that the region to unassign, " +
1944                 encodedName + ", no longer exists -- confirm", ke);
1945               return;
1946             } catch (KeeperException ke) {
1947               LOG.error("Unexpected zk state", ke);
1948             } catch (DeserializationException de) {
1949               LOG.error("Failed parse", de);
1950             }
1951           }
1952           // If we get here, don't understand whats going on -- abort.
1953           server.abort("Unexpected ZK exception creating node CLOSING", e);
1954           return;
1955         }
1956         state = regionStates.updateRegionState(region, RegionState.State.PENDING_CLOSE);
1957       } else if (force && (state.isPendingClose() || state.isClosing())) {
1958         LOG.debug("Attempting to unassign region " + region.getRegionNameAsString() +
1959           " which is already " + state.getState()  +
1960           " but forcing to send a CLOSE RPC again ");
1961         state.updateTimestampToNow();
1962       } else {
1963         LOG.debug("Attempting to unassign region " +
1964           region.getRegionNameAsString() + " but it is " +
1965           "already in transition (" + state.getState() + ", force=" + force + ")");
1966         return;
1967       }
1968 
1969       unassign(region, state, versionOfClosingNode, dest, true);
1970     } finally {
1971       lock.unlock();
1972     }
1973   }

2541     private void actOnTimeOut(RegionState regionState) {
2542       HRegionInfo regionInfo = regionState.getRegion();
2543       LOG.info("Regions in transition timed out:  " + regionState);
2544       // Expired! Do a retry.
2545       switch (regionState.getState()) {
2546       case CLOSED:
2547         LOG.info("Region " + regionInfo.getEncodedName()
2548             + " has been CLOSED for too long, waiting on queued "
2549             + "ClosedRegionHandler to run or server shutdown");
2550         // Update our timestamp.
2551         regionState.updateTimestampToNow();
2552         break;
2553       case OFFLINE:
2554         LOG.info("Region has been OFFLINE for too long, " + "reassigning "
2555             + regionInfo.getRegionNameAsString() + " to a random server");
2556         invokeAssign(regionInfo);
2557         break;
2558       case PENDING_OPEN:
2559         LOG.info("Region has been PENDING_OPEN for too "
2560             + "long, reassigning region=" + regionInfo.getRegionNameAsString());
2561         invokeAssign(regionInfo);
2562         break;
2563       case OPENING:
2564         processOpeningState(regionInfo);
2565         break;
2566       case OPEN:
2567         LOG.error("Region has been OPEN for too long, " +
2568             "we don't know where region was opened so can't do anything");
2569         regionState.updateTimestampToNow();
2570         break;
2571 
2572       case PENDING_CLOSE:
2573         LOG.info("Region has been PENDING_CLOSE for too "
2574             + "long, running forced unassign again on region="
2575             + regionInfo.getRegionNameAsString());
2576         invokeUnassign(regionInfo);
2577         break;
2578       case CLOSING:
2579         LOG.info("Region has been CLOSING for too " +
2580           "long, this should eventually complete or the server will " +
2581           "expire, send RPC again");
2582         invokeUnassign(regionInfo);
2583         break;

2586       case SPLITTING:
