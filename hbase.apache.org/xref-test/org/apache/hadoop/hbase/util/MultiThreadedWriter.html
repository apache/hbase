38  /** Creates multiple threads that write key/values into the */
39  public class MultiThreadedWriter extends MultiThreadedAction {
40    private static final Log LOG = LogFactory.getLog(MultiThreadedWriter.class);
41  
42    private long minColumnsPerKey = 1;
43    private long maxColumnsPerKey = 10;
44    private Set<HBaseWriterThread> writers = new HashSet<HBaseWriterThread>();
45  
46    private boolean isMultiPut = false;
47  
48    /**
49     * A temporary place to keep track of inserted keys. This is written to by
50     * all writers and is drained on a separate thread that populates
51     * {@link #insertedUpToKey}, the maximum key in the contiguous range of keys
52     * being inserted. This queue is supposed to stay small.
53     */
54    private BlockingQueue<Long> insertedKeys =
55        new ArrayBlockingQueue<Long>(10000);
56  
57    /**
58     * This is the current key to be inserted by any thread. Each thread does an
59     * atomic get and increment operation and inserts the current value.
60     */
61    private AtomicLong nextKeyToInsert = new AtomicLong();
62  
63    /**
64     * The highest key in the contiguous range of keys .
65     */
66    private AtomicLong insertedUpToKey = new AtomicLong();
67  
68    /** The sorted set of keys NOT inserted by the writers */
69    private Set<Long> failedKeySet = new ConcurrentSkipListSet<Long>();
70  
71    /**
72     * The total size of the temporary inserted key set that have not yet lined
73     * up in a our contiguous sequence starting from startKey. Supposed to stay
74     * small.
75     */
76    private AtomicLong insertedKeyQueueSize = new AtomicLong();
77  
78    /** Enable this if used in conjunction with a concurrent reader. */
79    private boolean trackInsertedKeys;

96    @Override
97    public void start(long startKey, long endKey, int numThreads)
98        throws IOException {
99      super.start(startKey, endKey, numThreads);
100 
101     if (verbose) {
102       LOG.debug("Inserting keys [" + startKey + ", " + endKey + ")");
103     }
104 
105     nextKeyToInsert.set(startKey);
106     insertedUpToKey.set(startKey - 1);
107 
108     for (int i = 0; i < numThreads; ++i) {
109       HBaseWriterThread writer = new HBaseWriterThread(i);
110       writers.add(writer);
111     }
112 
113     if (trackInsertedKeys) {
114       new Thread(new InsertedKeysTracker()).start();
115       numThreadsWorking.incrementAndGet();
116     }
117 
118     startThreads(writers);
119   }

121   public static byte[] longToByteArrayKey(long rowKey) {
122     return LoadTestKVGenerator.md5PrefixedKey(rowKey).getBytes();
123   }

125   private class HBaseWriterThread extends Thread {
126     private final HTable table;
127 
128     private final Random random = new Random();
129     private final LoadTestKVGenerator dataGenerator = new LoadTestKVGenerator(
130         minDataSize, maxDataSize);
131 
132     public HBaseWriterThread(int writerId) throws IOException {
133       setName(getClass().getSimpleName() + "_" + writerId);
134       table = new HTable(conf, tableName);
135     }
136 
137     public void run() {
138       try {
139         long rowKey;
140         while ((rowKey = nextKeyToInsert.getAndIncrement()) < endKey) {
141           long numColumns = minColumnsPerKey + Math.abs(random.nextLong())
142               % (maxColumnsPerKey - minColumnsPerKey);
143           numKeys.addAndGet(1);
144           if (isMultiPut) {
145             multiPutInsertKey(rowKey, 0, numColumns);
146           } else {
147             for (long col = 0; col < numColumns; ++col) {
148               insert(rowKey, col);
149             }
150           }
151           if (trackInsertedKeys) {
152             insertedKeys.add(rowKey);
153           }
154         }
155       } finally {
156         try {
157           table.close();
158         } catch (IOException e) {
159           LOG.error("Error closing table", e);
160         }
161         numThreadsWorking.decrementAndGet();
162       }
163     }
164 
165     public void insert(long rowKey, long col) {
166       Put put = new Put(longToByteArrayKey(rowKey));
167       String colAsStr = String.valueOf(col);
168       put.add(columnFamily, Bytes.toBytes(colAsStr),
169           dataGenerator.generateRandomSizeValue(rowKey, colAsStr));
170       try {
171         long start = System.currentTimeMillis();
172         table.put(put);
173         numCols.addAndGet(1);
174         totalOpTimeMs.addAndGet(System.currentTimeMillis() - start);
175       } catch (IOException e) {
176         failedKeySet.add(rowKey);
177         LOG.error("Failed to insert: " + rowKey);
178         e.printStackTrace();
179       }
180     }
181 
182     public void multiPutInsertKey(long rowKey, long startCol, long endCol) {
183       if (verbose) {
184         LOG.debug("Preparing put for key = " + rowKey + ", cols = ["
185             + startCol + ", " + endCol + ")");
186       }
187 
188       if (startCol >= endCol) {
189         return;
190       }
191 
192       Put put = new Put(LoadTestKVGenerator.md5PrefixedKey(
193           rowKey).getBytes());
194       byte[] columnQualifier;
195       byte[] value;
196       for (long i = startCol; i < endCol; ++i) {
197         String qualStr = String.valueOf(i);
198         columnQualifier = qualStr.getBytes();
199         value = dataGenerator.generateRandomSizeValue(rowKey, qualStr);
200         put.add(columnFamily, columnQualifier, value);
201       }
202 
203       try {
204         long start = System.currentTimeMillis();
205         table.put(put);
206         numCols.addAndGet(endCol - startCol);
207         totalOpTimeMs.addAndGet(
208             System.currentTimeMillis() - start);
209       } catch (IOException e) {
210         failedKeySet.add(rowKey);
211         e.printStackTrace();
212       }
213     }
214   }

216   /**
217    * A thread that keeps track of the highest key in the contiguous range of
218    * inserted keys.
219    */
220   private class InsertedKeysTracker implements Runnable {
221 
222     @Override
223     public void run() {
224       Thread.currentThread().setName(getClass().getSimpleName());
225       try {
226         long expectedKey = startKey;
227         Queue<Long> sortedKeys = new PriorityQueue<Long>();
228         while (expectedKey < endKey) {
229           // Block until a new element is available.
230           Long k;
231           try {
232             k = insertedKeys.poll(1, TimeUnit.SECONDS);
233           } catch (InterruptedException e) {
234             LOG.info("Inserted key tracker thread interrupted", e);
235             break;
236           }
237           if (k == null) {
238             continue;
239           }
240           if (k == expectedKey) {
241             // Skip the "sorted key" queue and consume this key.
242             insertedUpToKey.set(k);
243             ++expectedKey;
244           } else {
245             sortedKeys.add(k);
246           }
247 
248           // See if we have a sequence of contiguous keys lined up.
249           while (!sortedKeys.isEmpty()
250               && ((k = sortedKeys.peek()) == expectedKey)) {
251             sortedKeys.poll();
252             insertedUpToKey.set(k);
253             ++expectedKey;
254           }
255 
256           insertedKeyQueueSize.set(insertedKeys.size() + sortedKeys.size());
257         }
258       } catch (Exception ex) {
259         LOG.error("Error in inserted key tracker", ex);
260       } finally {
261         numThreadsWorking.decrementAndGet();
262       }
263     }
264 
265   }
