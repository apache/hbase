40  /**
41   * A reference to the top or bottom half of a store file where 'bottom' is the first half
42   * of the file containing the keys that sort lowest and 'top' is the second half
43   * of the file with keys that sort greater than those of the bottom half.  The file referenced
44   * lives under a different region.  References are made at region split time.
45   *
46   * <p>References work with a special half store file type.  References know how
47   * to write out the reference format in the file system and are what is juggled
48   * when references are mixed in with direct store files.  The half store file
49   * type is used reading the referred to file.
50   *
51   * <p>References to store files located over in some other region look like
52   * this in the file system
53   * <code>1278437856009925445.3323223323</code>:
54   * i.e. an id followed by hash of the referenced region.
55   * Note, a region is itself not splittable if it has instances of store file
56   * references.  References are cleaned up by compactions.
57   */
58  @InterfaceAudience.Private
59  public class Reference implements Writable {
60    private byte [] splitkey;
61    private Range region;
62  
63    /**
64     * For split HStoreFiles, it specifies if the file covers the lower half or
65     * the upper half of the key range
66     */
67    static enum Range {
68      /** HStoreFile contains upper half of key range */
69      top,
70      /** HStoreFile contains lower half of key range */
71      bottom
72    }
73  
74    /**
75     * @param splitRow
76     * @return A {@link Reference} that points at top half of a an hfile
77     */
78    public static Reference createTopReference(final byte [] splitRow) {
79      return new Reference(splitRow, Range.top);
80    }
81  
82    /**
83     * @param splitRow
84     * @return A {@link Reference} that points at the bottom half of a an hfile
85     */
86    public static Reference createBottomReference(final byte [] splitRow) {
87      return new Reference(splitRow, Range.bottom);
88    }
89  
90    /**
91     * Constructor
92     * @param splitRow This is row we are splitting around.
93     * @param fr
94     */
95    Reference(final byte [] splitRow, final Range fr) {
96      this.splitkey = splitRow == null?  null: KeyValue.createFirstOnRow(splitRow).getKey();
97      this.region = fr;
98    }

154   public static boolean isTopFileRegion(final Range r) {
155     return r.equals(Range.top);
156   }

158   public Path write(final FileSystem fs, final Path p)
159   throws IOException {
160     FSDataOutputStream out = fs.create(p, false);
161     try {
162       out.write(toByteArray());
163     } finally {
164       out.close();
165     }
166     return p;
167   }

169   /**
170    * Read a Reference from FileSystem.
171    * @param fs
172    * @param p
173    * @return New Reference made from passed <code>p</code>
174    * @throws IOException
175    */
176   public static Reference read(final FileSystem fs, final Path p)
177   throws IOException {
178     InputStream in = fs.open(p);
179     try {
180       // I need to be able to move back in the stream if this is not a pb serialization so I can
181       // do the Writable decoding instead.
182       in = in.markSupported()? in: new BufferedInputStream(in);
183       int pblen = ProtobufUtil.lengthOfPBMagic();
184       in.mark(pblen);
185       byte [] pbuf = new byte[pblen];
186       int read = in.read(pbuf);
187       if (read != pblen) throw new IOException("read=" + read + ", wanted=" + pblen);
188       // WATCHOUT! Return in middle of function!!!
189       if (ProtobufUtil.isPBMagicPrefix(pbuf)) return convert(FSProtos.Reference.parseFrom(in));
190       // Else presume Writables.  Need to reset the stream since it didn't start w/ pb.
191       // We won't bother rewriting thie Reference as a pb since Reference is transitory.
192       in.reset();
193       Reference r = new Reference();
194       DataInputStream dis = new DataInputStream(in);
195       // Set in = dis so it gets the close below in the finally on our way out.
196       in = dis;
197       r.readFields(dis);
198       return r;
199     } finally {
200       in.close();
201     }
202   }

204   FSProtos.Reference convert() {
205     FSProtos.Reference.Builder builder = FSProtos.Reference.newBuilder();
206     builder.setRange(isTopFileRegion(getFileRegion())?
207       FSProtos.Reference.Range.TOP: FSProtos.Reference.Range.BOTTOM);
208     builder.setSplitkey(ByteString.copyFrom(getSplitKey()));
209     return builder.build();
210   }
211 
212   static Reference convert(final FSProtos.Reference r) {
213     Reference result = new Reference();
214     result.splitkey = r.getSplitkey().toByteArray();
215     result.region = r.getRange() == FSProtos.Reference.Range.TOP? Range.top: Range.bottom;
216     return result;
217   }

219   /**
220    * Use this instead of {@link #toByteArray()} when writing to a stream and you want to use
221    * the pb mergeDelimitedFrom (w/o the delimiter, pb reads to EOF which may not be what ou want).
222    * @return This instance serialized as a delimited protobuf w/ a magic pb prefix.
223    * @throws IOException
224    * @see {@link #toByteArray()}
225    */
226   byte [] toByteArray() throws IOException {
227     return ProtobufUtil.prependPBMagic(convert().toByteArray());
228   }
