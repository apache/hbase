42  /**
43   * Scanner class that contains the <code>.META.</code> table scanning logic
44   * and uses a Retryable scanner. Provided visitors will be called
45   * for each row.
46   *
47   * Although public visibility, this is not a public-facing API and may evolve in
48   * minor releases.
49   *
50   * <p> Note that during concurrent region splits, the scanner might not see
51   * META changes across rows (for parent and daughter entries) consistently.
52   * see HBASE-5986, and {@link BlockingMetaScannerVisitor} for details. </p>
53   */
54  @InterfaceAudience.Private
55  public class MetaScanner {

111    * Scans the meta table and calls a visitor on each RowResult. Uses a table
112    * name and a row name to locate meta regions. And it only scans at most
113    * <code>rowLimit</code> of rows.
114    *
115    * @param configuration HBase configuration.
116    * @param visitor Visitor object. Closes the visitor before returning.
117    * @param tableName User table name in meta table to start scan at.  Pass
118    * null if not interested in a particular table.
119    * @param row Name of the row at the user table. The scan will start from
120    * the region row where the row resides.
121    * @param rowLimit Max of processed rows. If it is less than 0, it
122    * will be set to default value <code>Integer.MAX_VALUE</code>.
123    * @param metaTableName Meta table to scan, root or meta.
124    * @throws IOException e
125    */
126   public static void metaScan(Configuration configuration,
127       final MetaScannerVisitor visitor, final byte[] tableName,
128       final byte[] row, final int rowLimit, final byte[] metaTableName)
129       throws IOException {
130     try {
131       HConnectionManager.execute(new HConnectable<Void>(configuration) {
132         @Override
133         public Void connect(HConnection connection) throws IOException {
134           metaScan(conf, connection, visitor, tableName, row, rowLimit,
135               metaTableName);
136           return null;
137         }
138       });
139     } finally {
140       visitor.close();
141     }

144   private static void metaScan(Configuration configuration, HConnection connection,
145       MetaScannerVisitor visitor, byte [] tableName, byte[] row,
146       int rowLimit, final byte [] metaTableName)
147   throws IOException {
148     int rowUpperLimit = rowLimit > 0 ? rowLimit: Integer.MAX_VALUE;
149 
150     // if row is not null, we want to use the startKey of the row's region as
151     // the startRow for the meta scan.
152     byte[] startRow;
153     if (row != null) {
154       // Scan starting at a particular row in a particular table
155       assert tableName != null;
156       byte[] searchRow =
157         HRegionInfo.createRegionName(tableName, row, HConstants.NINES,
158           false);
159       HTable metaTable = null;
160       try {
161         metaTable = new HTable(configuration, HConstants.META_TABLE_NAME);
162         Result startRowResult = metaTable.getRowOrBefore(searchRow,
163             HConstants.CATALOG_FAMILY);
164         if (startRowResult == null) {
165           throw new TableNotFoundException("Cannot find row in .META. for table: "
166               + Bytes.toString(tableName) + ", row=" + Bytes.toStringBinary(searchRow));
167         }
168         HRegionInfo regionInfo = getHRegionInfo(startRowResult);
169         if (regionInfo == null) {
170           throw new IOException("HRegionInfo was null or empty in Meta for " +
171             Bytes.toString(tableName) + ", row=" + Bytes.toStringBinary(searchRow));
172         }
173 
174         byte[] rowBefore = regionInfo.getStartKey();
175         startRow = HRegionInfo.createRegionName(tableName, rowBefore,
176             HConstants.ZEROES, false);
177       } finally {
178         if (metaTable != null) {
179           metaTable.close();
180         }
181       }
182     } else if (tableName == null || tableName.length == 0) {
183       // Full META scan
184       startRow = HConstants.EMPTY_START_ROW;
185     } else {
186       // Scan META for an entire table
187       startRow = HRegionInfo.createRegionName(
188           tableName, HConstants.EMPTY_START_ROW, HConstants.ZEROES, false);
189     }
190 
191     // Scan over each meta region
192     ScannerCallable callable;
193     int rows = Math.min(rowLimit, configuration.getInt(
194         HConstants.HBASE_META_SCANNER_CACHING,
195         HConstants.DEFAULT_HBASE_META_SCANNER_CACHING));
196     do {
197       final Scan scan = new Scan(startRow).addFamily(HConstants.CATALOG_FAMILY);
198       if (LOG.isDebugEnabled()) {
199         LOG.debug("Scanning " + Bytes.toString(metaTableName) +
200           " starting at row=" + Bytes.toStringBinary(startRow) + " for max=" +
201           rowUpperLimit + " rows using " + connection.toString());
202       }
203       callable = new ScannerCallable(connection, metaTableName, scan, null);
204       // Open scanner
205       callable.withRetries();
206 
207       int processedRows = 0;
208       try {
209         callable.setCaching(rows);
210         done: do {
211           if (processedRows >= rowUpperLimit) {
212             break;
213           }
214           //we have all the rows here
215           Result [] rrs = callable.withRetries();
216           if (rrs == null || rrs.length == 0 || rrs[0].size() == 0) {
217             break; //exit completely
218           }
219           for (Result rr : rrs) {
220             if (processedRows >= rowUpperLimit) {
221               break done;
222             }
223             if (!visitor.processRow(rr))
224               break done; //exit completely
225             processedRows++;
226           }
227           //here, we didn't break anywhere. Check if we have more rows
228         } while(true);
229         // Advance the startRow to the end key of the current region
230         startRow = callable.getHRegionInfo().getEndKey();
231       } finally {
232         // Close scanner
233         callable.setClose();
234         callable.withRetries();
235       }
236     } while (Bytes.compareTo(startRow, HConstants.LAST_ROW) != 0);
237   }

239   /**
240    * Returns HRegionInfo object from the column
241    * HConstants.CATALOG_FAMILY:HConstants.REGIONINFO_QUALIFIER of the catalog
242    * table Result.
243    * @param data a Result object from the catalog table scan
244    * @return HRegionInfo or null
245    */
246   public static HRegionInfo getHRegionInfo(Result data) {
247     byte [] bytes =
248       data.getValue(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER);
249     if (bytes == null) return null;
250     HRegionInfo info = HRegionInfo.parseFromOrNull(bytes);
251     if (LOG.isDebugEnabled()) {
252       LOG.debug("Current INFO from scan results = " + info);
253     }
254     return info;
255   }

268   /**
269    * Lists all of the regions currently in META.
270    * @param conf
271    * @param offlined True if we are to include offlined regions, false and we'll
272    * leave out offlined regions from returned list.
273    * @return List of all user-space regions.
274    * @throws IOException
275    */
276   public static List<HRegionInfo> listAllRegions(Configuration conf, final boolean offlined)
277   throws IOException {
278     final List<HRegionInfo> regions = new ArrayList<HRegionInfo>();
279     MetaScannerVisitor visitor = new BlockingMetaScannerVisitor(conf) {
280         @Override
281         public boolean processRowInternal(Result result) throws IOException {
282           if (result == null || result.isEmpty()) {
283             return true;
284           }
285 
286           HRegionInfo regionInfo = getHRegionInfo(result);
287           if (regionInfo == null) {
288             LOG.warn("Null REGIONINFO_QUALIFIER: " + result);
289             return true;
290           }
291 
292           // If region offline AND we are not to include offlined regions, return.
293           if (regionInfo.isOffline() && !offlined) return true;
294           regions.add(regionInfo);
295           return true;
296         }
297     };
298     metaScan(conf, visitor);
299     return regions;
300   }

302   /**
303    * Lists all of the table regions currently in META.
304    * @param conf
305    * @param offlined True if we are to include offlined regions, false and we'll
306    * leave out offlined regions from returned list.
307    * @return Map of all user-space regions to servers
308    * @throws IOException
309    */
310   public static NavigableMap<HRegionInfo, ServerName> allTableRegions(Configuration conf,
311       final byte [] tablename, final boolean offlined) throws IOException {
312     final NavigableMap<HRegionInfo, ServerName> regions =
313       new TreeMap<HRegionInfo, ServerName>();
314     MetaScannerVisitor visitor = new TableMetaScannerVisitor(conf, tablename) {
315       @Override
316       public boolean processRowInternal(Result rowResult) throws IOException {
317         HRegionInfo info = getHRegionInfo(rowResult);
318         ServerName serverName = HRegionInfo.getServerName(rowResult);
319 
320         if (!(info.isOffline() || info.isSplit())) {
321           regions.put(new UnmodifyableHRegionInfo(info), serverName);
322         }
323         return true;
324       }
325     };
326     metaScan(conf, visitor, tablename);
327     return regions;
328   }

352   /**
353    * A MetaScannerVisitor that provides a consistent view of the table's
354    * META entries during concurrent splits (see HBASE-5986 for details). This class
355    * does not guarantee ordered traversal of meta entries, and can block until the
356    * META entries for daughters are available during splits.
357    */
358   public static abstract class BlockingMetaScannerVisitor
359     extends MetaScannerVisitorBase {
360 
361     private static final int DEFAULT_BLOCKING_TIMEOUT = 10000;
362     private Configuration conf;
363     private TreeSet<byte[]> daughterRegions = new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR);
364     private int blockingTimeout;
365     private HTable metaTable;
366 
367     public BlockingMetaScannerVisitor(Configuration conf) {
368       this.conf = conf;
369       this.blockingTimeout = conf.getInt(HConstants.HBASE_CLIENT_OPERATION_TIMEOUT,
370           DEFAULT_BLOCKING_TIMEOUT);
371     }
372 
373     public abstract boolean processRowInternal(Result rowResult) throws IOException;
374 
375     @Override
376     public void close() throws IOException {
377       super.close();
378       if (metaTable != null) {
379         metaTable.close();
380         metaTable = null;
381       }
382     }
383 
384     public HTable getMetaTable() throws IOException {
385       if (metaTable == null) {
386         metaTable = new HTable(conf, HConstants.META_TABLE_NAME);
387       }
388       return metaTable;
389     }
390 
391     @Override
392     public boolean processRow(Result rowResult) throws IOException {
393       HRegionInfo info = getHRegionInfo(rowResult);
394       if (info == null) {
395         return true;
396       }
397 
398       if (daughterRegions.remove(info.getRegionName())) {
399         return true; //we have already processed this row
400       }
401 
402       if (info.isSplitParent()) {
403         /* we have found a parent region which was split. We have to ensure that it's daughters are
404          * seen by this scanner as well, so we block until they are added to the META table. Even
405          * though we are waiting for META entries, ACID semantics in HBase indicates that this
406          * scanner might not see the new rows. So we manually query the daughter rows */
407         PairOfSameType<HRegionInfo> daughters = HRegionInfo.getDaughterRegions(rowResult);
408         HRegionInfo splitA = daughters.getFirst();
409         HRegionInfo splitB = daughters.getSecond();
410 
411         HTable metaTable = getMetaTable();
412         long start = System.currentTimeMillis();
413         Result resultA = getRegionResultBlocking(metaTable, blockingTimeout,
414             splitA.getRegionName());
415         if (resultA != null) {
416           processRow(resultA);
417           daughterRegions.add(splitA.getRegionName());
418         } else {
419           throw new RegionOfflineException("Split daughter region " +
420               splitA.getRegionNameAsString() + " cannot be found in META.");
421         }
422         long rem = blockingTimeout - (System.currentTimeMillis() - start);
423 
424         Result resultB = getRegionResultBlocking(metaTable, rem,
425             splitB.getRegionName());
426         if (resultB != null) {
427           processRow(resultB);
428           daughterRegions.add(splitB.getRegionName());
429         } else {
430           throw new RegionOfflineException("Split daughter region " +
431               splitB.getRegionNameAsString() + " cannot be found in META.");
432         }
433       }
434 
435       return processRowInternal(rowResult);
436     }
437 
438     private Result getRegionResultBlocking(HTable metaTable, long timeout, byte[] regionName)
439         throws IOException {
440       if (LOG.isDebugEnabled()) {
441         LOG.debug("blocking until region is in META: " + Bytes.toStringBinary(regionName));
442       }
443       long start = System.currentTimeMillis();
444       while (System.currentTimeMillis() - start < timeout) {
445         Get get = new Get(regionName);
446         Result result = metaTable.get(get);
447         HRegionInfo info = getHRegionInfo(result);
448         if (info != null) {
449           return result;
450         }
451         try {
452           Thread.sleep(10);
453         } catch (InterruptedException ex) {
454           Thread.currentThread().interrupt();
455           break;
456         }
457       }
458       return null;
459     }
460   }

462   /**
463    * A MetaScannerVisitor for a table. Provides a consistent view of the table's
464    * META entries during concurrent splits (see HBASE-5986 for details). This class
465    * does not guarantee ordered traversal of meta entries, and can block until the
466    * META entries for daughters are available during splits.
467    */
468   public static abstract class TableMetaScannerVisitor extends BlockingMetaScannerVisitor {
469     private byte[] tableName;
470 
471     public TableMetaScannerVisitor(Configuration conf, byte[] tableName) {
472       super(conf);
473       this.tableName = tableName;
474     }
475 
476     @Override
477     public final boolean processRow(Result rowResult) throws IOException {
478       HRegionInfo info = getHRegionInfo(rowResult);
479       if (info == null) {
480         return true;
481       }
482       if (!(Bytes.equals(info.getTableName(), tableName))) {
483         return false;
484       }
485       return super.processRow(rowResult);
486     }
