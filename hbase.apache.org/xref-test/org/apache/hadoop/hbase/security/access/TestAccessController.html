179   private static AccessControlProtos.GrantRequest newGrantRequest(
180       String username, byte[] table, byte[] family, byte[] qualifier,
181       AccessControlProtos.Permission.Action... actions) {
182     AccessControlProtos.Permission.Builder permissionBuilder =
183         AccessControlProtos.Permission.newBuilder();
184     for (AccessControlProtos.Permission.Action a : actions) {
185       permissionBuilder.addAction(a);
186     }
187     if (table != null) {
188       permissionBuilder.setTable(ByteString.copyFrom(table));
189     }
190     if (family != null) {
191       permissionBuilder.setFamily(ByteString.copyFrom(family));
192     }
193     if (qualifier != null) {
194       permissionBuilder.setQualifier(ByteString.copyFrom(qualifier));
195     }
196 
197     return AccessControlProtos.GrantRequest.newBuilder()
198         .setPermission(
199             AccessControlProtos.UserPermission.newBuilder()
200                 .setUser(ByteString.copyFromUtf8(username))
201                 .setPermission(permissionBuilder.build())
202         ).build();
203   }

205   public void verifyAllowed(User user, PrivilegedExceptionAction... actions) throws Exception {
206     for (PrivilegedExceptionAction action : actions) {
207       try {
208         user.runAs(action);
209       } catch (AccessDeniedException ade) {
210         fail("Expected action to pass for user '" + user.getShortName() + "' but was denied");
211       }
212     }
213   }
214 
215   public void verifyAllowed(PrivilegedExceptionAction action, User... users) throws Exception {
216     for (User user : users) {
217       verifyAllowed(user, action);
218     }
219   }
220 
221   public void verifyDenied(User user, PrivilegedExceptionAction... actions) throws Exception {
222     for (PrivilegedExceptionAction action : actions) {
223       try {
224         user.runAs(action);
225         fail("Expected AccessDeniedException for user '" + user.getShortName() + "'");
226       } catch (RetriesExhaustedWithDetailsException e) {
227         // in case of batch operations, and put, the client assembles a
228         // RetriesExhaustedWithDetailsException instead of throwing an
229         // AccessDeniedException
230         boolean isAccessDeniedException = false;
231         for (Throwable ex : e.getCauses()) {
232           if (ex instanceof ServiceException) {
233             ServiceException se = (ServiceException)ex;
234             if (se.getCause() != null && se.getCause() instanceof AccessDeniedException) {
235               isAccessDeniedException = true;
236               break;
237             }
238           } else if (ex instanceof AccessDeniedException) {
239             isAccessDeniedException = true;
240             break;
241           }
242         }
243         if (!isAccessDeniedException) {
244           fail("Not receiving AccessDeniedException for user '" + user.getShortName() + "'");
245         }
246       } catch (AccessDeniedException ade) {
247         // expected result
248       }
249     }
250   }
251 
252   public void verifyDenied(PrivilegedExceptionAction action, User... users) throws Exception {
253     for (User user : users) {
254       verifyDenied(user, action);
255     }
256   }
