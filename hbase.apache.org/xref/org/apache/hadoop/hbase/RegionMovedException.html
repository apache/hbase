18  package org.apache.hadoop.hbase;
19  
20  import org.apache.commons.logging.Log;
21  import org.apache.commons.logging.LogFactory;
22  import org.apache.hadoop.classification.InterfaceAudience;
23  import org.apache.hadoop.classification.InterfaceStability;
24  import org.apache.hadoop.ipc.RemoteException;
25  
26  /**
27   * Subclass if the server knows the region is now on another server.
28   * This allows the client to call the new region server without calling the master.
29   */
30  @InterfaceAudience.Private
31  @InterfaceStability.Evolving
32  public class RegionMovedException extends NotServingRegionException {
33    private static final Log LOG = LogFactory.getLog(RegionMovedException.class);
34    private static final long serialVersionUID = -7232903522310558397L;
35  
36    private final String hostname;
37    private final int port;
38  
39    private static final String HOST_FIELD = "hostname=";
40    private static final String PORT_FIELD = "port=";
41  
42    public RegionMovedException(final String hostname, final int port) {
43      super();
44      this.hostname = hostname;
45      this.port = port;
46    }
47  
48    public String getHostname() {
49      return hostname;
50    }
51  
52    public int getPort() {
53      return port;
54    }
55  
56    /**
57     * For hadoop.ipc internal call. Do NOT use.
58     * We have to parse the hostname to recreate the exception.
59     * The input is the one generated by {@link #getMessage()}
60     */
61    public RegionMovedException(String s) {
62      int posHostname = s.indexOf(HOST_FIELD) + HOST_FIELD.length();
63      int posPort = s.indexOf(PORT_FIELD) + PORT_FIELD.length();
64  
65      String tmpHostname = null;
66      int tmpPort = -1;
67      try {
68        tmpHostname = s.substring(posHostname, s.indexOf(' ', posHostname));
69        tmpPort = Integer.parseInt(s.substring(posPort, s.indexOf('.', posPort)));
70      } catch (Exception ignored) {
71        LOG.warn("Can't parse the hostname and the port from this string: " + s + ", "+
72          "Continuing");
73      }
74  
75      hostname = tmpHostname;
76      port = tmpPort;
77    }
78  
79    @Override
80    public String getMessage() {
81      return "Region moved to: " + HOST_FIELD + hostname + " " + PORT_FIELD + port + ".";
82    }
83  
84    /**
85     * Look for a RegionMovedException in the exception:
86     *  - hadoop.ipc wrapped exceptions
87     *  - nested exceptions
88     * Returns null if we didn't find the exception or if it was not readable.
89     */
90    public static RegionMovedException find(Object exception) {
91      if (exception == null || !(exception instanceof Throwable)){
92        return null;
93      }
94  
95      Throwable cur = (Throwable)exception;
96      RegionMovedException res = null;
97  
98      while (res == null && cur != null) {
99        if (cur instanceof RegionMovedException) {
100         res = (RegionMovedException) cur;
101       } else {
102         if (cur instanceof RemoteException) {
103           RemoteException re = (RemoteException) cur;
104           Exception e = re.unwrapRemoteException(RegionMovedException.class);
105           if (e == null){
106             e = re.unwrapRemoteException();
107           }
108           // unwrapRemoteException can return the exception given as a parameter when it cannot
109           //  unwrap it. In this case, there is no need to look further
110           // noinspection ObjectEquality
111           if (e != re){
112             res =  find(e);
113           }
114         }
115         cur = cur.getCause();
116       }
117     }
118 
119     if (res != null && (res.getPort() < 0 || res.getHostname() == null)){
120       // We failed to parse the exception. Let's act as we don't find the exception.
121       return null;
122     } else {
123       return res;
124     }
125   }
126 }
