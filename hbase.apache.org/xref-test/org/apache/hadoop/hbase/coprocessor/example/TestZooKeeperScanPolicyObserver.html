46  @Category(MediumTests.class)
47  public class TestZooKeeperScanPolicyObserver {
48    private static final Log LOG = LogFactory.getLog(TestZooKeeperScanPolicyObserver.class);
49    private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
50    private static final byte[] F = Bytes.toBytes("fam");
51    private static final byte[] Q = Bytes.toBytes("qual");
52    private static final byte[] R = Bytes.toBytes("row");
53  
54    @BeforeClass
55    public static void setUpBeforeClass() throws Exception {
56      // Test we can first start the ZK cluster by itself
57      Configuration conf = TEST_UTIL.getConfiguration();
58      conf.setStrings(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY,
59          ZooKeeperScanPolicyObserver.class.getName());
60      TEST_UTIL.startMiniZKCluster();
61      TEST_UTIL.startMiniCluster();
62    }
63  
64    @AfterClass
65    public static void tearDownAfterClass() throws Exception {
66      TEST_UTIL.shutdownMiniCluster();
67    }
68  
69    @Test
70    public void testScanPolicyObserver() throws Exception {
71      byte[] tableName = Bytes.toBytes("testScanPolicyObserver");
72      HTableDescriptor desc = new HTableDescriptor(tableName);
73      HColumnDescriptor hcd = new HColumnDescriptor(F)
74      .setMaxVersions(10)
75      .setTimeToLive(1);
76      desc.addFamily(hcd);
77      TEST_UTIL.getHBaseAdmin().createTable(desc);
78      HTable t = new HTable(new Configuration(TEST_UTIL.getConfiguration()), tableName);
79      long now = EnvironmentEdgeManager.currentTimeMillis();
80  
81      ZooKeeperWatcher zkw = HConnectionManager.getConnection(TEST_UTIL.getConfiguration())
82          .getZooKeeperWatcher();
83      ZooKeeper zk = zkw.getRecoverableZooKeeper().getZooKeeper();
84      ZKUtil.createWithParents(zkw, ZooKeeperScanPolicyObserver.node);
85      // let's say test last backup was 1h ago
86      // using plain ZK here, because RecoverableZooKeeper add extra encoding to the data
87      zk.setData(ZooKeeperScanPolicyObserver.node, Bytes.toBytes(now - 3600*1000), -1);
88  
89      LOG.debug("Set time: "+Bytes.toLong(Bytes.toBytes(now - 3600*1000)));
90  
91      // sleep for 1s to give the ZK change a chance to reach the watcher in the observer.
92      // TODO: Better to wait for the data to be propagated
93      Thread.sleep(1000);
94  
95      long ts = now - 2000;
96      Put p = new Put(R);
97      p.add(F, Q, ts, Q);
98      t.put(p);
99      p = new Put(R);
100     p.add(F, Q, ts+1, Q);
101     t.put(p);
102 
103     // these two should be expired but for the override
104     // (their ts was 2s in the past)
105     Get g = new Get(R);
106     g.setMaxVersions(10);
107     Result r = t.get(g);
108     // still there?
109     assertEquals(2, r.size());
110 
111     TEST_UTIL.flush(tableName);
112     TEST_UTIL.compact(tableName, true);
113 
114     g = new Get(R);
115     g.setMaxVersions(10);
116     r = t.get(g);
117     // still there?
118     assertEquals(2, r.size());
119     zk.setData(ZooKeeperScanPolicyObserver.node, Bytes.toBytes(now), -1);
120     LOG.debug("Set time: "+now);
121 
122     TEST_UTIL.compact(tableName, true);
123 
124     g = new Get(R);
125     g.setMaxVersions(10);
126     r = t.get(g);
127     // should be gone now
128     assertEquals(0, r.size());
129     t.close();
130   }
