33  /**
34   * A region server that will OOME.
35   * Everytime {@link #put(regionName, Put)} is called, we add
36   * keep around a reference to the batch.  Use this class to test OOME extremes.
37   * Needs to be started manually as in
38   * <code>${HBASE_HOME}/bin/hbase ./bin/hbase org.apache.hadoop.hbase.OOMERegionServer start</code>.
39   */
40  public class OOMERegionServer extends HRegionServer {
41    private List<Put> retainer = new ArrayList<Put>();
42  
43    public OOMERegionServer(HBaseConfiguration conf) throws IOException, InterruptedException {
44      super(conf);
45    }
46  
47    public void put(byte [] regionName, Put put)
48    throws IOException {
49      try {
50        MutateRequest request =
51          RequestConverter.buildMutateRequest(regionName, put);
52        super.mutate(null, request);
53        for (int i = 0; i < 30; i++) {
54          // Add the batch update 30 times to bring on the OOME faster.
55          this.retainer.add(put);
56        }
57      } catch (ServiceException se) {
58        throw ProtobufUtil.getRemoteException(se);
59      }
60    }
61  
62    public static void main(String[] args) {
63      new HRegionServerCommandLine(OOMERegionServer.class).doMain(args);
64    }
65  }
