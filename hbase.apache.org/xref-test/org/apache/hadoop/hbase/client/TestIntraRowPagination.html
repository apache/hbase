34  /**
35   * Test scan/get offset and limit settings within one row through HRegion API.
36   */
37  @Category(SmallTests.class)
38  public class TestIntraRowPagination {
39  
40    private static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
41  
42    /**
43     * Test from client side for scan with maxResultPerCF set
44     *
45     * @throws Exception
46     */
47    @Test
48    public void testScanLimitAndOffset() throws Exception {
49      //byte [] TABLE = HTestConst.DEFAULT_TABLE_BYTES;
50      byte [][] ROWS = HTestConst.makeNAscii(HTestConst.DEFAULT_ROW_BYTES, 2);
51      byte [][] FAMILIES = HTestConst.makeNAscii(HTestConst.DEFAULT_CF_BYTES, 3);
52      byte [][] QUALIFIERS = HTestConst.makeNAscii(HTestConst.DEFAULT_QUALIFIER_BYTES, 10);
53  
54      HTableDescriptor htd = new HTableDescriptor(HTestConst.DEFAULT_TABLE_BYTES);
55      HRegionInfo info = new HRegionInfo(HTestConst.DEFAULT_TABLE_BYTES, null, null, false);
56      for (byte[] family : FAMILIES) {
57        HColumnDescriptor hcd = new HColumnDescriptor(family);
58        htd.addFamily(hcd);
59      }
60      HRegion region =
61          HRegion.createHRegion(info, TEST_UTIL.getDataTestDir(), TEST_UTIL.getConfiguration(), htd);
62      try {
63        Put put;
64        Scan scan;
65        Result result;
66        boolean toLog = true;
67  
68        List<KeyValue> kvListExp = new ArrayList<KeyValue>();
69  
70        int storeOffset = 1;
71        int storeLimit = 3;
72        for (int r = 0; r < ROWS.length; r++) {
73          put = new Put(ROWS[r]);
74          for (int c = 0; c < FAMILIES.length; c++) {
75            for (int q = 0; q < QUALIFIERS.length; q++) {
76              KeyValue kv = new KeyValue(ROWS[r], FAMILIES[c], QUALIFIERS[q], 1,
77                  HTestConst.DEFAULT_VALUE_BYTES);
78              put.add(kv);
79              if (storeOffset <= q && q < storeOffset + storeLimit) {
80                kvListExp.add(kv);
81              }
82            }
83          }
84          region.put(put);
85        }
86  
87        scan = new Scan();
88        scan.setRowOffsetPerColumnFamily(storeOffset);
89        scan.setMaxResultsPerColumnFamily(storeLimit);
90        RegionScanner scanner = region.getScanner(scan);
91        List<KeyValue> kvListScan = new ArrayList<KeyValue>();
92        List<KeyValue> results = new ArrayList<KeyValue>();
93        while (scanner.next(results) || !results.isEmpty()) {
94          kvListScan.addAll(results);
95          results.clear();
96        }
97        result = new Result(kvListScan);
98        TestScannersFromClientSide.verifyResult(result, kvListExp, toLog,
99            "Testing scan with storeOffset and storeLimit");
100     } finally {
101       region.close();
102     }
103   }
