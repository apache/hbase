22  import java.util.List;
23  import java.util.TreeSet;

26  import org.apache.hadoop.classification.InterfaceStability;
27  import org.apache.hadoop.hbase.KeyValue;
28  import org.apache.hadoop.hbase.client.Put;
29  import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
30  import org.apache.hadoop.mapreduce.Reducer;
31  import org.apache.hadoop.util.StringUtils;

33  /**
34   * Emits sorted Puts.
35   * Reads in all Puts from passed Iterator, sorts them, then emits
36   * Puts in sorted order.  If lots of columns per row, it will use lots of
37   * memory sorting.
38   * @see HFileOutputFormat
39   * @see KeyValueSortReducer
40   */
41  @InterfaceAudience.Public
42  @InterfaceStability.Stable
43  public class PutSortReducer extends
44      Reducer<ImmutableBytesWritable, Put, ImmutableBytesWritable, KeyValue> {
45    
46    @Override
47    protected void reduce(
48        ImmutableBytesWritable row,
49        java.lang.Iterable<Put> puts,
50        Reducer<ImmutableBytesWritable, Put,
51                ImmutableBytesWritable, KeyValue>.Context context)
52        throws java.io.IOException, InterruptedException
53    {
54      // although reduce() is called per-row, handle pathological case
55      long threshold = context.getConfiguration().getLong(
56          "putsortreducer.row.threshold", 2L * (1<<30));
57      Iterator<Put> iter = puts.iterator();
58      while (iter.hasNext()) {
59        TreeSet<KeyValue> map = new TreeSet<KeyValue>(KeyValue.COMPARATOR);
60        long curSize = 0;
61        // stop at the end or the RAM threshold
62        while (iter.hasNext() && curSize < threshold) {
63          Put p = iter.next();
64          for (List<KeyValue> kvs : p.getFamilyMap().values()) {
65            for (KeyValue kv : kvs) {
66              map.add(kv);
67              curSize += kv.getLength();
68            }
69          }
70        }
71        context.setStatus("Read " + map.size() + " entries of " + map.getClass()
72            + "(" + StringUtils.humanReadableInt(curSize) + ")");
73        int index = 0;
74        for (KeyValue kv : map) {
75          context.write(row, kv);
76          if (index > 0 && index % 100 == 0)
77            context.setStatus("Wrote " + index);
78        }
79  
80        // if we have more entries to process
81        if (iter.hasNext()) {
82          // force flush because we cannot guarantee intra-row sorted order
83          context.write(null, null);
84        }
85      }
86    }
87  }
