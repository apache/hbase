18  package org.apache.hadoop.hbase.client;
19  
20  import java.io.IOException;
21  import java.util.ArrayList;
22  import java.util.Arrays;
23  import java.util.Collections;
24  import java.util.List;
25  
26  import org.apache.hadoop.classification.InterfaceAudience;
27  import org.apache.hadoop.classification.InterfaceStability;
28  import org.apache.hadoop.hbase.HConstants;
29  import org.apache.hadoop.hbase.util.Bytes;
30  
31  /**
32   * Performs multiple mutations atomically on a single row.
33   * Currently {@link Put} and {@link Delete} are supported.
34   *
35   * The mutations are performed in the order in which they
36   * were added.
37   */
38  @InterfaceAudience.Public
39  @InterfaceStability.Evolving
40  public class RowMutations implements Row {
41    private List<Mutation> mutations = new ArrayList<Mutation>();
42    private byte [] row;
43    private static final byte VERSION = (byte)0;
44  
45    /** Constructor for Writable. DO NOT USE */
46    public RowMutations() {}
47  
48    /**
49     * Create an atomic mutation for the specified row.
50     * @param row row key
51     */
52    public RowMutations(byte [] row) {
53      if(row == null || row.length > HConstants.MAX_ROW_LENGTH) {
54        throw new IllegalArgumentException("Row key is invalid");
55      }
56      this.row = Arrays.copyOf(row, row.length);
57    }
58  
59    /**
60     * Add a {@link Put} operation to the list of mutations
61     * @param p The {@link Put} to add
62     * @throws IOException
63     */
64    public void add(Put p) throws IOException {
65      internalAdd(p);
66    }
67  
68    /**
69     * Add a {@link Delete} operation to the list of mutations
70     * @param d The {@link Delete} to add
71     * @throws IOException
72     */
73    public void add(Delete d) throws IOException {
74      internalAdd(d);
75    }
76  
77    private void internalAdd(Mutation m) throws IOException {
78      int res = Bytes.compareTo(this.row, m.getRow());
79      if(res != 0) {
80        throw new IOException("The row in the recently added Put/Delete " +
81            Bytes.toStringBinary(m.getRow()) + " doesn't match the original one " +
82            Bytes.toStringBinary(this.row));
83      }
84      mutations.add(m);
85    }
86  
87    @Override
88    public int compareTo(Row i) {
89      return Bytes.compareTo(this.getRow(), i.getRow());
90    }
91  
92    @Override
93    public byte[] getRow() {
94      return row;
95    }
96  
97    /**
98     * @return An unmodifiable list of the current mutations.
99     */
100   public List<Mutation> getMutations() {
101     return Collections.unmodifiableList(mutations);
102   }
103 }
