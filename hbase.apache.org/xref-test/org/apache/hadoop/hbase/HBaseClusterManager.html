31  /**
32   * A default cluster manager for HBase. Uses SSH, and hbase shell scripts
33   * to manage the cluster. Assumes Unix-like commands are available like 'ps',
34   * 'kill', etc. Also assumes the user running the test has enough "power" to start & stop
35   * servers on the remote machines (for example, the test user could be the same user as the
36   * user the daemon isrunning as)
37   */
38  @InterfaceAudience.Private
39  public class HBaseClusterManager extends ClusterManager {

41    /**
42     * Executes commands over SSH
43     */
44    static class RemoteShell extends Shell.ShellCommandExecutor {
45  
46      private String hostname;
47  
48      private String sshCmd = "/usr/bin/ssh";
49      private String sshOptions = System.getenv("HBASE_SSH_OPTS"); //from conf/hbase-env.sh
50  
51      public RemoteShell(String hostname, String[] execString, File dir, Map<String, String> env,
52          long timeout) {
53        super(execString, dir, env, timeout);
54        this.hostname = hostname;
55      }
56  
57      public RemoteShell(String hostname, String[] execString, File dir, Map<String, String> env) {
58        super(execString, dir, env);
59        this.hostname = hostname;
60      }
61  
62      public RemoteShell(String hostname, String[] execString, File dir) {
63        super(execString, dir);
64        this.hostname = hostname;
65      }
66  
67      public RemoteShell(String hostname, String[] execString) {
68        super(execString);
69        this.hostname = hostname;
70      }
71  
72      @Override
73      public String[] getExecString() {
74        return new String[] {
75            "bash", "-c",
76            StringUtils.join(new String[] { sshCmd,
77                sshOptions == null ? "" : sshOptions,
78                hostname,
79                "\"" + StringUtils.join(super.getExecString(), " ") + "\""
80            }, " ")};
81      }
82  
83      @Override
84      public void execute() throws IOException {
85        super.execute();
86      }
87  
88      public void setSshCmd(String sshCmd) {
89        this.sshCmd = sshCmd;
90      }
91  
92      public void setSshOptions(String sshOptions) {
93        this.sshOptions = sshOptions;
94      }
95  
96      public String getSshCmd() {
97        return sshCmd;
98      }
99  
100     public String getSshOptions() {
101       return sshOptions;
102     }

105   /**
106    * Provides command strings for services to be executed by Shell. CommandProviders are
107    * pluggable, and different deployments(windows, bigtop, etc) can be managed by
108    * plugging-in custom CommandProvider's or ClusterManager's.
109    */
110   static abstract class CommandProvider {
111 
112     enum Operation {
113       START, STOP, RESTART
114     }
115 
116     public abstract String getCommand(ServiceType service, Operation op);
117 
118     public String isRunningCommand(ServiceType service) {
119       return findPidCommand(service);
120     }
121 
122     protected String findPidCommand(ServiceType service) {
123       return String.format("ps aux | grep proc_%s | grep -v grep | tr -s ' ' | cut -d ' ' -f2",
124           service);
125     }
126 
127     public String signalCommand(ServiceType service, String signal) {
128       return String.format("%s | xargs kill -s %s", findPidCommand(service), signal);
129     }
130   }
131 
132   /**
133    * CommandProvider to manage the service using bin/hbase-* scripts
134    */
135   static class HBaseShellCommandProvider extends CommandProvider {
136     private String getHBaseHome() {
137       return System.getenv("HBASE_HOME");
138     }
139 
140     private String getConfig() {
141       String confDir = System.getenv("HBASE_CONF_DIR");
142       if (confDir != null) {
143         return String.format("--config %s", confDir);
144       }
145       return "";
146     }
147 
148     @Override
149     public String getCommand(ServiceType service, Operation op) {
150       return String.format("%s/bin/hbase-daemon.sh %s %s %s", getHBaseHome(), getConfig(),
151           op.toString().toLowerCase(), service);
152     }
153   }
154 
155   public HBaseClusterManager() {
156     super();
157   }
158 
159   protected CommandProvider getCommandProvider(ServiceType service) {
160     //TODO: make it pluggable, or auto-detect the best command provider, should work with
161     //hadoop daemons as well
162     return new HBaseShellCommandProvider();
163   }

165   /**
166    * Execute the given command on the host using SSH
167    * @return pair of exit code and command output
168    * @throws IOException if something goes wrong.
169    */
170   private Pair<Integer, String> exec(String hostname, String... cmd) throws IOException {
171     LOG.info("Executing remote command: " + StringUtils.join(cmd, " ") + " , hostname:" + hostname);
172 
173     RemoteShell shell = new RemoteShell(hostname, cmd);
174     shell.execute();
175 
176     LOG.info("Executed remote command, exit code:" + shell.getExitCode()
177         + " , output:" + shell.getOutput());
178 
179     return new Pair<Integer, String>(shell.getExitCode(), shell.getOutput());
180   }
181 
182   private void exec(String hostname, ServiceType service, Operation op) throws IOException {
183     exec(hostname, getCommandProvider(service).getCommand(service, op));
184   }
185 
186   @Override
187   public void start(ServiceType service, String hostname) throws IOException {
188     exec(hostname, service, Operation.START);
189   }
190 
191   @Override
192   public void stop(ServiceType service, String hostname) throws IOException {
193     exec(hostname, service, Operation.STOP);
194   }
195 
196   @Override
197   public void restart(ServiceType service, String hostname) throws IOException {
198     exec(hostname, service, Operation.RESTART);
199   }
200 
201   @Override
202   public void signal(ServiceType service, String signal, String hostname) throws IOException {
203     exec(hostname, getCommandProvider(service).signalCommand(service, signal));
204   }
205 
206   @Override
207   public boolean isRunning(ServiceType service, String hostname) throws IOException {
208     String ret = exec(hostname, getCommandProvider(service).isRunningCommand(service))
209         .getSecond();
210     return ret.length() > 0;
211   }
