88    /**
89     * Does {@link MetaReader#getRegion(CatalogTracker, byte[])} and a write
90     * against .META. while its hosted server is restarted to prove our retrying
91     * works.
92     * @throws IOException
93     * @throws InterruptedException
94     */
95    @Test public void testRetrying()
96    throws IOException, InterruptedException {
97      final String name = "testRetrying";
98      LOG.info("Started " + name);
99      final byte [] nameBytes = Bytes.toBytes(name);
100     HTable t = UTIL.createTable(nameBytes, HConstants.CATALOG_FAMILY);
101     int regionCount = UTIL.createMultiRegions(t, HConstants.CATALOG_FAMILY);
102     // Test it works getting a region from just made user table.
103     final List<HRegionInfo> regions =
104       testGettingTableRegions(CT, nameBytes, regionCount);
105     MetaTask reader = new MetaTask(CT, "reader") {
106       @Override
107       void metaTask() throws Throwable {
108         testGetRegion(this.ct, regions.get(0));
109         LOG.info("Read " + regions.get(0).getEncodedName());
110       }
111     };
112     MetaTask writer = new MetaTask(CT, "writer") {
113       @Override
114       void metaTask() throws Throwable {
115         MetaEditor.addRegionToMeta(this.ct, regions.get(0));
116         LOG.info("Wrote " + regions.get(0).getEncodedName());
117       }
118     };
119     reader.start();
120     writer.start();
121 
122     // We're gonna check how it takes. If it takes too long, we will consider
123     //  it as a fail. We can't put that in the @Test tag as we want to close
124     //  the threads nicely
125     final long timeOut = 180000;
126     long startTime = System.currentTimeMillis();
127 
128     try {
129       // Make sure reader and writer are working.
130       assertTrue(reader.isProgressing());
131       assertTrue(writer.isProgressing());
132 
133       // Kill server hosting meta -- twice  . See if our reader/writer ride over the
134       // meta moves.  They'll need to retry.
135       for (int i = 0; i < 2; i++) {
136         LOG.info("Restart=" + i);
137         UTIL.ensureSomeRegionServersAvailable(2);
138         int index = -1;
139         do {
140           index = UTIL.getMiniHBaseCluster().getServerWithMeta();
141         } while (index == -1 &&
142           startTime + timeOut < System.currentTimeMillis());
143 
144         if (index != -1){
145           UTIL.getMiniHBaseCluster().abortRegionServer(index);
146           UTIL.getMiniHBaseCluster().waitOnRegionServer(index);
147         }
148       }
149 
150       assertTrue("reader: " + reader.toString(), reader.isProgressing());
151       assertTrue("writer: " + writer.toString(), writer.isProgressing());
152     } catch (IOException e) {
153       throw e;
154     } finally {
155       reader.stop = true;
156       writer.stop = true;
157       reader.join();
158       writer.join();
159       t.close();
160     }
161     long exeTime = System.currentTimeMillis() - startTime;
162     assertTrue("Timeout: test took " + exeTime / 1000 + " sec", exeTime < timeOut);
163   }

165   /**
166    * Thread that runs a MetaReader/MetaEditor task until asked stop.
167    */
168   abstract static class MetaTask extends Thread {
169     boolean stop = false;
170     int count = 0;
171     Throwable t = null;
172     final CatalogTracker ct;
173 
174     MetaTask(final CatalogTracker ct, final String name) {
175       super(name);
176       this.ct = ct;
177     }
178 
179     @Override
180     public void run() {
181       try {
182         while(!this.stop) {
183           LOG.info("Before " + this.getName()+ ", count=" + this.count);
184           metaTask();
185           this.count += 1;
186           LOG.info("After " + this.getName() + ", count=" + this.count);
187           Thread.sleep(100);
188         }
189       } catch (Throwable t) {
190         LOG.info(this.getName() + " failed", t);
191         this.t = t;
192       }
193     }
194 
195     boolean isProgressing() throws InterruptedException {
196       int currentCount = this.count;
197       while(currentCount == this.count) {
198         if (!isAlive()) return false;
199         if (this.t != null) return false;
200         Thread.sleep(10);
201       }
202       return true;
203     }
204 
205     @Override
206     public String toString() {
207       return "count=" + this.count + ", t=" +
208         (this.t == null? "null": this.t.toString());
209     }
210 
211     abstract void metaTask() throws Throwable;
212   }
