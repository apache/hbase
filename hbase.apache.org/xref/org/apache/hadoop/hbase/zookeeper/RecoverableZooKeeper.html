44  /**
45   * A zookeeper that can handle 'recoverable' errors.
46   * To handle recoverable errors, developers need to realize that there are two 
47   * classes of requests: idempotent and non-idempotent requests. Read requests 
48   * and unconditional sets and deletes are examples of idempotent requests, they 
49   * can be reissued with the same results. 
50   * (Although, the delete may throw a NoNodeException on reissue its effect on 
51   * the ZooKeeper state is the same.) Non-idempotent requests need special 
52   * handling, application and library writers need to keep in mind that they may 
53   * need to encode information in the data or name of znodes to detect 
54   * retries. A simple example is a create that uses a sequence flag. 
55   * If a process issues a create("/x-", ..., SEQUENCE) and gets a connection 
56   * loss exception, that process will reissue another 
57   * create("/x-", ..., SEQUENCE) and get back x-111. When the process does a 
58   * getChildren("/"), it sees x-1,x-30,x-109,x-110,x-111, now it could be 
59   * that x-109 was the result of the previous create, so the process actually 
60   * owns both x-109 and x-111. An easy way around this is to use "x-process id-" 
61   * when doing the create. If the process is using an id of 352, before reissuing
62   * the create it will do a getChildren("/") and see "x-222-1", "x-542-30", 
63   * "x-352-109", x-333-110". The process will know that the original create 
64   * succeeded an the znode it created is "x-352-109".
65   * @see "http://wiki.apache.org/hadoop/ZooKeeper/ErrorHandling"
66   */

95    public RecoverableZooKeeper(String quorumServers, int sessionTimeout,
96        Watcher watcher, int maxRetries, int retryIntervalMillis) 
97    throws IOException {
98      this.zk = new ZooKeeper(quorumServers, sessionTimeout, watcher);
99      this.retryCounterFactory =
100       new RetryCounterFactory(maxRetries, retryIntervalMillis);
101 
102     // the identifier = processID@hostName
103     this.identifier = ManagementFactory.getRuntimeMXBean().getName();
104     LOG.info("The identifier of this process is " + identifier);
105     this.id = Bytes.toBytes(identifier);
106 
107     this.watcher = watcher;
108     this.sessionTimeout = sessionTimeout;
109     this.quorumServers = quorumServers;
110     salter = new SecureRandom();
111   }

114         throws IOException, InterruptedException {
115     LOG.info("Closing dead ZooKeeper connection, session" +
116       " was: 0x"+Long.toHexString(zk.getSessionId()));
117     zk.close();
118     this.zk = new ZooKeeper(this.quorumServers,

121       " is: 0x"+Long.toHexString(zk.getSessionId()));

124   /**
125    * delete is an idempotent operation. Retry before throwing exception.
126    * This function will not throw NoNodeException if the path does not
127    * exist.
128    */
129   public void delete(String path, int version)
130   throws InterruptedException, KeeperException {
131     RetryCounter retryCounter = retryCounterFactory.create();
132     boolean isRetry = false; // False for first attempt, true for all retries.
133     while (true) {
134       try {
135         zk.delete(path, version);
136         return;
137       } catch (KeeperException e) {
138         switch (e.code()) {
139           case NONODE:
140             if (isRetry) {
141               LOG.info("Node " + path + " already deleted. Assuming a " +
142                   "previous attempt succeeded.");
143               return;
144             }
145             LOG.warn("Node " + path + " already deleted, retry=" + isRetry);
146             throw e;
147 
148           case CONNECTIONLOSS:
149           case SESSIONEXPIRED:
150           case OPERATIONTIMEOUT:
151             retryOrThrow(retryCounter, e, "delete");
152             break;
153 
154           default:
155             throw e;
156         }
157       }
158       retryCounter.sleepUntilNextRetry();
159       retryCounter.useRetry();
160       isRetry = true;
161     }
162   }

218   private void retryOrThrow(RetryCounter retryCounter, KeeperException e,
219       String opName) throws KeeperException {
220     LOG.warn("Possibly transient ZooKeeper exception: " + e);
221     if (!retryCounter.shouldRetry()) {
222       LOG.error("ZooKeeper " + opName + " failed after "
223         + retryCounter.getMaxRetries() + " retries");
224       throw e;
225     }
226   }

338   /**
339    * setData is NOT an idempotent operation. Retry may cause BadVersion Exception
340    * Adding an identifier field into the data to check whether 
341    * badversion is caused by the result of previous correctly setData
342    * @return Stat instance
343    */
344   public Stat setData(String path, byte[] data, int version)
345   throws KeeperException, InterruptedException {
346     RetryCounter retryCounter = retryCounterFactory.create();
347     byte[] newData = appendMetaData(data);
348     boolean isRetry = false;
349     while (true) {
350       try {
351         return zk.setData(path, newData, version);
352       } catch (KeeperException e) {
353         switch (e.code()) {
354           case CONNECTIONLOSS:
355           case SESSIONEXPIRED:
356           case OPERATIONTIMEOUT:
357             retryOrThrow(retryCounter, e, "setData");
358             break;
359           case BADVERSION:
360             if (isRetry) {
361               // try to verify whether the previous setData success or not
362               try{
363                 Stat stat = new Stat();
364                 byte[] revData = zk.getData(path, false, stat);
365                 if(Bytes.compareTo(revData, newData) == 0) {
366                   // the bad version is caused by previous successful setData
367                   return stat;
368                 }
369               } catch(KeeperException keeperException){
370                 // the ZK is not reliable at this moment. just throwing exception
371                 throw keeperException;
372               }
373             }
374           // throw other exceptions and verified bad version exceptions
375           default:
376             throw e;
377         }
378       }
379       retryCounter.sleepUntilNextRetry();
380       retryCounter.useRetry();
381       isRetry = true;
382     }
383   }

385   /**
386    * <p>
387    * NONSEQUENTIAL create is idempotent operation. 
388    * Retry before throwing exceptions.
389    * But this function will not throw the NodeExist exception back to the
390    * application.
391    * </p>
392    * <p>
393    * But SEQUENTIAL is NOT idempotent operation. It is necessary to add 
394    * identifier to the path to verify, whether the previous one is successful 
395    * or not.
396    * </p>
397    * 
398    * @return Path
399    */
400   public String create(String path, byte[] data, List<ACL> acl,
401       CreateMode createMode)
402   throws KeeperException, InterruptedException {
403     byte[] newData = appendMetaData(data);
404     switch (createMode) {
405       case EPHEMERAL:
406       case PERSISTENT:
407         return createNonSequential(path, newData, acl, createMode);
408 
409       case EPHEMERAL_SEQUENTIAL:
410       case PERSISTENT_SEQUENTIAL:
411         return createSequential(path, newData, acl, createMode);
412 
413       default:
414         throw new IllegalArgumentException("Unrecognized CreateMode: " + 
415             createMode);
416     }
417   }

419   private String createNonSequential(String path, byte[] data, List<ACL> acl, 
420       CreateMode createMode) throws KeeperException, InterruptedException {
421     RetryCounter retryCounter = retryCounterFactory.create();
422     boolean isRetry = false; // False for first attempt, true for all retries.
423     while (true) {
424       try {
425         return zk.create(path, data, acl, createMode);
426       } catch (KeeperException e) {
427         switch (e.code()) {
428           case NODEEXISTS:
429             if (isRetry) {
430               // If the connection was lost, there is still a possibility that
431               // we have successfully created the node at our previous attempt,
432               // so we read the node and compare. 
433               byte[] currentData = zk.getData(path, false, null);
434               if (currentData != null &&
435                   Bytes.compareTo(currentData, data) == 0) { 
436                 // We successfully created a non-sequential node
437                 return path;
438               }
439               LOG.error("Node " + path + " already exists with " + 
440                   Bytes.toStringBinary(currentData) + ", could not write " +
441                   Bytes.toStringBinary(data));
442               throw e;
443             }
444             LOG.info("Node " + path + " already exists and this is not a " +
445                 "retry");
446             throw e;
447 
448           case CONNECTIONLOSS:
449           case SESSIONEXPIRED:
450           case OPERATIONTIMEOUT:
451             retryOrThrow(retryCounter, e, "create");
452             break;
453 
454           default:
455             throw e;
456         }
457       }
458       retryCounter.sleepUntilNextRetry();
459       retryCounter.useRetry();
460       isRetry = true;
461     }
462   }

464   private String createSequential(String path, byte[] data, 
465       List<ACL> acl, CreateMode createMode)
466   throws KeeperException, InterruptedException {
467     RetryCounter retryCounter = retryCounterFactory.create();
468     boolean first = true;
469     String newPath = path+this.identifier;
470     while (true) {
471       try {
472         if (!first) {
473           // Check if we succeeded on a previous attempt
474           String previousResult = findPreviousSequentialNode(newPath);
475           if (previousResult != null) {
476             return previousResult;
477           }
478         }
479         first = false;
480         return zk.create(newPath, data, acl, createMode);
481       } catch (KeeperException e) {
482         switch (e.code()) {
483           case CONNECTIONLOSS:
484           case SESSIONEXPIRED:
485           case OPERATIONTIMEOUT:
486             retryOrThrow(retryCounter, e, "create");
487             break;
488 
489           default:
490             throw e;
491         }
492       }
493       retryCounter.sleepUntilNextRetry();
494       retryCounter.useRetry();
495     }
496   }

498   private String findPreviousSequentialNode(String path)
499     throws KeeperException, InterruptedException {
500     int lastSlashIdx = path.lastIndexOf('/');
501     assert(lastSlashIdx != -1);
502     String parent = path.substring(0, lastSlashIdx);
503     String nodePrefix = path.substring(lastSlashIdx+1);
504 
505     List<String> nodes = zk.getChildren(parent, false);
506     List<String> matching = filterByPrefix(nodes, nodePrefix);
507     for (String node : matching) {
508       String nodePath = parent + "/" + node;
509       Stat stat = zk.exists(nodePath, false);
510       if (stat != null) {
511         return nodePath;
512       }
513     }
514     return null;
515   }

517   public byte[] removeMetaData(byte[] data) {
518     if(data == null || data.length == 0) {
519       return data;
520     }
521     // check the magic data; to be backward compatible
522     byte magic = data[0];
523     if(magic != MAGIC) {
524       return data;
525     }
526 
527     int idLength = Bytes.toInt(data, ID_LENGTH_OFFSET);
528     int dataLength = data.length-MAGIC_SIZE-ID_LENGTH_SIZE-idLength;
529     int dataOffset = MAGIC_SIZE+ID_LENGTH_SIZE+idLength;
530 
531     byte[] newData = new byte[dataLength];
532     System.arraycopy(data, dataOffset, newData, 0, dataLength);
533     return newData;
534   }

536   private byte[] appendMetaData(byte[] data) {
537     if(data == null || data.length == 0){
538       return data;
539     }
540     byte[] salt = Bytes.toBytes(salter.nextLong());
541     int idLength = id.length + salt.length;
542     byte[] newData = new byte[MAGIC_SIZE+ID_LENGTH_SIZE+idLength+data.length];
543     int pos = 0;
544     pos = Bytes.putByte(newData, pos, MAGIC);
545     pos = Bytes.putInt(newData, pos, idLength);
546     pos = Bytes.putBytes(newData, pos, id, 0, id.length);
547     pos = Bytes.putBytes(newData, pos, salt, 0, salt.length);
548     pos = Bytes.putBytes(newData, pos, data, 0, data.length);
549     return newData;
550   }

576   /**
577    * Filters the given node list by the given prefixes.
578    * This method is all-inclusive--if any element in the node list starts
579    * with any of the given prefixes, then it is included in the result.
580    *
581    * @param nodes the nodes to filter
582    * @param prefixes the prefixes to include in the result
583    * @return list of every element that starts with one of the prefixes
584    */
585   private static List<String> filterByPrefix(List<String> nodes, 
586       String... prefixes) {
587     List<String> lockChildren = new ArrayList<String>();
588     for (String child : nodes){
589       for (String prefix : prefixes){
590         if (child.startsWith(prefix)){
591           lockChildren.add(child);
592           break;
593         }
594       }
595     }
596     return lockChildren;
597   }
