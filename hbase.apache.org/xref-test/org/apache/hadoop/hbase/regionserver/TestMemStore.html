79     * Test memstore snapshot happening while scanning.
80     * @throws IOException

82    public void testScanAcrossSnapshot() throws IOException {
83      int rowCount = addRows(this.memstore);
84      List<KeyValueScanner> memstorescanners = this.memstore.getScanners();
85      Scan scan = new Scan();
86      List<KeyValue> result = new ArrayList<KeyValue>();
87      MultiVersionConsistencyControl.resetThreadReadPoint(mvcc);
88      ScanInfo scanInfo = new ScanInfo(null, 0, 1, HConstants.LATEST_TIMESTAMP, false,
89          0, this.memstore.comparator);
90      ScanType scanType = ScanType.USER_SCAN;
91      StoreScanner s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
92      int count = 0;
93      try {
94        while (s.next(result)) {
95          LOG.info(result);
96          count++;
97          // Row count is same as column count.
98          assertEquals(rowCount, result.size());
99          result.clear();
100       }
101     } finally {
102       s.close();
103     }
104     assertEquals(rowCount, count);
105     for (KeyValueScanner scanner : memstorescanners) {
106       scanner.close();
107     }
108 
109     MultiVersionConsistencyControl.resetThreadReadPoint(mvcc);
110     memstorescanners = this.memstore.getScanners();
111     // Now assert can count same number even if a snapshot mid-scan.
112     s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
113     count = 0;
114     try {
115       while (s.next(result)) {
116         LOG.info(result);
117         // Assert the stuff is coming out in right order.
118         assertTrue(Bytes.compareTo(Bytes.toBytes(count), result.get(0).getRow()) == 0);
119         count++;
120         // Row count is same as column count.
121         assertEquals(rowCount, result.size());
122         if (count == 2) {
123           this.memstore.snapshot();
124           LOG.info("Snapshotted");
125         }
126         result.clear();
127       }
128     } finally {
129       s.close();
130     }
131     assertEquals(rowCount, count);
132     for (KeyValueScanner scanner : memstorescanners) {
133       scanner.close();
134     }
135     memstorescanners = this.memstore.getScanners();
136     // Assert that new values are seen in kvset as we scan.
137     long ts = System.currentTimeMillis();
138     s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
139     count = 0;
140     int snapshotIndex = 5;
141     try {
142       while (s.next(result)) {
143         LOG.info(result);
144         // Assert the stuff is coming out in right order.
145         assertTrue(Bytes.compareTo(Bytes.toBytes(count), result.get(0).getRow()) == 0);
146         // Row count is same as column count.
147         assertEquals("count=" + count + ", result=" + result, rowCount, result.size());
148         count++;
149         if (count == snapshotIndex) {
150           this.memstore.snapshot();
151           this.memstore.clearSnapshot(this.memstore.getSnapshot());
152           // Added more rows into kvset.  But the scanner wont see these rows.
153           addRows(this.memstore, ts);
154           LOG.info("Snapshotted, cleared it and then added values (which wont be seen)");
155         }
156         result.clear();
157       }
158     } finally {
159       s.close();
160     }
161     assertEquals(rowCount, count);
162   }

168   public void testScanAcrossSnapshot2() throws IOException {
169     // we are going to the scanning across snapshot with two kvs
170     // kv1 should always be returned before kv2
171     final byte[] one = Bytes.toBytes(1);
172     final byte[] two = Bytes.toBytes(2);
173     final byte[] f = Bytes.toBytes("f");
174     final byte[] q = Bytes.toBytes("q");
175     final byte[] v = Bytes.toBytes(3);
176 
177     final KeyValue kv1 = new KeyValue(one, f, q, v);
178     final KeyValue kv2 = new KeyValue(two, f, q, v);
179 
180     // use case 1: both kvs in kvset
181     this.memstore.add(kv1.clone());
182     this.memstore.add(kv2.clone());
183     verifyScanAcrossSnapshot2(kv1, kv2);
184 
185     // use case 2: both kvs in snapshot
186     this.memstore.snapshot();
187     verifyScanAcrossSnapshot2(kv1, kv2);
188 
189     // use case 3: first in snapshot second in kvset
190     this.memstore = new MemStore();
191     this.memstore.add(kv1.clone());
192     this.memstore.snapshot();
193     this.memstore.add(kv2.clone());
194     verifyScanAcrossSnapshot2(kv1, kv2);
195   }

197   private void verifyScanAcrossSnapshot2(KeyValue kv1, KeyValue kv2)
198       throws IOException {
199     MultiVersionConsistencyControl.resetThreadReadPoint(mvcc);
200     List<KeyValueScanner> memstorescanners = this.memstore.getScanners();
201     assertEquals(1, memstorescanners.size());
202     final KeyValueScanner scanner = memstorescanners.get(0);
203     scanner.seek(KeyValue.createFirstOnRow(HConstants.EMPTY_START_ROW));
204     assertEquals(kv1, scanner.next());
205     assertEquals(kv2, scanner.next());
206     assertNull(scanner.next());
207   }

209   private void assertScannerResults(KeyValueScanner scanner, KeyValue[] expected)
210       throws IOException {
211     scanner.seek(KeyValue.createFirstOnRow(new byte[]{}));
212     List<KeyValue> returned = Lists.newArrayList();
213 
214     while (true) {
215       KeyValue next = scanner.next();
216       if (next == null) break;
217       returned.add(next);
218     }
219 
220     assertTrue(
221         "Got:\n" + Joiner.on("\n").join(returned) +
222         "\nExpected:\n" + Joiner.on("\n").join(expected),
223         Iterables.elementsEqual(Arrays.asList(expected), returned));
224     assertNull(scanner.peek());
225   }

268    * Regression test for HBASE-2616, HBASE-2670.
269    * When we insert a higher-memstoreTS version of a cell but with
270    * the same timestamp, we still need to provide consistent reads
271    * for the same scanner.
272    */
273   public void testMemstoreEditsVisibilityWithSameKey() throws IOException {
274     final byte[] row = Bytes.toBytes(1);
275     final byte[] f = Bytes.toBytes("family");
276     final byte[] q1 = Bytes.toBytes("q1");
277     final byte[] q2 = Bytes.toBytes("q2");
278     final byte[] v1 = Bytes.toBytes("value1");
279     final byte[] v2 = Bytes.toBytes("value2");
280 
281     // INSERT 1: Write both columns val1
282     MultiVersionConsistencyControl.WriteEntry w =
283         mvcc.beginMemstoreInsert();
284 
285     KeyValue kv11 = new KeyValue(row, f, q1, v1);
286     kv11.setMemstoreTS(w.getWriteNumber());
287     memstore.add(kv11);
288 
289     KeyValue kv12 = new KeyValue(row, f, q2, v1);
290     kv12.setMemstoreTS(w.getWriteNumber());
291     memstore.add(kv12);
292     mvcc.completeMemstoreInsert(w);
293 
294     // BEFORE STARTING INSERT 2, SEE FIRST KVS
295     MultiVersionConsistencyControl.resetThreadReadPoint(mvcc);
296     KeyValueScanner s = this.memstore.getScanners().get(0);
297     assertScannerResults(s, new KeyValue[]{kv11, kv12});
298 
299     // START INSERT 2: Write both columns val2
300     w = mvcc.beginMemstoreInsert();
301     KeyValue kv21 = new KeyValue(row, f, q1, v2);
302     kv21.setMemstoreTS(w.getWriteNumber());
303     memstore.add(kv21);
304 
305     KeyValue kv22 = new KeyValue(row, f, q2, v2);
306     kv22.setMemstoreTS(w.getWriteNumber());
307     memstore.add(kv22);
308 
309     // BEFORE COMPLETING INSERT 2, SEE FIRST KVS
310     MultiVersionConsistencyControl.resetThreadReadPoint(mvcc);
311     s = this.memstore.getScanners().get(0);
312     assertScannerResults(s, new KeyValue[]{kv11, kv12});
313 
314     // COMPLETE INSERT 2
315     mvcc.completeMemstoreInsert(w);
316 
317     // NOW SHOULD SEE NEW KVS IN ADDITION TO OLD KVS.
318     // See HBASE-1485 for discussion about what we should do with
319     // the duplicate-TS inserts
320     MultiVersionConsistencyControl.resetThreadReadPoint(mvcc);
321     s = this.memstore.getScanners().get(0);
322     assertScannerResults(s, new KeyValue[]{kv21, kv11, kv22, kv12});
323   }

376   private static class ReadOwnWritesTester extends Thread {
377     static final int NUM_TRIES = 1000;
378 
379     final byte[] row;
380 
381     final byte[] f = Bytes.toBytes("family");
382     final byte[] q1 = Bytes.toBytes("q1");
383 
384     final MultiVersionConsistencyControl mvcc;
385     final MemStore memstore;
386 
387     AtomicReference<Throwable> caughtException;
388 
389 
390     public ReadOwnWritesTester(int id,
391                                MemStore memstore,
392                                MultiVersionConsistencyControl mvcc,
393                                AtomicReference<Throwable> caughtException)
394     {
395       this.mvcc = mvcc;
396       this.memstore = memstore;
397       this.caughtException = caughtException;
398       row = Bytes.toBytes(id);
399     }
400 
401     public void run() {
402       try {
403         internalRun();
404       } catch (Throwable t) {
405         caughtException.compareAndSet(null, t);
406       }
407     }
408 
409     private void internalRun() throws IOException {
410       for (long i = 0; i < NUM_TRIES && caughtException.get() == null; i++) {
411         MultiVersionConsistencyControl.WriteEntry w =
412           mvcc.beginMemstoreInsert();
413 
414         // Insert the sequence value (i)
415         byte[] v = Bytes.toBytes(i);
416 
417         KeyValue kv = new KeyValue(row, f, q1, i, v);
418         kv.setMemstoreTS(w.getWriteNumber());
419         memstore.add(kv);
420         mvcc.completeMemstoreInsert(w);
421 
422         // Assert that we can read back
423         MultiVersionConsistencyControl.resetThreadReadPoint(mvcc);
424 
425         KeyValueScanner s = this.memstore.getScanners().get(0);
426         s.seek(kv);
427 
428         KeyValue ret = s.next();
429         assertNotNull("Didnt find own write at all", ret);
430         assertEquals("Didnt read own writes",
431                      kv.getTimestamp(), ret.getTimestamp());
432       }
433     }
434   }

436   public void testReadOwnWritesUnderConcurrency() throws Throwable {
437 
438     int NUM_THREADS = 8;
439 
440     ReadOwnWritesTester threads[] = new ReadOwnWritesTester[NUM_THREADS];
441     AtomicReference<Throwable> caught = new AtomicReference<Throwable>();
442 
443     for (int i = 0; i < NUM_THREADS; i++) {
444       threads[i] = new ReadOwnWritesTester(i, memstore, mvcc, caught);
445       threads[i].start();
446     }
447 
448     for (int i = 0; i < NUM_THREADS; i++) {
449       threads[i].join();
450     }
451 
452     if (caught.get() != null) {
453       throw caught.get();
454     }
455   }

529    * @throws InterruptedException

531   public void testGetNextRow() throws Exception {
532     MultiVersionConsistencyControl.resetThreadReadPoint();
533     addRows(this.memstore);
534     // Add more versions to make it a little more interesting.
535     Thread.sleep(1);
536     addRows(this.memstore);
537     KeyValue closestToEmpty = this.memstore.getNextRow(KeyValue.LOWESTKEY);
538     assertTrue(KeyValue.COMPARATOR.compareRows(closestToEmpty,
539       new KeyValue(Bytes.toBytes(0), System.currentTimeMillis())) == 0);
540     for (int i = 0; i < ROW_COUNT; i++) {
541       KeyValue nr = this.memstore.getNextRow(new KeyValue(Bytes.toBytes(i),
542         System.currentTimeMillis()));
543       if (i + 1 == ROW_COUNT) {
544         assertEquals(nr, null);
545       } else {
546         assertTrue(KeyValue.COMPARATOR.compareRows(nr,
547           new KeyValue(Bytes.toBytes(i + 1), System.currentTimeMillis())) == 0);
548       }
549     }
550     //starting from each row, validate results should contain the starting row
551     for (int startRowId = 0; startRowId < ROW_COUNT; startRowId++) {
552       ScanInfo scanInfo = new ScanInfo(FAMILY, 0, 1, Integer.MAX_VALUE, false,
553           0, this.memstore.comparator);
554       ScanType scanType = ScanType.USER_SCAN;
555       InternalScanner scanner = new StoreScanner(new Scan(
556           Bytes.toBytes(startRowId)), scanInfo, scanType, null,
557           memstore.getScanners());
558       List<KeyValue> results = new ArrayList<KeyValue>();
559       for (int i = 0; scanner.next(results); i++) {
560         int rowId = startRowId + i;
561         assertTrue("Row name",
562           KeyValue.COMPARATOR.compareRows(results.get(0),
563           Bytes.toBytes(rowId)) == 0);
564         assertEquals("Count of columns", QUALIFIER_COUNT, results.size());
565         List<KeyValue> row = new ArrayList<KeyValue>();
566         for (KeyValue kv : results) {
567           row.add(kv);
568         }
569         isExpectedRowWithoutTimestamps(rowId, row);
570         // Clear out set.  Otherwise row results accumulate.
571         results.clear();
572       }
573     }
574   }

799   /**
800    * Test a pathological pattern that shows why we can't currently
801    * use the MSLAB for upsert workloads. This test inserts data
802    * in the following pattern:
803    * 
804    * - row0001 through row1000 (fills up one 2M Chunk)
805    * - row0002 through row1001 (fills up another 2M chunk, leaves one reference
806    *   to the first chunk
807    * - row0003 through row1002 (another chunk, another dangling reference)
808    * 
809    * This causes OOME pretty quickly if we use MSLAB for upsert
810    * since each 2M chunk is held onto by a single reference.
811    */
812   public void testUpsertMSLAB() throws Exception {
813     Configuration conf = HBaseConfiguration.create();
814     conf.setBoolean(MemStore.USEMSLAB_KEY, true);
815     memstore = new MemStore(conf, KeyValue.COMPARATOR);
816     
817     int ROW_SIZE = 2048;
818     byte[] qualifier = new byte[ROW_SIZE - 4];
819     
820     MemoryMXBean bean = ManagementFactory.getMemoryMXBean();
821     for (int i = 0; i < 3; i++) { System.gc(); }
822     long usageBefore = bean.getHeapMemoryUsage().getUsed();
823     
824     long size = 0;
825     long ts=0;
826     
827     for (int newValue = 0; newValue < 1000; newValue++) {
828       for (int row = newValue; row < newValue + 1000; row++) {
829         byte[] rowBytes = Bytes.toBytes(row);
830         size += memstore.updateColumnValue(rowBytes, FAMILY, qualifier, newValue, ++ts);
831       }
832     }
833     System.out.println("Wrote " + ts + " vals");
834     for (int i = 0; i < 3; i++) { System.gc(); }
835     long usageAfter = bean.getHeapMemoryUsage().getUsed();
836     System.out.println("Memory used: " + (usageAfter - usageBefore)
837         + " (heapsize: " + memstore.heapSize() +
